28  package graph;
29  
30  import java.util.ArrayList;
31  
32  /**
33   * This class implements an undirected graph. It does so by defining a new type
34   * called UndirectedGraphNode which aliases the inNeighbors and the outNeighbors
35   * together.
36   *
37   * @author milind
38   * @param <NodeData>
39   * The data type stored at each node
40   */
41  public class UndirectedGraph<NodeData> extends DirectedGraph<NodeData> {
42  
43    @Override
44    public Node<NodeData> createNode(NodeData n) {
45      return new UndirectedGraphNode(n);
46    }
47  
48    @Override
49    public boolean isDirected() {
50      return false;
51    }
52  
53    /**
54     * The only difference between this class and GraphNode is that inNeighbors
55     * and outNeighbors are aliased to one another.
56     *
57     * @author milind
58     */
59    protected class UndirectedGraphNode extends GraphNode {
60      public UndirectedGraphNode(NodeData n) {
61        data = n;
62        inNeighbors = new ArrayList<GraphNode>();
63        outNeighbors = inNeighbors;
64      }
65    }
66  }
