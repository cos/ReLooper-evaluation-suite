28  package graph;
29  
30  import java.util.Collection;
31  
32  /**
33   * This variant of graph provides explicit Edge objects which can have arbitrary
34   * data (e.g. weights) associated with them
35   *
36   * @author milind
37   */
38  public interface EdgeGraph<NodeData, EdgeData> extends Graph<NodeData> {
39  
40    /**
41     * Factory method to create a new Edge object with associated EdgeData which will connect src to dest
42     * Note that this <b>does not</b> add the edge to the graph. This merely constructs the edge object.
43     * To add the edge to the graph, call @link{addEdge}.
44     *
45     * @param src  The source of the edge
46     * @param dest The dest of the edge
47     * @param e    The data to associate with the edge
48     * @return The newly created Edge object.
49     */
50    Edge<EdgeData> createEdge(Node<NodeData> src, Node<NodeData> dest, EdgeData e);
51  
52    /**
53     * Given two nodes, src and dest, returns the edge (src, dest) connecting them, if if exists
54     *
55     * @param src  The source node for the edge
56     * @param dest The destination node for the edge
57     * @return The edge (src, dest), if it exists, null otherwise
58     */
59    Edge<EdgeData> getEdge(Node<NodeData> src, Node<NodeData> dest);
60  
61    /**
62     * Remove an edge from the graph
63     *
64     * @param e The edge to remove
65     * @return true if the edge was removed
66     */
67    boolean removeEdge(Edge<EdgeData> e);
68  
69    /**
70     * Add an edge to the graph
71     *
72     * @param e The edge to add
73     * @return true if the edge was not already in the graph
74     */
75    boolean addEdge(Edge<EdgeData> e);
76  
77    /**
78     * Check whether an edge is in the graph
79     *
80     * @param e The edge to check
81     * @return true if e is in the graph
82     */
83    boolean hasEdge(Edge<EdgeData> e);
84  
85    /**
86     * Given an edge (src, dest), get src
87     *
88     * @param e The edge to examine
89     * @return If e = (src, dest), returns src
90     */
91    Node<NodeData> getSource(Edge<EdgeData> e);
92  
93    /**
94     * Given an edge (src, dest), get dest
95     *
96     * @param e The edge to examine
97     * @return IF e = (src, dest), returns dest
98     */
99    Node<NodeData> getDest(Edge<EdgeData> e);
100 
101   /**
102    * Return the edges leading away from n
103    *
104    * @param n The node to get edges for
105    * @return The set of all edges (n, m) in the graph
106    */
107   Collection<? extends Edge<EdgeData>> getOutEdges(Node<NodeData> n);
108 
109   /**
110    * Return the edges leading to n
111    *
112    * @param n The node to get edges for
113    * @return The set of all edges (m, n) in the graph
114    */
115   Collection<? extends Edge<EdgeData>> getInEdges(Node<NodeData> n);
116 
117   /**
118    * Get the data associated with an edge
119    *
120    * @param e The edge in question
121    * @return The EdgeData object associated with e
122    */
123   EdgeData getEdgeData(Edge<EdgeData> e);
124 
125   /**
126    * Set the data associated with an edge
127    *
128    * @param e The edge in question
129    * @param d The data to associate with e
130    * @return The data previously associated with e
131    */
132   EdgeData setEdgeData(Edge<EdgeData> e, EdgeData d);
133 }
