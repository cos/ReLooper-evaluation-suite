28  package graph;
29  
30  /**
31   * This interface represents a graph that allows programmers to refer to a
32   * node's edges by a particular index
33   *
34   * @author milind
35   * @param <NodeData>
36   * The type of object stored at each node
37   */
38  public interface IndexedGraph<NodeData> extends Graph<NodeData> {
39  
40    /**
41     * Set a particular neighbor of a given node
42     *
43     * @param src   The node whose neighbor to set
44     * @param index The particular neighbor to set
45     * @param dest  The new neighbor
46     */
47    void setNeighbor(Node<NodeData> src, int index, Node<NodeData> dest);
48  
49    /**
50     * Get a particular neighbor of a given node
51     *
52     * @param src   The node whose neighbor to get
53     * @param index The particular neighbor to get
54     * @return The neighbor at index
55     */
56    Node<NodeData> getNeighbor(Node<NodeData> src, int index);
57  
58    /**
59     * Remove a particular neighbor of a given node. Note that this is equivalent
60     * to calling setNeighbor(src, index, null)
61     *
62     * @param src   The node whose neighbor to remove
63     * @param index The neighbor to remove
64     */
65    void removeNeighbor(Node<NodeData> src, int index);
66  }
