41  public class DirectedEdgeGraph<NodeData, EdgeData> implements EdgeGraph<NodeData, EdgeData> {
42    Set<EdgeGraphNode> nodes;
43  
44    public DirectedEdgeGraph() {
45      nodes = Collections.synchronizedSet(new HashSet<EdgeGraphNode>());
46    }
47  
48    @Override
49    public boolean addEdge(Edge<EdgeData> e) {
50      GraphEdge ge = (GraphEdge) e;
51      EdgeGraphNode src = ge.getSrc();
52      EdgeGraphNode dest = ge.getDest();
53  
54      // if src.addOutEdge(dest, ge), then return dest.addInEdge(src, ge) [true],
55      // else return false
56      return (src.addOutEdge(dest, ge)) ? dest.addInEdge(src, ge) : false;
57    }
58  
59    @Override
60    public Edge<EdgeData> createEdge(Node<NodeData> src, Node<NodeData> dest, EdgeData e) {
61      return new GraphEdge((EdgeGraphNode) src, (EdgeGraphNode) dest, e);
62    }
63  
64    @Override
65    public Node<NodeData> getDest(Edge<EdgeData> e) {
66      return ((GraphEdge) e).getDest();
67    }
68  
69    @Override
70    public Edge<EdgeData> getEdge(Node<NodeData> src, Node<NodeData> dest) {
71      return ((EdgeGraphNode) src).getOutEdge((EdgeGraphNode) dest);
72    }

74    @Override
75    public Collection<? extends Edge<EdgeData>> getInEdges(Node<NodeData> n) {
76      return ((EdgeGraphNode) n).getInEdges();
77    }

80    public Collection<? extends Edge<EdgeData>> getOutEdges(Node<NodeData> n) {
81      return ((EdgeGraphNode) n).getOutEdges();

85    public Node<NodeData> getSource(Edge<EdgeData> e) {
86      return ((GraphEdge) e).src;

89    @Override
90    public boolean hasEdge(Edge<EdgeData> e) {
91      GraphEdge ge = (GraphEdge) e;
92      return (ge.getSrc().hasOutNeighbor(ge.getDest()));
93    }
94  
95    @Override
96    public boolean removeEdge(Edge<EdgeData> e) {
97      GraphEdge ge = (GraphEdge) e;
98      EdgeGraphNode src = ge.getSrc();
99      EdgeGraphNode dest = ge.getDest();
100     return (src.removeOutEdge(dest)) ? dest.removeInEdge(src) : false;
101   }

123   @Override
124   public boolean removeNeighbor(Node<NodeData> src, Node<NodeData> dest) {
125     EdgeGraphNode gsrc = (EdgeGraphNode) src;
126     EdgeGraphNode gdest = (EdgeGraphNode) dest;
127 
128     //if gsrc.removeOutEdge(gdest), then return gdest.removeInEdge(gsrc) [true], else return false.
129     return (gsrc.removeOutEdge(gdest)) ? gdest.removeInEdge(gsrc) : false;
130   }

133   public EdgeData getEdgeData(Edge<EdgeData> e) {

181   @Override
182   public boolean hasNeighbor(Node<NodeData> src, Node<NodeData> dest) {
183     EdgeGraphNode esrc = (EdgeGraphNode) src;
184     EdgeGraphNode edest = (EdgeGraphNode) dest;
185     return (esrc.hasOutNeighbor(edest));
186   }

188   @Override
189   public boolean removeNode(Node<NodeData> n) {
190     removeConnectingEdges((EdgeGraphNode) n);
191     return nodes.remove(n);
192   }
193 
194   protected void removeConnectingEdges(EdgeGraphNode n) {
195     Collection<EdgeGraphNode> outNeighbors = n.getOutNeighborsCopy();
196     for (EdgeGraphNode g : outNeighbors) {
197       removeNeighbor(n, g);
198     }
199 
200     Collection<EdgeGraphNode> inNeighbors = n.getInNeighborsCopy();
201     for (EdgeGraphNode g : inNeighbors) {
202       removeNeighbor(g, n);
203     }
204   }
205 
206   @Override
207   public NodeData setNodeData(Node<NodeData> n, NodeData d) {
208     EdgeGraphNode egn = (EdgeGraphNode) n;
209     NodeData retval = egn.data;
210     egn.data = d;
211     return retval;
212   }

214   protected class EdgeGraphNode implements Node<NodeData> {
215     protected Map<EdgeGraphNode, GraphEdge> inEdges;
216     protected Map<EdgeGraphNode, GraphEdge> outEdges;
217 
218     protected NodeData data;
219 
220     EdgeGraphNode() {
221     }
222 
223     EdgeGraphNode(NodeData d) {
224       inEdges = new HashMap<EdgeGraphNode, GraphEdge>();
225       outEdges = new HashMap<EdgeGraphNode, GraphEdge>();
226       data = d;
227     }
228 
229     final protected boolean hasInNeighbor(EdgeGraphNode n) {
230       return inEdges.containsKey(n);
231     }
232 
233     /**
234      * Record the edge (n, this)
235      *
236      * @param n The source of the edge
237      * @param e The edge object
238      * @return True if the edge was successfully added, false otherwise
239      */
240     protected boolean addInEdge(EdgeGraphNode n, GraphEdge e) {
241       if (hasInNeighbor(n)) {
242         return false;
243       }
244       inEdges.put(n, e);
245       return true;
246     }
247 
248     /**
249      * Remove the edge (if any) (n, this)
250      *
251      * @param n The source of the edge
252      * @return true if an edge was removed, false otherwise
253      */
254     protected boolean removeInEdge(EdgeGraphNode n) {
255       if (!hasInNeighbor(n)) {
256         return false;
257       }
258       inEdges.remove(n);
259       return true;
260     }
261 
262     /**
263      * Return the edge (n, this)
264      *
265      * @param n The source of the edge
266      * @return The edge object connecting n to this (if one exists)
267      */
268     protected GraphEdge getInEdge(EdgeGraphNode n) {
269       return inEdges.get(n);
270     }
271 
272     /**
273      * Get all the edges (n, this)
274      *
275      * @return A collection of all edges (n, this)
276      */
277     protected Collection<GraphEdge> getInEdges() {
278       return inEdges.values();
279     }
280 
281     final protected Collection<EdgeGraphNode> getInNeighbors() {
282       return inEdges.keySet();
283     }
284 
285     final protected Collection<EdgeGraphNode> getInNeighborsCopy() {
286       return new ArrayList<EdgeGraphNode>(inEdges.keySet());
287     }
288 
289     final protected boolean hasOutNeighbor(EdgeGraphNode n) {
290       return outEdges.containsKey(n);
291     }
292 
293     /**
294      * Record the edge (this, n)
295      *
296      * @param n The target of the edge
297      * @param e The edge object
298      * @return true if the edge was successfully added, false otherwise
299      */
300     protected boolean addOutEdge(EdgeGraphNode n, GraphEdge e) {
301       if (hasOutNeighbor(n)) {
302         return false;
303       }
304       outEdges.put(n, e);
305       return true;
306     }
307 
308     /**
309      * Remove the edge (if any) (this, n)
310      *
311      * @param n The target of the edge
312      * @return true if an edge was removed, false otherwise
313      */
314     protected boolean removeOutEdge(EdgeGraphNode n) {
315       if (!hasOutNeighbor(n)) {
316         return false;
317       }
318       outEdges.remove(n);
319       return true;
320     }
321 
322     /**
323      * Return the edge (this, n)
324      *
325      * @param n The target of the edge
326      * @return The edge object connecting this to n (if one exists)
327      */
328     protected GraphEdge getOutEdge(EdgeGraphNode n) {
329       return outEdges.get(n);
330     }
331 
332     /**
333      * Get all the edges (this, n)
334      *
335      * @return A collection of all edges (this, n)
336      */
337     protected Collection<GraphEdge> getOutEdges() {
338       return outEdges.values();
339     }
340 
341     final protected Collection<EdgeGraphNode> getOutNeighbors() {
342       return outEdges.keySet();
343     }
344 
345     final protected Collection<EdgeGraphNode> getOutNeighborsCopy() {
346       return new ArrayList<EdgeGraphNode>(outEdges.keySet());
347     }
348 
349     @Override
350     public NodeData getData() {
351       return getNodeData(this);
352     }
353 
354     @Override
355     public NodeData setData(NodeData n) {
356       return setNodeData(this, n);
357     }
358   }

360   protected class GraphEdge implements Edge<EdgeData> {
361 
362     protected EdgeGraphNode src;
363     protected EdgeGraphNode dest;
364 
365     protected EdgeData d;
366 
367     public GraphEdge(EdgeData d) {
368       this.d = d;
369     }
370 
371     public GraphEdge(EdgeGraphNode src, EdgeGraphNode dest, EdgeData d) {
372       this(d);
373       this.src = src;
374       this.dest = dest;
375     }
376 
377     final protected EdgeGraphNode getOpposite(EdgeGraphNode n) {
378       return (n == src) ? dest : src;
379     }
380 
381     final protected EdgeGraphNode getSrc() {
382       return src;
383     }
384 
385     final protected EdgeGraphNode getDest() {
386       return dest;
387     }
388 
389     @Override
390     public EdgeData getData() {
391       return getEdgeData(this);
392     }
393 
394     @Override
395     public EdgeData setData(EdgeData e) {
396       return setEdgeData(this, e);
397     }
398   }
399 
400   @Override
401   public boolean isDirected() {
402     return true;
403   }
