41  public final class SerialBarneshut {
42    private static int nbodies; // number of bodies in system
43    private static int ntimesteps; // number of time steps to run
44    private static double dtime; // length of one time step
45    private static double eps; // potential softening parameter
46    private static double tol; // tolerance for stopping recursion, should be less than 0.57 for 3D case to bound error
47  
48    private static double dthf, epssq, itolsq;
49    private static OctTreeLeafNodeData body[]; // the n bodies
50    private static double diameter, centerx, centery, centerz;
51    private static int curr;
52  
53    private static boolean isFirstRun = true;

56    private static void ReadInput(String filename) {
57      double vx, vy, vz;
58      Scanner in = null;
59      try {
60        in = new Scanner(new BufferedReader(new FileReader(filename)));
61        in.useLocale(Locale.US);
62      } catch (FileNotFoundException e) {
63        System.err.println(e);
64        System.exit(-1);
65      }
66      nbodies = in.nextInt();
67      ntimesteps = in.nextInt();
68      dtime = in.nextDouble();
69      eps = in.nextDouble();
70      tol = in.nextDouble();
71      dthf = 0.5 * dtime;
72      epssq = eps * eps;
73      itolsq = 1.0 / (tol * tol);
74      if (isFirstRun) {
75        System.err.println("configuration: " + nbodies + " bodies, " + ntimesteps + " time steps");
76        System.err.println();
77      }
78      body = new OctTreeLeafNodeData[nbodies];
79      for (int i = 0; i < nbodies; i++) {
80        body[i] = new OctTreeLeafNodeData();
81      }
82      for (int i = 0; i < nbodies; i++) {
83        body[i].mass = in.nextDouble();
84        body[i].posx = in.nextDouble();
85        body[i].posy = in.nextDouble();
86        body[i].posz = in.nextDouble();
87        vx = in.nextDouble();
88        vy = in.nextDouble();
89        vz = in.nextDouble();
90        body[i].setVelocity(vx, vy, vz);
91      }
92    }

95    private static void ComputeCenterAndDiameter() {
96      double minx, miny, minz;
97      double maxx, maxy, maxz;
98      double posx, posy, posz;
99      minx = miny = minz = Double.MAX_VALUE;
100     maxx = maxy = maxz = Double.MIN_VALUE;
101     for (int i = 0; i < nbodies; i++) {
102       posx = body[i].posx;
103       posy = body[i].posy;
104       posz = body[i].posz;
105       if (minx > posx) {
106         minx = posx;
107       }
108       if (miny > posy) {
109         miny = posy;
110       }
111       if (minz > posz) {
112         minz = posz;
113       }
114       if (maxx < posx) {
115         maxx = posx;
116       }
117       if (maxy < posy) {
118         maxy = posy;
119       }
120       if (maxz < posz) {
121         maxz = posz;
122       }
123     }
124     diameter = maxx - minx;
125     if (diameter < (maxy - miny)) {
126       diameter = (maxy - miny);
127     }
128     if (diameter < (maxz - minz)) {
129       diameter = (maxz - minz);
130     }
131     centerx = (maxx + minx) * 0.5;
132     centery = (maxy + miny) * 0.5;
133     centerz = (maxz + minz) * 0.5;
134   }

137   static void ComputeCenterOfMass(ArrayIndexedGraph<OctTreeNodeData> octree, Node<OctTreeNodeData> root) { // recursively summarizes info about subtrees
138     double m, px = 0.0, py = 0.0, pz = 0.0;
139     OctTreeNodeData n = octree.getNodeData(root);
140     int j = 0;
141     n.mass = 0.0;
142     for (int i = 0; i < 8; i++) {
143       Node<OctTreeNodeData> child = octree.getNeighbor(root, i);
144       if (child != null) {
145         if (i != j) {
146           octree.removeNeighbor(root, i);
147           octree.setNeighbor(root, j, child); // move non-null children to the front (needed later to make other code faster)
148         }
149         j++;
150         OctTreeNodeData ch = octree.getNodeData(child);
151         if (ch instanceof OctTreeLeafNodeData) {
152           body[curr++] = (OctTreeLeafNodeData) ch; // sort bodies in tree order (approximation of putting nearby nodes together for locality)
153         } else {
154           ComputeCenterOfMass(octree, child);
155         }
156         m = ch.mass;
157         n.mass += m;
158         px += ch.posx * m;
159         py += ch.posy * m;
160         pz += ch.posz * m;
161       }
162     }
163     m = 1.0 / n.mass;
164     n.posx = px * m;
165     n.posy = py * m;
166     n.posz = pz * m;
167   }

170   static void Insert(ArrayIndexedGraph<OctTreeNodeData> octree, Node<OctTreeNodeData> root, OctTreeLeafNodeData b, double r) { // builds the tree
171     double x = 0.0, y = 0.0, z = 0.0;
172     OctTreeNodeData n = octree.getNodeData(root);
173     int i = 0;
174     if (n.posx < b.posx) {
175       i = 1;
176       x = r;
177     }
178     if (n.posy < b.posy) {
179       i += 2;
180       y = r;
181     }
182     if (n.posz < b.posz) {
183       i += 4;
184       z = r;
185     }
186     Node<OctTreeNodeData> child = octree.getNeighbor(root, i);
187     if (child == null) {
188       Node<OctTreeNodeData> newnode = octree.createNode(b);
189       octree.addNode(newnode);
190       octree.setNeighbor(root, i, newnode);
191     } else {
192       double rh = 0.5 * r;
193       OctTreeNodeData ch = octree.getNodeData(child);
194       if (!(ch instanceof OctTreeLeafNodeData)) {
195         Insert(octree, child, b, rh);
196       } else {
197         Node<OctTreeNodeData> newnode = octree.createNode(new OctTreeNodeData(n.posx - rh + x, n.posy - rh + y, n.posz - rh + z));
198         octree.addNode(newnode);
199         Insert(octree, newnode, b, rh);
200         Insert(octree, newnode, (OctTreeLeafNodeData) ch, rh);
201         octree.setNeighbor(root, i, newnode);
202       }
203     }
204   }

207   public static void main(String args[]) {
208     long runtime, lasttime, mintime, run;
209 
210     runtime = 0;
211     lasttime = Long.MAX_VALUE;
212     mintime = Long.MAX_VALUE;
213     run = 0;
214     while (((run < 3) || (Math.abs(lasttime-runtime)*64 > Math.min(lasttime, runtime))) && (run < 7)) {
215       System.gc();  System.gc();  System.gc();  System.gc();  System.gc();
216       runtime = run(args);
217       if (runtime < mintime) mintime = runtime;
218       run++;
219     }
220     System.err.println("minimum runtime: " + mintime + " ms");
221     System.err.println("");
222   }

225   public static long run(String args[]) {
226     if (isFirstRun) {
227       System.err.println("Lonestar Benchmark Suite v2.1");
228       System.err.println("Copyright (C) 2007, 2008, 2009 The University of Texas at Austin");
229       System.err.println("http://iss.ices.utexas.edu/lonestar/");
230       System.err.println();
231       System.err.println("application: BarnesHut (serial version)");
232       System.err.println("Simulation of the gravitational forces in a galactic");
233       System.err.println("cluster using the Barnes-Hut n-body algorithm");
234       System.err.println("http://iss.ices.utexas.edu/lonestar/barneshut.html");
235       System.err.println();
236     }
237     if (args.length != 1) {
238       System.err.println("arguments: input_file_name");
239       System.exit(-1);
240     }
241     ReadInput(args[0]);
242     long runtime = 0;
243     
244     long totaltimeid = Time.getNewTimeId();
245     for (int step = 0; step < ntimesteps; step++) { // time-step the system
246       ComputeCenterAndDiameter();
247       ArrayIndexedGraph<OctTreeNodeData> octree = new ArrayIndexedGraph<OctTreeNodeData>(8);
248       Node<OctTreeNodeData> root = octree.createNode(new OctTreeNodeData(centerx, centery, centerz)); // create the tree's root
249       octree.addNode(root);
250       double radius = diameter * 0.5;
251       for (int i = 0; i < nbodies; i++) {
252         Insert(octree, root, body[i], radius); // grow the tree by inserting each body
253       }
254       curr = 0;
255       // summarize subtree info in each internal node (plus restructure tree and sort bodies for performance reasons)
256       ComputeCenterOfMass(octree, root);
257 
258       long id = Time.getNewTimeId();
259       for (OctTreeLeafNodeData n : body) {
260         // compute the acceleration of each body (consumes most of the total runtime)
261         n.ComputeForce(octree, root, diameter, itolsq, step, dthf, epssq);
262       }
263       runtime += Time.elapsedTime(id);
264       for (int i = 0; i < nbodies; i++) {
265         body[i].Advance(dthf, dtime); // advance the position and velocity of each body
266       }
267     } // end of time step
268     long totaltime = Time.elapsedTime(totaltimeid);
269     
270     System.err.println("Total time: "+totaltime);
271 
272     if (isFirstRun) {
273       DecimalFormat df = new DecimalFormat("0.0000E00");
274       for (int i = 0; i < nbodies; i++) {
275         System.out.println(df.format(body[i].posx) + " " + df.format(body[i].posy) + " " + df.format(body[i].posz)); // print result
276       }
277       System.out.println();
278     }
279     System.err.println("runtime: " + runtime + " ms");
280 
281     isFirstRun = false;
282     return runtime;
283   }
