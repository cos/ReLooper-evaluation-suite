42  public final class ParallelBarneshut {
43      private static int nbodies; // number of bodies in system
44      private static int ntimesteps; // number of time steps to run
45      private static double dtime; // length of one time step
46      private static double eps; // potential softening parameter
47      private static double tol; // tolerance for stopping recursion, should be
48                                  // less than 0.57 for 3D case to bound error
49  
50      private static double dthf, epssq, itolsq;
51      private static ParallelArray<OctTreeLeafNodeData> body; // the n bodies
52      private static double diameter, centerx, centery, centerz;
53      private static int curr;
54  
55      private static boolean isFirstRun = true;

57      private static void ReadInput(String filename) {
58          Scanner in1 = null;
59          try {
60              in1 = new Scanner(new BufferedReader(new FileReader(filename)));
61              in1.useLocale(Locale.US);
62          } catch (FileNotFoundException e) {
63              System.err.println(e);
64              System.exit(-1);
65          }
66          final Scanner in = in1;
67          nbodies = in.nextInt();
68          ntimesteps = in.nextInt();
69          dtime = in.nextDouble();
70          eps = in.nextDouble();
71          tol = in.nextDouble();
72          dthf = 0.5 * dtime;
73          epssq = eps * eps;
74          itolsq = 1.0 / (tol * tol);
75          if (isFirstRun) {
76              System.err.println("configuration: " + nbodies + " bodies, " + ntimesteps + " time steps");
77              System.err.println();
78          }
79          body = ParallelArray.createUsingHandoff(new OctTreeLeafNodeData[nbodies], ParallelArray.defaultExecutor());
80          body.replaceWithGeneratedValueSeq(new Generator<OctTreeLeafNodeData>() {
81              @Override
82              public OctTreeLeafNodeData op() {
83                  return new OctTreeLeafNodeData();
84              }
85          });
86          body.applySeq(new Procedure<OctTreeLeafNodeData>() {
87  
88              @Override
89              public void op(OctTreeLeafNodeData el) {
90                  el.mass = in.nextDouble();
91                  el.posx = in.nextDouble();
92                  el.posy = in.nextDouble();
93                  el.posz = in.nextDouble();
94                  double vx, vy, vz;
95                  vx = in.nextDouble();
96                  vy = in.nextDouble();
97                  vz = in.nextDouble();
98                  el.setVelocity(vx, vy, vz);
99              }
100         });
101     }

103     private static void ComputeCenterAndDiameter() {
104         final double minx[] = new double[1], miny[] = new double[1], minz[] = new double[1];
105         final double maxx[] = new double[1], maxy[] = new double[1], maxz[] = new double[1];
106         minx[0] = miny[0] = minz[0] = Double.MAX_VALUE;
107         maxx[0] = maxy[0] = maxz[0] = Double.MIN_VALUE;
108         body.applySeq(new Procedure<OctTreeLeafNodeData>() {
109 
110             @Override
111             public void op(OctTreeLeafNodeData b) {
112                 // TODO Auto-generated method stub
113                 double posx = b.posx;
114                 double posy = b.posy;
115                 double posz = b.posz;
116                 if (minx[0] > posx) {
117                     minx[0] = posx;
118                 }
119                 if (miny[0] > posy) {
120                     miny[0] = posy;
121                 }
122                 if (minz[0] > posz) {
123                     minz[0] = posz;
124                 }
125                 if (maxx[0] < posx) {
126                     maxx[0] = posx;
127                 }
128                 if (maxy[0] < posy) {
129                     maxy[0] = posy;
130                 }
131                 if (maxz[0] < posz) {
132                     maxz[0] = posz;
133                 }
134             }
135         });
136             
137         diameter = maxx[0] - minx[0];
138         if (diameter < (maxy[0] - miny[0])) {
139             diameter = (maxy[0] - miny[0]);
140         }
141         if (diameter < (maxz[0] - minz[0])) {
142             diameter = (maxz[0] - minz[0]);
143         }
144         centerx = (maxx[0] + minx[0]) * 0.5;
145         centery = (maxy[0] + miny[0]) * 0.5;
146         centerz = (maxz[0] + minz[0]) * 0.5;
147     }

149     static void ComputeCenterOfMass(ArrayIndexedGraph<OctTreeNodeData> octree, Node<OctTreeNodeData> root) { // recursively
150                                                                                                                 // summarizes
151                                                                                                                 // info
152                                                                                                                 // about
153                                                                                                                 // subtrees
154         double m, px = 0.0, py = 0.0, pz = 0.0;
155         OctTreeNodeData n = octree.getNodeData(root);
156         int j = 0;
157         n.mass = 0.0;
158         for (int i = 0; i < 8; i++) {
159             Node<OctTreeNodeData> child = octree.getNeighbor(root, i);
160             if (child != null) {
161                 if (i != j) {
162                     octree.removeNeighbor(root, i);
163                     octree.setNeighbor(root, j, child); // move non-null
164                                                         // children to the front
165                                                         // (needed later to make
166                                                         // other code faster)
167                 }
168                 j++;
169                 OctTreeNodeData ch = octree.getNodeData(child);
170                 if (ch instanceof OctTreeLeafNodeData) {
171                     OctTreeLeafNodeData[] bodyArray = body.getArray();
172                     bodyArray[curr++] = (OctTreeLeafNodeData) ch; // sort bodies
173                                                                     // in tree
174                                                                     // order
175                                                                     // (approximation
176                                                                     // of
177                                                                     // putting
178                                                                     // nearby
179                                                                     // nodes
180                                                                     // together
181                                                                     // for
182                                                                     // locality)
183                 } else {
184                     ComputeCenterOfMass(octree, child);
185                 }
186                 m = ch.mass;
187                 n.mass += m;
188                 px += ch.posx * m;
189                 py += ch.posy * m;
190                 pz += ch.posz * m;
191             }
192         }
193         m = 1.0 / n.mass;
194         n.posx = px * m;
195         n.posy = py * m;
196         n.posz = pz * m;
197     }

199     static void Insert(ArrayIndexedGraph<OctTreeNodeData> octree, Node<OctTreeNodeData> root, OctTreeLeafNodeData b,
200             double r) { // builds the tree
201         double x = 0.0, y = 0.0, z = 0.0;
202         OctTreeNodeData n = octree.getNodeData(root);
203         int i = 0;
204         if (n.posx < b.posx) {
205             i = 1;
206             x = r;
207         }
208         if (n.posy < b.posy) {
209             i += 2;
210             y = r;
211         }
212         if (n.posz < b.posz) {
213             i += 4;
214             z = r;
215         }
216         Node<OctTreeNodeData> child = octree.getNeighbor(root, i);
217         if (child == null) {
218             Node<OctTreeNodeData> newnode = octree.createNode(b);
219             octree.addNode(newnode);
220             octree.setNeighbor(root, i, newnode);
221         } else {
222             double rh = 0.5 * r;
223             OctTreeNodeData ch = octree.getNodeData(child);
224             if (!(ch instanceof OctTreeLeafNodeData)) {
225                 Insert(octree, child, b, rh);
226             } else {
227                 Node<OctTreeNodeData> newnode = octree.createNode(new OctTreeNodeData(n.posx - rh + x, n.posy - rh + y,
228                         n.posz - rh + z));
229                 octree.addNode(newnode);
230                 Insert(octree, newnode, b, rh);
231                 Insert(octree, newnode, (OctTreeLeafNodeData) ch, rh);
232                 octree.setNeighbor(root, i, newnode);
233             }
234         }
235     }

237     public static void main(String[] args) {
238         long runtime, lasttime, mintime, run;
239 
240         runtime = 0;
241         lasttime = Long.MAX_VALUE;
242         mintime = Long.MAX_VALUE;
243         run = 0;
244         while (((run < 3) || (Math.abs(lasttime - runtime) * 64 > Math.min(lasttime, runtime))) && (run < 7)) {
245             System.gc();
246             System.gc();
247             System.gc();
248             System.gc();
249             System.gc();
250             runtime = run(args);
251             if (runtime < mintime)
252                 mintime = runtime;
253             run++;
254         }
255         System.err.println("minimum runtime: " + mintime + " ms");
256         System.err.println("");
257     }

259     public static long run(String args[]) {
260         if (isFirstRun) {
261             System.err.println("Lonestar Benchmark Suite v2.1");
262             System.err.println("Copyright (C) 2007, 2008, 2009 The University of Texas at Austin");
263             System.err.println("http://iss.ices.utexas.edu/lonestar/");
264             System.err.println();
265             System.err.println("application: BarnesHut (serial version)");
266             System.err.println("Simulation of the gravitational forces in a galactic");
267             System.err.println("cluster using the Barnes-Hut n-body algorithm");
268             System.err.println("http://iss.ices.utexas.edu/lonestar/barneshut.html");
269             System.err.println();
270         }
271         if (args.length != 1) {
272             System.err.println("arguments: input_file_name");
273             System.exit(-1);
274         }
275         ReadInput(args[0]);
276         long runtime = 0;
277 
278         for (int step = 0; step < ntimesteps; step++) {
279             // time-step the system
280             ComputeCenterAndDiameter();
281             final ArrayIndexedGraph<OctTreeNodeData> octree = new ArrayIndexedGraph<OctTreeNodeData>(8);
282             final Node<OctTreeNodeData> root = octree.createNode(new OctTreeNodeData(centerx, centery, centerz));
283             // create the tree's root
284             octree.addNode(root);
285             final double radius = diameter * 0.5;
286             body.applySeq(new Procedure<OctTreeNodeData>() {
287                 @Override
288                 public void op(OctTreeNodeData b) {
289                     Insert(octree, root, (OctTreeLeafNodeData) b, radius);
290                     // grow the tree by inserting each body
291                 }
292             });
293 
294             curr = 0;
295             // summarize subtree info in each internal node (plus restructure
296             // tree and sort bodies for performance reasons)
297             ComputeCenterOfMass(octree, root);
298 
299             long id = Time.getNewTimeId();
300 
301             final int finalizedStep = step;
302             // this is to overcome the anonymous class final requirement
303 
304             body.apply(new Procedure<OctTreeLeafNodeData>() {
305 
306                 @Override
307                 public void op(OctTreeLeafNodeData n) {
308                     n.ComputeForce(octree, root, diameter, itolsq, finalizedStep, dthf, epssq);
309                 }
310             });
311 
312             runtime += Time.elapsedTime(id);
313 
314             body.apply(new Procedure<OctTreeLeafNodeData>() {
315 
316                 @Override
317                 public void op(OctTreeLeafNodeData n) {
318                     n.Advance(dthf, dtime); // advance the position and velocity
319                                             // of each body
320                 }
321             });
322         } // end of time step
323 
324         if (isFirstRun) {
325             final DecimalFormat df = new DecimalFormat("0.0000E00");
326             body.applySeq(new Procedure<OctTreeLeafNodeData>() {
327 
328                 @Override
329                 public void op(OctTreeLeafNodeData b) {
330                     System.out.println(df.format(b.posx) + " " + df.format(b.posy) + " "
331                             + df.format(b.posz)); // print result
332                 }
333                 
334             });
335             System.out.println();
336         }
337         System.err.println("runtime: " + runtime + " ms");
338 
339         isFirstRun = false;
340         return runtime;
341     }
