25  package barnesHut;
26  
27  class OctTreeNodeData { // the internal nodes are cells that summarize their children's properties
28    double mass;
29    double posx;
30    double posy;
31    double posz;
32  
33    OctTreeNodeData(double px, double py, double pz) {
34      mass = 0.0;
35      posx = px;
36      posy = py;
37      posz = pz;
38    }
39  
40    @Override
41    public String toString() {
42      String result = "mass = " + mass;
43      result += "pos = (" + posx + "," + posy + "," + posz + ")";
44      return result;
45    }
46  
47    public double posx() {
48      return posx;
49    }
50  
51    public double posy() {
52      return posy;
53    }
54  
55    public double posz() {
56      return posz;
57    }
58  }
