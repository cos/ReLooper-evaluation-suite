28  package util;
29  
30  import java.lang.management.GarbageCollectorMXBean;
31  import java.lang.management.ManagementFactory;
32  import java.util.List;
33  import java.util.Map;
34  import java.util.HashMap;
35  
36  /**
37   * Class that implements alternative timing methods, such as measuring the interval
38   * between two events without considering any time spent in garbage collection. The
39   * implementation is not thread-safe.
40   */
41  public final class Time {
42  
43    private final static List<GarbageCollectorMXBean> garbageCollectorMXBeans;
44    private static long counter;
45    private static Map<Long, Pair<Long, Long>> times;
46  
47    static {
48      garbageCollectorMXBeans = ManagementFactory.getGarbageCollectorMXBeans();
49      counter = Long.MIN_VALUE;
50      times = new HashMap<Long, Pair<Long, Long>>();
51    }
52  
53    /**
54     * Starts a new time counter, returning its identifier.
55     * @return An id identifying the current counter.
56     */
57    public static long getNewTimeId() {
58      return getNewTimeId(5);
59    }
60  
61    /**
62     * Starts a new time counter, returning its identifier.
63     * @param collections Number of invocations to System.gc() prior to start the timer.
64     * @return An id identifying the current counter.
65     */
66    public static long getNewTimeId(int collections) {
67      // Before gathering any time statistics, it is useful to run the garbage collector
68      // so it does not interfere with the results.
69      for (int i = 0; i < collections; i++) {
70        System.gc();
71      }
72      counter++;
73      times.put(counter, new Pair<Long, Long>(milliTime(), milliGcTime()));
74      return counter;
75    }
76  
77    /**
78     * Returns the elapsed time since the given counter was created. The returned value
79     * does not include any time spent by the JVM in garbage collection.
80     * @param id Id generated by getNewTimeId.
81     * @return Elapsed time, in milliseconds.
82     */
83    public static long elapsedTime(long id) {
84      Pair<Long, Long> startTimes = times.get(id);
85      return elapsedTime(startTimes.getFirst(), startTimes.getSecond());
86    }
87  
88    private static long elapsedTime(long startTime, long startGcTime) {
89      return milliTime() - startTime - milliGcTime() + startGcTime;
90    }
91  
92    /**
93     * Returns the elapsed time since the given counter was created. The user can choose whether to
94     * include the time spent by the JVM in garbage collection or not.
95     * @param id Id generated by getNewTimeId.
96     * @param Whether to include the time spent in garbage collection in the result or not.
97     * @return Elapsed time, in milliseconds.
98     */
99    public static long elapsedTime(long id, boolean includeGc) {
100     Pair<Long, Long> elapsedGcTimes = elapsedAndGcTime(id);
101     long elapsedTime = elapsedGcTimes.getFirst();
102     long gcTime = elapsedGcTimes.getSecond();
103     return includeGc ? elapsedTime : elapsedTime - gcTime;
104   }
105 
106   /**
107    * Returns both the total and the GC elapsed times since the counter was created.
108    * @param id Id generated by getNewTimeId.
109    * @return Total and GC elapsed times, in milliseconds. The GC time is always smaller or equal
110    * than the total time.
111    */
112   public static Pair<Long, Long> elapsedAndGcTime(long id) {
113     // first take times
114     long milliTime = milliTime();
115     long milliGcTime = milliGcTime();
116     // now do the calculations
117     Pair<Long, Long> startTimes = times.get(id);
118     long startTime = startTimes.getFirst();
119     long startGcTime = startTimes.getSecond();
120     return new Pair<Long, Long>(milliTime - startTime, milliGcTime - startGcTime);
121   }
122 
123   private static long milliTime() {
124     return System.nanoTime() / 1000000;
125   }
126 
127   public static long milliGcTime() {
128     long result = 0;
129     for (GarbageCollectorMXBean garbageCollectorMXBean : garbageCollectorMXBeans) {
130       // the collection time is already in milliseconds,
131       result += Math.max(0, garbageCollectorMXBean.getCollectionTime());
132     }
133     return result;
134   }
135 }
