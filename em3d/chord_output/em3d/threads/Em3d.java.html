2   /**
3    * Java implementation of the <tt>em3d</tt> Olden benchmark.  This Olden
4    * benchmark models the propagation of electromagnetic waves through
5    * objects in 3 dimensions. It is a simple computation on an irregular
6    * bipartite graph containing nodes representing electric and magnetic
7    * field values.
8    * <p/>
9    * <p><cite>
10   * D. Culler, A. Dusseau, S. Goldstein, A. Krishnamurthy, S. Lumetta, T. von
11   * Eicken and K. Yelick. "Parallel Programming in Split-C".  Supercomputing
12   * 1993, pages 262-273.
13   * </cite>
14   */
15  public class Em3d {
16  
17    static int nodect = 50;
18    static int conn = 10;
19    static int numiter = 3;
20  
21    /**
22     * The main roitine that creates the irregular, linked data structure
23     * that represents the electric and magnetic fields and propagates the
24     * waves through the graph.
25     *
26     * @param args the command line arguments
27     */
28    public static void main(String[] args) {
29      BiGraph graph = BiGraph.create(nodect, conn);
30  
31      for (int i = 0; i < numiter; i++) {
32        try {
33          graph.compute();
34      } catch (InterruptedException e) {
35          // TODO Auto-generated catch block
36          e.printStackTrace();
37      }
38      }
39  
40      graph.printGraph();
41    }
42  }
