1   package em3d.threads;
2   import java.util.*;
3   
4   /**
5    * This class implements nodes (both E- and H-nodes) of the EM graph. Sets
6    * up random neighbors and propagates field values among neighbors.
7    */
8   class ENode {
9     /**
10     * The value of the node.
11     */
12    double value;
13    /**
14     * Array of nodes to which we send our value.
15     */
16    ENode[] toNodes;
17    /**
18     * Array of nodes from which we receive values.
19     */
20    ENode[] fromNodes;
21    /**
22     * Coefficients on the fromNodes edges
23     */
24    double[] coeffs;
25    /**
26     * The number of fromNodes edges
27     */
28    int fromCount;
29    /**
30     * Used to create the fromEdges - keeps track of the number of edges that have
31     * been added
32     */
33    int fromLength;
34  
35    /**
36     * Constructor for a node with given `degree'.   The value of the
37     * node is initialized to a random value.
38     */
39    public ENode(int degree) {
40      value = MathFP.randDouble();
41      // create empty array for holding toNodes
42      toNodes = new ENode[degree];
43      fromNodes = null;
44      coeffs = null;
45      fromCount = 0;
46      fromLength = 0;
47    }
48  
49    /**
50     * Create the linked list of E or H nodes.  We create a table which is used
51     * later to create links among the nodes.
52     *
53     * @param size   the no. of nodes to create
54     * @param degree the out degree of each node
55     * @return a table containing all the nodes.
56     */
57    static ENode[] fillTable(int size, int degree) {
58      ENode[] table = new ENode[size];
59      for (int i = 0; i < size; i++) {
60        table[i] = new ENode(degree);
61      }
62  
63      return table;
64    }
65  
66    /**
67     * Create unique `degree' neighbors from the nodes given in nodeTable.
68     * We do this by selecting a random node from the give nodeTable to
69     * be neighbor. If this neighbor has been previously selected, then
70     * a different random neighbor is chosen.
71     *
72     * @param nodeTable the list of nodes to choose from.
73     */
74    void makeUniqueNeighbors(ENode[] nodeTable) {
75      for (int filled = 0; filled < toNodes.length; filled++) {
76        int k;
77        ENode otherNode = null;
78  
79        do {
80          // generate a random number in the correct range
81          int index = MathI.randInt();
82          if (index < 0) {
83            index = -index;
84          }
85  
86          index = index % nodeTable.length;
87  
88          // find a node with the random index in the given table
89          otherNode = (ENode) nodeTable[index];
90  
91          for (k = 0; k < filled; k++) {
92            if (otherNode == toNodes[filled]) {
93              break;
94            }
95          }
96        } while (k < filled);
97  
98        // other node is definitely unique among "filled" toNodes
99        toNodes[filled] = otherNode;
100 
101       // update fromCount for the other node
102       otherNode.fromCount = otherNode.fromCount + 1;
103     }
104   }
105 
106   /**
107    * Allocate the right number of FromNodes for this node. This
108    * step can only happen once we know the right number of from nodes
109    * to allocate. Can be done after unique neighbors are created and known.
110    * <p/>
111    * It also initializes random coefficients on the edges.
112    */
113   void makeFromNodes() {
114     fromNodes = new ENode[fromCount]; // nodes fill be filled in later
115     coeffs = new double[fromCount];
116   }
117 
118   /**
119    * Fill in the fromNode field in "other" nodes which are pointed to
120    * by this node.
121    */
122   void updateFromNodes() {
123     int i;
124     for (i = 0; i < this.toNodes.length; ++i) {
125       ENode otherNode = this.toNodes[i];
126       int count = otherNode.fromLength;
127       otherNode.fromLength = otherNode.fromLength + 1;
128       otherNode.fromNodes[count] = this;
129       otherNode.coeffs[count] = MathFP.randDouble();
130     }
131   }
132 
133   /**
134    * Get the new value of the current node based on its neighboring
135    * from_nodes and coefficients.
136    */
137   void computeNewValue() {
138     for (int i = 0; i < fromCount; i++) {
139       value -= coeffs[i] * fromNodes[i].value;
140     }
141   }
142 
143   public void printNode() {
144     System.out.print("value " + value + ", from_count " + fromCount);
145     }
146 }
