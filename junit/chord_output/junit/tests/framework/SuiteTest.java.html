13   */
14  public class SuiteTest extends TestCase {
15      protected TestResult fResult;
16      public SuiteTest(String name) {
17          super(name);
18      }
19      @Override
20      protected void setUp() {
21          fResult= new TestResult(); 
22      }
23      public static Test suite() {
24          TestSuite suite= new TestSuite("Suite Tests");
25          // build the suite manually, because some of the suites are testing
26          // the functionality that automatically builds suites
27          suite.addTest(new SuiteTest("testNoTestCases"));
28          suite.addTest(new SuiteTest("testOneTestCase"));
29          suite.addTest(new SuiteTest("testNotPublicTestCase"));
30          suite.addTest(new SuiteTest("testNotVoidTestCase"));
31          suite.addTest(new SuiteTest("testNotExistingTestCase"));
32          suite.addTest(new SuiteTest("testInheritedTests"));
33          suite.addTest(new SuiteTest("testOneTestCaseEclipseSeesSameStructureAs381"));       
34          suite.addTest(new SuiteTest("testNoTestCaseClass"));
35          suite.addTest(new SuiteTest("testShadowedTests"));
36          suite.addTest(new SuiteTest("testAddTestSuite"));
37          suite.addTest(new SuiteTest("testCreateSuiteFromArray"));
38          
39          return suite;
40      }
41      public void testInheritedTests() {
42          TestSuite suite= new TestSuite(InheritedTestCase.class);
43          suite.run(fResult);
44          assertTrue(fResult.wasSuccessful());
45          assertEquals(2, fResult.runCount());
46      }
47      public void testNoTestCaseClass() {
48          Test t= new TestSuite(NoTestCaseClass.class);
49          t.run(fResult);
50          assertEquals(1, fResult.runCount());  // warning test
51          assertTrue(! fResult.wasSuccessful());
52      }
53      public void testNoTestCases() {
54          Test t= new TestSuite(NoTestCases.class);
55          t.run(fResult);
56          assertTrue(fResult.runCount() == 1);  // warning test
57          assertTrue(fResult.failureCount() == 1);
58          assertTrue(! fResult.wasSuccessful());
59      }
60      public void testNotExistingTestCase() {
61          Test t= new SuiteTest("notExistingMethod");
62          t.run(fResult);
63          assertTrue(fResult.runCount() == 1);  
64          assertTrue(fResult.failureCount() == 1);
65          assertTrue(fResult.errorCount() == 0);
66      }
67      public void testNotPublicTestCase() {
68          TestSuite suite= new TestSuite(NotPublicTestCase.class);
69          // 1 public test case + 1 warning for the non-public test case
70          assertEquals(2, suite.countTestCases());
71      }
72      public void testNotVoidTestCase() {
73          TestSuite suite= new TestSuite(NotVoidTestCase.class);
74          assertTrue(suite.countTestCases() == 1);
75      }
76      public void testOneTestCase() {
77          TestSuite t= new TestSuite(OneTestCase.class);
78          t.run(fResult);
79          assertTrue(fResult.runCount() == 1);  
80          assertTrue(fResult.failureCount() == 0);
81          assertTrue(fResult.errorCount() == 0);
82          assertTrue(fResult.wasSuccessful());
83      }
84      public void testOneTestCaseEclipseSeesSameStructureAs381() {
85          TestSuite t= new TestSuite(ThreeTestCases.class);
86          assertEquals(3, Collections.list(t.tests()).size());
87      }
88      public void testShadowedTests() {
89          TestSuite suite= new TestSuite(OverrideTestCase.class);
90          suite.run(fResult);
91          assertEquals(1, fResult.runCount());
92      }
93      public void testAddTestSuite() {
94          TestSuite suite= new TestSuite();
95          suite.addTestSuite(OneTestCase.class);
96          suite.run(fResult);
97          assertEquals(1, fResult.runCount());
98      }
99      public void testCreateSuiteFromArray() {
100         TestSuite suite = new TestSuite(OneTestCase.class, DoublePrecisionAssertTest.class);
101         assertEquals(2, suite.testCount());
102         assertEquals("junit.tests.framework.DoublePrecisionAssertTest" , ((TestSuite)suite.testAt(1)).getName());
103         assertEquals("junit.tests.framework.OneTestCase" , ((TestSuite)suite.testAt(0)).getName());
104     }
