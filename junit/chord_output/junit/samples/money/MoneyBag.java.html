1   package junit.samples.money;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * A MoneyBag defers exchange rate conversions. For example adding 
8    * 12 Swiss Francs to 14 US Dollars is represented as a bag 
9    * containing the two Monies 12 CHF and 14 USD. Adding another
10   * 10 Swiss francs gives a bag with 22 CHF and 14 USD. Due to 
11   * the deferred exchange rate conversion we can later value a 
12   * MoneyBag with different exchange rates.
13   *
14   * A MoneyBag is represented as a list of Monies and provides 
15   * different constructors to create a MoneyBag. 
16   */ 
17  public class MoneyBag implements IMoney {
18      private List<Money> fMonies= new ArrayList<Money>(5);
19  
20      public static IMoney create(IMoney m1, IMoney m2) {
21          MoneyBag result= new MoneyBag();
22          m1.appendTo(result);
23          m2.appendTo(result);
24          return result.simplify();
25      }
26      public IMoney add(IMoney m) {
27          return m.addMoneyBag(this);
28      }
29      public IMoney addMoney(Money m) { 
30          return MoneyBag.create(m, this);
31      }
32      public IMoney addMoneyBag(MoneyBag s) {
33          return MoneyBag.create(s, this);
34      }
35      void appendBag(MoneyBag aBag) {
36          for (Money each : aBag.fMonies)
37              appendMoney(each);
38      }
39      void appendMoney(Money aMoney) {
40          if (aMoney.isZero()) return;
41          IMoney old= findMoney(aMoney.currency());
42          if (old == null) {
43              fMonies.add(aMoney);
44              return;
45          }
46          fMonies.remove(old);
47          Money sum= (Money) old.add(aMoney);
48          if (sum.isZero()) 
49              return;
50          fMonies.add(sum);
51      }
52      @Override
53      public boolean equals(Object anObject) {
54          if (isZero())
55              if (anObject instanceof IMoney)
56                  return ((IMoney)anObject).isZero();
57  
58          if (anObject instanceof MoneyBag) {
59              MoneyBag aMoneyBag= (MoneyBag)anObject;
60              if (aMoneyBag.fMonies.size() != fMonies.size())
61                  return false;
62  
63              for (Money each : fMonies)
64                  if (! aMoneyBag.contains(each))
65                      return false;
66              return true;
67          }
68          return false;
69      }
70      private Money findMoney(String currency) {
71          for (Money each : fMonies)
72              if (each.currency().equals(currency))
73                  return each;
74          return null;
75      }
76      private boolean contains(Money m) {
77          Money found= findMoney(m.currency());
78          if (found == null) return false;
79          return found.amount() == m.amount();
80      }
81      @Override
82      public int hashCode() {
83          int hash= 0;
84          for (Money each : fMonies)
85              hash^= each.hashCode();
86          return hash;
87      }
88      public boolean isZero() {
89          return fMonies.size() == 0;
90      }
91      public IMoney multiply(int factor) {
92          MoneyBag result= new MoneyBag();
93          if (factor != 0)
94              for (Money each : fMonies)
95                  result.appendMoney((Money) each.multiply(factor));
96          return result;
97      }
98      public IMoney negate() {
99          MoneyBag result= new MoneyBag();
100         for (Money each : fMonies)
101             result.appendMoney((Money) each.negate());
102         return result;
103     }
104     private IMoney simplify() {
105         if (fMonies.size() == 1)
106             return fMonies.iterator().next();
107         return this;
108     }
109     public IMoney subtract(IMoney m) {
110         return add(m.negate());
111     }
112     @Override
113     public String toString() {
114         StringBuffer buffer = new StringBuffer();
115         buffer.append("{");
116         for (Money each : fMonies)
117             buffer.append(each);
118         buffer.append("}");
119         return buffer.toString();
120     }
121     public void appendTo(MoneyBag m) {
122         m.appendBag(this);
123     }
124 }
