38  /**
39   * Provides most of the functionality specific to a Runner that implements a
40   * "parent node" in the test tree, with children defined by objects of some data
41   * type {@code T}. (For {@link BlockJUnit4ClassRunner}, {@code T} is
42   * {@link Method} . For {@link Suite}, {@code T} is {@link Class}.) Subclasses
43   * must implement finding the children of the node, describing each child, and
44   * running each child. ParentRunner will filter and sort children, handle
45   * {@code @BeforeClass} and {@code @AfterClass} methods, 
46   * handle annotated {@link ClassRule}s, create a composite
47   * {@link Description}, and run children sequentially.
48   */
49  public abstract class ParentRunner<T> extends Runner implements Filterable,
50          Sortable {
51      private final TestClass fTestClass;
52  
53      private Sorter fSorter= Sorter.NULL;
54  
55      private List<T> fFilteredChildren= null;
56      
57      private RunnerScheduler fScheduler= new RunnerScheduler() { 
58          public void schedule(Runnable childStatement) {
59              childStatement.run();
60          }
61      
62          public void finished() {
63              // do nothing
64          }
65      };

67      /**
68       * Constructs a new {@code ParentRunner} that will run {@code @TestClass}
69       * @throws InitializationError 
70       */
71      protected ParentRunner(Class<?> testClass) throws InitializationError {
72          fTestClass= new TestClass(testClass);
73          validate();
74      }

103     /**
104      * Adds to {@code errors} a throwable for each problem noted with the test class (available from {@link #getTestClass()}).
105      * Default implementation adds an error for each method annotated with
106      * {@code @BeforeClass} or {@code @AfterClass} that is not
107      * {@code public static void} with no arguments.
108      */
109     protected void collectInitializationErrors(List<Throwable> errors) {
110         validatePublicVoidNoArgMethods(BeforeClass.class, true, errors);
111         validatePublicVoidNoArgMethods(AfterClass.class, true, errors);
112     }
113 
114     /**
115      * Adds to {@code errors} if any method in this class is annotated with
116      * {@code annotation}, but:
117      * <ul>
118      * <li>is not public, or
119      * <li>takes parameters, or
120      * <li>returns something other than void, or
121      * <li>is static (given {@code isStatic is false}), or
122      * <li>is not static (given {@code isStatic is true}).
123      */
124     protected void validatePublicVoidNoArgMethods(Class<? extends Annotation> annotation,
125             boolean isStatic, List<Throwable> errors) {
126         List<FrameworkMethod> methods= getTestClass().getAnnotatedMethods(annotation);
127 
128         for (FrameworkMethod eachTestMethod : methods)
129             eachTestMethod.validatePublicVoidNoArg(isStatic, errors);
130     }

132     /** 
133      * Constructs a {@code Statement} to run all of the tests in the test class. Override to add pre-/post-processing. 
134      * Here is an outline of the implementation:
135      * <ul>
136      * <li>Call {@link #runChild(Object, RunNotifier)} on each object returned by {@link #getChildren()} (subject to any imposed filter and sort).</li>
137      * <li>ALWAYS run all non-overridden {@code @BeforeClass} methods on this class
138      * and superclasses before the previous step; if any throws an
139      * Exception, stop execution and pass the exception on.
140      * <li>ALWAYS run all non-overridden {@code @AfterClass} methods on this class
141      * and superclasses before any of the previous steps; all AfterClass methods are
142      * always executed: exceptions thrown by previous steps are combined, if
143      * necessary, with exceptions from AfterClass methods into a
144      * {@link MultipleFailureException}.
145      * </ul>
146      * @param notifier
147      * @return {@code Statement}
148      */
149     protected Statement classBlock(final RunNotifier notifier) {
150         Statement statement= childrenInvoker(notifier);
151         statement= withBeforeClasses(statement);
152         statement= withAfterClasses(statement);
153         statement= withClassRules(statement);
154         return statement;
155     }
156 
157     /**
158      * Returns a {@link Statement}: run all non-overridden {@code @BeforeClass} methods on this class
159      * and superclasses before executing {@code statement}; if any throws an
160      * Exception, stop execution and pass the exception on.
161      */
162     protected Statement withBeforeClasses(Statement statement) {
163         List<FrameworkMethod> befores= fTestClass
164                 .getAnnotatedMethods(BeforeClass.class);
165         return befores.isEmpty() ? statement :
166             new RunBefores(statement, befores, null);
167     }
168 
169     /**
170      * Returns a {@link Statement}: run all non-overridden {@code @AfterClass} methods on this class
171      * and superclasses before executing {@code statement}; all AfterClass methods are
172      * always executed: exceptions thrown by previous steps are combined, if
173      * necessary, with exceptions from AfterClass methods into a
174      * {@link MultipleFailureException}.
175      */
176     protected Statement withAfterClasses(Statement statement) {
177         List<FrameworkMethod> afters= fTestClass
178                 .getAnnotatedMethods(AfterClass.class);
179         return afters.isEmpty() ? statement : 
180             new RunAfters(statement, afters, null);
181     }

183     /**
184      * Returns a {@link Statement}: apply all 
185      * static fields assignable to {@link TestRule}
186      * annotated with {@link ClassRule}.
187      *
188      * @param statement
189      *            the base statement
190      * @return a RunRules statement if any class-level {@link Rule}s are
191      *         found, or the base statement
192      */
193     private Statement withClassRules(Statement statement) {
194         List<TestRule> classRules= classRules();
195         return classRules.isEmpty() ? statement :
196             new RunRules(statement, classRules, getDescription());
197     }
198 
199     /**
200      * @return the {@code ClassRule}s that can transform the block that runs
201      *         each method in the tested class.
202      */
203     protected List<TestRule> classRules() {
204         List<TestRule> results= new ArrayList<TestRule>();
205         for (FrameworkField field : classRuleFields())
206             results.add(getClassRule(field));
207         return results;
208     }
209 
210     private TestRule getClassRule(final FrameworkField field) {
211         try {
212             return (TestRule) field.get(null);
213         } catch (IllegalAccessException e) {
214             throw new RuntimeException(
215                     "How did getAnnotatedFields return a field we couldn't access?");
216         }
217     }

219     /**
220      * @return list of {@link FrameworkField}s annotated with {@link Rule}
221      */
222     protected List<FrameworkField> classRuleFields() {
223         return fTestClass.getAnnotatedFields(ClassRule.class);
224     }

226     /**
227      * Returns a {@link Statement}: Call {@link #runChild(Object, RunNotifier)}
228      * on each object returned by {@link #getChildren()} (subject to any imposed
229      * filter and sort)
230      */
231     protected Statement childrenInvoker(final RunNotifier notifier) {
232         return new Statement() {
233             @Override
234             public void evaluate() {
235                 runChildren(notifier);
236             }
237         };
238     }
239 
240     private void runChildren(final RunNotifier notifier) {
241         for (final T each : getFilteredChildren())
242             fScheduler.schedule(new Runnable() {            
243                 public void run() {
244                     ParentRunner.this.runChild(each, notifier);
245                 }
246             });
247         fScheduler.finished();
248     }

268     /**
269      * Runs a {@link Statement} that represents a leaf (aka atomic) test.
270      */
271     protected final void runLeaf(Statement statement, Description description,
272             RunNotifier notifier) {
273         EachTestNotifier eachNotifier= new EachTestNotifier(notifier, description);
274         eachNotifier.fireTestStarted();
275         try {
276             statement.evaluate();
277         } catch (AssumptionViolatedException e) {
278             eachNotifier.addFailedAssumption(e);
279         } catch (Throwable e) {
280             eachNotifier.addFailure(e);
281         } finally {
282             eachNotifier.fireTestFinished();
283         }
284     }

290     @Override
291     public Description getDescription() {
292         Description description= Description.createSuiteDescription(getName(),
293                 fTestClass.getAnnotations());
294         for (T child : getFilteredChildren())
295             description.addChild(describeChild(child));
296         return description;
297     }
298 
299     @Override
300     public void run(final RunNotifier notifier) {
301         EachTestNotifier testNotifier= new EachTestNotifier(notifier,
302                 getDescription());
303         try {
304             Statement statement= classBlock(notifier);
305             statement.evaluate();
306         } catch (AssumptionViolatedException e) {
307             testNotifier.fireTestIgnored();
308         } catch (StoppedByUserException e) {
309             throw e;
310         } catch (Throwable e) {
311             testNotifier.addFailure(e);
312         }
313     }

319     public void filter(Filter filter) throws NoTestsRemainException {
320         for (Iterator<T> iter = getFilteredChildren().iterator(); iter.hasNext(); ) {
321             T each = iter.next();
322             if (shouldRun(filter, each))
323                 try {
324                     filter.apply(each);
325                 } catch (NoTestsRemainException e) {
326                     iter.remove();
327                 }
328             else
329                 iter.remove();
330         }
331         if (getFilteredChildren().isEmpty()) {
332             throw new NoTestsRemainException();
333         }
334     }
335 
336     public void sort(Sorter sorter) {
337         fSorter= sorter;
338         for (T each : getFilteredChildren())
339             sortChild(each);
340         Collections.sort(getFilteredChildren(), comparator());
341     }

347     private void validate() throws InitializationError {
348         List<Throwable> errors= new ArrayList<Throwable>();
349         collectInitializationErrors(errors);
350         if (!errors.isEmpty())
351             throw new InitializationError(errors);
352     }
353 
354     private List<T> getFilteredChildren() {
355         if (fFilteredChildren == null)
356             fFilteredChildren = new ArrayList<T>(getChildren());
357         return fFilteredChildren;
358     }
359 
360     private void sortChild(T child) {
361         fSorter.apply(child);
362     }
363 
364     private boolean shouldRun(Filter filter, T each) {
365         return filter.shouldRun(describeChild(each));
366     }
367 
368     private Comparator<? super T> comparator() {
369         return new Comparator<T>() {
370             public int compare(T o1, T o2) {
371                 return fSorter.compare(describeChild(o1), describeChild(o2));
372             }
373         };
374     }
