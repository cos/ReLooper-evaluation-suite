30  /**
31   * Implements the JUnit 4 standard test case class model, as defined by the
32   * annotations in the org.junit package. Many users will never notice this
33   * class: it is now the default test class runner, but it should have exactly
34   * the same behavior as the old test class runner ({@code JUnit4ClassRunner}).
35   * 
36   * BlockJUnit4ClassRunner has advantages for writers of custom JUnit runners
37   * that are slight changes to the default behavior, however:
38   * 
39   * <ul>
40   * <li>It has a much simpler implementation based on {@link Statement}s,
41   * allowing new operations to be inserted into the appropriate point in the
42   * execution flow.
43   * 
44   * <li>It is published, and extension and reuse are encouraged, whereas {@code
45   * JUnit4ClassRunner} was in an internal package, and is now deprecated.
46   * </ul>
47   */
48  public class BlockJUnit4ClassRunner extends ParentRunner<FrameworkMethod> {

54       *             if the test class is malformed.

57          super(klass);

64      @Override
65      protected void runChild(final FrameworkMethod method, RunNotifier notifier) {
66          Description description= describeChild(method);
67          if (method.getAnnotation(Ignore.class) != null) {
68              notifier.fireTestIgnored(description);
69          } else {
70              runLeaf(methodBlock(method), description, notifier);
71          }
72      }

74      @Override
75      protected Description describeChild(FrameworkMethod method) {
76          return Description.createTestDescription(getTestClass().getJavaClass(),
77                  testName(method), method.getAnnotations());
78      }

82          return computeTestMethods();

108      * Adds to {@code errors} if the test class has more than one constructor,
109      * or if the constructor takes parameters. Override if a subclass requires
110      * different validation rules.
111      */
112     protected void validateConstructor(List<Throwable> errors) {
113         validateOnlyOneConstructor(errors);
114         validateZeroArgConstructor(errors);
115     }

117     /**
118      * Adds to {@code errors} if the test class has more than one constructor
119      * (do not override)
120      */
121     protected void validateOnlyOneConstructor(List<Throwable> errors) {
122         if (!hasOneConstructor()) {
123             String gripe= "Test class should have exactly one public constructor";
124             errors.add(new Exception(gripe));
125         }
126     }
127 
128     /**
129      * Adds to {@code errors} if the test class's single constructor takes
130      * parameters (do not override)
131      */
132     protected void validateZeroArgConstructor(List<Throwable> errors) {
133         if (hasOneConstructor()
134                 && !(getTestClass().getOnlyConstructor().getParameterTypes().length == 0)) {
135             String gripe= "Test class should have exactly one public zero-argument constructor";
136             errors.add(new Exception(gripe));
137         }
138     }
139 
140     private boolean hasOneConstructor() {
141         return getTestClass().getJavaClass().getConstructors().length == 1;
142     }

144     /**
145      * Adds to {@code errors} for each method annotated with {@code @Test},
146      * {@code @Before}, or {@code @After} that is not a public, void instance
147      * method with no arguments.
148      * 
149      * @deprecated unused API, will go away in future version
150      */
151     @Deprecated
152     protected void validateInstanceMethods(List<Throwable> errors) {
153         validatePublicVoidNoArgMethods(After.class, false, errors);
154         validatePublicVoidNoArgMethods(Before.class, false, errors);
155         validateTestMethods(errors);
156 
157         if (computeTestMethods().size() == 0)
158             errors.add(new Exception("No runnable methods"));
159     }

161     private void validateFields(List<Throwable> errors) {
162         for (FrameworkField each : getTestClass()
163                 .getAnnotatedFields(Rule.class))
164             validateRuleField(each.getField(), errors);
165     }
166 
167     private void validateRuleField(Field field, List<Throwable> errors) {
168         Class<?> type= field.getType();
169         if (!isMethodRule(type) && !isTestRule(type))
170             errors.add(new Exception("Field " + field.getName()
171                     + " must implement MethodRule or TestRule"));
172         if (!Modifier.isPublic(field.getModifiers()))
173             errors.add(new Exception("Field " + field.getName()
174                     + " must be public"));
175     }
176 
177     private boolean isTestRule(Class<?> type) {
178         return TestRule.class.isAssignableFrom(type);
179     }
180 
181     @SuppressWarnings("deprecation")
182     private boolean isMethodRule(Class<?> type) {
183         return org.junit.rules.MethodRule.class.isAssignableFrom(type);
184     }

188      * is not a public, void instance method with no arguments.

190     protected void validateTestMethods(List<Throwable> errors) {
191         validatePublicVoidNoArgMethods(Test.class, false, errors);

196      * the test class's no-argument constructor (validation should have ensured

211     /**
212      * Returns a Statement that, when executed, either returns normally if
213      * {@code method} passes, or throws an exception if {@code method} fails.
214      * 
215      * Here is an outline of the default implementation:
216      * 
217      * <ul>
218      * <li>Invoke {@code method} on the result of {@code createTest()}, and
219      * throw any exceptions thrown by either operation.
220      * <li>HOWEVER, if {@code method}'s {@code @Test} annotation has the {@code
221      * expecting} attribute, return normally only if the previous step threw an
222      * exception of the correct type, and throw an exception otherwise.
223      * <li>HOWEVER, if {@code method}'s {@code @Test} annotation has the {@code
224      * timeout} attribute, throw an exception if the previous step takes more
225      * than the specified number of milliseconds.
226      * <li>ALWAYS run all non-overridden {@code @Before} methods on this class
227      * and superclasses before any of the previous steps; if any throws an
228      * Exception, stop execution and pass the exception on.
229      * <li>ALWAYS run all non-overridden {@code @After} methods on this class
230      * and superclasses after any of the previous steps; all After methods are
231      * always executed: exceptions thrown by previous steps are combined, if
232      * necessary, with exceptions from After methods into a
233      * {@link MultipleFailureException}.
234      * <li>ALWAYS allow {@code @Rule} fields to modify the execution of the
235      * above steps. A {@code Rule} may prevent all execution of the above steps,
236      * or add additional behavior before and after, or modify thrown exceptions.
237      * For more information, see {@link TestRule}
238      * </ul>
239      * 
240      * This can be overridden in subclasses, either by overriding this method,
241      * or the implementations creating each sub-statement.
242      */
243     protected Statement methodBlock(FrameworkMethod method) {
244         Object test;
245         try {
246             test= new ReflectiveCallable() {
247                 @Override
248                 protected Object runReflectiveCall() throws Throwable {
249                     return createTest();
250                 }
251             }.run();
252         } catch (Throwable e) {
253             return new Fail(e);
254         }
255 
256         Statement statement= methodInvoker(method, test);
257         statement= possiblyExpectingExceptions(method, test, statement);
258         statement= withPotentialTimeout(method, test, statement);
259         statement= withBefores(method, test, statement);
260         statement= withAfters(method, test, statement);
261         statement= withRules(method, test, statement);
262         return statement;
263     }

276     /**
277      * Returns a {@link Statement}: if {@code method}'s {@code @Test} annotation
278      * has the {@code expecting} attribute, return normally only if {@code next}
279      * throws an exception of the correct type, and throw an exception
280      * otherwise.
281      * 
282      * @deprecated Will be private soon: use Rules instead
283      */
284     @Deprecated
285     protected Statement possiblyExpectingExceptions(FrameworkMethod method,
286             Object test, Statement next) {
287         Test annotation= method.getAnnotation(Test.class);
288         return expectsException(annotation) ? new ExpectException(next,
289                 getExpectedException(annotation)) : next;
290     }

292     /**
293      * Returns a {@link Statement}: if {@code method}'s {@code @Test} annotation
294      * has the {@code timeout} attribute, throw an exception if {@code next}
295      * takes more than the specified number of milliseconds.
296      * 
297      * @deprecated Will be private soon: use Rules instead
298      */
299     @Deprecated
300     protected Statement withPotentialTimeout(FrameworkMethod method,
301             Object test, Statement next) {
302         long timeout= getTimeout(method.getAnnotation(Test.class));
303         return timeout > 0 ? new FailOnTimeout(next, timeout) : next;
304     }

307      * Returns a {@link Statement}: run all non-overridden {@code @Before}

340     private Statement withRules(FrameworkMethod method, Object target,
341             Statement statement) {
342         Statement result= statement;
343         result= withMethodRules(method, target, result);
344         result= withTestRules(method, target, result);
345         return result;
346     }

348     @SuppressWarnings("deprecation")
349     private Statement withMethodRules(FrameworkMethod method, Object target,
350             Statement result) {
351         List<TestRule> testRules= getTestRules(target);
352         for (org.junit.rules.MethodRule each : getMethodRules(target))
353             if (! testRules.contains(each))
354                 result= each.apply(result, method, target);
355         return result;
356     }

358     @SuppressWarnings("deprecation")
359     private List<org.junit.rules.MethodRule> getMethodRules(Object target) {
360         return getTestClass().getAnnotatedFieldValues(target,
361                 Rule.class, org.junit.rules.MethodRule.class);
362     }

364     /**
365      * Returns a {@link Statement}: apply all non-static {@link Value} fields
366      * annotated with {@link Rule}.
367      *
368      * @param statement The base statement
369      * @return a RunRules statement if any class-level {@link Rule}s are
370      *         found, or the base statement
371      */
372     private Statement withTestRules(FrameworkMethod method, Object target,
373             Statement statement) {
374         List<TestRule> testRules= getTestRules(target);
375         return testRules.isEmpty() ? statement :
376             new RunRules(statement, testRules, describeChild(method));
377     }
378 
379     private List<TestRule> getTestRules(Object target) {
380         return getTestClass().getAnnotatedFieldValues(target,
381                 Rule.class, TestRule.class);
382     }

384     private Class<? extends Throwable> getExpectedException(Test annotation) {
385         if (annotation == null || annotation.expected() == None.class)
386             return null;
387         else
388             return annotation.expected();
389     }
390 
391     private boolean expectsException(Test annotation) {
392         return getExpectedException(annotation) != null;
393     }

395     private long getTimeout(Test annotation) {
396         if (annotation == null)
397             return 0;
398         return annotation.timeout();
