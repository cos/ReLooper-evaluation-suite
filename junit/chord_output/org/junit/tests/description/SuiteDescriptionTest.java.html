8   public class SuiteDescriptionTest {
9       Description childless = Description.createSuiteDescription("a");
10      Description anotherChildless = Description.createSuiteDescription("a");
11      Description namedB = Description.createSuiteDescription("b");
12      
13      Description twoKids = descriptionWithTwoKids("foo", "bar");
14      Description anotherTwoKids = descriptionWithTwoKids("foo", "baz");
15  
16      @Test public void equalsIsCorrect() {   
17          assertEquals(childless, anotherChildless);
18          assertFalse(childless.equals(namedB));
19          assertEquals(childless, twoKids);
20          assertEquals(twoKids, anotherTwoKids);
21          assertFalse(twoKids.equals(new Integer(5)));
22      }
23  
24      @Test public void hashCodeIsReasonable() {
25          assertEquals(childless.hashCode(), anotherChildless.hashCode());
26          assertFalse(childless.hashCode() == namedB.hashCode());
27      }
28      
29      private Description descriptionWithTwoKids(String first, String second) {
30          Description twoKids = Description.createSuiteDescription("a");
31          twoKids.addChild(Description.createTestDescription(getClass(), first));
32          twoKids.addChild(Description.createTestDescription(getClass(), second));
33          return twoKids;
34      }
