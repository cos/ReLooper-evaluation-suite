3   package org.junit.tests.assertion;
4   
5   import static org.hamcrest.CoreMatchers.equalTo;
6   import static org.junit.Assert.assertThat;
7   import static org.junit.Assert.assertTrue;
8   import static org.junit.Assert.fail;
9   
10  import java.util.ArrayList;
11  import java.util.Collections;
12  import java.util.List;
13  
14  import org.junit.Test;
15  import org.junit.runners.model.MultipleFailureException;
16  
17  /**
18   * Tests for {@link org.junit.runners.model.MultipleFailureException}
19   *
20   * @author kcooney@google.com (Kevin Cooney)
21   */
22  public class MultipleFailureExceptionTest {
23  
24      @Test
25      public void assertEmptyDoesNotThrowForEmptyList() throws Throwable {
26          MultipleFailureException.assertEmpty(Collections.<Throwable>emptyList());
27      }
28      
29      @Test(expected=ExpectedException.class)
30      public void assertEmptyRethrowsSingleThrowable() throws Throwable {
31          MultipleFailureException.assertEmpty(
32                  Collections.<Throwable>singletonList(new ExpectedException("pesto")));
33      }
34      
35      @Test
36      public void assertEmptyThrowsMutipleFailureExceptionForManyThrowables() throws Throwable {
37          List<Throwable> errors = new ArrayList<Throwable>();
38          errors.add(new ExpectedException("basil"));
39          errors.add(new RuntimeException("garlic"));
40          
41          try {
42              MultipleFailureException.assertEmpty(errors);
43              fail();
44          } catch (MultipleFailureException expected) {
45              assertThat(expected.getFailures(), equalTo(errors));
46              assertTrue(expected.getMessage().startsWith("There were 2 errors:\n"));
47              assertTrue(expected.getMessage().contains("ExpectedException(basil)\n"));
48              assertTrue(expected.getMessage().contains("RuntimeException(garlic)"));
49          }
50      }
51  
52  
53      private static class ExpectedException extends RuntimeException {
54          private static final long serialVersionUID = 1L;
55  
56          public ExpectedException(String message) {
57              super(message);
58          }
59      }
60  }
