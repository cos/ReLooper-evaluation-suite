28      @Test

31      }

33      @Theory
34      public void bothFails(int value, Matcher<Integer> first,
35              Matcher<Integer> second) {
36          assumeTrue(!(first.matches(value) && second.matches(value)));
37          assertThat(value, not(both(first).and(second)));
38      }
39  
40      @Theory
41      public <T> void descriptionIsSensible(Matcher<T> first, Matcher<T> second) {
42          Matcher<?> both= both(first).and(second);
43          assertThat(both.toString(), containsString(first.toString()));
44          assertThat(both.toString(), containsString(second.toString()));
45      }
46  
47      @Test
48      public void eitherPasses() {
49          assertThat(3, either(is(3)).or(is(4)));
50      }
51  
52      @Theory
53      public <T> void threeAndsWork(Matcher<Integer> first,
54              Matcher<Integer> second, Matcher<Integer> third, int value) {
55          assumeTrue(first.matches(value) && second.matches(value)
56                  && third.matches(value));
57          assertThat(value, both(first).and(second).and(third));
58      }
59  
60      @Theory
61      public <T> void threeOrsWork(Matcher<Integer> first,
62              Matcher<Integer> second, Matcher<Integer> third, int value) {
63          assumeTrue(first.matches(value) || second.matches(value)
64                  || third.matches(value));
65          assertThat(value, either(first).or(second).or(third));
66      }
67      
68      @Test public void subclassesAreOkInSecondPositionOnly() {
69          assertThat(3, both(is(Integer.class)).and(is(3)));
