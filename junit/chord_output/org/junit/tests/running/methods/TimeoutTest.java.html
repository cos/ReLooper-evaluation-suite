123     private long runAndTime(Class<?> clazz) {
124         JUnitCore core= new JUnitCore();
125         long startTime= System.currentTimeMillis();
126         core.run(clazz);
127         long totalTime = System.currentTimeMillis() - startTime;
128         return totalTime;
129     }
