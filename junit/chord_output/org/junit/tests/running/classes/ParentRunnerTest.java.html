39          log= "";
40          ParentRunner<?> runner= new BlockJUnit4ClassRunner(FruitTest.class);
41          runner.setScheduler(new RunnerScheduler() {
42              public void schedule(Runnable childStatement) {
43                  log+= "before ";
44                  childStatement.run();
45                  log+= "after ";
46              }
47  
48              public void finished() {
49                  log+= "afterAll ";
50              }
51          });
52  
53          runner.run(new RunNotifier());
54          assertEquals("before apple after before banana after afterAll ", log);
55      }

57      @Test
58      public void testMultipleFilters() throws Exception {
59          JUnitCore junitCore= new JUnitCore();
60          Request request= Request.aClass(ExampleTest.class);
61          Request requestFiltered= request.filterWith(new Exclude("test1"));
62          Request requestFilteredFiltered= requestFiltered
63                  .filterWith(new Exclude("test2"));
64          Result result= junitCore.run(requestFilteredFiltered);
65          assertThat(result.getFailures(), isEmpty());
66          assertEquals(1, result.getRunCount());
67      }

69      private Matcher<List<?>> isEmpty() {
70          return new TypeSafeMatcher<List<?>>() {
71              public void describeTo(org.hamcrest.Description description) {
72                  description.appendText("is empty");
73              }
74  
75              @Override
76              public boolean matchesSafely(List<?> item) {
77                  return item.size() == 0;
78              }
79          };
80      }
