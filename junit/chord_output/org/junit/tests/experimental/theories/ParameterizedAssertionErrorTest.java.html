11  import org.junit.experimental.theories.Theory;
12  import org.junit.experimental.theories.internal.ParameterizedAssertionError;
13  import org.junit.runner.RunWith;
14  
15  @RunWith(Theories.class)
16  public class ParameterizedAssertionErrorTest {
17      @DataPoint
18      public static final String METHOD_NAME= "methodName";
19  
20      @DataPoint
21      public static final NullPointerException NULL_POINTER_EXCEPTION= new NullPointerException();
22  
23      @DataPoint
24      public static Object[] NO_OBJECTS= new Object[0];
25  
26      @DataPoint
27      public static ParameterizedAssertionError A= new ParameterizedAssertionError(
28              NULL_POINTER_EXCEPTION, METHOD_NAME);
29  
30      @DataPoint
31      public static ParameterizedAssertionError B= new ParameterizedAssertionError(
32              NULL_POINTER_EXCEPTION, METHOD_NAME);
33  
34      @DataPoint
35      public static ParameterizedAssertionError B2= new ParameterizedAssertionError(
36              NULL_POINTER_EXCEPTION, "methodName2");
37  
38      @Theory
39      public void equalParameterizedAssertionErrorsHaveSameToString(
40              ParameterizedAssertionError a, ParameterizedAssertionError b) {
41          assumeThat(a, is(b));
42          assertThat(a.toString(), is(b.toString()));
43      }
44  
45      @Theory
46      public void differentParameterizedAssertionErrorsHaveDifferentToStrings(
47              ParameterizedAssertionError a, ParameterizedAssertionError b) {
48          assumeThat(a, not(b));
49          assertThat(a.toString(), not(b.toString()));
50      }
51  
52      @Theory
53      public void equalsReturnsTrue(Throwable targetException, String methodName,
54              Object[] params) {
55          assertThat(new ParameterizedAssertionError(targetException, methodName,
56                  params), is(new ParameterizedAssertionError(targetException,
57                  methodName, params)));
58      }
59  
60      @Theory(nullsAccepted= false)
61      public void buildParameterizedAssertionError(String methodName, String param) {
62          assertThat(new ParameterizedAssertionError(new RuntimeException(),
63                  methodName, param).toString(), containsString(methodName));
64      }
65  
66      @Test
67      public void canJoinWhenToStringFails() {
68          assertThat(ParameterizedAssertionError.join(" ", new Object() {
69              @Override
70              public String toString() {
71                  throw new UnsupportedOperationException();
72              }
73          }), is("[toString failed]"));
74      }
75  }
