17  
18  @RunWith(Theories.class)
19  public class PrintableResultTest {
20      @SuppressWarnings("unchecked")
21      @Theory(nullsAccepted= false)
22      public void backTraceHasGoodToString(String descriptionName,
23              final String stackTraceClassName) {
24          Failure failure= new Failure(Description
25                  .createSuiteDescription(descriptionName), new Throwable() {
26              private static final long serialVersionUID= 1L;
27  
28              @Override
29              public StackTraceElement[] getStackTrace() {
30                  return new StackTraceElement[] { new StackTraceElement(
31                          stackTraceClassName, "methodName", "fileName", 1) };
32              }
33          });
34  
35          assertThat(new PrintableResult(asList(failure)).toString(), allOf(
36                  containsString(descriptionName), containsString(stackTraceClassName)));
37      }
38  
39      @DataPoint
40      public static String SHELL_POINT= "Shell Point";
41  
42      @Theory
43      public void includeMultipleFailures(String secondExceptionName) {
44          PrintableResult backtrace= new PrintableResult(Arrays.asList(
45                  new Failure(Description.createSuiteDescription("firstName"),
46                          new RuntimeException("firstException")), new Failure(
47                          Description.createSuiteDescription("secondName"),
48                          new RuntimeException(secondExceptionName))));
49          assertThat(backtrace.toString(), containsString(secondExceptionName));
50      }
51  }
