6   import static org.junit.Assert.assertEquals;

17  /**
18   * Tests to exercise class-level rules.
19   */
20  public class ClassRulesTest {
21      public static class Counter extends ExternalResource {
22          public int count = 0;
23  
24          @Override
25          protected void before() throws Throwable {
26              count++;
27          }       
28      }
29      
30      public static class ExampleTestWithClassRule {
31          @ClassRule
32          public static Counter counter= new Counter();
33  
34          @Test
35          public void firstTest() {
36              assertEquals(1, counter.count);
37          }
38  
39          @Test
40          public void secondTest() {
41              assertEquals(1, counter.count);
42          }
43      }
44  
45      @Test
46      public void ruleIsAppliedOnce() {
47          ExampleTestWithClassRule.counter.count= 0;
48          JUnitCore.runClasses(ExampleTestWithClassRule.class);
49          assertEquals(1, ExampleTestWithClassRule.counter.count);
50      }
51  
52      public static class SubclassOfTestWithClassRule extends
53              ExampleTestWithClassRule {
54  
55      }
56  
57      @Test
58      public void ruleIsIntroducedAndEvaluatedOnSubclass() {
59          ExampleTestWithClassRule.counter.count= 0;
60          JUnitCore.runClasses(SubclassOfTestWithClassRule.class);
61          assertEquals(1, ExampleTestWithClassRule.counter.count);
62      }
63      
64      public static class CustomCounter implements TestRule {
65          public int count = 0;
66          
67          public Statement apply(final Statement base, Description description) {
68              return new Statement() {                
69                  @Override
70                  public void evaluate() throws Throwable {
71                      count++;
72                      base.evaluate();
73                  }
74              };
75          }       
76      }
77      
78      public static class ExampleTestWithCustomClassRule {
79          @ClassRule
80          public static CustomCounter counter= new CustomCounter();
81  
82          @Test
83          public void firstTest() {
84              assertEquals(1, counter.count);
85          }
86  
87          @Test
88          public void secondTest() {
89              assertEquals(1, counter.count);
90          }
91      }
92      
93  
94      @Test
95      public void customRuleIsAppliedOnce() {
96          ExampleTestWithCustomClassRule.counter.count= 0;
97          Result result= JUnitCore.runClasses(ExampleTestWithCustomClassRule.class);
98          assertTrue(result.wasSuccessful());
99          assertEquals(1, ExampleTestWithCustomClassRule.counter.count);
100     }
