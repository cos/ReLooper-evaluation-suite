7    * 
8    * <pre>

12   * 
13   *  &#064;Test

15   *      assertEquals(&quot;testA&quot;, name.getMethodName());
16   *  }
17   * 
18   *  &#064;Test
19   *  public void testB() {
20   *      assertEquals(&quot;testB&quot;, name.getMethodName());
21   *  }
22   * }
23   * </pre>
24   */
25  public class TestName extends TestWatcher {
26      private String fName;
27  
28      @Override

31      }
32  
33      /**
