283   /**
284     * Method for calculating the returns on a given rate path, via the
285     * definition for the instantaneous compounded return.
286     *       u_i = \ln{\frac{S_i}{S_{i-1}}}
287     * 
288     * @return the return, as defined.
289     * @exception DemoException thrown if there is a problem with the
290     *                          calculation.
291     */
292   public ReturnPath getReturnCompounded() throws DemoException {
293     if( pathValue == null || nAcceptedPathValue == 0 ) {
294       throw new DemoException("The Rate Path has not been defined!");
295     }
296     double[] returnPathValue = new double[nAcceptedPathValue];
297     returnPathValue[0] = 0.0;
298     try{
299       for(int i=1; i< nAcceptedPathValue; i++ ) {
300     returnPathValue[i] = Math.log(pathValue[i] / pathValue[i-1]);
301       }
302     } catch( ArithmeticException aex ) {
303       throw new DemoException("Error in getReturnLogarithm:"+aex.toString());
304     }
305     ReturnPath rPath = new ReturnPath(returnPathValue, nAcceptedPathValue, 
306     ReturnPath.COMPOUNDED);
307     //
308     // Copy the PathId information to the ReturnPath object.
309     rPath.copyInstanceVariables(this);
310     rPath.estimatePath();
311     return(rPath);
312   }
313   /**
314     * Method for calculating the returns on a given rate path, via the
315     * definition for the instantaneous non-compounded return.
316     *       u_i = \frac{S_i - S_{i-1}}{S_i}
317     * 
318     * @return the return, as defined.
319     * @exception DemoException thrown if there is a problem with the
320     *                          calculation.
321     */
322   public ReturnPath getReturnNonCompounded() throws DemoException {
323     if( pathValue == null || nAcceptedPathValue == 0 ) {
324       throw new DemoException("The Rate Path has not been defined!");
325     }
326     double[] returnPathValue = new double[nAcceptedPathValue];
327     returnPathValue[0] = 0.0;
328     try{
329       for(int i=1; i< nAcceptedPathValue; i++ ) {
330     returnPathValue[i] = (pathValue[i] - pathValue[i-1])/pathValue[i];
331       }
332     } catch( ArithmeticException aex ) {
333       throw new DemoException("Error in getReturnPercentage:"+aex.toString());
334     }
335     ReturnPath rPath = new ReturnPath(returnPathValue, nAcceptedPathValue, 
336     ReturnPath.NONCOMPOUNDED);
337     //
338     // Copy the PathId information to the ReturnPath object.
339     rPath.copyInstanceVariables(this);
340     rPath.estimatePath();
341     return(rPath);
342   }

346   /**
347     * Method for reading in data file, in a given format.
348     * Namely:
349       <pre>
350       881003,0.0000,14.1944,13.9444,14.0832,2200050,0
351       881004,0.0000,14.1668,14.0556,14.1668,1490850,0
352       ...
353       990108,35.8125,36.7500,35.5625,35.8125,4381200,0
354       990111,35.8125,35.8750,34.8750,35.1250,3920800,0
355       990112,34.8750,34.8750,34.0000,34.0625,3577500,0
356       </pre>
357     * <p>Where the fields represent, one believes, the following:
358     * <ol>
359     *   <li>The date in 'YYMMDD' format</li>
360     *   <li>Open</li>
361     *   <li>High</li>
362     *   <li>Low</li>
363     *   <li>Last</li>
364     *   <li>Volume</li>
365     *   <li>Open Interest</li>
366     * </ol>
367     * One will probably make use of the closing price, but this can be
368     * redefined via the class variable <code>DATUMFIELD</code>.  Note that
369     * since the read in data are then used to compute the return, this would
370     * be a good place to trap for zero values in the data, which will cause
371     * all sorts of problems.
372     *
373     * @param dirName the directory in which to search for the data file.
374     * @param filename the data filename itself.
375     * @exception DemoException thrown if there was a problem with the data
376     *                          file.
377     */
378   private void readRatesFile(String dirName, String filename) throws DemoException {
379     java.io.File ratesFile = new File(dirName, filename);
380     java.io.BufferedReader in;
381     if( ! ratesFile.canRead() ) {
382       throw new DemoException("Cannot read the file "+ratesFile.toString());
383     }
384     try{
385       in = new BufferedReader(new FileReader(ratesFile));
386     } catch( FileNotFoundException fnfex ) {
387       throw new DemoException(fnfex.toString());
388     }
389     //
390     // Proceed to read all the lines of data into a Vector object.
391     int iLine=0, initNlines=100, nLines=0;
392     
393     String aLine;
394     java.util.Vector allLines = new Vector(initNlines);
395     try{
396       while( (aLine = in.readLine()) != null ) {
397     iLine++;
398     //
399     // Note, I'm not entirely sure whether the object passed in is copied
400     // by value, or just its reference.
401     allLines.addElement(aLine);
402       }
403     } catch( IOException ioex ) {
404       throw new DemoException("Problem reading data from the file "+ioex.toString());
405     }
406     nLines = iLine;
407     //
408     // Now create an array to store the rates data.
409     this.pathValue = new double[nLines];
410     this.pathDate  = new int[nLines];
411     nAcceptedPathValue=0;
412     iLine=0;
413     for( java.util.Enumeration e = allLines.elements(); e.hasMoreElements(); ) {
414       aLine = (String) e.nextElement();
415       String[] field = Utilities.splitString(",",aLine);
416       int aDate = Integer.parseInt("19"+field[0]);
417       //
418       // static double Double.parseDouble() method is a feature of JDK1.2!
419       double aPathValue = Double.valueOf(field[DATUMFIELD]).doubleValue();
420       if( (aDate <= MINIMUMDATE) || (Math.abs(aPathValue) < EPSILON) ) {
421     dbgPrintln("Skipped erroneous data in "+filename+" indexed by date="+field[0]+".");
422       } else {
423     pathDate[iLine] = aDate;
424     pathValue[iLine] = aPathValue;
425     iLine++;
426       }
427     }
428     //
429     // Record the actual number of accepted data points.
430     nAcceptedPathValue = iLine;
431     //
432     // Now to fill in the structures from the 'PathId' class.
433     set_name(ratesFile.getName());
434     set_startDate(pathDate[0]);
435     set_endDate(pathDate[nAcceptedPathValue-1]);
436     set_dTime((double)(1.0/365.0));
437   }
