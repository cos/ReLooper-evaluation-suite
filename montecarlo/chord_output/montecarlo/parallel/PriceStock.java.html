23  package montecarlo.parallel;
24  /**
25    * Class to do the work in the Application demonstrator, in particular
26    * the pricing of the stock path generated by Monte Carlo.  The run
27    * method will generate a single sequence with the required statistics,
28    * estimate its volatility, expected return rate and final stock price
29    * value.
30    *
31    * @author H W Yau
32    * @version $Revision: 1.5 $ $Date: 1999/02/16 18:52:15 $
33    */
34  public class PriceStock extends Universal {
35  
36    //------------------------------------------------------------------------
37    // Class variables.
38    //------------------------------------------------------------------------
39    /**
40      * Class variable for determining whether to switch on debug output or
41      * not.
42      */
43    public static boolean DEBUG=true;
44    /**
45      * Class variable for defining the debug message prompt.
46      */
47    protected static String prompt="PriceStock> ";
48  
49    //------------------------------------------------------------------------
50    // Instance variables.
51    //------------------------------------------------------------------------
52    /**
53      * The Monte Carlo path to be generated.
54      */
55    private MonteCarloPath mcPath;
56    /**
57      * String identifier for a given task.
58      */
59    private String taskHeader;
60    /**
61      * Random seed from which the Monte Carlo sequence is started.
62      */
63    private long randomSeed=-1;
64    /**
65      * Initial stock price value.
66      */
67    private double pathStartValue=Double.NaN;
68    /**
69      * Object which represents the results from a given computation task.
70      */
71    private ToResult result;
72    private double expectedReturnRate=Double.NaN;
73    private double volatility=Double.NaN;
74    private double volatility2=Double.NaN;
75    private double finalStockPrice=Double.NaN;
76    private double[] pathValue;
77  
78    //------------------------------------------------------------------------
79    // Constructors.
80    //------------------------------------------------------------------------
81    /**
82      * Default constructor.
83      */
84    public PriceStock() {
85      super();
86      mcPath = new MonteCarloPath();
87      set_prompt(prompt);
88      set_DEBUG(DEBUG);
89    }
90    //------------------------------------------------------------------------
91    // Methods.
92    //------------------------------------------------------------------------
93    //------------------------------------------------------------------------
94    // Methods which implement the Slaveable interface.
95    //------------------------------------------------------------------------
96    /**
97      * Method which is passed in the initialisation data common to all tasks,
98      * and then unpacks them for use by this object.
99      *
100     * @param obj Object representing data which are common to all tasks.
101     */
102   public void setInitAllTasks(Object obj) {
103     ToInitAllTasks initAllTasks = (ToInitAllTasks) obj;
104     mcPath.set_name(initAllTasks.get_name());
105     mcPath.set_startDate(initAllTasks.get_startDate());
106     mcPath.set_endDate(initAllTasks.get_endDate());
107     mcPath.set_dTime(initAllTasks.get_dTime());
108     mcPath.set_returnDefinition(initAllTasks.get_returnDefinition());
109     mcPath.set_expectedReturnRate(initAllTasks.get_expectedReturnRate());
110     mcPath.set_volatility(initAllTasks.get_volatility());
111     int nTimeSteps = initAllTasks.get_nTimeSteps();
112     mcPath.set_nTimeSteps(nTimeSteps);
113     this.pathStartValue = initAllTasks.get_pathStartValue();
114     mcPath.set_pathStartValue(pathStartValue);
115     mcPath.set_pathValue(new double[nTimeSteps]);
116     mcPath.set_fluctuations(new double[nTimeSteps]);
117   }
118   /**
119     * Method which is passed in the data representing each task, which then
120     * unpacks it for use by this object.
121     *
122     * @param obj Object representing the data which defines a given task.
123     */
124   public void setTask(Object obj) {
125     ToTask task = (ToTask) obj;
126     this.taskHeader     = task.get_header();
127     this.randomSeed     = task.get_randomSeed();
128   }
129   /**
130     * The business end.  Invokes the necessary computation routine, for a
131     * a given task.
132     */
133   public void run() {
134     try{
135       mcPath.computeFluctuationsGaussian(randomSeed);
136       mcPath.computePathValue(pathStartValue);
137       RatePath rateP = new RatePath(mcPath);
138       ReturnPath returnP = rateP.getReturnCompounded();
139       returnP.estimatePath();
140       expectedReturnRate = returnP.get_expectedReturnRate();
141       volatility = returnP.get_volatility();
142       volatility2 = returnP.get_volatility2();
143       finalStockPrice = rateP.getEndPathValue();
144       pathValue = mcPath.get_pathValue();
145     } catch( DemoException demoEx ) {
146       errPrintln(demoEx.toString());
147     }
148   }
149   /*
150    * Method which returns the results of a computation back to the caller.
151    *
152    * @return An object representing the computed results.
153    */
154   public Object getResult() {
155     String resultHeader = "Result of task with Header="+taskHeader+": randomSeed="+randomSeed+": pathStartValue="+pathStartValue;
156     ToResult res = new ToResult(resultHeader,expectedReturnRate,volatility,
157     volatility2,finalStockPrice,pathValue);
158     return (Object) res;
159   }
160 }
