34    /**
35      * Static method which behaves like the Unix `which' command.  OS
36      * specific dependencies are handled by the Java.lang.System
37      * properties.
38      *
39      * @param executable The executable to search for.
40      * @param pathEnv    The list of paths in which to search, in the style of the
41      *                   OS's PATH environment variable.
42      * @return           The full pathname of where the executable lives,
43      *                   or failing that the error message "<executable> not found.".
44      */
45    public static String which(String executable, String pathEnv) {
46      String executablePath;
47      String paths[];
48  
49      paths=splitString(System.getProperty("path.separator"),pathEnv);
50      for(int i=0; i<paths.length; i++ ) {
51        if( paths[i].length() > 0 ) {
52      java.io.File pathFile=new java.io.File(paths[i]);
53      if( pathFile.isDirectory() ) {
54        String filesInDirectory[];
55        filesInDirectory=pathFile.list();
56        for(int j=0; j<filesInDirectory.length; j++ ) {
57          if( DEBUG ) {
58            System.out.println("DBG: Matching "+filesInDirectory[j]);
59          }
60          if( filesInDirectory[j].equals(executable) ) {
61            executablePath=paths[i]+System.getProperty("file.separator")+executable;
62            return executablePath;
63          }
64        }
65      } else {
66        if( DEBUG ) {
67          System.out.println("DBG: path "+paths[i]+" is not a directory!");
68        }
69      }
70        }
71      } /* for i */
72      executablePath=executable+" not found.";
73      return executablePath;
74    }

76    /**
77      * Static method which behaves like the Perl join() function.
78      *
79      * @param joinChar    The character on which to join.
80      * @param stringArray The array of strings to join.
81      * @return            A string of the joined string array.
82      */
83    public static String joinString(String joinChar,String stringArray[]) {
84      return joinString(joinChar,stringArray,0);
85    }
86    /**
87      * Static method which behaves like the Perl join() function.
88      *
89      * @param joinChar    The character on which to join.
90      * @param stringArray The array of strings to join.
91      * @param index       The array index on which to start joining.
92      * @return            A string of the joined string array.
93      */
94    public static String joinString(String joinChar,String stringArray[],int index) {
95      String methodName="join";
96      StringBuffer tmpString;
97  
98      int nStrings = java.lang.reflect.Array.getLength(stringArray);
99      if( nStrings <= index ) {
100       tmpString = new StringBuffer();
101     } else {
102       tmpString = new StringBuffer(stringArray[index]);
103       for(int i=(index+1); i < nStrings; i++) {
104     tmpString.append(joinChar).append(stringArray[i]);
105       }
106     }
107     return tmpString.toString();
108   }
109   /**
110     * Static method which behaves like the Perl split() function.
111     *
112     * @param splitChar The character on which to split.
113     * @param arg       The string to be split.
114     * @return          A string array of the split string.
115     */
116   public static String[] splitString(String splitChar,String arg) {
117     String methodName="split";
118 
119     String myArgs[];
120     int nArgs=0;
121     int foundIndex=0, fromIndex=0;
122     
123     while( (foundIndex=arg.indexOf(splitChar,fromIndex)) > -1 ) {
124       nArgs++;
125       fromIndex=foundIndex+1;
126     }
127     if( DEBUG ) {
128       System.out.println("DBG "+className+"."+methodName+": "+nArgs);
129     }
130     myArgs = new String[nArgs+1];
131     nArgs=0;
132     fromIndex=0;
133     while( (foundIndex=arg.indexOf(splitChar,fromIndex)) > -1 ) {
134       if( DEBUG ) {
135     System.out.println("DBG "+className+"."+methodName+": "+fromIndex+" "+foundIndex);
136       }
137       myArgs[nArgs]=arg.substring(fromIndex,foundIndex);
138       nArgs++;
139       fromIndex=foundIndex+1;
140     }
141     myArgs[nArgs]=arg.substring(fromIndex);
142     return myArgs;
143   }
