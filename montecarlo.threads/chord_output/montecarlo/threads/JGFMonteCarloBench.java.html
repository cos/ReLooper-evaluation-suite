37    }

39    public void JGFinitialise(){

45    public void JGFapplication(){ 
46  
47      JGFInstrumentor.startTimer("Section3:MonteCarlo:Run");

57    public void JGFvalidate(){
58     double refval[] = {-0.0333976656762814,-0.03215796752868655};
59     double dev = Math.abs(AppDemo.JGFavgExpectedReturnRateMC - refval[size]);
60     if (dev > 1.0e-12 ){
61       System.out.println("Validation failed");
62       System.out.println(" expectedReturnRate= " + AppDemo.JGFavgExpectedReturnRateMC + "  " + dev + "  " + size);
63      }
64    }
65  
66    public void JGFtidyup(){    
67  
68      System.gc();
69    }
70  
71    public static void main(String[] args) {
72        JGFMonteCarloBench a = new JGFMonteCarloBench(2);
73        a.JGFrun(2);
74    }
75  
76    public void JGFrun(int size){
77  
78      JGFInstrumentor.addTimer("Section3:MonteCarlo:Total", "Solutions",size);
79      JGFInstrumentor.addTimer("Section3:MonteCarlo:Run", "Samples",size);
80  
81      JGFsetsize(size); 
82  
83      JGFInstrumentor.startTimer("Section3:MonteCarlo:Total");
84  
85      JGFinitialise(); 
86      JGFapplication(); 
87      JGFvalidate(); 
88      JGFtidyup(); 
89  
90      JGFInstrumentor.stopTimer("Section3:MonteCarlo:Total");
91  
92      JGFInstrumentor.addOpsToTimer("Section3:MonteCarlo:Run", (double) input[1] );
93      JGFInstrumentor.addOpsToTimer("Section3:MonteCarlo:Total", 1);
94  
95      JGFInstrumentor.printTimer("Section3:MonteCarlo:Run"); 
96      JGFInstrumentor.printTimer("Section3:MonteCarlo:Total");
