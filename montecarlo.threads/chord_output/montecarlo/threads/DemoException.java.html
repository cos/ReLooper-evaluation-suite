24  package montecarlo.threads;
25  
26  /**
27    * Error exception, for use by the Applications Demonstrator code.
28    * With optional field for selecting whether to print debug information
29    * via a stack trace.
30    *
31    * @author H W Yau
32    * @version $Revision: 1.4 $ $Date: 1999/02/16 18:51:14 $
33    */
34  public class DemoException extends java.lang.Exception {
35    /**
36      * Flag for selecting whether to print the stack-trace dump.
37      */
38    public static boolean DEBUG=true;
39  
40    /**
41      * Default constructor.
42      */
43    public DemoException() {
44      super();
45      if( DEBUG ) {
46        printStackTrace();
47      }
48    }
49    /**
50      * Default constructor for reporting an error message.
51      */
52    public DemoException(String s) {
53      super(s);
54      if( DEBUG ) {
55        printStackTrace();
56      }
57    }
58    /**
59      * Default constructor for reporting an error code.
60      */
61    public DemoException(int ierr) {
62      super(String.valueOf(ierr));
63      if( DEBUG ) {
64        printStackTrace();
65      }
66    }
67  }
