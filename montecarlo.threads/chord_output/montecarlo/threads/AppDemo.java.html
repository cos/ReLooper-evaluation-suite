151   public void runThread() {
152     results = new Vector(nRunsMC);
153 
154        Runnable thobjects[] = new Runnable [JGFMonteCarloBench.nthreads];
155        Thread th[] = new Thread [JGFMonteCarloBench.nthreads];
156 
157         for(int i=1;i<JGFMonteCarloBench.nthreads;i++) {
158             thobjects[i] = new AppDemoThread(i,nRunsMC);
159             th[i] = new Thread(thobjects[i]);
160             th[i].start();
161         }
162 
163         thobjects[0] = new AppDemoThread(0,nRunsMC);
164         thobjects[0].run();
165 
166 
167         for(int i=1;i<JGFMonteCarloBench.nthreads;i++) {
168             try {
169                 th[i].join();
170             }
171             catch (InterruptedException e) {}
172         }

187   /**
188     * Generates the parameters for the given Monte Carlo simulation.
189     *
190     * @param nRunsMC the number of tasks, and hence Monte Carlo paths to
191     *        produce.
192     */
193   private void initTasks(int nRunsMC) {
194     tasks = new Vector(nRunsMC);
195     for( int i=0; i < nRunsMC; i++ ) {
196       String header="MC run "+String.valueOf(i);
197       ToTask task = new ToTask(header, (long)i*11);
198       tasks.addElement((Object) task);
199     }
200   }
201   /**
202     * Method for doing something with the Monte Carlo simulations.
203     * It's probably not mathematically correct, but shall take an average over
204     * all the simulated rate paths.
205     *
206     * @exception DemoException thrown if there is a problem with reading in
207     *            any values.
208     */
209   private void processResults() throws DemoException{
210     double avgExpectedReturnRateMC = 0.0;
211     double avgVolatilityMC = 0.0;
212     double runAvgExpectedReturnRateMC = 0.0;
213     double runAvgVolatilityMC = 0.0;
214     ToResult returnMC;
215     if( nRunsMC != results.size() ) {
216       errPrintln("Fatal: TaskRunner managed to finish with no all the results gathered in!");
217       System.exit(-1);
218     }
219     //
220     // Create an instance of a RatePath, for accumulating the results of the
221     // Monte Carlo simulations.
222     RatePath avgMCrate = new RatePath(nTimeStepsMC, "MC", 19990109, 19991231, dTime);
223     for( int i=0; i < nRunsMC; i++ ) {
224       // First, create an instance which is supposed to generate a
225       // particularly simple MC path.
226       returnMC = (ToResult) results.elementAt(i);
227       avgMCrate.inc_pathValue(returnMC.get_pathValue());
228       avgExpectedReturnRateMC += returnMC.get_expectedReturnRate();
229       avgVolatilityMC         += returnMC.get_volatility();
230       runAvgExpectedReturnRateMC = avgExpectedReturnRateMC /((double)(i+1));
231       runAvgVolatilityMC = avgVolatilityMC / ((double)(i+1));
232     } // for i;
233     avgMCrate.inc_pathValue((double)1.0/((double)nRunsMC));
234     avgExpectedReturnRateMC /= nRunsMC;
235     avgVolatilityMC         /= nRunsMC;
236     /*
237     try{
238       Thread.sleep(200);
239     } catch( InterruptedException intEx) {
240       errPrintln(intEx.toString());
241     }
242 
243     */
244 
245     JGFavgExpectedReturnRateMC = avgExpectedReturnRateMC;
246 
247 //    dbgPrintln("Average over "+nRunsMC+": expectedReturnRate="+
248 //    avgExpectedReturnRateMC+" volatility="+avgVolatilityMC + JGFavgExpectedReturnRateMC);
249   }

355 class AppDemoThread implements Runnable {
356 
357     int id,nRunsMC;
358 
359     public AppDemoThread(int id,int nRunsMC) {
360         this.id = id;
361         this.nRunsMC=nRunsMC;
362 
363     }
364 
365 
366     public void run() {
367 
368          PriceStock ps;
369         // Now do the computation.
370 
371 
372         int ilow, iupper, slice;
373 
374         slice = (nRunsMC + JGFMonteCarloBench.nthreads-1)/JGFMonteCarloBench.nthreads;
375 
376         ilow = id*slice;
377         iupper = (id+1)*slice;
378         if (id==JGFMonteCarloBench.nthreads-1) iupper=nRunsMC;
379 
380         for( int iRun=ilow; iRun < iupper; iRun++ ) {
381         ps = new PriceStock();
382         ps.setInitAllTasks(AppDemo.initAllTasks);
383         ps.setTask(AppDemo.tasks.elementAt(iRun));
384         ps.run();
385             AppDemo.results.addElement(ps.getResult());
386         }
387     }
388 }
