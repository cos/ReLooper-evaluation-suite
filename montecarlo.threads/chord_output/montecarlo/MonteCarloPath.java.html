387   /**
388     * Method for writing out the values from a Monte Carlo path into a
389     * data file.  This can then be fed back in as a test.
390     * The data are written in the following format:
391       <pre>
392       881003,0.0000,14.1944,13.9444,14.0832,2200050,0
393       881004,0.0000,14.1668,14.0556,14.1668,1490850,0
394       ...
395       990108,35.8125,36.7500,35.5625,35.8125,4381200,0
396       990111,35.8125,35.8750,34.8750,35.1250,3920800,0
397       990112,34.8750,34.8750,34.0000,34.0625,3577500,0
398       </pre>
399     * <p>Where the fields represent, one believes, the following:
400     * <ol>
401     *   <li>The date in 'YYMMDD' format</li>
402     *   <li>Open</li>
403     *   <li>High</li>
404     *   <li>Low</li>
405     *   <li>Last</li>
406     *   <li>Volume</li>
407     *   <li>Open Interest</li>
408     * </ol>
409     * One will probably make use of the closing price, but this can be
410     * redefined via the class variable <code>DATUMFIELD</code>.  Note that
411     * since the read in data are then used to compute the return, this would
412     * be a good place to trap for zero values in the data, which will cause
413     * all sorts of problems.
414     *
415     * @param dirName the directory in which to write the data file.
416     * @param filename the data filename itself.
417     * @exception DemoException thrown if there was a problem with the data
418     *                          file.
419     */
420   public void writeFile(String dirName, String filename) throws DemoException {
421     try{
422       java.io.File ratesFile = new File(dirName, filename);
423       if( ratesFile.exists() && ! ratesFile.canWrite() )
424     throw new DemoException("Cannot write to specified filename!");
425       java.io.PrintWriter out = new PrintWriter(new BufferedWriter(
426       new FileWriter(ratesFile)));
427       for(int i=0; i < nTimeSteps; i++) {
428     out.print("19990101,");
429     for(int j=1; j<DATUMFIELD; j++ ) {
430       out.print("0.0000,");
431     }
432     out.print(pathValue[i]+",");
433     out.println("0.0000,0.0000");
434       }
435       out.close();
436     } catch(java.io.IOException ioex) {
437       throw new DemoException(ioex.toString());
438     }
439   }

452   /**
453     * Method for calculating the sequence of fluctuations, based around
454     * a Gaussian distribution of given mean and variance, as defined
455     * in this class' instance variables.  Mapping from Gaussian
456     * distribution of (0,1) to (mean-drift,volatility) is done via
457     * Ito's lemma on the log of the stock price.
458     * 
459     * @param randomSeed The psuedo-random number seed value, to start off a
460     *                   given sequence of Gaussian fluctuations.
461     * @exception DemoException thrown if there are any problems with
462     *                          the computation.
463     */
464   public void computeFluctuationsGaussian(long randomSeed) throws DemoException {
465     if( nTimeSteps > fluctuations.length )
466       throw new DemoException("Number of timesteps requested is greater than the allocated array!");
467     //
468     // First, make use of the passed in seed value.
469     Random rnd;
470     if( randomSeed == -1 ) {
471       rnd = new Random();
472     } else {
473       rnd = new Random(randomSeed);
474     }
475     //
476     // Determine the mean and standard-deviation, from the mean-drift and volatility.
477     double mean = (expectedReturnRate-0.5*volatility*volatility)*get_dTime();
478     double sd   = volatility*Math.sqrt(get_dTime());
479     double gauss, meanGauss=0.0, variance=0.0;
480     for( int i=0; i < nTimeSteps; i++ ) {
481       gauss = rnd.nextGaussian();
482       meanGauss+= gauss;
483       variance+= (gauss*gauss);
484       //
485       // Now map this onto a general Gaussian of given mean and variance.
486       fluctuations[i] = mean + sd*gauss;
487       //      dbgPrintln("gauss="+gauss+" fluctuations="+fluctuations[i]);
488     }
489     meanGauss/=(double)nTimeSteps;
490     variance /=(double)nTimeSteps;
491     //    dbgPrintln("meanGauss="+meanGauss+" variance="+variance);
492   }
493   /**
494     * Method for calculating the sequence of fluctuations, based around
495     * a Gaussian distribution of given mean and variance, as defined
496     * in this class' instance variables.  Mapping from Gaussian
497     * distribution of (0,1) to (mean-drift,volatility) is done via
498     * Ito's lemma on the log of the stock price.  This overloaded method
499     * is for when the random seed should be decided by the system.
500     * 
501     * @exception DemoException thrown if there are any problems with
502     *                          the computation.
503     */
504   public void computeFluctuationsGaussian() throws DemoException {
505     computeFluctuationsGaussian((long)-1);
506   }
507   /**
508     * Method for calculating the corresponding rate path, given the
509     * fluctuations and starting rate value.
510     * 
511     * @param startValue the starting value of the rate path, to be
512     *                   updated with the precomputed fluctuations.
513     * @exception DemoException thrown if there are any problems with
514     *                          the computation.
515     */
516   public void computePathValue(double startValue) throws DemoException {
517     pathValue[0] = startValue;
518     if( returnDefinition == ReturnPath.COMPOUNDED || 
519     returnDefinition == ReturnPath.NONCOMPOUNDED) {
520       for(int i=1; i < nTimeSteps; i++ ) {
521     pathValue[i] = pathValue[i-1] * Math.exp(fluctuations[i]);
522       }
523     } else {
524       throw new DemoException("Unknown or undefined update method.");
525     }
526   }
