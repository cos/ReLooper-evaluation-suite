93      private int yy_advance ()
94          throws java.io.IOException {
95          int next_read;
96          int i;
97          int j;
98  
99          if (yy_buffer_index < yy_buffer_read) {
100             return yy_buffer[yy_buffer_index++];
101         }
102 
103         if (0 != yy_buffer_start) {
104             i = yy_buffer_start;
105             j = 0;
106             while (i < yy_buffer_read) {
107                 yy_buffer[j] = yy_buffer[i];
108                 ++i;
109                 ++j;
110             }
111             yy_buffer_end = yy_buffer_end - yy_buffer_start;
112             yy_buffer_start = 0;
113             yy_buffer_read = j;
114             yy_buffer_index = j;
115             next_read = yy_reader.read(yy_buffer,
116                     yy_buffer_read,
117                     yy_buffer.length - yy_buffer_read);
118             if (-1 == next_read) {
119                 return YY_EOF;
120             }
121             yy_buffer_read = yy_buffer_read + next_read;
122         }
123 
124         while (yy_buffer_index >= yy_buffer_read) {
125             if (yy_buffer_index >= yy_buffer.length) {
126                 yy_buffer = yy_double(yy_buffer);
127             }
128             next_read = yy_reader.read(yy_buffer,
129                     yy_buffer_read,
130                     yy_buffer.length - yy_buffer_read);
131             if (-1 == next_read) {
132                 return YY_EOF;
133             }
134             yy_buffer_read = yy_buffer_read + next_read;
135         }
136         return yy_buffer[yy_buffer_index++];
137     }

203     private int[][] unpackFromString(int size1, int size2, String st) {
204         int colonIndex = -1;
205         String lengthString;
206         int sequenceLength = 0;
207         int sequenceInteger = 0;
208 
209         int commaIndex;
210         String workString;
211 
212         int res[][] = new int[size1][size2];
213         for (int i= 0; i < size1; i++) {
214             for (int j= 0; j < size2; j++) {
215                 if (sequenceLength != 0) {
216                     res[i][j] = sequenceInteger;
217                     sequenceLength--;
218                     continue;
219                 }
220                 commaIndex = st.indexOf(',');
221                 workString = (commaIndex==-1) ? st :
222                     st.substring(0, commaIndex);
223                 st = st.substring(commaIndex+1);
224                 colonIndex = workString.indexOf(':');
225                 if (colonIndex == -1) {
226                     res[i][j]=Integer.parseInt(workString);
227                     continue;
228                 }
229                 lengthString =
230                     workString.substring(colonIndex+1);
231                 sequenceLength=Integer.parseInt(lengthString);
232                 workString=workString.substring(0,colonIndex);
233                 sequenceInteger=Integer.parseInt(workString);
234                 res[i][j] = sequenceInteger;
235                 sequenceLength--;
236             }
237         }
238         return res;
239     }
