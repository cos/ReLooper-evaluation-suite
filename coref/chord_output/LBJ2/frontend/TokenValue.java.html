4   /**
5     * Objects of this class are returned by LBJ2's scanner to its parser.  It
6     * simply holds some information about the token and provides easy access to
7     * a few primitive parsing routines.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class TokenValue
12  {
13    /** The line on which the token is found in the source file. */
14    public int line;
15    /**
16      * The byte offset in the file at which the token is found in the source
17      * file.
18     **/
19    public int byteOffset;
20    /** The text in the source file that comprises the token. */
21    public String text;
22    /** The name of the source file. */
23    public String filename;   
24  
25    /** Default constructor.  Does nothing. */
26    TokenValue() { }
27  
28    /**
29      * Full constructor.
30      *
31      * @param text       The text in the source that comprises the token.
32      * @param line       The line on which the token is found in the source.
33      * @param byteOffset The byte offset in the file at which the token is
34      *                   found in the source.
35      * @param filename   The name of the source file.
36     **/
37    TokenValue(String text, int line, int byteOffset, String filename)
38    {
39      this.text = text; 
40      this.line = line; 
41      this.byteOffset = byteOffset;
42      this.filename = filename;
43    }
44  
45    /**
46      * Return the token's text in a <code>String</code>.
47      *
48      * @return The token's text.
49     **/
50    public String toString() { return text; }
51  
52    /**
53      * Attempts to parse the token's text as if it represented an integer.
54      *
55      * @return The integer that the token's text represents.
56     **/
57    public int toInt() { return Integer.parseInt(text); }
58  
59    /**
60      * Attempts to parse the token's text as if it represented an integer.
61      *
62      * @return The integer that the token's text represents.
63     **/
64    public long toLong() { return Long.parseLong(text); }
65  
66    /**
67      * Attempts to parse the token's text as if it represented a double
68      * precision floating point value.
69      *
70      * @return The double precision floating point value that the token's text
71      *         represents.
72     **/
73    public double toFloat() { return Float.parseFloat(text); }
74  
75    /**
76      * Attempts to parse the token's text as if it represented a double
77      * precision floating point value.
78      *
79      * @return The double precision floating point value that the token's text
80      *         represents.
81     **/
82    public double toDouble() { return Double.parseDouble(text); }
83  
84    /**
85      * Attempts to parse the token's text as if it represented a boolean value.
86      *
87      * @return The boolean value that the token's text represents.
88     **/
89    public boolean toBoolean() { return text.equals("true"); }
90  
91    /**
92      * Attempts to parse the token's text as if it represented a character
93      * value.
94      *
95      * @return The character value that the token's text represents.
96     **/
97    public String toChar() { return text.substring(1, text.length() - 1); }
98  }
