4   /**
5     * This class prints useful error messages for scanner generated errrors.
6     *
7     * @author Nick Rizzolo
8    **/
9   class ScannerError
10  {
11    /**
12      * Signal an error due to an unterminated comment appearing in the token
13      * stream.
14     **/
15    public static void unterminatedCommentError()
16    {
17      System.err.println("Unterminated comment.");
18    }
19  
20    /**
21      * Signal an error due to discovering an end-of-comment marker while not
22      * scanning a comment.
23     **/
24    public static void commentEndWithoutBegin()
25    {
26      System.err.println("Comment ending encountered without beginning.");
27    }
28  
29    /**
30      * Signal an error due to an invalid character (one which is not specified
31      * as being allowed by the language definition) in the source text.
32     **/
33    public static void illegalCharacterError()
34    {
35      System.err.println("Illegal character");
36    }
37  
38    /**
39      * Signal an error in scanning which does not fall into any of the above
40      * categories.
41     **/
42    public static void otherError()
43    { 
44      System.err.println("Other");
45    }
46  }
