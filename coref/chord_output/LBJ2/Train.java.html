30    /**
31      * Generates a <code>String</code> containing the name of the specified
32      * <code>Throwable</code> and its stack trace.
33      *
34      * @param t  <code>Throwable</code>.
35      * @return   The generated message.
36     **/
37    private static String stackTrace(Throwable t)
38    {
39      String message = "  " + t + "\n";
40      StackTraceElement[] elements = t.getStackTrace();
41      if (elements.length == 0) message += "    no stack trace available\n";
42      for (int i = 0; i < elements.length; ++i)
43        message += "    " + elements[i] + "\n";
44      return message;
45    }

48    /**
49      * Run the <code>javac</code> compiler with the specified arguments in
50      * addition to those specified on the command line.
51      *
52      * @param arguments  The arguments to send to <code>javac</code>.
53      * @return           <code>true</code> iff errors were encountered.
54     **/
55    public static boolean runJavac(String arguments)
56    {
57      Process javac = null;
58      String pathArguments = "-classpath \"" + Main.classPath + "\" -sourcepath \""
59                             + Main.sourcePath + "\"";
60  
61      if (Main.generatedSourceDirectory != null)
62      {
63        String gsd = Main.generatedSourceDirectory;
64        int packageIndex = -1;
65        if (!AST.globalSymbolTable.getPackage().equals(""))
66          packageIndex =
67            gsd.lastIndexOf(File.separator + AST.globalSymbolTable.getPackage()
68                                             .replace('.', File.separatorChar));
69        if (packageIndex != -1) gsd = gsd.substring(0, packageIndex);
70        pathArguments += File.pathSeparator + gsd;
71      }
72  
73      if (Main.classPackageDirectory != null)
74        pathArguments += " -d " + Main.classPackageDirectory;
75  
76      String command = Configuration.javac + " " + Main.javacArguments + " "
77                       + pathArguments + " " + arguments;
78  
79      try { javac = Runtime.getRuntime().exec(command); }
80      catch (Exception e)
81      {
82        System.err.println("Failed to execute 'javac': " + e);
83        System.exit(1);
84      }
85  
86      BufferedReader error =
87        new BufferedReader(new InputStreamReader(javac.getErrorStream()));
88      try
89      {
90        for (String line = error.readLine(); line != null;
91             line = error.readLine())
92          System.out.println(line);
93      }
94      catch (Exception e)
95      {
96        System.err.println("Error reading STDERR from 'javac': " + e);
97        System.exit(1);
98      }
99  
100     int exit = 0;
101     try { exit = javac.waitFor(); }
102     catch (Exception e)
103     {
104       System.err.println("Error waiting for 'javac' to terminate: " + e);
105       System.exit(1);
106     }
107 
108     return exit != 0;
109   }

151   /**
152     * Adds an edge from dependor to dependency in the
153     * {@link #learnerDependencies} graph.  If <code>dependency</code> is
154     * <code>null</code>, no new list item is added, but the
155     * <code>HashSet</code> associated with <code>dependor</code> is still
156     * created if it didn't already exist.
157     *
158     * @param dependor   The name of the node doing the depending.
159     * @param dependency The name of the node depended on.
160    **/
161   private void addDependency(String dependor, String dependency)
162   {
163     HashSet dependencies = (HashSet) learnerDependencies.get(dependor);
164 
165     if (dependencies == null)
166     {
167       dependencies = new HashSet();
168       learnerDependencies.put(dependor, dependencies);
169     }
170 
171     if (dependency != null) dependencies.add(dependency);
172   }

175   /**
176     * This method initializes the {@link #learnerDependencies} graph such
177     * that the entry for each learner contains the names of all learners that
178     * depend on it, except that cycles are broken by preferring that learners
179     * appearing earlier in the source get trained first.
180    **/
181   protected void fillLearnerDependorsDAG()
182   {
183     threads =
184       (TrainingThread[]) threadMap.values().toArray(new TrainingThread[0]);
185     Arrays.sort(threads,
186                 new Comparator()
187                 {
188                   public int compare(Object o1, Object o2)
189                   {
190                     TrainingThread t1 = (TrainingThread) o1;
191                     TrainingThread t2 = (TrainingThread) o2;
192                     return t2.byteOffset - t1.byteOffset;
193                   }
194                 });
195 
196     for (int i = 0; i < threads.length - 1; ++i)
197       for (int j = i + 1; j < threads.length; ++j)
198       {
199         if (SemanticAnalysis.isDependentOn(threads[i].getName(),
200                                            threads[j].getName()))
201           addDependency(threads[i].getName(), threads[j].getName());
202         else if (SemanticAnalysis.isDependentOn(threads[j].getName(),
203                                                 threads[i].getName()))
204           addDependency(threads[j].getName(), threads[i].getName());
205       }
206   }

209   /**
210     * This method updates the {@link #learnerDependencies} graph by removing
211     * the specified name from every dependencies list, and then starts every
212     * thread that has no more dependencies.
213     *
214     * @param name The name of a learner whose training has completed.
215    **/
216   protected void executeReadyThreads(String name)
217   {
218     LinkedList ready = new LinkedList();
219 
220     synchronized (learnerDependencies)
221     {
222       for (Iterator I = learnerDependencies.entrySet().iterator();
223            I.hasNext(); )
224       {
225         Map.Entry e = (Map.Entry) I.next();
226         HashSet dependencies = (HashSet) e.getValue();
227         dependencies.remove(name);
228         if (dependencies.size() == 0) ready.add(e.getKey());
229       }
230     }
231 
232     for (Iterator I = ready.iterator(); I.hasNext(); )
233     {
234       TrainingThread thread = null;
235 
236       synchronized (threadMap)
237       {
238         thread = (TrainingThread) threadMap.remove(I.next());
239       }
240 
241       if (thread != null)
242       {
243         thread.start();
244 
245         if (!Main.concurrentTraining)
246         {
247           try { thread.join(); }
248           catch (InterruptedException e)
249           {
250             System.err.println("LBJ ERROR: Training of " + thread.getName()
251                                + " has been interrupted.");
252             fatalError = true;
253           }
254         }
255       }
256     }
257   }

260   /**
261     * Runs this pass on all nodes of the indicated type.
262     *
263     * @param ast  The node to process.
264    **/
265   public void run(AST ast)
266   {
267     if (RevisionAnalysis.noChanges) return;
268     TRAINING = true;
269     threadMap = new HashMap();
270     learnerDependencies = new HashMap();
271 
272     String files = "";
273     for (Iterator I = Main.fileNames.iterator(); I.hasNext(); )
274       files += " " + I.next();
275 
276     System.out.println("Compiling generated code");
277     if (runJavac(files)) return;
278 
279     runOnChildren(ast);
280 
281     fillLearnerDependorsDAG();
282     executeReadyThreads(null);
283 
284     for (int i = 0; i < threads.length; ++i)
285     {
286       try { threads[i].join(); }
287       catch (InterruptedException e)
288       {
289         System.err.println("LBJ ERROR: Training of " + threads[i].getName()
290                            + " has been interrupted.");
291         fatalError = true;
292       }
293     }
294 
295     if (!fatalError && newCode)
296     {
297       System.out.println("Compiling generated code");
298       runJavac(files);
299     }
300 
301     TRAINING = false;
302   }

305   /**
306     * Runs this pass on all nodes of the indicated type.
307     *
308     * @param lce  The node to process.
309    **/
310   public void run(LearningClassifierExpression lce)
311   {
312     runOnChildren(lce);
313 
314     String lceName = lce.name.toString();
315     if (lce.parser != null
316           && RevisionAnalysis.revisionStatus.get(lceName)
317              .equals(RevisionAnalysis.UNAFFECTED)
318         || lce.parser == null
319            && !RevisionAnalysis.revisionStatus.get(lceName)
320                .equals(RevisionAnalysis.REVISED))
321       return;
322 
323     newCode = true;
324 
325     // First, instantiate the learner.
326     Class learnerClass = ast.symbolTable.classForName(lceName);
327     if (learnerClass == null)
328     {
329       reportError(lce.line,
330                   "Could not locate class for classifier '" + lceName + "'.");
331       return;
332     }
333 
334     Learner learner = null;
335     try { learner = (Learner) learnerClass.newInstance(); }
336     catch (Exception e)
337     {
338       reportError(lce.line,
339                   "Could not instantiate learner '" + lceName + "': " + e);
340       return;
341     }
342 
343     // Get the unique instance.
344     Method getInstance = null;
345     try
346     {
347       getInstance =
348         learnerClass.getDeclaredMethod("getInstance", new Class[0]);
349     }
350     catch (Exception e)
351     {
352       reportError(lce.line, "Could not access method '" + lceName
353                             + ".getInstance()': " + e);
354       return;
355     }
356 
357     try { learner = (Learner) getInstance.invoke(null, null); }
358     catch (Exception e)
359     {
360       reportError(lce.line,
361                   "Could not get unique instance of '" + lce.parser.name
362                   + "': " + e + ", caused by");
363       System.err.print(stackTrace(e.getCause()));
364       return;
365     }
366 
367     if (learner == null)
368     {
369       System.err.println("Could not get unique instance of '" + lceName
370                          + "'.");
371       System.exit(1);
372     }
373 
374     Parser parser = null;
375     if (lce.parser != null)
376     {
377       // Next, instantiate the parser.
378       Method getParser = null;
379       try
380       {
381         getParser = learnerClass.getDeclaredMethod("getParser", new Class[0]);
382       }
383       catch (Exception e)
384       {
385         reportError(lce.line, "Could not access method '" + lceName
386                               + ".getParser()': " + e);
387         return;
388       }
389 
390       try { parser = (Parser) getParser.invoke(null, null); }
391       catch (Exception e)
392       {
393         reportError(lce.line,
394                     "Could not instantiate parser '" + lce.parser.name + "': "
395                     + e + ", caused by");
396         Throwable cause = e.getCause();
397         System.err.print(stackTrace(cause));
398 
399         if (cause instanceof ExceptionInInitializerError)
400         {
401           System.err.println("... caused by");
402           System.err.print(
403               stackTrace(((ExceptionInInitializerError) cause).getCause()));
404         }
405 
406         return;
407       }
408     }
409 
410     TrainingThread thread =
411       new TrainingThread(lceName, lce.byteOffset, learnerClass, learner,
412                          parser, lce);
413     threadMap.put(lceName, thread);
414     addDependency(lceName, null);
415   }

448   /**
449     * This class contains the code that trains a learning classifier.  It is a
450     * subclass of <code>Thread</code> so that it may be executed concurrently.
451     *
452     * @author Nick Rizzolo
453    **/
454   protected class TrainingThread extends Thread
455   {
456     /** The byte offset at which the learner appeared. */
457     public int byteOffset;
458     /** The class of the learner to train. */
459     protected Class learnerClass;
460     /** The learner to train. */
461     protected Learner learner;
462     /** The parser from which training objects are obtained. */
463     protected Parser parser;
464     /** The expression that specified the learner. */
465     protected LearningClassifierExpression lce;
466 
467 
468     /**

474       * @param l    The learner.
475       * @param p    The parser.
476       * @param lce  The expression that specified the learner.
477      **/
478     public TrainingThread(String n, int b, Class c, Learner l, Parser p,
479                           LearningClassifierExpression lce)
480     {
481       super(n);
482       byteOffset = b;
483       learnerClass = c;
484       learner = l;
485       parser = p;
486       this.lce = lce;
487     }

490     /** Performs the training and then generates the new code. */
491     public void run()
492     {
493       if (parser != null)
494       {
495         // if a progressOutput value is specified in the lce, override that
496         // which was provided via command line. If neither are provided,
497         // default is 0.
498         if (lce.progressOutput != null)
499         {
500           int p = Integer.parseInt(lce.progressOutput.value);
501           progressOutput = p;
502         }
503 
504         int examples = 0;
505 
506         // declare these outside of the conditional so they can be referenced
507         // later
508         // store the old labeler and extractor
509         Classifier labeler = learner.getLabeler();
510         Parser oldparser = parser;
511 
512         System.out.println("Training " + getName());
513 
514         // ----------------- Feature Vector Pre-Extraction -----------------
515         boolean preExtract =
516           lce.preExtract != null
517           && Boolean.valueOf(lce.preExtract.value).booleanValue();
518         if (preExtract)
519         {
520           String exampleFilePath = getName() + ".ex";
521           if (Main.generatedSourceDirectory != null)
522             exampleFilePath = Main.generatedSourceDirectory + File.separator
523                               + exampleFilePath;
524           File exampleFile = new File(exampleFilePath);
525 
526           if (lce.rounds != null && exampleFile.exists()
527               && !exampleFile.delete())
528           {
529             System.err.println("LBJ ERROR: Can't delete " + exampleFile);
530             System.exit(1);
531           }
532 
533           DataOutputStream dos = null;
534           try
535           {
536             dos =
537               new DataOutputStream(
538                   new FileOutputStream(exampleFilePath, true));
539           }
540           catch (Exception e)
541           {
542             System.err.println("Can't create feature vector stream for '"
543                                + getName() + "': " + e);
544             System.exit(1);
545           }
546 
547           LinkedHashMap lexicon = new LinkedHashMap();
548           int nextFeatureIndex = 0;
549           Classifier extractor = learner.getExtractor();
550 
551           for (Object example = parser.next(); example != null;
552                example = parser.next())
553           {
554             if (progressOutput > 0 && examples % progressOutput == 0)
555               System.out.println("  " + getName() + ", pre-extract: "
556                                  + examples + " examples at " + new Date());
557 
558             try
559             {
560               // if we run into a fold separator, write a -1
561               if (example == FoldSeparator.separator)
562               {
563                 dos.writeInt(-1);
564               }
565               // otherwise, act normally
566               else
567               {
568                 ++examples;
569                 FeatureVector vector = extractor.classify(example);
570                 vector.addLabels(labeler.classify(example));
571 
572                 dos.writeInt(vector.labels.size());
573                 for (Iterator I = vector.labels.iterator(); I.hasNext(); )
574                 {
575                   Feature f = (Feature) I.next();
576                   Integer index = (Integer) lexicon.get(f);
577 
578                   if (index == null)
579                   {
580                     f.intern();
581                     index = new Integer(nextFeatureIndex++);
582                     lexicon.put(f, index);
583                   }
584 
585                   dos.writeInt(index.intValue());
586                 }
587 
588                 dos.writeInt(vector.features.size());
589                 for (Iterator I = vector.features.iterator(); I.hasNext(); )
590                 {
591                   Feature f = (Feature) I.next();
592                   Integer index = (Integer) lexicon.get(f);
593 
594                   if (index == null)
595                   {
596                     f.intern();
597                     index = new Integer(nextFeatureIndex++);
598                     lexicon.put(f, index);
599                   }
600 
601                   dos.writeInt(index.intValue());
602                 }
603               }
604             }
605             catch (Exception e)
606             {
607               System.err.println("Can't write example for '" + getName()
608                                  + "': " + e);
609               e.printStackTrace();
610               System.exit(1);
611             }
612           }
613 
614           try { dos.close(); }
615           catch (Exception e)
616           {
617             System.err.println("Can't close feature vector stream for '"
618                                + getName() + "': " + e);
619             System.exit(1);
620           }
621 
622           if (progressOutput > 0)
623             System.out.println("  " + getName() + ", pre-extract: " + examples
624                                + " examples processed at " + new Date());
625 
626           ObjectOutputStream oos = null;
627           String lexiconFilePath = getName() + ".lex";
628           if (Main.generatedSourceDirectory != null)
629             lexiconFilePath = Main.generatedSourceDirectory + File.separator
630                               + lexiconFilePath;
631 
632           try
633           {
634             oos =
635               new ObjectOutputStream(new FileOutputStream(lexiconFilePath));
636           }
637           catch (Exception e)
638           {
639             System.err.println("Can't create lexicon stream for '" + getName()
640                                + "': " + e);
641             System.exit(1);
642           }
643 
644           Feature[] lexiconArray =
645             (Feature[]) lexicon.keySet().toArray(new Feature[0]);
646           try { oos.writeObject(lexiconArray); }
647           catch (Exception e)
648           {
649             System.err.println("Can't write lexicon for '" + getName() + "': "
650                                + e);
651             System.exit(1);
652           }
653 
654           try { oos.close(); }
655           catch (Exception e)
656           {
657             System.err.println("Can't close lexicon stream for '"
658                 + getName() + "': " + e);
659             System.exit(1);
660           }
661 
662           // Setting up the learner to pull from the file
663           final String returnType = lce.labeler.returnType.getTypeName();
664           final String[] allowable =
665             new String[lce.labeler.returnType.values.size()];
666 
667           ConstantList.ConstantListIterator I =
668             lce.labeler.returnType.values.listIterator();
669           for (int i = 0; I.hasNext(); ++i)
670             allowable[i] = I.nextItem().value;
671 
672           if (labeler.getOutputType().equals("discrete"))
673             learner.setLabeler(
674                 new LabelVectorReturner()
675                 {
676                   public String getOutputType() { return returnType; }
677                   public String[] allowableValues() { return allowable; }
678                   public String discreteValue(Object e)
679                   {
680                     return ((DiscreteFeature) classify(e).firstFeature())
681                            .getValue();
682                   }
683                 });
684           else if (labeler.getOutputType().equals("real"))
685             learner.setLabeler(
686                 new LabelVectorReturner()
687                 {
688                   public String getOutputType() { return returnType; }
689                   public String[] allowableValues() { return allowable; }
690                   public double realValue(Object e)
691                   {
692                     return ((RealFeature) classify(e).firstFeature())
693                            .getValue();
694                   }
695                 });
696           else if (labeler.getOutputType().equals("discrete[]"))
697             learner.setLabeler(
698                 new LabelVectorReturner()
699                 {
700                   public String getOutputType() { return returnType; }
701                   public String[] allowableValues() { return allowable; }
702                   public String[] discreteValueArray(Object e)
703                   {
704                     return classify(e).discreteValueArray();
705                   }
706                 });
707           else if (labeler.getOutputType().equals("real[]"))
708             learner.setLabeler(
709                 new LabelVectorReturner()
710                 {
711                   public String getOutputType() { return returnType; }
712                   public String[] allowableValues() { return allowable; }
713                   public double[] realValueArray(Object e)
714                   {
715                     return classify(e).realValueArray();
716                   }
717                 });
718           else
719             learner.setLabeler(
720                 new LabelVectorReturner()
721                 {
722                   public String getOutputType() { return returnType; }
723                   public String[] allowableValues() { return allowable; }
724                 });
725 
726           // set the parser to be a FeatureVectorParser, so we can train from
727           // the data we just extracted
728           parser = new FeatureVectorParser(exampleFilePath, lexiconArray);
729         }
730 
731 
732         // ------------------- Cross Validation -----------------------------
733         if (lce.K != null)
734         {
735           int k = Integer.parseInt(lce.K.value);
736 
737           if (k > 1 || lce.splitStrategy == FoldParser.SplitStrategy.manual)
738           {
739             FoldParser.SplitStrategy splitStrategy = lce.splitStrategy;
740 
741             System.out.print("  " + getName() + ": Cross Validation: ");
742             if (k != -1) System.out.print("k = " + k + ", ");
743             System.out.print("Split = " + splitStrategy.getName());
744 
745             int rounds = 1;
746             if (lce.rounds != null)
747               rounds = Integer.parseInt(lce.rounds.value);
748             if (rounds != 1) System.out.print(", Rounds = " + rounds);
749             System.out.println();
750 
751             FoldParser foldParser = null;
752             // If we pre-extracted, we know how many examples there are
753             // already; otherwise FoldParser will have to compute it.
754             if (preExtract)
755               foldParser =
756                 new FoldParser(parser, k, splitStrategy, 0, false, examples);
757             else
758               foldParser = new FoldParser(parser, k, splitStrategy, 0, false);
759             k = foldParser.getK();
760 
761             // An array to store the results of k-fold testing
762             double[] results = new double[k];
763 
764             Classifier oracle = learner.getLabeler();
765             TestingMetric testingMetric = null;
766 
767             if (lce.testingMetric != null)
768             {
769               // i basically copied this section from where Nick gets an
770               // instance of Parser
771 
772               Method getTestingMetric = null;
773               try
774               {
775                 getTestingMetric =
776                   learnerClass.getDeclaredMethod("getTestingMetric",
777                                                  new Class[0]);
778               }
779               catch (Exception e)
780               {
781                 reportError(lce.line,
782                             "Could not access method'" + getName()
783                             + ".getTestingMetric()': " + e);
784                 return;
785               }
786 
787               try
788               {
789                 testingMetric =
790                   (TestingMetric) getTestingMetric.invoke(null, null);
791               }
792               catch (Exception e)
793               {
794                 reportError(lce.line,
795                             "Could not instantiate testing metric '"
796                             + lce.parser.name + "': " + e + ", caused by");
797                 System.err.print(stackTrace(e.getCause()));
798                 return;
799               }
800             }
801             else
802             {
803               // if no user-implemented testing metric was specified,
804               // default to the built-in accuracy metric
805               testingMetric = new Accuracy();
806             }
807 
808             boolean usingAccuracy = testingMetric instanceof Accuracy;
809             String metricName = testingMetric.getClass().getName();
810             while (metricName.indexOf('.') != -1)
811               metricName = metricName.substring(metricName.indexOf('.') + 1);
812 
813             // Go through the example file K times, learning a different
814             // subset each time.
815             for (int i = 0; i < k; foldParser.setPivot(++i))
816             {
817               // train on all subsets except the i'th
818               System.out.println("  " + getName()
819                                  + ": training against subset " + i);
820 
821               // go through training once per round
822               for (int r = 0; r < rounds; ++r)
823               {
824                 examples = 0;
825 
826                 for (Object example = foldParser.next(); example != null;
827                      example = foldParser.next(), ++examples)
828                 {
829                   if (progressOutput > 0 && examples % progressOutput == 0)
830                     System.out.println(
831                         "  " + getName() + ", round " + r + ", " + examples
832                         + " examples processed at " + new Date());
833                   learner.learn(example);
834                 }
835 
836                 if (progressOutput > 0)
837                   System.out.println("  " + getName() + ", " + examples
838                                      + " examples processed at "
839                                      + new Date());
840 
841                 foldParser.reset();
842               }
843 
844               learner.doneLearning();
845               foldParser.setFromPivot(true);
846               results[i] = testingMetric.test(oracle, learner, foldParser);
847 
848               System.out.print("  " + getName() + ": subset " + i);
849               double rounded = results[i];
850               if (usingAccuracy) rounded *= 100;
851               rounded = Math.round(rounded * 100000) / 100000.0;
852               System.out.print(" " + metricName + ": " + rounded);
853               if (usingAccuracy) System.out.print("%");
854               System.out.println();
855 
856               foldParser.setFromPivot(false);
857               learner.forget();
858             }
859 
860 
861             // x: the mean of the testing results
862             double x = 0.0;
863             for (int i = 0; i < k; ++i)
864               x += results[i];
865             x /= (double) k;
866 
867             // s: the standard deviation of the testing results
868             double s = 0.0;
869             for (int i = 0; i < k; ++i)
870               s += (results[i] - x) * (results[i] - x);
871             s /= (double) (k - 1);
872             s = Math.sqrt(s);
873 
874             // sem: estimated standard error of the mean
875             double sem = s / Math.sqrt(k);
876 
877             // alpha: unconfidence level
878             double alpha = Double.parseDouble(lce.alpha.value);
879 
880             double t = StudentT.tTable(k - 1, alpha);
881             double halfInterval = t * sem;
882 
883             if (usingAccuracy)
884             {
885               x *= 100;
886               halfInterval *= 100;
887             }
888 
889             halfInterval = Math.round(halfInterval * 100000) / 100000.0;
890             x = Math.round(x * 100000) / 100000.0;
891 
892             System.out.print("  " + getName() + ": " + (100 * (1 - alpha))
893                              + "% confidence interval: " + x);
894             if (usingAccuracy) System.out.print("%");
895             System.out.print(" +/- " + halfInterval);
896             if (usingAccuracy) System.out.print("%");
897             System.out.println();
898 
899             parser.reset();
900 
901             System.out.println("  " + getName()
902                                + ": training on entire training set");
903           } // end of 'if k > 1' section
904         }
905 
906         // ------------------- Final Training --------------------------------
907         if (lce.rounds == null)
908         {
909           examples = 0;
910 
911           for (Object example = parser.next(); example != null;
912                example = parser.next())
913           {
914             if (progressOutput > 0 && examples % progressOutput == 0)
915               System.out.println("  " + getName() + ": " + examples
916                                  + " examples processed at " + new Date());
917 
918             if (example != FoldSeparator.separator)
919             {
920               learner.learn(example);
921               ++examples;
922             }
923           }
924 
925           if (progressOutput > 0)
926             System.out.println("  " + getName() + ": " + examples
927                                + " examples processed at " + new Date());
928         }
929         else
930         {
931           int rounds = Integer.parseInt(lce.rounds.value);
932 
933           for (int i = 1; i <= rounds; ++i)
934           {
935             examples = 0;
936 
937             for (Object example = parser.next(); example != null;
938                  example = parser.next())
939             {
940               if (progressOutput > 0 && examples % progressOutput == 0)
941                 System.out.println(
942                     "  " + getName() + ", round " + i + ": " + examples
943                     + " examples processed at " + new Date());
944 
945               if (example != FoldSeparator.separator)
946               {
947                 learner.learn(example);
948                 ++examples;
949               }
950             }
951 
952             if (progressOutput > 0)
953               System.out.println("  " + getName() + ", round " + i + ": "
954                                  + examples + " examples processed at "
955                                  + new Date());
956             parser.reset();
957           }
958         }
959 
960         // Resetting the labeler and parser
961         if (preExtract)
962         {
963           learner.setLabeler(labeler);
964           parser = oldparser;
965         }
966 
967         learner.doneLearning();
968       }
969 
970       String lcFilePath = getName() + ".lc";
971       if (Main.classDirectory != null)
972         lcFilePath = Main.classDirectory + File.separator + lcFilePath;
973 
974       // Write the unique instance.
975       ObjectOutputStream oos = null;
976       try
977       {
978         oos = new ObjectOutputStream(
979                 new GZIPOutputStream(new FileOutputStream(lcFilePath)));
980       }
981       catch (Exception e)
982       {
983         System.err.println("Can't create object stream for '" + getName()
984                            + "': " + e);
985         System.exit(1);
986       }
987 
988       try { oos.writeObject(learner); }
989       catch (Exception e)
990       {
991         System.err.println("Can't write to object stream for '" + getName()
992                            + "': " + e);
993         System.exit(1);
994       }
995 
996       try { oos.close(); }
997       catch (Exception e)
998       {
999         System.err.println("Can't close object stream for '" + getName()
1000                           + "': " + e);
1001        System.exit(1);
1002      }
1003
1004      Field fieldIsTraining = null;
1005      try { fieldIsTraining = learnerClass.getField("isTraining"); }
1006      catch (Exception e)
1007      {
1008        System.err.println("Can't access " + learnerClass
1009                           + "'s isTraining field: " + e);
1010        System.exit(1);
1011      }
1012
1013      try { fieldIsTraining.setBoolean(learner, false); }
1014      catch (Exception e)
1015      {
1016        System.err.println("Can't set " + learnerClass
1017                           + "'s isTraining field: " + e);
1018        System.exit(1);
1019      }
1020
1021      // Write the new code.
1022      PrintStream out = TranslateToJava.open(lce);
1023      if (out == null) return;
1024
1025      out.println(TranslateToJava.disclaimer);
1026      out.print("// ");
1027      TranslateToJava.compressAndPrint(lce.shallow(), out);
1028      out.println("\n");
1029
1030      ast.symbolTable.generateHeader(out);
1031
1032      if (lce.cacheIn != null)
1033      {
1034        String field = lce.cacheIn.toString();
1035        boolean cachedInMap = field.equals(ClassifierAssignment.mapCache);
1036        if (cachedInMap) out.println("import java.util.WeakHashMap;");
1037      }
1038
1039      out.println("\n");
1040      if (lce.comment != null) out.println(lce.comment);
1041
1042      out.println("\n\npublic class " + getName() + " extends "
1043                  + lce.learnerName);
1044      out.println("{");
1045      out.println("  public static boolean isTraining = false;");
1046      out.println("  private static java.net.URL lcFilePath;");
1047      out.println("  private static " + getName() + " instance;");
1048      out.println("  public static " + getName() + " getInstance()");
1049      out.println("  {");
1050      out.println("    if (instance == null)");
1051      out.println("      instance = (" + lce.name + ") "
1052                  + "Classifier.binaryRead(lcFilePath, \"" + lce.name
1053                  + "\");");
1054      out.println("    return instance;");
1055      out.println("  }\n");
1056
1057      out.println("  static");
1058      out.println("  {");
1059      out.println("    lcFilePath = " + getName() + ".class.getResource(\""
1060                  + getName() + ".lc\");\n");
1061
1062      out.println("    if (lcFilePath == null)");
1063      out.println("    {");
1064      out.println("      System.err.println(\"ERROR: Can't locate "
1065                  + getName() + ".lc in the class path.\");");
1066      out.println("      System.exit(1);");
1067      out.println("    }");
1068      out.println("  }\n");
1069
1070      out.println("  public void save()");
1071      out.println("  {");
1072      out.println("    if (instance == null) return;\n");
1073
1074      out.println("    if (lcFilePath.toString().indexOf(\".jar!\" + "
1075                  + "java.io.File.separator) != -1)");
1076      out.println("    {");
1077      out.println("      System.err.println(\"WARNING: " + getName() + ".lc "
1078                  + "is part of a jar file.  It will be written to the "
1079                  + "current directory.  Use 'jar -u' to update the jar "
1080                  + "file.  To avoid seeing this message in the future, "
1081                  + "unpack the jar file and put the unpacked files on your "
1082                  + "class path instead.\");");
1083      out.println(
1084          "      instance.binaryWrite(System.getProperty(\"user.dir\") + "
1085          + "java.io.File.separator + \"" + getName() + ".lc\", \""
1086          + getName() + "\");");
1087      out.println("    }");
1088      out.println("    else instance.binaryWrite(lcFilePath.getPath(), \""
1089                  + getName() + "\");");
1090      out.println("  }\n");
1091
1092      TranslateToJava.generateLearnerBody(out, lce, true);
1093
1094      out.println("}\n");
1095      out.close();
1096
1097      executeReadyThreads(getName());
1098    }
1099  }
