6   /**
7     * Contains utility methods used by the classifiers in this package.
8    **/
9   public class Util

11    /**
12      * Determines what ranges the two given real values fall in, considers both
13      * sets of ranges to be feature vectors with appropriately named features,
14      * then subtracts the second vector from the first.
15     **/
16    public static LinkedList rankingRangeFeatures(double range, Double real1,
17                                                  Double real2)
18    {
19      HashMap<String, Integer> features = new HashMap<String, Integer>();
20  
21      if (real1 != null)
22      {
23        increment(features, name1(range, real1), 1);
24        increment(features, name2(range, real1), 1);
25      }
26  
27      if (real2 != null)
28      {
29        increment(features, name1(range, real2), -1);
30        increment(features, name2(range, real2), -1);
31      }
32  
33      LinkedList result = new LinkedList();
34  
35      for (Map.Entry<String, Integer> entry : features.entrySet())
36      {
37        result.add(entry.getKey());
38        result.add(entry.getValue());
39      }
40  
41      return result;
42    }
43  
44  
45    private static String name1(double range, double real)
46    {
47      return "" + (Math.floor(real / range) * range);
48    }
49  
50  
51    private static String name2(double range, double real)
52    {
53      return "" + ((Math.floor(real / range + .5) - .5) * range);
54    }
55  
56  
57    private static void increment(HashMap<String, Integer> features,
58                                  String name, int delta)
59    {
60      Integer I = features.get(name);
61      if (I == null) I = 0;
62      features.put(name, I + delta);
63    }
64  }
