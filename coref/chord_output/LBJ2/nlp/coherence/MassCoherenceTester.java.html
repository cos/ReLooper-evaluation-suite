12  /**
13    * This class may be used to produce a report of the accuracy of a coherence
14    * classifier.
15    *
16    * <h4>Usage</h4>
17    * <blockquote><pre>
18    *   java LBJ2.nlp.coherence.MassCoherenceTester \
19    *          &lt;labeled data&gt; &lt;coherence classifier&gt; ...
20    * </pre></blockquote>
21    *
22    * <h4>Input</h4>
23    * <code>&lt;labeled data&gt;</code> is the name of a file containing names
24    * of ACE 2004 documents.  <code>&lt;coherence classifier&gt;</code> is the
25    * fully qualified name of the coherence classifier to test.  Multiple such
26    * classifiers can be specified.
27    *
28    * <h4>Output</h4>
29    * A table of resutls for each classifier tested in a variety of different
30    * settings.
31   **/
32  public class MassCoherenceTester
33  {
34    /** For now, this progam does not understand verbosity. */
35    private static final int verbosity = 0;
36  
37  
38    private static final String[] columnLabels =
39    {
40      "Name", "SS20", "SE20", "NCSPO", "NNPTPSPO", "FPLSPO", "P-NCSPO",
41      "P-NNPTPSPO", "P-FPLSPO"
42    };
43  
44  
45    private static final Classifier[] labelers =
46      new Classifier[columnLabels.length - 1];
47    static
48    {
49      labelers[0] = new CoherenceLabel();
50      for (int i = 1; i < 5; ++i) labelers[i] = labelers[0];
51      labelers[5] = new B3Label();
52      for (int i = 6; i < 8; ++i) labelers[i] = labelers[5];
53    }
54  
55  
56    public static void main(String[] args)
57    {
58      String coherenceName = null;
59      String filename = null;
60  
61      try
62      {
63        filename = args[0];
64        if (args.length < 2) throw new Exception();
65      }
66      catch (Exception e)
67      {
68        System.err.println(
69              "usage: java LBJ2.nlp.coherence.MassCoherenceTester \\\n"
70            + "              <labeled data> <coherence classifier> ...");
71        System.exit(1);
72      }
73  
74      String total = "" + (args.length - 1);
75      Parser[] parsers = getParsers(filename);
76      double[][] results = new double[args.length - 1][parsers.length];
77      String[] rowLabels = new String[args.length - 1];
78  
79      for (int i = 1; i < args.length; ++i)
80      {
81        Learner cohere = ClassUtils.getLearner(args[i]);
82        rowLabels[i - 1] = args[i].substring(args[i].lastIndexOf('.') + 1);
83        String current = "" + i;
84        while (current.length() < total.length()) current = " " + current;
85        System.err.println("Processing classifier " + current + " / "
86                           + total + ": " + rowLabels[i - 1] + ", at "
87                           + new Date());
88  
89        for (int j = 0; j < parsers.length; ++j)
90        {
91          TestDiscrete tester = new TestDiscrete();
92  
93          for (Object e = parsers[j].next(); e != null; e = parsers[j].next())
94          {
95            String label = labelers[j].discreteValue(e);
96            String prediction = cohere.discreteValue(e);
97            tester.reportPrediction(prediction, label);
98  
99            if (verbosity > 0)
100           {
101             Document[] pair = (Document[]) e;
102             System.err.println(
103                 pair[0].getName() + ": " + cohere.scores(e).get(label));
104             if (verbosity > 1)
105               System.err.println(cohere.getExtractor().classify(e));
106           }
107         }
108 
109         results[i - 1][j] = 100 * tester.getOverallStats()[0];
110         parsers[j].reset();
111       }
112     }
113 
114     String[] output =
115       TableFormat.tableFormat(columnLabels, rowLabels, results);
116     for (int i = 0; i < output.length; ++i)
117       System.out.println(output[i]);
118   }

121   private static Parser[] getParsers(String filename)
122   {
123     Parser[] result = new Parser[columnLabels.length - 1];
124 
125     result[0] =
126       new DocumentPairParser(
127           new ACE2004DocumentParser(filename), 20, false, true);
128     result[1] =
129       new DocumentPairParser(
130           new ACE2004DocumentParser(filename), 20, true, true);
131 
132     Classifier coref = new HCBLDataCoref();
133 
134     result[2] =
135       new DocumentPairParser(
136           new ACE2004DocumentParser(filename),
137           new PredictedDocumentParser(
138             new ACE2004DocumentParser(filename), coref),
139           true);
140 
141     result[5] =
142       new DocumentPairParser(
143           new PredictedDocumentParser(
144             new ACE2004DocumentParser(filename),
145             new PerturbedLabels()),
146           new PredictedDocumentParser(
147             new ACE2004DocumentParser(filename), coref),
148           true);
149 
150     coref = new HCNNPTPDataCoref();
151 
152     result[3] =
153       new DocumentPairParser(
154           new ACE2004DocumentParser(filename),
155           new PredictedDocumentParser(
156             new ACE2004DocumentParser(filename), coref),
157           true);
158 
159     result[6] =
160       new DocumentPairParser(
161           new PredictedDocumentParser(
162             new ACE2004DocumentParser(filename),
163             new PerturbedLabels()),
164           new PredictedDocumentParser(
165             new ACE2004DocumentParser(filename), coref),
166           true);
167 
168     //coref = new FPLDataCoref();
169 
170     result[4] =
171       new DocumentPairParser(
172           new ACE2004DocumentParser(filename),
173           new PredictedDocumentParser(
174             new ACE2004DocumentParser(filename), coref),
175           true);
176 
177     result[7] =
178       new DocumentPairParser(
179           new PredictedDocumentParser(
180             new ACE2004DocumentParser(filename),
181             new PerturbedLabels()),
182           new PredictedDocumentParser(
183             new ACE2004DocumentParser(filename), coref),
184           true);
185 
186     return result;
187   }
