4   import LBJ2.learn.*;
5   import LBJ2.parse.Parser;
6   import LBJ2.nlp.coref.*;
7   import LBJ2.util.ClassUtils;
8

10  /**
11    * This class may be used to produce a report of the accuracy of a coherence
12    * classifier.
13    *
14    * <h4>Usage</h4>
15    * <blockquote><pre>
16    *   java LBJ2.nlp.coherence.CoherenceTester &lt;coherence classifier&gt; \
17    *                                           &lt;labeled data&gt; \
18    *                                           &lt;shuffle pair order&gt; \
19    *                                           &lt;examples per document&gt; \
20    *                                           &lt;shuffleIds&gt; \
21    *                                           [&lt;verbosity&gt;]
22    * </pre></blockquote>
23    * or
24    * <blockquote><pre>
25    *   java LBJ2.nlp.coherence.CoherenceTester &lt;coherence classifier&gt; \
26    *                                           &lt;labeled data&gt; \
27    *                                           &lt;shuffle pair order&gt; \
28    *                                           &lt;coref classifier&gt; \
29    *                                           [&lt;verbosity&gt;]
30    * </pre></blockquote>
31    * or
32    * <blockquote><pre>
33    *   java LBJ2.nlp.coherence.CoherenceTester &lt;coherence classifier&gt; \
34    *                                           &lt;labeled data&gt; \
35    *                                           &lt;shuffle pair order&gt; \
36    *                                           &lt;coref classifier 1&gt; \
37    *                                           &lt;coref classifier 2&gt; \
38    *                                           [&lt;verbosity&gt;]
39    * </pre></blockquote>
40    * or
41    * <blockquote><pre>
42    *   java LBJ2.nlp.coherence.CoherenceTester &lt;coherence classifier&gt; \
43    *                                           &lt;labeled data&gt; \
44    *                                           &lt;shuffle pair order&gt; \
45    *                                           &lt;coref classifier&gt; \
46    *                                           &lt;change probability&gt; \
47    *                                           &lt;new ID probability&gt; \
48    *                                           [&lt;verbosity&gt;]
49    * </pre></blockquote>
50    *
51    * <h4>Input</h4>
52    * <p> The first command line form above is used to produce document pairs in
53    * which one of the documents is coref gold and the other is a shuffled
54    * version of gold.  The second form creates pairs of gold labels and
55    * predictions from the specified coref classifier (which can either be an
56    * actual <code>Classifier</code> or a directory containing the predictions
57    * of a classifier).  The third form takes two such classifiers, considering
58    * whichever produces the higher <i>B<sup>3</sup></i> score to be the more
59    * coherent.  The fourth command line form is similar to the third, except it
60    * is assumed that one of the coref classifiers will be
61    * <code>LBJ2.nlp.coref.PerturbedLabels</code>.
62    *
63    * <p> All forms start with the same three command line arguments.
64    * <code>&lt;coherence classifier&gt;</code> is the fully qualified name of
65    * the coherence classifier to test.  <code>&lt;labeled data&gt;</code> is
66    * the name of a file containing names of ACE 2004 documents.
67    * <code>&lt;shuffle pair order&gt;</code> must be exactly <code>true</code>
68    * or <code>false</code>, and it controls whether the order of the documents
69    * in each document-pair-example will be randomly permuted.
70    *
71    * <p> The rest of the command line arguments are particular to the forms
72    * they appear in.  <code>&lt;examples per document&gt;</code> is the maximum
73    * number of shuffled document pairs to create per original document.
74    * <code>&lt;shuffleIds&gt;</code> will force entity IDs to be shuffled when
75    * it appears explicitly as <code>true</code> and sentence ordering to be
76    * shuffled when it appears explicitly as <code>false</code>.  The always
77    * optional <code>&lt;verbosity&gt;</code> is an interger; the greater the
78    * value, the more output is produced on <code>STDOUT</code>.
79    * <code>&lt;coref classifier&gt;</code> can be one of two things.  First, it
80    * can be the fully qualified name of a coref classifier that makes a
81    * <code>true</code> or <code>false</code> prediction about a pair of
82    * mentions stored in a {@link Mention}<code>[]</code>.  Second, it can be a
83    * directory within which files labeled with predictions are stored.
84    * <code>&lt;change probability&gt;</code> and <code>&lt;new ID
85    * probability&gt;</code> are parameters for
86    * <code>LBJ2.nlp.coref.PerturbedLabels</code>.
87    *
88    * <h4>Output</h4>
89    * <p> The main output is generated by the
90    * <code>LBJ2.classify.TestDiscrete</code> class.  See its online
91    * documentation at
92    * <a href="http://flake.cs.uiuc.edu/~rizzolo/LBJ2/library/LBJ2/classify/TestDiscrete.html">http://flake.cs.uiuc.edu/~rizzolo/LBJ2/library/LBJ2/classify/TestDiscrete.html</a>.
93    * This output is generated regardless of the value of
94    * <code>&lt;verbosity&gt;</code>.  When <code>&lt;verbosity&gt;</code> is
95    * greater than 0, the name of each document followed by the coherence
96    * classifier's unnormalized score on that document is printed.
97   **/
98  public class CoherenceTester
99  {
100   public static void main(String[] args)
101   {
102     String coherenceName = null;
103     String filename = null;
104     boolean shufflePairOrder = false;
105     boolean automaticShuffling = false;
106     boolean perturbed = false;
107     String predictedDataRoot1 = null;
108     String predictedDataRoot2 = null;
109     int examplesPerDocument = 0;
110     boolean shuffleIds = false;
111     Classifier corefClassifier1 = null;
112     Classifier corefClassifier2 = null;
113     double changeProbability = 0;
114     double newIDProbability = 0;
115     int verbosity = 0;
116 
117     try
118     {
119       coherenceName = args[0];
120       filename = args[1];
121       shufflePairOrder = args[2].equals("true");
122       automaticShuffling = Character.isDigit(args[3].charAt(0));
123 
124       if (automaticShuffling)
125       {
126         examplesPerDocument = Integer.parseInt(args[3]);
127         if (examplesPerDocument < 1) throw new Exception();
128         shuffleIds = args[4].equals("true");
129         if (args.length == 6) verbosity = Integer.parseInt(args[5]);
130         if (args.length > 6) throw new Exception();
131       }
132       else
133       {
134         String corefName = args[3];
135         corefClassifier1 = ClassUtils.getClassifier(corefName, false);
136 
137         if (corefClassifier1 == null)
138         {
139           predictedDataRoot1 = args[3];
140           corefName = null;
141         }
142 
143         if (args.length == 5 && Character.isDigit(args[4].charAt(0)))
144           verbosity = Integer.parseInt(args[4]);
145         else if (args.length >= 5)
146         {
147           perturbed = true;
148 
149           try
150           {
151             changeProbability = Double.parseDouble(args[4]);
152             newIDProbability = Double.parseDouble(args[5]);
153           }
154           catch (Exception e) { perturbed = false; }
155 
156           if (perturbed)
157           {
158             if (args.length == 7) verbosity = Integer.parseInt(args[6]);
159             if (args.length > 7) throw new Exception();
160           }
161           else
162           {
163             corefName = args[4];
164             corefClassifier2 = ClassUtils.getClassifier(corefName, false);
165 
166             if (corefClassifier2 == null)
167             {
168               predictedDataRoot2 = args[4];
169               corefName = null;
170             }
171 
172             if (args.length == 6) verbosity = Integer.parseInt(args[5]);
173             if (args.length > 6) throw new Exception();
174           }
175         }
176       }
177 
178       if (verbosity < 0) throw new Exception();
179     }
180     catch (Exception e)
181     {
182       System.err.println(
183   "usage: java LBJ2.nlp.coherence.CoherenceTester <coherence classifier> \\\n"
184 + "                                               <labeled data> \\\n"
185 + "                                               <shuffle pair order> \\\n"
186 + "                                               <examples per document> \\\n"
187 + "                                               <shuffleIds> \\\n"
188 + "                                               [<verbosity>]\n"
189 + " or\n"
190 + "       java LBJ2.nlp.coherence.CoherenceTester <coherence classifier> \\\n"
191 + "                                               <labeled data> \\\n"
192 + "                                               <shuffle pair order> \\\n"
193 + "                                               <coref classifier> \\\n"
194 + "                                               [<verbosity>]\n"
195 + " or\n"
196 + "       java LBJ2.nlp.coherence.CoherenceTester <coherence classifier> \\\n"
197 + "                                               <labeled data> \\\n"
198 + "                                               <shuffle pair order> \\\n"
199 + "                                               <coref classifier 1> \\\n"
200 + "                                               <coref classifier 2> \\\n"
201 + "                                               [<verbosity>]\n"
202 + " or\n"
203 + "       java LBJ2.nlp.coherence.CoherenceTester <coherence classifier> \\\n"
204 + "                                               <labeled data> \\\n"
205 + "                                               <shuffle pair order> \\\n"
206 + "                                               <coref classifier> \\\n"
207 + "                                               <change probability> \\\n"
208 + "                                               <new ID probability> \\\n"
209 + "                                               [<verbosity>]");
210       System.exit(1);
211     }
212 
213     Learner cohere = ClassUtils.getLearner(coherenceName);
214     Classifier labeler = new CoherenceLabel();
215     TestDiscrete tester = new TestDiscrete();
216     Parser parser = null;
217 
218     if (automaticShuffling)
219       parser =
220         new DocumentPairParser(new ACE2004DocumentParser(filename),
221                                examplesPerDocument, shuffleIds,
222                                shufflePairOrder);
223     else if (predictedDataRoot2 == null && corefClassifier2 == null)
224     {
225       Parser p1 = null;
226       Parser p2 = null;
227 
228       if (perturbed)
229       {
230         p1 =
231           new PredictedDocumentParser(
232               new ACE2004DocumentParser(filename),
233               new PerturbedLabels());
234         PerturbedLabels.setParameters(changeProbability, newIDProbability);
235         labeler = new B3Label(verbosity);
236       }
237       else p1 = new ACE2004DocumentParser(filename);
238 
239       if (predictedDataRoot1 != null)
240         p2 = new PrePredictedDocumentParser(predictedDataRoot1, filename);
241       else
242         p2 =
243           new PredictedDocumentParser(
244               new ACE2004DocumentParser(filename),
245               corefClassifier1);
246 
247       parser = new DocumentPairParser(p1, p2, shufflePairOrder);
248     }
249     else
250     {
251       Parser p1 = null;
252       Parser p2 = null;
253 
254       if (predictedDataRoot1 != null)
255         p1 = new PrePredictedDocumentParser(predictedDataRoot1, filename);
256       else
257         p1 = new PredictedDocumentParser(new ACE2004DocumentParser(filename),
258                                          corefClassifier1);
259 
260       if (predictedDataRoot2 != null)
261         p2 = new PrePredictedDocumentParser(predictedDataRoot2, filename);
262       else
263         p2 = new PredictedDocumentParser(new ACE2004DocumentParser(filename),
264                                          corefClassifier2);
265 
266       parser = new DocumentPairParser(p1, p2, shufflePairOrder);
267       labeler = new B3Label(verbosity);
268     }
269 
270     for (Object e = parser.next(); e != null; e = parser.next())
271     {
272       String label = labeler.discreteValue(e);
273       String prediction = cohere.discreteValue(e);
274       tester.reportPrediction(prediction, label);
275 
276       if (verbosity > 0)
277       {
278         Document[] pair = (Document[]) e;
279         System.out.println(
280             pair[0].getName() + ": " + cohere.scores(e).get(label));
281         if (verbosity > 1)
282           System.out.println(cohere.getExtractor().classify(e));
283       }
284     }
285 
286     tester.printPerformance(System.out);
287   }
288 }
