3   import LBJ2.parse.*;
4   import LBJ2.nlp.*;
5   
6   
7   /**
8     * This parser takes the {@link LBJ2.nlp.Word}s in the representation created
9     * by another {@link LBJ2.parse.Parser} and creates a new representation
10    * consisting of {@link Token}s.  The input parser is actually expected to
11    * return a {@link LBJ2.parse.LinkedVector} populated by
12    * {@link LBJ2.nlp.Word}s with each call to {@link LBJ2.parse.Parser#next()}.
13    * The {@link Token}s returned by calls to this class's {@link #next()}
14    * method are also contained in {@link LBJ2.parse.LinkedVector}s representing
15    * sentences which are accessible via the
16    * {@link LBJ2.parse.LinkedVector#parent} field.
17    *
18    * @author Nick Rizzolo
19   **/
20  public class PlainToTokenParser implements Parser
21  {
22    /**
23      * A parser creating a representation consisting of {@link LBJ2.nlp.Word}s.
24     **/
25    protected Parser parser;
26    /** The next token to return. */
27    protected Token next;
28  
29  
30    /**
31      * The only constructor.
32      *
33      * @param p  A parser creating a representation consisting of
34      *           {@link LBJ2.nlp.Word}s.
35     **/
36    public PlainToTokenParser(Parser p) { parser = p; }
37  
38  
39    /**
40      * This method returns {@link Token}s until the input is exhausted, at
41      * which point it returns <code>null</code>.
42     **/
43    public Object next()
44    {
45      while (next == null)
46      {
47        LinkedVector words = (LinkedVector) parser.next();
48        if (words == null) return null;
49        Word w = (Word) words.get(0);
50        Token t = new Token(w, null, null);
51  
52        for (w = (Word) w.next; w != null; w = (Word) w.next)
53        {
54          t.next = new Token(w, t, null);
55          t = (Token) t.next;
56        }
57  
58        LinkedVector tokens = new LinkedVector(t);
59        next = (Token) tokens.get(0);
60      }
61  
62      Token result = next;
63      next = (Token) next.next;
64      return result;
65    }
66  
67  
68    /** Sets this parser back to the beginning of the raw data. */
69    public void reset()
70    {
71      parser.reset();
72      next = null;
73    }
74  }
