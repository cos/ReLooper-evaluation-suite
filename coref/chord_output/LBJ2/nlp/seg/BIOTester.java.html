9   /**
10    * This class may be used to produce a detailed report of the <i>segment by
11    * segment</i> performance of a given classifier on given labeled testing
12    * data.  Segment by segment performance is computed by using a specified
13    * {@link Token} classifier to induce the predicted segments, and then
14    * computing precision, recall, and F<sub>1</sub> measures on those segments.
15    * A predicted segment is judged as different than a labeled segment if the
16    * two segments start or end at different {@link Token}s, or if they have
17    * different types.
18    *
19    * <p> It is assumed that both of the specified {@link Token} classifiers
20    * (one giving labels and the other giving predictions) produce discrete
21    * predicitions of the form <code>B-<i>type</i></code>,
22    * <code>I-<i>type</i></code>, and <code>O</code> to represent the beginning
23    * of a segment of type <i>type</i>, a token inside a segment of type
24    * <i>type</i>, and a token outside of any segment respectively.
25    *
26    * <p> It is also assumed that the specified {@link LBJ2.parse.Parser}
27    * produces {@link Token}s linked to each other via the <code>previous</code>
28    * and <code>next</code> fields inherited from
29    * {@link LBJ2.parse.LinkedChild}.  In order to invoke this class as a
30    * program on the command line, it must also be the case that the parser
31    * implements a constructor with a single <code>String</code> argument.
32    *
33    * <h4>Command Line Usage</h4>
34    * <blockquote><code>
35    *   java LBJ2.nlp.seg.BIOTester &lt;classifier&gt; &lt;labeler&gt;
36    *                               &lt;parser&gt; &lt;test file&gt;
37    * </code></blockquote>
38    *
39    * <h3>Input</h3>
40    * The first three arguments must be fully qualified class names.  The fourth
41    * is the name of a file containing labeled testing data to be parsed by the
42    * parser.
43    *
44    * <h3>Output</h3>
45    * The output is generated by the {@link LBJ2.classify.TestDiscrete} class.
46    *
47    * @author Nick Rizzolo
48   **/
49  public class BIOTester
50  {
51    /** A BIO classifier that classifies {@link Token}s. */
52    protected Classifier classifier;
53    /** A BIO classifier that produces the true labels of the {@link Token}s. */
54    protected Classifier labeler;
55    /** A parser that produces {@link Token}s. */
56    protected Parser parser;
57  
58  
59    /**
60      * Initializing constructor.
61      *
62      * @param c  The value for {@link #classifier}.
63      * @param l  The value for {@link #labeler}.
64      * @param p  The value for {@link #parser}.
65     **/
66    public BIOTester(Classifier c, Classifier l, Parser p)
67    {
68      classifier = c;
69      labeler = l;
70      parser = p;
71    }
72

74    /**
75      * This method runs the tester, packaging the results in a
76      * <code>TestDiscrete</code> object.
77      *
78      * @return The performance results.
79     **/
80    public TestDiscrete test()
81    {
82      TestDiscrete results = new TestDiscrete();
83      results.addNull("O");
84  
85      for (Token t = (Token) parser.next(); t != null;
86           t = (Token) parser.next())
87      {
88        Vector vector = new Vector();
89        for (; t.next != null; t = (Token) parser.next()) vector.add(t);
90        vector.add(t);
91  
92        int N = vector.size();
93        String[] predictions = new String[N], labels = new String[N];
94  
95        for (int i = 0; i < N; ++i)
96        {
97          predictions[i] = classifier.discreteValue(vector.get(i));
98          labels[i] = labeler.discreteValue(vector.get(i));
99        }
100 
101       for (int i = 0; i < N; ++i)
102       {
103         String p = "O", l = "O";
104         int pEnd = -1, lEnd = -1;
105         Token w = (Token) vector.get(i);
106 
107         if (predictions[i].startsWith("B-")
108             || predictions[i].startsWith("I-")
109                && (i == 0
110                    || !predictions[i - 1]
111                        .endsWith(predictions[i].substring(2))))
112         {
113           p = predictions[i].substring(2);
114           pEnd = i;
115           while (pEnd + 1 < N && predictions[pEnd + 1].equals("I-" + p))
116             ++pEnd;
117         }
118 
119         if (labels[i].startsWith("B-")
120             || labels[i].startsWith("I-")
121                && (i == 0 || !labels[i - 1].endsWith(labels[i].substring(2))))
122         {
123           l = labels[i].substring(2);
124           lEnd = i;
125           while (lEnd + 1 < N && labels[lEnd + 1].equals("I-" + l)) ++lEnd;
126         }
127 
128         if (!p.equals("O") || !l.equals("O"))
129         {
130           if (pEnd == lEnd) results.reportPrediction(p, l);
131           else
132           {
133             if (!p.equals("O")) results.reportPrediction(p, "O");
134             if (!l.equals("O")) results.reportPrediction("O", l);
135           }
136         }
137       }
138     }
139 
140     return results;
141   }

144   /**
145     * The command line program simply instantiates an object of this class and
146     * calls its {@link #test()} method.
147    **/
148   public static void main(String[] args)
149   {
150     String classifierName = null;
151     String labelerName = null;
152     String parserName = null;
153     String inputFile = null;
154 
155     try
156     {
157       classifierName = args[0];
158       labelerName = args[1];
159       parserName = args[2];
160       inputFile = args[3];
161       if (args.length > 4) throw new Exception();
162     }
163     catch (Exception e)
164     {
165       System.err.println(
166 "usage: java LBJ2.nlp.seg.BIOTester <classifier> <labeler> <parser> <test file>");
167       System.exit(1);
168     }
169 
170     Classifier classifier = ClassUtils.getClassifier(classifierName);
171     Classifier labeler = ClassUtils.getClassifier(labelerName);
172     Parser parser =
173       ClassUtils.getParser(parserName, new Class[]{ String.class },
174                            new String[]{ inputFile });
175 
176     new BIOTester(classifier, labeler, parser)
177       .test().printPerformance(System.out);
178   }
