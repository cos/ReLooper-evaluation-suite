3   import LBJ2.nlp.Word;

6   /**
7     * Simple extension of the {@link LBJ2.nlp.Word} class from LBJ's NLP
8     * library.  Two extra fields are provided to store the token's type.  First,
9     * {@link #type} is intended to store the type of the token <i>as computed by
10    * some classifier</i>.  Second, {@link #label} is intended for use only when
11    * the data is labeled.
12    *
13    * @author Nick Rizzolo
14   **/
15  public class Token extends Word
16  {
17    /** This field is used to store a computed type tag. */
18    public String type;
19    /** This field stores the type tag found in labeled data. */
20    public String label;
21  
22  
23    /**
24      * A <code>Token</code> can be constructed from a <code>Word</code>
25      * object representing the same word, a <code>Token</code> representing
26      * the previous word in the sentence, and the type label found in the data.
27      *
28      * @param w    Represents the same word as the <code>Token</code> being
29      *             constructed.
30      * @param p    The previous word in the sentence.
31      * @param type The type label for this word from the data.
32     **/
33    public Token(Word w, Token p, String type)
34    {
35      super(w.form, w.partOfSpeech, w.lemma, w.wordSense, p, w.start, w.end);
36      label = type;
37    }
38  
39  
40    /**
41      * Produces a simple <code>String</code> representation of this word in
42      * which the {@link #label} field appears followed by the word's part
43      * of speech and finally the form (i.e., spelling) of the word all
44      * surrounded by parentheses.
45     **/
46    public String toString()
47    {
48      return "(" + label + " " + partOfSpeech + " " + form + ")";
49    }
50  }
