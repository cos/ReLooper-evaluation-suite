7   /**
8     * This class converts the <code>String</code> names of POS tags into
9     * discrete integer values.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class POS implements Serializable
14  {
15    /** The actual part of speech is encoded as an integer. */
16    private int value;
17  
18  
19    /**
20      * Constructor that initializes the <code>value</code> variable.  This
21      * constructor should never be called, since all possible parts of speech
22      * are numerated by name as <code>public static final</code> variables.
23      *
24      * @param v  The value of the new part of speech object.
25     **/
26    public POS(int v) { value = v; }
27  
28  
29    /** Returns the name of the part of speech represented by this object. */
30    public String toString() { return names[value]; }
31  
32    /** Returns the token that represents the same part of speech. */
33    public String toToken() { return tokens[value]; }
34  
35  
36    /** Returns the integer used to represent this part of speech tag. */
37    public int getValue() { return value; }
38  
39  
40    /**
41      * Two <code>POS</code> objects are equal iff their <code>value</code>
42      * variables are equal.
43      *
44      * @param o  The object to compare against this <code>POS</code> object.
45      * @return   <code>true</code> iff the input object is a <code>POS</code>
46      *           object with the same <code>value</code>.
47     **/
48    public boolean equals(Object o)
49    {
50      if (!(o instanceof POS)) return false;
51      return ((POS) o).value == value;
52    }
53  
54  
55    /** Simply returns the <code>value</code> variable. */
56    public int hashCode() { return value; }

129   /**
130     * Converts a <code>POS</code> object to the token that represents the same
131     * part of speech.
132     *
133     * @param tag  The <code>POS</code> object to convert.
134     * @return     The token representing the same part of speech.
135    **/
136   public static String toToken(POS tag) { return tokens[tag.value]; }
137 
138 
139   /**
140     * Converts a token to the <code>POS</code> object that represents the same
141     * part of speech.
142     *
143     * @param s  The token to convert.
144     * @return   The <code>POS</code> object representing the same part of
145     *           speech.
146    **/
147   public static POS fromToken(String s)
148   {
149     POS result = (POS) fromTokens.get(s);
150     assert result != null
151          : "LBJ ASSERTION FAILED: Unrecognized POS tag: '" + s + "'";
152     return result;
153   }
