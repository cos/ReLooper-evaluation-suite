8   /**
9    * This class implements a dummy inference variable returning mention pair
10   * scores that have been precomputed by an external coreference classifier.
11   **/
12  public class DataCoref extends LBJ2.learn.Learner {
13      /**
14       * A score set with scores of 0 for variables representing identity
15       * coreference.
16       **/
17      private static final ScoreSet zeroScore = new ScoreSet(new String[] {
18              "false", "true" }, new double[] { 0, 0 });
19      /** The name of the file from which this learner reads its data. */
20      private static final String scoresDirectory = 
21          "data" + File.separator+ "data" + File.separator+ "ace_tides_multling_train" + File.separator + "NoConstraints";
22          //"data/data/ace_tides_multling_train/NoConstraints";
23      // "/home/binhle2/workspaces/coref/coref/data/ace_tides_multling_train/NoConstraints";
24      /**
25       * The threshold adjustment, as if supplied to
26       * <code>LinearThresholdUnit.setThreshold(double)</code>.
27       **/
28      private static double thresholdAdjustment = -5.5;
29      /** The name of the currently loaded scores file. */
30      private static String currentScoresFile = new String();
31      /**
32       * A cache of all <code>ScoreSet</code>s this learner is capable of
33       * returning.
34       **/
35      private static HashMap<String, ScoreSet> scores = new HashMap<String, ScoreSet>();
36  
37      /**
38       * Determines if <code>DataCoref</code> currently has the predictions for
39       * the specified document cached.
40       * 
41       * @param name
42       *            The name of the document.
43       * @return <code>true</code> iff the predictions for the specified document
44       *         are currently cached.
45       **/
46      public static boolean isCached(String name) {
47          return (name + ".scores").equals(currentScoresFile);
48      }
49  
50      /**
51       * Reads this classifier's internal representation into memory.
52       * 
53       * @param file
54       *            The name of the file to read it from.
55       * @return A map storing the scores for every possible example object.
56       **/
57      private static void readData(String file) {
58          if (file.equals(currentScoresFile))
59              return;
60          currentScoresFile = file;
61          final String[] dirs = { "dev", "test", "train" };
62  
63          for (int i = 0; i < 3; ++i) {
64              file = scoresDirectory + File.separator + dirs[i] + File.separator + currentScoresFile;
65              if (new File(file).exists())
66                  break;
67          }
68  
69          scores = new HashMap<String, ScoreSet>();
70          Parser parser = new LineByLine(file) {
71              public Object next() {
72                  return readLine();
73              }
74          };
75  
76          for (String line = (String) parser.next(); line != null; line = (String) parser
77                  .next()) {
78              if (line.matches("^\\s*$"))
79                  continue;
80              String[] fields = line.split("\\s+");
81              double score = Double.parseDouble(fields[1]);
82              String[] ids = fields[0].split("\\|");
83              scores.put(fields[0], new ScoreSet(
84                      new String[] { "false", "true" }, new double[] { -score,
85                              score }));
86              scores.put(ids[0] + "|" + ids[2] + "|" + ids[1], new ScoreSet(
87                      new String[] { "false", "true" }, new double[] { -score,
88                              score }));
89          }
90      }

104     public void setThreshold(double t) {

139     /**
140      * Looks up the scores for the given example in {@link #scores}.
141      * 
142      * @param example
143      *            The example for which scores are desired.
144      * @return The score set for the given example.
145      **/
146     public ScoreSet scores(Object example) {
147         Document.Mention[] pair = (Document.Mention[]) example;
148         String docName = pair[0].getDocument().getName();
149         readData(docName + ".scores");
150 
151         if (pair[0].getMentionID().equals(pair[1].getMentionID()))
152             return zeroScore;
153         String key = pair[0].getDocument().getName() + "|"
154                 + pair[0].getMentionID() + "|" + pair[1].getMentionID();
155 
156         if (!scores.containsKey(key)) {
157             System.err.println("ERROR: No entry for mention pair '" + key + "'");
158             System.exit(1);
159         }
160 
161         ScoreSet result = (ScoreSet) scores.get(key).clone();
162         result.getScore("true").score -= thresholdAdjustment;
163         result.getScore("false").score += thresholdAdjustment;
164         return result;
165     }

167     /**
168      * Returns the value with the highest score as computed by
169      * {@link #scores(Object)}.
170      * 
171      * @param example
172      *            The example to classify.
173      * @return The classification for the given example.
174      **/
175     public String discreteValue(Object example) {
176         return scores(example).highScoreValue();
177     }

179     /**
180      * Looks up the score set for the given example and then returns a feature

183      * 
184      * @param example
185      *            The example for which a classification is desired.
186      * @return A vector containing a single feature whose value is set as
187      *         described above.
188      **/
189     public FeatureVector classify(Object example) {
190         return new FeatureVector(new DiscreteFeature(containingPackage, name,
191                 discreteValue(example)));
192     }
