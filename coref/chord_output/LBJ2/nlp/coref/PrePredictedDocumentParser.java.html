3   import java.util.*;
4   import java.io.*;
5   import LBJ2.parse.LineByLine;

8   /**
9     * Reads the <code>.pred.apf.xml</code> files that have the same base
10    * filename as the usual <code>.apf.xml</code> files from the specified root
11    * folder.  The same input files that {@link ACE2004DocumentParser} take as
12    * input are accepted by this parser.  <code>.pred.apf.xml</code> files have
13    * the same format as <code>.apf.xml</code> files, so this class simply
14    * serves to redirect the input from the files named in the input file to the
15    * corresponding <code>.pred.apf.xml</code> files.
16   **/
17  public class PrePredictedDocumentParser extends LineByLine
18  {
19    /**
20      * Possible subdirectories of the prefix in which a prediction file may
21      * reside.
22     **/
23    protected static String[] subdirectories = { "train", "dev", "test" };
24  
25  
26    /** This parser reads prediction labeled documents. */
27    protected ACE2004DocumentParser parser;
28  
29  
30    /**
31      * Initializing constructor.
32      *
33      * @param root The root direcotry in which to look for prediction files.
34      * @param file The name of a file containing <code>.apf.xml</code> file
35      *             names, one per line.
36     **/
37    public PrePredictedDocumentParser(String root, String file)
38    {
39      super(file);
40      LinkedList<String> files = new LinkedList<String>();
41  
42      for (String name = readLine(); name != null; name = readLine())
43        if (!name.equals(""))
44        {
45          int lastSlash = name.lastIndexOf(File.separatorChar);
46          name = name.substring(lastSlash + 1);
47          name = name.substring(0, name.length() - 8) + ".pred.apf.xml";
48          File f = new File(root + File.separator + subdirectories[0] + File.separator + name);
49          int i = 1;
50          for (; i < subdirectories.length && !f.exists(); ++i)
51            f = new File(root + File.separator + subdirectories[i] + File.separator + name);
52          --i;
53  
54          if (!f.exists())
55          {
56            System.out.println("Can't find " + name);
57            System.exit(1);
58          }
59  
60          files.add(f.getAbsolutePath());
61        }
62  
63      parser = new ACE2004DocumentParser(files.toArray(new String[0]), "");
64    }
65  
66  
67    /** The next pre-predicted document. */
68    public Object next() { return parser.next(); }
69  
70  
71    /** Starts parsing over again. */
72    public void reset() { parser.reset(); }
73  }
