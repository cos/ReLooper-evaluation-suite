8   /**
9     * Implements a beam search over the document renditions returned by an
10    * instance of {@link ClusterMerger}.
11   **/
12  public class ClusterMergerParser implements Parser
13  {
14    /** A file containing the names of files to be parsed. */
15    private String filename;
16    /** A parser returning labeled documents. */
17    private Parser parser;
18    /** The document are currently merging clusters over. */
19    private Document currentDocument;
20    /**
21      * A coreference classifier that takes a pair of {@link Mention}s as input.
22     **/
23    private Classifier coref;
24    /** The maximum number of document renditions in the beam. */
25    private int beamWidth;
26    /** The maximum number of mergings allowed to create a new rendition. */
27    protected int maxDepth;
28    /** The beam. */
29    protected LinkedList<ClusterMerger> beam;
30    /**
31      * When all documents in the {@link #beam} have been processed, this
32      * becomes the new beam.
33     **/
34    private LinkedList<ClusterMerger> nextLevel;
35    /** This comparator puts the "best" documents first in the beam. */
36    private Comparator<ClusterMerger> key;
37    /**
38      * Remembers the depth of the most recent document returned by
39      * {@link #next()}.
40     **/
41    private int lastDepth;
42    /** Filters the cluster merging process. */
43    private ClusterMerger.Filter filter;
44  
45  
46    /**
47      * Use of this constructor implies that document renditions will be ranked
48      * by gold <i>B<sup>3</sup></i> score.
49      *
50      * @param file The name of the input file.
51      * @param c    A coreference classifier.
52      * @param b    Value for {@link #beamWidth}.
53      * @param d    Value for {@link #maxDepth}.
54     **/
55    public ClusterMergerParser(String file, Classifier c, int b, int d)
56    {
57      this(file, c, b, d, new GoldKey(), new ClusterMerger().new Filter());
58    }
59  
60    /**
61      * Use of this constructor implies that document renditions will be ranked
62      * by gold <i>B<sup>3</sup></i> score.
63      *
64      * @param file The name of the input file.
65      * @param c    A coreference classifier.
66      * @param b    Value for {@link #beamWidth}.
67      * @param d    Value for {@link #maxDepth}.
68      * @param f    A filter used during the search.
69     **/
70    public ClusterMergerParser(String file, Classifier c, int b, int d,
71                               ClusterMerger.Filter f)
72    {
73      this(file, c, b, d, new GoldKey(), f);
74    }
75  
76    /**
77      * Use of this constructor implies that document renditions will be ranked
78      * by coherence score.
79      *
80      * @param file   The name of the input file.
81      * @param coref  A coreference classifier.
82      * @param b      Value for {@link #beamWidth}.
83      * @param d      Value for {@link #maxDepth}.
84      * @param cohere The coherence classifier whose scores will be used to rank
85      *               document renditions.
86     **/
87    public ClusterMergerParser(String file, Classifier coref, int b, int d,
88                               Classifier cohere)
89    {
90      this(file, coref, b, d, new CoherenceKey(cohere),
91           new ClusterMerger().new Filter());
92    }

94    /**
95      * Use of this constructor implies that document renditions will be ranked
96      * by coherence score.
97      *
98      * @param file   The name of the input file.
99      * @param coref  A coreference classifier.
100     * @param b      Value for {@link #beamWidth}.
101     * @param d      Value for {@link #maxDepth}.
102     * @param cohere The coherence classifier whose scores will be used to rank
103     *               document renditions.
104     * @param f      A filter used during the search.
105    **/
106   public ClusterMergerParser(String file, Classifier coref, int b, int d,
107                              Classifier cohere, ClusterMerger.Filter f)
108   {
109     this(file, coref, b, d, new CoherenceKey(cohere), f);
110   }

112   /**
113     * Initializing constructor.
114     *
115     * @param file The name of the input file.
116     * @param c    A coreference classifier.
117     * @param b    Value for {@link #beamWidth}.
118     * @param d    Value for {@link #maxDepth}.
119     * @param k    Value for {@link #key}; probably an instance of
120     *             {@link ClusterMergerParser.GoldKey} or
121     *             {@link ClusterMergerParser.CoherenceKey}.
122     * @param f    A filter used during the search.
123    **/
124   protected ClusterMergerParser(String file, Classifier c, int b, int d,
125                                 Comparator<ClusterMerger> k,
126                                 ClusterMerger.Filter f)
127   {
128     filename = file;
129     parser = new ACE2004DocumentParser(filename);
130     beam = new LinkedList<ClusterMerger>();
131     nextLevel = new LinkedList<ClusterMerger>();
132     coref = c;
133     beamWidth = b;
134     maxDepth = d;
135     key = k;
136     filter = f;
137   }
138 
139 
140   /**
141     * Return the next document rendition.
142     *
143     * @return The next document rendition.
144    **/
145   public Object next()
146   {
147     if (beam.size() == 0)
148     {
149       if (nextLevel.size() > 0)
150       {
151         beam = nextLevel;
152         nextLevel = new LinkedList<ClusterMerger>();
153       }
154       else
155       {
156         currentDocument = (Document) parser.next();
157         if (currentDocument == null) return null;
158         beam.add(new ClusterMerger(currentDocument, coref, filter));
159       }
160     }
161 
162     ClusterMerger result = updateBeam();
163     lastDepth = result.getDepth();
164     return result.getDocument();
165   }
166 
167 
168   /** Returns the value of {@link #lastDepth}. */
169   public int getLastDepth() { return lastDepth; }
170 
171 
172   /**
173     * Pop the first document in the beam, expand on it, resort, trim, and
174     * return the popped document.
175     *
176     * @return The popped document from the top of the beam.
177    **/
178   protected ClusterMerger updateBeam()
179   {
180     ClusterMerger top = beam.removeFirst();
181     if (top.getDepth() >= maxDepth) return top;
182 
183     for (ClusterMerger c = top.next(); c != null; c = top.next())
184     {
185       if (nextLevel.size() == 0 || key.compare(c, nextLevel.getFirst()) < 0)
186         nextLevel.addFirst(c);
187       else
188       {
189         ListIterator<ClusterMerger> I =
190           nextLevel.listIterator(nextLevel.size());
191         for (ClusterMerger p = I.previous(); key.compare(c, p) < 0;
192              p = (ClusterMerger) I.previous());
193         I.next();
194         I.add(c);
195       }
196 
197       if (nextLevel.size() > beamWidth) nextLevel.removeLast();
198     }
199 
200     return top;
201   }
202 
203 
204   /** Sets this parser back to the beginning of the raw data. */
205   public void reset()
206   {
207     parser.reset();
208     beam = new LinkedList<ClusterMerger>();
209     nextLevel = new LinkedList<ClusterMerger>();
210   }

213   /**
214     * Simply a comparator that orders documents with better
215     * <i>B<sup>3</sup></i> scores first.
216    **/
217   private static class GoldKey implements Comparator<ClusterMerger>
218   {
219     public int compare(ClusterMerger c1, ClusterMerger c2)
220     {
221       return c2.compareTo(c1);
222     }
223   }
224 
225 
226   /**
227     * A comparator that orders documents with higher coherence score first.
228    **/
229   private static class CoherenceKey implements Comparator<ClusterMerger>
230   {
231     /**
232       * A coherence classifier which takes a pair of documents as input and
233       * returns <code>true</code> iff the first document is more coherent.
234      **/
235     private Classifier cohere;
236 
237 
238     /**
239       * Initializing constructor.
240       *
241       * @param c  Value for {@link #cohere}.
242      **/
243     public CoherenceKey(Classifier c) { cohere = c; }
244 
245 
246     public int compare(ClusterMerger c1, ClusterMerger c2)
247     {
248       if (cohere.discreteValue(
249             new Document[]{ c1.getDocument(), c2.getDocument() })
250           .equals("true"))
251         return -1;
252       return 1;
253     }
254   }
