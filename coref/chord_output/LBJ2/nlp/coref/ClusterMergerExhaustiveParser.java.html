7   /**
8     * Returns all the document renditions resulting from merging clusters
9     * starting from the rendition produced by a coreference classifier.
10   **/
11  public class ClusterMergerExhaustiveParser extends ClusterMergerParser
12  {
13    /**
14      * Initializing constructor.
15      *
16      * @param file A file containing the names of files to be parsed.
17      * @param c    A value for {@link #coref}.
18      * @param d    A value for {@link #maxDepth}.
19     **/
20    public ClusterMergerExhaustiveParser(String file, Classifier c, int d)
21    {
22      this(file, c, d, new ClusterMerger().new Filter());
23    }
24  
25    /**
26      * Initializing constructor.
27      *
28      * @param file A file containing the names of files to be parsed.
29      * @param c    A value for {@link #coref}.
30      * @param d    A value for {@link #maxDepth}.
31      * @param f    A filter used during the search.
32     **/
33    public ClusterMergerExhaustiveParser(String file, Classifier c, int d,
34                                         ClusterMerger.Filter f)
35    {
36      super(file, c, 0, d, (Comparator<ClusterMerger>) null, f);
37    }
38  
39  
40    /**
41      * Pop the first document in the beam, expand on it, resort, trim, and
42      * return the popped document.
43      *
44      * @return The popped document from the top of the beam.
45     **/
46    protected ClusterMerger updateBeam()
47    {
48      for (boolean process = false; !process; )
49      {
50        ClusterMerger top = beam.getFirst();
51        process = top.getDepth() >= maxDepth;
52  
53        if (!process)
54        {
55          ClusterMerger merger = top.next();
56          if (merger != null) beam.addFirst(merger);
57          else process = true;
58        }
59      }
60  
61      return beam.removeFirst();
62    }
63  }
