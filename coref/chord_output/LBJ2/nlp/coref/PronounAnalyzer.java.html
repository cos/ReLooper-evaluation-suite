9   /**
10    * This program compiles some statistics about the pronouns in a dataset.
11    *
12    * <h4>Usage</h4>
13    * <blockquote><pre>
14    *   java LBJ2.nlp.coref.PronounAnalyzer &lt;classifier&gt; \
15    *                                       &lt;test data&gt; \
16    *                                       [&lt;verbosity=0&gt;]
17    * </pre></blockquote>
18    *
19    * <h4>Input</h4>
20    * <p> <code>&lt;classifier&gt;</code> is the fully qualified class name of
21    * the classifier to test.  <code>&lt;test data&gt;</code> is the name of a
22    * file containing test data.  The higher the integer supplied for
23    * <code>&lt;verbosity&gt;</code>, the more information will be output to
24    * <code>STDOUT</code>.
25    *
26    * <h4>Output</h4>
27    * A table with various counts related to how pronouns are clustered to other
28    * mentions.
29   **/
30  public class PronounAnalyzer extends CoreferenceTester
31  {
32    private static final String[] rowLabels =
33      new String[]{ "pred", "bestMerge", "gold" };
34    private static final String[] columnLabels =
35      new String[]{ "Cluster", "Sing-good", "Sing-bad", "NAM", "NOM", "PRO",
36                    "PRE", "Total" };
37    private static final int[] sigDigits = new int[7];
38  
39  
40    /**
41      * Computes <i>F<sub>1</sub></i>.
42      *
43      * @param p  The precision.
44      * @param r  The recall.
45      * @return The <i>F<sub>1</sub></i>.
46     **/
47    private static double F1(double p, double r) { return 2 * p * r / (p + r); }

50    public static void main(String[] args)
51    {
52      String classifierName = null;
53      String testFile = null;
54      double threshold = 0;
55      int verbosity = 0;
56  
57      try
58      {
59        classifierName = args[0];
60        testFile = args[1];
61        if (args.length > 2) verbosity = Integer.parseInt(args[2]);
62        if (args.length > 3) throw new Exception();
63      }
64      catch (Exception e)
65      {
66        System.err.println(
67    "usage: java LBJ2.nlp.coref.PronounAnalyzer <classifier> <test data> \\\n"
68  + "                                           [<verbosity=0>]");
69        System.exit(1);
70      }
71  
72      Classifier classifier = ClassUtils.getClassifier(classifierName);
73  
74      double[][] results =
75        analyzePronouns(classifier, new ACE2004DocumentParser(testFile),
76                        verbosity);
77      results = TableFormat.transpose(results);
78      double[][] table = new double[results.length + 1][];
79      System.arraycopy(results, 0, table, 0, results.length);
80      table[results.length] = new double[rowLabels.length];
81      table = TableFormat.transpose(table);
82      for (int i = 0; i < rowLabels.length; ++i)
83        for (int j = 0; j < results.length; ++j)
84          table[i][results.length] += table[i][j];
85  
86      String[] output =
87        TableFormat.tableFormat(columnLabels, rowLabels, table, sigDigits,
88                                new int[]{ 0 });
89      for (int i = 0; i < output.length; ++i)
90        System.out.println(output[i]);
91    }

94    /**
95      * Tests the specified classifier on the specified testing data.
96      *
97      * @param classifier The specified classifier.
98      * @param parser     A document parser.
99      * @param verbosity  The level of verbosity for output sent to
100     *                   <code>STDOUT</code>.
101     * @return The total pronoun counts over the data set, as described in
102     *         {@link #analyzePronouns(Classifier,Document)}.
103    **/
104   public static double[][] analyzePronouns(Classifier classifier,
105                                            ACE2004DocumentParser parser,
106                                            int verbosity)
107   {
108     int index = 1;
109     double[][] result = new double[rowLabels.length][columnLabels.length - 2];
110 
111     for (Document d = (Document) parser.next(); d != null;
112          d = (Document) parser.next(), ++index)
113     {
114       if (verbosity > 0) System.out.print(index + ": ");
115       double[][] results =
116         new PronounAnalyzer(verbosity).analyzePronouns(classifier, d);
117       for (int i = 0; i < result.length; ++i)
118         for (int j = 0; j < result[0].length; ++j)
119           result[i][j] += results[i][j];
120     }
121 
122     return result;
123   }

137   /**
138     * Computes statistics about pronoun usage both in the gold labels and in
139     * the predictions of a classifier.  In particular, we look to see how
140     * often pronouns occur as singletons and in clusters containing mentions
141     * of other types.  The first dimension of the returned array selects
142     * either the prediction stats (index 0) or the gold stats (index 1).  The
143     * second dimension contains counts for the following scenarios starting
144     * from index 0: singleton, cluster contains another mention of type
145     * <code>NAM</code>, <code>NOM</code>, <code>PRO</code>, or
146     * <code>PRE</code>.
147     *
148     * @param classifier The classifier whose predictions will be merged.
149     * @param document   The document which the classifier is classifying.
150    **/
151   public double[][] analyzePronouns(Classifier classifier, Document document)
152   {
153     if (verbosity > 0)
154       System.out.println(document.getName() + ": " + new Date());
155 
156     double[][] result = new double[rowLabels.length][columnLabels.length - 2];
157     initializeClusterCache(document.totalMentions());
158     Document[] documents = { new Document(document), document };
159     documents[0].fillInPredictions(classifier, verbosity);
160 
161     for (int i = 0; i < documents[0].sentences(); ++i)
162       for (int j = 0; j < documents[0].mentionsInSentence(i); ++j)
163       {
164         Document.Mention[] m =
165           { documents[0].getMention(i, j), documents[1].getMention(i, j) };
166         if (!m[0].getType().equals("PRO")) continue;
167         if (verbosity > 1)
168           System.out.print("mention " + m[0].getIndexInDocument() + ": ");
169 
170         for (int k = 0; k < 2; ++k)
171         {
172           HashSet<Document.Mention> cluster =
173             getCluster(documents[k], k, m[k]);
174 
175           if (cluster.size() == 1)
176           {
177             if (k == 0)
178             {
179               cluster = getCluster(documents[1], 1, m[1]);
180 
181               if (cluster.size() == 1)
182               {
183                 ++result[0][0];
184                 if (verbosity > 1) System.out.println("really a singleton");
185               }
186               else
187               {
188                 ++result[0][1];
189                 if (verbosity > 1) System.out.println("not a singleton");
190               }
191             }
192             else ++result[2][0];
193           }
194           else
195           {
196             boolean[] types = new boolean[6];
197 
198             for (Document.Mention n : cluster)
199             {
200               if (m[k] == n) continue;
201 
202               if (n.getType().equals("NAM")) types[2] = true;
203               else if (n.getType().equals("NOM")) types[3] = true;
204               else if (n.getType().equals("PRO")) types[4] = true;
205               else if (n.getType().equals("PRE")) types[5] = true;
206               else
207               {
208                 System.err.println("ERROR: unrecognized mention type: '"
209                                    + n.getType() + "'");
210                 System.exit(1);
211               }
212             }
213 
214             for (int t = 2; t < 6; ++t)
215               if (types[t]) ++result[2 * k][t];
216 
217             if (verbosity > 1 && k == 0)
218               System.out.println("not predicted a singleton");
219           }
220         }
221       }
222 
223     int[] c = mergeSingletonPronouns(documents[0], documents[1], verbosity);
224     result[1][0] = c[0];
225     result[1][1] = c[1];
226 
227     for (int i = 0; i < documents[0].sentences(); ++i)
228       for (int j = 0; j < documents[0].mentionsInSentence(i); ++j)
229       {
230         Document.Mention m = documents[0].getMention(i, j);
231         if (!m.getType().equals("PRO")) continue;
232         HashSet<Document.Mention> cluster = getCluster(documents[0], 0, m);
233 
234         if (cluster.size() > 1)
235         {
236           boolean[] types = new boolean[6];
237 
238           for (Document.Mention n : cluster)
239           {
240             if (m == n) continue;
241 
242             if (n.getType().equals("NAM")) types[2] = true;
243             else if (n.getType().equals("NOM")) types[3] = true;
244             else if (n.getType().equals("PRO")) types[4] = true;
245             else if (n.getType().equals("PRE")) types[5] = true;
246             else
247             {
248               System.err.println("ERROR: unrecognized mention type: '"
249                                  + n.getType() + "'");
250               System.exit(1);
251             }
252           }
253 
254           for (int t = 2; t < 6; ++t)
255             if (types[t]) ++result[1][t];
256         }
257       }
258 
259     return result;
260   }
