7   /**
8     * This class implements a classifier which takes the predictions of another
9     * coreference classifier and then merges the singleton pronouns.
10   **/
11  public abstract class MergedPronouns extends Classifier
12  {
13    /**
14      * A classifier to start with whose singleton pronouns will then be merged.
15     **/
16    protected Classifier coref;
17    /**
18      * Caches the current document, which will contain this classifier's
19      * predictions.
20     **/
21    protected Document document;
22  
23  
24    /**
25      * Initializing constructor.
26      *
27      * @param n  The name of the classifier.
28      * @param c  The coref classifier.
29     **/
30    public MergedPronouns(String n, Classifier c)
31    {
32      super(n);
33      coref = c;
34    }
35  
36  
37    /** Returns the input type of this classifier. */
38    public String getInputType() { return "[LLBJ2.nlp.coref.Document$Mention;"; }
39    /** Returns the feature output type of this classifier. */
40    public String getOutputType() { return "discrete"; }
41    /**
42      * Returns the values that a feature returned by this classifier is allowed
43      * to take.  This classifier is Boolean, and its features are either
44      * <code>"false"</code> or <code>"true"</code>.
45     **/
46    public String[] allowableValues() { return DiscreteFeature.BooleanValues; }
47  
48  
49    /**
50      * Returns the value after merging pronouns.
51      *
52      * @param example  The example to classify.
53      * @return The classification for the given example.
54     **/
55    public String discreteValue(Object example)
56    {
57      Document.Mention[] pair = (Document.Mention[]) example;
58      Document labeled = pair[0].getDocument();
59  
60      if (document == null || !document.getName().equals(labeled.getName()))
61      {
62        document = new Document(labeled);
63        document.fillInPredictions(coref, 0);
64        new CoreferenceTester(0).mergeSingletonPronouns(document, labeled, 0);
65      }
66  
67      int i0 = pair[0].getSentenceIndex();
68      int j0 = pair[0].getIndexInSentence();
69      int i1 = pair[1].getSentenceIndex();
70      int j1 = pair[1].getIndexInSentence();
71      return "" + document.getMention(i0, j0).getEntityID()
72                  .equals(document.getMention(i1, j1).getEntityID());
73    }
74  
75  
76    /**
77      * Looks up the score set for the given example and then returns a feature
78      * whose value had the highest score in the score set wrapped in a feature
79      * vector.
80      *
81      * @param example  The example for which a classification is desired.
82      * @return A vector containing a single feature whose value is set as
83      *         described above.
84     **/
85    public FeatureVector classify(Object example)
86    {
87      return
88        new FeatureVector(
89            new DiscreteFeature(containingPackage, name,
90                                discreteValue(example)));
91    }
92  
93  
94    /** Returns a hash code for this object. */
95    public int hashCode() { return name.hashCode(); }
96  }
