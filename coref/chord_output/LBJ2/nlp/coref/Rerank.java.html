9   /**
10    * An inference algorithm that re-ranks a <i>k</i>-best list of best-link
11    * coref results based on coherence scores.  This inference algorithm is
12    * specifically designed to work with coreference classifiers that take pairs
13    * of <code>LBJ2.nlp.coref.Document.Mention</code>s in an array as input and
14    * coherence classifiers that take pairs of
15    * <code>LBJ2.nlp.coref.Document</code>s in an array as input and will not
16    * work with any other classifiers.
17   **/
18  public class Rerank extends Inference
19  {
20    /** The classifier being reranked. */
21    protected DataCoref classifier;
22    /** A coherence classifier to do the reranking. */
23    protected Learner cohere;
24    /** The size of the <i>k</i>-best list of coref results to rerank. */
25    protected int K;
26    /**
27      * All coreference scores are multiplied by this weight, and all coherence
28      * scores are multiplied by 1 minus this weight.
29     **/
30    protected double alpha;
31    /** The "head" document, more conveniently accessible. */
32    protected Document headDocument;
33    /** All mentions in the document. */
34    protected Document.Mention[] mentions;
35    /**
36      * The value at location <i>(i, j)</i> in this array stores the score of
37      * the best-link coref classifier when mention <i>i</i> chooses mention
38      * <i>j</i> as its "best link".
39     **/
40    protected ScoreEntry[][] scores;

104   /** Performs the re-ranking inference. */
105   protected void infer() throws Exception
106   {
107     if (scores != null) return;
108     System.out.println(headDocument.getName() + "  "
109                        + headDocument.totalMentions() + " mentions");
110     instantiateVariables();
111     Normalizer norm = getNormalizer(cohere);
112 
113     PredictedDocument[] kBest = new PredictedDocument[K];
114     LinkedList<PredictedDocument> kBestDifferent =
115       new LinkedList<PredictedDocument>();
116     PredictedDocument.NextBests[] nextBests =
117       new PredictedDocument.NextBests[K];
118     kBest[0] = new PredictedDocument(new int[mentions.length]);
119     /*
120     kBest[0].coherenceScore =
121       (1 - alpha)
122       * norm.normalize(
123           cohere.scores(new Document[]{ kBest[0].document, new Document() }))
124         .get("true");
125         */
126     kBest[0].bCubed();
127     kBestDifferent.add(kBest[0]);
128     kBest[0].original = true;
129 
130     HashSet<String> seen = new HashSet<String>();
131     String identifier = kBest[0].toString();
132     seen.add(identifier);
133 
134     for (int i = 1; i < K; ++i)
135     {
136       nextBests[i - 1] = kBest[i - 1].getNextBests();
137       int bestJ = -1;
138       double bestScore = -Double.MAX_VALUE;
139 
140       for (int j = 0; j < i; ++j)
141       {
142         double current = nextBests[j].nextScore();
143 
144         if (current > bestScore)
145         {
146           bestScore = current;
147           bestJ = j;
148         }
149       }
150 
151       if (bestJ == -1)
152       {
153         PredictedDocument[] temp = new PredictedDocument[i];
154         System.arraycopy(kBest, 0, temp, 0, i);
155         kBest = temp;
156         break;
157       }
158 
159       kBest[i] = nextBests[bestJ].next();
160 
161       identifier = kBest[i].toString();
162       if (seen.contains(identifier)) continue;
163       seen.add(identifier);
164       kBestDifferent.add(kBest[i]);
165 
166       /*
167       kBest[i].coherenceScore =
168         (1 - alpha)
169         * norm.normalize(
170             cohere.scores(
171               new Document[]{ kBest[i].document, new Document() }))
172           .get("true");
173           */
174       kBest[i].bCubed();
175     }
176 
177     Collections.sort(kBestDifferent);
178     kBestDifferent.getFirst().setVariables();
179 
180     for (PredictedDocument doc : kBestDifferent)
181     {
182       if (doc.original) System.out.print("* ");
183       else System.out.print("  ");
184       System.out.println(doc.scoresToString());
185     }
186 
187     System.out.println();
188   }

191   /**
192     * Creates the first order variables involved in this inference problem as
193     * well as initializing the {@link #mentions} and {@link #scores} member
194     * variables.
195    **/
196   protected void instantiateVariables()
197   {
198     Normalizer norm = getNormalizer(classifier);
199     double normZero =
200       alpha
201       * norm.normalize(new ScoreSet(new Score[]{ new Score("", 0) })).get("");
202     /*
203     mentions =
204       (Document.Mention[]) headDocument.toArray(new Document.Mention[0]);
205       */
206     scores = new ScoreEntry[mentions.length][];
207 
208     for (int i = 0; i < mentions.length; ++i)
209     {
210       scores[i] = new ScoreEntry[i + 1];
211 
212       for (int j = 0; j < i; ++j)
213       {
214         FirstOrderVariable v =
215           new FirstOrderVariable(
216               classifier,
217               headDocument.getMentionPair(mentions[j], mentions[i]));
218         variables.put(v, v);
219         scores[i][j] =
220           new ScoreEntry(j,
221                          alpha * norm.normalize(v.getScores()).get("true"));
222       }
223 
224       scores[i][i] = new ScoreEntry(-1, normZero);
225       Arrays.sort(scores[i]);
226     }
227   }

245   /**
246     * Simply a wrapper for a ({@link Document#Mention}<code>[]</code>, score)
247     * pair.  An instance of this class can also represent the case where a
248     * mention does not have an antecedent.  In this case, {@link #index} will
249     * be set to -1.
250    **/
251   protected class ScoreEntry implements Comparable
252   {
253     /**
254       * The index of the earlier occurring mention in {@link Rerank#mentions};
255       * the other mention in the pair is determined externally to this class.
256      **/
257     public int index;
258     /** The score associated with the mention pair. */
259     public double score;
260 
261 
262     /**
263       * Initializing constructor.
264       *
265       * @param i  A value for {@link #index}.
266       * @param s  A value for {@link #score}.
267      **/
268     public ScoreEntry(int i, double s)
269     {
270       index = i;
271       score = s;
272     }

275     /**
276       * This method insures that instances of this class will be sorted in
277       * decreasing order of their {@link #score}.
278       *
279       * @param o  An object to compare with.
280       * @return   A value less than 0 if this score is greater than the
281       *           input's, 0 if they are equal, and a value greater than 0
282       *           otherwise.
283      **/
284     public int compareTo(Object o)
285     {
286       if (!(o instanceof ScoreEntry)) return -1;
287       ScoreEntry e = (ScoreEntry) o;
288       return new Double(e.score).compareTo(new Double(score));
289     }

570     /**
571       * An iterator of sorts for iterating through the next best solutions
572       * after this one.
573      **/
574     protected class NextBests
575     {
576       /**
577         * Index into the {@link LBJ2.nlp.coref.Rerank.PredictedDocument#nextI}
578         * array.
579        **/
580       protected int index;
581 
582 
583       /**
584         * Returns <code>true</code> iff there is another index that can be
585         * incremented.
586        **/
587       public boolean hasNext()
588       {
589         return index < nextI.length
590                && indexes[nextI[index][0]] + 1
591                   < scores[nextI[index][0]].length;
592       }
593 
594 
595       /**
596         * Returns the score of the next best document prediction in this
597         * iteration.
598         *
599         * @return The score of the next best document prediction in this
600         *         iteration, or <code>-Double.MAX_VALUE</code> if there are no
601         *         more document predictions in this iteration.
602        **/
603       public double nextScore()
604       {
605         if (!hasNext()) return -Double.MAX_VALUE;
606         return
607           corefScore
608           - scores[nextI[index][0]][indexes[nextI[index][0]]].score
609           + scores[nextI[index][0]][indexes[nextI[index][0]] + 1].score;
610       }
611 
612 
613       /**
614         * Create and return the next best document prediction in this
615         * iteration.
616         *
617         * @return The next best document prediction in this iteration, or
618         *         <code>null</code> if there are no more document predictions
619         *         in this iteration.
620        **/
621       public PredictedDocument next()
622       {
623         if (!hasNext()) return null;
624         int[] newIndexes = (int[]) indexes.clone();
625         ++newIndexes[nextI[index][0]];
626         PredictedDocument result =
627           new PredictedDocument(newIndexes, nextScore());
628         ++index;
629         return result;
630       }
631     }
