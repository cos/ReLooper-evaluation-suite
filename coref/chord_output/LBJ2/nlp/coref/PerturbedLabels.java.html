3   import java.util.*;
4   import LBJ2.classify.*;

7   /**
8     * Takes the labels present in a given document and perturbs them slightly.
9     * For each mention in a given document, if the decision is made to change
10    * that mention's ID, it can either be given a new ID not previously found in
11    * the document, or it can be randomly assigned the same ID as some mention
12    * that came before it.  The decision to make a change is governed by
13    * {@link #changeProbability}, while the decision of how to change is
14    * governed by {@link #newIDProbability}.
15   **/
16  public class PerturbedLabels extends Classifier
17  {
18    /** The probability that a mention's ID will be changed. */
19    private static double changeProbability = .1;
20    /**
21      * The probability that a mention will be assigned a new ID, given that we
22      * have decided to change it at all.
23     **/
24    private static double newIDProbability = .35;
25  
26  
27    /**
28      * Sets the values of {@link #changeProbability} and
29      * {@link #newIDProbability}.
30      *
31      * @param c  The new value for {@link #changeProbability}.
32      * @param n  The new value for {@link #newIDProbability}.
33     **/
34    public static void setParameters(double c, double n)
35    {
36      changeProbability = c;
37      newIDProbability = n;
38    }
39  
40  
41    /** Remembers the predictions this classifier makes for a given document. */
42    protected WeakHashMap<Document, String[]> predictionMap;
43  
44  
45    /** Sets the classifier's name to the default. */
46    public PerturbedLabels()
47    {
48      super("LBJ2.nlp.coref.PerturbedLabels");
49      predictionMap = new WeakHashMap<Document, String[]>();
50    }
51  
52  
53    /** Returns the input type of this classifier. */
54    public String getInputType() { return "[LLBJ2.nlp.coref.Document$Mention;"; }
55    /** Returns the feature output type of this classifier. */
56    public String getOutputType() { return "discrete"; }
57    /**
58      * Returns the values that a feature returned by this classifier is allowed
59      * to take.  This classifier is Boolean, and its features are either
60      * <code>"false"</code> or <code>"true"</code>.
61     **/
62    public String[] allowableValues() { return DiscreteFeature.BooleanValues; }
63  
64  
65    /**
66      * Specifically designed to classify {@link Mention}<code>[]</code>s.
67      *
68      * @param ex The example object to classify.
69      * @return The prediction of this classifier.
70     **/
71    public String discreteValue(Object ex)
72    {
73      Document.Mention[] pair = (Document.Mention[]) ex;
74      Document d = pair[0].getDocument();
75      String[] predictions = predictionMap.get(d);
76  
77      if (predictions == null)
78      {
79        predictions = new String[d.totalMentions()];
80        predictionMap.put(d, predictions);
81  
82        for (int i = 0; i < d.sentences(); ++i)
83          for (int j = 0; j < d.mentionsInSentence(i); ++j)
84          {
85            Document.Mention m = d.getMention(i, j);
86            predictions[m.getIndexInDocument()] = m.getEntityID();
87          }
88  
89        Random r = new Random(d.getName().hashCode());
90        int nextID = 0;
91  
92        for (int i = 0; i < predictions.length; ++i)
93        {
94          if (r.nextDouble() >= changeProbability) continue;
95  
96          if (r.nextDouble() < newIDProbability)
97          {
98            predictions[i] = "e" + nextID++;
99            continue;
100         }
101 
102         if (i == 0) continue;
103         predictions[i] = predictions[r.nextInt(i)];
104       }
105     }
106 
107     return "" + predictions[pair[0].getIndexInDocument()]
108                 .equals(predictions[pair[1].getIndexInDocument()]);
109   }
110 
111 
112   /**
113     * Specifically designed to classify {@link Mention}<code>[]</code>s.
114     *
115     * @param example  The example object to classify.
116     * @return A vector containing the single feature representing the
117     *         prediction of this classifier.
118    **/
119   public FeatureVector classify(Object example)
120   {
121     return
122       new FeatureVector(
123           new DiscreteFeature(containingPackage, name,
124                               discreteValue(example)));
125   }
