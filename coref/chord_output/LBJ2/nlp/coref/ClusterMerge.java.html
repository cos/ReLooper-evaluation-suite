4   import LBJ2.classify.*;

9   /**
10    * Attempts to merge clusters produced by a coreference classifier via a
11    * beam search.  This inference algorithm is specifically designed to work
12    * with coreference classifiers that take pairs of
13    * <code>LBJ2.nlp.coref.Document.Mention</code>s in an array as input and
14    * coherence classifiers that take pairs of
15    * <code>LBJ2.nlp.coref.Document</code>s in an array as input and will not
16    * work with any other classifiers.
17   **/
18  public class ClusterMerge extends Inference
19  {
20    /** The classifier being reranked. */
21    private DataCoref classifier;
22    /** A coherence classifier to do the reranking. */
23    private Learner cohere;
24    /** The "head" document, more conveniently accessible. */
25    private Document headDocument;
26    /**
27      * All coreference scores are multiplied by this weight, and all coherence
28      * scores are multiplied by 1 minus this weight.
29     **/
30    private double alpha;
31  
32  
33    /**
34      * Don't use this constructor, because it doesn't set values for
35      * {@link #classifier} or {@link #cohere}.
36     **/
37    public ClusterMerge() { this(null, null, .5); }
38  
39    /**
40      * Initializes member variables, but not the head object.
41      *
42      * @param c  The classifier being reranked.
43      * @param cc A coherence classifier to do the re-ranking.
44      * @param a  A value for {@link #alpha}.
45     **/
46    public ClusterMerge(DataCoref c, Learner cc, double a)
47    {
48      this(null, c, cc, a);
49    }
50  
51    /**
52      * Initializes member variables including the head object.
53      *
54      * @param h  The head object.
55      * @param c  The classifier being reranked.
56      * @param cc A coherence classifier to do the re-ranking.
57      * @param a  A value for {@link #alpha}.
58     **/
59    public ClusterMerge(Object h, DataCoref c, Learner cc, double a)
60    {
61      super(h);
62      classifier = c;
63      cohere = cc;
64      alpha = a;
65    }
66  
67  
68    /** Performs the cluster merging inference. */
69    protected void infer() throws Exception
70    {
71  
72      instantiateVariables();
73      Normalizer norm = getNormalizer(cohere);
74  
75    }
76  
77  
78    /**
79      * Creates the first order variables involved in this inference problem.
80     **/
81    private void instantiateVariables()
82    {
83      Normalizer norm = getNormalizer(classifier);
84  
85      for (int i = 0; i < headDocument.sentences(); ++i)
86        for (int j = 0; j < headDocument.mentionsInSentence(i); ++j)
87        {
88          for (int k = 0; k < i; ++k)
89            for (int l = 0; l < headDocument.mentionsInSentence(k); ++l)
90            {
91              FirstOrderVariable v =
92                new FirstOrderVariable(
93                    classifier,
94                    headDocument.getMentionPair(headDocument.getMention(k, l),
95                                                headDocument.getMention(i, j)));
96              variables.put(v, v);
97            }
98  
99          for (int l = 0; l < j; ++l)
100         {
101           FirstOrderVariable v =
102             new FirstOrderVariable(
103                 classifier,
104                 headDocument.getMentionPair(headDocument.getMention(i, l),
105                                             headDocument.getMention(i, j)));
106           variables.put(v, v);
107         }
108       }
109   }
110 
111 
112   /**
113     * Retrieves the value of the specified variable as identified by the
114     * classifier and the object that produce that variable.
115     *
116     * @param c  The classifier producing the variable.
117     * @param o  The object from which the variable is produced.
118     * @return   The current value of the requested variable.
119    **/
120   public String valueOf(Learner c, Object o) throws Exception
121   {
122     infer();
123     return getVariable(new FirstOrderVariable(c, o)).getValue();
124   }
