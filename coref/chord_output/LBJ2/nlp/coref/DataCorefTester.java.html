8   /**
9     * This program is intended to test a coreference classifier that uses a
10    * {@link DataCoref} classifier in some way with the <i>B<sup>3</sup></i>
11    * evaluation metric.
12    *
13    * <h4>Usage</h4>
14    * <blockquote><pre>
15    *   java LBJ2.nlp.coref.DataCorefTester &lt;classifier&gt; \
16    *                                       &lt;test data&gt; \
17    *                                       &lt;threshold&gt; \
18    *                                       [&lt;verbosity=0&gt;]
19    * </pre></blockquote>
20    *
21    * <h4>Input</h4>
22    * <p> <code>&lt;classifier&gt;</code> is the fully qualified class name of
23    * the classifier to test.  <code>&lt;test data&gt;</code> is the name of a
24    * file containing test data.  <code>&lt;threshold&gt;</code> is used to set
25    * the threshold adjustment of the classifier.  The higher the integer
26    * supplied for <code>&lt;verbosity&gt;</code>, the more information will be
27    * output to <code>STDOUT</code>.
28    *
29    * <h4>Output</h4>
30    * <p> Precision, recall, and <i>F<sub>1</sub></i> are sent to
31    * <code>STDOUT</code> as computed by <i>B<sup>3</sup></i>.
32   **/
33  public class DataCorefTester
34  {
35    public static void main(String[] args)
36    {
37      String classifierName = null;
38      String testFile = null;
39      double threshold = 0;
40      int verbosity = 0;
41  
42      try
43      {
44        classifierName = args[0];
45        testFile = args[1];
46        threshold = Double.parseDouble(args[2]);
47        if (args.length > 3) verbosity = Integer.parseInt(args[3]);
48        if (args.length > 4) throw new Exception();
49      }
50      catch (Exception e)
51      {
52        System.err.println(
53    "usage: java LBJ2.nlp.coref.DataCorefTester <classifier> <test data> \\\n"
54  + "                                           <threshold> [<verbosity=0>]");
55        System.exit(1);
56      }
57  
58      Class classifierClass = null;
59      DataCoref classifier =
60        (DataCoref) ClassUtils.getClassifier(classifierName);
61      classifier.setThreshold(threshold);
62  
63      double[] results =
64        CoreferenceTester.test(classifier, testFile, verbosity);
65  
66      System.out.println("precision: " + results[0]);
67      System.out.println("recall:    " + results[1]);
68      System.out.println("F1:        " + results[2]);
69    }
70  }
