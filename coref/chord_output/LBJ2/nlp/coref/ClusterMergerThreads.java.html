29  /**
30   * Use this program to determine the extent of the correlation between the
31   * performances of a coreference and a coherence classifier on perturbations of
32   * a coreference labeling as produced by merging clusters.
33   * 
34   * <h4>Usage</h4> <blockquote>
35   * 
36   * <pre>
37   *   java LBJ2.nlp.coref.ClusterMerger &lt;coreference classifier&gt; \
38   *                                     &lt;coherence classifier&gt; \
39   *                                     &lt;test data&gt; \
40   *                                     &lt;base&gt; \
41   *                                     [&lt;verbosity=0&gt;]
42   * </pre>
43   * 
44   * </blockquote>
45   * 
46   * <h4>Input</h4>
47   * <p>
48   * <code>&lt;coreference classifier&gt;</code> is the fully qualified class name
49   * of the classifier involved in the correlation, which will also produce the
50   * initial clustering for the search procedure.
51   * <code>&lt;coherence classifier&gt;</code> is the fully qualified class name
52   * of the other classifier involved in the correlation. <code>&lt;test
53   * data&gt;</code> is the name of a file containing test data. The higher the
54   * integer supplied for <code>&lt;verbosity&gt;</code>, the more information
55   * will be output to <code>STDOUT</code>.

57   * <h4>Output</h4> GNU plot commands to create a correlation graph.

120     /**
121      * Initializes the search procedure for a given document with a specified
122      * filter.
123      * 
124      * @param d
125      *            The document whose clusters will be merged.
126      * @param c
127      *            A coreference classifier that takes pairs of {@link Mention}s
128      *            as input.
129      * @param f
130      *            A filter used during the search.
131      **/
132     public ClusterMergerThreads(Document d, Classifier c, Filter f) {
133         if (f == null)
134             filter = new Filter();
135         else
136             filter = f.clone(this);
137         predicted = new Document(d);
138         clusterNames = new String[predicted.fillInPredictions(c, 0)];
139         for (int i = 0; i < clusterNames.length; ++i)
140             clusterNames[i] = "e" + i;
141 
142         clusterSizes = new int[clusterNames.length];
143         for (int i = 0; i < predicted.sentences(); ++i)
144             for (int j = 0; j < predicted.mentionsInSentence(i); ++j) {
145                 Document.Mention m = predicted.getMention(i, j);
146                 int cluster = Integer.parseInt(m.getEntityID().substring(1));
147                 ++clusterSizes[cluster];
148             }
149 
150         depth = i = 0;
151         j = 0;
152         increment();
153     }

160     /**
161      * Return the next cluster merging or <code>null</code> if there are no
162      * more.
163      **/
164     public ClusterMergerThreads next() {
165         if (clusterNames == null || j >= clusterNames.length)
166             return null;
167         ClusterMergerThreads result = new ClusterMergerThreads();
168         result.predicted = predicted;
169         result.clusterNames = clusterNames.clone();
170         result.clusterNames[j] = result.clusterNames[i];
171         result.clusterSizes = clusterSizes.clone();
172         result.clusterSizes[i] += result.clusterSizes[j];
173         result.clusterSizes[j] = 0;
174         result.i = i;
175         result.j = j;
176         result.filter = filter.clone(result);
177         result.increment();
178         result.depth = depth + 1;
179         increment();
180         return result;
181     }

183     /**
184      * Sets the {@link #i} and {@link #j} variables to point to the next cluster
185      * pair to be merged together.
186      **/
187     private void increment() {
188         do {
189             if (++j == clusterNames.length) {
190                 for (++i; i < clusterNames.length && clusterSizes[i] == 0; ++i)
191                     ;
192                 j = i + 1;
193             }
194         } while (j < clusterNames.length && (clusterSizes[j] == 0 || filter.reject()));
195     }

283     /**
284      * Computes the correlation coefficients for the given data.
285      * 
286      * @param pairs
287      *            The data arranged as an array of pairs, the first element of
288      *            each pair coming from one variable to be correlated, and the
289      *            second element of each pair coming from the other.
290      * @param start
291      *            The index of the first pair in the correlation.
292      * @param end
293      *            One more than the index of the last pair in the correlation.
294      * @return The correlation coefficient <i>R</i> as well as the coefficients
295      *         <i>a</i> and <i>b</i> such that the line <i>ax + b</i> is the
296      *         best fit for the data, in the order mentioned, in an array.
297      **/
298     private static double[] correlate(double[][] pairs, int start, int end) {
299         double N = end - start;
300         if (N == 1)
301             return new double[] { 1, 0, pairs[start][1] };
302         double sumX = 0, sumY = 0, sumX2 = 0, sumY2 = 0, sumXY = 0;
303 
304         for (int i = start; i < end; ++i) {
305             sumX += pairs[i][0];
306             sumY += pairs[i][1];
307             sumX2 += pairs[i][0] * pairs[i][0];
308             sumY2 += pairs[i][1] * pairs[i][1];
309             sumXY += pairs[i][0] * pairs[i][1];
310         }
311 
312         double R = (sumXY - sumX * sumY / N) / Math.sqrt((sumX2 - sumX * sumX / N) * (sumY2 - sumY * sumY / N));
313         double a = (sumXY - sumX * sumY / N) / (sumX2 - sumX * sumX / N);
314         double b = (sumY - a * sumX) / N;
315         return new double[] { R, a, b };
316     }

551         for (String fileName : fileNames) {
552             File f = new File(fileName);
553 
554             if (!f.exists())
555                 throw new IllegalArgumentException("Delete: no such file or directory: " + fileName);
556 
557             if (!f.canWrite())
558                 throw new IllegalArgumentException("Delete: write protected: " + fileName);
559 
560             if (f.isDirectory()) {
561                 String[] files = f.list();
562                 if (files.length > 0)
563                     throw new IllegalArgumentException("Delete: directory not empty: " + fileName);
564             }
565 
566             boolean success = f.delete();

569                 throw new IllegalArgumentException("Delete: deletion failed");
570         }

766     /**
767      * Evaluates the specified document's coreference labels as well as the
768      * specified coherence classifier's ability to distinguish it from the gold
769      * labeled version and returns the results in an array.
770      * 
771      * @param document
772      *            The document to be evaluated.
773      * @param depth
774      *            The level in the search at which the document was produced.
775      * @param cohere
776      *            The coherence classifier.
777      * @param base
778      *            The base of the exponentiation in the sigmoid function applied
779      *            to the coherence classifier's output. Setting this parameter
780      *            to 0 results in no sigmoid function being applied.
781      * @return Coherence and coreference performance followed by the value of
782      *         <code>depth</code>, in that order, in an array.
783      **/
784     private static double[] getResults(Document document, int depth, Learner cohere, double base) {
785         Document[] pair = null;
786         if (new Random(document.getName().hashCode()).nextBoolean())
787             pair = new Document[] { document.getLabeled(), document };
788         else
789             pair = new Document[] { document, document.getLabeled() };
790         String label = coherenceLabeler.discreteValue(pair);
791         double score = cohere.scores(pair).get(label);
792         if (base > 0)
793             score = 1.0 / (1.0 + Math.pow(base, -score));
794 
795         double F1 = new CoreferenceTester(0).getF1(document, document.getLabeled());
796         return new double[] { score, F1, depth };
797     }

799     /**
800      * This method returns a new {@link ClusterMergerThreads.Filter} object that is
801      * "unattached" in the sense that it does not belong to any
802      * <code>ClusterMerger</code> object referenced anywhere else in the code.
803      * Of course, it is attached to some <code>ClusterMerger</code> object, but
804      * that object does not contain anything useful and should not be
805      * referenced. Thus, the object returned by this method should only be used
806      * as input to the various constructors in this package.
807      * 
808      * @param specifier
809      *            Can be <code>"none"</code> to create a
810      *            {@link ClusterMergerThreads.Filter}, <code>"size"</code> to create a
811      *            {@link ClusterMergerThreads.SizeFilter}, <code>"b3D"</code> to create
812      *            a {@link ClusterMergerThreads.B3DistanceFilter}, or
813      *            <code>"clusterD"</code> to create a
814      *            {@link ClusterMergerThreads.ClusterDistanceFilter}.
815      * @param argument
816      *            Passed to the constructors of
817      *            {@link ClusterMergerThreads.B3DistanceFilter} and
818      *            {@link ClusterMergerThreads.ClusterDistanceFilter}, and ignored by
819      *            other filters.
820      **/
821     public static Filter getUnattachedFilter(String specifier, double argument) {
822         Filter filter = null;
823 
824         if (specifier.equals("none"))
825             filter = new ClusterMergerThreads().new Filter();
826         else if (specifier.equals("size"))
827             filter = new ClusterMergerThreads().new SizeFilter();
828         else if (specifier.equals("b3D"))
829             filter = new ClusterMergerThreads().new B3DistanceFilter(argument);
830         else if (specifier.equals("clusterD"))
831             filter = new ClusterMergerThreads().new ClusterDistanceFilter(argument);
832         else {
833             System.err.println("Unrecognized filter specifier: '" + specifier + "'");
834             System.exit(1);
835         }
836 
837         return filter;
838     }

876     public class ClusterDistanceFilter extends Filter {
877         private double threshold;
878 
879         public ClusterDistanceFilter(double t) {
880             threshold = t;
881         }
882 
883         protected boolean reject() {
884             Vector<Vector<Document.Mention>> clusters = new Vector<Vector<Document.Mention>>();
885             clusters.addAll(getDocument().getAllChains());
886             double highest = -Double.MAX_VALUE;
887             int I = i, J = j;
888             for (int k = J - 1; k >= 0; --k)
889                 if (clusterSizes[k] == 0)
890                     --J;
891             for (int k = I - 1; k >= 0; --k)
892                 if (clusterSizes[k] == 0)
893                     --I;
894 
895             for (Document.Mention mi : clusters.get(I))
896                 for (Document.Mention mj : clusters.get(J))
897                     highest = Math.max(highest, coref.scores(new Document.Mention[] { mi, mj }).get("true"));
898 
899             return highest < threshold;
900         }
901 
902         protected Filter clone(ClusterMergerThreads merger) {
903             return merger.new ClusterDistanceFilter(threshold);
904         }
