3   public class StatisticsUtility {
4   
5       private int numberOfDocs;    // the maximum number of documents in the corpus 
6       private double[] execTimePerDoc; 
7       private double[] timeFirstParsing;
8       private double[] timeSecondParsing;
9       private double[] timeFirstIteration;
10      private int[] iterationsPerDocument;
11      
12      public StatisticsUtility(int numberOfDocuments) {
13          numberOfDocs = numberOfDocuments;
14          execTimePerDoc = new double[numberOfDocs];
15          timeFirstParsing = new double[numberOfDocs];
16          timeSecondParsing = new double[numberOfDocs];
17          timeFirstIteration = new double[numberOfDocs];
18          iterationsPerDocument = new int[numberOfDocs];
19      }
20  
21      public void setTimeForDocument(int indexOfDocument, double timeForThisDoc) {
22          execTimePerDoc[indexOfDocument] = timeForThisDoc;
23      }
24      
25      public double averageExecTimePerDoc(){
26          double totalTime = 0.0;
27          for (int i = 0; i < numberOfDocs; ++i){
28              totalTime += execTimePerDoc[i];
29          }
30          return totalTime / numberOfDocs;
31      }
32  
33      public void setTimeForFirstParsing(int indexDoc, double time) {
34          timeFirstParsing[indexDoc] = time;
35      }
36  
37      public void setTimeForSecondParsing(int indexOfDocument, double d) {
38          timeSecondParsing[indexOfDocument] = d;
39      }
40  
41      public void setTimeForFirstIteration(int indexOfDocument,
42              double firstIterationTime) {
43          timeFirstIteration[indexOfDocument] = firstIterationTime;
44      }

46      public void printStatistics() {
47          System.out.printf("The average time to process each document: %.2f\n", averageExecTimePerDoc());
48          
49          double maxTime = 0.0;
50          double minTime = 10000000000.00; //
51          int maxIndex = 0, minIndex = 0;
52          double totalFirstParsing = 0.0;
53          for (int i = 0; i < numberOfDocs; i++) {
54              System.out.println("document " + i + "========");
55              
56              double timePerDoc = (execTimePerDoc[i] + timeFirstParsing[i]);
57              System.out.printf("total execution time per doc: %.2f\n", timePerDoc);
58              System.out.printf("timeFirstParsing: %.2f\n", timeFirstParsing[i]);
59              System.out.printf("timeSeconParsing: %.2f\n", timeSecondParsing[i]);
60              System.out.printf("timeFirstIteration: %.2f\n", timeFirstIteration[i]);
61              System.out.println("iterations per Document: " + iterationsPerDocument[i]);
62              totalFirstParsing += timeFirstParsing[i];
63              
64              if (timePerDoc > maxTime) {
65                  maxTime = timePerDoc;
66                  maxIndex = i;
67              }
68              
69              if (timePerDoc < minTime) {
70                  minTime = timePerDoc;
71                  minIndex = i;
72              }
73          }
74          
75          System.out.printf("Total first parsing time: %.2f\n", totalFirstParsing);
76          System.out.printf("Document %d has MIN exec time of %.2f\n", minIndex, minTime);
77          System.out.printf("Document %d has MAX exec time of %.2f\n", maxIndex, maxTime);
78      }
