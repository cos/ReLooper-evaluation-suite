1   package LBJ2.nlp.coref;
2   
3   import java.io.*;
4   import LBJ2.classify.Classifier;
5   import LBJ2.util.ClassUtils;
6   
7   
8   /**
9     * A program that writes documents to HTML files with coreference predictions
10    * highlighted.
11    *
12    * <h4>Usage</h4>
13    * <blockquote><pre>
14    *   java LBJ2.nlp.coref.CorefToHTML &lt;test data&gt; &lt;directory&gt; \
15    *                                   [&lt;classifier&gt;]
16    * </pre></blockquote>
17    *
18    * <h4>Input</h4>
19    * <p> <code>&lt;test data&gt;</code> is the name of a
20    * file containing test data.  <code>&lt;directory&gt;</code> is the name of
21    * a directory into which the HTML will be generated.
22    * <code>&lt;classifier&gt;</code> is the fully qualified class name of the
23    * classifier to test.  If no classifier is specified, gold labels are used
24    * instead.
25    *
26    * <h4>Output</h4>
27    * HTML files are created in the specified directory displaying the files in
28    * the test data highlighted with coreference labels or predictions.
29   **/
30  public class CorefToHTML
31  {
32    public static void main(String[] args)
33    {
34      String classifierName = null;
35      String testFile = null;
36      String directory = null;
37  
38      try
39      {
40        testFile = args[0];
41        directory = args[1];
42        if (args.length == 3) classifierName = args[2];
43        if (args.length > 3) throw new Exception();
44      }
45      catch (Exception e)
46      {
47        System.out.println(
48          "usage: java LBJ2.nlp.coref.CorefToHTML <test data> <directory> \\\n"
49        + "                                       [<classifier>]");
50        System.exit(1);
51      }
52  
53      Classifier classifier = null;
54  
55      if (classifierName == null) directory += "/gold";
56      else
57      {
58        directory += "/" + classifierName;
59        classifier = ClassUtils.getClassifier(classifierName);
60      }
61  
62      new File(directory).mkdirs();
63      ACE2004DocumentParser parser = new ACE2004DocumentParser(testFile);
64      PrintStream out = null;
65  
66      try
67      {
68        out = new PrintStream(new FileOutputStream(directory + "/index.html"));
69      }
70      catch (Exception e)
71      {
72        System.err.println(
73            "Can't open " + directory + "/index.html for output: " + e);
74        System.exit(1);
75      }
76  
77      String testFileName = testFile;
78      int index = testFileName.lastIndexOf('/');
79      testFileName = testFileName.substring(index + 1);
80  
81      out.println("<html>");
82      out.println("<head>");
83      out.println("<title>Files from " + testFileName + "</title>");
84      out.println("</head>");
85      out.println("<body><ul>");
86  
87      for (Document d = (Document) parser.next(); d != null;
88           d = (Document) parser.next())
89      {
90        if (classifier != null) d.fillInPredictions(classifier, 0);
91        d.writeHTML(directory);
92        out.println(
93            "<li> <a href=\"" + d.getName() + "/all.html\">" + d.getName());
94      }
95  
96      out.println("</ul></body></html>");
97      out.close();
98    }
99  }
