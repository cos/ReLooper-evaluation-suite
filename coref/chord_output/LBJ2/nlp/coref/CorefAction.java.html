13  public class CorefAction {
14      private static final long serialVersionUID = -1072536969137444539L;
15  
16  
17      
18      static Classifier corefClassifier;
19      static Learner coherenceClassifier;
20      static Filter filter;
21      private Document initialDoc; // pre-computed to get around blocking IO
22      private ClusterMergerParser parser;
23      static Classifier coherenceLabeler;
24  
25      static int verbosity;
26      static double base;
27      static boolean DEBUG;
28      static PrintStream out;
29      static PrintStream performanceOutput;
30      private double totalP = 0;
31      private double totalR = 0;
32      private int totalMentions = 0;
33      private LinkedList<double[]> results = new LinkedList<double[]>();
34      static TestDiscrete tester;
35  
36      private int indexOfDocument;
37  
38      private StatisticsUtility statistics;
39  
40      public LinkedList<double[]> getResults() {
41          return results;
42      }
43

45          return totalMentions;
46      }

49          return totalP;
50      }

53          return totalR;

56      public CorefAction(ClusterMergerParser parser, Document initialDoc, int i) {
57  
58          this.parser = parser;
59          this.initialDoc = initialDoc;
60  
61          this.indexOfDocument = i;
62      }
63  
64      protected void compute() {
65          int counter = 0;
66          Document currentPredicted = null;
67          double currentF1 = 0;
68          double bestScore = -Double.MAX_VALUE, bestP = 0, bestR = 0;
69  
70          long startDoc = System.nanoTime();
71  
72          boolean firstIteration = true;
73          boolean secondParsing = true;
74          double firstIterationTime = 0; // to record the IO time
75  
76          int iterationCount = 0;
77          for (Document d = initialDoc; d != null;) {
78  
79              // The branch is taken on every brand new document, not on another
80              // partitioning of the same document
81              if (currentPredicted == null
82                      || currentPredicted.getLabeled() != d.getLabeled()) {
83  
84                  if (currentPredicted != null) {
85                      totalP += bestP;
86                      totalR += bestR;
87                      totalMentions += currentPredicted.totalMentions();
88                      bestScore = -Double.MAX_VALUE;
89                      bestP = bestR = 0;
90                  }
91  
92  //              System.out.println("Processing document " + indexOfDocument
93  //                      + ": " + d.getName() + " on thread " + Thread.currentThread());
94  
95                  currentPredicted = new Document(d.getLabeled());
96                  currentPredicted.fillInPredictions(corefClassifier, 0);
97                  currentF1 = new CoreferenceTester(verbosity).getF1(
98                          currentPredicted, d.getLabeled());
99              }
100 
101             Document[] pair = null;
102             boolean predictedFirst = new Random(d.getName().hashCode())
103                     .nextBoolean();
104             if (predictedFirst)
105                 pair = new Document[] { currentPredicted, d };
106             else
107                 pair = new Document[] { d, currentPredicted };
108 
109             double score = coherenceClassifier.scores(pair).get(
110                     "" + !predictedFirst);
111             if (base > 0)
112                 score = 1.0 / (1.0 + Math.pow(base, -score));
113 
114             double[] dB3 = new CoreferenceTester(verbosity).test(d, d
115                     .getLabeled()); // an array of size 3
116             double dF1 = 0;
117             if (dB3[0] + dB3[1] != 0)
118                 dF1 = F1(dB3[0], dB3[1]) / d.totalMentions();
119 
120             double difference = dF1 - currentF1;
121             if (Double.doubleToLongBits(score) != Double
122                     .doubleToLongBits(Double.NaN))
123                 results.add(new double[] { score, difference,
124                         parser.getLastDepth() });
125 
126             if (score > bestScore) {
127                 bestScore = score;
128                 bestP = dB3[0];
129                 bestR = dB3[1];
130             }
131 
132             String label = coherenceLabeler.discreteValue(pair).equals(
133                     "" + !predictedFirst) ? "higher" : "lower";
134             String prediction = score > 0 ? "higher" : "lower";
135             tester.reportPrediction(prediction, label);
136 
137             if (DEBUG) {
138                 String number = "" + counter++;
139                 while (number.length() < 3)
140                     number = "0" + number;
141                 number += "." + parser.getLastDepth();
142                 d.writeHTML("debug/" + number);
143                 out.println("<li> <a href=\"" + number + "/" + d.getName()
144                         + "/all.html\">" + number + "</a>");
145             }
146             if (firstIteration) {
147                 firstIteration = false;
148                 firstIterationTime = (System.nanoTime() - startDoc) / 1000000.00;
149                 statistics.setTimeForFirstIteration(indexOfDocument,
150                         firstIterationTime);
151             }
152 
153             if (secondParsing) {
154                 secondParsing = false;
155                 long startSecondParsing = System.nanoTime();
156                 d = (Document) parser.next();
157                 statistics.setTimeForSecondParsing(indexOfDocument, (System
158                         .nanoTime() - startSecondParsing) / 1000000.00);
159             } else
160                 d = (Document) parser.next();
161             iterationCount++;
162 
163         } // end of the loop
164 
165         totalP += bestP;
166         totalR += bestR;
167         totalMentions += currentPredicted.totalMentions();
168 
169         long endDoc = System.nanoTime();
170         double timeForThisDoc = (endDoc - startDoc) / 1000000.00;
171         statistics.setTimeForDocument(indexOfDocument, timeForThisDoc);
172         statistics.setIterationPerDocument(indexOfDocument, iterationCount);
173     }
174 
175     private static double F1(double p, double r) {
176         return 2 * p * r / (p + r);
177     }
178 }
