6   /**
7     * This parser returns arrays of <code>String</code>s representing the rows
8     * of a file in column format.  The input file is assumed to contain fields
9     * of non-whitespace characters separated by any amount of whitespace, one
10    * line of which is usually used to represent a word in a corpus.  This
11    * parser breaks a given line into one <code>String</code> per field,
12    * omitting all of the whitespace.  They are then returned in an array via
13    * <code>next()</code>.  If the input line is empty or contains only
14    * whitespace, a zero length array will be returned.
15    *
16    * @author Nick Rizzolo
17   **/
18  public class ColumnFormat extends LineByLine
19  {
20    /**
21      * Creates the parser.
22      *
23      * @param file The name of the file to parse.
24     **/
25    public ColumnFormat(String file) { super(file); }
26  
27  
28    /**
29      * Returns an array of <code>String</code>s representing the information in
30      * the columns of this row.
31     **/
32    public Object next()
33    {
34      String line = readLine();
35      if (line == null) return null;
36      if (line.equals("")) return new String[0];
37      return line.split("\\s+");
38    }
39  }
