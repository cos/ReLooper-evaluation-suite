6   /**
7     * Use this parser to return <code>LinkedVector</code> objects representing
8     * sentences given file names of POS bracket form files to parse.  These
9     * files are expected to have one sentence per line, and the format of each
10    * line is as follows: <br><br>
11    *
12    * <code>(pos1 spelling1) (pos2 spelling2) ... (posN spellingN)</code>
13    * <br><br>
14    *
15    * It is also expected that there will be exactly one space between a part of
16    * speech and the corresponding spelling and between a closing parenthesis
17    * and an opening parenthesis.
18    *
19    * @author Nick Rizzolo
20   **/
21  public class POSBracketToVector extends LineByLine
22  {
23    /**
24      * Creates the parser.
25      *
26      * @param file The file to parse.
27     **/
28    public POSBracketToVector(String file) { super(file); }
29  
30  
31    /**
32      * Retrieves the next <code>LinkedVector</code> from the files being
33      * parsed.
34     **/
35    public Object next()
36    {
37      String line = readLine();
38      if (line == null) return null;
39      return parsePOSBracketForm(line);
40    }
41  
42  
43    /**
44      * Given a single line of textual input (containing all and only the words
45      * in a single sentence) in the format shown above, this method parses and
46      * returns a <code>LinkedVector</code>.
47      *
48      * @param line A single line of text.
49      * @return     A <code>LinkedVector</code> representing the input text.
50     **/
51    public static LinkedVector parsePOSBracketForm(String line)
52    {
53      String[] tokens = line.trim().split(" ");
54      if (tokens.length == 0
55          || tokens.length == 1 && (tokens[0] == null || tokens[0].equals("")))
56        return new LinkedVector();
57  
58      int spaceIndex = line.indexOf(' ');
59      spaceIndex = line.indexOf(' ', spaceIndex + 1);
60      Word w = new Word(tokens[1].substring(0, tokens[1].length() - 1),
61                        tokens[0].substring(1),
62                        0,
63                        spaceIndex - 1);
64  
65      for (int i = 2; i < tokens.length; i += 2)
66      {
67        int start = spaceIndex + 1;
68        spaceIndex = line.indexOf(' ', spaceIndex + 1);
69        spaceIndex = line.indexOf(' ', spaceIndex + 1);
70  
71        w.next =
72          new Word(tokens[i + 1].substring(0, tokens[i + 1].length() - 1),
73                   tokens[i].substring(1),
74                   w,
75                   start,
76                   spaceIndex - 1);
77        w = (Word) w.next;
78      }
79  
80      return new LinkedVector(w);
81    }
82  
83  
84    /**
85      * Given textual input in the format shown below, this method parses and
86      * returns the <code>Word</code> that the text represents.  Expected
87      * format: <br><br>
88      *
89      * <code>(pos spelling)</code>
90      *
91      * @param text     Text representing a word in POS bracket form.
92      * @param previous The word that came before this word in the sentence.
93      * @return         A <code>Word</code> represented by the input text or
94      *                 <code>null</code> if the input does not represent a
95      *                 <code>Word</code>.
96     **/
97    public static Word parsePOSBracketForm(String text, Word previous)
98    {
99      if (text.charAt(0) != '(' || text.charAt(text.length() - 1) != ')')
100       return null;
101     String[] tokens = text.split(" ");
102     if (tokens.length != 2) return null;
103     return new Word(tokens[1].substring(0, tokens[1].length() - 1),
104                     tokens[0].substring(1),
105                     previous);
106   }
107 }
