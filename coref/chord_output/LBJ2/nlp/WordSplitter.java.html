3   import LBJ2.parse.*;

6   /**
7     * This parser takes the plain, unannotated {@link Sentence}s returned by
8     * another parser (e.g., {@link SentenceSplitter}) and splits them into
9     * {@link Word} objects.  Entire sentences now represented as
10    * {@link LBJ2.parse.LinkedVector}s are then returned one at a time by calls
11    * to the <code>next()</code> method.
12    *
13    * <p> A {@link #main(String[])} method is also implemented which applies
14    * this class to plain text in a straight-forward way.
15    *
16    * @author Nick Rizzolo
17   **/
18  public class WordSplitter implements Parser
19  {
20    /**
21      * Run this program on a file containing plain text, and it will produce
22      * the same text on <code>STDOUT</code> rearranged so that each line
23      * contains exactly one sentence, and so that character sequences deemed to
24      * be "words" are delimited by whitespace.
25      *
26      * <p> Usage:
27      * <code> java LBJ2.nlp.WordSplitter &lt;file name&gt; </code>
28      *
29      * @param args The command line arguments.
30     **/
31    public static void main(String[] args)
32    {
33      String filename = null;
34  
35      try
36      {
37        filename = args[0];
38        if (args.length > 1) throw new Exception();
39      }
40      catch (Exception e)
41      {
42        System.err.println("usage: java LBJ2.nlp.WordSplitter <file name>");
43        System.exit(1);
44      }
45  
46      WordSplitter splitter = new WordSplitter(new SentenceSplitter(filename));
47  
48      for (LinkedVector s = (LinkedVector) splitter.next(); s != null;
49           s = (LinkedVector) splitter.next())
50      {
51        if (s.size() > 0)
52        {
53          Word w = (Word) s.get(0);
54          System.out.print(w.form);
55          for (w = (Word) w.next ; w != null; w = (Word) w.next)
56            System.out.print(" " + w.form);
57        }
58  
59        System.out.println();
60      }
61    }
62  
63  
64    /** The {@link Sentence} returning parser. */
65    protected Parser parser;
66  
67  
68    /**
69      * Initializing constructor.
70      *
71      * @param p  The {@link Sentence} returning parser.
72     **/
73    public WordSplitter(Parser p) { parser = p; }
74  
75  
76    /**
77      * Returns {@link LBJ2.parse.LinkedVector}s of {@link Word} objects one at
78      * a time.
79     **/
80    public Object next()
81    {
82      Sentence sentence = (Sentence) parser.next();
83      if (sentence == null) return null;
84      return sentence.wordSplit();
85    }
86  
87  
88    /** Sets this parser back to the beginning of the raw data. */
89    public void reset() { parser.reset(); }
90  }
