107   /**
108     * Creates and returns a <code>LinkedVector</code> representation of this
109     * sentence in which every <code>LinkedChild</code> is a <code>Word</code>.
110     * Offset information is respected and propagated.
111     *
112     * @see    Word
113     * @return A <code>LinkedVector</code> representation of this sentence.
114    **/
115   public LinkedVector wordSplit()
116   {
117     LinkedList boundaries = new LinkedList();
118 
119     // Whitespace always signals a word boundary.
120     Matcher m = Pattern.compile("\\s+").matcher(text);
121     while (m.find())
122     {
123       myAdd(boundaries, m.start() - 1, ")whitespace");
124       myAdd(boundaries, m.end(), "(whitespace");
125     }
126 
127     // The beginning and end of the text are also word boundaries, unless
128     // there's whitespace there.
129     if (boundaries.size() > 0
130         && ((Integer) boundaries.getLast()).intValue() >= text.length())
131       boundaries.removeLast();
132     else myAdd(boundaries, text.length() - 1, ")$");
133 
134     if (boundaries.size() > 1
135         && ((Integer) boundaries.getFirst()).intValue() == -1)
136       boundaries.removeFirst();
137     else myAdd(boundaries, 0, "(^");
138 
139     Pattern pNoSpaceOrDigit = Pattern.compile("[^\\s\\d]");
140     Pattern pDigit = Pattern.compile("\\d");
141     Pattern pDigitCommaNoDigit = Pattern.compile("\\d,\\D");
142     Pattern pNoDigitCommaDigit = Pattern.compile("\\D,\\d");
143 
144     // Commas are separate words unless they're part of a number.
145     for (int i = text.indexOf(','); i != -1; i = text.indexOf(',', i + 1))
146     {
147       if (i > 0
148           && (pNoSpaceOrDigit.matcher(text.substring(i - 1, i)).find()
149               || i + 1 == text.length()
150                  && pDigit.matcher(text.substring(i - 1, i)).find()
151               || i + 1 < text.length()
152                  && pDigitCommaNoDigit
153                     .matcher(text.substring(i - 1, i + 2)).find()))
154       {
155         myAdd(boundaries, i - 1, ")comma1");
156         myAdd(boundaries, i, "(comma1");
157       }
158 
159       if (i + 1 < text.length()
160           && (pNoSpaceOrDigit.matcher(text.substring(i + 1, i + 2)).find()
161               || i == 0 && pDigit.matcher(text.substring(i + 1, i + 2)).find()
162               || i > 0
163                  && pNoDigitCommaDigit.matcher(text.substring(i - 1, i + 2))
164                                       .find()))
165       {
166         myAdd(boundaries, i, ")comma2");
167         myAdd(boundaries, i + 1, "(comma2");
168       }
169     }

171     Pattern pApostropheMask = Pattern.compile("[^\\s,']");
172     Pattern pAbbreviation = Pattern.compile("[A-Za-z]'[A-Za-z]");
173     Pattern pPossessive = Pattern.compile("s[^A-Za-z']");
174     Pattern pShortWill = Pattern.compile("ll[^A-Za-z']");
175 
176     // Apostrophes are handled by making consecutive occurrences a single
177     // separate word and treating all other occurences as abbreviations which
178     // should not be separated, with the following exceptions which are
179     // considered contractions:
180     //    '         Plural possessive (must follow the letter 's')
181     //    'd        "I'd", "he'd", "they'd"
182     //    'll       "I'll", "he'll", "they'll"
183     //    'm        "I'm"
184     //    're       "they're"
185     //    's        Possessive
186     //    've       "I've", "they've"
187     //    n't       "can't", "won't", "shouldn't", "aren't"
188     for (int i = text.indexOf('\''); i != -1; i = text.indexOf('\'', i + 1))
189     {
190       if (i - 1 > 0 && Character.isLetter(text.charAt(i - 2))
191           && text.charAt(i - 1) == 'n' && i + 1 < text.length()
192           && text.charAt(i + 1) == 't'
193           && (i + 2 == text.length()
194               || !Character.isLetter(text.charAt(i + 2))
195                  && text.charAt(i + 2) != '\''))
196       {
197         myAdd(boundaries, i - 2, ")n't");
198         myAdd(boundaries, i - 1, "(n't");
199       }
200       else
201         if (i > 0
202             && (pApostropheMask.matcher(text.substring(i - 1, i)).find()
203                    && i + 1 < text.length() && text.charAt(i + 1) == '\''
204                 || text.charAt(i - 1) == 's'
205                    && (i + 1 == text.length()
206                        || !Character.isLetter(text.charAt(i + 1))
207                           && text.charAt(i + 1) != '\'')
208                 || Character.isLetter(text.charAt(i - 1))
209                    && (i + 1 < text.length()
210                           && (i + 2 == text.length()
211                               || !Character.isLetter(text.charAt(i + 2))
212                                  && text.charAt(i + 2) != '\'')
213                           && (text.charAt(i + 1) == 'd'
214                               || text.charAt(i + 1) == 'm'
215                               || text.charAt(i + 1) == 's')
216                        || i + 2 < text.length()
217                           && (i + 3 == text.length()
218                               || !Character.isLetter(text.charAt(i + 3))
219                                  && text.charAt(i + 3) != '\'')
220                           && (text.substring(i + 1, i + 3).equals("ll")
221                               || text.substring(i + 1, i + 3).equals("re")
222                               || text.substring(i + 1, i + 3).equals("ve")))
223                 || text.charAt(i - 1) == '.' && i - 1 > 0
224                    && Character.isLetter(text.charAt(i - 2))
225                    && i + 1 < text.length()
226                    && (i + 2 == text.length()
227                        || !Character.isLetter(text.charAt(i + 2))
228                           && text.charAt(i + 2) != '\'')
229                    && text.charAt(i + 1) == 's'))
230         {
231           myAdd(boundaries, i - 1, ")contraction1");
232           myAdd(boundaries, i, "(contraction1");
233         }
234 
235       if (i + 1 < text.length()
236           && pApostropheMask.matcher(text.substring(i + 1, i + 2)).find()
237           && (!Character.isLetter(text.charAt(i + 1))
238               || i > 0 && text.charAt(i - 1) == '\''))
239       {
240         myAdd(boundaries, i, ")contraction2");
241         myAdd(boundaries, i + 1, "(contraction2");
242       }
243     }

245     Pattern pColonMask = Pattern.compile("[^\\s,':]");
246     Pattern pColonSeparator = Pattern.compile("\\d:\\d");
247     // Colons get separated into their own word unless it looks like they're
248     // part of a time (or some other useful structure involving digits) or a
249     // URL.
250     for (int i = text.indexOf(':'); i != -1; i = text.indexOf(':', i + 1))
251       if (!(i >= 2 && i + 2 < text.length()
252               && pColonSeparator.matcher(text.substring(i - 2, i + 3)).find()
253             || i > 2 && i + 2 < text.length()
254                && (text.substring(i - 2, i + 3).equals("tp://")
255                    || text.substring(i - 2, i + 3).equals("TP://"))
256             || partOfURL(i)))
257       {
258         if (i >= 1 && pColonMask.matcher(text.substring(i - 1, i)).find())
259         {
260           myAdd(boundaries, i - 1, ")colon1");
261           myAdd(boundaries, i, "(colon1");
262         }
263 
264         if (i + 1 < text.length()
265             && pColonMask.matcher(text.substring(i + 1, i + 2)).find())
266         {
267           myAdd(boundaries, i, ")colon2");
268           myAdd(boundaries, i + 1, "(colon2");
269         }
270       }
271 
272     Pattern pSlashMask = Pattern.compile("[^\\s,':/]");
273     Pattern pSlashSeparator = Pattern.compile("\\d/\\d");
274 
275     // Slashes get separated into their own word unless it looks like they're
276     // part of a date (or some other useful structure involving digits) or a
277     // URL.
278     for (int i = text.indexOf('/'); i != -1; i = text.indexOf('/', i + 1))
279       if (!(i >= 2 && i + 2 < text.length()
280               && pSlashSeparator.matcher(text.substring(i - 2, i + 3)).find()
281             || i > 3 && i + 1 < text.length()
282                && (text.substring(i - 3, i + 2).equals("tp://")
283                    || text.substring(i - 3, i + 2).equals("TP://"))
284             || i > 4
285                && (text.substring(i - 4, i + 1).equals("tp://")
286                    || text.substring(i - 4, i + 1).equals("TP://"))
287             || partOfURL(i)))
288       {
289         if (i >= 1 && pSlashMask.matcher(text.substring(i - 1, i)).find())
290         {
291           myAdd(boundaries, i - 1, ")slash1");
292           myAdd(boundaries, i, "(slash1");
293         }
294 
295         if (i + 1 < text.length()
296             && pSlashMask.matcher(text.substring(i + 1, i + 2)).find())
297         {
298           myAdd(boundaries, i, ")slash2");
299           myAdd(boundaries, i + 1, "(slash2");
300         }
301       }
302 
303     Pattern pDashMask = Pattern.compile("[^\\s,':/-]");
304     Pattern pDashSeparator = Pattern.compile("\\w-\\w");
305     Pattern pNegative1 = Pattern.compile("-\\.?\\d");
306     Pattern pNegative2 = Pattern.compile("\\s-\\.?\\d");
307 
308     // Dashes get separated into their own words unless it looks like they're
309     // part of some useful structure like a compound word, a number, or a URL.
310     for (int i = text.indexOf('-'); i != -1; i = text.indexOf('-', i + 1))
311       if (!(i + 1 < text.length() && i >= 1
312               && pDashSeparator.matcher(text.substring(i - 1, i + 2)).find()
313             || (i + 2 < text.length()
314                 && (i == 0
315                       && pNegative1.matcher(text.substring(i, i + 3)).find()
316                     || i > 0
317                        && pNegative2.matcher(text.substring(i - 1, i + 3))
318                                     .find()))
319             || partOfURL(i)))
320       {
321         if (i >= 1 && pDashMask.matcher(text.substring(i - 1, i)).find())
322         {
323           myAdd(boundaries, i - 1, ")dash1");
324           myAdd(boundaries, i, "(dash1");
325         }
326 
327         if (i + 1 < text.length()
328             && pDashMask.matcher(text.substring(i + 1, i + 2)).find())
329         {
330           myAdd(boundaries, i, ")dash2");
331           myAdd(boundaries, i + 1, "(dash2");
332         }
333       }
334 
335     Pattern pDollarMask = Pattern.compile("[^\\s,':/\\$-]");
336     Pattern pMoney1 = Pattern.compile("\\$\\.?\\d");
337     Pattern pMoney2 = Pattern.compile("(\\s|-)\\$\\.?\\d");
338 
339     // Dollar signs get separated into their own words unless it looks like
340     // they're in fact delimiting the start of a dollar amount, or are part of
341     // a URL.
342     for (int i = text.indexOf('$'); i != -1; i = text.indexOf('$', i + 1))
343       if (!(i == 0 && i + 2 < text.length()
344               && pMoney1.matcher(text.substring(i, i + 3)).find()
345             || i > 0 && i + 2 < text.length()
346                && pMoney2.matcher(text.substring(i - 1, i + 3)).find()
347             || partOfURL(i)))
348       {
349         if (i >= 1 && pDollarMask.matcher(text.substring(i - 1, i)).find())
350         {
351           myAdd(boundaries, i - 1, ")dollar1");
352           myAdd(boundaries, i, "(dollar1");
353         }
354 
355         if (i + 1 < text.length()
356             && pDollarMask.matcher(text.substring(i + 1, i + 2)).find())
357         {
358           myAdd(boundaries, i, ")dollar2");
359           myAdd(boundaries, i + 1, "(dollar2");
360         }
361       }
362 
363     Pattern pBeforeElipsis = Pattern.compile("[^\\s,':/\\$\\.-]\\.\\.\\.");
364     Pattern pAfterElipsis = Pattern.compile("\\.\\.\\.[^\\s,':/\\$\\.-]");
365 
366     // Three or more consecutive periods form their own word.
367     for (int i = text.indexOf('.'); i != -1; i = text.indexOf('.', i + 1))
368     {
369       if (i > 0 && i + 2 < text.length()
370           && pBeforeElipsis.matcher(text.substring(i - 1, i + 3)).find())
371       {
372         myAdd(boundaries, i - 1, ")ellipsis1");
373         myAdd(boundaries, i, "(ellipsis1");
374       }
375 
376       if (i >= 2 && i + 1 < text.length()
377           && pAfterElipsis.matcher(text.substring(i - 2, i + 2)).find())
378       {
379         myAdd(boundaries, i, ")ellipsis2");
380         myAdd(boundaries, i + 1, "(ellipsis2");
381       }
382     }
383 
384     // If the last occurrence of a period in the sentence comes after all
385     // occurrences of letters and digits, it is an end of sentence marker
386     // which constitutes its own word, unless it appears immediately after two
387     // other periods.
388     int period = text.lastIndexOf('.');
389     if (period != -1)
390     {
391       boolean endOfSentence = true;
392       for (int i = period + 1; i < text.length() && endOfSentence; ++i)
393         endOfSentence = !Character.isLetterOrDigit(text.charAt(i));
394 
395       if (endOfSentence)
396       {
397         if (period >= 1
398             && (text.charAt(period - 1) != '.' || period == 1
399                 || text.charAt(period - 2) != '.')
400             && pDollarMask.matcher(text.substring(period - 1, period)).find())
401         {
402           myAdd(boundaries, period - 1, ")period1");
403           myAdd(boundaries, period, "(period1");
404         }
405 
406         if (period + 1 < text.length()
407             && (period == 0 || text.charAt(period - 1) != '.' || period == 1
408                 || text.charAt(period - 2) != '.')
409             && pDollarMask.matcher(text.substring(period + 1, period + 2))
410                           .find())
411         {
412           myAdd(boundaries, period, ")period2");
413           myAdd(boundaries, period + 1, "(period2");
414         }
415       }
416       else period = -1;
417     }
418 
419     // All other punctuation marks constitute their own words, unless they
420     // appear immediately after themselves (consecutive identical punctuation
421     // marks form a single word) or are part of a URL.
422     Pattern pPunctuation = Pattern.compile("[^\\s\\w,'\\.:/\\$-]");
423     m = pPunctuation.matcher(text);
424 
425     while (m.find())
426       if (!partOfURL(m.start()))
427       {
428         if (m.start() + 1 < text.length()
429             && text.charAt(m.start()) != text.charAt(m.start() + 1)
430             && m.start() + 1 != period
431             && pPunctuation
432                .matcher(text.substring(m.start() + 1, m.start() + 2)).find())
433         {
434           myAdd(boundaries, m.start(), ")punctuation1");
435           myAdd(boundaries, m.start() + 1, "(punctuation1");
436         }
437       }
438 
439     m = Pattern.compile("[^\\s\\w,'\\.:/\\$-]\\w|\\w[^\\s\\w,'\\.:/\\$-]")
440         .matcher(text);
441     while (m.find())
442       if (!partOfURL(m.start()))
443       {
444         myAdd(boundaries, m.start(), ")punctuation2");
445         myAdd(boundaries, m.start() + 1, "(punctuation2");
446       }
447 
448     // Now we just have to create the LinkedVector.
449     Integer[] temp =
450       (Integer[]) boundaries.toArray(new Integer[boundaries.size()]);
451     int[] I = new int[temp.length];
452     for (int i = 0; i < I.length; ++i) I[i] = temp[i].intValue();
453     Arrays.sort(I);
454 
455     Word w = new Word(text.substring(I[0], I[1] + 1), I[0] + start,
456                       I[1] + start);
457     for (int i = 2; i < I.length; i += 2)
458     {
459       w.next = new Word(text.substring(I[i], I[i + 1] + 1),
460                         w,
461                         I[i] + start,
462                         I[i + 1] + start);
463       w = (Word) w.next;
464     }
465 
466     inURL = null;
467     return new LinkedVector(w);
468   }

471   /**
472     * Does a simple check to determine if the symbol at the specified index in
473     * the specified string is likely to be part of a URL.  If the specified
474     * text contains any of the following strings before the specified symbol,
475     * and there is no whitespace in between the two, the specified symbol is
476     * deemed likely to be part of a URL.
477     *
478     * @param index  The index of the symbol in question.
479     * @return       <code>true</code> if and only if the specified symbol
480     *               appears to be part of a URL.
481    **/
482   private boolean partOfURL(int index)
483   {
484     if (inURL != null) return inURL[index];
485     inURL = new boolean[text.length()];
486 
487     String pattern = "(?i)(" + protocols[0];
488     for (int i = 1; i < protocols.length; ++i) pattern += "|" + protocols[i];
489     pattern += ")://\\S+|[a-zA-Z0-9][a-zA-Z0-9-]*\\.(" + topLevelDomains[0];
490     for (int i = 0; i < topLevelDomains.length; ++i)
491       pattern += "|" + topLevelDomains[i];
492     pattern += ")(/\\S+)?";
493 
494     Matcher m = Pattern.compile(pattern).matcher(text);
495     while (m.find())
496       for (int i = m.start(); i < m.end(); ++i) inURL[i] = true;
497     return inURL[index];
498   }
