6   /**
7     * Use this parser to return <code>Word</code> objects given file names of
8     * POS bracket form files to parse.  These files are expected to have one
9     * sentence per line, and the format of each line is as follows: <br><br>
10    *
11    * <code>(pos1 spelling1) (pos2 spelling2) ... (posN spellingN)</code>
12    * <br><br>
13    *
14    * It is also expected that there will be exactly one space between a part of
15    * speech and the corresponding spelling and between a closing parenthesis
16    * and an opening parenthesis.
17    *
18    * @deprecated As of LBJ release 2.0.4, the functionality of this class has
19    *             been superceded by the {@link LBJ2.parse.ChildrenFromVectors}
20    *             parser used in conjunction with {@link POSBracketToVector}.
21    * @author     Nick Rizzolo
22   **/
23  public class POSBracketToWord extends POSBracketToVector
24  {
25    /**
26      * The next word to return, or <code>null</code> if we need a new sentence.
27     **/
28    private Word currentWord;
29  
30  
31    /**
32      * Adds the given file name to the queue.
33      *
34      * @param file The file name to add to the queue.
35     **/
36    public POSBracketToWord(String file) { super(file); }
37  
38  
39    /**
40      * Retrieves the next sentence from the files being parsed.
41      *
42      * @return A <code>LinkedVector</code> representation of the next sentence.
43     **/
44    public Object next()
45    {
46      if (currentWord == null)
47      {
48        LinkedVector vector = (LinkedVector) super.next();
49        if (vector != null) currentWord = (Word) vector.get(0);
50      }
51  
52      Word result = currentWord;
53      if (currentWord != null) currentWord = (Word) currentWord.next;
54      return result;
55    }
56  }
