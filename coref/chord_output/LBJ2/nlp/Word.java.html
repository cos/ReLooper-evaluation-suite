6   /**
7     * Implementation of a word for natural language processing.  Please note
8     * that in general, one can only count on the <code>form</code> and
9     * <code>capitalized</code> fields described below having meaningful values.
10    * The <code>form</code> field can be assumed to be filled in because it's
11    * hard to imagine a situation in which a <code>Word</code> object should be
12    * created without any knowledge of how that word appeared in text.  The
13    * <code>capitalized</code> field is computed from the <code>form</code> by
14    * this class' constructor.
15    *
16    * <p> <i>All other fields must be obtained or computed externally.  Space is
17    * provided for them in this class' implementation as a convenience, since we
18    * expect the user will make frequent use of these fields.</i>
19    *
20    * <p> This class extends from {@link LBJ2.parse.LinkedChild}.  Of course,
21    * this means that objects of this class contain references to both the
22    * previous and the next word in the sentence.  Constructors are available
23    * that take the previous word as an argument, setting that reference.  Thus,
24    * a useful technique for constructing all the words in a sentence will
25    * involve code that looks like this (where <code>form</code> is a
26    * {@link java.lang.String}):
27    *
28    * <blockquote>
29    *   <code>
30    *   Word current = new Word(form);<br>
31    *   <i>a loop of some sort</i><br>
32    *   {<br>
33    *   &nbsp;&nbsp;&nbsp;&nbsp;current.next = new Word(form, current);<br>
34    *   &nbsp;&nbsp;&nbsp;&nbsp;current = current.next;<br>
35    *   }<br>
36    * </blockquote>
37    *
38    * @author Nick Rizzolo
39   **/
40  public class Word extends LinkedChild
41  {
42    /** The actual text from the corpus that represents the word. */
43    public String form;
44    /**
45      * Whether or not the word is capitalized is determined automatically by
46      * the constructor.
47     **/
48    public boolean capitalized;
49    /** Names the part of speech of this word. */
50    public String partOfSpeech;
51    /** The base form of the word. */
52    public String lemma;
53    /** An indication of the meaning or usage of this instance of this word. */
54    public String wordSense;
55

57    /**
58      * When all that is known is the spelling of the word.
59      *
60      * @param f  The actual text of the word.
61     **/
62    public Word(String f) { this(f, null, null); }
63  
64    /**
65      * Sets the actual text and the part of speech.
66      *
67      * @param f    The actual text of the word.
68      * @param pos  A token representing the word's part of speech.
69     **/
70    public Word(String f, String pos) { this(f, pos, null); }
71  
72    /**
73      * This constructor is useful when the sentence is being parsed forwards.
74      *
75      * @param f  The actual text of the word.
76      * @param p  The word that came before this one in the sentence.
77     **/
78    public Word(String f, Word p) { this(f, null, p); }
79  
80    /**
81      * This constructor is useful when the sentence is being parsed forwards.
82      *
83      * @param f    The actual text of the word.
84      * @param pos  A token representing the word's part of speech.
85      * @param p    The word that came before this one in the sentence.
86     **/
87    public Word(String f, String pos, Word p) { this(f, pos, p, -1, -1); }
88  
89    /**
90      * When you have offset information.
91      *
92      * @param f      The actual text of the word.
93      * @param start  The offset into the parent document at which the first
94      *               character of this word is found.
95      * @param end    The offset into the parent document at which the last
96      *               character of this word is found.
97     **/
98    public Word(String f, int start, int end)
99    {
100     this(f, null, null, start, end);
101   }

103   /**
104     * When you have offset information.
105     *
106     * @param f      The actual text of the word.
107     * @param pos    A token representing the word's part of speech.
108     * @param start  The offset into the parent document at which the first
109     *               character of this word is found.
110     * @param end    The offset into the parent document at which the last
111     *               character of this word is found.
112    **/
113   public Word(String f, String pos, int start, int end)
114   {
115     this(f, pos, null, start, end);
116   }

119     * This constructor is useful when the sentence is being parsed forwards.
120     *
121     * @param f      The actual text of the word.
122     * @param p      The word that came before this one in the sentence.
123     * @param start  The offset into the parent document at which the first
124     *               character of this word is found.
125     * @param end    The offset into the parent document at which the last
126     *               character of this word is found.
127    **/
128   public Word(String f, Word p, int start, int end)

130     this(f, null, p, start, end);
131   }

133   /**
134     * This constructor is useful when the sentence is being parsed forwards.
135     *
136     * @param f      The actual text of the word.
137     * @param pos    A token representing the word's part of speech.
138     * @param p      The word that came before this one in the sentence.
139     * @param start  The offset into the parent document at which the first
140     *               character of this word is found.
141     * @param end    The offset into the parent document at which the last
142     *               character of this word is found.
143    **/
144   public Word(String f, String pos, Word p, int start, int end)
145   {
146     this(f, pos, null, null, p, start, end);
147   }

149   /**
150     * This constructor is useful when the sentence is being parsed forwards.
151     *
152     * @param f      The actual text of the word.
153     * @param pos    A token representing the word's part of speech.
154     * @param l      The base form of the word.
155     * @param sense  The sense of the word.
156     * @param p      The word that came before this one in the sentence.
157     * @param start  The offset into the parent document at which the first
158     *               character of this word is found.
159     * @param end    The offset into the parent document at which the last
160     *               character of this word is found.
161    **/
162   public Word(String f, String pos, String l, String sense, Word p, int start,
163               int end)
164   {
165     super(p, start, end);
166     form = f;
167     capitalized = f != null && f.length() > 0
168                   && Character.isUpperCase(f.charAt(0));
169     partOfSpeech = pos;
170     POS.fromToken(partOfSpeech);
171     lemma = l;
172     wordSense = sense;
173   }
174 
175 
176   /**
177     * The string representation of a word is its POS bracket form, or, if the
178     * part of speech is not available, it is just the spelling of the word.
179     * Note that the POS bracket form of a word also entails displaying left
180     * brackets (<code>"("</code>, <code>"["</code>, and <code>"{"</code>) as
181     * <code>"-LRB-"</code> and right brackets (<code>")"</code>,
182     * <code>"]"</code>, <code>"}"</code>) as <code>"-RRB-"</code>.
183     *
184     * @return The POS bracket form of this word, or just the spelling of the
185     *         word if the part of speech is not available.
186    **/
187   public String toString()
188   {
189     if (partOfSpeech == null) return form;
190     String form = this.form;
191 
192     if (form.length() == 1)
193     {
194       if ("([{".indexOf(form.charAt(0)) != -1) form = "-LRB-";
195       if (")]}".indexOf(form.charAt(0)) != -1) form = "-RRB-";
196     }
197 
198     return "(" + partOfSpeech + " " + form + ")";
199   }
