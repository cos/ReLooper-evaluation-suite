8   /**
9     * Use this class to extract sentences from plain text.  The user constructs
10    * an object of this class with the file name of a document written in
11    * natural English (i.e., with no annotations added or any type of
12    * preprocessing performed).  <b>It should be noted that this class will
13    * interpret empty lines that appear in the input as paragraph
14    * boundaries.</b>
15    *
16    * <p> The user can then retrieve <code>Sentence</code>s one at a time with
17    * the <code>next()</code> method, or all at once with the
18    * <code>splitAll()</code> method.  The returned <code>Sentence</code>s'
19    * <code>start</code> and <code>end</code> fields represent offsets into the
20    * file they were extracted from.  Every character in between those two
21    * offsets inclusive, including extra spaces, newlines, etc., is included in
22    * the <code>Sentence</code> as it appeared in the paragraph.
23    *
24    * <p> A {@link #main(String[])} method is also implemented which applies
25    * this class to plain text in a straight-forward way.
26    *
27    * @see    Sentence
28    * @author Nick Rizzolo
29   **/
30  public class SentenceSplitter extends LineByLine
31  {
32    /**
33      * Run this program on a file containing plain text, and it will produce
34      * the same text rearranged so that each line contains exactly one sentence
35      * on <code>STDOUT</code>.
36      *
37      * <p> Usage:
38      * <code> java LBJ2.nlp.SentenceSplitter &lt;file name&gt; </code>
39      *
40      * @param args The command line arguments.
41     **/
42    public static void main(String[] args)
43    {
44      String filename = null;
45  
46      try
47      {
48        filename = args[0];
49        if (args.length > 1) throw new Exception();
50      }
51      catch (Exception e)
52      {
53        System.err.println("usage: java LBJ2.nlp.SentenceSplitter <file name>");
54        System.exit(1);
55      }
56  
57      SentenceSplitter splitter = new SentenceSplitter(filename);
58  
59      for (Sentence s = (Sentence) splitter.next(); s != null;
60           s = (Sentence) splitter.next())
61      {
62        StringBuffer buffer = new StringBuffer(s.text);
63  
64        for (int i = 0; i < buffer.length(); ++i)
65        {
66          char c = buffer.charAt(i);
67          if (c == '\n' || c == '\r' || c == '\f') buffer.setCharAt(i, ' ');
68        }
69  
70        System.out.println(buffer);
71      }
72    }

135   /**
136     * This method is used to extract a paragraph at a time from the input.
137     *
138     * @return The extracted paragraph, or a string containing only whitespace
139     *         if no text remains in the input.
140    **/
141   protected String getParagraph()
142   {
143     String paragraph = "", line;
144 
145     for (line = readLine(); line != null && line.trim().equals("");
146          line = readLine())
147       paragraph += line + "\n";
148 
149     for (; line != null && !line.trim().equals(""); line = readLine())
150       paragraph += line + "\n";
151 
152     if (line != null) paragraph += line + "\n";
153     return paragraph;
154   }

157   /**
158     * Retrieves the next sentence off the queue and returns it.
159     *
160     * @return The next sentence found or <code>null</code> if there are no
161     *         more sentences.
162    **/
163   public Object next()
164   {
165     if (sentences.size() == 0)
166     {
167       String paragraph = getParagraph();
168       if (!paragraph.trim().equals("")) process(paragraph);
169       currentOffset += paragraph.length();
170     }
171 
172     if (sentences.size() == 0) return null;
173     return sentences.removeFirst();
174   }

177   /**
178     * Retrieves every sentence found in the input paragraphs that have been
179     * provided so far in array form.
180     *
181     * @return All sentences in the input paragraphs.
182    **/
183   public Sentence[] splitAll()
184   {
185     for (String paragraph = getParagraph(); !paragraph.trim().equals("");
186          paragraph = getParagraph())
187     {
188       if (!paragraph.trim().equals("")) process(paragraph);
189       currentOffset += paragraph.length();
190     }
191 
192     return (Sentence[]) sentences.toArray(new Sentence[sentences.size()]);
193   }

196   /**
197     * This method does the actual work, deciding where sentences begin and end
198     * and populating the <code>sentences</code> member variable.
199     *
200     * @param paragraph  The paragraph to process.
201    **/
202   protected void process(String paragraph)
203   {
204     if (paragraph.trim().equals("")) return;
205     Matcher m = Pattern.compile("([^-\\s]-\n\\s*|\\S)+").matcher(paragraph);
206     LinkedList w = new LinkedList();
207     while (m.find()) w.add(new Word(m.group(), m.start(), m.end() - 1));
208     Word[] words = (Word[]) w.toArray(new Word[w.size()]);
209 
210     int sentenceStart = words[0].start;
211     boolean dumpTrailingWords = true;
212     for (int i = 0; i < words.length; ++i)
213     {
214       int punctuationIndex = words[i].form.lastIndexOf('.');
215 
216       int index = words[i].form.lastIndexOf('?');
217       if (index > punctuationIndex) punctuationIndex = index;
218 
219       index = words[i].form.lastIndexOf('!');
220       if (index > punctuationIndex) punctuationIndex = index;
221 
222       if (punctuationIndex != -1)
223       {
224         Word next1 = (i + 1 < words.length) ? words[i + 1] : null;
225         Word next2 = (i + 2 < words.length) ? words[i + 2] : null;
226 
227         if (boundary(punctuationIndex, words[i], next1, next2))
228         {
229           sentences.add(
230               new Sentence(paragraph.substring(sentenceStart,
231                                                words[i].end + 1),
232                            currentOffset + sentenceStart,
233                            currentOffset + words[i].end));
234           if (i + 1 < words.length) sentenceStart = words[i + 1].start;
235           else dumpTrailingWords = false;
236         }
237       }
238     }
239 
240     if (dumpTrailingWords)
241       sentences.add(
242           new Sentence(paragraph.substring(sentenceStart,
243                                            words[words.length - 1].end + 1),
244                        currentOffset + sentenceStart,
245                        currentOffset + words[words.length - 1].end));
246   }

249   /**
250     * Determines whether the given punctuation represents the end of a
251     * sentence based on elements of the paragraph immediately surrounding the
252     * punctuation.
253     *
254     * @param index      The index of the punctuation in question in its word.
255     * @param word       The word containing the punctuation.
256     * @param next1      The word one after the word containing the
257     *                   punctuation.
258     * @param next2      The word two after the word containing the
259     *                   punctuation.
260     * punctuation before 
261    **/
262   protected boolean boundary(int index, Word word, Word next1, Word next2)
263   {
264     char punctuation = word.form.charAt(index);
265     Word prefix = new Word(word.form.substring(0, index));
266     Word suffix = new Word(word.form.substring(index + 1));
267     Word root = new Word(prefix.form);
268     while (root.form.length() > 0
269            && "\"'`{[(".indexOf(root.form.charAt(0)) != -1)
270       root.form = root.form.substring(1);
271 
272     if ("yahoo!".equalsIgnoreCase(root.form + punctuation)) return false;
273 
274     if (punctuation == '?' || punctuation == '!')
275       return next1 == null
276              || suffix.form.equals("")
277                 && (next1.capitalized || startsWithQuote(next1)
278                     || next1.form.equals(".")
279                     || next2 != null && next2.capitalized
280                        && (next1.form.equals("--")
281                            || next1.form.equals("-RBR-")))
282              || isClose(suffix) && hasStartMarker(next1);
283 
284     if (next1 == null) return true;
285 
286     if (suffix.form.equals(""))
287     {
288       if (startsWithQuote(next1) || startsWithOpenBracket(next1)) return true;
289 
290       if (next1.form.equals("-RBR-") && next2.form.equals("--"))
291         return false;
292 
293       if (isClosingBracket(next1)) return true;
294 
295       if (prefix.form.equals("") && next1.form.equals("."))
296         return false;
297 
298       if (next1.form.equals(".")) return true;
299 
300       if (next1.form.equals("--") && next2.capitalized
301           && endsWithQuote(prefix))
302         return false;
303 
304       if (next1.form.equals("--")
305             && (next2.capitalized || startsWithQuote(next2)))
306         return true;
307 
308       if (next1.capitalized || Character.isDigit(next1.form.charAt(0)))
309         return isTerminal(root)
310           || !((root.form.equals("p.m")
311                 || root.form.equals("a.m"))
312                  && isTimeZone(next1)
313                || isHonorific(root) || startsWithQuote(prefix)
314                || startsWithOpenBracket(prefix)
315                || Pattern.compile("^([A-Z]\\.)*[A-Z]$")
316                          .matcher(prefix.form).find()
317                   && !sentenceBeginner(next1));
318     }
319 
320     return isClose(suffix) && hasStartMarker(next1) && !isHonorific(root);
321   }

337     * Determines whether the first character of the argument is any of the
338     * three varieties of quotes: ' " `.
339     *

341     * @return   <code>true</code> if and only if the first character of the
342     *           argument is any of the three varieties of quotes.

344   protected boolean startsWithQuote(Word w)
345   {
346     if (w.form.length() == 0) return false;
347     return w.form.charAt(0) == '\'' || w.form.charAt(0) == '"'
348            || w.form.charAt(0) == '`';
