8   /**
9     * This class represents a method call.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class MethodInvocation extends StatementExpression
14  {
15    /**
16      * (&oslash;) This expression evaluates to the object whose method will be
17      * called.
18     **/
19    public Expression parentObject;
20    /** (&not;&oslash;) The name of the method to be invoked. */
21    public Name name;
22    /** (&not;&oslash;) The argument expressions passed to the method. */
23    public ExpressionList arguments;
24    /**
25      * Filled in by the <code>SemanticAnalysis</code> pass, this variable is
26      * set to <code>true</code> iff this invocation represents a classifier
27      * invocation.
28      *
29      * @see LBJ2.SemanticAnalysis
30     **/
31    public boolean isClassifierInvocation;
32    /**
33      * Filled in by the <code>SemanticAnalysis</code> pass, this variable is
34      * set to <code>true</code> iff this invocation is the argument of a
35      * learning classifier expression's evaluate clause.
36      *
37      * @see LBJ2.SemanticAnalysis
38     **/
39    public boolean isEvaluateArgument;
40    /**
41      * The <code>SemanticAnalysis</code> pass will let this
42      * <code>MethodInvocation</code> know if it is the immediate
43      * <code>value</code> child of a <code>SenseStatement</code> by setting
44      * this flag.
45      *
46      * @see LBJ2.SemanticAnalysis
47     **/
48    public boolean isSensedValue;

51    /**
52      * Initializing constructor.  Line and byte offset information is taken
53      * from the representation of the name.
54      *
55      * @param n  The name of the method being invoked.
56     **/
57    public MethodInvocation(Name n) { this(n, new ExpressionList()); }

59    /**
60      * Initializing constructor.  Line and byte offset information is taken
61      * from the representation of the name.
62      *
63      * @param n  The name of the method being invoked.
64      * @param a  The argument expressions passed to the method.
65     **/
66    public MethodInvocation(Name n, ExpressionList a)
67    {
68      this(null, n, a, n.line, n.byteOffset);
69    }

71    /**
72      * Parser's constructor.  Line and byte offset information is taken from
73      * the representation of the name.
74      *
75      * @param p  Represents the object whose method is being invoked.
76      * @param n  Token representing the name of the method being invoked.
77     **/
78    public MethodInvocation(Expression p, TokenValue n)
79    {
80      this(p, n, new ExpressionList());
81    }

83    /**
84      * Parser's constructor.  Line and byte offset information is taken from
85      * the representation of the name.
86      *
87      * @param p  Represents the object whose method is being invoked.
88      * @param n  Token representing the name of the method being invoked.
89      * @param a  The argument expressions passed to the method.
90     **/
91    public MethodInvocation(Expression p, TokenValue n, ExpressionList a)
92    {
93      this(p, new Name(n), a, n.line, n.byteOffset);
94    }
95  
96    /**
97      * Full constructor.
98      *
99      * @param p          Represents the object whose method is being invoked.
100     * @param n          The name of the method being invoked.
101     * @param a          The argument expressions passed to the method.
102     * @param line       The line on which the source code represented by this
103     *                   node is found.
104     * @param byteOffset The byte offset from the beginning of the source file
105     *                   at which the source code represented by this node is
106     *                   found.
107    **/
108   public MethodInvocation(Expression p, Name n, ExpressionList a, int line,
109                           int byteOffset)
110   {
111     super(line, byteOffset);
112     parentObject = p;
113     name = n;
114     arguments = a;
115     isClassifierInvocation = isEvaluateArgument = isSensedValue = false;
116   }

119   /**
120     * Returns a set of <code>Argument</code>s storing the name and type of
121     * each variable that is a subexpression of this expression.  This method
122     * cannot be run before <code>SemanticAnalysis</code> runs.
123    **/
124   public HashSet getVariableTypes()
125   {
126     HashSet result = new HashSet();
127     if (parentObject != null) result.addAll(parentObject.getVariableTypes());
128     result.addAll(name.getVariableTypes(true));
129     result.addAll(arguments.getVariableTypes());
130     return result;
131   }

134   /**
135     * Determines if there are any quantified variables in this expression.
136     * This method cannot be run before <code>SemanticAnalysis</code> runs.
137    **/
138   public boolean containsQuantifiedVariable()
139   {
140     return parentObject != null && parentObject.containsQuantifiedVariable()
141            || name.containsQuantifiedVariable(true)
142            || arguments.containsQuantifiedVariable();
143   }

146   /** Sets the <code>isSensedValue</code> flag. */
147   public void senseValueChild() { isSensedValue = true; }

150   /**
151     * Returns a hash code value for java hash structures.
152     *
153     * @return A hash code for this object.
154    **/
155   public int hashCode()
156   {
157     int result = name.hashCode() + arguments.hashCode();
158     if (parentObject != null) result += parentObject.hashCode();
159     return result;
160   }

163   /**
164     * Indicates whether some other object is "equal to" this one.
165     *
166     * @return <code>true</code> iff this object is the same as the argument.
167    **/
168   public boolean equals(Object o)
169   {
170     if (!(o instanceof MethodInvocation)) return false;
171     MethodInvocation i = (MethodInvocation) o;
172     return (parentObject == null && i.parentObject == null
173             || parentObject != null && parentObject.equals(i.parentObject))
174            && name.equals(i.name) && arguments.equals(i.arguments);
175   }

178   /**
179     * Returns an iterator used to successively access the children of this
180     * node.
181     *
182     * @return An iterator used to successively access the children of this
183     *         node.
184    **/
185   public ASTNodeIterator iterator()
186   {
187     ASTNodeIterator I = new ASTNodeIterator(parentObject == null ? 2 : 3);
188     if (parentObject != null) I.children[0] = parentObject;
189     I.children[I.children.length - 2] = name;
190     I.children[I.children.length - 1] = arguments;
191     return I;
192   }

195   /**
196     * Creates a new object with the same primitive data, and recursively
197     * creates new member data objects as well.
198     *
199     * @return The clone node.
200    **/
201   public Object clone()
202   {
203     return
204       new MethodInvocation(
205           (parentObject == null) ? null : (Expression) parentObject.clone(),
206           (Name) name.clone(), (ExpressionList) arguments.clone(), -1, -1);
207   }

219   /**
220     * Writes a string representation of this <code>ASTNode</code> to the
221     * specified buffer.  The representation written is parsable by the LBJ2
222     * compiler, but not very readable.
223     *
224     * @param buffer The buffer to write to.
225    **/
226   public void write(StringBuffer buffer)
227   {
228     if (parenthesized) buffer.append("(");
229 
230     if (parentObject != null)
231     {
232       parentObject.write(buffer);
233       buffer.append(".");
234     }
235 
236     name.write(buffer);
237     buffer.append("(");
238     arguments.write(buffer);
239     buffer.append(")");
240 
241     if (parenthesized) buffer.append(")");
242   }
