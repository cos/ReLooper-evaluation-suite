6   /**
7     * The root node of LBJ2's AST.  The parser will create only one of these and
8     * return it as the ultimate result of parsing.  It currently contains a
9     * single (optional) <code>PackageDeclaration</code>, a <code>List</code> of
10    * <code>ImportDeclaration</code>s (also optional), the global
11    * <code>SymbolTable</code>, and a <code>List</code> containing
12    * <code>ClassifierAssignment</code>s, <code>ConstraintDeclaration</code>s,
13    * and <code>InferenceDeclaration</code>s.
14    *
15    * <p>
16    * The <code>PackageDeclaration</code> specifies what package will contain
17    * the generated classes.  The <code>ImportDeclaration</code>s work similarly
18    * to <code>import</code> declarations in regular Java code: they allow the
19    * user to name classes in other packages without using their full package
20    * names.  The list of <code>Declaration</code>s comprises the LBJ2 program.
21    * The global <code>SymbolTable</code> simply associates variable names with
22    * their type information in the program.
23    *
24    * @author Nick Rizzolo
25   **/
26  public class AST extends ASTNode
27  {
28    /** The <code>symbolTable</code> variable mirrors this variable. */
29    public static final SymbolTable globalSymbolTable = new SymbolTable();
30  
31  
32    /**
33      * (&oslash;) An optional declaration of the package that generated classes
34      * should be a part of.
35     **/
36    public PackageDeclaration packageDeclaration;
37    /**
38      * (&not;&oslash;) The list of import statements at the top of the source
39      * file.
40     **/
41    public ImportList imports;
42    /**
43      * (&not;&oslash;) The list of classifier, constraint, and inference
44      * declarations representing the LBJ2 program.
45     **/
46    public DeclarationList declarations;
47  
48  
49    /**
50      * Initializes just the statement list.  Line and byte offset information
51      * are taken from the statement list.
52      *
53      * @param d  The declarations comprising the program.
54     **/
55    public AST(DeclarationList d)
56    {
57      this(null, new ImportList(), d, d.line, d.byteOffset);
58    }
59  
60    /**
61      * Initializes both lists.  Line and byte offset information are taken from
62      * the <code>import</code> list.
63      *
64      * @param i  The <code>import</code> declarations.
65      * @param d  The declarations comprising the program.
66     **/
67    public AST(ImportList i, DeclarationList d)
68    {
69      this(null, i, d, i.line, i.byteOffset);
70    }
71  
72    /**
73      * Initializes package declaration and statement list.  Line and byte
74      * offset information are taken from the <code>package</code> declaration.
75      *
76      * @param p  The <code>package</code> declaration.
77      * @param d  The declarations comprising the program.
78     **/
79    public AST(PackageDeclaration p, DeclarationList d)
80    {
81      this(p, new ImportList(), d, p.line, p.byteOffset);
82    }
83  
84    /**
85      * Initializes all member variables.  Line and byte offset information are
86      * taken from the <code>package</code> declaration.
87      *
88      * @param p  The <code>package</code> declaration.
89      * @param i  The <code>import</code> declarations.
90      * @param d  The declarations comprising the program.
91     **/
92    public AST(PackageDeclaration p, ImportList i, DeclarationList d)
93    {
94      this(p, i, d, p.line, p.byteOffset);
95    }
96  
97    /**
98      * Full constructor.
99      *
100     * @param p          The <code>package</code> declaration.
101     * @param i          The <code>import</code> declarations.
102     * @param d          The declarations comprising the program.
103     * @param line       The line on which the source code represented by this
104     *                   node is found.
105     * @param byteOffset The byte offset from the beginning of the source file
106     *                   at which the source code represented by this node is
107     *                   found.
108    **/
109   public AST(PackageDeclaration p, ImportList i, DeclarationList d, int line,
110              int byteOffset)
111   {
112     super(line, byteOffset);
113     packageDeclaration = p;
114     imports = i;
115     declarations = d;
116     symbolTable = globalSymbolTable;
117   }
118 
119 
120   /**
121     * Returns an iterator used to successively access the children of this
122     * node.
123     *
124     * @return An iterator used to successively access the children of this
125     *         node.
126    **/
127   public ASTNodeIterator iterator()
128   {
129     ASTNodeIterator I =
130       new ASTNodeIterator(packageDeclaration == null ? 2 : 3);
131     if (packageDeclaration != null) I.children[0] = packageDeclaration;
132     I.children[I.children.length - 2] = imports;
133     I.children[I.children.length - 1] = declarations;
134     return I;
135   }
136 
137 
138   /**
139     * Creates a new object with the same primitive data, and recursively
140     * creates new member data objects as well.
141     *
142     * @return The clone node.
143    **/
144   public Object clone()
145   {
146     PackageDeclaration p = null;
147     if (packageDeclaration != null)
148       p = (PackageDeclaration) packageDeclaration.clone();
149 
150     return new AST(p, (ImportList) imports.clone(),
151                    (DeclarationList) declarations.clone(), -1, -1);
152   }
153 
154 
155   /**
156     * Ensures that the correct <code>run()</code> method is called for this
157     * type of node.
158     *
159     * @param pass The pass whose <code>run()</code> method should be called.
160    **/
161   public void runPass(Pass pass) { pass.run(this); }
162 
163 
164   /**
165     * Writes a string representation of this <code>ASTNode</code> to the
166     * specified buffer.  The representation written is parsable by the LBJ2
167     * compiler, but not very readable.
168     *
169     * @param buffer The buffer to write to.
170    **/
171   public void write(StringBuffer buffer)
172   {
173     if (packageDeclaration != null)
174     {
175       packageDeclaration.write(buffer);
176       buffer.append(" ");
177     }
178 
179     imports.write(buffer);
180     buffer.append(" ");
181     declarations.write(buffer);
182   }
183 }
