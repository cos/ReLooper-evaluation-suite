6   /**
7     * Represents a continue statement.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class ContinueStatement extends Statement
12  {
13    /** (&oslash;) The label identifying the loop to continue, if any. */
14    public String label;
15  
16  
17    /**
18      * Full constructor.
19      *
20      * @param l          The label of the loop to continue.
21      * @param line       The line on which the source code represented by this
22      *                   node is found.
23      * @param byteOffset The byte offset from the beginning of the source file
24      *                   at which the source code represented by this node is
25      *                   found.
26     **/
27    public ContinueStatement(String l, int line, int byteOffset)
28    {
29      super(line, byteOffset);
30      label = l;
31    }
32  
33  
34    /**
35      * Returns an iterator used to successively access the children of this
36      * node.
37      *
38      * @return An iterator used to successively access the children of this
39      *         node.
40     **/
41    public ASTNodeIterator iterator() { return new ASTNodeIterator(0); }
42  
43  
44    /**
45      * Creates a new object with the same primitive data, and recursively
46      * creates new member data objects as well.
47      *
48      * @return The clone node.
49     **/
50    public Object clone() { return new ContinueStatement(label, -1, -1); }
51  
52  
53    /**
54      * Ensures that the correct <code>run()</code> method is called for this
55      * type of node.
56      *
57      * @param pass The pass whose <code>run()</code> method should be called.
58     **/
59    public void runPass(Pass pass) { pass.run(this); }
60  
61  
62    /**
63      * Writes a string representation of this <code>ASTNode</code> to the
64      * specified buffer.  The representation written is parsable by the LBJ2
65      * compiler, but not very readable.
66      *
67      * @param buffer The buffer to write to.
68     **/
69    public void write(StringBuffer buffer)
70    {
71      buffer.append("continue");
72      if (label != null) buffer.append(" " + label);
73      buffer.append(";");
74    }
75  }
