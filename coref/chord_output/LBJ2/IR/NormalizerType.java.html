6   /**
7     * A normalizer type is simply a place holder indicating that the name it is
8     * associated with in the symbol table is a normalizer function.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class NormalizerType extends Type
13  {
14    /** Default constructor. */
15    public NormalizerType()
16    {
17      super(-1, -1);
18  
19      try { myClass = Class.forName("LBJ2.learn.Normalizer"); }
20      catch (Exception e)
21      {
22        System.err.println("Class 'LBJ2.learn.Normalizer' not found.  "
23                           + "Aborting.");
24        System.exit(1);
25      }
26    }
27

29    /**
30      * Any two <code>NormalizerType</code>s are equivalent.
31      *
32      * @param o  The object whose equality with this object needs to be tested.
33      * @return   <code>true</code> if the two objects are equal, and
34      *           <code>false</code> otherwise.
35     **/
36    public boolean equals(Object o) { return o instanceof NormalizerType; }
