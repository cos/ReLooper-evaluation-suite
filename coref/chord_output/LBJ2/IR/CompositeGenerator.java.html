6   /**
7     * Represents a generator composed from several other classifiers.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class CompositeGenerator extends ClassifierExpression
12  {
13    /** (&not;&oslash;) The list of classifiers composing this classifier. */
14    public ClassifierExpressionList components;
15  
16  
17    /**
18      * Full constructor.  The line and byte offset are each set to -1.
19      *
20      * @param c  The list of components.
21     **/
22    public CompositeGenerator(ClassifierExpressionList c)
23    {
24      super(-1, -1);
25      components = c;
26    }
27  
28    /**
29      * Parser's constructor.
30      *
31      * @param e1 One <code>ClassifierExpression</code> to add.
32      * @param e2 Another <code>ClassifierExpression</code> to add.
33     **/
34    public CompositeGenerator(ClassifierExpression e1, ClassifierExpression e2)
35    {
36      super(e1.line, e2.byteOffset);
37  
38      if (e1 instanceof CompositeGenerator)
39      {
40        components = ((CompositeGenerator) e1).components;
41        if (e2 instanceof CompositeGenerator)
42          components.addAll(((CompositeGenerator) e2).components);
43        else components.add(e2);
44      }
45      else
46      {
47        components = new ClassifierExpressionList(e1);
48        components.add(e2);
49      }
50    }
51  
52  
53    /**
54      * Returns an iterator used to successively access the children of this
55      * node.
56      *
57      * @return An iterator used to successively access the children of this
58      *         node.
59     **/
60    public ASTNodeIterator iterator()
61    {
62      ASTNodeIterator I = new ASTNodeIterator(1);
63      I.children[0] = components;
64      return I;
65    }
66  
67  
68    /**
69      * Creates a new object with the same primitive data, and recursively
70      * creates new member data objects as well.
71      *
72      * @return The clone node.
73     **/
74    public Object clone()
75    {
76      return
77        new CompositeGenerator((ClassifierExpressionList) components.clone());
78    }
79  
80  
81    /**
82      * Ensures that the correct <code>run()</code> method is called for this
83      * type of node.
84      *
85      * @param pass The pass whose <code>run()</code> method should be called.
86     **/
87    public void runPass(Pass pass) { pass.run(this); }
88  
89  
90    /**
91      * Writes a string representation of this <code>ASTNode</code> to the
92      * specified buffer.  The representation written is parsable by the LBJ2
93      * compiler, but not very readable.
94      *
95      * @param buffer The buffer to write to.
96     **/
97    public void write(StringBuffer buffer)
98    {
99      if (parenthesized) buffer.append("(");
100     components.write(buffer);
101     if (parenthesized) buffer.append(")");
102   }
103 
104 
105   /**
106     * Creates a <code>StringBuffer</code> containing a shallow representation
107     * of this <code>ASTNode</code>.
108     *
109     * @return A <code>StringBuffer</code> containing a shallow text
110     *         representation of the given node.
111    **/
112   public StringBuffer shallow()
113   {
114     StringBuffer buffer = new StringBuffer();
115     returnType.write(buffer);
116     buffer.append(" ");
117     name.write(buffer);
118     buffer.append("(");
119     argument.write(buffer);
120     buffer.append(") <- ");
121     for (ClassifierExpressionList.ClassifierExpressionListIterator I =
122            components.listIterator();
123          I.hasNext(); )
124       buffer.append(I.nextItem().name + ", ");
125     return buffer;
126   }
127 }
