7   /**
8     * Currently, this is just a wrapper class for <code>LinkedList</code>.  The
9     * code that uses it looks a little cleaner when casts are all taken care of
10    * automatically.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class ConstantList extends List
15  {
16    /** Default constructor. */
17    public ConstantList() { super(-1, -1, ", "); }
18  
19    /**
20      * Initializing constructor.  Requires its argument to be
21      * non-<code>null</code>.
22      *
23      * @param c  A single <code>Constant</code> with which to initialize this
24      *           list.
25     **/
26    public ConstantList(Constant c)
27    {
28      super(c.line, c.byteOffset, ", ");
29      list.add(c);
30    }
31  
32    /**
33      * Creates an entire list from an array of values.
34      *
35      * @param a  The array of constant values.
36     **/
37    public ConstantList(String[] a)
38    {
39      super(-1, -1, ", ");
40      for (int i = 0; i < a.length; ++i) list.add(new Constant(a[i]));
41    }
42  
43  
44    /**
45      * Adds another <code>Constant</code> to the end of the list.
46      *
47      * @param c  A reference to the <code>Constant</code> to be added.
48     **/
49    public void add(Constant c) { list.add(c); }
50  
51  
52    /**
53      * Adds all the <code>Constant</code>s in another <code>ConstantList</code>
54      * to the end of this <code>ConstantList</code>.
55      *
56      * @param l  The list to be added.
57     **/
58    public void addAll(ConstantList l) { list.addAll(l.list); }
59  
60  
61    /**
62      * Transforms the list into an array of expressions.
63      *
64      * @return An array of constants containing references to every constant in
65      *         the list.
66     **/
67    public Constant[] toArray()
68    {
69      return (Constant[]) list.toArray(new Constant[list.size()]);
70    }
71  
72  
73    /**
74      * Two <code>ConstantList</code>s are equal when they contain the same
75      * elements in the same order as evaluated by the
76      * <code>Constant.equals(Object)</code> method.
77      *
78      * @param o  The object to test equality with.
79      * @return   <code>true</code> iff this <code>ConstantList</code> is
80      *           equivalent to the specified object as described above.
81     **/
82    public boolean equals(Object o)
83    {
84      if (!(o instanceof ConstantList)) return false;
85      ConstantList list = (ConstantList) o;
86      if (list.size() != size()) return false;
87  
88      ASTNodeIterator I = iterator();
89      for (ASTNodeIterator J = list.iterator(); J.hasNext(); )
90        if (!I.next().equals(J.next())) return false;
91      return true;
92    }
93

95    /**
96      * Returns an iterator used specifically to access the elements of this
97      * list.
98      *
99      * @return An iterator used specifically to access the elements of this
100     *         list.
101    **/
102   public ConstantListIterator listIterator()
103   {
104     return new ConstantListIterator();
105   }

108   /**
109     * Returns an iterator used to successively access the children of this
110     * node.
111     *
112     * @return An iterator used to successively access the children of this
113     *         node.
114    **/
115   public ASTNodeIterator iterator() { return listIterator(); }

118   /**
119     * Creates a new object with the same primitive data, and recursively
120     * creates new member data objects as well.
121     *
122     * @return The clone node.
123    **/
124   public Object clone()
125   {
126     ConstantList clone = new ConstantList();
127     for (Iterator i = list.iterator(); i.hasNext(); )
128       clone.list.add(((Constant) i.next()).clone());
129     return clone;
130   }

135     * type of node.
136     *
137     * @param pass The pass whose <code>run()</code> method should be called.
138    **/
139   public void runPass(Pass pass) { pass.run(this); }

142   /**
143     * Used to iterate though the children of a list of AST nodes.  The entire
144     * interface of <code>java.util.ListIterator</code> is exposed through this
145     * class.
146     *
147     * @author Nick Rizzolo
148    **/
149   public class ConstantListIterator extends NodeListIterator
150   {
151     /**
152       * Returns the next AST node in the list.  This method may be called
153       * repeatedly to iterate through the list, or intermixed with calls to
154       * <code>previous()</code> to go back and forth.  (Note that alternating
155       * calls to <code>next()</code> and <code>previous()</code> will return
156       * the same element repeatedly.)
157       *
158       * @return The next AST node in the list.
159      **/
160     public Constant nextItem() { return (Constant) I.next(); }
161 
162 
163     /**
164       * Returns the previous element in the list. This method may be called
165       * repeatedly to iterate through the list backwards, or intermixed with
166       * calls to next to go back and forth. (Note that alternating calls to
167       * next and previous will return the same element repeatedly.)
168       *
169       * @return The previous AST node in the list.
170      **/
171     public Constant previousItem() { return (Constant) I.previous(); }
172   }
