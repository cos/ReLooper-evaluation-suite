1   package LBJ2.IR;

3   import LBJ2.Pass;

6   /**
7     * Represents an assertion statement.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class AssertStatement extends Statement
12  {
13    /**
14      * (&not;&oslash;) The condition that must hold; otherwise, an assertion
15      * error is generated.
16     **/
17    public Expression condition;
18    /**
19      * (&oslash;) Represents the error message in the assertion error, if any.
20     **/
21    public Expression message;
22  
23  
24    /**
25      * Initializing constructor.
26      *
27      * @param c          The condition that must hold.
28      * @param line       The line on which the source code represented by this
29      *                   node is found.
30      * @param byteOffset The byte offset from the beginning of the source file
31      *                   at which the source code represented by this node is
32      *                   found.
33     **/
34    public AssertStatement(Expression c, int line, int byteOffset)
35    {
36      this(c, null, line, byteOffset);
37    }
38  
39    /**
40      * Full constructor.
41      *
42      * @param c          The condition that must hold.
43      * @param m          The error message.
44      * @param line       The line on which the source code represented by this
45      *                   node is found.
46      * @param byteOffset The byte offset from the beginning of the source file
47      *                   at which the source code represented by this node is
48      *                   found.
49     **/
50    public AssertStatement(Expression c, Expression m, int line, int byteOffset)
51    {
52      super(line, byteOffset);
53      condition = c;
54      message = m;
55    }
56  
57  
58    /**
59      * Returns an iterator used to successively access the children of this
60      * node.
61      *
62      * @return An iterator used to successively access the children of this
63      *         node.
64     **/
65    public ASTNodeIterator iterator()
66    {
67      ASTNodeIterator I = new ASTNodeIterator(message == null ? 1 : 2);
68      I.children[0] = condition;
69      if (message != null) I.children[1] = message;
70      return I;
71    }
72  
73  
74    /**
75      * Creates a new object with the same primitive data, and recursively
76      * creates new member data objects as well.
77      *
78      * @return The clone node.
79     **/
80    public Object clone()
81    {
82      return new AssertStatement(
83          (Expression) condition.clone(),
84          (message == null ? null : (Expression) message.clone()), -1, -1);
85    }
86  
87  
88    /**
89      * Ensures that the correct <code>run()</code> method is called for this
90      * type of node.
91      *
92      * @param pass The pass whose <code>run()</code> method should be called.
93     **/
94    public void runPass(Pass pass) { pass.run(this); }
95  
96  
97    /**
98      * Writes a string representation of this <code>ASTNode</code> to the
99      * specified buffer.  The representation written is parsable by the LBJ2
100     * compiler, but not very readable.
101     *
102     * @param buffer The buffer to write to.
103    **/
104   public void write(StringBuffer buffer)
105   {
106     buffer.append("assert ");
107     condition.write(buffer);
108     if (message != null)
109     {
110       buffer.append(" : ");
111       message.write(buffer);
112     }
113     buffer.append(";");
114   }
115 }
