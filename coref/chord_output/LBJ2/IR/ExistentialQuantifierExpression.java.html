7   /**
8     * An existential quantifier has the form:
9     * <blockquote>
10    *   <code>exists <i>argument</i> in (<i>expression</i>)
11    *   <i>constraint-expression</i>
12    * </blockquote>
13    * where <code><i>expression</i></code> must evaluate to a
14    * <code>Collection</code>, and the existential quantifier expression is
15    * sastisfied iff <code><i>constraint-expression</i></code> is satisfied for
16    * any setting of <code><i>argument</i></code> taken from the
17    * <code>Collection</code>.
18    *
19    * @author Nick Rizzolo
20   **/
21  public class ExistentialQuantifierExpression
22         extends QuantifiedConstraintExpression
23  {
24    /**
25      * Full constructor.
26      *
27      * @param line       The line on which the source code represented by this
28      *                   node is found.
29      * @param byteOffset The byte offset from the beginning of the source file
30      *                   at which the source code represented by this node is
31      *                   found.
32      * @param a          The quantification variable specification.
33      * @param c          Evaluates to the collection of objects.
34      * @param co         The quantified constraint.
35     **/
36    public ExistentialQuantifierExpression(int line, int byteOffset, Argument a,
37                                           Expression c,
38                                           ConstraintExpression co)
39    {
40      super(line, byteOffset, a, c, co);
41    }
42  
43    /**
44      * Parser's constructor.  Line and byte offset information are taken from
45      * the token.
46      *
47      * @param t  The token containing line and byte offset information.
48      * @param a  The quantification variable specification.
49      * @param c  Evaluates to the collection of objects.
50      * @param co The quantified constraint.
51     **/
52    public ExistentialQuantifierExpression(TokenValue t, Argument a,
53                                           Expression c,
54                                           ConstraintExpression co)
55    {
56      this(t.line, t.byteOffset, a, c, co);
57    }
58  
59  
60    /**
61      * Creates a new object with the same primitive data, and recursively
62      * creates new member data objects as well.
63      *
64      * @return The clone node.
65     **/
66    public Object clone()
67    {
68      return
69        new ExistentialQuantifierExpression(
70            -1, -1, (Argument) argument.clone(),
71            (Expression) collection.clone(),
72            (ConstraintExpression) constraint.clone());
73    }
74  
75  
76    /**
77      * Ensures that the correct <code>run()</code> method is called for this
78      * type of node.
79      *
80      * @param pass The pass whose <code>run()</code> method should be called.
81     **/
82    public void runPass(Pass pass) { pass.run(this); }
83  
84  
85    /**
86      * Writes a string representation of this <code>ASTNode</code> to the
87      * specified buffer.  The representation written is parsable by the LBJ2
88      * compiler, but not very readable.
89      *
90      * @param buffer The buffer to write to.
91     **/
92    public void write(StringBuffer buffer)
93    {
94      buffer.append("exists ");
95      argument.write(buffer);
96      buffer.append(" in (");
97      collection.write(buffer);
98      buffer.append(") ");
99      constraint.write(buffer);
100   }
101 }
