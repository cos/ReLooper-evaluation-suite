12  /**
13    * A symbol table is simply a <code>HashMap</code> associating names with
14    * their types.  This class also assumes responsibility for determining type
15    * information for externally defined names.
16    *
17    * <p> The global symbol table also keeps track of <code>package</code> and
18    * <code>import</code> declarations so that when a name cannot be found
19    * otherwise, it is searched for in the imported packages.
20    *
21    * @author Nick Rizzolo
22   **/
23  public class SymbolTable
24  {
25    /**
26      * Given the name of a class, which may or may not be fully qualified, this
27      * method returns the absolute path to the file with the same name and the
28      * given extension.  This method was written for use by the LBJ compiler
29      * and its generated code.  The user should not need to call it.
30      *
31      * @param name       The name of the class.
32      * @param extension  The extension of the file to search for.
33      * @param path       Paths in which to search for name.extension.
34      * @return           An object representing the file or <code>null</code>
35      *                   if it can't be located.
36     **/
37    protected static File findFile(String name, String extension, String path)
38    {
39      String fqName = name.replace('.', File.separatorChar) + "." + extension;
40      String[] paths = path.split("\\" + File.pathSeparator);
41  
42      for (int i = 0; i < paths.length; ++i)
43      {
44        File file = new File(paths[i] + File.separator + fqName);
45        if (file.exists()) return file;
46      }
47  
48      return null;
49    }

181   /**
182     * Retrieves the type associated with the given name.  If this table does
183     * not contain the name, imported packages are searched.
184     *
185     * @param name The name to retrieve type information for.
186     * @return     The type associated with the given name or <code>null</code>
187     *             if no type information could be found.
188    **/
189   public Type get(String name)
190   {
191     if (localContainsKey(name)) return (Type) table.get(name);
192     if (parent != null) return parent.get(name);
193     if (external.containsKey(name)) return (Type) external.get(name);
194 
195     Type result = null;
196     Class c = classForName(name);
197 
198     if (c != null)
199     {
200       if (ParameterizedConstraint.class.isAssignableFrom(c))
201       {
202         ParameterizedConstraint constraint = null;
203         try { constraint = (ParameterizedConstraint) c.newInstance(); }
204         catch (Exception e)
205         {
206           System.err.println("Can't instantiate parameterized constraint '"
207               + c + "'.  Make sure there is a public, no argument "
208               + "constructor defined:");
209           e.printStackTrace();
210           System.exit(1);
211         }
212 
213         result =
214           new ConstraintType(Type.parseType(constraint.getInputType()));
215       }
216       else if (Classifier.class.isAssignableFrom(c))
217       {
218         Classifier classifier = null;
219         try { classifier = (Classifier) c.newInstance(); }
220         catch (Exception e)
221         {
222           System.err.println("Can't instantiate classifier '" + c
223               + "'.  Make sure there is a public, no argument constructor "
224               + "defined:");
225           e.printStackTrace();
226           System.exit(1);
227         }
228 
229         result =
230           new ClassifierType(
231               Type.parseType(classifier.getInputType()),
232               new ClassifierReturnType(
233                 classifier.getOutputType(),
234                 new ConstantList(classifier.allowableValues())),
235               Learner.class.isAssignableFrom(c));
236       }
237       else if (Inference.class.isAssignableFrom(c))
238       {
239         Inference inference = null;
240         try { inference = (Inference) c.newInstance(); }
241         catch (Exception e)
242         {
243           System.err.println("Can't instantiate inference '" + c
244               + "'.  Make sure there is a public, no argument constructor "
245               + "defined:");
246           e.printStackTrace();
247           System.exit(1);
248         }
249 
250         Type headType = new ReferenceType(new Name(inference.getHeadType()));
251         String[] headFinderTypeStrings = inference.getHeadFinderTypes();
252         Type[] headFinderTypes = new Type[headFinderTypeStrings.length];
253         for (int i = 0; i < headFinderTypeStrings.length; ++i)
254           headFinderTypes[i] =
255             new ReferenceType(new Name(headFinderTypeStrings[i]));
256 
257         result = new InferenceType(headType, headFinderTypes);
258       }
259       else if (Normalizer.class.isAssignableFrom(c))
260       {
261         try { c.newInstance(); }
262         catch (Exception e)
263         {
264           System.err.println("Can't instantiate normalizer '" + c
265               + "'.  Make sure there is a public, no argument constructor "
266               + "defined:");
267           e.printStackTrace();
268           System.exit(1);
269         }
270 
271         result = new NormalizerType();
272       }
273     }
274 
275     external.put(name, result);
276 
277     return result;
278   }

313   /**
314     * Attempts to locate the named class in the current package and any
315     * imported packages.  If the corresponding Java source file is found and
316     * either the class file does not exist or its time of last modification is
317     * earlier than the java file's, it is recompiled.  If no class with the
318     * specified name is found, <code>null</code> is returned.
319     *
320     * @param name The name of the class to search for.
321     * @return     The <code>Class</code> object representing that class.
322    **/
323   public Class classForName(Name name)
324   {
325     if (parent != null) return parent.classForName(name);
326 
327     Class result = null;
328 
329     LinkedList prefixes = new LinkedList();
330     prefixes.add("");
331     if (!sourcePackage.equals("")) prefixes.add(sourcePackage + ".");
332     prefixes.add("java.lang.");
333 
334     for (Iterator I = imported.iterator(); I.hasNext(); )
335     {
336       String s = (String) I.next();
337       if (s.endsWith(".*")) prefixes.add(s.substring(0, s.length() - 1));
338       else if (s.endsWith("." + name.name[0]))
339         prefixes.add(s.substring(0, s.length() - name.name[0].length()));
340     }
341 
342     for (Iterator I = prefixes.iterator(); I.hasNext() && result == null; )
343     {
344       String prefix = (String) I.next();
345       String fqName = prefix + name;
346       File javaFile = findFile(fqName, "java", LBJ2.Main.sourcePath);
347 
348       if (javaFile != null)
349       {
350         File classFile = findFile(fqName, "class", LBJ2.Main.classPath);
351 
352         if ((classFile == null
353              || javaFile.lastModified() > classFile.lastModified())
354             && LBJ2.Train.runJavac(javaFile.toString()))
355           System.exit(1);
356       }
357 
358       try { result = Class.forName(fqName); }
359       catch (Exception e) { }
360       catch (NoClassDefFoundError e) { }
361 
362       for (int i = 0; i < name.name.length - 1 && result == null; ++i)
363       {
364         fqName = prefix + name.name[0];
365         for (int j = 1; j <= i; ++j) fqName += "." + name.name[j];
366         javaFile = findFile(fqName, "java", LBJ2.Main.sourcePath);
367 
368         if (javaFile != null)
369         {
370           File classFile = findFile(fqName, "class", LBJ2.Main.classPath);
371 
372           if ((classFile == null
373                || javaFile.lastModified() > classFile.lastModified())
374               && LBJ2.Train.runJavac(javaFile.toString()))
375             System.exit(1);
376         }
377 
378         for (int j = i + 1; j < name.name.length; ++j)
379           fqName += "$" + name.name[j];
380 
381         try { result = Class.forName(fqName); }
382         catch (Exception e) { }
383         catch (NoClassDefFoundError e) { }
384       }
385     }
386 
387     return result;
388   }

564   /**
565     * Prints this table and all its children recursively to
566     * <code>STDOUT</code>.
567     *
568     * @param indent The level of indentation.
569    **/
570   public void print(String indent)
571   {
572     if (parent == null)
573     {
574       if (sourcePackage.equals(""))
575         System.out.println("Package: " + sourcePackage);
576       System.out.println("Imported:");
577       for (Iterator I = imported.iterator(); I.hasNext(); )
578         System.out.println("  " + I.next());
579     }
580 
581     System.out.println(indent + "Symbols:");
582     String[] symbols = (String[]) table.keySet().toArray(new String[0]);
583     Arrays.sort(symbols);
584     for (int i = 0; i < symbols.length; ++i)
585       System.out.println(indent + "  " + symbols[i] + " -> "
586                          + table.get(symbols[i]));
587 
588     if (children.size() > 0)
589     {
590       for (Iterator I = children.iterator(); I.hasNext(); )
591       {
592         System.out.println();
593         ((SymbolTable) I.next()).print(indent + "  ");
594       }
595     }
596   }
