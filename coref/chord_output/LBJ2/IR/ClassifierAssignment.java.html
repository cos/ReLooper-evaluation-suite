7   /**
8     * Represents the assignment of a classifier expression to a method
9     * signature.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class ClassifierAssignment extends Declaration
14  {
15    /**
16      * This value is used in place of the field access which appears as an
17      * argument to <code>cachedin</code> to indicate that in fact,
18      * <code>cached</code> was used instead.
19     **/
20    public static final String mapCache = "!!!";
21  
22  
23    /** (&not;&oslash;) The return type of the classifier. */
24    public ClassifierReturnType returnType;
25    /** (&not;&oslash;) The input specification of the classifier. */
26    public Argument argument;
27    /** (&not;&oslash;) The expression representing the classifier. */
28    public ClassifierExpression expression;
29    /**
30      * (&oslash;) The expression representing the field to cache this
31      * classifier's result in.
32     **/
33    public Name cacheIn;
34  
35  
36    /**
37      * Full constructor.  Line and byte offset information is taken from the
38      * type.
39      *
40      * @param co A Javadoc comment associated with the declaration.
41      * @param t  The return type of the classifier.
42      * @param n  The classifier's name.
43      * @param a  The input specification of the classifier.
44      * @param e  The expression representing the classifier.
45      * @param ca The expression representing the field to cache this
46      *           classifier's result in.
47     **/
48    public ClassifierAssignment(String co, ClassifierReturnType t, Name n,
49                                Argument a, ClassifierExpression e, Name ca)
50    {
51      super(co, n, t.line, t.byteOffset);
52      returnType = t;
53      argument = a;
54      expression = e;
55      cacheIn = ca;
56    }
57  
58    /**
59      * Parser's constructor.  Line and byte offset information is taken from
60      * the type.
61      *
62      * @param t  The return type of the classifier.
63      * @param i  The identifier token representing the classifier's name.
64      * @param a  The input specification of the classifier.
65      * @param e  The expression representing the classifier.
66     **/
67    public ClassifierAssignment(ClassifierReturnType t, TokenValue i,
68                                Argument a, ClassifierExpression e)
69    {
70      this(null, t, new Name(i), a, e, null);
71    }
72  
73    /**
74      * Parser's constructor.  Line and byte offset information is taken from
75      * the type.
76      *
77      * @param t  The return type of the classifier.
78      * @param i  The identifier token representing the classifier's name.
79      * @param a  The input specification of the classifier.
80      * @param e  The expression representing the classifier.
81      * @param c  The expression representing the field to cache this
82      *           classifier's result in.
83     **/
84    public ClassifierAssignment(ClassifierReturnType t, TokenValue i,
85                                Argument a, ClassifierExpression e, Name c)
86    {
87      this(null, t, new Name(i), a, e, c);
88    }
89  
90  
91    /**
92      * Returns the type of the declaration.
93      *
94      * @return The type of the declaration.
95     **/
96    public Type getType()
97    {
98      return
99        new ClassifierType(argument.getType(), returnType,
100                          expression instanceof LearningClassifierExpression);
101   }

103 
104   /**
105     * Returns an iterator used to successively access the children of this
106     * node.
107     *
108     * @return An iterator used to successively access the children of this
109     *         node.
110    **/
111   public ASTNodeIterator iterator()
112   {
113     ASTNodeIterator I = new ASTNodeIterator(cacheIn == null ? 4 : 5);
114     I.children[0] = returnType;
115     I.children[1] = name;
116     I.children[2] = argument;
117     I.children[3] = expression;
118     if (cacheIn != null) I.children[4] = cacheIn;
119     return I;
120   }
121 
122 
123   /**
124     * Creates a new object with the same primitive data, and recursively
125     * creates new member data objects as well.
126     *
127     * @return The clone node.
128    **/
129   public Object clone()
130   {
131     return
132       new ClassifierAssignment(comment,
133                                (ClassifierReturnType) returnType.clone(),
134                                (Name) name.clone(),
135                                (Argument) argument.clone(),
136                                (ClassifierExpression) expression.clone(),
137                                (Name)
138                                  (cacheIn == null ? null : cacheIn.clone()));
139   }

142   /**
143     * Ensures that the correct <code>run()</code> method is called for this
144     * type of node.
145     *
146     * @param pass The pass whose <code>run()</code> method should be called.
147    **/
148   public void runPass(Pass pass) { pass.run(this); }
149

151   /**
152     * Writes a string representation of this <code>ASTNode</code> to the
153     * specified buffer.  The representation written is parsable by the LBJ2
154     * compiler, but not very readable.
155     *
156     * @param buffer The buffer to write to.
157    **/
158   public void write(StringBuffer buffer)
159   {
160     returnType.write(buffer);
161     buffer.append(" ");
162     name.write(buffer);
163     buffer.append("(");
164     argument.write(buffer);
165     buffer.append(") ");
166 
167     if (cacheIn != null)
168     {
169       buffer.append("cached");
170 
171       if (!cacheIn.toString().equals(ClassifierAssignment.mapCache))
172       {
173         buffer.append("in ");
174         cacheIn.write(buffer);
175       }
176 
177       buffer.append(" ");
178     }
179 
180     buffer.append("<- ");
181     expression.write(buffer);
182   }
