6   /**
7     * Abstract node class that every AST node extends. <br><br>
8     *
9     * Every AST node must
10    * <ul>
11    *   <li>
12    *     have a unique ID number and store the line and byte offset where it
13    *     was found in the source file.
14    *   <li>
15    *     include a special symbol at the beginning of the Javadoc comment of
16    *     each of its AST node child member variables indicating whether that
17    *     member variable is allowed to be set to <code>null</code>.  The symbol
18    *     (&oslash;) is used to indicate that the variable is allowed to be
19    *     <code>null</code>, and the symbol (&not;&oslash;) indicates that the
20    *     variable will never be <code>null</code>.
21    *   <li>
22    *     return an <code>ASTNodeIterator</code> through the abstract
23    *     <code>iterator()</code> method that traverses its children in the
24    *     order in which the code would be executed.  If the node is composed of
25    *     declarative constructs rather than executable code, the iterator will
26    *     traverse those constructs in the order in which they appear in the
27    *     source.
28    *   <li>
29    *     override the <code>clone()</code> method such that it returns a deep
30    *     copy of itself, except that the line and byte offset of the copy are
31    *     both set to <code>-1</code>.
32    *   <li>
33    *     override the <code>runPass(Pass)</code> method so that it simply calls
34    *     the <code>Pass</code>'s <code>run(.)</code> method whose argument type
35    *     is the type of the node.
36    *   <li>
37    *     return a <code>String</code> representation of itself through the
38    *     <code>toString()</code> and <code>write(StringBuffer)</code> methods.
39    *     Only the <code>write(StringBuffer)</code> method need be overridden by
40    *     each node, as <code>toString()</code> will simply invoke
41    *     <code>write(StringBuffer)</code> to produce its result.  (This is much
42    *     more efficient than having <code>toString()</code> call its childrens'
43    *     <code>toString()</code> methods recursively through the AST and
44    *     concatenating them all together.)  The <code>String</code> produced
45    *     will not be very readable (e.g., it won't contain any new lines), but
46    *     it will be compilable by the LBJ compiler.
47    * </ul>
48    *
49    * Most <code>ASTNode</code>s will also contain more than one constructor.
50    * There will be one constructor that includes all references to its children
51    * as well as line and byte offset information, etc.  This constructor is
52    * commonly used by the node's <code>clone()</code> method.  In addition,
53    * there may be at least one constructor designed to be more useful for the
54    * JavaCUP parser, taking <code>TokenValue</code>s as input.
55    *
56    * @author Nick Rizzolo
57   **/
58  public abstract class ASTNode
59  {
60    /** Keeps track of how many nodes have been created. */
61    private static int nextID = 0;
62    /** Stores the ID of this node as provided by <code>nextID</code>. */
63    public int nodeID;
64  
65    /** The line on which the source code represented by this node is found. */
66    public int line;
67    /**
68      * The byte offset from the beginning of the source file at which the
69      * source code represented by this node is found.
70     **/
71    public int byteOffset;
72    /** The table of variable types representing this node's scope. */
73    public SymbolTable symbolTable;
74  
75  
76    /** Default constructor. */
77    public ASTNode() { this(-1, -1); }
78  
79  
80    /**
81      * Initializing constructor.  This constructor is called via the
82      * <code>super</code> operator from every other node.
83      *
84      * @param line       The line on which the source code represented by this
85      *                   node is found.
86      * @param byteOffset The byte offset from the beginning of the source file
87      *                   at which the source code represented by this node is
88      *                   found.
89     **/
90    public ASTNode(int line, int byteOffset)
91    {
92      nodeID = nextID++;
93      this.line = line;
94      this.byteOffset = byteOffset;
95    }
96

98    /**
99      * Returns an iterator used to successively access the children of this
100     * node.
101     *
102     * @return An iterator used to successively access the children of this
103     *         node.
104    **/
105   abstract public ASTNodeIterator iterator();

108   /**
109     * Creates a new object with the same primitive data, and recursively
110     * creates new member data objects as well.
111     *
112     * @return The clone node.
113    **/
114   public Object clone()
115   {
116     System.err.println("WARNING: clone() not defined for class '"
117                        + this.getClass().getName() + "'");
118     return null;
119   }

122   /**
123     * Ensures that the correct <code>run()</code> method is called for this
124     * type of node.
125     *
126     * @param pass The pass whose <code>run()</code> method should be called.
127    **/
128   abstract public void runPass(Pass pass);
129 
130 
131   /**
132     * Calls the <code>write(StringBuffer)</code> method to produce a string
133     * representation of this node.
134     *
135     * @return A textual representation of this node.
136    **/
137   public String toString()
138   {
139     StringBuffer buffer = new StringBuffer();
140     write(buffer);
141     return "" + buffer;
142   }

145   /**
146     * Writes a string representation of this <code>ASTNode</code> to the
147     * specified buffer.  The representation written is parsable by the LBJ2
148     * compiler, but not very readable.
149     *
150     * @param buffer The buffer to write to.
151    **/
152   abstract public void write(StringBuffer buffer);
153 }
