3   import LBJ2.Pass;

6   /**
7     * An expression statement is a statement composed only of a single
8     * expression, as opposed to a statement involving control flow.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class ExpressionStatement extends Statement
13  {
14    /** (&not;&oslash;) The expression being used as a statement. */
15    public StatementExpression expression;
16  
17  
18    /**
19      * Initializing constructor.  Line and byte offset information are taken
20      * from the expression's representation.
21      *
22      * @param e  The expression being used as a statement.
23     **/
24    public ExpressionStatement(StatementExpression e)
25    {
26      super(e.line, e.byteOffset);
27      expression = e;
28    }
29  
30  
31    /**
32      * Returns an iterator used to successively access the children of this
33      * node.
34      *
35      * @return An iterator used to successively access the children of this
36      *         node.
37     **/
38    public ASTNodeIterator iterator()
39    {
40      ASTNodeIterator I = new ASTNodeIterator(1);
41      I.children[0] = expression;
42      return I;
43    }
44  
45  
46    /**
47      * Creates a new object with the same primitive data, and recursively
48      * creates new member data objects as well.
49      *
50      * @return The clone node.
51     **/
52    public Object clone()
53    {
54      return new ExpressionStatement((StatementExpression) expression.clone());
55    }
56  
57  
58    /**
59      * Ensures that the correct <code>run()</code> method is called for this
60      * type of node.
61      *
62      * @param pass The pass whose <code>run()</code> method should be called.
63     **/
64    public void runPass(Pass pass) { pass.run(this); }
65  
66  
67    /**
68      * Writes a string representation of this <code>ASTNode</code> to the
69      * specified buffer.  The representation written is parsable by the LBJ2
70      * compiler, but not very readable.
71      *
72      * @param buffer The buffer to write to.
73     **/
74    public void write(StringBuffer buffer)
75    {
76      expression.write(buffer);
77      buffer.append(";");
78    }
79  }
