6   /**
7     * This class represents an expression creating an array.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class ArrayCreationExpression extends Expression
12  {
13    /**
14      * (&not;&oslash;) The most basic type of elements in the array (i.e., it
15      * will not be an <code>ArrayType</code>).
16     **/
17    public Type elementType;
18    /**
19      * (&not;&oslash;) Describes the size of each dimension in the new array.
20     **/
21    public ExpressionList sizes;
22    /**
23      * The total number of dimensions, including those for
24      * which no size is given.
25     **/
26    public int dimensions;
27    /** (&oslash;) Initial values for the new array. */
28    public ArrayInitializer initializer;
29  
30  
31    /**
32      * Initializing constructor.
33      *
34      * @param t          The element type.
35      * @param l          The list of dimension size expressions.
36      * @param d          The total number of dimensions in the array.
37      * @param line       The line on which the source code represented by this
38      *                   node is found.
39      * @param byteOffset The byte offset from the beginning of the source file
40      *                   at which the source code represented by this node is
41      *                   found.
42     **/
43    public ArrayCreationExpression(Type t, ExpressionList l, int d, int line,
44                                   int byteOffset)
45    {
46      super(line, byteOffset);
47      elementType = t;
48      sizes = l;
49      dimensions = d;
50      initializer = null;
51    }
52  
53    /**
54      * Initializing constructor.
55      *
56      * @param t          The element type.
57      * @param d          The total number of dimensions in the array.
58      * @param a          An initializing expression for the array.
59      * @param line       The line on which the source code represented by this
60      *                   node is found.
61      * @param byteOffset The byte offset from the beginning of the source file
62      *                   at which the source code represented by this node is
63      *                   found.
64     **/
65    public ArrayCreationExpression(Type t, int d, ArrayInitializer a, int line,
66                                   int byteOffset)
67    {
68      this(t, new ExpressionList(), d, line, byteOffset);
69      initializer = a;
70    }
71  
72  
73    /**
74      * Returns a hash code value for java hash structures.
75      *
76      * @return A hash code for this object.
77     **/
78    public int hashCode()
79    {
80      return elementType.hashCode() + sizes.hashCode()
81             + (initializer == null ? 0 : initializer.hashCode());
82    }
83  
84  
85    /**
86      * Indicates whether some other object is "equal to" this one.
87      *
88      * @return <code>true</code> iff this object is the same as the argument.
89     **/
90    public boolean equals(Object o)
91    {
92      if (!(o instanceof ArrayCreationExpression)) return false;
93      ArrayCreationExpression a = (ArrayCreationExpression) o;
94      return elementType.equals(a.elementType) && sizes.equals(a.sizes)
95             && (initializer == null && a.initializer == null
96                 || initializer != null && initializer.equals(a.initializer));
97    }

99  
100   /**
101     * Returns an iterator used to successively access the children of this
102     * node.
103     *
104     * @return An iterator used to successively access the children of this
105     *         node.
106    **/
107   public ASTNodeIterator iterator()
108   {
109     ASTNodeIterator I = new ASTNodeIterator(initializer == null ? 2 : 3);
110     I.children[0] = elementType;
111     I.children[1] = sizes;
112     if (initializer != null) I.children[2] = initializer;
113     return I;
114   }

116 
117   /**
118     * Creates a new object with the same primitive data, and recursively
119     * creates new member data objects as well.
120     *
121     * @return The clone node.
122    **/
123   public Object clone()
124   {
125     return initializer == null
126       ? new ArrayCreationExpression((Type) elementType.clone(),
127                                     (ExpressionList) sizes.clone(),
128                                     dimensions, -1, -1)
129       : new ArrayCreationExpression((Type) elementType.clone(),
130                                     dimensions,
131                                     (ArrayInitializer) initializer.clone(),
132                                     -1, -1);
133   }

136   /**
137     * Ensures that the correct <code>run()</code> method is called for this
138     * type of node.
139     *
140     * @param pass The pass whose <code>run()</code> method should be called.
141    **/
142   public void runPass(Pass pass) { pass.run(this); }

145   /**
146     * Writes a string representation of this <code>ASTNode</code> to the
147     * specified buffer.  The representation written is parsable by the LBJ2
148     * compiler, but not very readable.
149     *
150     * @param buffer The buffer to write to.
151    **/
152   public void write(StringBuffer buffer)
153   {
154     if (parenthesized) buffer.append("(");
155     buffer.append("new ");
156     elementType.write(buffer);
157 
158     int i = 0;
159     for (ASTNodeIterator I = sizes.iterator(); I.hasNext(); ++i)
160     {
161       buffer.append("[");
162       I.next().write(buffer);
163       buffer.append("]");
164     }
165 
166     for (; i < dimensions; ++i) buffer.append("[]");
167     if (initializer != null) initializer.write(buffer);
168     if (parenthesized) buffer.append(")");
169   }
