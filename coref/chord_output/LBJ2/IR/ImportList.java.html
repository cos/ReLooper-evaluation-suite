7   /**
8     * Currently, this is just a wrapper class for <code>LinkedList</code>.  The
9     * code that uses it looks a little cleaner when casts are all taken care of
10    * automatically.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class ImportList extends List
15  {
16    /** Default constructor. */
17    public ImportList() { super(-1, -1, " "); }
18  
19    /**
20      * Initializing constructor.  Requires its argument to be
21      * non-<code>null</code>.
22      *
23      * @param d  A single <code>ImportDeclaration</code> with which to
24      *           initialize this list.
25     **/
26    public ImportList(ImportDeclaration d)
27    {
28      super(d.line, d.byteOffset, " ");
29      list.add(d);
30    }
31  
32  
33    /**
34      * Adds another <code>ImportDeclaration</code> to the end of the list.
35      *
36      * @param d  A reference to the <code>ImportDeclaration</code> to be added.
37     **/
38    public void add(ImportDeclaration d) { list.add(d); }
39  
40  
41    /**
42      * Adds all the <code>ImportDeclaration</code>s in another
43      * <code>ImportList</code> to the end of this
44      * <code>ImportList</code>.
45      *
46      * @param d  The list to be added.
47     **/
48    public void addAll(ImportList d) { list.addAll(d.list); }
49  
50  
51    /**
52      * Transforms the list into an array of statements.
53      *
54      * @return An array of statements containing references to every statement
55      *         in the list.
56     **/
57    public ImportDeclaration[] toArray()
58    {
59      return
60        (ImportDeclaration[]) list.toArray(new ImportDeclaration[list.size()]);
61    }
62  
63  
64    /**
65      * Returns an iterator used specifically to access the elements of this
66      * list.
67      *
68      * @return An iterator used specifically to access the elements of this
69      *         list.
70     **/
71    public ImportListIterator listIterator()
72    {
73      return new ImportListIterator();
74    }
75  
76  
77    /**
78      * Returns an iterator used to successively access the children of this
79      * node.
80      *
81      * @return An iterator used to successively access the children of this
82      *         node.
83     **/
84    public ASTNodeIterator iterator() { return listIterator(); }
85  
86  
87    /**
88      * Creates a new object with the same primitive data, and recursively
89      * creates new member data objects as well.
90      *
91      * @return The clone node.
92     **/
93    public Object clone()
94    {
95      ImportList clone = new ImportList();
96      for (Iterator i = list.iterator(); i.hasNext(); )
97        clone.list.add(((ImportDeclaration) i.next()).clone());
98      return clone;
99    }

102   /**
103     * Ensures that the correct <code>run()</code> method is called for this
104     * type of node.
105     *
106     * @param pass The pass whose <code>run()</code> method should be called.
107    **/
108   public void runPass(Pass pass) { pass.run(this); }
109 
110 
111   /**
112     * Used to iterate though the children of a list of AST nodes.  The entire
113     * interface of <code>java.util.ListIterator</code> is exposed through this
114     * class.
115     *
116     * @author Nick Rizzolo
117    **/
118   public class ImportListIterator extends NodeListIterator
119   {
120     /**
121       * Returns the next AST node in the list.  This method may be called
122       * repeatedly to iterate through the list, or intermixed with calls to
123       * <code>previous()</code> to go back and forth.  (Note that alternating
124       * calls to <code>next()</code> and <code>previous()</code> will return
125       * the same element repeatedly.)
126       *
127       * @return The next AST node in the list.
128      **/
129     public ImportDeclaration nextItem()
130     {
131       return (ImportDeclaration) I.next();
132     }
133 
134 
135     /**
136       * Returns the previous element in the list. This method may be called
137       * repeatedly to iterate through the list backwards, or intermixed with
138       * calls to next to go back and forth. (Note that alternating calls to
139       * next and previous will return the same element repeatedly.)
140       *
141       * @return The previous AST node in the list.
142      **/
143     public ImportDeclaration previousItem()
144     {
145       return (ImportDeclaration) I.previous();
146     }
147   }
148 }
