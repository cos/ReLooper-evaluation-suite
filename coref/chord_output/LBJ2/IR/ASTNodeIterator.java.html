4   /**
5     * Used to iterate though the children of an AST node.  It is assumed that
6     * the <code>children</code> array never contains a <code>null</code>
7     * reference.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class ASTNodeIterator
12  {
13    /** The nodes iterated through by this iterator. */
14    public ASTNode[] children;
15    /** Index into the <code>children</code> array. */
16    protected int index;
17  
18  
19    /** Initializes <code>index</code>, but not <code>children</code>. */
20    public ASTNodeIterator() { index = 0; }
21  
22    /**
23      * The <code>children</code> array will have the specified length.
24      *
25      * @param l  The number of children to iterate through.
26     **/
27    public ASTNodeIterator(int l)
28    {
29      this();
30      children = new ASTNode[l];
31    }
32  
33  
34    /**
35      * Determines whether there are any child nodes left to be accessed.
36      *
37      * @return <code>true</code> iff there are child nodes remaining.
38     **/
39    public boolean hasNext()
40    {
41      return children != null && index < children.length;
42    }
43  
44  
45    /**
46      * Returns the next child AST node.
47      *
48      * @return The next child AST node.
49     **/
50    public ASTNode next()
51    {
52      return children == null || index == children.length
53             ? null : children[index++];
54    }
55  
56  
57    /** Restarts the iterator. */
58    public void reset() { index = 0; }
59  }
