8   /**
9     * This class represents a scalar variable.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class Name extends Expression
14  {
15    /**
16      * (&not;&oslash;) These strings appeared with dots between them to form
17      * the name in the source.
18     **/
19    public String[] name;
20    /**
21      * The number of matched brackets appearing after a single identifier;
22      * supports variable declarations.
23     **/
24    public int dimensions;

27    /**
28      * Takes a fully specified name (eg java.lang.String) as input.
29      *
30      * @param n  A fully specified name.
31     **/
32    public Name(String n) { this(n, -1, -1); }
33  
34    /**
35      * Takes a fully specified name (eg java.lang.String) as input.
36      *
37      * @param n  A fully specified name.
38     **/
39    public Name(String n, int line, int byteOffset)
40    {
41      super(line, byteOffset);
42      name = n.split("\\.");
43      dimensions = 0;
44    }
45  
46    /**
47      * Should only be called by the <code>clone()</code> method.
48      *
49      * @param n  The value of the <code>name</code> variable.
50     **/
51    protected Name(String[] n) { this(n, -1, -1); }
52  
53    /**
54      * Should only be called by the <code>clone()</code> method.
55      *
56      * @param n  The value of the <code>name</code> variable.
57     **/
58    protected Name(String[] n, int line, int byteOffset)
59    {
60      super(line, byteOffset);
61      name = (String[]) n.clone();
62      dimensions = 0;
63    }
64  
65    /**
66      * Parser's constructor.
67      *
68      * @param token  The parser's token for the identifier.
69     **/
70    public Name(TokenValue token)
71    {
72      super(token.line, token.byteOffset);
73      name = new String[1];
74      name[0] = "" + token;
75      dimensions = 0;
76    }
77  
78    /**
79      * Parser's constructor.
80      *
81      * @param n      A name that needs another identifier added to it.
82      * @param token  The parser's token for the identifier.
83     **/
84    public Name(Name n, TokenValue token)
85    {
86      super(n.line, n.byteOffset);
87      name = new String[n.name.length + 1];
88      for (int i = 0; i < n.name.length; ++i) name[i] = n.name[i];
89      name[n.name.length] = "" + token;
90    }

96      * @return The length of the <code>name</code> array.
97     **/
98    public int length() { return name.length; }

101   /**
102     * Returns a new <code>Name</code> object that is the same as this
103     * <code>Name</code> object, except the last identifier has been removed.
104    **/
105   public Name cutLast()
106   {
107     String[] n = new String[name.length - 1];
108     for (int i = 0; i < n.length; ++i) n[i] = name[i];
109     return new Name(n);
110   }

113   /**
114     * Returns a set of <code>Argument</code>s storing the name and type of
115     * each variable that is a subexpression of this expression.  This method
116     * cannot be run before <code>SemanticAnalysis</code> runs.
117    **/
118   public HashSet getVariableTypes()
119   {
120     HashSet result = new HashSet();
121 
122     Type type = null;
123     if (name.length == 1) type = typeCache;
124     else type = symbolTable.get(name[0]);
125 
126     if (type != null
127         && (type instanceof ArrayType || type instanceof ReferenceType
128             || type instanceof PrimitiveType))
129       result.add(new Argument(type, name[0]));
130 
131     return result;
132   }

135   /**
136     * Returns a set of <code>Argument</code>s storing the name and type of
137     * each variable that is a subexpression of this expression.  This method
138     * cannot be run before <code>SemanticAnalysis</code> runs.
139     *
140     * @param b  Flag set if this name is the name of an invoked method.
141    **/
142   public HashSet getVariableTypes(boolean b)
143   {
144     if (!b || name.length > 1) return getVariableTypes();
145     return new HashSet();
146   }

149   /**
150     * Determines if there are any quantified variables in this expression.
151     * This method cannot be run before <code>SemanticAnalysis</code> runs.
152    **/
153   public boolean containsQuantifiedVariable()
154   {
155     HashSet types = getVariableTypes();
156     if (types.size() == 0) return false;
157     return
158       ((Argument) types.iterator().next()).getType().quantifierArgumentType;
159   }

162   /**
163     * Determines if there are any quantified variables in this expression.
164     * This method cannot be run before <code>SemanticAnalysis</code> runs.
165     *
166     * @param b  Flag set if this name is the name of an invoked method.
167    **/
168   public boolean containsQuantifiedVariable(boolean b)
169   {
170     if (!b || name.length > 1) return containsQuantifiedVariable();
171     return false;
172   }

175   /**
176     * Returns a hash code value for java hash structures.
177     *
178     * @return A hash code for this object.
179    **/
180   public int hashCode()
181   {
182     int code = 0;
183     for (int i = 0; i < name.length; ++i) code += name[i].hashCode();
184     return code;
185   }

188   /**
189     * Indicates whether some other object is "equal to" this one.
190     *
191     * @return <code>true</code> iff this object is the same as the argument.
192    **/
193   public boolean equals(Object o)
194   {
195     if (!(o instanceof Name)) return false;
196     Name n = (Name) o;
197     if (n.name.length != name.length) return false;
198     boolean equal = true;
199     for (int i = 0; i < name.length && equal; ++i)
200       if (!name[i].equals(n.name[i])) equal = false;
201     return equal;
202   }

233   /**
234     * Writes a string representation of this <code>ASTNode</code> to the
235     * specified buffer.  The representation written is parsable by the LBJ2
236     * compiler, but not very readable.
237     *
238     * @param buffer The buffer to write to.
239    **/
240   public void write(StringBuffer buffer)
241   {
242     if (parenthesized) buffer.append("(");
243 
244     if (name.length > 0)
245     {
246       buffer.append(name[0]);
247       for (int i = 1; i < name.length; ++i)
248       {
249         buffer.append(".");
250         buffer.append(name[i]);
251       }
252     }
253 
254     if (parenthesized) buffer.append(")");
255   }
