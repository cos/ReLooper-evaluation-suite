4   /**
5     * Abstract classifier expression class.  The member variables of this class
6     * are filled in either during parsing or during semantic analysis, but in
7     * any case, they do not represent AST children that exist in the source.
8     *
9     * @author Nick Rizzolo
10   **/
11  public abstract class ClassifierExpression extends ASTNode
12                                             implements LBJ2.CodeGenerator
13  {
14    /**
15      * (&oslash;) The text of a Javadoc comment that may be associated with
16      * this classifier.
17     **/
18    public String comment;
19    /** Expression describing what is being declared. */
20    public Name name;
21    /** The return type of the declared classifier. */
22    public ClassifierReturnType returnType;
23    /** Specification of the classifier's input. */
24    public Argument argument;
25    /** Indicates whether this expression was parenthesized in the source. */
26    public boolean parenthesized = false;
27    /**
28      * The expression representing the field to cache this classifier's result
29      * in.
30     **/
31    public Name cacheIn;
32  
33  
34    /**
35      * Default constructor.
36      *
37      * @param line       The line on which the source code represented by this
38      *                   node is found.
39      * @param byteOffset The byte offset from the beginning of the source file
40      *                   at which the source code represented by this node is
41      *                   found.
42     **/
43    ClassifierExpression(int line, int byteOffset) { super(line, byteOffset); }
44  
45  
46    /** Returns the name of the <code>ClassifierExpression</code>. */
47    public String getName() { return "" + name; }
48  
49  
50    /**
51      * Returns the line number on which this AST node is found in the source
52      * (starting from line 0).  This method exists to fulfull the
53      * implementation of <code>CodeGenerator</code>.
54      * @see LBJ2.CodeGenerator
55     **/
56    public int getLine() { return line; }
57  
58  
59    /**
60      * Sets the <code>cacheIn</code> member variable to the argument.
61      *
62      * @param c  The new expression for the <code>cacheIn</code> member
63      *           variable.
64     **/
65    public void setCacheIn(Name c) { cacheIn = c; }
66  
67  
68    /**
69      * Creates a <code>StringBuffer</code> containing a shallow representation
70      * of this <code>ClassifierExpression</code>.
71      *
72      * @return A <code>StringBuffer</code> containing a shallow text
73      *         representation of the given node.
74     **/
75    abstract public StringBuffer shallow();
76  }
