6   /**
7     * An inference can be invoked as a method with the name of a learning
8     * classifier involved in that inference as its lone argument.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class InferenceInvocation extends ClassifierExpression
13  {
14    /** (&not;&oslash;) The name of the inference to invoke. */
15    public Name inference;
16    /** (&not;&oslash;) The name of the argument learning classifier. */
17    public Name classifier;
18  
19  
20    /**
21      * Initializing constructor.  Line and byte offset information are taken
22      * from the name of the inference.
23      *
24      * @param i  The name of the inference.
25      * @param c  The name of the classifier.
26     **/
27    public InferenceInvocation(Name i, Name c)
28    {
29      super(i.line, i.byteOffset);
30      inference = i;
31      classifier = c;
32    }
33  
34  
35    /**
36      * Returns a hash code value for java hash structures.
37      *
38      * @return A hash code for this object.
39     **/
40    public int hashCode()
41    {
42      return inference.hashCode() + classifier.hashCode();
43    }
44  
45  
46    /**
47      * Indicates whether some other object is "equal to" this one.
48      *
49      * @return <code>true</code> iff this object is the same as the argument.
50     **/
51    public boolean equals(Object o)
52    {
53      if (!(o instanceof InferenceInvocation)) return false;
54      InferenceInvocation i = (InferenceInvocation) o;
55      return inference.equals(i.inference) && classifier.equals(i.classifier);
56    }
57  
58  
59    /**
60      * Returns an iterator used to successively access the children of this
61      * node.
62      *
63      * @return An iterator used to successively access the children of this
64      *         node.
65     **/
66    public ASTNodeIterator iterator()
67    {
68      ASTNodeIterator I = new ASTNodeIterator(2);
69      I.children[0] = inference;
70      I.children[1] = classifier;
71      return I;
72    }
73  
74  
75    /**
76      * Creates a new object with the same primitive data, and recursively
77      * creates new member data objects as well.
78      *
79      * @return The clone node.
80     **/
81    public Object clone()
82    {
83      return
84        new InferenceInvocation((Name) inference.clone(),
85                                (Name) classifier.clone());
86    }

89    /**
90      * Ensures that the correct <code>run()</code> method is called for this
91      * type of node.
92      *
93      * @param pass The pass whose <code>run()</code> method should be called.
94     **/
95    public void runPass(Pass pass) { pass.run(this); }

98    /**
99      * Writes a string representation of this <code>ASTNode</code> to the
100     * specified buffer.  The representation written is parsable by the LBJ2
101     * compiler, but not very readable.
102     *
103     * @param buffer The buffer to write to.
104    **/
105   public void write(StringBuffer buffer)
106   {
107     if (parenthesized) buffer.append("(");
108     inference.write(buffer);
109     buffer.append("(");
110     classifier.write(buffer);
111     buffer.append(")");
112     if (parenthesized) buffer.append(")");
113   }

116   /**
117     * Creates a <code>StringBuffer</code> containing a shallow representation
118     * of this <code>ASTNode</code>.
119     *
120     * @return A <code>StringBuffer</code> containing a shallow text
121     *         representation of the given node.
122    **/
123   public StringBuffer shallow()
124   {
125     StringBuffer buffer = new StringBuffer();
126     returnType.write(buffer);
127     buffer.append(" ");
128     name.write(buffer);
129     buffer.append("(");
130     argument.write(buffer);
131     buffer.append(") ");
132 
133     if (cacheIn != null)
134     {
135       buffer.append("cached");
136 
137       if (!cacheIn.toString().equals(ClassifierAssignment.mapCache))
138       {
139         buffer.append("in ");
140         cacheIn.write(buffer);
141       }
142 
143       buffer.append(" ");
144     }
145 
146     buffer.append("<- ");
147     inference.write(buffer);
148     buffer.append("(");
149     classifier.write(buffer);
150     buffer.append(")");
151     return buffer;
152   }
