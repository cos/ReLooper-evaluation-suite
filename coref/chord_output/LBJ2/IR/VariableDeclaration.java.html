6   /**
7     * Represents a local variable declaration.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class VariableDeclaration extends Statement
12  {
13    /** Whether or not the argument was modified as final. */
14    public boolean isFinal;
15    /** (&not;&oslash;) The type of the declared variable. */
16    public Type type;
17    /** (&not;&oslash;) The names of variables declared in this statement. */
18    public NameList names;
19    /**
20      * (&not;&oslash;) The initializing expressions for the declared variables,
21      * <code>null</code> being an allowable value.
22     **/
23    public ExpressionList initializers;
24  
25  
26    /**
27      * Parser's constructor, leaving the type to be filled in later.
28      *
29      * @param n  The name of the declared variable.
30     **/
31    public VariableDeclaration(Name n) { this(n, null); }
32  
33    /**
34      * Parser's constructor, leaving the type to be filled in later.
35      *
36      * @param n  The name of the declared variable.
37      * @param i  The initializing expression for the declared variable.
38     **/
39    public VariableDeclaration(Name n, Expression i)
40    {
41      super(-1, -1);  // Line and byte offset information will be filled in
42                      // later.
43      names = new NameList(n);
44      initializers = new ExpressionList(i);
45      type = null;
46      isFinal = false;
47    }
48  
49    /**
50      * Full constructor.
51      *
52      * @param t  The type of the declared variables.
53      * @param n  The names of the declared variables.
54      * @param i  The initializing expressions for the declared variables.
55      * @param f  Whether or not the variables were declared as final.
56     **/
57    public VariableDeclaration(Type t, NameList n, ExpressionList i, boolean f)
58    {
59      super(t.line, t.byteOffset);
60      type = t;
61      names = n;
62      initializers = i;
63      isFinal = f;
64    }
65  
66  
67    /**
68      * Adds the declarations in the specified declaration statement to the
69      * declarations in this statement.
70      *
71      * @param v  The variables to be added.
72     **/
73    public void addVariables(VariableDeclaration v)
74    {
75      names.addAll(v.names);
76      initializers.addAll(v.initializers);
77    }
78  
79  
80    /**
81      * Setting this declaration statement's type also sets its line and byte
82      * offset information.
83      *
84      * @param t  The new type for this variable declaration statement.
85     **/
86    public void setType(Type t)
87    {
88      type = t;
89      line = type.line;
90      byteOffset = type.byteOffset;
91    }
92  
93  
94    /**
95      * Returns an iterator used to successively access the children of this
96      * node.
97      *
98      * @return An iterator used to successively access the children of this
99      *         node.
100    **/
101   public ASTNodeIterator iterator()
102   {
103     ASTNodeIterator I = new ASTNodeIterator(3);
104     I.children[0] = type;
105     I.children[1] = names;
106     I.children[2] = initializers;
107     return I;
108   }
109 
110 
111   /**
112     * Creates a new object with the same primitive data, and recursively
113     * creates new member data objects as well.
114     *
115     * @return The clone node.
116    **/
117   public Object clone()
118   {
119     return
120       new VariableDeclaration((Type) type.clone(), (NameList) names.clone(),
121                               (ExpressionList) initializers.clone(), isFinal);
122   }
123 
124 
125   /**
126     * Ensures that the correct <code>run()</code> method is called for this
127     * type of node.
128     *
129     * @param pass The pass whose <code>run()</code> method should be called.
130    **/
131   public void runPass(Pass pass) { pass.run(this); }
132 
133 
134   /**
135     * Writes a string representation of this <code>ASTNode</code> to the
136     * specified buffer.  The representation written is parsable by the LBJ2
137     * compiler, but not very readable.
138     *
139     * @param buffer The buffer to write to.
140    **/
141   public void write(StringBuffer buffer)
142   {
143     if (isFinal) buffer.append("final ");
144     type.write(buffer);
145 
146     ASTNodeIterator N = names.iterator();
147     ExpressionList.ExpressionListIterator I = initializers.listIterator();
148     buffer.append(" ");
149     N.next().write(buffer);
150 
151     Expression i = I.nextItem();
152     if (i != null)
153     {
154       buffer.append(" = ");
155       i.write(buffer);
156     }
157 
158     while (N.hasNext())
159     {
160       buffer.append(", ");
161       N.next().write(buffer);
162       i = I.nextItem();
163       if (i != null)
164       {
165         buffer.append(" = ");
166         i.write(buffer);
167       }
168     }
169 
170     buffer.append(";");
171   }
