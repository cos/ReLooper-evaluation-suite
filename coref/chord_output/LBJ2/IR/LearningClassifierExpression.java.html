284   /**
285     * Parser's supervised learning constructor.
286     *
287     * @param l          The expression representing the labeler classifier.
288     * @param cl         A list of clauses.
289     * @param line       The line on which the source code represented by this
290     *                   node is found.
291     * @param byteOffset The byte offset from the beginning of the source file
292     *                   at which the source code represented by this node is
293     *                   found.
294    **/
295   public LearningClassifierExpression(ClassifierExpression l, LinkedList cl,
296                                       int line, int byteOffset)
297   {
298     super(line, byteOffset);
299 
300     labeler = l;
301     extractor = null;
302     parser = null;
303     learnerConstructor = null;
304     checkDiscreteValues = false;
305     alpha = new Constant("" + defaultAlpha);
306 
307     for (Iterator I = cl.iterator(); I.hasNext(); )
308     {
309       Clause c = (Clause) I.next();
310 
311       if (c.type == Clause.USING)
312       {
313         extractor = (ClassifierExpression) c.argument;
314         ++usingClauses;
315       }
316       else if (c.type == Clause.FROM)
317       {
318         parser = (InstanceCreationExpression) c.argument;
319         rounds = c.rounds;
320         ++fromClauses;
321       }
322       else if (c.type == Clause.WITH)
323       {
324         if (c.learnerParameterBlock == null)
325           learnerConstructor = (InstanceCreationExpression) c.argument;
326         else
327         {
328           learnerName = (Name) c.argument;
329           learnerParameterBlock = c.learnerParameterBlock;
330         }
331 
332         ++withClauses;
333       }
334       else if (c.type == Clause.EVALUATE)
335       {
336         evaluation = (Expression) c.argument;
337         ++evaluateClauses;
338       }
339       else if (c.type == Clause.CVAL)
340       {
341         if (c.argument != null)
342           K = (Constant) c.argument;
343         else
344           // if there was a cval clause, but k was not given, set it to -1 so
345           // crossvalidation gets invoked anyway, this occurs legally in the
346           // case of manual separation
347           K = new Constant("-1");
348 
349         // handle default action
350         if (c.splitStrategy != null)
351           splitStrategy = c.splitStrategy;
352         else
353           splitStrategy = FoldParser.SplitStrategy.sequential;
354 
355         ++cvalClauses;
356       }
357       else if (c.type == Clause.TESTINGMETRIC)
358       {
359         testingMetric = (InstanceCreationExpression) c.argument;
360         ++testingMetricClauses;
361       }
362       else if (c.type == Clause.ALPHA)
363       {
364         alpha = (Constant) c.argument;
365         ++alphaClauses;
366       }
367       else if (c.type == Clause.PREEXTRACT)
368       {
369         preExtract = (Constant) c.argument;
370         ++preExtractClauses;
371       }
372       else if (c.type == Clause.PROGRESSOUTPUT)
373       {
374         progressOutput = (Constant) c.argument;
375         ++progressOutputClauses;
376       }
377     }
378   }

739     public static final int USING = 0;
740     /** Value of the <code>type</code> variable. */
741     public static final int FROM = 1;

801     /**
802       * Initializing constructor.
803       *
804       * @param t  The type.
805       * @param a  The argument node.

807     public Clause(int t, ASTNode a) { this(t, a, (Constant) null); }
808 
809     /**
810       * Full constructor.
811       *
812       * @param t  The type.
813       * @param a  The argument node.
814       * @param r  Could represent the number of training repetitions or the
815       *           split strategy.
816     **/
817     public Clause(int t, ASTNode a, Constant r)
818     {
819       type = t;
820 
821       if (t == CVAL)
822       {
823         if (r == null)
824         {
825           if (!(a instanceof Constant
826                 && ((Constant) a).value.equals("\"manual\"")))
827           {
828             System.err.println(
829               "K must be specified if not using the manual split strategy.");
830             System.exit(1);
831           }
832 
833           splitStrategy = FoldParser.SplitStrategy.manual;
834         }
835         else
836         {
837           argument = a;
838           String s = r.value;
839           if (s.startsWith("\"") && s.endsWith("\"") && s.length() >= 2)
840             s = s.substring(1, s.length() - 1);
841 
842           if (s.equals("random"))
843             splitStrategy = FoldParser.SplitStrategy.random;
844           else if (s.equals("sequential"))
845             splitStrategy = FoldParser.SplitStrategy.sequential;
846           else if (s.equals("kth"))
847             splitStrategy = FoldParser.SplitStrategy.kth;
848           else if (s.equals("manual"))
849             splitStrategy = FoldParser.SplitStrategy.manual;
850           else
851           {
852             System.err.println("Error: '" + s + "' is not a split strategy.");
853             System.exit(1);
854           }
855         }
856       }
857       else
858       {
859         argument = a;
860         rounds = r;
861       }
862     }
