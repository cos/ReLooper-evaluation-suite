7   /**
8     * Currently, this is just a wrapper class for <code>LinkedList</code>.  The
9     * code that uses it looks a little cleaner when casts are all taken care of
10    * automatically.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class StatementList extends List
15  {
16    /** Default constructor. */
17    public StatementList() { super(-1, -1, " "); }
18  
19    /**
20      * Initializing constructor.
21      *
22      * @param line       The line on which the source code represented by this
23      *                   node is found.
24      * @param byteOffset The byte offset from the beginning of the source file
25      *                   at which the source code represented by this node is
26      *                   found.
27     **/
28    public StatementList(int line, int byteOffset)
29    {
30      super(line, byteOffset, " ");
31    }
32  
33    /**
34      * Initializing constructor.  Requires its argument to be
35      * non-<code>null</code>.
36      *
37      * @param s  A single <code>Statement</code> with which to initialize this
38      *           list.
39     **/
40    public StatementList(Statement s)
41    {
42      super(s.line, s.byteOffset, " ");
43      list.add(s);
44    }
45  
46  
47    /**
48      * Adds another <code>Statement</code> to the end of the list.
49      *
50      * @param s  A reference to the <code>Statement</code> to be added.
51     **/
52    public void add(Statement s) { list.add(s); }
53  
54  
55    /**
56      * Adds all the <code>Statement</code>s in another
57      * <code>StatementList</code> to the end of this
58      * <code>StatementList</code>.
59      *
60      * @param s  The list to be added.
61     **/
62    public void addAll(StatementList s) { list.addAll(s.list); }
63  
64  
65    /**
66      * Transforms the list into an array of statements.
67      *
68      * @return An array of statements containing references to every statement
69      *         in the list.
70     **/
71    public Statement[] toArray()
72    {
73      return (Statement[]) list.toArray(new Statement[list.size()]);
74    }
75  
76  
77    /**
78      * Returns an iterator used specifically to access the elements of this
79      * list.
80      *
81      * @return An iterator used specifically to access the elements of this
82      *         list.
83     **/
84    public StatementListIterator listIterator()
85    {
86      return new StatementListIterator();
87    }
88  
89  
90    /**
91      * Returns an iterator used to successively access the children of this
92      * node.
93      *
94      * @return An iterator used to successively access the children of this
95      *         node.
96     **/
97    public ASTNodeIterator iterator() { return listIterator(); }
98  
99  
100   /**
101     * Creates a new object with the same primitive data, and recursively
102     * creates new member data objects as well.
103     *
104     * @return The clone node.
105    **/
106   public Object clone()
107   {
108     StatementList clone = new StatementList();
109     for (Iterator i = list.iterator(); i.hasNext(); )
110       clone.list.add(((Statement) i.next()).clone());
111     return clone;
112   }

115   /**
116     * Ensures that the correct <code>run()</code> method is called for this
117     * type of node.
118     *
119     * @param pass The pass whose <code>run()</code> method should be called.
120    **/
121   public void runPass(Pass pass) { pass.run(this); }
122

124   /**
125     * Used to iterate though the children of a list of AST nodes.  The entire
126     * interface of <code>java.util.ListIterator</code> is exposed through this
127     * class.
128     *
129     * @author Nick Rizzolo
130    **/
131   public class StatementListIterator extends NodeListIterator
132   {
133     /**
134       * Returns the next AST node in the list.  This method may be called
135       * repeatedly to iterate through the list, or intermixed with calls to
136       * <code>previous()</code> to go back and forth.  (Note that alternating
137       * calls to <code>next()</code> and <code>previous()</code> will return
138       * the same element repeatedly.)
139       *
140       * @return The next AST node in the list.
141      **/
142     public Statement nextItem() { return (Statement) I.next(); }
143 
144 
145     /**
146       * Returns the previous element in the list. This method may be called
147       * repeatedly to iterate through the list backwards, or intermixed with
148       * calls to next to go back and forth. (Note that alternating calls to
149       * next and previous will return the same element repeatedly.)
150       *
151       * @return The previous AST node in the list.
152      **/
153     public Statement previousItem() { return (Statement) I.previous(); }
154   }
155 }
