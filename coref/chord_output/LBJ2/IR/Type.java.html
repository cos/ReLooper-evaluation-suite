4   /**
5     * Abstract class representing the type of a variable or the return type of a
6     * method.
7     *
8     * @author Nick Rizzolo
9    **/
10  public abstract class Type extends ASTNode
11  {
12    /**
13      * This method takes a Java <code>Class</code> object and generates an
14      * LBJ2 <code>Type</code> that represents the same type.
15      *
16      * @param c  The <code>Class</code> to "translate".
17      * @return   The <code>Type</code> object that represents the same type as
18      *           the argument.
19     **/
20    public static Type parseType(Class c) { return parseType(c.getName()); }

23    /**
24      * This method takes a Java type encoding and generates an LBJ2
25      * <code>Type</code> that represents the same type.
26      *
27      * @param encoding The encoding to decode.
28      * @return The <code>Type</code> object that represents the same type as
29      *         the argument.
30     **/
31    public static Type parseType(String encoding)
32    {
33      int dimensions = 0;
34      while (encoding.charAt(dimensions) == '[') ++dimensions;
35  
36      if (dimensions == 0)
37      {
38        if (encoding.equals("boolean"))
39          return new PrimitiveType(PrimitiveType.BOOLEAN);
40        else if (encoding.equals("byte"))
41          return new PrimitiveType(PrimitiveType.BYTE);
42        else if (encoding.equals("char"))
43          return new PrimitiveType(PrimitiveType.CHAR);
44        else if (encoding.equals("double"))
45          return new PrimitiveType(PrimitiveType.DOUBLE);
46        else if (encoding.equals("float"))
47          return new PrimitiveType(PrimitiveType.FLOAT);
48        else if (encoding.equals("int"))
49          return new PrimitiveType(PrimitiveType.INT);
50        else if (encoding.equals("long"))
51          return new PrimitiveType(PrimitiveType.LONG);
52        else if (encoding.equals("short"))
53          return new PrimitiveType(PrimitiveType.SHORT);
54      }
55  
56      Type result = null;
57      if (dimensions > 0)
58      {
59        switch (encoding.charAt(dimensions))
60        {
61          case 'Z': result = new PrimitiveType(PrimitiveType.BOOLEAN); break;
62          case 'B': result = new PrimitiveType(PrimitiveType.BYTE);    break;
63          case 'C': result = new PrimitiveType(PrimitiveType.CHAR);    break;
64          case 'D': result = new PrimitiveType(PrimitiveType.DOUBLE);  break;
65          case 'F': result = new PrimitiveType(PrimitiveType.FLOAT);   break;
66          case 'I': result = new PrimitiveType(PrimitiveType.INT);     break;
67          case 'J': result = new PrimitiveType(PrimitiveType.LONG);    break;
68          case 'S': result = new PrimitiveType(PrimitiveType.SHORT);   break;
69          case 'L': break;
70          default:
71            System.err.println("ERROR: Can't parse type '" + encoding + "'");
72            new Exception().printStackTrace();
73            System.exit(1);
74        }
75      }
76  
77      if (result == null)
78      {
79        String referenceString;
80        if (dimensions == 0) referenceString = encoding;
81        else
82          referenceString =
83            encoding.substring(dimensions + 1, encoding.length() - 1);
84  
85        for (int i = referenceString.indexOf('$'); i != -1;
86             i = referenceString.indexOf('$'))
87          referenceString = referenceString.substring(0, i) + "."
88                            + referenceString.substring(i + 1);
89  
90        result = new ReferenceType(new Name(referenceString));
91      }
92  
93      for (int i = 0; i < dimensions; ++i) result = new ArrayType(result);
94      return result;
95    }
