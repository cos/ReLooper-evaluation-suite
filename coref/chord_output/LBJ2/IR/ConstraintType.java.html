7   /**
8     * A constraint's type is defined by what it takes as input.  This class does
9     * not represent a syntax that appears in the source - it is constructed only
10    * during semantic analysis.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class ConstraintType extends ClassifierType
15  {
16    /**
17      * Initializing constructor.
18      *
19      * @param i  The classifier's input type.
20     **/
21    public ConstraintType(Type i)
22    {
23      super(i,
24            new ClassifierReturnType(
25                ClassifierReturnType.DISCRETE,
26                new ConstantList(DiscreteFeature.BooleanValues)),
27            false);
28  
29      try { myClass = Class.forName("LBJ2.infer.ParameterizedConstraint"); }
30      catch (Exception e)
31      {
32        System.err.println("Class 'LBJ2.infer.ParameterizedConstraint' not "
33                           + "found.  Aborting.");
34        System.exit(1);
35      }
36    }
37  
38  
39    /**
40      * Two <code>ConstraintType</code>s are equivalent when their input types
41      * match.
42      *
43      * @param o  The object whose equality with this object needs to be tested.
44      * @return   <code>true</code> if the two objects are equal, and
45      *           <code>false</code> otherwise.
46     **/
47    public boolean equals(Object o)
48    {
49      return o instanceof ConstraintType
50             && input.equals(((ConstraintType) o).input);
51    }

54    /**
55      * Returns an iterator used to successively access the children of this
56      * node.
57      *
58      * @return An iterator used to successively access the children of this
59      *         node.
60     **/
61    public ASTNodeIterator iterator()
62    {
63      ASTNodeIterator I = new ASTNodeIterator(1);
64      I.children[0] = input;
65      return I;
66    }

71      * creates new member data objects as well.
72      *
73      * @return The clone node.
74     **/
75    public Object clone() { return new ConstraintType((Type) input.clone()); }

80      * type of node.

84    public void runPass(Pass pass) { pass.run(this); }
