8   /**
9     * An "at most" quantifier has the form:
10    * <blockquote>
11    *   <code>atmost <i>expression</i> of <i>argument</i> in
12    *   (<i>expression</i>) <i>constraint-expression</i>
13    * </blockquote>
14    * where the first <code><i>expression</i></code> must evaluate to an
15    * <code>int</code>, the second <code><i>expression</i></code> must evaluate
16    * to a <code>Collection</code>, and the "at most" quantifier expression is
17    * sastisfied iff when taking settings of <code><i>argument</i></code> from
18    * the <code>Collection</code>, <code><i>constraint-expression</i></code> is
19    * satisfied at most as many times as the integer the first
20    * <code><i>expression</i></code> evaluates to.
21    *
22    * @author Nick Rizzolo
23   **/
24  public class AtMostQuantifierExpression
25         extends QuantifiedConstraintExpression
26  {
27    /**
28      * (&not;&oslash;) This expression evaluates to an integer representing the
29      * maximum number of objects that must satisfy the child constraint
30      * expression in order for this quantified constraint expression to be
31      * satisfied.
32     **/
33    public Expression upperBound;
34    /**
35      * Filled in by <code>SemanticAnalysis</code>, this flag is set if
36      * <code>upperBound</code> contains any quantified variables.
37     **/
38    public boolean upperBoundIsQuantified;
39  
40  
41    /**
42      * Full constructor.
43      *
44      * @param line       The line on which the source code represented by this
45      *                   node is found.
46      * @param byteOffset The byte offset from the beginning of the source file
47      *                   at which the source code represented by this node is
48      *                   found.
49      * @param ub         The upper bound expression.
50      * @param a          The quantification variable specification.
51      * @param c          Evaluates to the collection of objects.
52      * @param co         The quantified constraint.
53     **/
54    public AtMostQuantifierExpression(int line, int byteOffset,
55                                      Expression ub, Argument a, Expression c,
56                                      ConstraintExpression co)
57    {
58      super(line, byteOffset, a, c, co);
59      upperBound = ub;
60    }
61  
62    /**
63      * Parser's constructor.  Line and byte offset information are taken from
64      * the token.
65      *
66      * @param t  The token containing line and byte offset information.
67      * @param ub The upper bound expression.
68      * @param a  The quantification variable specification.
69      * @param c  Evaluates to the collection of objects.
70      * @param co The quantified constraint.
71     **/
72    public AtMostQuantifierExpression(TokenValue t, Expression ub, Argument a,
73                                      Expression c, ConstraintExpression co)
74    {
75      this(t.line, t.byteOffset, ub, a, c, co);
76    }
77  
78  
79    /**
80      * Returns a set of <code>Argument</code>s storing the name and type of
81      * each variable that is a subexpression of this expression.  This method
82      * cannot be run before <code>SemanticAnalysis</code> runs.
83     **/
84    public HashSet getVariableTypes()
85    {
86      HashSet result = super.getVariableTypes();
87      result.addAll(upperBound.getVariableTypes());
88      return result;
89    }
90  
91  
92    /**
93      * Returns an iterator used to successively access the children of this
94      * node.
95      *
96      * @return An iterator used to successively access the children of this
97      *         node.
98     **/
99    public ASTNodeIterator iterator()
100   {
101     ASTNodeIterator I = new ASTNodeIterator(4);
102     I.children[0] = upperBound;
103     I.children[1] = argument;
104     I.children[2] = collection;
105     I.children[3] = constraint;
106     return I;
107   }

109 
110   /**
111     * Creates a new object with the same primitive data, and recursively
112     * creates new member data objects as well.
113     *
114     * @return The clone node.
115    **/
116   public Object clone()
117   {
118     return
119       new AtMostQuantifierExpression(
120           -1, -1, (Expression) upperBound.clone(),
121           (Argument) argument.clone(),
122           (Expression) collection.clone(),
123           (ConstraintExpression) constraint.clone());
124   }

127   /**
128     * Ensures that the correct <code>run()</code> method is called for this
129     * type of node.
130     *
131     * @param pass The pass whose <code>run()</code> method should be called.
132    **/
133   public void runPass(Pass pass) { pass.run(this); }

136   /**
137     * Writes a string representation of this <code>ASTNode</code> to the
138     * specified buffer.  The representation written is parsable by the LBJ2
139     * compiler, but not very readable.
140     *
141     * @param buffer The buffer to write to.
142    **/
143   public void write(StringBuffer buffer)
144   {
145     buffer.append("atmost ");
146     upperBound.write(buffer);
147     buffer.append(" of ");
148     argument.write(buffer);
149     buffer.append(" in (");
150     collection.write(buffer);
151     buffer.append(") ");
152     constraint.write(buffer);
153   }
