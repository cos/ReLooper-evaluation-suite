7   /**
8     * Represents the declaration of a constraint.  Constraints declared in an
9     * LBJ source file are mainly used to constrain inferences, but they may also
10    * be used exactly as if they were a classifier with return type
11    * <code>discrete{false, true}</code>.
12    *
13    * @author Nick Rizzolo
14   **/
15  public class ConstraintDeclaration extends Declaration
16                                     implements LBJ2.CodeGenerator
17  {
18    /** (&not;&oslash;) The input specification of the constraint. */
19    public Argument argument;
20    /** (&not;&oslash;) Statements making up the body of the constraint. */
21    public Block body;
22  
23  
24    /**
25      * Full constructor.
26      *
27      * @param c          A Javadoc comment associated with the declaration.
28      * @param line       The line on which the source code represented by this
29      *                   node is found.
30      * @param byteOffset The byte offset from the beginning of the source file
31      *                   at which the source code represented by this node is
32      *                   found.
33      * @param n          The constraint's name.
34      * @param a          The input specification of the constraint.
35      * @param b          The code block representing the constraint.
36     **/
37    public ConstraintDeclaration(String c, int line, int byteOffset, Name n,
38                                 Argument a, Block b)
39    {
40      super(c, n, line, byteOffset);
41      argument = a;
42      body = b;
43    }
44  
45    /**
46      * Parser's constructor.  Line and byte offset information is taken from
47      * the first token.
48      *
49      * @param t  The first token indicates line and byte offset information.
50      * @param i  The identifier token representing the constraint's name.
51      * @param a  The input specification of the constraint.
52      * @param b  The code block representing the constraint.
53     **/
54    public ConstraintDeclaration(TokenValue t, TokenValue i, Argument a,
55                                 Block b)
56    {
57      this(null, t.line, t.byteOffset, new Name(i), a, b);
58    }
59  
60  
61    /**
62      * Returns the type of the declaration.
63      *
64      * @return The type of the declaration.
65     **/
66    public Type getType() { return new ConstraintType(argument.getType()); }
67  
68  
69    /** Returns the name of the <code>ConstraintDeclaration</code>. */
70    public String getName() { return "" + name; }
71  
72  
73    /**
74      * Returns the line number on which this AST node is found in the source
75      * (starting from line 0).  This method exists to fulfull the
76      * implementation of <code>CodeGenerator</code>.
77      * @see LBJ2.CodeGenerator
78     **/
79    public int getLine() { return line; }
80  
81  
82    /** Returns a shallow textual representation of this AST node. */
83    public StringBuffer shallow()
84    {
85      StringBuffer buffer = new StringBuffer();
86      write(buffer);
87      return buffer;
88    }
89  
90  
91    /**
92      * Returns an iterator used to successively access the children of this
93      * node.
94      *
95      * @return An iterator used to successively access the children of this
96      *         node.
97     **/
98    public ASTNodeIterator iterator()
99    {
100     ASTNodeIterator I = new ASTNodeIterator(3);
101     I.children[0] = name;
102     I.children[1] = argument;
103     I.children[2] = body;
104     return I;
105   }

107 
108   /**
109     * Creates a new object with the same primitive data, and recursively
110     * creates new member data objects as well.
111     *
112     * @return The clone node.
113    **/
114   public Object clone()
115   {
116     return
117       new ConstraintDeclaration(comment, -1, -1, (Name) name.clone(),
118                                (Argument) argument.clone(),
119                                (Block) body.clone());
120   }

124     * Ensures that the correct <code>run()</code> method is called for this
125     * type of node.
126     *
127     * @param pass The pass whose <code>run()</code> method should be called.
128    **/
129   public void runPass(Pass pass) { pass.run(this); }

132   /**
133     * Writes a string representation of this <code>ASTNode</code> to the
134     * specified buffer.  The representation written is parsable by the LBJ2
135     * compiler, but not very readable.
136     *
137     * @param buffer The buffer to write to.
138    **/
139   public void write(StringBuffer buffer)
140   {
141     buffer.append("constraint ");
142     name.write(buffer);
143     buffer.append("(");
144     argument.write(buffer);
145     buffer.append(") ");
146     body.write(buffer);
147   }
