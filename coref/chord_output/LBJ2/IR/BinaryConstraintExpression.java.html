6   /**
7     * This class represents a constraint expression involving a binary operator.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class BinaryConstraintExpression extends ConstraintExpression
12  {
13    /** (&not;&oslash;) The binary operation. */
14    public Operator operation;
15    /** (&not;&oslash;) The left hand side of the binary expression. */
16    public ConstraintExpression left;
17    /** (&not;&oslash;) The right hand side of the binary expression. */
18    public ConstraintExpression right;
19  
20  
21    /**
22      * Initializing constructor.  Line and byte offset information are taken
23      * from the representation of the operator.
24      *
25      * @param op Reference to the operator's representation.
26      * @param l  Reference to the left hand side's representation.
27      * @param r  Reference to the right hand side's representation.
28     **/
29    public BinaryConstraintExpression(Operator op, ConstraintExpression l,
30                                      ConstraintExpression r)
31    {
32      super(op.line, op.byteOffset);
33      operation = op;
34      left = l;
35      right = r;
36    }
37  
38  
39    /**
40      * Returns an iterator used to successively access the children of this
41      * node.
42      *
43      * @return An iterator used to successively access the children of this
44      *         node.
45     **/
46    public ASTNodeIterator iterator()
47    {
48      ASTNodeIterator I = new ASTNodeIterator(3);
49      I.children[0] = left;
50      I.children[1] = operation;
51      I.children[2] = right;
52      return I;
53    }
54  
55  
56    /**
57      * Creates a new object with the same primitive data, and recursively
58      * creates new member data objects as well.
59      *
60      * @return The clone node.
61     **/
62    public Object clone()
63    {
64      return
65        new BinaryConstraintExpression((Operator) operation.clone(),
66                                       (ConstraintExpression) left.clone(),
67                                       (ConstraintExpression) right.clone());
68    }
69  
70  
71    /**
72      * Ensures that the correct <code>run()</code> method is called for this
73      * type of node.
74      *
75      * @param pass The pass whose <code>run()</code> method should be called.
76     **/
77    public void runPass(Pass pass) { pass.run(this); }
78  
79  
80    /**
81      * Writes a string representation of this <code>ASTNode</code> to the
82      * specified buffer.  The representation written is parsable by the LBJ2
83      * compiler, but not very readable.
84      *
85      * @param buffer The buffer to write to.
86     **/
87    public void write(StringBuffer buffer)
88    {
89      if (parenthesized) buffer.append("(");
90      left.write(buffer);
91      buffer.append(" ");
92      operation.write(buffer);
93      buffer.append(" ");
94      right.write(buffer);
95      if (parenthesized) buffer.append(")");
96    }
97  }
