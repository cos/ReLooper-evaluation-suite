3   import LBJ2.Pass;

6   /**
7     * Representation of an expression that casts a value to another type.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class CastExpression extends Expression
12  {
13    /** (&not;&oslash;) The type to cast to. */
14    public Type type;
15    /** (&not;&oslash;) The expression whose value should be casted. */
16    public Expression expression;
17  
18  
19    /**
20      * Initializing constructor.  Line and byte offset information are taken
21      * from the type's representation.
22      *
23      * @param t  Reference to the object representing the cast's type.
24      * @param e  Reference to the object representing the expression to cast.
25     **/
26    public CastExpression(Type t, Expression e)
27    {
28      super(t.line, t.byteOffset);
29      type = t;
30      expression = e;
31    }
32  
33  
34    /**
35      * Returns an iterator used to successively access the children of this
36      * node.
37      *
38      * @return An iterator used to successively access the children of this
39      *         node.
40     **/
41    public ASTNodeIterator iterator()
42    {
43      ASTNodeIterator I = new ASTNodeIterator(2);
44      I.children[0] = type;
45      I.children[1] = expression;
46      return I;
47    }
48  
49  
50    /**
51      * Creates a new object with the same primitive data, and recursively
52      * creates new member data objects as well.
53      *
54      * @return The clone node.
55     **/
56    public Object clone()
57    {
58      return
59        new CastExpression((Type) type.clone(),
60                           (Expression) expression.clone());
61    }
62  
63  
64    /**
65      * Ensures that the correct <code>run()</code> method is called for this
66      * type of node.
67      *
68      * @param pass The pass whose <code>run()</code> method should be called.
69     **/
70    public void runPass(Pass pass) { pass.run(this); }
71  
72  
73    /**
74      * Writes a string representation of this <code>ASTNode</code> to the
75      * specified buffer.  The representation written is parsable by the LBJ2
76      * compiler, but not very readable.
77      *
78      * @param buffer The buffer to write to.
79     **/
80    public void write(StringBuffer buffer)
81    {
82      if (parenthesized) buffer.append("(");
83      buffer.append("(");
84      type.write(buffer);
85      buffer.append(") ");
86      expression.write(buffer);
87      if (parenthesized) buffer.append(")");
88    }
89  }
