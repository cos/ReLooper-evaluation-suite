7   /**
8     * This class represents the atom of the LBJ constraint expression: the
9     * (in)equality comparison.  The application of a learning classifier to an
10    * example object is here compared to either another learning classifier
11    * application or to an arbitrary Java expression evaluating to a
12    * <code>String</code>.
13    *
14    * @author Nick Rizzolo
15   **/
16  public class ConstraintEqualityExpression extends ConstraintExpression
17  {
18    /**
19      * (&not;&oslash;) Represents either an equality or an inequality
20      * comparison.
21     **/
22    public Operator operation;
23    /** (&not;&oslash;) The expression on the left hand side of the operator. */
24    public Expression left;
25    /**
26      * Filled in by <code>SemanticAnalysis</code>, this flag is set if
27      * <code>left</code> represents the invocation of a discrete learner.
28     **/
29    public boolean leftIsDiscreteLearner;
30    /**
31      * Filled in by <code>SemanticAnalysis</code>, this flag is set if
32      * <code>left</code> contains any quantified variables.
33     **/
34    public boolean leftIsQuantified;
35    /**
36      * (&not;&oslash;) The expression on the right hand side of the operator.
37     **/
38    public Expression right;
39    /**
40      * Filled in by <code>SemanticAnalysis</code>, this flag is set if
41      * <code>right</code> represents the invocation of a discrete learner.
42     **/
43    public boolean rightIsDiscreteLearner;
44    /**
45      * Filled in by <code>SemanticAnalysis</code>, this flag is set if
46      * <code>right</code> contains any quantified variables.
47     **/
48    public boolean rightIsQuantified;
49  
50  
51    /**
52      * Full constructor.  Line and byte offset information are taken from the
53      * operator.
54      *
55      * @param o  The equality comparison operator.
56      * @param l  The expression on the left of the operator.
57      * @param r  The expression on the right of the operator.
58     **/
59    public ConstraintEqualityExpression(Operator o, Expression l, Expression r)
60    {
61      super(o.line, o.byteOffset);
62      operation = o;
63      left = l;
64      right = r;
65    }
66  
67  
68    /**
69      * Returns a set of <code>Argument</code>s storing the name and type of
70      * each variable that is a subexpression of this expression.  This method
71      * cannot be run before <code>SemanticAnalysis</code> runs.
72     **/
73    public HashSet getVariableTypes()
74    {
75      HashSet result = left.getVariableTypes();
76      result.addAll(right.getVariableTypes());
77      return result;
78    }
79  
80  
81    /**
82      * Determines if there are any quantified variables in this expression.
83      * This method cannot be run before <code>SemanticAnalysis</code> runs.
84     **/
85    public boolean containsQuantifiedVariable()
86    {
87      return left.containsQuantifiedVariable()
88             || right.containsQuantifiedVariable();
89    }
90  
91  
92    /**
93      * Returns an iterator used to successively access the children of this
94      * node.
95      *
96      * @return An iterator used to successively access the children of this
97      *         node.
98     **/
99    public ASTNodeIterator iterator()
100   {
101     ASTNodeIterator I = new ASTNodeIterator(3);
102     I.children[0] = left;
103     I.children[1] = operation;
104     I.children[2] = right;
105     return I;
106   }
107 
108 
109   /**
110     * Creates a new object with the same primitive data, and recursively
111     * creates new member data objects as well.
112     *
113     * @return The clone node.
114    **/
115   public Object clone()
116   {
117     return
118       new ConstraintEqualityExpression((Operator) operation.clone(),
119                                        (Expression) left.clone(),
120                                        (Expression) right.clone());
121   }

124   /**
125     * Ensures that the correct <code>run()</code> method is called for this
126     * type of node.
127     *
128     * @param pass The pass whose <code>run()</code> method should be called.
129    **/
130   public void runPass(Pass pass) { pass.run(this); }
131 
132 
133   /**
134     * Writes a string representation of this <code>ASTNode</code> to the
135     * specified buffer.  The representation written is parsable by the LBJ2
136     * compiler, but not very readable.
137     *
138     * @param buffer The buffer to write to.
139    **/
140   public void write(StringBuffer buffer)
141   {
142     left.write(buffer);
143     buffer.append(" ");
144     operation.write(buffer);
145     buffer.append(" ");
146     right.write(buffer);
147   }
148 }
