6   /**
7     * This class represents a conditional expression.  Conditional expressions
8     * have the form <code>(c ? t : e)</code>.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class Conditional extends Expression
13  {
14    /** (&not;&oslash;) The condition of the conditional expression. */
15    public Expression condition;
16    /**
17      * (&not;&oslash;) The expression to evaluate if the condition evaluates to
18      * <code>true</code>.
19     **/
20    public Expression thenClause;
21    /**
22      * (&not;&oslash;) The expression to evaluate if the condition evaluates to
23      * <code>false</code>.
24     **/
25    public Expression elseClause;
26  
27  
28    /**
29      * Full constructor.
30      *
31      * @param c          The condition.
32      * @param t          The expression to evaluate if the condition is
33      *                   <code>true</code>.
34      * @param e          The expression to evaluate if the condition is
35      *                   <code>false</code>.
36      * @param line       The line on which the source code represented by this
37      *                   node is found.
38      * @param byteOffset The byte offset from the beginning of the source file
39      *                   at which the source code represented by this node is
40      *                   found.
41     **/
42    public Conditional(Expression c, Expression t, Expression e, int line,
43                       int byteOffset)
44    {
45      super(line, byteOffset);
46      condition = c;
47      thenClause = t;
48      elseClause = e;
49    }
50  
51  
52    /**
53      * Returns a hash code value for java hash structures.
54      *
55      * @return A hash code for this object.
56     **/
57    public int hashCode()
58    {
59      return condition.hashCode() + thenClause.hashCode()
60             + elseClause.hashCode();
61    }
62  
63  
64    /**
65      * Indicates whether some other object is "equal to" this one.
66      *
67      * @return <code>true</code> iff this object is the same as the argument.
68     **/
69    public boolean equals(Object o)
70    {
71      if (!(o instanceof Conditional)) return false;
72      Conditional c = (Conditional) o;
73      return condition.equals(c.condition) && thenClause.equals(c.thenClause)
74             && elseClause.equals(c.elseClause);
75    }
76  
77  
78    /**
79      * Returns an iterator used to successively access the children of this
80      * node.
81      *
82      * @return An iterator used to successively access the children of this
83      *         node.
84     **/
85    public ASTNodeIterator iterator()
86    {
87      ASTNodeIterator I = new ASTNodeIterator(3);
88      I.children[0] = condition;
89      I.children[1] = thenClause;
90      I.children[2] = elseClause;
91      return I;
92    }
93  
94  
95    /**
96      * Creates a new object with the same primitive data, and recursively
97      * creates new member data objects as well.
98      *
99      * @return The clone node.
100    **/
101   public Object clone()
102   {
103     return new Conditional((Expression) condition.clone(),
104                            (Expression) thenClause.clone(),
105                            (Expression) elseClause.clone(), -1, -1);
106   }
107 
108 
109   /**
110     * Ensures that the correct <code>run()</code> method is called for this
111     * type of node.
112     *
113     * @param pass The pass whose <code>run()</code> method should be called.
114    **/
115   public void runPass(Pass pass) { pass.run(this); }
116 
117 
118   /**
119     * Writes a string representation of this <code>ASTNode</code> to the
120     * specified buffer.  The representation written is parsable by the LBJ2
121     * compiler, but not very readable.
122     *
123     * @param buffer The buffer to write to.
124    **/
125   public void write(StringBuffer buffer)
126   {
127     if (parenthesized) buffer.append("(");
128     condition.write(buffer);
129     buffer.append(" ? ");
130     thenClause.write(buffer);
131     buffer.append(" : ");
132     elseClause.write(buffer);
133     if (parenthesized) buffer.append(")");
134   }
135 }
