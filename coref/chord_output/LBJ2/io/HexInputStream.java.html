6   /**
7     * This class receives input from another <code>InputStream</code> assuming
8     * that data is little endian, hexidecimal text, converts that text to bytes,
9     * and makes those bytes available through its interface.  The most common
10    * usage of this class will involve passing it to the constructor of another
11    * <code>InputStream</code>.  For instance: <br><br>
12    *
13    * <pre>
14    *   ObjectInputStream ois =
15    *     new ObjectInputStream(
16    *       new GZIPInputStream(
17    *         new HexInputStream(new FileInputStream(fileName))));
18    * </pre>
19    *
20    * @see HexOutputStream
21    * @author Nick Rizzolo
22   **/
23  public class HexInputStream extends InputStream
24  {
25    /** Characters representing a hexidecimal digit. */
26    private static final String digits = "0123456789ABCDEF";
27  
28    /**
29      * The <code>InputStream</code> from which yet-to-be-converted input
30      * should be received.
31     **/
32    private InputStream in;
33  
34  
35    /**
36      * Initializes this stream with another input stream.
37      *
38      * @param i  The input stream from which yet-to-be-converted input should
39      *           be received.
40     **/
41    public HexInputStream(InputStream i) { in = i; }
42  
43  
44    /**
45      * Reads the next byte of data from the input stream.  The value is
46      * returned as an <code>int</code> in the range 0 to 255.  If no byte is
47      * available because the end of the stream has been reached, the value -1
48      * is returned.  This method blocks until input data is available, the end
49      * of the stream is detected, or an exception is thrown.
50      *
51      * @return The next byte of data, or -1 if the end of the stream is
52      *         reached.
53     **/
54    public int read() throws IOException
55    {
56      int d1 = in.read();
57      if (d1 == -1) return -1;
58      int d2 = in.read();
59      if (d2 == -1)
60        throw new IOException("HexInputStream: Unexpected end of file");
61  
62      int i1 = digits.indexOf((char) d1);
63      if (i1 == -1)
64        throw new IOException("HexInputStream: Invalid input character: '"
65                              + ((char) d1) + "' (" + d1 + ")");
66      int i2 = digits.indexOf((char) d2);
67      if (i2 == -1)
68        throw new IOException("HexInputStream: Invalid input character: '"
69                              + ((char) d2) + "' (" + d2 + ")");
70  
71      return (i2 << 4) | i1;
72    }
73

75    /**
76      * This method has the same effect as <code>read(b, 0, b.length)</code>.
77      *
78      * @param b  A buffer in which the converted input is stored.
79      * @return   The total number of bytes read into the buffer, or -1 if there
80      *           is no more data because the end of the stream was previously
81      *           reached.
82     **/
83    public int read(byte[] b) throws IOException
84    {
85      return read(b, 0, b.length);
86    }

89    /**
90      * Reads up to <code>len</code> bytes of data from another input stream
91      * into an array of bytes.  An attempt is made to read as many as
92      * <code>len</code> bytes, but a smaller number may be read, possibly zero.
93      * The number of bytes actually read is returned as an integer. <br><br>
94      *
95      * This method blocks until input data is available, end of file is
96      * detected, or an exception is thrown. <br><br>
97      *
98      * If <code>b</code> is <code>null</code>, a
99      * <code>NullPointerException</code> is thrown. <br><br>
100     *
101     * If <code>off</code> is negative, or <code>len</code> is negative, or
102     * <code>off+len</code> is greater than the length of the array
103     * <code>b</code>, then an <code>IndexOutOfBoundsException</code> is
104     * thrown.  <br><br>
105     *
106     * If <code>len</code> is zero, then no bytes are read and 0 is returned;
107     * otherwise, there is an attempt to read at least one byte.  If no byte is
108     * available because the stream is at end of file, the value -1 is
109     * returned; otherwise, at least one byte is read and stored into
110     * <code>b</code>. <br><br>
111     *
112     * The first byte read is stored into element <code>b[off]</code>, the next
113     * one into <code>b[off+1]</code>, and so on.  The number of bytes read is,
114     * at most, equal to <code>len</code>.  Let <i>k</i> be the number of bytes
115     * actually read; these bytes will be stored in elements
116     * <code>b[off]</code> through <code>b[off+k-1]</code>, leaving elements
117     * <code>b[off+k]</code> through <code>b[off+len-1]</code> unaffected.
118     * <br><br>
119     *
120     * In every case, elements <code>b[0]</code> through <code>b[off-1]</code>
121     * and elements <code>b[off+len]</code> through <code>b[b.length-1]</code>
122     * are unaffected. <br><br>
123     *
124     * If the first byte cannot be read for any reason other than end of file,
125     * then an <code>IOException</code> is thrown.  In particular, an
126     * <code>IOException</code> is thrown if the input stream has been closed.
127     *
128     * @param b    A buffer into which the converted input is stored.
129     * @param off  The offset in the buffer at which to begin writing.
130     * @param len  The amount of bytes to be received and written into the
131     *             buffer.
132     * @return     The total number of bytes read into the buffer, or -1 if
133     *             there is no more data because the end of the stream has been
134     *             reached.
135    **/
136   public int read(byte[] b, int off, int len) throws IOException
137   {
138     byte[] hex = new byte[2 * len];
139     int bytesRead = in.read(hex);
140     if (bytesRead == -1) return -1;
141     if (bytesRead % 2 == 1)
142       throw new IOException("HexInputStream: Unexpected end of file");
143 
144     for (int i = 0; i < bytesRead; i += 2)
145     {
146       int d1 = digits.indexOf((char) hex[i]);
147       if (d1 == -1)
148         throw new IOException("HexInputStream: Invalid input character: '"
149                               + ((char) hex[i]) + "' (" + ((int) hex[i])
150                               + ")");
151       int d2 = digits.indexOf((char) hex[i + 1]);
152       if (d2 == -1)
153         throw new IOException("HexInputStream: Invalid input character: '"
154                               + ((char) hex[i + 1]) + "' ("
155                               + ((int) hex[i + 1]) + ")");
156 
157       b[i / 2] = (byte) ((d2 << 4) | d1);
158     }
159 
160     return bytesRead / 2;
161   }

164   /**
165     * Skips over and discards <code>n</code> bytes of data from this input
166     * stream.  The skip method may, for a variety of reasons, end up skipping
167     * over some smaller number of bytes, possibly 0.  This may result from any
168     * of a number of conditions; reaching end of file before <code>n</code>
169     * bytes have been skipped is only one possibility.  The actual number of
170     * bytes skipped is returned. If <code>n</code> is negative, no bytes are
171     * skipped.
172     *
173     * @param n  The number of bytes to be skipped.
174     * @return   The actual number of bytes skipped.
175    **/
176   public long skip(long n) throws IOException { return in.skip(n * 2); }

179   /**
180     * Returns the number of bytes that can be read (or skipped over) from this
181     * input stream without blocking by the next caller of a method for this
182     * input stream.  The next caller might be the same thread or or another
183     * thread.
184     *
185     * @return The number of bytes that can be read from this input stream
186     *         without blocking.
187    **/
188   public int available() throws IOException { return in.available() / 2; }

198   /**
199     * Marks the current position in this input stream.  A subsequent call to
200     * the <code>reset</code> method repositions this stream at the last marked
201     * position so that subsequent reads re-read the same bytes.
202     *
203     * The <code>readlimit</code> argument tells this input stream to allow
204     * that many bytes to be read before the mark position gets invalidated.
205     *
206     * The general contract of mark is that, if the method
207     * <code>markSupported</code> returns <code>true</code>, the stream somehow
208     * remembers all the bytes read after the call to mark and stands ready to
209     * supply those same bytes again if and whenever the method
210     * <code>reset</code> is called. However, the stream is not required to
211     * remember any data at all if more than <code>readlimit</code> bytes are
212     * read from the stream before <code>reset</code> is called.
213     *
214     * @param readlimit  The maximum limit of bytes that can be read before the
215     *                   mark position becomes invalid.
216    **/
217   public void mark(int readlimit) { in.mark(readlimit * 2); }
218 
219 
220   /**
221     * Repositions this stream to the position at the time the
222     * <code>mark</code> method was last called on this input stream.
223    **/
224   public void reset() throws IOException { in.reset(); }
225 
226 
227   /**
228     * Tests if this input stream supports the mark and reset methods.  Whether
229     * or not <code>mark</code> and <code>reset</code> are supported is an
230     * invariant property of the provided input stream instance.
231     *
232     * @return <code>true</code> iff the provided input stream instance
233     *         supports the <code>mark</code> and <code>reset</code> methods.
234    **/
235   public boolean markSupported() { return in.markSupported(); }
