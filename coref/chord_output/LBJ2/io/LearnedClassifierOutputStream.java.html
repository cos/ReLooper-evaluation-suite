3   import java.io.*;

6   /**
7     * This stream assumes it will be sent characters that are not <code>"</code>
8     * (double quote), rearranging those characters by adding spacing, double
9     * quotes, addition symbols, and commas so that the result is a
10    * <code>String</code> expression list suitable for inclusion in a Java
11    * source file.  It was created for use by the LBJ2 compiler when generating
12    * the source file that represents a trained classifier.
13    *
14    * @see HexInputStream
15    * @author Nick Rizzolo
16   **/
17  public class LearnedClassifierOutputStream extends OutputStream
18  {
19    /**
20      * The <code>PrintStream</code> to which converted output should be sent.
21     **/
22    private PrintStream out;
23    /** Internal buffer for data not yet written. */
24    private StringBuffer buffer;
25    /** Keeps track of how many lines have been written by this stream. */
26    private int lines;
27  
28  
29    /**
30      * Initializes this stream with a <code>PrintStream</code>.
31      *
32      * @param o  The <code>PrintStream</code> to which converted output should
33      *           be sent.
34     **/
35    public LearnedClassifierOutputStream(PrintStream o)
36    {
37      out = o;
38      buffer = new StringBuffer();
39      lines = 0;
40    }
41  
42  
43    /**
44      * All writes go through this method which interacts with the buffer,
45      * deciding what bytes to write when.
46     **/
47    private void writeIfFull()
48    {
49      while (buffer.length() >= 70)
50      {
51        String line = buffer.substring(0, 70);
52        buffer.delete(0, 70);
53  
54        char op = '+';
55        if (++lines == 1) op = ' ';
56        else if (lines % 936 == 1) op = ',';
57  
58        out.println("    " + op + " \"" + line + "\"");
59      }
60    }
61  
62  
63    /**
64      * Writes the specified byte to this output stream.  The general contract
65      * for <code>write</code> is that one byte is written to the output stream.
66      * The byte to be written is the eight low-order bits of the argument
67      * <code>b</code>.  The 24 high-order bits of <code>b</code> are ignored.
68      *
69      * @param b  The byte to be written.
70     **/
71    public void write(int b) throws IOException
72    {
73      buffer.append((char) b);
74      writeIfFull();
75    }
76  
77  
78    /**
79      * Writes <code>b.length</code> bytes from the specified byte array to this
80      * output stream.  The general contract for <code>write(b)</code> is that
81      * it should have exactly the same effect as the call <code>write(b, 0,
82      * b.length)</code>.
83      *
84      * @param b  The bytes to be written.
85     **/
86    public void write(byte[] b) throws IOException
87    {
88      for (int i = 0; i < b.length; ++i) buffer.append((char) b[i]);
89      writeIfFull();
90    }
91  
92  
93    /**
94      * Writes <code>len</code> bytes from the specified byte array starting at
95      * offset <code>off</code> to this output stream.  The general contract for
96      * <code>write(b, off, len)</code> is that some of the bytes in the array
97      * <code>b</code> are written to the output stream in order; element
98      * <code>b[off]</code> is the first byte written and
99      * <code>b[off+len-1]</code> is the last byte written by this operation.
100     * <br><br>
101     *
102     * If <code>b</code> is <code>null</code>, a
103     * <code>NullPointerException</code> is thrown. <br><br>
104     *
105     * If <code>off</code> is negative, or <code>len</code> is negative, or
106     * <code>off+len</code> is greater than the length of the array
107     * <code>b</code>, then an <code>IndexOutOfBoundsException</code> is
108     * thrown.
109     *
110     * @param b    A buffer containing the bytes to be written.
111     * @param off  The offset of the first byte to be written.
112     * @param len  The amount of bytes to be written.
113    **/
114   public void write(byte[] b, int off, int len) throws IOException
115   {
116     for (int i = off; i < off + len; ++i) buffer.append((char) b[i]);
117     writeIfFull();
118   }
119 
120 
121   /**
122     * Flushes this output stream and forces any buffered output bytes to be
123     * written out.  The general contract of <code>flush</code> is that calling
124     * it is an indication that, if any bytes previously written have been
125     * buffered by the implementation of the output stream, such bytes should
126     * immediately be written to their intended destination.
127    **/
128   public void flush() throws IOException
129   {
130     writeIfFull();
131 
132     if (buffer.length() > 0)
133     {
134       char op = '+';
135       if (++lines == 1) op = ' ';
136       else if (lines % 936 == 1) op = ',';
137 
138       out.println("    " + op + " \"" + buffer + "\"");
139 
140       buffer.delete(0, buffer.length());
141     }
142 
143     out.flush();
144   }
145 
146 
147   /**
148     * Closes this output stream and releases any system resources associated
149     * with this stream.  The general contract of <code>close</code> is that it
150     * closes the output stream.  A closed stream cannot perform output
151     * operations and cannot be reopened.
152    **/
153   public void close() throws IOException { flush(); }
154 }
