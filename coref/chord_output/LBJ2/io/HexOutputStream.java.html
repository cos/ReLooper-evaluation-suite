6   /**
7     * This class will convert whatever data is sent to it into little endian,
8     * hexidecimal text and send that text on to another
9     * <code>OutputStream</code>.  The most common usage of this class will
10    * involve passing it to the constructor of another
11    * <code>OutputStream</code>.  For instance: <br><br>
12    *
13    * <pre>
14    *   ObjectOutputStream oos =
15    *     new ObjectOutputStream(
16    *       new GZIPOutputStream(
17    *         new HexOutputStream(new FileOutputStream(fileName))));
18    * </pre>
19    *
20    * @see HexInputStream
21    * @author Nick Rizzolo
22   **/
23  public class HexOutputStream extends OutputStream
24  {
25    /** Characters representing a hexidecimal digit. */
26    private static final String digits = "0123456789ABCDEF";
27  
28    /**
29      * The <code>OutputStream</code> to which converted output should be sent.
30     **/
31    private OutputStream out;
32  
33  
34    /**
35      * Initializes this stream with another output stream.
36      *
37      * @param o  The output stream to which converted output should be sent.
38     **/
39    public HexOutputStream(OutputStream o) { out = o; }
40  
41  
42    /**
43      * Writes the specified byte to this output stream.  The general contract
44      * for <code>write</code> is that one byte is written to the output stream.
45      * The byte to be written is the eight low-order bits of the argument
46      * <code>b</code>.  The 24 high-order bits of <code>b</code> are ignored.
47      *
48      * @param b  The byte to be written.
49     **/
50    public void write(int b) throws IOException
51    {
52      b &= 255;
53      out.write(digits.charAt(b & 15));
54      out.write(digits.charAt((b & 240) >> 4));
55    }
56  
57  
58    /**
59      * Writes <code>b.length</code> bytes from the specified byte array to this
60      * output stream.  The general contract for <code>write(b)</code> is that
61      * it should have exactly the same effect as the call <code>write(b, 0,
62      * b.length)</code>.
63      *
64      * @param b  The bytes to be written.
65     **/
66    public void write(byte[] b) throws IOException
67    {
68      write(b, 0, b.length);
69    }
70  
71  
72    /**
73      * Writes <code>len</code> bytes from the specified byte array starting at
74      * offset <code>off</code> to this output stream.  The general contract for
75      * <code>write(b, off, len)</code> is that some of the bytes in the array
76      * <code>b</code> are written to the output stream in order; element
77      * <code>b[off]</code> is the first byte written and
78      * <code>b[off+len-1]</code> is the last byte written by this operation.
79      * <br><br>
80      *
81      * If <code>b</code> is <code>null</code>, a
82      * <code>NullPointerException</code> is thrown. <br><br>
83      *
84      * If <code>off</code> is negative, or <code>len</code> is negative, or
85      * <code>off+len</code> is greater than the length of the array
86      * <code>b</code>, then an <code>IndexOutOfBoundsException</code> is
87      * thrown.
88      *
89      * @param b    A buffer containing the bytes to be written.
90      * @param off  The offset of the first byte to be written.
91      * @param len  The amount of bytes to be written.
92     **/
93    public void write(byte[] b, int off, int len) throws IOException
94    {
95      byte[] hex = new byte[2 * len];
96      for (int i = 0; i < len; ++i)
97      {
98        hex[2 * i] = (byte) digits.charAt(b[off + i] & 15);
99        hex[2 * i + 1] = (byte) digits.charAt((b[off + i] & 240) >> 4);
100     }
101 
102     out.write(hex);
103   }
104 
105 
106   /**
107     * Flushes this output stream and forces any buffered output bytes to be
108     * written out.  The general contract of <code>flush</code> is that calling
109     * it is an indication that, if any bytes previously written have been
110     * buffered by the implementation of the output stream, such bytes should
111     * immediately be written to their intended destination.
112    **/
113   public void flush() throws IOException { out.flush(); }
114 
115 
116   /**
117     * Closes this output stream and releases any system resources associated
118     * with this stream.  The general contract of <code>close</code> is that it
119     * closes the output stream.  A closed stream cannot perform output
120     * operations and cannot be reopened.
121    **/
122   public void close() throws IOException { out.close(); }
123 }
