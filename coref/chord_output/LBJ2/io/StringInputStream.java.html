3   import java.io.*;

6   /**
7     * This class allows an application to create an input stream in which the
8     * bytes read are supplied by the contents of a string.
9     *
10    * <p>
11    * Only strings that contain only printable, ASCII characters will be
12    * handled correctly by this class.  If the string contains any other
13    * characters, the behavior of this class is undefined.
14    *
15    * @author Nick Rizzolo
16   **/
17  public class StringInputStream extends InputStream
18  {
19    /** The string from which bytes are read. */
20    protected String buffer;
21  
22    /** The index of the next character to read from the input stream buffer. */
23    protected int position;
24  
25    /** The number of valid characters in the input stream buffer. */
26    protected int size;
27  
28  
29    /**
30      * Creates a <code>StringInputStream</code> to read data from the specified
31      * string.
32      *
33      * @param s  The underlying input buffer.
34     **/
35    public StringInputStream(String s)
36    {
37      buffer = s;
38      position = 0;
39      size = s.length();
40    }
41  
42  
43    /**
44      * Reads the next byte of data from this input stream.  The value of the
45      * next <code>char</code> in the string is returned as an <code>int</code>
46      * in the range <code>0</code> to <code>255</code>.  If no
47      * <code>char</code> is available because the end of the string has been
48      * reached, the value <code>-1</code> is returned.
49      *
50      * <p>
51      * The <code>read</code> method of <code>StringInputStream</code> will not
52      * block.  It simply converts the next character available to an
53      * <code>int</code> and returns that <code>int</code>.
54      *
55      * @return The next byte of data, or <code>-1</code> if the end of the
56      *         stream is reached.
57     **/
58    public int read()
59    {
60      return (position < size) ? (int) buffer.charAt(position++) : -1;
61    }
62  
63  
64    /**
65      * Reads up to <code>len char</code>s from this input stream into an array
66      * of bytes.
67      *
68      * <p>
69      * The <code>read</code> method of <code>StringInputStream</code> cannot
70      * block.  It simply casts the characters in this input stream's buffer
71      * into <code>byte</code>s, storing them in the byte array argument.
72      *
73      * @param b    The buffer into which the data is read.
74      * @param off  The first index in <code>b</code> in which bytes should be
75      *             stored.
76      * @param len  The maximum number of bytes read.
77      * @return     The total number of bytes read into the buffer, or
78      *             <code>-1</code> if there is no more data because the end of
79      *             the stream has been reached.
80     **/
81    public int read(byte b[], int off, int len)
82    {
83      if (position >= size) return -1;
84      if (len < 1) return 0;
85  
86      if (position + len > size) len = size - position;
87      int result = len;
88      for (; len > 0; --len) b[off++] = (byte) buffer.charAt(position++);
89      return result;
90    }
91  
92  
93    /**
94      * Skips <code>n</code> bytes of input from this input stream.  Fewer bytes
95      * might be skipped if the end of the input stream is reached.
96      *
97      * @param n  The number of bytes to be skipped.
98      * @return   The actual number of bytes skipped.
99     **/
100   public long skip(long n)
101   {
102     if (n <= 0) return 0;
103     if (n > size - position) n = size - position;
104     position += n;
105     return n;
106   }
107 
108 
109   /**
110     * Returns the number of bytes that can be read from the input stream
111     * without blocking.
112     *
113     * @return The value of <code>size - position</code>, which is the number
114     *         of bytes remaining to be read from the input buffer.
115    **/
116   public int available() { return size - position; }
117 
118 
119   /**
120     * Resets the input stream to begin reading from the first character of
121     * this input stream's underlying buffer.
122    **/
123   public synchronized void reset() { position = 0; }
124 }
