6   /**
7     * Simply turns each score <i>s</i> in the {@link ScoreSet} returned by the
8     * specified {@link Normalizer} into <i>log(s)</i>.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class Log extends Normalizer
13  {
14    /** This normalizer runs before applying the <i>log</i> function. */
15    protected Normalizer first;
16  
17  
18    /** This constructor provided for use by the LBJ compiler only. */
19    public Log() { }
20  
21    /**
22      * Initializing constructor.
23      *
24      * @param n  This normalizer runs before applying the <i>log</i> function.
25     **/
26    public Log(Normalizer n) { first = n; }
27  
28  
29    /**
30      * Normalizes the given <code>ScoreSet</code>; its scores are modified in
31      * place before it is returned.
32      *
33      * @param scores The set of scores to normalize.
34      * @return       The normalized set of scores.
35     **/
36    public ScoreSet normalize(ScoreSet scores)
37    {
38      Score[] array = first.normalize(scores).toArray();
39      for (int i = 0; i < array.length; ++i)
40        array[i].score = Math.log(array[i].score);
41      return scores;
42    }
43  }
