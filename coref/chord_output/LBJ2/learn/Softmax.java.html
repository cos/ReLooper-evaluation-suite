6   /**
7     * The softmax normalization function replaces each score with the fraction
8     * of its exponential out of the sum of all scores' exponentials.  In other
9     * words, each score <code>x<sub>i</sub></code> is replaced by
10    * <code>exp(x<sub>i</sub>) / sum<sub>j</sub> exp(x<sub>j</sub>)</code>.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class Softmax extends Normalizer
15  {
16    /**
17      * Normalizes the given <code>ScoreSet</code>; its scores are modified in
18      * place before it is returned.
19      *
20      * @param scores The set of scores to normalize.
21      * @return       The normalized set of scores.
22     **/
23    public ScoreSet normalize(ScoreSet scores)
24    {
25      Score[] array = scores.toArray();
26      double sum = 0;
27      for (int i = 0; i < array.length; ++i)
28      {
29        array[i].score = Math.exp(array[i].score);
30        sum += array[i].score;
31      }
32  
33      for (int i = 0; i < array.length; ++i) array[i].score /= sum;
34      return scores;
35    }
36  }
