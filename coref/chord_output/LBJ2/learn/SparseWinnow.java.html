6   /**
7     * Simple sparse Winnow implementation.  It is assumed that
8     * {@link Learner#labeler} is a single discrete classifier whose returned
9     * feature values are available through the
10    * {@link LBJ2.classify.Classifier#allowableValues()} method.  The second
11    * value returned from {@link LBJ2.classify.Classifier#allowableValues()} is
12    * treated as "positive", and it is assumed there are exactly 2 allowable
13    * values.  Assertions will produce error messages if these assumptions do
14    * not hold.
15    *
16    * <p> This algorithm's user-configurable parameters are stored in member
17    * fields of this class.  They may be set via either a constructor that names
18    * each parameter explicitly or a constructor that takes an instance of
19    * {@link LBJ2.learn.SparseWinnow.Parameters Parameters} as input.  The
20    * documentation in each member field in this class indicates the default
21    * value of the associated parameter when using the former type of
22    * constructor.  The documentation of the associated member field in the
23    * {@link LBJ2.learn.SparseWinnow.Parameters Parameters} class indicates the
24    * default value of the parameter when using the latter type of constructor.
25    *
26    * @author Nick Rizzolo
27   **/

313   /**
314     * This method performs an update <code>w_i *= base<sup>x_i</sup></code>,
315     * initalizing weights in the weight vector as needed.
316     *
317     * @param example  The example object.
318     * @param base     As described above.
319    **/
320   public void update(Object example, double base)
321   {
322     SparseWeightVector.WeightIterator wI =
323       weightVector.weightIterator(extractor.classify(example));
324 
325     while (wI.hasNext())
326     {
327       wI.next();
328       Double dw = wI.getWeight();
329       double w = dw == null ? initialWeight : dw.doubleValue();
330       double s = wI.getCurrentFeatureStrength();
331 
332       double multiplier = base;
333       if (s == 0) multiplier = 1;
334       else if (s != 1) multiplier = Math.pow(base, s);
335 
336       wI.setWeight(w * multiplier, initialWeight);
337     }
338 
339     bias *= base;
340   }

343   /**
344     * Writes the algorithm's internal representation as text.  In the first
345     * line of output, the name of the classifier is printed, followed by
346     * {@link #alpha}, {@link #beta},
347     * {@link LinearThresholdUnit#initialWeight},
348     * {@link LinearThresholdUnit#threshold},
349     * {@link LinearThresholdUnit#positiveThickness},
350     * {@link LinearThresholdUnit#negativeThickness}, and finally
351     * {@link LinearThresholdUnit#bias}.
352     *
353     * @param out  The output stream.
354    **/
355   public void write(PrintStream out)
356   {
357     out.println(name + ": " + alpha + ", " + beta + ", " + initialWeight
358                 + ", " + threshold + ", " + positiveThickness + ", "
359                 + negativeThickness + ", " + bias);
360     weightVector.write(out);
361   }
