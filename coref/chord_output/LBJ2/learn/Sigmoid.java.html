6   /**
7     * The sigmoid normalization function replaces each score
8     * <code>x<sub>i</sub></code> with
9     * <code>1 / (1 + exp(-alpha x<sub>i</sub>))</code>, where <code>alpha</code>
10    * is a user-specified constant.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class Sigmoid extends Normalizer
15  {
16    /** The user-specified constant described above. */
17    protected double alpha;
18  
19  
20    /** Default constructor; sets {@link #alpha} to 1. */
21    public Sigmoid() { this(1); }
22  
23    /**
24      * Initializing constructor.
25      *
26      * @param a  The setting for {@link #alpha}.
27     **/
28    public Sigmoid(double a) { alpha = a; }
29  
30  
31    /**
32      * Normalizes the given <code>ScoreSet</code>; its scores are modified in
33      * place before it is returned.
34      *
35      * @param scores The set of scores to normalize.
36      * @return       The normalized set of scores.
37     **/
38    public ScoreSet normalize(ScoreSet scores)
39    {
40      Score[] array = scores.toArray();
41      for (int i = 0; i < array.length; ++i)
42        array[i].score = 1 / (1 + Math.exp(-alpha * array[i].score));
43      return scores;
44    }
45  }
