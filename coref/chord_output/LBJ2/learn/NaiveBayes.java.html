9   /**
10    * Naive Bayes is a multi-class learner that uses prediction value counts and
11    * feature counts given a particular prediction value to select the most
12    * likely prediction value.  More precisely, a score <i>s<sub>v</sub></i> for
13    * a given prediction value <i>v</i> is computed such that
14    * <i>e<sup>s<sub>v</sub></sup></i> is proportional to
15    *
16    * <blockquote>
17    *   <i>P(v) Prod<sub>f</sub> P(f|v)</i>
18    * </blockquote>
19    *
20    * where <i>Prod</i> is a multiplication quantifier over <i>f</i>, and
21    * <i>f</i> stands for a feature.  The value corresponding to the highest
22    * score is selected as the prediction.  Feature values that were never
23    * observed given a particular prediction value during training are smoothed
24    * with a configurable constant that defaults to <i>e<sup>-15</sup></i>.
25    *
26    * <p> This {@link Learner} learns a <code>discrete</code> classifier from
27    * other <code>discrete</code> classifiers.  <i>Features coming from
28    * <code>real</code> classifiers are ignored</i>.  It is also assumed that a
29    * single discrete label feature will be produced in association with each
30    * example object.  A feature taking one of the values observed in that label
31    * feature will be produced by the learned classifier.
32    *
33    * <p> This algorithm's user-configurable parameters are stored in member
34    * fields of this class.  They may be set via either a constructor that names
35    * each parameter explicitly or a constructor that takes an instance of
36    * {@link LBJ2.learn.NaiveBayes.Parameters Parameters} as input.  The
37    * documentation in each member field in this class indicates the default
38    * value of the associated parameter when using the former type of
39    * constructor.  The documentation of the associated member field in the
40    * {@link LBJ2.learn.NaiveBayes.Parameters Parameters} class indicates the
41    * default value of the parameter when using the latter type of constructor.
42    *
43    * @see    NaiveBayesVector
44    * @author Nick Rizzolo
45   **/
46  public class NaiveBayes extends Learner

153     *
154     * @param example  An example of the desired learned classifier's behavior.
155    **/
156   public void learn(Object example)
157   {
158     String labelValue =
159       ((DiscreteFeature) labeler.classify(example).firstFeature()).getValue();
160     NaiveBayesVector labelVector = (NaiveBayesVector) network.get(labelValue);
161 
162     if (labelVector == null)
163     {
164       labelVector = new NaiveBayesVector();
165       network.put(labelValue, labelVector);
166     }
167 
168     labelVector.scaledAdd(extractor.classify(example));
169   }

176   /**
177     * The scores in the returned {@link ScoreSet} are the posterior
178     * probabilities of each possible label given the example.
179     *
180     * @param example  The object to make decisions about.
181     * @return         A set of scores indicating the degree to which each
182     *                 possible discrete classification value is associated
183     *                 with the given example object.
184    **/
185   public ScoreSet scores(Object example)
186   {
187     ScoreSet s = new ScoreSet();
188 
189     for (Iterator I = network.entrySet().iterator(); I.hasNext(); )
190     {
191       Map.Entry e = (Map.Entry) I.next();
192       NaiveBayesVector vector = (NaiveBayesVector) e.getValue();
193       double score = vector.dot(extractor.classify(example));
194       s.put((String) e.getKey(), score);
195     }
196 
197     Score[] original = s.toArray();
198     ScoreSet result = new ScoreSet();
199 
200     // This code would clearly run quicker if you computed each exp(score)
201     // ahead of time, and divided them each by their sum.  However, each score
202     // is likely to be a very negative number, so exp(score) may not be
203     // numerically stable.  Subtracting two scores, however, hopefully leaves
204     // you with a "less negative" number, so exp applied to the subtraction
205     // hopefully behaves better.
206 
207     for (int i = 0; i < original.length; ++i)
208     {
209       double score = 1;
210 
211       for (int j = 0; j < original.length; ++j)
212       {
213         if (i == j) continue;
214         score += Math.exp(original[j].score - original[i].score);
215       }
216 
217       result.put(original[i].value, 1 / score);
218     }
219 
220     return result;
221   }

224   /**
225     * Prediction value counts and feature counts given a particular prediction
226     * value are used to select the most likely prediction value.
227     *
228     * @param example  The example object.
229     * @return         A single discrete feature, set to the most likely value.
230    **/
231   public FeatureVector classify(Object example)
232   {
233     double bestScore = -Double.MAX_VALUE;
234     String bestValue = null;
235 
236     for (Iterator I = network.entrySet().iterator(); I.hasNext(); )
237     {
238       Map.Entry e = (Map.Entry) I.next();
239       NaiveBayesVector vector = (NaiveBayesVector) e.getValue();
240       double score = vector.dot(extractor.classify(example));
241 
242       if (score > bestScore)
243       {
244         bestValue = (String) e.getKey();
245         bestScore = score;
246       }
247     }
248 
249     if (bestValue == null) return new FeatureVector();
250     return
251       new FeatureVector(
252         new DiscreteFeature(containingPackage, name, bestValue,
253                             valueIndexOf(bestValue),
254                             (short) allowableValues().length));
255   }

429   /**
430     * Keeps track of all the counts associated with a given label.
431     * Features are associated with {@link NaiveBayes.Count}s.  Those not
432     * appearing in this vector are assumed to have a count of 0.  The
433     * invocation of either of the <code>scaledAdd</code> methods increments
434     * the prior count for the label.
435     *
436     * <p> {@link RealFeature}s are ignored by this vector - no counts will be
437     * associated with them.
438     *
439     * @author Nick Rizzolo
440    **/
441   protected class NaiveBayesVector extends SparseWeightVector
442   {
443     /**
444       * The prior count is the number of times either <code>scaledAdd</code>
445       * method has been called.
446      **/
447     protected Count priorCount;
448 
449 
450     /** Simply instantiates {@link SparseWeightVector#weights}. */
451     public NaiveBayesVector() { this(new HashMap()); }
452 
453     /**
454       * Simply initializes {@link SparseWeightVector#weights}.
455       *
456       * @param w  A map of weights.
457      **/
458     public NaiveBayesVector(HashMap w)
459     {
460       super(w);
461       priorCount = new Count();
462     }
463

469     public Count getPrior() { return priorCount; }

471 
472     /**
473       * Produces an iterator that accesses the data in this vector associated
474       * with the features in the given vector.

477       * @return         A weight iterator.

479     public WeightIterator weightIterator(FeatureVector example)

485     /**
486       * Takes the dot product of this vector with the given vector, using the
487       * hard coded smoothing weight.
488       *
489       * @param example  A vector of features extracted from an example object.
490       * @return         The computed dot product.
491      **/
492     public double dot(FeatureVector example)
493     {
494       return dot(example, priorCount.getLog() + smoothing);
495     }
496 
497 
498     /**
499       * Takes the dot product of this vector with the given vector,
500       * using the specified default weight when encountering a feature that is
501       * not yet present in this vector.  Here, weights are taken as
502       * <i>log(feature count / prior count)</i>.  The output of this method is
503       * related to the empirical probability of the example <i>e</i> as
504       * follows: <br><br>
505       *
506       * <i>exp(dot(e)) / (sum of all labels' prior counts)) =</i><br>
507       * <i>P(e's label && e)</i>
508       *
509       * @param example  A vector of features extracted from an example object.
510       * @param defaultW The default weight.
511       * @return         The computed dot product.
512      **/
513     public double dot(FeatureVector example, double defaultW)
514     {
515       WeightIterator I = weightIterator(example);
516       double sum = (1 - I.totalFeatures()) * priorCount.getLog();
517 
518       while (I.hasNext())
519       {
520         I.next();
521         Double dw = I.getWeight();
522         sum += dw == null ? defaultW : dw.doubleValue();
523       }
524 
525       return sum;
526     }

529     /**
530       * This method is similar to the implementation in
531       * {@link SparseWeightVector} except that the <code>factor</code>
532       * and <code>defaultW</code> arguments are ignored and
533       * {@link NaiveBayes.NaiveBayesVector.NaiveBayesIterator#incrementCount()}
534       * is called instead of
535       * {@link SparseWeightVector.WeightIterator#setWeight(double)}.
536       *
537       * @param example  A vector of features extracted from an example object.
538       * @param factor   The scaling factor.
539       * @param defaultW The default weight.
540      **/
541     public void scaledAdd(FeatureVector example, double factor,
542                           double defaultW)
543     {
544       WeightIterator I = weightIterator(example);
545       priorCount.increment();
546 
547       while (I.hasNext())
548       {
549         I.next();
550         ((NaiveBayesIterator) I).incrementCount();
551       }
552     }

555     /**
556       * Converts this <code>NaiveBayesVector</code> into a
557       * <code>String</code>.
558       *
559       * @return A <code>String</code> holding a textual representation of this
560       *         vector.
561      **/
562     public String toString()
563     {
564       Map.Entry[] entries =
565         (Map.Entry[]) weights.entrySet().toArray(new Map.Entry[0]);
566       Arrays.sort(entries,
567                   new Comparator()
568                   {
569                     public int compare(Object o1, Object o2)
570                     {
571                       Map.Entry e1 = (Map.Entry) o1;
572                       Map.Entry e2 = (Map.Entry) o2;
573                       return ((Feature) e1.getKey()).compareTo(e2.getKey());
574                     }
575                   });
576 
577       int i, biggest = "priorCount".length();
578       for (i = 0; i < entries.length; ++i)
579       {
580         String key = entries[i].getKey().toString();
581         if (key.length() > biggest) biggest = key.length();
582       }
583 
584       if (biggest % 2 == 0) biggest += 2;
585       else ++biggest;
586 
587       StringBuffer result = new StringBuffer();
588       result.append("priorCount");
589       for (i = "priorCount".length(); i < biggest; ++i) result.append(" ");
590       result.append(priorCount + "\n");
591 
592       for (i = 0; i < entries.length; ++i)
593       {
594         String key = entries[i].getKey().toString();
595         result.append(key);
596         for (int j = 0; key.length() + j < biggest; ++j) result.append(" ");
597 
598         Object weight = entries[i].getValue();
599         if (weight instanceof Count) result.append(weight + "\n");
600         else
601         {
602           Count[] w = (Count[]) weight;
603           result.append(w[0] + "\n");
604           for (int j = 1; j < w.length; ++j)
605           {
606             for (int k = 0; k < biggest; ++k) result.append(" ");
607             result.append(w[j] + "\n");
608           }
609         }
610       }
611 
612       result.append("\n");
613       return result.toString();
614     }

620       *
621       * @return A copy of this <code>NaiveBayesVector</code>.
622      **/
623     public Object clone()
624     {
625       NaiveBayesVector clone = new NaiveBayesVector();
626       clone.priorCount = (Count) priorCount.clone();
627 
628       for (Iterator I = weights.entrySet().iterator(); I.hasNext(); )

634         else clone.weights.put(e.getKey(), ((Count[]) value).clone());
635       }

641     /**
642       * This class simultaneously iterates over the features in a given vector
643       * and their corresponding weights from a
644       * {@link NaiveBayes.NaiveBayesVector}.  Member methods give read and
645       * write access to the weight that an object of this class currently
646       * points to, and changes are reflected in the
647       * {@link NaiveBayes.NaiveBayesVector}.  This iterator automatically
648       * skips {@link RealFeature}s.
649       *
650       * @author Nick Rizzolo
651      **/
652     protected class NaiveBayesIterator extends WeightIterator
653     {
654       /**
655         * Iterates through the features corresponding to the weights to be
656         * iterated over for the current composite child.
657        **/
658       protected ListIterator I;
659       /** The feature corresponding to the current weight. */
660       protected DiscreteFeature currentFeature;
661       /** The array of weights, if applicable. */
662       protected Count[] countArray;
663 
664 
665       /**
666         * This constructor selects a slice of weights from the
667         * {@link NaiveBayes.NaiveBayesVector} representing all those weights
668         * corresponding to features in the given vector.
669         *
670         * @param example  A vector of features extracted from an example
671         *                 object.
672        **/
673       public NaiveBayesIterator(FeatureVector example)
674       {
675         super(example);
676         I = vector.listIterator();
677         countArray = null;
678         skipReals();
679       }
680 
681 
682       /**
683         * Repositions this iterator to point just before the next weight
684         * corresponding to a discrete feature.  If there is no next discrete
685         * feature, this iterator will either continue to point to the same
686         * weight as before or run off the end of the vector.
687        **/
688       protected void skipReals()
689       {
690         if (!I.hasNext()) return;
691 
692         Feature f = null;
693         do { f = (Feature) I.next(); }
694         while (!(f instanceof DiscreteFeature) && I.hasNext());
695 
696         if (f instanceof DiscreteFeature) I.previous();
697       }
698 
699 
700       /**
701         * Determines whether a call to {@link #next()} will redirect this
702         * iterator to point to a different weight.
703         *
704         * @return <code>true</code> iff there are more weights in the
705         *         iteration.
706        **/
707       public boolean hasNext() { return I.hasNext(); }
708 
709 
710       /**
711         * Repositions this iterator to point to the next weight in the
712         * iteration.  If there is no next weight, this iterator will continue
713         * to point to the same weight as before.
714        **/
715       public void next()
716       {
717         if (!hasNext()) return;
718 
719         currentFeature = (DiscreteFeature) I.next();
720         if (currentFeature.totalValues() >= 2)
721         {
722           if (currentFeature.fromArray() && arrayIndex++ == -1
723               || !currentFeature.fromArray())
724           {
725             DiscreteFeature key =
726               new DiscreteFeature(currentFeature.getPackage(),
727                                   currentFeature.getIdentifier(), "");
728             countArray = (Count[]) weights.get(key);
729           }
730         }
731         else
732         {
733           arrayIndex = -1;
734           countArray = null;
735         }
736 
737         skipReals();
738       }
739 
740 
741       /** Restart the iteration. */
742       public void reset()
743       {
744         I = vector.listIterator();
745         countArray = null;
746         arrayIndex = -1;
747       }

764         * @return The strength of the current feature in the given example.

774       /**
775         * Returns the count associated with the current feature's value.
776         *
777         * @return The count associated with the current feature's value, or
778         *         <code>null</code> if the location in the weight vector
779         *         pointed to by this iterator is empty.
780        **/
781       public Integer getCount()
782       {
783         if (currentFeature.totalValues() >= 2)
784         {
785           if (countArray != null)
786           {
787             if (currentFeature.fromArray())
788               return
789                 new Integer(
790                   countArray[arrayIndex * currentFeature.totalValues()
791                              + currentFeature.getValueIndex()]
792                   .getCount());
793             return
794               new Integer(countArray[currentFeature.getValueIndex()]
795                           .getCount());
796           }
797         }
798         else
799         {
800           Count c = (Count) weights.get(currentFeature);
801           if (c == null) return null;
802           return new Integer(c.getCount());
803         }
804 
805         return null;
806       }

823               return
824                 new Double(
825                   countArray[arrayIndex * currentFeature.totalValues()
826                              + currentFeature.getValueIndex()]
827                   .getLog());

830           }
831         }

837         }

852       /**
853         * Increments the count associated with the value of the current
854         * feature.
855        **/
856       public void incrementCount()
857       {
858         if (currentFeature.totalValues() >= 2)
859         {
860           if (currentFeature.fromArray())
861           {
862             DiscreteArrayFeature daf = (DiscreteArrayFeature) currentFeature;
863             if (countArray == null)
864             {
865               countArray =
866                 new Count[daf.getArrayLength() * daf.totalValues()];
867               for (int i = 0; i < countArray.length; ++i)
868                 countArray[i] = new Count();
869 
870               DiscreteFeature key =
871                 new DiscreteFeature(daf.getPackage(), daf.getIdentifier(),
872                                     "");
873               key.intern();
874               weights.put(key, countArray);
875             }
876 
877             countArray[arrayIndex * daf.totalValues() + daf.getValueIndex()]
878               .increment();
879           }
880           else
881           {
882             if (countArray == null)
883             {
884               countArray = new Count[currentFeature.totalValues()];
885               for (int i = 0; i < countArray.length; ++i)
886                 countArray[i] = new Count();
887 
888               DiscreteFeature key =
889                 new DiscreteFeature(currentFeature.getPackage(),
890                                     currentFeature.getIdentifier(), "");
891               key.intern();
892               weights.put(key, countArray);
893             }
894 
895             countArray[currentFeature.getValueIndex()].increment();
896           }
897         }
898         else
899         {
900           Count c = (Count) weights.get(currentFeature);
901 
902           if (c == null)
903           {
904             c = new Count();
905             currentFeature.intern();
906             weights.put(currentFeature, c);
907           }
908 
909           c.increment();
910         }
911       }
