8   /**
9     * An approximation to voted Perceptron, in which a weighted average of the
10    * weight vectors arrived at during training becomes the weight vector used
11    * to make predictions after training.
12    *
13    * <p> During training, after each example <i>e<sub>i</sub></i> is processed,
14    * the weight vector <i>w<sub>i</sub></i> becomes the active weight vector
15    * used to make predictions on future training examples.  If a mistake was
16    * made on <i>e<sub>i</sub></i>, <i>w<sub>i</sub></i> will be different than
17    * <i>w<sub>i - 1</sub></i>.  Otherwise, it will remain unchanged.
18    *
19    * <p> After training, each distinct weight vector arrived at during training
20    * is associated with an integer weight equal to the number of examples whose
21    * training made that weight vector active.  A new weight vector
22    * <i>w<sup>*</sup></i> is computed by taking the average of all these weight
23    * vectors weighted as described.  <i>w<sup>*</sup></i> is used to make all
24    * predictions returned to the user through methods such as
25    * {@link Classifier#classify(Object)} or
26    * {@link Classifier#discreteValue(Object)}.
27    *
28    * <p> The above description is a useful way to think about the operation of
29    * this {@link Learner}.  However, the user should note that this
30    * implementation never explicitly stores <i>w<sup>*</sup></i>.  Instead, it
31    * is computed efficiently on demand.  Thus, interspersed online training and
32    * evaluation is efficient and operates as expected.
33    *
34    * <p> It is assumed that {@link Learner#labeler} is a single discrete
35    * classifier that produces the same feature for every example object and
36    * that the values that feature may take are available through the
37    * {@link Classifier#allowableValues()} method.  The second value returned
38    * from {@link Classifier#allowableValues()} is treated as "positive", and it
39    * is assumed there are exactly 2 allowable values.  Assertions will produce
40    * error messages if these assumptions do not hold.
41    *
42    * @author Nick Rizzolo
43   **/
44  public class SparseAveragedPerceptron extends SparsePerceptron

189   /**
190     * The score of the specified object is equal to <code>w * x + bias</code>
191     * where <code>*</code> is dot product, <code>w</code> is the weight
192     * vector, and <code>x</code> is the feature vector produced by the
193     * extractor.
194     *
195     * @param example  The example object.
196     * @return         The result of the dot product plus the bias.
197    **/
198   public double score(Object example)
199   {
200     double result =
201       weightVector.dot(extractor.classify(example), initialWeight);
202     int examples = ((AveragedWeightVector) weightVector).getExamples();
203     if (examples > 0)
204       result += (examples * bias - averagedBias) / (double) examples;
205     return result;
206   }

209   /**
210     * Scales the feature vector produced by the extractor by the learning rate
211     * and adds it to the weight vector.
212     *
213     * @param example  The example object.
214    **/
215   public void promote(Object example)
216   {
217     bias += learningRate;
218     int examples = ((AveragedWeightVector) weightVector).getExamples();
219     averagedBias += examples * learningRate;
220     weightVector.scaledAdd(extractor.classify(example), learningRate,
221                            initialWeight);
222   }

225   /**
226     * Scales the feature vector produced by the extractor by the learning rate
227     * and subtracts it from the weight vector.
228     *
229     * @param example  The example object.
230    **/
231   public void demote(Object example)
232   {
233     bias -= learningRate;
234     int examples = ((AveragedWeightVector) weightVector).getExamples();
235     averagedBias -= examples * learningRate;
236     weightVector.scaledAdd(extractor.classify(example), -learningRate,
237                            initialWeight);
238   }

241   /**
242     * This method works just like {@link SparsePerceptron#learn(Object)},
243     * except it assumes that {@link #weightVector} references a
244     * {@link SparseAveragedPerceptron.AveragedWeightVector} and it notifies
245     * that vector when it got an example correct in addition to updating it
246     * when it makes a mistake.
247     *
248     * @param example  The example object.
249    **/
250   public void learn(Object example)
251   {
252     Feature l = labeler.classify(example).firstFeature();
253     assert l != null
254       : "An LTU's label classifier must always produce the same feature.";
255     assert l instanceof DiscreteFeature
256       : "An LTU's label classifier must always produce a single discrete "
257         + "feature.";
258 
259     DiscreteFeature labelFeature = (DiscreteFeature) l;
260     assert labelFeature.valueEquals(allowableValues[0])
261            || labelFeature.valueEquals(allowableValues[1])
262       : "Example has unallowed label value.";
263 
264     boolean label = labelFeature.getValueIndex() == 1
265                     || labelFeature.getValueIndex() == -1
266                        && labelFeature.valueEquals(allowableValues[1]);
267 
268     AveragedWeightVector awv = (AveragedWeightVector) weightVector;
269     double s =
270       awv.simpleDot(extractor.classify(example), initialWeight) + bias;
271     if (label && s < threshold + positiveThickness) promote(example);
272     else if (!label && s >= threshold - negativeThickness) demote(example);
273     else ((AveragedWeightVector) weightVector).correctExample();
274   }

327     * This implementation of a sparse weight vector associates two
328     * <code>double</code>s with each {@link Feature}.  The first plays the
329     * role of the usual weight vector, and the second accumulates multiples of

331     * average.

384     /**
385       * Takes the dot product of the regular, non-averaged, Perceptron weight
386       * vector with the given vector, using the specified default weight when
387       * a feature is not yet present in this vector.
388       *
389       * @param example  A vector of features extracted from an example object.
390       * @param defaultW An initial weight for new features.
391       * @return         The computed dot product.
392      **/
393     public double simpleDot(FeatureVector example, double defaultW)
394     {
395       WeightIterator I = weightIterator(example);
396       double sum = 0;
397 
398       while (I.hasNext())
399       {
400         I.next();
401         Double dw = ((AveragedWeightIterator) I).getSimpleWeight();
402         double w = dw == null ? defaultW : dw.doubleValue();
403         sum += w * I.getCurrentFeatureStrength();
404       }
405 
406       return sum;
407     }

410     /**
411       * Performs pairwise addition of the feature values in the given vector
412       * scaled by the given factor, modifying this weight vector, using the
413       * specified default weight when a feature from the given vector is not
414       * yet present in this vector.
415       *
416       * @param example  A vector of features extracted from an example object.
417       * @param factor   The scaling factor.
418       * @param defaultW An initial weight for new features.
419      **/
420     public void scaledAdd(FeatureVector example, double factor,
421                           double defaultW)
422     {
423       WeightIterator I = weightIterator(example);
424 
425       while (I.hasNext())
426       {
427         I.next();
428         Double dw = ((AveragedWeightIterator) I).getSimpleWeight();
429         double w = dw == null ? defaultW : dw.doubleValue();
430         I.setWeight(w + I.getCurrentFeatureStrength() * factor, defaultW);
431       }
432 
433       ++examples;
434     }

443     public Object clone()
444     {
445       AveragedWeightVector clone = new AveragedWeightVector();
446 
447       for (Iterator I = weights.entrySet().iterator(); I.hasNext(); )
448       {
449         Map.Entry e = (Map.Entry) I.next();
450         double[] value = (double[]) e.getValue();
451         clone.weights.put(e.getKey(), value.clone());
452       }

455     }

478     protected class AveragedWeightIterator
479               extends SparseWeightVector.WeightIterator

481       /**

492       /**
493         * Returns the double precision value pointed to by this iterator.
494         *
495         * @return The double precision value pointed to by this iterator, or
496         *         <code>null</code> if the location in the weight vector
497         *         pointed to by this iterator is empty.
498        **/
499       public Double getSimpleWeight()
500       {
501         if (isDiscrete)
502         {
503           if (currentFeature.totalValues() >= 2)
504           {
505             if (currentFeature.fromArray())
506             {
507               if (weightArray != null)
508               {
509                 if (currentFeature.totalValues() == 2)
510                   return new Double(weightArray[2 * arrayIndex]);
511                 else
512                 {
513                   DiscreteFeature df = (DiscreteFeature) currentFeature;
514                   return
515                     new Double(weightArray[2 * (arrayIndex * df.totalValues()
516                                                 + df.getValueIndex())]);
517                 }
518               }
519             }
520             else if (currentFeature.totalValues() == 2)
521             {
522               DiscreteFeature key =
523                 new DiscreteFeature(currentFeature.getPackage(),
524                                     currentFeature.getIdentifier(), "");
525               double[] w = (double[]) weights.get(key);
526               if (w != null) return new Double(w[0]);
527             }
528             else if (weightArray != null)
529             {
530               DiscreteFeature df = (DiscreteFeature) currentFeature;
531               return new Double(weightArray[2 * df.getValueIndex()]);
532             }
533           }
534           else
535           {
536             double[] w = (double[]) weights.get(currentFeature);
537             if (w != null) return new Double(w[0]);
538           }
539         }
540         else if (currentFeature.fromArray())
541         {
542           if (weightArray != null)
543             return new Double(weightArray[2 * arrayIndex]);
544         }
545         else
546         {
547           RealFeature key =
548             new RealFeature(currentFeature.getPackage(),
549                             currentFeature.getIdentifier(), 0);
550           double[] w = (double[]) weights.get(key);
551           if (w != null) return new Double(w[0]);
552         }
553 
554         return null;
555       }

558       /**
559         * Returns the weighted average value pointed to by this iterator.
560         *
561         * @return The weighted average value pointed to by this iterator, or
562         *         <code>null</code> if the location in the weight vector
563         *         pointed to by this iterator is empty.
564        **/
565       public Double getWeight()
566       {
567         if (isDiscrete)
568         {
569           if (currentFeature.totalValues() >= 2)
570           {
571             if (currentFeature.fromArray())
572             {
573               if (weightArray != null)
574               {
575                 int i = 0;
576 
577                 if (currentFeature.totalValues() == 2) i = 2 * arrayIndex;
578                 else
579                 {
580                   DiscreteFeature df = (DiscreteFeature) currentFeature;
581                   i = 2 * (arrayIndex * df.totalValues()
582                            + df.getValueIndex());
583                 }
584 
585                 return
586                   new Double((examples * weightArray[i] - weightArray[i + 1])
587                              / (double) examples);
588               }
589             }
590             else if (currentFeature.totalValues() == 2)
591             {
592               DiscreteFeature key =
593                 new DiscreteFeature(currentFeature.getPackage(),
594                                     currentFeature.getIdentifier(), "");
595               double[] w = (double[]) weights.get(key);
596               if (w != null)
597                 return
598                   new Double((examples * w[0] - w[1]) / (double) examples);
599             }
600             else if (weightArray != null)
601             {
602               DiscreteFeature df = (DiscreteFeature) currentFeature;
603               int i = 2 * df.getValueIndex();
604               return
605                 new Double((examples * weightArray[i] - weightArray[i + 1])
606                            / (double) examples);
607             }
608           }
609           else
610           {
611             double[] w = (double[]) weights.get(currentFeature);
612             if (w != null)
613               return new Double((examples * w[0] - w[1]) / (double) examples);
614           }
615         }
616         else if (currentFeature.fromArray())
617         {
618           if (weightArray != null)
619           {
620             int i = 2 * arrayIndex;
621             return
622               new Double((examples * weightArray[i] - weightArray[i + 1])
623                          / (double) examples);
624           }
625         }
626         else
627         {
628           RealFeature key =
629             new RealFeature(currentFeature.getPackage(),
630                             currentFeature.getIdentifier(), 0);
631           double[] w = (double[]) weights.get(key);
632           if (w != null)
633             return new Double((examples * w[0] - w[1]) / (double) examples);
634         }
635 
636         return null;
637       }

650       /**
651         * Modifies the weight vector.
652         *
653         * @param w  The new value for the weight pointed to by this iterator.
654         * @param d  The default value for other weights incidentally created
655         *           by this invocation.
656        **/
657       public void setWeight(double w, double d)
658       {
659         if (isDiscrete)
660         {
661           if (currentFeature.totalValues() >= 2)
662           {
663             if (currentFeature.fromArray())
664             {
665               if (currentFeature.totalValues() == 2)
666               {
667                 if (weightArray == null)
668                 {
669                   DiscreteArrayFeature daf =
670                     (DiscreteArrayFeature) currentFeature;
671                   weightArray = new double[2 * daf.getArrayLength()];
672                   DiscreteFeature key =
673                     new DiscreteFeature(daf.getPackage(), daf.getIdentifier(),
674                                         "");
675                   key.intern();
676                   weights.put(key, weightArray);
677 
678                   if (d != 0)
679                     for (int i = 0; i < weightArray.length; i += 2)
680                       weightArray[i] = d;
681                 }
682 
683                 int i = 2 * arrayIndex;
684                 double difference = w - weightArray[i];
685                 weightArray[i] = w;
686                 weightArray[i + 1] += examples * difference;
687               }
688               else
689               {
690                 DiscreteArrayFeature daf =
691                   (DiscreteArrayFeature) currentFeature;
692                 if (weightArray == null)
693                 {
694                   weightArray =
695                     new double[2 * daf.getArrayLength() * daf.totalValues()];
696                   DiscreteFeature key =
697                     new DiscreteFeature(daf.getPackage(), daf.getIdentifier(),
698                                         "");
699                   key.intern();
700                   weights.put(key, weightArray);
701 
702                   if (d != 0)
703                     for (int i = 0; i < weightArray.length; i += 2)
704                       weightArray[i] = d;
705                 }
706 
707                 int i = 2 * (arrayIndex * daf.totalValues()
708                              + daf.getValueIndex());
709                 double difference = w - weightArray[i];
710                 weightArray[i] = w;
711                 weightArray[i + 1] += examples * difference;
712               }
713             }
714             else if (currentFeature.totalValues() == 2)
715             {
716               DiscreteFeature key =
717                 new DiscreteFeature(currentFeature.getPackage(),
718                                     currentFeature.getIdentifier(), "");
719               double[] a = (double[]) weights.get(key);
720 
721               if (a == null)
722               {
723                 a = new double[2];
724                 key.intern();
725                 weights.put(key, a);
726                 a[0] = d;
727               }
728 
729               double difference = w - a[0];
730               a[0] = w;
731               a[1] += examples * difference;
732             }
733             else
734             {
735               if (weightArray == null)
736               {
