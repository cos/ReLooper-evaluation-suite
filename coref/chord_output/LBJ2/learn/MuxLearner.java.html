8   /**
9     * A <code>MuxLearner</code> uses one of many <code>Learner</code>s indexed
10    * by the first feature in an example to produce a classification.  During
11    * training, the features produced by the first child classifier of this
12    * classifier's composite generator feature extractor are taken to determine
13    * which <code>Learner</code>s will learn from the training object.  There
14    * must be one <code>Feature</code> produced by the labeler for each
15    * <code>Feature</code> produced by the first child classifier.  If this
16    * classifier's feature extractor is not a composite generator, the first
17    * feature it produces is the only one taken.
18    *
19    * <p> It is assumed that the <code>Learner</code> being multiplexed expects
20    * a single label feature on each training example, and that the feature(s)
21    * used to do the multiplexing are <code>DiscreteFeature</code>(s).
22    * Furthermore, if this classifier's feature extractor is a composite
23    * generator, it must produce the same number of features as this
24    * classifier's labeler, and they must correspond to each other in the order
25    * produced.
26    *
27    * <p> This algorithm's user-configurable parameters are stored in member
28    * fields of this class.  They may be set via either a constructor that names
29    * each parameter explicitly or a constructor that takes an instance of
30    * {@link LBJ2.learn.MuxLearner.Parameters Parameters} as input.  The
31    * documentation in each member field in this class indicates the default
32    * value of the associated parameter when using the former type of
33    * constructor.  The documentation of the associated member field in the
34    * {@link LBJ2.learn.MuxLearner.Parameters Parameters} class indicates the
35    * default value of the parameter when using the latter type of constructor.
36    *
37    * @author Nick Rizzolo
38   **/
39  public class MuxLearner extends Learner
40  {

43    /** Default for {@link #defaultPrediction}. */
44    public static final String defaultDefaultPrediction = null;

186   public void setExtractor(Classifier e)
187   {
188     super.setExtractor(e);
189 
190     try
191     {
192       compositeChildren = extractor.getCompositeChildren();
193       select = (Classifier) compositeChildren.removeFirst();
194     }
195     catch (UnsupportedOperationException ex)
196     {
197       compositeChildren = null;
198       select = null;
199     }

203   /**
204     * The training example is multiplexed to the appropriate
205     * <code>Learner</code>(s).
206     *
207     * @param example  The example object.
208    **/
209   public void learn(Object example)
210   {
211     FeatureVector labels = labeler.classify(example);
212     FeatureVector selections = null;
213     FeatureVector vector = null;
214 
215     if (select != null)
216     {
217       selections = select.classify(example);
218       vector = new FeatureVector();
219       for (Iterator I = compositeChildren.iterator(); I.hasNext(); )
220         vector.addFeatures(((Classifier) I.next()).classify(example));
221 
222       assert selections.size() == labels.size()
223            : "MuxLearner ERROR: Learner selections and labels have differing "
224              + "sizes: " + labels + ", " + selections;
225     }
226     else
227     {
228       vector = (FeatureVector) extractor.classify(example).clone();
229       selections = new FeatureVector((Feature) vector.features.removeFirst());
230       for (int i = 1; i < labels.size(); ++i)
231         selections.addFeature((Feature) vector.features.removeFirst());
232     }
233 
234     Iterator L = labels.iterator();
235     for (Iterator I = selections.iterator(); I.hasNext(); )
236     {
237       vector.addLabel((Feature) L.next());
238 
239       DiscreteFeature df = (DiscreteFeature) I.next();
240       Learner l = network.get(df.getValue());
241 
242       if (l == null)
243       {
244         l = (Learner) baseLearner.clone();
245         network.put(df.getValue(), l);
246       }
247 
248       l.learn(vector);
249 
250       vector = (FeatureVector) vector.clone();
251       vector.labels.clear();
252     }
253   }

260   /**
261     * Produces a set of scores indicating the degree to which each possible
262     * discrete classification value is associated with the given example
263     * object.  These scores are just the scores produced by the multiplexed
264     * <code>Learner</code>'s <code>scores(Object)</code> method.
265     *
266     * @see   Learner#scores(Object)
267     * @param example  The example object.
268    **/
269   public ScoreSet scores(Object example)
270   {
271     String selection = null;
272     FeatureVector vector = null;
273 
274     if (select != null)
275     {
276       selection =
277         ((DiscreteFeature) select.classify(example).firstFeature())
278         .getValue();
279       vector = new FeatureVector();
280       for (Iterator I = compositeChildren.iterator(); I.hasNext(); )
281         vector.addFeatures(((Classifier) I.next()).classify(example));
282     }
283     else
284     {
285       vector = (FeatureVector) extractor.classify(example).clone();
286       selection =
287         ((DiscreteFeature) vector.features.removeFirst()).getValue();
288     }
289 
290     Learner l = network.get(selection);
291 
292     if (l == null)
293       return
294         new ScoreSet(new String[]{ defaultPrediction }, new double[]{ 1 });
295 
296     return l.scores(vector);
297   }

300   /**
301     * This method performs the multiplexing and returns the output of the
302     * selected <code>Learner</code>.
303     *
304     * @param example  The example object.
305     * @return         The output of the selected <code>Learner</code>.
306    **/
307   public FeatureVector classify(Object example)
308   {
309     String selection = null;
310     FeatureVector vector = null;
311 
312     if (select != null)
313     {
314       selection =
315         ((DiscreteFeature) select.classify(example).firstFeature())
316         .getValue();
317       vector = new FeatureVector();
318       for (Iterator I = compositeChildren.iterator(); I.hasNext(); )
319         vector.addFeatures(((Classifier) I.next()).classify(example));
320     }
321     else
322     {
323       vector = (FeatureVector) extractor.classify(example).clone();
324       selection =
325         ((DiscreteFeature) vector.features.removeFirst()).getValue();
326     }
327 
328     Learner l = network.get(selection);
329 
330     if (l == null)
331       return
332         new FeatureVector(
333           new DiscreteFeature(containingPackage, name, defaultPrediction,
334                               valueIndexOf(defaultPrediction),
335                               (short) allowableValues().length));
336 
337     return l.classify(vector);
338   }

341   /**
342     * Writes the algorithm's internal representation as text.
343     *
344     * @param out  The output stream.
345    **/
346   public void write(PrintStream out)
347   {
348     Map.Entry<String, Learner>[] entries =
349       network.entrySet().toArray(new Map.Entry[0]);
350     Arrays.sort(entries,
351                 new Comparator<Map.Entry<String, Learner>>()
352                 {
353                   public int compare(Map.Entry<String, Learner> e1, Map.Entry<String, Learner> e2)
354                   {
355                     return e1.getKey().compareTo(e2.getKey());
356                   }
357                 });
358 
359     for (int i = 0; i < entries.length; ++i)
360     {
361       out.println("select: " + entries[i].getKey());
362       ((Learner) entries[i].getValue()).write(out);
363     }
364 
365     out.println("End of MuxLearner");
366   }

369   /** Returns a deep clone of this learning algorithm. */
370   public Object clone()
371   {
372     MuxLearner clone = null;
373     try { clone = (MuxLearner) super.clone(); }
374     catch (Exception e)
375     {
376       System.err.println("Error cloning MuxLearner: " + e);
377       e.printStackTrace();
378       System.exit(1);
379     }
380 
381     clone.baseLearner = (Learner) baseLearner.clone();
382     clone.network = new HashMap();
383     for (Map.Entry<String, Learner> e : network.entrySet())
384     {
385       clone.network.put(e.getKey(), (Learner) e.getValue().clone());
386     }
387 
388     return clone;
389   }
