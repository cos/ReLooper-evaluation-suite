8   /**
9     * This class is used as a weight vector in sparse learning algorithms.
10    * {@link Feature}s are associated with <code>Double</code>s and/or with
11    * <code>double[]</code>s representing the weights of the features they
12    * produce.  Features not appearing in the vector are assumed to have the
13    * {@link #defaultWeight}.
14    *
15    * @author Nick Rizzolo
16   **/
17  public class SparseWeightVector implements Cloneable, Serializable
18  {
19    /**
20      * When a feature appears in an example but not in this vector, it is
21      * assumed to have this weight.
22     **/
23    protected static final double defaultWeight = 0;
24

27    protected HashMap weights;

62    {
63      return dot(example, defaultWeight);

67    /**
68      * Takes the dot product of this <code>SparseWeightVector</code> with the
69      * argument vector, using the specified default weight when one is not yet
70      * present in this vector.
71      *
72      * @param example  A vector of features extracted from an example object.
73      * @param defaultW The default weight.
74      * @return         The computed dot product.
75     **/
76    public double dot(FeatureVector example, double defaultW)
77    {
78      WeightIterator I = weightIterator(example);
79      double sum = 0;
80  
81      while (I.hasNext())
82      {
83        I.next();
84        Double dw = I.getWeight();
85        double w = dw == null ? defaultW : dw.doubleValue();
86        sum += w * I.getCurrentFeatureStrength();
87      }
88  
89      return sum;
90    }

118   /**
119     * Self-modifying vector addition where the argument vector is first scaled
120     * by the given factor.
121     *
122     * @param example  A vector of features extracted from an example object.
123     * @param factor   The scaling factor.
124     * @param defaultW An initial weight for previously unseen features.
125    **/
126   public void scaledAdd(FeatureVector example, double factor, double defaultW)
127   {
128     WeightIterator I = weightIterator(example);
129 
130     while (I.hasNext())
131     {
132       I.next();
133       Double dw = I.getWeight();
134       double w = dw == null ? defaultW : dw.doubleValue();
135       I.setWeight(w + I.getCurrentFeatureStrength() * factor, defaultW);
136     }
137   }

144   /**
145     * Converts this <code>SparseWeightVector</code> into a
146     * <code>String</code>.
147     *
148     * @return A <code>String</code> holding a textual representation of this
149     *         vector.
150    **/
151   public String toString()
152   {
153     Map.Entry[] entries =
154       (Map.Entry[]) weights.entrySet().toArray(new Map.Entry[weights.size()]);
155     Arrays.sort(entries,
156                 new Comparator()
157                 {
158                   public int compare(Object o1, Object o2)
159                   {
160                     Map.Entry e1 = (Map.Entry) o1;
161                     Map.Entry e2 = (Map.Entry) o2;
162                     return ((Feature) e1.getKey()).compareTo(e2.getKey());
163                   }
164                 });
165 
166     int i, biggest = 0;
167     for (i = 0; i < entries.length; ++i)
168     {
169       String key = entries[i].getKey().toString();
170       if (key.length() > biggest) biggest = key.length();
171     }
172 
173     if (biggest % 2 == 0) biggest += 2;
174     else ++biggest;
175 
176     StringBuffer result = new StringBuffer();
177     for (i = 0; i < entries.length; ++i)
178     {
179       String key = entries[i].getKey().toString();
180       result.append(key);
181       for (int j = 0; key.length() + j < biggest; ++j) result.append(" ");
182 
183       Object weight = entries[i].getValue();
184       if (weight instanceof Double) result.append(weight + "\n");
185       else
186       {
187         double[] w = (double[]) weight;
188         result.append(w[0] + "\n");
189         for (int j = 1; j < w.length; ++j)
190         {
191           for (int k = 0; k < biggest; ++k) result.append(" ");
192           result.append(w[j] + "\n");
193         }
194       }
195     }
196 
197     result.append("\n");
198     return result.toString();
199   }

212   /**
213     * Returns a copy of this <code>SparseWeightVector</code> in which the
214     * {@link #weights} variable has been cloned deeply.
215     *
216     * @return A copy of this <code>SparseWeightVector</code>.
217    **/
218   public Object clone()
219   {
220     SparseWeightVector clone = new SparseWeightVector();
221 
222     for (Iterator I = weights.entrySet().iterator(); I.hasNext(); )
223     {
224       Map.Entry e = (Map.Entry) I.next();
225       Object value = e.getValue();
226       if (value instanceof Double) clone.weights.put(e.getKey(), value);
227       else clone.weights.put(e.getKey(), ((double[]) value).clone());
228     }
229 
230     return clone;
231   }

234   /**
235     * This class simultaneously iterates over the features in the given vector
236     * and their corresponding weights from a {@link SparseWeightVector}.
237     * Member methods give read and write access to the weight that an object
238     * of this class currently points to, and changes are reflected in the
239     * {@link SparseWeightVector}.
240     *
241     * @author Nick Rizzolo
242    **/
243   protected class WeightIterator
244   {
245     /** The feature vector to iterate through. */
246     protected FeatureVector vector;
247     /**
248       * Iterates through the features corresponding to the weights to be
249       * iterated over for the current composite child.
250      **/
251     protected Iterator I;
252     /** The feature corresponding to the current weight. */
253     protected Feature currentFeature;
254     /** <code>true</code> iff the current feature is discrete. */
255     protected boolean isDiscrete;
256     /** The array of weights, if applicable. */
257     protected double[] weightArray;
258     /** Pointer into <code>weightArray</code>. */
259     protected int arrayIndex;
260 
261 
262     /**
263       * This constructor selects a slice of weights from the
264       * {@link SparseWeightVector} representing all those weights
265       * corresponding to features in the given vector.
266       *
267       * @param example  A vector of features extracted from an example object.
268      **/
269     public WeightIterator(FeatureVector example)
270     {
271       currentFeature = null;
272       vector = example;
273       I = vector.iterator();
274       weightArray = null;
275       arrayIndex = -1;
276     }
277 
278 
279     /** Returns the total number of features to iterate through. */
280     public int totalFeatures() { return vector.size(); }
281 
282 
283     /**
284       * Determines whether a call to <code>next()</code> will redirect this
285       * iterator to point to a different weight.
286       *
287       * @return <code>true</code> iff there are more weights in the iteration.
288      **/
289     public boolean hasNext() { return I.hasNext(); }
290 
291 
292     /**
293       * Repositions this iterator to point to the next weight in the
294       * iteration.  If there is no next weight, this iterator will continue to
295       * point to the same weight as before.
296      **/
297     public void next()
298     {
299       if (!hasNext()) return;
300 
301       Feature nextFeature = (Feature) I.next();
302       if (currentFeature != null && currentFeature.fromArray()
303           && (!nextFeature.fromArray()
304               || !currentFeature.nameEquals(nextFeature)))
305       {
306         arrayIndex = -1;
307         weightArray = null;
308       }
309 
310       currentFeature = nextFeature;
311       if (currentFeature.fromArray())
312       {
313         if (arrayIndex++ == -1)
314         {
315           isDiscrete = currentFeature instanceof DiscreteFeature;
316           if (currentFeature.totalValues() > 0)
317           {
318             DiscreteFeature key =
319               new DiscreteFeature(currentFeature.getPackage(),
320                                   currentFeature.getIdentifier(), "");
321             weightArray = (double[]) weights.get(key);
322           }
323           else if (!isDiscrete)
324           {
325             RealFeature key =
326               new RealFeature(currentFeature.getPackage(),
327                               currentFeature.getIdentifier(), 0);
328             weightArray = (double[]) weights.get(key);
329           }
330         }
331       }
332       else
333       {
334         isDiscrete = currentFeature instanceof DiscreteFeature;
335         if (currentFeature.totalValues() > 2)
336         {
337           DiscreteFeature key =
338             new DiscreteFeature(currentFeature.getPackage(),
339                                 currentFeature.getIdentifier(), "");
340           weightArray = (double[]) weights.get(key);
341         }
342         else weightArray = null;
343       }
344     }

364     /**
365       * Convenience method for determining the current feature's strength in
366       * the given example.  The strength of a discrete feature is always 1,
367       * except when that feature is known to be boolean and has a value of
368       * <code>false</code>, in which case the strength is 0.  The strength of
369       * a real feature is just its value.
370       *
371       * @return The strength of the current feature in the given example.
372      **/
373     public double getCurrentFeatureStrength()
374     {
375       if (currentFeature.totalValues() == 2)
376         return ((DiscreteFeature) currentFeature).getValueIndex();
377       if (isDiscrete) return 1;
378       return ((RealFeature) currentFeature).getValue();
379     }

382     /**
383       * Returns the double precision value pointed to by this iterator.
384       *
385       * @return The double precision value pointed to by this iterator, or
386       *         <code>null</code> if the location in the weight vector pointed
387       *         to by this iterator is empty.
388      **/
389     public Double getWeight()
390     {
391       if (isDiscrete)
392       {
393         if (currentFeature.totalValues() >= 2)
394         {
395           if (currentFeature.fromArray())
396           {
397             if (weightArray != null)
398             {
399               if (currentFeature.totalValues() == 2)
400                 return new Double(weightArray[arrayIndex]);
401               else
402               {
403                 DiscreteFeature df = (DiscreteFeature) currentFeature;
404                 return new Double(weightArray[arrayIndex * df.totalValues()
405                                               + df.getValueIndex()]);
406               }
407             }
408           }
409           else if (currentFeature.totalValues() == 2)
410           {
411             DiscreteFeature key =
412               new DiscreteFeature(currentFeature.getPackage(),
413                                   currentFeature.getIdentifier(), "");
414             return (Double) weights.get(key);
415           }
416           else if (weightArray != null)
417           {
418             DiscreteFeature df = (DiscreteFeature) currentFeature;
419             return new Double(weightArray[df.getValueIndex()]);
420           }
421         }
422         else return (Double) weights.get(currentFeature);
423       }
424       else if (currentFeature.fromArray())
425       {
426         if (weightArray != null) return new Double(weightArray[arrayIndex]);
427       }
428       else
429       {
430         RealFeature key =
431           new RealFeature(currentFeature.getPackage(),
432                           currentFeature.getIdentifier(), 0);
433         return (Double) weights.get(key);
434       }
435 
436       return null;
437     }

450     /**
451       * Modifies the weight vector.
452       *
453       * @param w  The new value for the weight pointed to by this iterator.
454       * @param d  The default value for other weights incidentally created by
455       *           this invocation.
456      **/
457     public void setWeight(double w, double d)
458     {
459       if (isDiscrete)
460       {
461         if (currentFeature.totalValues() >= 2)
462         {
463           if (currentFeature.fromArray())
464           {
465             if (currentFeature.totalValues() == 2)
466             {
467               if (weightArray == null)
468               {
469                 DiscreteArrayFeature daf =
470                   (DiscreteArrayFeature) currentFeature;
471                 weightArray = new double[daf.getArrayLength()];
472                 DiscreteFeature key =
473                   new DiscreteFeature(daf.getPackage(), daf.getIdentifier(),
474                                       "");
475                 key.intern();
476                 weights.put(key, weightArray);
477 
478                 if (d != 0)
479                   for (int i = 0; i < weightArray.length; ++i)
480                     weightArray[i] = d;
481               }
482 
483               weightArray[arrayIndex] = w;
484             }
485             else
486             {
487               DiscreteArrayFeature daf =
488                 (DiscreteArrayFeature) currentFeature;
489               if (weightArray == null)
490               {
491                 weightArray =
492                   new double[daf.getArrayLength() * daf.totalValues()];
493                 DiscreteFeature key =
494                   new DiscreteFeature(daf.getPackage(), daf.getIdentifier(),
495                                       "");
496                 key.intern();
497                 weights.put(key, weightArray);
498 
499                 if (d != 0)
500                   for (int i = 0; i < weightArray.length; ++i)
501                     weightArray[i] = d;
502               }
503 
504               weightArray[arrayIndex * daf.totalValues()
505                           + daf.getValueIndex()] = w;
506             }
507           }
508           else if (currentFeature.totalValues() == 2)
509           {
510             DiscreteFeature key =
511               new DiscreteFeature(currentFeature.getPackage(),
512                                   currentFeature.getIdentifier(), "");
513             key.intern();
514             weights.put(key, new Double(w));
515           }
516           else
517           {
518             if (weightArray == null)
519             {
520               weightArray = new double[currentFeature.totalValues()];
521               DiscreteFeature key =
522                 new DiscreteFeature(currentFeature.getPackage(),
523                                     currentFeature.getIdentifier(), "");
524               key.intern();
525               weights.put(key, weightArray);
526 
527               if (d != 0)
528                 for (int i = 0; i < weightArray.length; ++i)
529                   weightArray[i] = d;
530             }
531 
532             weightArray[((DiscreteFeature) currentFeature).getValueIndex()] =
533               w;
534           }
535         }
536         else
537         {
538           currentFeature.intern();
539           weights.put(currentFeature, new Double(w));
540         }
541       }
542       else if (currentFeature.fromArray())
543       {
544         if (weightArray == null)
545         {
546           RealArrayFeature raf = (RealArrayFeature) currentFeature;
547           weightArray = new double[raf.getArrayLength()];
548           RealFeature key =
549             new RealFeature(raf.getPackage(), raf.getIdentifier(), 0);
550           key.intern();
551           weights.put(key, weightArray);
552 
553           if (d != 0)
554             for (int i = 0; i < weightArray.length; ++i)
555               weightArray[i] = d;
556         }
557 
558         weightArray[arrayIndex] = w;
559       }
560       else
561       {
562         RealFeature key =
563           new RealFeature(currentFeature.getPackage(),
564                           currentFeature.getIdentifier(), 0);
565         key.intern();
566         weights.put(key, new Double(w));
567       }
568     }
569   }
