10  /**
11    * This is the cross validation testing metric which LBJ defaults to when
12    * none is specified.  It simply counts the number of correct predictions and
13    * returns that number divided by the total number of examples which it used
14    * for testing.
15    *
16    * @author Dan Muriello
17   **/
18  public class Accuracy implements TestingMetric
19  {
20    /** Standard constructor, takes no arguments and does nothing. */
21    public Accuracy() {}
22  
23  
24    /**
25      * The <code>test</code> method is what LBJ calls during its testing stage.
26      *
27      * @param classifier The classifier whose accuracy is being measured.
28      * @param oracle     A classifier that returns the label of each example.
29      * @param parser     A parser to supply the example objects.
30     **/
31    public double test(Classifier classifier, Classifier oracle, Parser parser)
32    {
33      TestDiscrete tester =
34        TestDiscrete.testDiscrete(classifier, oracle, parser);
35      return tester.getOverallStats()[0];
36    }
37  }
