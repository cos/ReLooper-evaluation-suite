8   /**
9     * The Binary MIRA learning algorithm implementation.  This algorithm
10    * operates very similarly to {@link SparsePerceptron} with a thick
11    * separator, except the learning rate is a function of each training
12    * example's margin.  When the weight vector has made a mistake, the full
13    * {@link SparsePerceptron#learningRate} will be used.  When the weight
14    * vector did not make a mistake, {@link SparsePerceptron#learningRate} is
15    * multiplied by the following value before the update takes place.
16    *
17    * <p>
18    * <blockquote>
19    * <i>(beta/2 - y(w*x)) / ||x||<sup>2</sup></i>
20    * </blockquote>
21    *
22    * <p> In the expression above, <i>w</i> is the weight vector, <i>y</i>
23    * represents the label of the example vector <i>x</i>, <i>*</i> stands for
24    * inner product, and <i>beta</i> is a user supplied parameter.  If this
25    * expression turns out to be non-positive (i.e., if <i>y(w*x) >=
26    * beta/2</i>), no update is made for that example.
27    *
28    * <p> It is assumed that {@link Learner#labeler} is a single discrete
29    * classifier that produces the same feature for every example object and
30    * that the values that feature may take are available through the
31    * {@link Classifier#allowableValues()} method.  The second value returned
32    * from {@link Classifier#allowableValues()} is treated as "positive", and it
33    * is assumed there are exactly 2 allowable values.  Assertions will produce
34    * error messages if these assumptions do not hold.
35    *
36    * <p> This algorithm's user-configurable parameters are stored in member
37    * fields of this class.  They may be set via either a constructor that names
38    * each parameter explicitly or a constructor that takes an instance of
39    * {@link LBJ2.learn.SparsePerceptron.Parameters Parameters} as input.
40    * The documentation in each member field in this class indicates the default
41    * value of the associated parameter when using the former type of
42    * constructor.  The documentation of the associated member field in the
43    * {@link LBJ2.learn.SparsePerceptron.Parameters Parameters} class
44    * indicates the default value of the parameter when using the latter type of
45    * constructor.
46    *
47    * @author Arindam Saha
48   **/
49  public class BinaryMIRA extends SparsePerceptron {

197     /**
198      * Evaluates the example object with the
199    * {@link LinearThresholdUnit#score(Object)}, updates the
200      * learning rate, and finally updates
201    * {@link LinearThresholdUnit#weightVector} as described above.
202    *
203      * @param example The example object
204      */
205     public void learn(Object example) {
206         Feature l = labeler.classify(example).firstFeature();
207         assert l != null
208           : "An LTU's label classifier must always produce the same feature.";
209         assert l instanceof DiscreteFeature
210           : "An LTU's label classifier must always produce a single discrete "
211             + "feature.";
212 
213         DiscreteFeature labelFeature = (DiscreteFeature) l;
214         assert labelFeature.valueEquals(allowableValues[0])
215                || labelFeature.valueEquals(allowableValues[1])
216           : "Example has unallowed label value.";
217 
218         boolean label = labelFeature.getValueIndex() == 1
219                         || labelFeature.getValueIndex() == -1
220                            && labelFeature.valueEquals(allowableValues[1]);
221 
222         double s = score(example);
223         updateLearningRate(example, s, label);
224 
225         if (!nearlyEqualTo(learningRate, 0.0)) {
226             if (label) promote(example);
227             else demote(example);
228         }
229     }

232     /**
233      * Determines if <code>a</code> is nearly equal to <code>b</code> based on
234      * the value of the <code>TOLERANCE</code> member variable.
235      *
236      * @param a The first value
237      * @param b The second value
238      * @return  True if they are nearly equal, false otherwise.
239      */
240     private static boolean nearlyEqualTo(double a, double b) {
241         return -TOLERANCE < a - b && a - b < TOLERANCE;
242     }
243 
244 
245     /**
246      * Updates the {@link #learningRate} member variable.
247      *
248      * @param example The example object
249      * @param s       The score for the example object
250      * @param label   The label
251      */
252     private void updateLearningRate(Object example, double s, boolean label) {
253         double labelVal = label? 1: -1;
254 
255         double x = (beta / 2 - labelVal * s)
256                / (extractor.classify(example).L2NormSquared() + 1);
257 
258         if (x < 0) learningRate = 0;
259         else if (x < 1 || nearlyEqualTo(x, 1.0)) learningRate = x;
260         else learningRate = 1;
261     learningRate *= saveLearningRate;
262     }

265     /**
266       * Writes the algorithm's internal representation as text.  In the first
267       * line of output, the name of the classifier is printed, followed by
268       * {@link SparsePerceptron#learningRate}, {@link #beta},
269     * {@link LinearThresholdUnit#initialWeight},
270     * {@link LinearThresholdUnit#threshold},
271       * {@link LinearThresholdUnit#positiveThickness},
272     * {@link LinearThresholdUnit#negativeThickness},
273       * and finally {@link LinearThresholdUnit#bias}.
274       *
275       * @param out  The output stream.
276      **/
277     public void write(PrintStream out) {
278         out.println(name + ": " + learningRate + ", " + beta + ", "
279                 + initialWeight + ", " + threshold + ", " + positiveThickness
280                 + ", " + negativeThickness + ", " + bias);
281         weightVector.write(out);
282     }
