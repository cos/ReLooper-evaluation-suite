6   /**
7     * Simple sparse Perceptron implementation.  It is assumed that
8     * {@link Learner#labeler} is a single discrete classifier that produces the
9     * same feature for every example object and that the values that feature may
10    * take are available through the
11    * {@link LBJ2.classify.Classifier#allowableValues()} method.  The second
12    * value returned from {@link LBJ2.classify.Classifier#allowableValues()} is
13    * treated as "positive", and it is assumed there are exactly 2 allowable
14    * values.  Assertions will produce error messages if these assumptions do
15    * not hold.
16    *
17    * <p> This algorithm's user-configurable parameters are stored in member
18    * fields of this class.  They may be set via either a constructor that names
19    * each parameter explicitly or a constructor that takes an instance of
20    * {@link LBJ2.learn.SparsePerceptron.Parameters Parameters} as input.
21    * The documentation in each member field in this class indicates the default
22    * value of the associated parameter when using the former type of
23    * constructor.  The documentation of the associated member field in the
24    * {@link LBJ2.learn.SparsePerceptron.Parameters Parameters} class
25    * indicates the default value of the parameter when using the latter type of
26    * constructor.
27    *
28    * @author Nick Rizzolo
29   **/
30  public class SparsePerceptron extends LinearThresholdUnit

237   /**
238     * Scales the feature vector produced by the extractor by the learning rate
239     * and adds it to the weight vector.
240     *
241     * @param example  The example object.
242    **/
243   public void promote(Object example)
244   {
245     weightVector.scaledAdd(extractor.classify(example), learningRate,
246                            initialWeight);
247     bias += learningRate;
248   }
249 
250 
251   /**
252     * Scales the feature vector produced by the extractor by the learning rate
253     * and subtracts it from the weight vector.
254     *
255     * @param example  The example object.
256    **/
257   public void demote(Object example)
258   {
259     weightVector.scaledAdd(extractor.classify(example), -learningRate,
260                            initialWeight);
261     bias -= learningRate;
262   }

265   /**
266     * Writes the algorithm's internal representation as text.  In the first
267     * line of output, the name of the classifier is printed, followed by
268     * {@link #learningRate}, {@link LinearThresholdUnit#initialWeight},
269     * {@link LinearThresholdUnit#threshold},
270     * {@link LinearThresholdUnit#positiveThickness},
271     * {@link LinearThresholdUnit#negativeThickness}, and finally
272     * {@link LinearThresholdUnit#bias}.
273     *
274     * @param out  The output stream.
275    **/
276   public void write(PrintStream out)
277   {
278     out.println(name + ": " + learningRate + ", " + initialWeight + ", "
279                 + threshold + ", " + positiveThickness + ", "
280                 + negativeThickness + ", " + bias);
281     weightVector.write(out);
282   }

285   /** Returns a deep clone of this learning algorithm. */
286   public Object clone()
287   {
288     SparsePerceptron clone = null;
289 
290     try { clone = (SparsePerceptron) super.clone(); }
291     catch (Exception e)
292     {
293       System.err.println("Error cloning SparsePerceptron: " + e);
294       System.exit(1);
295     }
296 
297     if (weightVector != null)
298       clone.weightVector = (SparseWeightVector) weightVector.clone();
299     return clone;
300   }
