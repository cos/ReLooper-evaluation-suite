7   /**
8     * Extend this class to create a new {@link Classifier} that learns to mimic
9     * one an oracle classifier given a feature extracting classifier and example
10    * objects.
11    *
12    * @author Nick Rizzolo
13   **/
14  public abstract class Learner extends Classifier
15  {
16    /** Stores the classifier used to produce labels. */
17    protected Classifier labeler;
18    /** Stores the classifiers used to produce features. */
19    protected Classifier extractor;
20  
21  
22    /**
23      * This constructor is used by the LBJ2 compiler; it should never be called
24      * by a programmer.
25     **/
26    protected Learner() { }
27  
28    /**
29      * Initializes the name.
30      *
31      * @param n  The name of the classifier.
32     **/
33    protected Learner(String n) { super(n); }
34  
35    /**
36      * Constructor for unsupervised learning.
37      *
38      * @param n  The name of the classifier.
39      * @param e  The feature extracting classifier.
40     **/
41    protected Learner(String n, Classifier e) { this(n, null, e); }
42  
43    /**
44      * Constructor for supervised learning.
45      *
46      * @param n  The name of the classifier.
47      * @param l  The labeling classifier.
48      * @param e  The feature extracting classifier.
49     **/
50    protected Learner(String n, Classifier l, Classifier e)
51    {
52      super(n);
53      setLabeler(l);
54      setExtractor(e);
55    }
56  
57  
58    /** Returns the labeler. */
59    public Classifier getLabeler() { return labeler; }
60  
61  
62    /**
63      * Sets the labeler.
64      *
65      * @param l  A labeling classifier.
66     **/
67    public void setLabeler(Classifier l) { labeler = l; }
68  
69  
70    /** Returns the extractor. */
71    public Classifier getExtractor() { return extractor; }
72  
73  
74    /**
75      * Sets the extractor.
76      *
77      * @param e  A feature extracting classifier.
78     **/
79    public void setExtractor(Classifier e) { extractor = e; }
80  
81  
82    /**
83      * Trains the learning algorithm given an object as an example.
84      *
85      * @param example  An example of the desired learned classifier's behavior.
86     **/
87    abstract public void learn(Object example);
88  
89  
90    /**
91      * Trains the learning algorithm given many objects as examples.  This
92      * implementation simply calls {@link #learn(Object)} on each of the
93      * objects in the input array.  It should be overridden if there is a more
94      * efficient implementation.
95      *
96      * @param examples Examples of the desired learned classifier's behavior.
97     **/
98    public void learn(Object[] examples)
99    {
100     for (int i = 0; i < examples.length; ++i) learn(examples[i]);
101   }
102 
103 
104   /**
105     * Overridden by subclasses to perform any required post-processing
106     * computations after all training examples have been observed through
107     * {@link #learn(Object)} and {@link #learn(Object[])}.  By default this
108     * method does nothing.
109    **/
110   public void doneLearning() { }
111 
112 
113   /**
114     * Overridden by subclasses to reinitialize the learner to the state it
115     * started at before any learning was performed.
116    **/
117   abstract public void forget();
118 
119 
120   /**
121     * Overridden by subclasses to store a binary representation of this
122     * learner in a pre-defined location.  By default this method does nothing.
123    **/
124   public void save() { }
125 
126 
127   /**
128     * Produces a set of scores indicating the degree to which each possible
129     * discrete classification value is associated with the given example
130     * object.  Learners that return a <code>real</code> feature or more than
131     * one feature may implement this method by simply returning
132     * <code>null</code>.
133     *
134     * @param example  The object to make decisions about.
135     * @return         A set of scores indicating the degree to which each
136     *                 possible discrete classification value is associated
137     *                 with the given example object.
138    **/
139   abstract public ScoreSet scores(Object example);
140 
141 
142   /**
143     * Writes the algorithm's internal representation as text.
144     *
145     * @param out  The output stream.
146    **/
147   abstract public void write(PrintStream out);
148 }
