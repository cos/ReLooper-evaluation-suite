4   /**
5     * Use this parser in conjunction with another parser that returns
6     * <code>LinkedVector</code>s, and this parser will return their
7     * <code>LinkedChild</code>ren.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class ChildrenFromVectors implements Parser
12  {
13    /** A parser that returns <code>LinkedVector</code>s. */
14    protected Parser parser;
15    /** The next child to be returned. */
16    protected LinkedChild next;
17  
18  
19    /**
20      * Creates the parser.
21      *
22      * @param p  A parser that returns <code>LinkedVector</code>s.
23     **/
24    public ChildrenFromVectors(Parser p) { parser = p; }
25  
26  
27    /**
28      * Returns the next <code>LinkedChild</code> parsed.
29      *
30      * @return The next <code>LinkedChild</code> parsed, or <code>null</code>
31      *         if there are no more children in the stream.
32     **/
33    public Object next()
34    {
35      while (next == null)
36      {
37        LinkedVector v = (LinkedVector) parser.next();
38        if (v == null) return null;
39        next = v.get(0);
40      }
41  
42      LinkedChild result = next;
43      next = next.next;
44      return result;
45    }
46  
47  
48    /** Sets this parser back to the beginning of the raw data. */
49    public void reset()
50    {
51      parser.reset();
52      next = null;
53    }
54  }
