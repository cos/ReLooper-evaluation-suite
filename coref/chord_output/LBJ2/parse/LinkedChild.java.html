6   /**
7     * A <code>LinkedChild</code> is the child of a <code>LinkedVector</code>.
8     * <code>LinkedVector</code>s use the methods of this interface to maintain
9     * links between parent and child.
10    *
11    * @see LinkedVector
12    * @author Nick Rizzolo
13   **/
14  public abstract class LinkedChild implements Cloneable, Serializable
15  {
16    /** A link to this child's parent. */
17    public LinkedVector parent;
18    /** A link to the previous child in the parent vector. */
19    public LinkedChild previous;
20    /** A link to the next child in the parent vector. */
21    public LinkedChild next;
22    /** The offset into the raw data input file at which this child starts. */
23    public int start;
24    /** The offset into the raw data input file at which this child ends. */
25    public int end;
26  
27  
28    /** Does nothing. */
29    protected LinkedChild() { }
30  
31    /**
32      * Useful when the information that this child represents is parsed
33      * forwards.
34      *
35      * @param p  The previous child in the parent vector.
36     **/
37    protected LinkedChild(LinkedChild p) { this(p, -1, -1); }
38  
39    /**
40      * Constructor that sets the byte offsets of this child.
41      *
42      * @param s  The offset at which this child starts.
43      * @param e  The offset at which this child ends.
44     **/
45    public LinkedChild(int s, int e) { this(null, s, e); }
46  
47    /**
48      * Useful when the information that this child represents is parsed
49      * forwards.
50      *
51      * @param p  The previous child in the parent vector.
52      * @param s  The offset at which this child starts.
53      * @param e  The offset at which this child ends.
54     **/
55    public LinkedChild(LinkedChild p, int s, int e)
56    {
57      previous = p;
58      start = s;
59      end = e;
60    }
61  
62  
63    /** Returns a shallow clone of this object. */
64    public Object clone()
65    {
66      LinkedChild clone = null;
67      try { clone = (LinkedChild) super.clone(); }
68      catch (Exception e)
69      {
70        System.err.println("Problem with LinkedChild clone: " + e);
71        System.exit(1);
72      }
73  
74      return clone;
75    }
76  }
