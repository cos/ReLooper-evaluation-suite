6   /**
7     * A <code>LinkedVector</code> is used to store a vector of
8     * <code>LinkedChild</code>ren which all maintain links between each other
9     * and the parent <code>LinkedVector</code>.
10    *
11    * @see LinkedChild
12    * @author Nick Rizzolo
13   **/
14  public class LinkedVector extends LinkedChild
15  {
16    /** The linked vector is simply represented as a vector of children. */
17    protected Vector children;
18  
19  
20    /** Initializes the vector. */
21    public LinkedVector() { children = new Vector(); }
22  
23    /**
24      * Constructor for when only a single child from anywhere in this vector is
25      * available.  It is assumed that the <code>previous</code> and
26      * <code>next</code> links are filled in by every child.
27      *
28      * @param c  Any child in this vector.
29     **/
30    public LinkedVector(LinkedChild c)
31    {
32      children = new Vector();
33  
34      while (c.previous != null) c = c.previous;
35      start = c.start;
36  
37      for (; c.next != null; c = c.next)
38      {
39        c.parent = this;
40        children.add(c);
41      }
42  
43      c.parent = this;
44      children.add(c);
45      end = c.end;
46    }
47  
48    /**
49      * Useful when the information that this child represents is parsed
50      * forwards.
51      *
52      * @param p  The previous child in the parent vector.
53     **/
54    public LinkedVector(LinkedVector p)
55    {
56      super(p);
57      children = new Vector();
58    }
59  
60    /**
61      * Constructor that sets the character offsets of this vector.
62      *
63      * @param s  The offset at which this sentence starts.
64      * @param e  The offset at which this sentence ends.
65     **/
66    public LinkedVector(int s, int e)
67    {
68      super(s, e);
69      children = new Vector();
70    }
71  
72    /**
73      * Constructor for when only a single child from anywhere in this vector is
74      * available.  It is assumed that the <code>previous</code> and
75      * <code>next</code> links are filled in by every child.
76      *
77      * @param c  Any child in this vector.
78      * @param s  The offset at which this sentence starts.
79      * @param e  The offset at which this sentence ends.
80     **/
81    public LinkedVector(LinkedChild c, int s, int e)
82    {
83      super(s, e);
84      children = new Vector();
85      while (c.previous != null) c = c.previous;
86      for (; c != null; c = c.next) add(c);
87    }
88  
89    /**
90      * Useful when the information that this child represents is parsed
91      * forwards.
92      *
93      * @param p  The previous child in the parent vector.
94      * @param s  The offset at which this sentence starts.
95      * @param e  The offset at which this sentence ends.
96     **/
97    public LinkedVector(LinkedVector p, int s, int e)
98    {
99      super(p, s, e);
100     children = new Vector();
101   }
102 
103 
104   /**
105     * Adds the specified child to the end of the vector, informing the child
106     * of its parent and index and linking the child to its only neighbor
107     * (which was previously the last child in the vector).
108     *
109     * @param c  The child to add.
110    **/
111   public boolean add(LinkedChild c)
112   {
113     c.parent = this;
114     if (children.size() > 0)
115     {
116       LinkedChild p = get(children.size() - 1);
117       p.next = c;
118       c.previous = p;
119     }
120 
121     return children.add(c);
122   }
123 
124 
125   /**
126     * Removes the child at the specified index.
127     *
128     * @param i  The index of the child to remove.
129     * @return   The child removed, or <code>null</code> if there was no child
130     *           at that index.
131    **/
132   public LinkedChild remove(int i)
133   {
134     LinkedChild before =
135       (i - 1 < 0) ? null : (LinkedChild) children.get(i - 1);
136     LinkedChild after =
137       (i + 1 >= children.size()) ? null : (LinkedChild) children.get(i + 1);
138 
139     if (before != null) before.next = after;
140     if (after != null) after.previous = before;
141 
142     LinkedChild removed = null;
143     try { removed = (LinkedChild) children.remove(i); }
144     catch (ArrayIndexOutOfBoundsException e) { return null; }
145 
146     removed.parent = null;
147     removed.next = removed.previous = null;
148     return removed;
149   }
150 
151 
152   /**
153     * Inserts the specified child into the specified index.  All children that
154     * previously had index greater than or equal to the specified index are
155     * shifted up one.
156     *
157     * @param c  The child to insert.
158     * @param i  The index at which to insert the child.
159     * @return   <code>true</code> if and only if the insert was successful.
160    **/
161   public boolean insert(LinkedChild c, int i)
162   {
163     try { children.insertElementAt(c, i); }
164     catch (ArrayIndexOutOfBoundsException e) { return false; }
165 
166     c.parent = this;
167     c.previous = (i - 1 < 0) ? null : (LinkedChild) children.get(i - 1);
168     c.next =
169       (i + 1 >= children.size()) ? null : (LinkedChild) children.get(i + 1);
170 
171     if (c.previous != null) c.previous.next = c;
172     if (c.next != null) c.next.previous = c;
173 
174     return true;
175   }
176 
177 
178   /**
179     * Retrieves the child at the specified index in the vector.
180     *
181     * @param i  The index from which to retrieve a child.
182     * @return   The child at the specified index, or <code>null</code> if
183     *           there was no child at that index.
184    **/
185   public LinkedChild get(int i)
186   {
187     try { return (LinkedChild) children.get(i); }
188     catch (ArrayIndexOutOfBoundsException e) { }
189     return null;
190   }
191 
192 
193   /**
194     * Returns the size of the vector.
195     *
196     * @return The size of the vector.
197    **/
198   public int size() { return children.size(); }
199 
200 
201   /**
202     * Returns a clone of this object that is deep in the sense that all of the
203     * children objects are cloned.
204     *
205     * @return A deep clone of this object.
206    **/
207   public Object clone()
208   {
209     LinkedVector clone = (LinkedVector) super.clone();
210     clone.children = (Vector) clone.children.clone();
211     // This may look inefficient, but there is a purpose.  Subclasses of
212     // LinkedVector that don't define any new non-primitive member fields will
213     // not need to override this method, as it already produces an object of
214     // the subclass's type.
215     for (int i = 0; i < size(); ++i)
216       clone.insert((LinkedChild) clone.remove(i).clone(), i);
217     return clone;
218   }
219 }
