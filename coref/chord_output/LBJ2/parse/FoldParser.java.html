6   /**
7     * Useful when performing <i>k</i>-fold cross validation, this parser filters
8     * the examples coming from another parser.  Conceptually, the examples from
9     * the original parser are first split into <i>k</i> "folds" (or partitions)
10    * depending on the selected splitting strategy.  A particular fold is then
11    * selected as the pivot, and this parser can be configured either to return
12    * all and only the examples from that fold, or all and only the examples
13    * from other folds.
14    *
15    * <p> The <i>k</i> folds are referred to by their indexes, which are 0, 1,
16    * ..., <i>k</i> - 1.  This index is used to select the pivot fold.
17    *
18    * @see    FoldParser.SplitStrategy
19    * @author Dan Muriello, Nick Rizzolo
20   **/
21  public class FoldParser implements Parser
22  {
23    /** The parser whose examples are being filtered. */
24    protected Parser parser;
25    /** The total number of folds. */
26    protected int K;
27    /** The way in which examples are partitioned into folds. */
28    protected SplitStrategy splitStrategy;
29    /**
30      * The examples from this fold are exclusively selected for or excluded
31      * from the set of examples returned by this parser.
32     **/
33    protected int pivot;
34    /** Whether examples will come from the pivot fold or not. */
35    protected boolean fromPivot;
36    /** The total number of examples coming from {@link #parser}. */
37    protected int examples;
38  
39    /** Keeps track of the index of the next example to be returned. */
40    protected int exampleIndex;
41    /** Keeps track of the current fold; used only in manual splitting. */
42    protected int fold;
43    /**
44      * A lower bound for an index relating to the pivot fold.  The index
45      * variable in question may either be {@link #exampleIndex} or
46      * {@link #shuffleIndex}.
47     **/
48    protected int lowerBound;
49    /**
50      * An upper bound for an index relating to the pivot fold.  The index
51      * variable in question may either be {@link #exampleIndex} or
52      * {@link #shuffleIndex}.
53     **/
54    protected int upperBound;
55    /**
56      * Used only by the random splitting strategy to remember which example
57      * indexes are in which folds.
58     **/
59    protected int[] shuffled;
60    /** An index pointing into {@link #shuffled}. */
61    protected int shuffleIndex;
62  
63  
64    /**
65      * Constructor for when you don't know how many examples are in the data.
66      * Using a constructor that allows specification of the number of examples
67      * in the data only saves computation when the splitting strategy is either
68      * sequential or random.
69      *
70      * @param parser The parser whose examples are being filtered.
71      * @param K      The total number of folds; this value is ignored if the
72      *               splitting strategy is manual.
73      * @param split  The way in which examples are partitioned into folds.
74      * @param pivot  The index of the pivot fold.
75      * @param f      Whether to extract examples from the pivot.
76     **/
77    public FoldParser(Parser parser, int K, SplitStrategy split, int pivot,
78                      boolean f)
79    {
80      this(parser, K, split, pivot, f, -1);
81    }
82  
83    /**
84      * Constructor for when you know neither how many examples are in the data
85      * nor <i>K</i>, i.e., how many folds are in the data.  This constructor
86      * can only be used when the splitting strategy is manual.  Using a
87      * constructor that allows specification of the number of examples in the
88      * data only saves computation when the splitting strategy is either
89      * sequential or random.
90      *
91      * @param parser The parser whose examples are being filtered.
92      * @param split  The way in which examples are partitioned into folds.
93      * @param pivot  The index of the pivot fold.
94      * @param f      Whether to extract examples from the pivot.
95     **/
96    public FoldParser(Parser parser, SplitStrategy split, int pivot, boolean f)
97    {
98      this(parser, -1, split, pivot, f, -1);
99    }
100 
101   /**
102     * Full constructor.
103     *
104     * @param parser The parser whose examples are being filtered.
105     * @param K      The total number of folds; this value is ignored if the
106     *               splitting strategy is manual.
107     * @param split  The way in which examples are partitioned into folds.
108     * @param pivot  The index of the pivot fold.
109     * @param f      Whether to extract examples from the pivot.
110     * @param e      The total number of examples coming from
111     *               <code>parser</code>, or -1 if unknown.
112    **/
113   public FoldParser(Parser parser, int K, SplitStrategy split, int pivot,
114                     boolean f, int e)
115   {
116     this.K = K;
117     splitStrategy = split;
118     fromPivot = f;
119     examples = e;
120 
121     if (examples == -1
122         && (splitStrategy == SplitStrategy.sequential
123             || splitStrategy == SplitStrategy.random))
124     {
125       ++examples;
126       for (Object example = parser.next(); example != null;
127            example = parser.next())
128         if (example != FoldSeparator.separator) ++examples;
129       parser.reset();
130     }
131 
132     if (splitStrategy == SplitStrategy.random)
133     {
134       shuffled = new int[examples];
135       for (int i = 0; i < examples; ++i) shuffled[i] = i;
136       Random r = new Random();
137 
138       for (int i = 0; i < examples; ++i)
139       {
140         int j = i + r.nextInt(examples - i);
141         int t = shuffled[i];
142         shuffled[i] = shuffled[j];
143         shuffled[j] = t;
144       }
145 
146       for (int i = 0; i < K; ++i)
147       {
148         setPivot(i);
149         Arrays.sort(shuffled, lowerBound, upperBound);
150       }
151     }
152 
153     if (splitStrategy == SplitStrategy.manual)
154     {
155       this.K = 1;
156       for (Object example = parser.next(); example != null;
157            example = parser.next())
158         if (example == FoldSeparator.separator) ++this.K;
159       parser.reset();
160     }
161 
162     setPivot(pivot);
163     this.parser = parser;
164   }

168     * Retrieves the value of {@link #K}, which may have been computed in the

171   public int getK() { return K; }

174   /**
175     * Sets the value of {@link #fromPivot}, which controls whether examples
176     * will be taken from the pivot fold or from all other folds.
177     *
178     * @param f  The new value for {@link #fromPivot}.
179    **/
180   public void setFromPivot(boolean f) { fromPivot = f; }

183   /**
184     * Sets the pivot fold, which also causes {@link #parser} to be reset.
185     *
186     * @param p  The index of the new pivot fold.
187    **/
188   public void setPivot(int p)
189   {
190     pivot = p;
191     if (p < K) reset();
192   }
193 
194 
195   /**
196     * Sets this parser back to the beginning of the raw data.  This means
197     * arranging for all relevant state variables to be reset appropriately as
198     * well, since the value of {@link #pivot} may have changed.
199     *
200     * @see #setPivot(int)
201    **/
202   public void reset()
203   {
204     if (parser != null) parser.reset();
205 
206     if (splitStrategy == SplitStrategy.sequential
207         || splitStrategy == SplitStrategy.random)
208     {
209       lowerBound = pivot * (examples / K) + Math.min(pivot, examples % K);
210       upperBound =
211         (pivot + 1) * (examples / K) + Math.min(pivot + 1, examples % K);
212     }
213 
214     if (splitStrategy == SplitStrategy.random) shuffleIndex = lowerBound;
215     if (splitStrategy == SplitStrategy.manual) fold = 0;
216     exampleIndex = 0;
217   }
218 
219 
220   /**
221     * Convenient for determining if the next example should be returned or
222     * not.
223     *
224     * @param example  The next example object.
225     * @return <code>true</code> iff the next example should be returned.
226    **/
227   protected boolean filter(Object example)
228   {
229     if (example == FoldSeparator.separator) return false;
230     if (splitStrategy == SplitStrategy.sequential)
231       return fromPivot
232              == (exampleIndex >= lowerBound && exampleIndex < upperBound);
233     if (splitStrategy == SplitStrategy.random)
234       return fromPivot
235              == (shuffleIndex < upperBound
236                  && shuffled[shuffleIndex] == exampleIndex);
237     if (splitStrategy == SplitStrategy.kth)
238       return fromPivot == (exampleIndex % K == pivot);
239     // splitStrategy == SplitStrategy.manual
240     return fromPivot == (fold == pivot);
241   }
242 
243 
244   /**
245     * Changes state to reflect retrieval of the next example from the parser.
246     *
247     * @param example  The previous example object.
248    **/
249   protected void increment(Object example)
250   {
251     if (example == FoldSeparator.separator)
252     {
253       if (splitStrategy == SplitStrategy.manual) ++fold;
254     }
255     else
256     {
257       if (splitStrategy == SplitStrategy.random)
258       {
259         if (shuffleIndex < upperBound
260             && shuffled[shuffleIndex] == exampleIndex)
261           ++shuffleIndex;
262       }
263 
264       ++exampleIndex;
265     }
266   }
267 
268 
269   /** Retrieves the next example object. */
270   public Object next()
271   {
272     Object result = parser.next();
273     for (; result != null && !filter(result); result = parser.next())
274       increment(result);
275     if (result != null) increment(result);
276     return result;
277   }

280   /**
281     * Immutable type representing the way in which examples are partitioned
282     * into folds.  When LBJ's self imposed restriction to use Java 1.4 is
283     * lifted, this class will be replaced by an <code>enum</code>.
284     *
285     * <p> The four implemented splitting strategies are described below.  Note
286     * that in all cases except "Manual", the size of the folds are as equal as
287     * possible, with any extra examples allocated to earlier folds.
288     *
289     * <blockquote>
290     * <dl>
291     *   <dt> <b>Sequential</b> </dt>
292     *   <dd> The examples are simply partitioned into sequential folds. </dd>
293     *   <dt> <b>k<sup>th</sup></b> </dt>
294     *   <dd> Every k<sup>th</sup> example is in the same fold. </dd>
295     *   <dt> <b>Random</b> </dt>
296     *   <dd> Examples are randomly assigned to folds. </dd>
297     *   <dt> <b>Manual</b> </dt>
298     *   <dd>
299     *      Same as sequential, except fold boundaries are indicated by an
300     *      appearance of the {@link FoldSeparator} in place of an example
301     *      object.
302     *   </dd>
303     * </dl>
304     * </blockquote>
305     *
306     * @author Nick Rizzolo
307   **/
308   public static class SplitStrategy
309   {
310     /** Represents the random split strategy. */
311     public static final SplitStrategy random = new SplitStrategy(0);
312     /** Represents the sequential split strategy. */
313     public static final SplitStrategy sequential = new SplitStrategy(1);
314     /**
315       * Represents the split strategy in which every k<sup>th</sup> example is
316       * part of the same fold.
317     **/
318     public static final SplitStrategy kth = new SplitStrategy(2);
319     /**
320       * Represents the split strategy in which the user manually inserts fold
321       * separation objects.
322     **/
323     public static final SplitStrategy manual = new SplitStrategy(3);
324 
325     /** The names of the different split strategies as strings. */
326     private static final String[] names =
327     {
328       "random", "sequential", "kth", "manual"
329     };
330 
331 
332     /** Can be used to index the {@link #names} array. */
333     private int index;
334 
335 
336     /** Initializes the object with an index. */
337     private SplitStrategy(int i) { index = i; }
338 
339 
340     /** Retrieves the name of the strategy represented by this object. */
341     public String getName() { return names[index]; }
342   }
