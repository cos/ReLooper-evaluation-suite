10  /**
11    * This class is a program that can evaluate any <code>Classifier</code>
12    * against an oracle <code>Classifier</code> on the objects returned from a
13    * <code>Parser</code>.
14    *
15    * <p> Usage:
16    * <blockquote>
17    *   <code>
18    *     java LBJ2.classify.TestDiscrete [-t &lt;n&gt;] &lt;classifier&gt;
19    *                                     &lt;oracle&gt; &lt;parser&gt;
20    *                                     &lt;input file&gt; [&lt;null label&gt;
21    *                                     [&lt;null label&gt; ...]]
22    *   </code>
23    * </blockquote>
24    *
25    * <p> <b>Options:</b> The <code>-t &lt;n&gt;</code> option is similar to the
26    * LBJ compiler's command line option of the same name.  When
27    * <code>&lt;n&gt;</code> is greater than 0, a time stamp is printed to
28    * <code>STDOUT</code> after every <code>&lt;n&gt;</code> examples are
29    * processed.
30    *
31    * <p> <b>Input:</b> The first three command line parameters are fully
32    * qualified class names, e.g.  <code>myPackage.myClassifier</code>.
33    * Next, <code>&lt;input file&gt;</code> is passed (as a <code>String</code>)
34    * to the constructor of <code>&lt;parser&gt;</code>.  The optional parameter
35    * <code>&lt;null label&gt;</code> identifies one of the possible labels
36    * produced by <code>&lt;oracle&gt;</code> as representing "no
37    * classification".  It is used during the computation of overall precision,
38    * recall, and F1 scores.  Finally, it is also assumed that
39    * <code>&lt;classifier&gt;</code> is discrete, and that its
40    * <code>discreteValue(Object)</code> method is implemented.
41    *
42    * <p> <b>Output:</b> First some timing information is presented.  The first
43    * time reported is the time taken to load the specified classifier's Java
44    * class into memory.  This reflects the time taken for LBJ to load the
45    * classifier's internal representation <b>if</b> the classifier does
46    * <b>not</b> make use of the <code>cachedin</code> keyword.  Next, the time
47    * taken to evaluate the first example is reported.  It isn't particularly
48    * informative unless the classifier <b>does</b> make use of the
49    * <code>cachedin</code> keyword.  In this case, it reflects the time LBJ
50    * takes to load the classifier's internal representation better than the
51    * first time reported.  Finally, the average time taken to execute the
52    * classifier's <code>discreteValue(Object)</code> method is reported.
53    *
54    * <p> After the timing information, an ASCII table is written to
55    * <code>STDOUT</code> reporting precision, recall, and F<sub>1</sub> scores
56    * itemized by the values that either the classifier or the oracle produced
57    * during the test.  The two rightmost columns are named
58    * <code>"LCount"</code> and <code>"PCount"</code> (standing for "labeled
59    * count" and "predicted count" respectively), and they report the number of
60    * times the oracle produced each label and the number of times the
61    * classifier predicted each label respectively.  If a "null label" is
62    * specified, overall precision, recall, and F<sub>1</sub> scores and a total
63    * count of non-null-labeled examples are reported at the bottom of the
64    * table.  In the last row, whether a "null label" is specified or not,
65    * overall accuracy is reported in the precision column.  In the count
66    * column, the total number of predictions (or labels, equivalently) is
67    * reported.
68   **/

75    /** References the parser supplying the testing objects. */
76    private static Parser parser;
77    /** The number of examples processed in between time stamp messages. */

82      * The entry point of this program.
83      *
84      * @param args The command line parameters.
85     **/
86    public static void main(String[] args)
87    {
88      long totalTime = -System.currentTimeMillis();
89      TestDiscrete tester = instantiate(args);

92                         + " seconds.");
93      testDiscrete(tester, classifier, oracle, parser, true, outputGranularity);
94    }

97    /**
98      * Tests the given discrete classifier against the given oracle using the
99      * given parser to provide the labeled testing data.  This simplified
100     * interface to
101     * {@link #testDiscrete(TestDiscrete,Classifier,Classifier,Parser,boolean,int)}
102     * assumes there are no null predictions and that output should not be
103     * generated on <code>STDOUT</code>.
104     *
105     * @param classifier The classifier to be tested.
106     * @param oracle     The classifier to test against.
107     * @param parser     The parser supplying the labeled example objects.
108    **/
109   public static TestDiscrete testDiscrete(Classifier classifier,
110                                           Classifier oracle, Parser parser)
111   {
112     return
113       testDiscrete(new TestDiscrete(), classifier, oracle, parser, false, 0);
114   }

117   /**
118     * Tests the given discrete classifier against the given oracle using the
119     * given parser to provide the labeled testing data.
120     *
121     * @param tester     An object of this class that has already been told via
122     *                   {@link #addNull(String)} which prediction values are
123     *                   considered to be null predictions.
124     * @param classifier The classifier to be tested.
125     * @param oracle     The classifier to test against.
126     * @param parser     The parser supplying the labeled example objects.
127     * @param output     Whether or not to produce output on
128     *                   <code>STDOUT</code>.
129     * @param outputGranularity
130     *                   The number of examples processed in between time stamp
131     *                   messages.
132     * @return           The same <code>TestDiscrete</code> object passed in
133     *                   the first argument, after being filled with
134     *                   statistics.
135    **/
136   public static TestDiscrete testDiscrete(TestDiscrete tester,
137                                           Classifier classifier,
138                                           Classifier oracle, Parser parser,
139                                           boolean output,
140                                           int outputGranularity)
141   {
142     int processed = 1;
143     long totalTime = 0;
144 
145     if (output && outputGranularity > 0)
146     {
147       System.out.println("0 examples tested at " + new Date());
148       Object example = parser.next();
149       if (example == null) return tester;
150       totalTime -= System.currentTimeMillis();
151       String prediction = classifier.discreteValue(example);
152       totalTime += System.currentTimeMillis();
153       System.out.println("First example processed in " + (totalTime / 1000.0)
154                          + " seconds.");
155 
156       String gold = oracle.discreteValue(example);
157       tester.reportPrediction(prediction, gold);
158 
159       for (example = parser.next(); example != null;
160            example = parser.next(), ++processed)
161       {
162         if (processed % outputGranularity == 0)
163           System.out.println(processed + " examples tested at " + new Date());
164 
165         totalTime -= System.currentTimeMillis();
166         prediction = classifier.discreteValue(example);
167         totalTime += System.currentTimeMillis();
168         assert prediction != null
169              : "Classifier returned null prediction for example " + example;
170 
171         gold = oracle.discreteValue(example);
172         tester.reportPrediction(prediction, gold);
173       }
174 
175       System.out.println(processed + " examples tested at " + new Date()
176                          + "\n");
177     }
178     else
179     {
180       Object example = parser.next();
181       if (example == null) return tester;
182       totalTime -= System.currentTimeMillis();
183       String prediction = classifier.discreteValue(example);
184       totalTime += System.currentTimeMillis();
185       if (output)
186         System.out.println("First example processed in "
187                            + (totalTime / 1000.0) + " seconds.");
188 
189       String gold = oracle.discreteValue(example);
190       tester.reportPrediction(prediction, gold);
191 
192       for (example = parser.next(); example != null;
193            example = parser.next(), ++processed)
194       {
195         totalTime -= System.currentTimeMillis();
196         prediction = classifier.discreteValue(example);
197         totalTime += System.currentTimeMillis();
198 
199         gold = oracle.discreteValue(example);
200         tester.reportPrediction(prediction, gold);
201       }
202     }
203 
204     if (output)
205     {
206       System.out.println("Average evaluation time: "
207                          + (totalTime / (1000.0 * processed)) + " seconds\n");
208       tester.printPerformance(System.out);
209     }
210 
211     return tester;
212   }

215   /**
216     * Given command line parameters representing the fully qualified names of
217     * the classifier to be tested, the oracle classifier to test against, the
218     * parser supplying the testing objects, and the input parameter to the
219     * parser's constructor this method instantiates all three objects.
220     *
221     * @param args The command line.
222     * @return     A new tester object containing the "null" labels.
223    **/
224   private static TestDiscrete instantiate(String[] args)
225   {
226     String classifierName = null, oracleName = null, parserName = null;
227     String inputFile = null;
228     TestDiscrete result = new TestDiscrete();
229 
230     try
231     {
232       int offset = 0;
233 
234       if (args[0].charAt(0) == '-')
235       {
236         if (!args[0].equals("-t")) throw new Exception();
237         outputGranularity = Integer.parseInt(args[1]);
238         offset = 2;
239       }
240 
241       classifierName = args[offset];
242       oracleName = args[offset + 1];
243       parserName = args[offset + 2];
244       inputFile = args[offset + 3];
245       for (int i = offset + 4; i < args.length; ++i) result.addNull(args[i]);
246     }
247     catch (Exception e)
248     {
249       System.err.println(
250       "usage:\n"
251     + "  java LBJ2.classify.TestDiscrete [-t <n>] <classifier> <oracle> \\\n"
252     + "                                  <parser> <input file> \\\n"
253     + "                                  [<null label> [<null label> ...]]");
254       System.exit(1);
255     }
256 
257     classifier = ClassUtils.getClassifier(classifierName);
258     oracle = ClassUtils.getClassifier(oracleName);
259     parser =
260       ClassUtils.getParser(parserName, new Class[]{ String.class },
261                            new String[]{ inputFile });
262 
263     return result;
264   }

386     * Takes a histogram implemented as a map and increments the count for the
387     * given key by the given amount.
388     *
389     * @param histogram  The histogram.
390     * @param key        The key whose count should be incremented.
391     * @param amount     The amount by which to increment.
392    **/
393   protected void histogramAdd(HashMap histogram, String key, int amount)
394   {     
395     Integer I = (Integer) histogram.get(key);
396     if (I == null) I = new Integer(0);
397     histogram.put(key, new Integer(I.intValue() + amount));
398   }

402     * Takes a histogram implemented as a map and retrieves the count for the
403     * given key.
404     *
405     * @param histogram  The histogram.
406     * @param key        The key whose count should be retrieved.
407     * @return           The count of the specified key.
408    **/
409   protected int histogramGet(HashMap histogram, String key)
410   {       
411     Integer I = (Integer) histogram.get(key);
412     if (I == null) I = new Integer(0);
413     return I.intValue();
414   }

417   /**
418     * Takes two histograms implemented as maps and adds the amounts found in
419     * the second histogram to the amounts found in the first.
420     *
421     * @param h1 The first histogram, whose values will be modified.
422     * @param h2 The second histogram, whose values will be added into the
423     *           first's.
424    **/
425   protected void histogramAddAll(HashMap h1, HashMap h2)
426   {
427     for (Iterator I = h2.entrySet().iterator(); I.hasNext(); )
428     {
429       Map.Entry e = (Map.Entry) I.next();
430       histogramAdd(h1, (String) e.getKey(),
431                    ((Integer) e.getValue()).intValue());
432     }
433   }

503   /**
504     * Returns the F<sub>beta</sub> score associated with the given label.
505     * F<sub>beta</sub> is defined as:
506     * <blockquote>
507     *   <i>F<sub>beta</sub> = (beta<sup>2</sup> + 1) * P * R</i>
508     *   <i>/ (beta<sup>2</sup> * P + R)</i>
509     * </blockquote>
510     *
511     * @param b  The value of beta.
512     * @param l  The given label.
513     * @return   The F<sub>beta</sub> score associated with <code>l</code>.
514    **/
515   public double getF(double b, String l)
516   {
517     double precision = getPrecision(l);
518     double recall = getRecall(l);
519     return (b * b + 1) * precision * recall / (b * b * precision + recall);
520   }

535   /**
536     * Computes overall the overall statistics precision, recall,
537     * F<sub>beta</sub>, and accuracy.  Note that these statistics are all
538     * equivalent unless "null" labels have been added.
539     *
540     * @param b  The value of beta.
541     * @return   An array in which the first element represents overall
542     *           precision, the second represents overall recall, then F1, and
543     *           finally accuracy.
544    **/
545   public double[] getOverallStats(double b)
546   {
547     String[] allClasses = getAllClasses();
548 
549     int totalCorrect = 0;
550     int totalPredicted = 0;
551     int notNullCorrect = 0;
552     int notNullPredicted = 0;
553     int notNullLabeled = 0;
554 
555     for (int i = 0; i < allClasses.length; ++i)
556     {
557       int correct = getCorrect(allClasses[i]);
558       int predicted = getPredicted(allClasses[i]);
559       int labeled = getLabeled(allClasses[i]);
560 
561       totalCorrect += correct;
562       totalPredicted += predicted;
563 
564       if (hasNulls() && !isNull(allClasses[i]))
565       {
566         notNullCorrect += correct;
567         notNullPredicted += predicted;
568         notNullLabeled += labeled;
569       }
570     } 
571 
572     double[] result = new double[4];
573     result[3] = totalCorrect / (double) totalPredicted;
574 
575     if (hasNulls())
576     {
577       result[0] = notNullCorrect / (double) notNullPredicted;
578       result[1] = notNullCorrect / (double) notNullLabeled;
579       result[2] = (b * b + 1) * result[0] * result[1]
580                   / (b * b * result[0] + result[1]);
581     }
582     else result[0] = result[1] = result[2] = result[3];
583 
584     return result;
585   }

588   /**
589     * Performance results are written to the given stream in the form of
590     * precision, recall, and F1 statistics.
591     *
592     * @param out        The stream to write to.
593    **/
594   public void printPerformance(PrintStream out)
595   {
596       
597     String[] allClasses = getAllClasses();
598     final HashSet n = nullLabels;
599     Arrays.sort(allClasses,
600                 new Comparator()
601                 {
602                   public int compare(Object o1, Object o2)
603                   {
604                     String s1 = (String) o1;
605                     String s2 = (String) o2;
606                     int n1 = n.contains(s1) ? 1 : 0;
607                     int n2 = n.contains(s2) ? 1 : 0;
608                     if (n1 != n2) return n1 - n2;
609                     return s1.compareTo(s2);
610                   }
611                 });
612 
613     int rows = allClasses.length + 1;
614     if (hasNulls()) ++rows;
615     String[] rowLabels = new String[rows];
616     System.arraycopy(allClasses, 0, rowLabels, 0, allClasses.length);
617     rowLabels[rows - 1] = "Accuracy";
618     if (hasNulls()) rowLabels[rows - 2] = "Overall";
619 
620     String[] columnLabels =
621       new String[]{ "Label", "Precision", "Recall", "F1", "LCount",
622                     "PCount" };
623 
624     int totalCorrect = 0;
625     int totalPredicted = 0;
626     int notNullCorrect = 0;
627     int notNullPredicted = 0;
628     int notNullLabeled = 0;
629 
630     Double[][] table = new Double[rows][];
631     Double zero = new Double(0);
632 
633     for (int i = 0; i < allClasses.length; ++i)
634     {
635       int correct = getCorrect(allClasses[i]);
636       int predicted = getPredicted(allClasses[i]);
637       int labeled = getLabeled(allClasses[i]);
638 
639       totalCorrect += correct;
640       totalPredicted += predicted;
641 
642       if (hasNulls() && !isNull(allClasses[i]))
643       {
644         notNullCorrect += correct;
645         notNullPredicted += predicted;
646         notNullLabeled += labeled;
647       }
648 
649       table[i] =
650         new Double[]{ zero, zero, zero, new Double(labeled),
651                       new Double(predicted) };
652 
653       if (predicted > 0)
654         table[i][0] = new Double(100 * correct / (double) predicted);
655       if (labeled > 0)
656         table[i][1] = new Double(100 * correct / (double) labeled);
657 
658       if (correct > 0)
659       {
660         double p = table[i][0].doubleValue();
661         double r = table[i][1].doubleValue();
662         table[i][2] = new Double(2 * p * r / (p + r));
663       }
664     }
665 
666     int[] dashRows = null;
667 
668     if (hasNulls())
669     {
670       table[rows - 2] =
671         new Double[]{ zero, zero, zero, new Double(notNullLabeled),
672                       new Double(notNullPredicted) };
673 
674       if (notNullPredicted > 0)
675         table[rows - 2][0] =
676           new Double(100 * notNullCorrect / (double) notNullPredicted);
677       if (notNullLabeled > 0)
678         table[rows - 2][1] =
679           new Double(100 * notNullCorrect / (double) notNullLabeled);
680 
681       if (notNullCorrect > 0)
682       {
683         double p = table[rows - 2][0].doubleValue();
684         double r = table[rows - 2][1].doubleValue();
685         table[rows - 2][2] = new Double(2 * p * r / (p + r));
686       }
687 
688       int nonNullLabels = allClasses.length - nullLabels.size();
689       dashRows = new int[]{ 0, nonNullLabels, allClasses.length };
690     }
691     else dashRows = new int[]{ 0, allClasses.length };
692 
693     double accuracy =
694       totalPredicted == 0 ? 0 : 100 * totalCorrect / (double) totalPredicted;
695     table[rows - 1] =
696       new Double[]{ new Double(accuracy), null, null, null,
697                     new Double(totalPredicted) };
698 
699     String[] result =
700       TableFormat.tableFormat(columnLabels, rowLabels, table,
701                               new int[]{ 3, 3, 3, 0, 0 }, dashRows);
702     for (int i = 0; i < result.length; ++i)
703       System.out.println(result[i]);
704   }
705 }
