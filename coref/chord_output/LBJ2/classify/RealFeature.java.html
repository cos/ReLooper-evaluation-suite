4   /**
5     * A real feature takes on any value representable by a <code>double</code>.
6     *
7     * @author Nick Rizzolo
8    **/
9   public class RealFeature extends Feature
10  {
11    /** The real value is represented as a double. */
12    protected double value;
13  
14  
15    /**
16      * Sets both the identifier and the value.
17      *
18      * @param p  The new real feature's package.
19      * @param i  The new <code>RealFeature</code>'s identifier.
20      * @param v  The new <code>RealFeature</code>'s value.
21     **/
22    public RealFeature(String p, String i, double v)
23    {
24      super(p, i);
25      value = v;
26    }
27  
28  
29    /** Returns the value of the <code>value</code> member variable. */
30    public double getValue() { return value; }
31  
32  
33    /**
34      * Gives a string representation of the value of this feature.
35      *
36      * @return A string representation of the value of this feature.
37     **/
38    public String getStringValue() { return "" + value; }
39  
40  
41    /**
42      * Determines whether or not the parameter is equivalent to the string
43      * representation of the value of this feature.
44      *
45      * @param v  The string to compare against.
46      * @return   <code>true</code> iff the parameter is equivalent to the
47      *           string representation of the value of this feature.
48     **/
49    public boolean valueEquals(String v) { return v.equals("" + value); }
50  
51  
52    /**
53      * Create a feature representing the conjunction of this feature with the
54      * given argument feature.
55      *
56      * @param f  The feature to conjunct with.
57      * @param c  The classifier producing the resulting feature.
58      * @return   A feature representing the conjunction of this feature and
59      *           <code>f</code>.
60     **/
61    public Feature conjunction(Feature f, Classifier c)
62    {
63      return f.conjunctWith(this, c);
64    }
65  
66  
67    /**
68      * Create a feature representing the conjunction of this feature with the
69      * given argument feature.
70      *
71      * @param f  The feature to conjunct with.
72      * @param c  The classifier producing the resulting feature.
73      * @return   A feature representing the conjunction of this feature and
74      *           <code>f</code>.
75     **/
76    protected Feature conjunctWith(DiscreteFeature f, Classifier c)
77    {
78      return f.conjunctWith(this, c);
79    }
80  
81  
82    /**
83      * Create a feature representing the conjunction of this feature with the
84      * given argument feature.
85      *
86      * @param f  The feature to conjunct with.
87      * @param c  The classifier producing the resulting feature.
88      * @return   A feature representing the conjunction of this feature and
89      *           <code>f</code>.
90     **/
91    protected Feature conjunctWith(DiscreteArrayFeature f, Classifier c)
92    {
93      return f.conjunctWith(this, c);
94    }
95  
96  
97    /**
98      * Create a feature representing the conjunction of this feature with the
99      * given argument feature.
100     *
101     * @param f  The feature to conjunct with.
102     * @param c  The classifier producing the resulting feature.
103     * @return   A feature representing the conjunction of this feature and
104     *           <code>f</code>.
105    **/
106   protected Feature conjunctWith(RealFeature f, Classifier c)
107   {
108     return new RealFeature(c.containingPackage, c.name, value * f.getValue());
109   }
110 
111 
112   /**
113     * Create a feature representing the conjunction of this feature with the
114     * given argument feature.
115     *
116     * @param f  The feature to conjunct with.
117     * @param c  The classifier producing the resulting feature.
118     * @return   A feature representing the conjunction of this feature and
119     *           <code>f</code>.
120    **/
121   protected Feature conjunctWith(RealArrayFeature f, Classifier c)
122   {
123     return
124       new RealArrayFeature(c.containingPackage, c.name, value * f.getValue(),
125                            f.getArrayIndex(), f.getArrayLength());
126   }
127 
128 
129   /**
130     * The hash code of a <code>RealFeature</code> is the sum of the hash codes
131     * of the containing package, the identifier, and the value.
132     *
133     * @return The hash code for this <code>Feature</code>.
134    **/
135   public int hashCode()
136   {
137     return super.hashCode() + new Double(value).hashCode();
138   }
139 
140 
141   /**
142     * Two <code>RealFeature</code>s are equivalent when their containing
143     * packages and identifiers are equivalent and their values are equal.
144     *
145     * @param o  The object with which to compare this <code>Feature</code>.
146     * @return   True iff the parameter is an equivalent <code>Feature</code>.
147    **/
148   public boolean equals(Object o)
149   {
150     return super.equals(o) && value == ((RealFeature) o).value;
151   }
152 
153 
154   /**
155     * Used to sort features via a lexicographic comparison of their packages
156     * and identifiers.
157     *
158     * @param o  An object to compare with.
159     * @return   The same thing that a <code>String</code> comparison of the
160     *           concatenations of the features' packages and identifiers would
161     *           return.
162    **/
163   public int compareTo(Object o)
164   {
165     int superComparison = super.compareTo(o);
166     if (superComparison != 0) return superComparison;
167     double difference = value - ((RealFeature) o).value;
168     if (difference < 0) return -1;
169     if (difference > 0) return 1;
170     return 0;
171   }
172 
173 
174   /**
175     * The string representation of a real feature is its identifier followed
176     * by its value surrounded by parentheses.
177     *
178     * @return The string representation of this <code>Feature</code>.
179    **/
180   public String toString()
181   {
182     return containingPackage + ":" + identifier + "(" + value + ")";
183   }
184 }
