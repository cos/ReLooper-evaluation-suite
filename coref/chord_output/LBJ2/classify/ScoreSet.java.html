6   /**
7     * A score set is simply a set of <code>Score</code>s.
8     *
9     * @see Score
10    * @author Nick Rizzolo
11   **/
12  public class ScoreSet implements Cloneable
13  {
14    /** The scores in this set, indexed by the discrete classification. */
15    private HashMap set;
16  
17  
18    /** Default constructor. */
19    public ScoreSet() { this(null, null); }
20  
21    /**
22      * The elements of the two argument arrays are assumed to be pair-wise
23      * associated with each other.
24      *
25      * @param values The classification values being scored.
26      * @param scores The scores of the classification values.
27     **/
28    public ScoreSet(String[] values, double[] scores)
29    {
30      set = new HashMap();
31      if (values == null || scores == null) return;
32      for (int i = 0; i < values.length && i < scores.length; ++i)
33        put(values[i], scores[i]);
34    }
35  
36    /**
37      * The elements of the array are added to the set.
38      *
39      * @param scores The scores to add to the set.
40     **/
41    public ScoreSet(Score[] scores)
42    {
43      set = new HashMap();
44      if (scores == null) return;
45      for (int i = 0; i < scores.length; ++i)
46        set.put(scores[i].value, scores[i]);
47    }
48  
49  
50    /** Returns the number of scores in this set. */
51    public int size() { return set.size(); }
52  
53  
54    /**
55      * Sets the score for a particular classification value.
56      *
57      * @param v  The classification value.
58      * @param s  The score.
59     **/
60    public void put(String v, double s) { set.put(v, new Score(v, s)); }
61  
62  
63    /**
64      * Retrieves the set of values that have scores associated with them in
65      * this score set.
66      *
67      * @return A set of <code>String</code>s.
68     **/
69    public Set values() { return set.keySet(); }
70  
71  
72    /**
73      * Returns the double precision score for a particular classification
74      * value.
75      *
76      * @param v  The classification value.
77      * @return   The associated score.
78     **/
79    public double get(String v) { return ((Score) set.get(v)).score; }
80  
81  
82    /**
83      * Retrieves the {@link Score} object associated with the given
84      * classification value.
85      *
86      * @param v  The classification value.
87      * @return   The associated {@link Score} object.
88     **/
89    public Score getScore(String v) { return (Score) set.get(v); }
90  
91  
92    /** Retrieves the value with the highest score in this set. */
93    public String highScoreValue()
94    {
95      String result = null;
96      double highScore = Double.NEGATIVE_INFINITY;
97  
98      for (Iterator I = set.entrySet().iterator(); I.hasNext(); )
99      {
100       Map.Entry e = (Map.Entry) I.next();
101       double score = ((Score) e.getValue()).score;
102       if (score > highScore)
103       {
104         highScore = score;
105         result = (String) e.getKey();
106       }
107     }
108 
109     return result;
110   }
111 
112 
113   /**
114     * Returns an array view of the <code>Score</code>s contained in this set.
115     *
116     * @return An array of <code>Score</code>s.
117    **/
118   public Score[] toArray()
119   {
120     return (Score[]) set.values().toArray(new Score[set.size()]);
121   }
122 
123 
124   /**
125     * The string representation of a <code>ScoreSet</code> is the
126     * concatenation of the string representations of each <code>Score</code>
127     * in the set sorted by value, separated by commas, and surrounded by curly
128     * braces.
129     *
130     * @return The string representation of a <code>ScoreSet</code>.
131    **/
132   public String toString()
133   {
134     String result = "{";
135 
136     if (set.size() > 0)
137     {
138       Score[] scores = toArray();
139       Arrays.sort(scores);
140       result += " " + scores[0];
141       for (int i = 1; i < scores.length; ++i) result += ", " + scores[i];
142     }
143 
144     return result + " }";
145   }
146 
147 
148   /**
149     * Produces a deep copy of this object.
150     *
151     * @return A deep copy of this object.
152    **/
153   public Object clone()
154   {
155     Score[] scores = toArray();
156     for (int i = 0; i < scores.length; ++i)
157       scores[i] = (Score) scores[i].clone();
158     return new ScoreSet(scores);
159   }
