4   /**
5     * A real array feature keeps track of its index in the classifier's returned
6     * array.
7     *
8     * @author Nick Rizzolo
9    **/
10  public class RealArrayFeature extends RealFeature
11  {
12    /** The feature's index in the returned array it is contained in. */
13    protected int arrayIndex;
14    /** The size of the returned array this feature is contained in. */
15    protected int arrayLength;
16  
17  
18    /**
19      * Sets all member variables.
20      *
21      * @param p  The new real feature's package.
22      * @param id The new real feature's identifier.
23      * @param v  The new real feature's value.
24      * @param i  The index of this feature in the returned array.
25      * @param l  The length of the array this feature is contained in.
26     **/
27    public RealArrayFeature(String p, String id, double v, int i, int l)
28    {
29      super(p, id, v);
30      arrayIndex = i;
31      arrayLength = l;
32    }
33  
34  
35    /**
36      * Determines if this feature comes from an array.
37      *
38      * @return <code>true</code>.
39     **/
40    public boolean fromArray() { return true; }

42  
43    /**
44      * If this feature is an array feature, call this method to set its array
45      * length; otherwise, this method has no effect.
46      *
47      * @param l  The new length.
48     **/
49    public void setArrayLength(int l) { arrayLength = l; }
50  
51  
52    /** Returns the array index of this feature. */
53    public int getArrayIndex() { return arrayIndex; }
54  
55  
56    /** Returns the length of the feature array that this feature comes from. */
57    public int getArrayLength() { return arrayLength; }

120   /**
121     * Create a feature representing the conjunction of this feature with the
122     * given argument feature.
123     *
124     * @param f  The feature to conjunct with.
125     * @param c  The classifier producing the resulting feature.
126     * @return   A feature representing the conjunction of this feature and
127     *           <code>f</code>.
128    **/
129   protected Feature conjunctWith(RealArrayFeature f, Classifier c)
130   {
131     int index = f.getArrayIndex() * arrayLength + arrayIndex;
132     int length = f.getArrayLength() * arrayLength;
133     return
134       new RealArrayFeature(c.containingPackage, c.name, value * f.getValue(),
135                            index, length);
136   }

139   /**
140     * The hash code of a <code>RealArrayFeature</code> is the sum of the hash
141     * codes of the containing package, the identifier, the value, and the
142     * array index.
143     *
144     * @return The hash code for this <code>Feature</code>.
145    **/
146   public int hashCode()
147   {
148     return super.hashCode() + new Integer(arrayIndex).hashCode();
149   }

152   /**
153     * Two <code>RealArrayFeature</code>s are equivalent when their containing
154     * packages, identifiers, indices, and values are equivalent.
155     *
156     * @param o  The object with which to compare this <code>Feature</code>.
157     * @return   True iff the parameter is an equivalent <code>Feature</code>.
158    **/
159   public boolean equals(Object o)
160   {
161     return super.equals(o) && arrayIndex == ((RealArrayFeature) o).arrayIndex;
162   }

165   /**
166     * Used to sort features via a lexicographic comparison of their packages
167     * and identifiers.
168     *
169     * @param o  An object to compare with.
170     * @return   The same thing that a <code>String</code> comparison of the
171     *           concatenations of the features' packages and identifiers would
172     *           return.
173    **/
174   public int compareTo(Object o)
175   {
176     if (!(o instanceof RealArrayFeature)) return -1;
177     RealArrayFeature f = (RealArrayFeature) o;
178     int c = containingPackage.compareTo(f.containingPackage);
179     if (c != 0) return c;
180     c = identifier.compareTo(f.identifier);
181     if (c != 0) return c;
182     c = arrayIndex - f.arrayIndex;
183     if (c != 0) return c;
184     double difference = value - ((RealFeature) o).value;
185     if (difference < 0) return -1;
186     if (difference > 0) return 1;
187     return 0;
188   }

191   /**
192     * The string representation of a real feature is its identifier followed
193     * by its array index surrounded by square brackets and its value
194     * surrounded by curly braces.
195     *
196     * @return The string representation of this <code>Feature</code>.
197    **/
198   public String toString()
199   {
200     return containingPackage + ":" + identifier + "[" + arrayIndex + "]("
201            + value + ")";
202   }
