6   /**
7     * This classifier applies another classifier to the example object and
8     * returns a Boolean feature (with value "true" or "false") indicating
9     * whether a given feature value appeared in the output of the classifier.
10    * This behavior differs from that of {@link ValueComparer} because it does
11    * not assume that the given classifier will return only a single feature.
12    *
13    * @author Nick Rizzolo
14   **/
15  public class MultiValueComparer extends ValueComparer
16  {
17    /**
18      * Constructor.
19      *
20      * @param c  The classifier whose value will be compared.
21      * @param v  The value to compare with.
22     **/
23    public MultiValueComparer(Classifier c, String v) { super(c, v); }
24  
25  
26    /**
27      * Returns a Boolean feature (with value "true" or "false") indicating
28      * whether the output of {@link ValueComparer#labeler} applied to the
29      * argument object contained the feature value referenced by
30      * {@link ValueComparer#value}.
31      *
32      * @param o  The object to make decisions about.
33      * @return   A feature vector containing the feature described above.
34     **/
35    public FeatureVector classify(Object o)
36    {
37      boolean prediction = false;
38  
39      for (Iterator I = labeler.classify(o).iterator();
40           I.hasNext() && !prediction; )
41        prediction = ((DiscreteFeature) I.next()).getValue().equals(value);
42  
43      short p = prediction ? (short) 1 : (short) 0;
44      return
45        new FeatureVector(
46            new DiscreteFeature("", "MultiValueComparer",
47                                DiscreteFeature.BooleanValues[p], p,
48                                (short) 2));
49    }
50  
51  
52    /**
53      * The <code>String</code> representation of a <code>ValueComparer</code>
54      * has the form <code>"ValueComparer(</code><i>child</i><code>)</code>,
55      * where <i>child</i> is the <code>String</code> representation of the
56      * classifier whose value is being compared.
57      *
58      * @return A string of the form described above.
59     **/
60    public String toString()
61    {
62      return "MultiValueComparer(" + labeler + ")";
63    }
64  }
