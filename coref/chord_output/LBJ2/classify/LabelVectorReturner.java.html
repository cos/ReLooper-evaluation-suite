4   /**
5     * This classifier expects a <code>FeatureVector</code> as input, and it
6     * returns the contents of its <code>labels</code> list in a new
7     * <code>FeatureVector</code> as output.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class LabelVectorReturner extends Classifier
12  {
13    /** Default constructor. */
14    public LabelVectorReturner() { super("LabelVectorReturner"); }
15  
16  
17    /**
18      * This method makes one or more decisions about a single object, returning
19      * those decisions as <code>Feature</code>s in a vector.
20      *
21      * @param o  The object to make decisions about.
22      * @return   A vector of <code>Feature</code>s about the input object.
23     **/
24    public FeatureVector classify(Object o)
25    {
26      FeatureVector vector = null;
27  
28      try { vector = (FeatureVector) o; }
29      catch (ClassCastException e)
30      {
31        System.err.println("LBJ2 ERROR: LabelVectorReturner received a '"
32                           + o.getClass().getName() + "' as input.");
33        System.exit(1);
34      }
35  
36      FeatureVector result = new FeatureVector();
37      result.features.addAll(vector.labels);
38      return result;
39    }
40  
41  
42    /**
43      * Returns a string describing the input type of this classifier.
44      *
45      * @return A string describing the input type of this classifier.
46     **/
47    public String getInputType() { return "LBJ2.classify.FeatureVector"; }
48  
49  
50    /**
51      * Returns a string describing the output type of this classifier.
52      *
53      * @return A string describing the output type of this classifier.
54     **/
55    public String getOutputType() { return "mixed%"; }
56  
57  
58    /** Simply returns the string <code>"LabelVectorReturner"</code>. */
59    public String toString() { return "LabelVectorReturner"; }
60  }
