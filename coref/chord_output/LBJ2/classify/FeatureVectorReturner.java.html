4   /**
5     * This classifier expects <code>FeatureVector</code>s as input, and it
6     * simply returns them as output.
7     *
8     * @author Nick Rizzolo
9    **/
10  public class FeatureVectorReturner extends Classifier
11  {
12    /** Default constructor. */
13    public FeatureVectorReturner() { super("FeatureVectorReturner"); }
14  
15  
16    /**
17      * This method makes one or more decisions about a single object, returning
18      * those decisions as <code>Feature</code>s in a vector.
19      *
20      * @param o  The object to make decisions about.
21      * @return   A vector of <code>Feature</code>s about the input object.
22     **/
23    public FeatureVector classify(Object o)
24    {
25      FeatureVector result = null;
26  
27      try { result = (FeatureVector) o; }
28      catch (ClassCastException e)
29      {
30        System.err.println("LBJ2 ERROR: FeatureVectorReturner received a '"
31                           + o.getClass().getName() + "' as input.");
32        System.exit(1);
33      }
34  
35      return result;
36    }
37  
38  
39    /**
40      * Returns a string describing the input type of this classifier.
41      *
42      * @return A string describing the input type of this classifier.
43     **/
44    public String getInputType() { return "LBJ2.classify.FeatureVector"; }
45  
46  
47    /**
48      * Returns a string describing the output type of this classifier.
49      *
50      * @return A string describing the output type of this classifier.
51     **/
52    public String getOutputType() { return "mixed%"; }
53  
54  
55    /** Simply returns the string <code>"FeatureVectorReturner"</code>. */
56    public String toString() { return "FeatureVectorReturner"; }
57  }
