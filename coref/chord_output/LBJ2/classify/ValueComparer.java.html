4   /**
5     * This classifier applies another classifier to the example object and
6     * returns a Boolean feature (with value "true" or "false") representing the
7     * equality of the argument classifier's feature value to a given value.
8     *
9     * @see    MultiValueComparer
10    * @author Nick Rizzolo
11   **/
12  public class ValueComparer extends Classifier
13  {
14    /** The classifier whose value will be compared. */
15    protected Classifier labeler;
16    /** The value to compare with. */
17    protected String value;
18  
19  
20    /**
21      * Constructor.
22      *
23      * @param c  The classifier whose value will be compared.
24      * @param v  The value to compare with.
25     **/
26    public ValueComparer(Classifier c, String v)
27    {
28      super("ValueComparer");
29      labeler = c;
30      value = v;
31    }
32  
33  
34    /** Sets the value of {@link #labeler}. */
35    public void setLabeler(Classifier l) { labeler = l; }
36  
37  
38    /**
39      * Returns a Boolean feature (with value "true" or "false") representing
40      * the equality of the output of {@link #labeler} applied to the argument
41      * object and {@link #value}.
42      *
43      * @param o  The object to make decisions about.
44      * @return   A feature vector containing the feature described above.
45     **/
46    public FeatureVector classify(Object o)
47    {
48      DiscreteFeature f = (DiscreteFeature) labeler.classify(o).firstFeature();
49      short prediction = f.getValue().equals(value) ? (short) 1 : (short) 0;
50      return
51        new FeatureVector(
52            new DiscreteFeature("", "ValueComparer",
53                                DiscreteFeature.BooleanValues[prediction],
54                                prediction, (short) 2));
55    }
56  
57  
58    /**
59      * Returns a string describing the input type of this classifier.
60      *
61      * @return A string describing the input type of this classifier.
62     **/
63    public String getInputType() { return labeler.getInputType(); }
64  
65  
66    /**
67      * Returns the array of allowable values that a feature returned by this
68      * classifier may take.
69      *
70      * @see    DiscreteFeature#BooleanValues
71      * @return <code>DiscreteFeature.BooleanValues</code>
72     **/
73    public String[] allowableValues() { return DiscreteFeature.BooleanValues; }
74  
75  
76    /**
77      * The <code>String</code> representation of a <code>ValueComparer</code>
78      * has the form <code>"ValueComparer(</code><i>child</i><code>)</code>,
79      * where <i>child</i> is the <code>String</code> representation of the
80      * classifier whose value is being compared.
81      *
82      * @return A string of the form described above.
83     **/
84    public String toString() { return "ValueComparer(" + labeler + ")"; }
85  }
