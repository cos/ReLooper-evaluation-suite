4   /**
5     * A score is a number produced by a learner that indicates the degree to
6     * which a particular discrete classification is appropriate for a given
7     * object.  The scores for all possible discrete classifications given an
8     * object need not be positive or sum to one.  A <code>Score</code> object
9     * simply contains a score and the associated discrete classification.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class Score implements Comparable, Cloneable
14  {
15    /** The discrete classification associated with this score. */
16    public String value;
17    /** The score. */
18    public double score;
19  
20  
21    /**
22      * Initializes both member variables.
23      *
24      * @param v  The discrete classification.
25      * @param s  The score.
26     **/
27    public Score(String v, double s)
28    {
29      value = v;
30      score = s;
31    }
32  
33  
34    /**
35      * This method is implemented so that a collection of <code>Score</code>s
36      * will be sorted first by value and then by score.
37      *
38      * @param o  The object to compare against.
39      * @return   A negative integer, zero, or a positive integer if this object
40      *           is less than, equal to, or greater than the specified object
41      *           respectively.
42     **/
43    public int compareTo(Object o)
44    {
45      if (!(o instanceof Score)) return -1;
46      Score s = (Score) o;
47      int result = value.compareTo(s.value);
48      if (result == 0)
49        result = new Double(score).compareTo(new Double(s.score));
50      return result;
51    }
52  
53  
54    /**
55      * The string representation of a <code>Score</code> is the value followed
56      * by the score separated by a colon.
57      *
58      * @return The string representation of a <code>Score</code>.
59     **/
60    public String toString() { return value + " : " + score; }
61  
62  
63    /**
64      * Produces a deep copy of this object.
65      *
66      * @return A deep copy of this object.
67     **/
68    public Object clone()
69    {
70      Object result = null;
71  
72      try { result = super.clone(); }
73      catch (Exception e)
74      {
75        System.err.println("Error cloning " + getClass().getName() + ":");
76        e.printStackTrace();
77        System.exit(1);
78      }
79  
80      return result;
81    }
82  }
