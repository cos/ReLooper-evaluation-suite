4   /**
5     * A discrete array feature keeps track of its index in the classifier's
6     * returned array as well as the total number of features in that array.
7     *
8     * @author Nick Rizzolo
9    **/
10  public class DiscreteArrayFeature extends DiscreteFeature
11  {
12    /** The feature's index in the returned array it is contained in. */
13    protected int arrayIndex;
14    /** The size of the returned array this feature is contained in. */
15    protected int arrayLength;
16  
17  
18    /**
19      * Sets the identifier, value, array index, and size of the containing
20      * array.  The value index and total allowable values, having not been
21      * specified, default to -1 and 0 respectively.
22      *
23      * @param p  The new discrete feature's package.
24      * @param id The new discrete feature's identifier.
25      * @param v  The new discrete feature's value.
26      * @param i  The index of this feature in the returned array.
27      * @param l  The length of the array this feature is contained in.
28     **/
29    public DiscreteArrayFeature(String p, String id, String v, int i, int l)
30    {
31      this(p, id, v, (short) -1, (short) 0, i, l);
32    }
33  
34    /**
35      * Sets all member variables.
36      *
37      * @param p  The new discrete feature's package.
38      * @param id The new discrete feature's identifier.
39      * @param v  The new discrete feature's value.
40      * @param vi The index corresponding to the value.
41      * @param t  The total allowable values for this feature.
42      * @param i  The index of this feature in the returned array.
43      * @param l  The length of the array this feature is contained in.
44     **/
45    public DiscreteArrayFeature(String p, String id, String v, short vi,
46                                short t, int i, int l)
47    {
48      super(p, id, v, vi, t);
49      arrayIndex = i;
50      arrayLength = l;
51    }
52  
53  
54    /** Returns the array index of this feature. */
55    public int getArrayIndex() { return arrayIndex; }
56  
57  
58    /** Returns the length of the feature array that this feature comes from. */
59    public int getArrayLength() { return arrayLength; }

62    /**
63      * Determines if this feature comes from an array.
64      *
65      * @return <code>true</code>.
66     **/
67    public boolean fromArray() { return true; }
68  
69  
70    /**
71      * If this feature is an array feature, call this method to set its array
72      * length; otherwise, this method has no effect.
73      *
74      * @param l  The new length.
75     **/
76    public void setArrayLength(int l) { arrayLength = l; }

79    /**
80      * Create a feature representing the conjunction of this feature with the
81      * given argument feature.
82      *
83      * @param f  The feature to conjunct with.

85      * @return   A feature representing the conjunction of this feature and
86      *           <code>f</code>.
87     **/
88    public Feature conjunction(Feature f, Classifier c)
89    {
90      return f.conjunctWith(this, c);
91    }

103   protected Feature conjunctWith(DiscreteFeature f, Classifier c)
104   {
105     return f.conjunctWith(this, c);

109   /**
110     * Create a feature representing the conjunction of this feature with the
111     * given argument feature.
112     *
113     * @param f  The feature to conjunct with.
114     * @param c  The classifier producing the resulting feature.
115     * @return   A feature representing the conjunction of this feature and
116     *           <code>f</code>.
117    **/
118   protected Feature conjunctWith(DiscreteArrayFeature f, Classifier c)
119   {
120     String[] allowable = c.allowableValues();
121     String result = f.getValue() + "&" + value;
122     int index = f.getArrayIndex() * arrayLength + arrayIndex;
123     int length = f.getArrayLength() * arrayLength;
124     if (allowable.length == 0)
125       return
126         new DiscreteArrayFeature(c.containingPackage, c.name, result, index,
127                                  length);
128     short v = c.valueIndexOf(result);
129     return
130       new DiscreteArrayFeature(c.containingPackage, c.name, result, v,
131                                (short) allowable.length, index, length);
132   }

135   /**
136     * Create a feature representing the conjunction of this feature with the
137     * given argument feature.
138     *
139     * @param f  The feature to conjunct with.
140     * @param c  The classifier producing the resulting feature.
141     * @return   A feature representing the conjunction of this feature and
142     *           <code>f</code>.
143    **/
144   protected Feature conjunctWith(RealFeature f, Classifier c)
145   {
146     String id = c.name + "_" + arrayIndex + "_" + value;
147     double result = f.getValue();
148     return new RealFeature(c.containingPackage, id, result);
149   }

152   /**
153     * Create a feature representing the conjunction of this feature with the
154     * given argument feature.
155     *
156     * @param f  The feature to conjunct with.
157     * @param c  The classifier producing the resulting feature.
158     * @return   A feature representing the conjunction of this feature and
159     *           <code>f</code>.
160    **/
161   protected Feature conjunctWith(RealArrayFeature f, Classifier c)
162   {
163     String id = c.name + "_" + arrayIndex + "_" + value;
164     return
165       new RealArrayFeature(c.containingPackage, id, f.getValue(),
166                            f.getArrayIndex(), f.getArrayLength());
167   }

170   /**
171     * The hash code of a <code>DiscreteArrayFeature</code> is the sum of the
172     * hash codes of the containing package, the identifier, the value and the
173     * array index.
174     *
175     * @return The hash code of this <code>Feature</code>.
176    **/
177   public int hashCode()
178   {
179     return super.hashCode() + new Integer(arrayIndex).hashCode();
180   }

183   /**
184     * Two <code>DiscreteArrayFeature</code>s are equivalent when their
185     * containing packages, identifiers, indices, and values are equivalent.
186     *
187     * @param o  The object with which to compare this <code>Feature</code>.
188     * @return   True iff the parameter is an equivalent <code>Feature</code>.
189    **/
190   public boolean equals(Object o)
191   {
192     return
193       super.equals(o) && arrayIndex == ((DiscreteArrayFeature) o).arrayIndex;
194   }

197   /**
198     * Used to sort features via a lexicographic comparison of their packages
199     * and identifiers.
200     *
201     * @param o  An object to compare with.
202     * @return   The same thing that a <code>String</code> comparison of the
203     *           concatenations of the features' packages and identifiers would
204     *           return.
205    **/
206   public int compareTo(Object o)
207   {
208     if (!(o instanceof DiscreteArrayFeature)) return -1;
209     DiscreteArrayFeature f = (DiscreteArrayFeature) o;
210     int c = containingPackage.compareTo(f.containingPackage);
211     if (c != 0) return c;
212     c = identifier.compareTo(f.identifier);
213     if (c != 0) return c;
214     c = arrayIndex - f.arrayIndex;
215     if (c != 0) return c;
216     return value.compareTo(f.value);
217   }

220   /**
221     * The string representation of a discrete feature is its identifier
222     * followed by its array index surrounded by square brackets and its value
223     * surrounded by curly braces.
224     *
225     * @return The string representation of this <code>Feature</code>.
226    **/
227   public String toString()
228   {
229     return containingPackage + ":" + identifier + "[" + arrayIndex + "]("
230            + value + ")";
231   }
