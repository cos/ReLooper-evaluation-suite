8   /**
9     * To be run after <code>SemanticAnalysis</code>, this pass compiles the list
10    * of files created by the LBJ2 compiler and removes them.
11    *
12    * @see    LBJ2.SemanticAnalysis
13    * @author Nick Rizzolo 
14   **/
15  public class Clean extends Pass
16  {
17    /** The list of files to be deleted. */
18    private LinkedList files;
19    /** The path at which a cleanable Java file will be found. */
20    private String javaPath;
21    /** The path at which a cleanable class file will be found. */
22    private String classPath;
23  
24  
25    /**
26      * Instantiates a pass that runs on an entire <code>AST</code>.
27      *
28      * @param ast  The program to run this pass on.
29     **/
30    public Clean(AST ast) { super(ast); }
31  
32  
33    /**
34      * Runs this pass on all nodes of the indicated type.
35      *
36      * @param ast  The node to process.
37     **/
38    public void run(AST ast)
39    {
40      files = new LinkedList();
41  
42      if (Main.generatedSourceDirectory != null)
43        javaPath = Main.generatedSourceDirectory + File.separator;
44      else javaPath = "";
45  
46      if (Main.classDirectory != null)
47        classPath = Main.classDirectory + File.separator;
48      else classPath = "";
49  
50      runOnChildren(ast);
51  
52      for (Iterator I = files.iterator(); I.hasNext(); )
53      {
54        Object o = I.next();
55        if (o instanceof String)
56        {
57          String name = (String) o;
58          File f = new File(name);
59          if (f.exists() && !f.delete())
60            reportError(0, "Could not delete '" + name + "'.");
61        }
62        else
63        {
64          String[] names = (String[]) o;
65          for (int i = 0; i < names.length; ++i)
66          {
67            File f = new File(names[i]);
68            if (f.exists() && !f.delete())
69              reportError(0, "Could not delete '" + names[i] + "'.");
70          }
71        }
72      }
73    }
74  
75  
76    /**
77      * Adds the default files (<code>*.java</code> and <code>*.class</code>)
78      * to the remove list.
79      *
80      * @param n  The node to add files with respect to.
81     **/
82    protected void defaultFiles(CodeGenerator n)
83    {
84      files.add(javaPath + n.getName() + ".java");
85      files.add(classPath + n.getName() + ".class");
86    }
87  
88  
89    /**
90      * Adds files generated for anonymous classes associated with a particular
91      * named classifier to the remove list.
92      *
93      * @param name The name of the classifier with which anonymous classes may
94      *             be associated.
95     **/
96    protected void anonymousFiles(Name name)
97    {
98      final String prefix = Main.sourceFileBase + "$" + name;
99      String directory = javaPath;
100     if (directory.equals("")) directory = System.getProperty("user.dir");
101     else directory = directory.substring(0, directory.length() - 1);
102 
103     String[] toAdd =
104       new File(directory).list(
105         new FilenameFilter()
106         {
107           public boolean accept(File directory, String n)
108           {
109             return n.startsWith(prefix);
110           }
111         });
112 
113     for (int i = 0; i < toAdd.length; ++i)
114       toAdd[i] = directory + File.separator + toAdd[i];
115     files.add(toAdd);
116 
117     directory = classPath;
118     if (directory.equals("")) directory = System.getProperty("user.dir");
119     else directory = directory.substring(0, directory.length() - 1);
120 
121     toAdd =
122       new File(directory).list(
123         new FilenameFilter()
124         {
125           public boolean accept(File directory, String n)
126           {
127             return n.startsWith(prefix);
128           }
129         });
130 
131     for (int i = 0; i < toAdd.length; ++i)
132       toAdd[i] = directory + File.separator + toAdd[i];
133     files.add(toAdd);
134   }

136   /**
137     * Runs this pass on all nodes of the indicated type.
138     *
139     * @param cn The node to process.
140    **/
141   public void run(ClassifierName cn)
142   {
143     if (cn.name == cn.referent) return;
144     defaultFiles(cn);
145   }

149     * Runs this pass on all nodes of the indicated type.
150     *
151     * @param cc The node to process.
152    **/
153   public void run(CodedClassifier cc) { defaultFiles(cc); }

188   /**
189     * Runs this pass on all nodes of the indicated type.
190     *
191     * @param lce  The node to process.
192    **/
193   public void run(LearningClassifierExpression lce)
194   {
195     defaultFiles(lce);
196     anonymousFiles(lce.name);
197     files.add(classPath + lce.name + ".lc");
198     files.add(javaPath + lce.name + ".ex");
199     files.add(javaPath + lce.name + ".lex");
200   }
