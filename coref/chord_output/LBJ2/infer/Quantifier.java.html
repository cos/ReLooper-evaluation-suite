6   /**
7     * A quantifier is a first order constraint parameterized by an object taken
8     * from a Java <code>Collection</code> of objects.
9     *
10    * @author Nick Rizzolo
11   **/
12  public abstract class Quantifier extends FirstOrderConstraint
13  {
14    /** The name of the quantification variable. */
15    protected String quantificationVariable;
16    /** The collection of objects to iterate over. */
17    protected Collection collection;
18    /** The constraint being quantified. */
19    protected FirstOrderConstraint constraint;
20    /**
21      * A list of the objects stored in the quantification variables of
22      * enclosing quantifiers.
23     **/
24    protected Vector enclosingQuantificationSettings;
25    /**
26      * The implementation of the functions that compute any parameters this
27      * quantifier may have.
28     **/
29    protected QuantifierArgumentReplacer replacer;
30  
31  
32    /**
33      * Initializing constructor.
34      *
35      * @param q    The name of the quantification variable.
36      * @param col  The collection of objects to iterate over.
37      * @param con  The constraint being quantified.
38     **/
39    public Quantifier(String q, Collection col, FirstOrderConstraint con)
40    {
41      this(q, col, con, null);
42    }
43  
44    /**
45      * This constructor specifies a variable setter for when this quantifier is
46      * itself quantified.
47      *
48      * @param q    The name of the quantification variable.
49      * @param col  The collection of objects to iterate over.
50      * @param con  The constraint being quantified.
51      * @param qar  The variable setter.
52     **/
53    public Quantifier(String q, Collection col, FirstOrderConstraint con,
54                      QuantifierArgumentReplacer qar)
55    {
56      quantificationVariable = q;
57      collection = col;
58      constraint = con;
59      replacer = qar;
60    }
61  
62  
63    /**
64      * Returns the children of this constraint in an array.
65      *
66      * @return The children of this constraint in an array.
67     **/
68    public Constraint[] getChildren()
69    {
70      return new FirstOrderConstraint[]{ constraint };
71    }
72  
73  
74    /**
75      * Makes sure that the <code>enclosingQuantificationSettings</code> vector
76      * exists, then adds a place holder for this quantifier's quantification
77      * variable setting.
78      *
79      * @return The index of this quantifier's quantification variable.
80     **/
81    protected int initialize()
82    {
83      if (enclosingQuantificationSettings == null)
84        enclosingQuantificationSettings = new Vector();
85      enclosingQuantificationSettings.add(null);
86      return enclosingQuantificationSettings.size() - 1;
87    }
88  
89  
90    /**
91      * Sets the variable map object stored in this object to the given
92      * argument; also instantiates all quantified variables and stores them in
93      * the map.
94      *
95      * @param m  The map in which to find unique copies of the variables.
96     **/
97    public void consolidateVariables(AbstractMap m)
98    {
99      int index = initialize();
100 
101     for (Iterator I = collection.iterator(); I.hasNext(); )
102     {
103       enclosingQuantificationSettings.set(index, I.next());
104       constraint.setQuantificationVariables(enclosingQuantificationSettings);
105       constraint.consolidateVariables(m);
106     }
107 
108     enclosingQuantificationSettings.removeElementAt(index);
109   }
110 
111 
112   /**
113     * The hash code of a <code>Quantifier</code> is the sum of the hash codes
114     * of its children plus three.
115     *
116     * @return The hash code for this <code>Quantifier</code>.
117    **/
118   public int hashCode()
119   {
120     int result = constraint.hashCode();
121     if (replacer != null) result += replacer.hashCode();
122     else result += collection.hashCode();
123     return result;
124   }
125 
126 
127   /**
128     * Two <code>Quantifier</code>s are equivalent when their children are
129     * equivalent.
130     *
131     * @return <code>true</code> iff the argument is an equivalent
132     *         <code>Quantifier</code>.
133    **/
134   public boolean equals(Object o)
135   {
136     if (!(o instanceof Quantifier)) return false;
137     Quantifier q = (Quantifier) o;
138     return replacer == q.replacer
139            && (replacer != null
140                || replacer == null && collection.equals(q.collection))
141            && constraint.equals(q.constraint);
142   }
143 }
