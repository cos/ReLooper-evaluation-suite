4   /**
5     * All classes for representing propositional constraints are derived from
6     * this base class.  A propositional constraint is:
7     *
8     * <ul>
9     *   <li> The constant <code>true</code> or the constant <code>false</code>.
10    *   <li> A variable name, which is an assertion of that variable's truth.
11    *   <li> The negation of a propositional constraint.
12    *   <li> The conjunction of two propositional constraints.
13    *   <li> The disjunction of two propositional constraints.
14    *   <li> An implication between two propositional constraints.
15    *   <li> A double implication between two propositional constraints.
16    * </ul>
17   **/
18  public abstract class PropositionalConstraint extends Constraint
19                                                implements Cloneable
20  {
21    /**
22      * Produces a new, logically simplified version of this constraint,
23      * preserving variable consolidation.
24      *
25      * @see    Constraint#consolidateVariables(java.util.AbstractMap)
26      * @return A logically simplified version of this constraint.
27     **/
28    abstract public PropositionalConstraint simplify();
29  
30  
31    /**
32      * Produces a new propositional constraint equivalent to this constraint
33      * and that contains no negated constraints other than variables.
34      *
35      * @return A constraint representing the negation of this constraint.
36     **/
37    abstract public PropositionalConstraint negate();
38  
39  
40    /**
41      * Produces a new, logically simplified version of this constraint in
42      * conjunctive normal form (CNF).
43      *
44      * @return The conjunctive normal form of this constraint.
45     **/
46    abstract public PropositionalConstraint CNF();
47  
48  
49    /**
50      * Produces a new, logically simplified version of this constraint in
51      * disjunctive normal form (DNF).
52      *
53      * @return The disjunctive normal form of this constraint.
54     **/
55    abstract public PropositionalConstraint DNF();
56  
57  
58    /**
59      * Compares topology to determine if this constraint is more general than
60      * the given constraint; <i>note: this method is not required to be correct
61      * when it answers <code>false</code></i>.
62      *
63      * @param c  The given constraint.
64      * @return   <code>true</code> if a topological analysis determined that
65      *           this constraint is more general than the given constraint.
66     **/
67    abstract public boolean moreGeneralThan(PropositionalConstraint c);

75      * @param c  The given implication.
76      * @return   <code>true</code> if a topological analysis determined that

78     **/
79    abstract public boolean moreSpecificThan(PropositionalImplication c);

167   /**
168     * Creates a string respresentation of this constraint using the string
169     * representations of the objects involved.  This method employs the
170     * <code>write(StringBuffer)</code> method to compute its output.
171    **/
172   public String toString()
173   {
174     StringBuffer result = new StringBuffer();
175     write(result);
176     return result.toString();
177   }

189   /**
190     * This method returns a shallow clone.
191     *
192     * @return A shallow clone.
193    **/
194   public Object clone()
195   {
196     Object clone = null;
197     try { clone = super.clone(); }
198     catch (Exception e)
199     {
200       System.err.println("Error cloning " + getClass().getName() + ":");
201       e.printStackTrace();
202       System.exit(1);
203     }
204 
205     return clone;
206   }
