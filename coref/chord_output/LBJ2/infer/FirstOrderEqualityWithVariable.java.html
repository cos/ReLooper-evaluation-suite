7   /**
8     * Represents the comparison of two classifier applications.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class FirstOrderEqualityWithVariable extends FirstOrderEquality
13  {
14    /** The variable on the left of the equality. */
15    protected FirstOrderVariable left;
16    /** The classifier application on the right of the equality. */
17    protected FirstOrderVariable right;

21      * Initializing constructor.

25      * @param r  The right classifier application.

27    public FirstOrderEqualityWithVariable(boolean e, FirstOrderVariable l,
28                                          FirstOrderVariable r)

30      this(e, l, r, null);

33    /**
34      * This constructor specifies a variable setter for when this equality is
35      * quantified.
36      *
37      * @param e    Indicates whether this is an equality or an inequality.
38      * @param l    The left classifier application.
39      * @param r    The right classifier application.
40      * @param ear  An argument replacer.
41     **/
42    public FirstOrderEqualityWithVariable(boolean e, FirstOrderVariable l,
43                                          FirstOrderVariable r,
44                                          EqualityArgumentReplacer ear)
45    {
46      super(e, ear);
47      left = l;
48      right = r;
49    }

52    /**
53      * Replaces all unquantified variables with the unique copy stored as a
54      * value of the given map; also instantiates all quantified variables and
55      * stores them in the given map.
56      *
57      * @param m  The map in which to find unique copies of the variables.
58     **/
59    public void consolidateVariables(java.util.AbstractMap m)
60    {
61      variableMap = m;
62      if (m.containsKey(left)) left = (FirstOrderVariable) m.get(left);
63      else m.put(left, left);
64      if (m.containsKey(right)) right = (FirstOrderVariable) m.get(right);
65      else m.put(right, right);
66    }

69    /**
70      * This method sets the given quantification variables to the given object
71      * references and evaluates the expressions involving those variables in
72      * this constraint's <code>FirstOrderEquality</code> children.
73      *
74      * @param o  The new object references for the enclosing quantification
75      *           variables, in order of nesting.
76     **/
77    public void setQuantificationVariables(Vector o)
78    {
79      if (replacer == null)
80      {
81        System.err.println(
82            "LBJ ERROR: Attempting to set quantification variable in "
83            + "FirstOrderEqualityWithVariable with no variable setter "
84            + "implementation provided.");
85        System.exit(1);
86      }
87  
88      replacer.setQuantificationVariables(o);
89  
90      if (!replacer.leftConstant)
91      {
92        left =
93          new FirstOrderVariable(left.getClassifier(),
94                                 replacer.getLeftObject());
95        if (variableMap != null && variableMap.containsKey(left))
96          left = (FirstOrderVariable) variableMap.get(left);
97      }
98  
99      if (!replacer.rightConstant)
100     {
101       right = new FirstOrderVariable(right.getClassifier(),
102                                      replacer.getRightObject());
103       if (variableMap != null && variableMap.containsKey(right))
104         right = (FirstOrderVariable) variableMap.get(right);
105     }
106   }

109   /** Determines whether the constraint is satisfied. */
110   public boolean evaluate()
111   {
112     return equality == left.getValue().equals(right.getValue());
113   }

116   /**
117     * Transforms this first order constraint into a propositional constraint.
118     *
119     * @return The propositionalized constraint.
120    **/
121   public PropositionalConstraint propositionalize()
122   {
123     Score[] leftScores = left.getScores().toArray();
124     Score[] rightScores = right.getScores().toArray();
125     if (leftScores.length == 0 || rightScores.length == 0)
126       return new PropositionalConstant(false);
127     if (leftScores.length == 1 && rightScores.length == 1)
128       return
129         new PropositionalConstant(leftScores[0].value
130                                   .equals(rightScores[0].value));
131 
132     PropositionalVariable[] leftVariables =
133       new PropositionalVariable[leftScores.length];
134     PropositionalVariable[] rightVariables =
135       new PropositionalVariable[rightScores.length];
136 
137     int size = 0;
138     for (int i = 0; i < leftScores.length; ++i)
139     {
140       boolean found = false;
141       for (int j = 0; j < rightScores.length && !found; ++j)
142       {
143         if (!leftScores[i].value.equals(rightScores[j].value)) continue;
144         found = true;
145         leftVariables[size] =
146           new PropositionalVariable(left.getClassifier(), left.getExample(),
147                                     leftScores[i].value);
148         rightVariables[size] =
149           new PropositionalVariable(right.getClassifier(), right.getExample(),
150                                     rightScores[j].value);
151         ++size;
152       }
153     }
154 
155     if (size == 0) return new PropositionalConstant(false);
156 
157     if (equality && size == leftScores.length && size == rightScores.length)
158       --size;
159 
160     PropositionalConstraint rightVariable = rightVariables[0];
161     if (!equality) rightVariable = new PropositionalNegation(rightVariable);
162     PropositionalConstraint result =
163       new PropositionalDisjunction(
164           new PropositionalNegation(leftVariables[0]),
165           rightVariable);
166     if (equality)
167       result =
168         new PropositionalConjunction(
169             result,
170             new PropositionalDisjunction(
171               new PropositionalNegation(rightVariable),
172               leftVariables[0]));
173 
174     for (int i = 1; i < size; ++i)
175     {
176       rightVariable = rightVariables[i];
177       if (!equality) rightVariable = new PropositionalNegation(rightVariable);
178       result =
179         new PropositionalConjunction(
180             result,
181             new PropositionalDisjunction(
182               new PropositionalNegation(leftVariables[i]),
183               rightVariable));
184       if (equality)
185         result =
186           new PropositionalConjunction(
187               result,
188               new PropositionalDisjunction(
189                 new PropositionalNegation(rightVariable),
190                 leftVariables[i]));
191     }
192 
193     return result;
194   }

197   /**
198     * The hash code of a <code>FirstOrderEqualityWithVariable</code> is the
199     * sum of the hash codes of its children plus 2.
200     *
201     * @return The hash code for this
202     *         <code>FirstOrderEqualityWithVariable</code>.
203    **/
204   public int hashCode()
205   {
206     if (replacer != null) return replacer.hashCode();
207     return left.hashCode() + right.hashCode() + 2;
208   }

211   /**
212     * Two <code>FirstOrderEqualityWithVariable</code>s are equivalent when
213     * their children are equivalent in either order.
214     *
215     * @return <code>true</code> iff the argument is a
216     *         <code>FirstOrderEqualityWithVariable</code> involving the same
217     *         variables.
218    **/
219   public boolean equals(Object o)
220   {
221     if (!(o instanceof FirstOrderEqualityWithVariable)) return false;
222     FirstOrderEqualityWithVariable n = (FirstOrderEqualityWithVariable) o;
223     return replacer == n.replacer
224            && (replacer != null
225                || replacer == null
226                   && (left.equals(n.left) && right.equals(n.right)
227                       || left.equals(n.right) && right.equals(n.left)));
228   }
