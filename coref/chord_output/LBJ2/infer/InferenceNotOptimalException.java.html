4   /**
5     * Exceptions of this type are thrown by the {@link ILPInference} class when
6     * the selected {@link ILPSolver} did not successfully find the optimal
7     * solution to the inference problem.  Instances of this class contain a
8     * reference to the {@link ILPSolver} instance so that the user can, for
9     * instance, call the {@link ILPSolver#write(java.lang.StringBuffer)} method.
10    *
11    * @author Nick Rizzolo
12   **/
13  public class InferenceNotOptimalException extends Exception
14  {
15    /** The ILP algorithm and problem representation that failed. */
16    private ILPSolver solver;
17    /** The head object of the inference problem. */
18    private Object head;
19  
20  
21    /**
22      * Initializing constructor.
23      *
24      * @param solver The ILP algorithm and problem representation that failed.
25      * @param head   The head object of the inference problem.
26     **/
27    public InferenceNotOptimalException(ILPSolver solver, Object head)
28    {
29      super(
30          "Failed to solve inference problem with the following head object: "
31          + head);
32      this.solver = solver;
33      this.head = head;
34    }
35  
36  
37    /** Retrieves the ILP problem instance, {@link #solver}. */
38    public ILPSolver getSolver() { return solver; }
39  
40  
41    /** Retrieves the head object, {@link #head}. */
42    public Object getHead() { return head; }
43  }
