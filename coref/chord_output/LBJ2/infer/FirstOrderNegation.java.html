3   import java.util.Vector;
4   
5   
6   /**
7     * Represents the negation operator applied to a first order constraint.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class FirstOrderNegation extends FirstOrderConstraint
12  {
13    /** The constraint that the negation is applied to. */
14    protected FirstOrderConstraint constraint;
15  
16  
17    /**
18      * Initializing constructor.
19      *
20      * @param c  The constraint to negate.
21     **/
22    public FirstOrderNegation(FirstOrderConstraint c) { constraint = c; }
23  
24  
25    /**
26      * Replaces all unquantified variables with the unique copy stored as a
27      * value of the given map; also instantiates all quantified variables and
28      * stores them in the given map.
29      *
30      * @param m  The map in which to find unique copies of the variables.
31     **/
32    public void consolidateVariables(java.util.AbstractMap m)
33    {
34      constraint.consolidateVariables(m);
35    }
36  
37  
38    /**
39      * This method sets the given quantification variables to the given object
40      * references and evaluates the expressions involving those variables in
41      * this constraint's <code>FirstOrderEquality</code> children.
42      *
43      * @param o  The new object references for the enclosing quantification
44      *           variables, in order of nesting.
45     **/
46    public void setQuantificationVariables(Vector o)
47    {
48      constraint.setQuantificationVariables(o);
49    }
50  
51  
52    /**
53      * Returns the children of this constraint in an array.
54      *
55      * @return The children of this constraint in an array.
56     **/
57    public Constraint[] getChildren() { return new FirstOrderConstraint[0]; }
58  
59  
60    /** Determines whether the constraint is satisfied. */
61    public boolean evaluate() { return !constraint.evaluate(); }
62  
63  
64    /**
65      * Transforms this first order constraint into a propositional constraint.
66      *
67      * @return The propositionalized constraint.
68     **/
69    public PropositionalConstraint propositionalize()
70    {
71      return new PropositionalNegation(constraint.propositionalize());
72    }
73  
74  
75    /**
76      * The hash code of a <code>FirstOrderNegation</code> is the hash code of
77      * its child constraint plus 1.
78      *
79      * @return The hash code for this <code>FirstOrderNegation</code>.
80     **/
81    public int hashCode() { return constraint.hashCode() + 1; }
82  
83  
84    /**
85      * Two <code>FirstOrderNegation</code>s are equivalent when their
86      * constraints are equivalent.
87      *
88      * @return <code>true</code> iff the argument is a
89      *         <code>FirstOrderNegation</code> of the same constraint.
90     **/
91    public boolean equals(Object o)
92    {
93      if (!(o instanceof FirstOrderNegation)) return false;
94      FirstOrderNegation n = (FirstOrderNegation) o;
95      return constraint.equals(n.constraint);
96    }
97  
98  
99    /**
100     * Calls the appropriate <code>visit(&middot;)</code> method of the given
101     * <code>Inference</code> for this <code>Constraint</code>, as per the
102     * visitor pattern.
103     *
104     * @param infer  The inference visiting this constraint.
105    **/
106   public void runVisit(Inference infer) { infer.visit(this); }
107 }
