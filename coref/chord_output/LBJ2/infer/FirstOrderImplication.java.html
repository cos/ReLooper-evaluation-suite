4   /**
5     * Represents an implication between two first order constraints.
6     *
7     * @author Nick Rizzolo
8    **/
9   public class FirstOrderImplication extends FirstOrderBinaryConstraint
10  {
11    /**
12      * Initializing constructor.
13      *
14      * @param l  The constraint on the left of the operator.
15      * @param r  The constraint on the right of the operator.
16     **/
17    public FirstOrderImplication(FirstOrderConstraint l, FirstOrderConstraint r)
18    {
19      super(l, r);
20    }
21  
22  
23    /** Determines whether the constraint is satisfied. */
24    public boolean evaluate() { return !left.evaluate() || right.evaluate(); }
25  
26  
27    /**
28      * Transforms this first order constraint into a propositional constraint.
29      *
30      * @return The propositionalized constraint.
31     **/
32    public PropositionalConstraint propositionalize()
33    {
34      return new PropositionalImplication(left.propositionalize(),
35                                          right.propositionalize());
36    }
37  
38  
39    /**
40      * The hash code of a <code>FirstOrderImplication</code> is the sum of the
41      * hash codes of its children plus two.
42      *
43      * @return The hash code for this <code>FirstOrderImplication</code>.
44     **/
45    public int hashCode() { return left.hashCode() + right.hashCode() + 2; }
46  
47  
48    /**
49      * Two <code>FirstOrderImplication</code>s are equivalent when they are
50      * topologically equivalent.
51      *
52      * @return <code>true</code> iff the argument is an equivalent
53      *         <code>FirstOrderImplication</code>.
54     **/
55    public boolean equals(Object o)
56    {
57      if (!(o instanceof FirstOrderImplication)) return false;
58      FirstOrderImplication i = (FirstOrderImplication) o;
59      return left.equals(i.left) && right.equals(i.right);
60    }
61  
62  
63    /**
64      * Calls the appropriate <code>visit(&middot;)</code> method of the given
65      * <code>Inference</code> for this <code>Constraint</code>, as per the
66      * visitor pattern.
67      *
68      * @param infer  The inference visiting this constraint.
69     **/
70    public void runVisit(Inference infer) { infer.visit(this); }
71  }
