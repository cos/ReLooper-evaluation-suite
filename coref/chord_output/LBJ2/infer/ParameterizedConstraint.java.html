6   /**
7     * This class represents an LBJ constraint as it appears in a source file.
8     * When given its input object, an object of this class generates objects of
9     * type <code>FirstOrderConstraint</code>.
10    * <code>ParameterizedConstraint</code>s are also Boolean
11    * <code>Classifier</code>s.
12    *
13    * <p> <code>ParameterizedConstraint</code> depends on extending classes to
14    * override the <code>discreteValue(Object)</code> method so that it returns
15    * <code>"true"</code> or <code>"false"</code> as appropriate.  When this is
16    * done, there is no need to override the <code>classify(Object)</code>
17    * method.
18    *
19    * @author Nick Rizzolo
20   **/
21  public abstract class ParameterizedConstraint extends Classifier
22  {
23    /** Default constructor. */
24    public ParameterizedConstraint() { }
25  
26    /**
27      * Initializes the name.
28      *
29      * @param n  The name of this constraint.
30     **/
31    public ParameterizedConstraint(String n) { super(n); }
32  
33  
34    /**
35      * This method makes one or more decisions about a single object, returning
36      * those decisions as <code>Feature</code>s in a vector.
37      *
38      * @param o  The object to make decisions about.
39      * @return   A vector of <code>Feature</code>s about the input object.
40     **/
41    public FeatureVector classify(Object o)
42    {
43      String value = discreteValue(o);
44      short index = (short) (value.equals("true") ? 1 : 0);
45      return
46        new FeatureVector(
47            new DiscreteFeature(containingPackage, name, value, index,
48                                (short) 2));
49    }
50  
51  
52    /**
53      * Returns the array of allowable values that a feature returned by this
54      * classifier may take.  If the array has length 0, it means either that
55      * the feature has discrete type and allowable values were not specified or
56      * that the feature has real or mixed type.  The default return value of
57      * this method is a 0 length array.<br><br>
58      *
59      * This method should be overridden by derived classes.
60      *
61      * @return The allowable values that a feature returned by this classifier
62      *         may take.
63     **/
64    public String[] allowableValues() { return DiscreteFeature.BooleanValues; }
65  
66  
67    /**
68      * This method builds a first order constraint based on the given input
69      * object.
70      *
71      * @param o  The object to build a constraint with respect to.
72      * @return   A first order constraint.
73     **/
74    public abstract FirstOrderConstraint makeConstraint(Object o);
75  }
