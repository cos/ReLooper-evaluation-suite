6   /**
7     * Represents the disjunction of first order constraints.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class FirstOrderDisjunction extends FirstOrderNAryConstraint
12  {
13    /**
14      * If either of the arguments is itself a
15      * <code>FirstOrderDisjunction</code>, its contents are flattened into
16      * this <code>FirstOrderDisjunction</code>.
17      *
18      * @param c1 One constraint to disjunct.
19      * @param c2 Another constraint to disjunct.
20     **/
21    public FirstOrderDisjunction(FirstOrderConstraint c1,
22                                 FirstOrderConstraint c2)
23    {
24      add(c1);
25      add(c2);
26    }
27  
28  
29    /**
30      * If the given constraint has the same type as this constraint, its terms
31      * are merged into this constraint; otherwise, it is added as a new term.
32      *
33      * @param c  The constraint to add.
34     **/
35    public void add(FirstOrderConstraint c)
36    {
37      if (c instanceof FirstOrderDisjunction)
38      {
39        Iterator I = ((FirstOrderDisjunction) c).children.iterator();
40        while (I.hasNext()) add((FirstOrderConstraint) I.next());
41      }
42      else children.add(c);
43    }
44  
45  
46    /** Determines whether the constraint is satisfied. */
47    public boolean evaluate()
48    {
49      for (Iterator I = children.iterator(); I.hasNext(); )
50        if (((FirstOrderConstraint) I.next()).evaluate()) return true;
51      return false;
52    }
53  
54  
55    /**
56      * Transforms this first order constraint into a propositional constraint.
57      *
58      * @return The propositionalized constraint.
59     **/
60    public PropositionalConstraint propositionalize()
61    {
62      if (children.size() == 0) return new PropositionalConstant(true);
63  
64      FirstOrderConstraint[] c =
65        (FirstOrderConstraint[]) children.toArray(new FirstOrderConstraint[0]);
66      if (c.length == 1) return c[0].propositionalize();
67  
68      PropositionalDisjunction result =
69        new PropositionalDisjunction(c[0].propositionalize(),
70                                     c[1].propositionalize());
71      for (int i = 2; i < c.length; ++i)
72        result = new PropositionalDisjunction(result, c[i].propositionalize());
73  
74      return result;
75    }
76  
77  
78    /**
79      * The hash code of a <code>FirstOrderDisjunction</code> is the sum of
80      * the hash codes of its children.
81      *
82      * @return The hash code for this <code>FirstOrderDisjunction</code>.
83     **/
84    public int hashCode()
85    {
86      int result = 0;
87      for (Iterator I = children.iterator(); I.hasNext(); )
88        result += I.next().hashCode();
89      return result;
90    }
91  
92  
93    /**
94      * Two <code>FirstOrderDisjunction</code>s are equivalent when they are
95      * topologically equivalent, respecting the associativity and commutivity
96      * of disjunction.
97      *
98      * @return <code>true</code> iff the argument is an equivalent
99      *         <code>FirstOrderDisjunction</code>.
100    **/
101   public boolean equals(Object o)
102   {
103     if (!(o instanceof FirstOrderDisjunction)) return false;
104     FirstOrderDisjunction d = (FirstOrderDisjunction) o;
105     return children.equals(d.children);
106   }

111     * <code>Inference</code> for this <code>Constraint</code>, as per the
112     * visitor pattern.
113     *
114     * @param infer  The inference visiting this constraint.
115    **/
116   public void runVisit(Inference infer) { infer.visit(this); }
