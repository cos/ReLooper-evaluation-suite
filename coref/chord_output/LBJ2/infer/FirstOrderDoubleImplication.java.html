4   /**
5     * Represents a double implication between two first order constraints.
6     *
7     * @author Nick Rizzolo
8    **/
9   public class FirstOrderDoubleImplication extends FirstOrderBinaryConstraint
10  {
11    /**
12      * Initializing constructor.
13      *
14      * @param l  The constraint on the left of the operator.
15      * @param r  The constraint on the right of the operator.
16     **/
17    public FirstOrderDoubleImplication(FirstOrderConstraint l,
18                                       FirstOrderConstraint r)
19    {
20      super(l, r);
21    }
22  
23  
24    /** Determines whether the constraint is satisfied. */
25    public boolean evaluate() { return left.evaluate() == right.evaluate(); }
26  
27  
28    /**
29      * Transforms this first order constraint into a propositional constraint.
30      *
31      * @return The propositionalized constraint.
32     **/
33    public PropositionalConstraint propositionalize()
34    {
35      return new PropositionalDoubleImplication(left.propositionalize(),
36                                                right.propositionalize());
37    }
38  
39  
40    /**
41      * The hash code of a <code>FirstOrderDoubleImplication</code> is the sum
42      * of the hash codes of its children plus three.
43      *
44      * @return The hash code for this <code>FirstOrderDoubleImplication</code>.
45     **/
46    public int hashCode() { return left.hashCode() + right.hashCode() + 3; }
47  
48  
49    /**
50      * Two <code>FirstOrderDoubleImplication</code>s are equivalent when
51      * they are topologically equivalent, respecting the commutativity of
52      * double implication.
53      *
54      * @return <code>true</code> iff the argument is an equivalent
55      *         <code>FirstOrderDoubleImplication</code>.
56     **/
57    public boolean equals(Object o)
58    {
59      if (!(o instanceof FirstOrderDoubleImplication)) return false;
60      FirstOrderDoubleImplication i = (FirstOrderDoubleImplication) o;
61      return left.equals(i.left) && right.equals(i.right)
62             || left.equals(i.right) && right.equals(i.left);
63    }
64  
65  
66    /**
67      * Calls the appropriate <code>visit(&middot;)</code> method of the given
68      * <code>Inference</code> for this <code>Constraint</code>, as per the
69      * visitor pattern.
70      *
71      * @param infer  The inference visiting this constraint.
72     **/
73    public void runVisit(Inference infer) { infer.visit(this); }
74  }
