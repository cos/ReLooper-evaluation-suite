7   /**
8     * Represents a classifier application.  An inference algorithm may change
9     * the value returned by the classifier application when satisfying
10    * constraints.
11    *
12    * @author Nick Rizzolo
13   **/
14  public class FirstOrderVariable implements Cloneable
15  {
16    /** The classifier being applied. */
17    private Learner classifier;
18    /** The classifier is applied to this example object. */
19    private Object example;
20    /** The scores of the possible values this variable might be set to. */
21    private ScoreSet scores;
22    /** The value imposed on the classifier when applied to the example. */
23    private String value;
24  
25  
26    /**
27      * Initializing constructor.
28      *
29      * @param c  The classifier being applied.
30      * @param e  The classifier is applied to this example object.
31     **/
32    public FirstOrderVariable(Learner c, Object e)
33    {
34      classifier = c;
35      example = e;
36    }
37  
38  
39    /** Retrieves the classifier. */
40    public Learner getClassifier() { return classifier; }
41  
42  
43    /** Retrieves the example object. */
44    public Object getExample() { return example; }
45  
46  
47    /** Retrieves the value this variable currently takes. */
48    public String getValue()
49    {
50      if (value == null)
51      {
52        if (scores == null) scores = classifier.scores(example);
53        value = scores.highScoreValue();
54      }
55  
56      return value;
57    }
58  
59  
60    /**
61      * Sets the value of this variable.
62      *
63      * @param v  The new value of this variable.
64     **/
65    public void setValue(String v) { value = v; }
66  
67  
68    /**
69      * Sets the example object.
70      *
71      * @param e  The new example object.
72     **/
73    public void setExample(Object e) { example = e; }
74  
75  
76    /** Retrieves the score of the current value of this variable. */
77    public double getScore()
78    {
79      if (scores == null) scores = classifier.scores(example);
80      return scores.get(getValue());
81    }
82  
83  
84    /** Retrieves all the scores for the values this variable may take. */
85    public ScoreSet getScores()
86    {
87      if (scores == null) scores = classifier.scores(example);
88      return scores;
89    }
90  
91  
92    /** Returns a string representation of this variable. */
93    public String toString()
94    {
95      return classifier + "(" + Inference.exampleToString(example) + ") = "
96             + value;
97    }
98  
99  
100   /**
101     * The hash code of a <code>FirstOrderVariable</code> is the hash code of
102     * the string representation of the classifier plus the system's hash code
103     * for the example object.
104     *
105     * @return The hash code of this <code>FirstOrderVariable</code>.
106    **/
107   public int hashCode()
108   {
109     return classifier.toString().hashCode()
110            + System.identityHashCode(example);
111   }
112 
113 
114   /**
115     * Two <code>FirstOrderVariable</code>s are equivalent when their
116     * classifiers are equivalent and they store the same example object.
117     *
118     * @param o  The object to test equivalence with.
119     * @return   <code>true</code> iff this object is equivalent to the
120     *           argument object.
121    **/
122   public boolean equals(Object o)
123   {
124     if (!(o instanceof FirstOrderVariable)) return false;
125     FirstOrderVariable v = (FirstOrderVariable) o;
126     return classifier.equals(v.classifier) && example == v.example;
127   }

130   /**
131     * This method returns a shallow clone.
132     *
133     * @return A shallow clone.
134    **/
135   public Object clone()
136   {
137     Object clone = null;
138 
139     try { clone = super.clone(); }
140     catch (Exception e)
141     {
142       System.err.println("Error cloning FirstOrderVariable:");
143       e.printStackTrace();
144       System.exit(1);
145     }
146 
147     return clone;
148   }
