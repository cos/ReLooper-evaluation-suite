4   /**
5     * Anonymous inner classes extending this class are instantiated by the code
6     * generated by the LBJ compiler when creating
7     * <code>FirstOrderConstraint</code> representations.  The methods of this
8     * class are used to compute new values for the arguments of a quantified
9     * <code>FirstOrderEquality</code>.  Only certain value returning methods are
10    * overridden.  The others will throw
11    * <code>UnsupportedOperationException</code>s.
12    *
13    * @see    LBJ2.infer.FirstOrderConstraint
14    * @see    LBJ2.infer.FirstOrderEquality
15    * @see    java.lang.UnsupportedOperationException
16    * @author Nick Rizzolo
17   **/
18  abstract public class EqualityArgumentReplacer extends ArgumentReplacer
19  {
20    /**
21      * This flag is set if the left hand side of the equality is not
22      * quantified.
23     **/
24    public boolean leftConstant;
25    /**
26      * This flag is set if the right hand side of the equality is not
27      * quantified.
28     **/
29    public boolean rightConstant;
30  
31  
32    /**
33      * Initializing constructor.
34      *
35      * @param c  The context of the corresponding equality, except for
36      *           quantification variables.
37     **/
38    public EqualityArgumentReplacer(Object[] c)
39    {
40      super(c);
41      leftConstant = rightConstant = false;
42    }
43  
44    /**
45      * Use this constructor to indicate which of the two arguments of the
46      * equality is in fact not quantified.
47      *
48      * @param c  The context of the corresponding equality, except for
49      *           quantification variables.
50      * @param r  Set to <code>false</code> if the unquantified argument is the
51      *           left; set to <code>true</code> if the unquantified argument is
52      *           the right.
53     **/
54    public EqualityArgumentReplacer(Object[] c, boolean r)
55    {
56      super(c);
57      leftConstant = !r;
58      rightConstant = r;
59    }
60  
61  
62    /**
63      * Computes the value on the left hand side of the equality.  This method
64      * needs to be overridden if it is to be called, since by default it simply
65      * throws an <code>UnsupportedOperationException</code>.
66     **/
67    public String getLeftValue()
68    {
69      throw new UnsupportedOperationException(
70          "LBJ ERROR: getLeftValue() not supported.");
71    }
72  
73  
74    /**
75      * Computes the value on the right hand side of the equality.  This method
76      * needs to be overridden if it is to be called, since by default it simply
77      * throws an <code>UnsupportedOperationException</code>.
78     **/
79    public String getRightValue()
80    {
81      throw new UnsupportedOperationException(
82          "LBJ ERROR: getRightValue() not supported.");
83    }
84  
85  
86    /**
87      * Computes the object on the left hand side of the equality.  This method
88      * needs to be overridden if it is to be called, since by default it simply
89      * throws an <code>UnsupportedOperationException</code>.
90     **/
91    public Object getLeftObject()
92    {
93      throw new UnsupportedOperationException(
94          "LBJ ERROR: getLeftObject() not supported.");
95    }
96  
97  
98    /**
99      * Computes the object on the right hand side of the equality.  This method
100     * needs to be overridden if it is to be called, since by default it simply
101     * throws an <code>UnsupportedOperationException</code>.
102    **/
103   public Object getRightObject()
104   {
105     throw new UnsupportedOperationException(
106         "LBJ ERROR: getRightObject() not supported.");
