4   /**
5     * Represents a propositional constraint involving a binary operator.
6     *
7     * @author Nick Rizzolo
8    **/
9   public abstract class PropositionalBinaryConstraint
10                  extends PropositionalConstraint
11  {
12    /** The constraint on the left of the operator. */
13    protected PropositionalConstraint left;
14    /** The constraint on the right of the operator. */
15    protected PropositionalConstraint right;
16  
17  
18    /**
19      * Initializing constructor.
20      *
21      * @param l  The constraint on the left of the operator.
22      * @param r  The constraint on the right of the operator.
23     **/
24    public PropositionalBinaryConstraint(PropositionalConstraint l,
25                                         PropositionalConstraint r)
26    {
27      left = l;
28      right = r;
29    }
30  
31  
32    /**
33      * Replaces all unquantified variables with the unique copy stored as a
34      * value of the given map; also instantiates all quantified variables and
35      * stores them in the given map.
36      *
37      * @param m  The map in which to find unique copies of the variables.
38     **/
39    public void consolidateVariables(java.util.AbstractMap m)
40    {
41      if (left instanceof PropositionalVariable)
42      {
43        if (m.containsKey(left)) left = (PropositionalVariable) m.get(left);
44        else m.put(left, left);
45      }
46      else left.consolidateVariables(m);
47  
48      if (right instanceof PropositionalVariable)
49      {
50        if (m.containsKey(right)) right = (PropositionalVariable) m.get(right);
51        else m.put(right, right);
52      }
53      else right.consolidateVariables(m);
54    }
55  
56  
57    /**
58      * Returns the children of this constraint in an array.
59      *
60      * @return The children of this constraint in an array.
61     **/
62    public Constraint[] getChildren()
63    {
64      return new PropositionalConstraint[]{ left, right };
65    }
66  }
