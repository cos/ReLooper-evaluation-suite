6   /**
7     * A first order constant is either <code>true</code> or <code>false</code>.
8     *
9     * @author Nick Rizzolo
10   **/
11  public class FirstOrderConstant extends FirstOrderConstraint
12  {
13    /** The constant value. */
14    private boolean constant;
15  
16  
17    /**
18      * Initializing constructor.
19      *
20      * @param v  The value of this constant.
21     **/
22    public FirstOrderConstant(boolean v) { constant = v; }
23  
24  
25    /**
26      * This method sets the given quantification variables to the given object
27      * references and evaluates the expressions involving those variables in
28      * this constraint's <code>FirstOrderEquality</code> children.
29      *
30      * @param o  The new object references for the enclosing quantification
31      *           variables, in order of nesting.
32     **/
33    public void setQuantificationVariables(Vector o) { }
34  
35  
36    /**
37      * Returns the children of this constraint in an array.
38      *
39      * @return The children of this constraint in an array.
40     **/
41    public Constraint[] getChildren() { return new FirstOrderConstraint[0]; }
42  
43  
44    /** Determines whether the constraint is satisfied. */
45    public boolean evaluate() { return constant; }
46  
47  
48    /**
49      * Replaces all unquantified variables with the unique copy stored as a
50      * value of the given map; also instantiates all quantified variables and
51      * stores them in the given map.
52      *
53      * @param m  The map in which to find unique copies of the variables.
54     **/
55    public void consolidateVariables(java.util.AbstractMap m) { }
56  
57  
58    /**
59      * Transforms this first order constraint into a propositional constraint.
60      *
61      * @return The propositionalized constraint.
62     **/
63    public PropositionalConstraint propositionalize()
64    {
65      return new PropositionalConstant(constant);
66    }
67  
68  
69    /**
70      * The hash code of a <code>FirstOrderConstant</code> is the hash code of
71      * the <code>Boolean</code> object formed from the constant.
72      *
73      * @return The hash code for this <code>FirstOrderConstant</code>.
74     **/
75    public int hashCode() { return new Boolean(constant).hashCode(); }
76  
77  
78    /**
79      * Two <code>FirstOrderConstant</code>s are equivalent when their constants
80      * are equal.
81      *
82      * @return <code>true</code> iff the argument is a
83      *         <code>FirstOrderConstant</code> set to the same value as this
84      *         constant.
85     **/
86    public boolean equals(Object o)
87    {
88      if (!(o instanceof FirstOrderConstant)) return false;
89      FirstOrderConstant c = (FirstOrderConstant) o;
90      return constant == c.constant;
91    }
92  
93  
94    /**
95      * Calls the appropriate <code>visit(&middot;)</code> method of the given
96      * <code>Inference</code> for this <code>Constraint</code>, as per the
97      * visitor pattern.
98      *
99      * @param infer  The inference visiting this constraint.
100    **/
101   public void runVisit(Inference infer) { infer.visit(this); }
102 }
