17      * If either of the arguments is itself a
18      * <code>PropositionalDisjunction</code>, its contents are flattened into
19      * this <code>PropositionalDisjunction</code>.
20      *

22      * @param c2 Another constraint to disjunct.

25                                    PropositionalConstraint c2)
26    {
27      add(c1);
28      add(c2);

33    public boolean evaluate()
34    {
35      for (Iterator I = children.iterator(); I.hasNext(); )
36        if (((PropositionalConstraint) I.next()).evaluate()) return true;
37      return false;
38    }

96    /**
97      * Uses DeMorgan's law to compute the negation of this constraint by
98      * distributing that negation to each child.
99      *
100     * @return A simplified constraint representing the negation of this
101     *         constraint.
102    **/
103   public PropositionalConstraint negate()
104   {
105     if (children.size() == 1)
106       return ((PropositionalConstraint) children.iterator().next()).negate();
107 
108     PropositionalConstraint[] array =
109       (PropositionalConstraint[])
110       children.toArray(new PropositionalConstraint[children.size()]);
111     for (int i = 0; i < array.length; ++i) array[i] = array[i].negate();
112 
113     PropositionalConjunction result =
114       new PropositionalConjunction(array[0], array[1]);
115     for (int i = 2; i < array.length; ++i) result.add(array[i]);
116     return result;
117   }

120   /**
121     * Produces a new, logically simplified version of this constraint in
122     * conjunctive normal form (CNF).
123     *
124     * @return The conjunctive normal form of this constraint.
125    **/
126   public PropositionalConstraint CNF()
127   {
128     PropositionalConstraint c = factor();
129     if (!(c instanceof PropositionalDisjunction)) return c.CNF();
130 
131     PropositionalDisjunction simplified = (PropositionalDisjunction) c;
132 
133     PropositionalDisjunction childrenCNF = new PropositionalDisjunction();
134     for (Iterator I = simplified.children.iterator(); I.hasNext(); )
135       childrenCNF.add(((PropositionalConstraint) I.next()).CNF());
136     if (childrenCNF.children.size() == 1)
137       return (PropositionalConstraint) childrenCNF.getChildren()[0];
138 
139     PropositionalConstraint[][] children =
140       new PropositionalConstraint[childrenCNF.children.size()][];
141     int i = 0;
142     boolean foundConjunction = false;
143     for (Iterator I = childrenCNF.children.iterator(); I.hasNext(); ++i)
144     {
145       PropositionalConstraint parent = (PropositionalConstraint) I.next();
146       if (parent instanceof PropositionalConjunction)
147       {
148         children[i] = (PropositionalConstraint[]) parent.getChildren();
149         foundConjunction = true;
150       }
151       else
152       {
153         children[i] = new PropositionalConstraint[1];
154         children[i][0] = parent;
155       }
156     }
157 
158     if (!foundConjunction) return childrenCNF;
159 
160     int[] indexes = new int[children.length];
161     PropositionalConstraint result =
162       new PropositionalDisjunction(children[0][0], children[1][0]);
163     for (i = 2; i < children.length; ++i)
164       result = new PropositionalDisjunction(result, children[i][0]);
165 
166     while (increment(children, indexes))
167     {
168       PropositionalConstraint combination =
169         new PropositionalDisjunction(children[0][indexes[0]],
170                                      children[1][indexes[1]]);
171       for (i = 2; i < children.length; ++i)
172         combination =
173           new PropositionalDisjunction(combination, children[i][indexes[i]]);
174       result = new PropositionalConjunction(result, combination);
175     }
176 
177     return result;
178   }

181   /**
182     * Utility method for iterating through all combinations of constraint
183     * children.
184     *
185     * @param c  Each element of this array is an array of children, exactly
186     *           one child of which appears in each combination.
187     * @param I  The indexes of the children in the current combination.
188     * @return   <code>true</code> iff <code>I</code> contains valid indexes
189     *           for a new combination; <code>false</code> iff if there are no
190     *           more combinations.
191    **/
192   public static boolean increment(PropositionalConstraint[][] c, int[] I)
193   {
194     int i = 0;
195     while (i < c.length && ++I[i] == c[i].length) ++i;
196     if (i == c.length) return false;
197     for (--i; i >= 0; --i) I[i] = 0;
198     return true;
199   }

371   /**
372     * Factoring a disjunction is the opposite of distributing a conjunction
373     * over a disjunction.
374     *
375     * @return A constraint that represents a factoring of this disjunction.
376    **/
377   public PropositionalConstraint factor()
378   {
379     PropositionalConstraint c = simplify();
380     if (!(c instanceof PropositionalDisjunction)) return c;
381     PropositionalDisjunction simplified = (PropositionalDisjunction) c;
382 
383     PropositionalConstraint[] best = new PropositionalConstraint[0];
384     while (best != null)
385     {
386       int bestConjunction = -1;
387       int bestOther = -1;
388       best = null;
389 
390       PropositionalConstraint[] children =
391         (PropositionalConstraint[]) simplified.getChildren();
392       Arrays.sort(children,
393           new Comparator()
394           {
395             public int compare(Object o1, Object o2)
396             {
397               if (o1 instanceof PropositionalConjunction)
398               {
399                 if (o2 instanceof PropositionalConjunction) return 0;
400                 return -1;
401               }
402 
403               if (o2 instanceof PropositionalConjunction) return 1;
404               return 0;
405             }
406           });
407 
408       for (int i = 0;
409            i < children.length - 1
410              && children[i] instanceof PropositionalConjunction;
411            ++i)
412         for (int j = i + 1; j < children.length; ++j)
413         {
414           PropositionalConstraint[] current =
415             ((PropositionalConjunction) children[i]).intersect(children[j]);
416           if (current != null
417               && (best == null || current.length > best.length))
418           {
419             best = current;
420             bestConjunction = i;
421             bestOther = j;
422           }
423         }
424 
425       if (best != null)
426       {
427         PropositionalConstraint toAdd = null;
428         if (best.length == 1) toAdd = best[0];
429         else
430         {
431           toAdd = new PropositionalConjunction(best[0], best[1]);
432           for (int i = 2; i < best.length; ++i)
433             toAdd = new PropositionalConjunction(toAdd, best[i]);
434         }
435 
436         if (children[bestOther] instanceof PropositionalConjunction)
437         {
438           PropositionalConstraint conjunct1 =
439             ((PropositionalConjunction) children[bestConjunction])
440             .subtract(best);
441           PropositionalConstraint conjunct2 =
442             ((PropositionalConjunction) children[bestOther]).subtract(best);
443 
444           toAdd =
445             new PropositionalConjunction(
446                 toAdd,
447                 new PropositionalDisjunction(conjunct1, conjunct2))
448             .simplify();
449         }
450 
451         simplified.children.remove(children[bestConjunction]);
452         simplified.children.remove(children[bestOther]);
453         simplified.add(toAdd);
454       }
455     }
456 
457     if (simplified.children.size() == 1)
458       return (PropositionalConstraint) simplified.getChildren()[0];
459     return simplified;
460   }

463   /**
464     * The intersection of two disjunctions is the set of all terms that are
465     * common to both disjunctions; the intersection of a disjunction and some
466     * other constraint <i>c</i> is <i>c</i> if <i>c</i> is contained in the
467     * disjunction and the empty set otherwise.
468     *
469     * @param c  The constraint to intersect with.
470     * @return   The set of common terms in array form or <code>null</code> if
471     *           there are none.
472    **/
473   public PropositionalConstraint[] intersect(PropositionalConstraint c)
474   {
475     if (!(c instanceof PropositionalDisjunction))
476     {
477       if (children.contains(c)) return new PropositionalConstraint[]{ c };
478       return null;
479     }
480 
481     PropositionalDisjunction disjunction = (PropositionalDisjunction) c;
482     LinkedList result = new LinkedList();
483     for (Iterator I = children.iterator(); I.hasNext(); )
484     {
485       Object next = I.next();
486       if (disjunction.children.contains(next)) result.add(next);
487     }
488 
489     if (result.size() == 0) return null;
490     return (PropositionalConstraint[])
491            result.toArray(new PropositionalConstraint[result.size()]);
492   }

495   /**
496     * Subtraction from a disjunction simply removes all of the specified
497     * terms from it; this method returns a new constraint representing the
498     * subtraction.
499     *
500     * @param terms  The terms to remove.
501     * @return       A new representation of this n-ary constraint with the
502     *               specified terms removed.
503    **/
504   public PropositionalConstraint subtract(PropositionalConstraint[] terms)
505   {
506     PropositionalDisjunction clone = (PropositionalDisjunction) clone();
507     for (int i = 0; i < terms.length; ++i) clone.children.remove(terms[i]);
508     if (clone.children.size() == 0) return new PropositionalConstant(false);
509     if (clone.children.size() == 1)
510       return (PropositionalConstraint) clone.getChildren()[0];
511     return clone;
512   }

515   /**
516     * Distributes the given conjunction over this disjunction.
517     *
518     * @return A simplified constraint representing the distribution of the
519     *         given conjunction over this disjunction.
520    **/
521   public PropositionalConstraint distribute(PropositionalConjunction c)
522   {
523     PropositionalConstraint[] array =
524       (PropositionalConstraint[])
525       children.toArray(new PropositionalConstraint[children.size()]);
526     for (int i = 0; i < array.length; ++i)
527     {
528       PropositionalConjunction clone = (PropositionalConjunction) c.clone();
529       clone.add(array[i]);
530       array[i] = clone;
531     }
532 
533     if (array.length == 1) return array[0].simplify();
534 
535     PropositionalDisjunction result =
536       new PropositionalDisjunction(array[0], array[1]);
537     for (int i = 2; i < array.length; ++i) result.add(array[i]);
538     return result.simplify();
539   }
