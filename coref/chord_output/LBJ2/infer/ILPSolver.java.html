4   /**
5     * Classes that implement this interface contain implementations of
6     * algorithms that solve Integer Linear Programming problems.
7     *
8     * @author Nick Rizzolo
9    **/
10  public interface ILPSolver
11  {
12    /**
13      * Sets the direction of the objective function.
14      *
15      * @param d  <code>true</code> if the objective function is to be
16      *           maximized.
17     **/
18    public void setMaximize(boolean d);
19  
20  
21    /**
22      * Adds a new Boolean variable (an integer variable constrained to take
23      * either the value 0 or the value 1) with the specified coefficient in the
24      * objective function to the problem.
25      *
26      * @param c  The objective function coefficient for the new Boolean
27      *           variable.
28      * @return   The indexes of the created variable.
29     **/
30    public int addBooleanVariable(double c);
31  
32  
33    /**
34      * Adds a general, multi-valued discrete variable, which is implemented as
35      * a set of Boolean variables, one per value of the discrete variable, with
36      * exactly one of those variables set <code>true</code> at any given time.
37      *
38      * @param c  The objective function coefficients for the new Boolean
39      *           variables.
40      * @return   The indexes of the newly created variables.
41     **/
42    public int[] addDiscreteVariable(double[] c);
43  
44  
45    /**
46      * Adds a general, multi-valued discrete variable, which is implemented as
47      * a set of Boolean variables, one per value of the discrete variable, with
48      * exactly one of those variables set <code>true</code> at any given time.
49      *
50      * @param c  An array of {@link LBJ2.classify.Score}s containing the
51      *           objective function coefficients for the new Boolean variables.
52      * @return   The indexes of the newly created variables.
53     **/
54    public int[] addDiscreteVariable(LBJ2.classify.Score[] c);
55  
56  
57    /**
58      * Adds a new fixed constraint to the problem.  The two array arguments
59      * must be the same length, as their elements correspond to each other.
60      * Variables whose coefficients are zero need not be mentioned.  Variables
61      * that are mentioned must have previously been added via
62      * {@link #addBooleanVariable(double)} or
63      * {@link #addDiscreteVariable(double[])}.  The resulting constraint has
64      * the form:
65      * <blockquote> <code>x<sub>i</sub> * a = b</code> </blockquote>
66      * where <code>x<sub>i</sub></code> represents the inference variables
67      * whose indexes are contained in the array <code>i</code> and
68      * <code>*</code> represents dot product.
69      *
70      * @param i  The indexes of the variables with non-zero coefficients.
71      * @param a  The coefficients of the variables with the given indexes.
72      * @param b  The new constraint will enforce equality with this constant.
73     **/
74    public void addEqualityConstraint(int[] i, double[] a, double b);
75  
76  
77    /**
78      * Adds a new lower bounded constraint to the problem.  The two array
79      * arguments must be the same length, as their elements correspond to each
80      * other.  Variables whose coefficients are zero need not be mentioned.
81      * Variables that are mentioned must have previously been added via
82      * {@link #addBooleanVariable(double)} or
83      * {@link #addDiscreteVariable(double[])}.  The resulting constraint has
84      * the form:
85      * <blockquote> <code>x<sub>i</sub> * a &gt;= b</code> </blockquote>
86      * where <code>x<sub>i</sub></code> represents the inference variables
87      * whose indexes are contained in the array <code>i</code> and
88      * <code>*</code> represents dot product.
89      *
90      * @param i  The indexes of the variables with non-zero coefficients.
91      * @param a  The coefficients of the variables with the given indexes.
92      * @param b  The lower bound for the new constraint.
93     **/
94    public void addGreaterThanConstraint(int[] i, double[] a, double b);
95  
96  
97    /**
98      * Adds a new upper bounded constraint to the problem.  The two array
99      * arguments must be the same length, as their elements correspond to each
100     * other.  Variables whose coefficients are zero need not be mentioned.
101     * Variables that are mentioned must have previously been added via
102     * {@link #addBooleanVariable(double)} or
103     * {@link #addDiscreteVariable(double[])}.  The resulting constraint has
104     * the form:
105     * <blockquote> <code>x<sub>i</sub> * a &lt;= b</code> </blockquote>
106     * where <code>x<sub>i</sub></code> represents the inference variables
107     * whose indexes are contained in the array <code>i</code> and
108     * <code>*</code> represents dot product.
109     *
110     * @param i  The indexes of the variables with non-zero coefficients.
111     * @param a  The coefficients of the variables with the given indexes.
112     * @param b  The upper bound for the new constraint.
113    **/
114   public void addLessThanConstraint(int[] i, double[] a, double b);
115 
116 
117   /**
118     * Solves the ILP problem, saving the solution internally.  This method may
119     * throw an exception if something doesn't go right.
120    **/
121   public boolean solve() throws Exception;
122 
123 
124   /**
125     * Tests whether the problem represented by this <code>ILPSolver</code>
126     * instance has been solved already.
127    **/
128   public boolean isSolved();
129 
130 
131   /**
132     * When the problem has been solved, use this method to retrieve the value
133     * of any Boolean inference variable.  The result of this method is
134     * undefined when the problem has not yet been solved.
135     *
136     * @param index  The index of the variable whose value is requested.
137     * @return       The value of the variable.
138    **/
139   public boolean getBooleanValue(int index);
140 
141 
142   /**
143     * This method clears the all constraints and variables out of the ILP
144     * solver's problem representation, bringing the <code>ILPSolver</code>
145     * instance back to the state it was in when first constructed.
146    **/
147   public void reset();
148 
149 
150   /**
151     * Creates a textual representation of the ILP problem in an algebraic
152     * notation.
153     *
154     * @param buffer The created textual representation will be appended here.
155    **/
156   public void write(StringBuffer buffer);
157 }
