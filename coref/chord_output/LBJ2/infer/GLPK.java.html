8   /**
9     * Uses the GNU Linear Programming Kit library to perform Integer Linear
10    * Programming over the variables, maximizing the sum of all
11    * learner-object-value triples selected while respecting the constraints.
12    * The difference between this implementation and <code>NaiveGLPK</code> is
13    * that this implementation attempts to save the generation of a few
14    * constraints by directly translating a broader variety of propositional
15    * logic subexpressions into as few constraints as possible.  For example,
16    * negated variables never need to be converted to positive variables with
17    * the addition of a new variable and constraint, like <code>NaiveGLPK</code>
18    * would generate.
19    *
20    * <p> This class assumes that the <code>constraint</code> variable inherited
21    * from class <code>Inference</code> is of type
22    * <code>FirstOrderConstraint</code>.
23    *
24    * @author     Nick Rizzolo
25    * @deprecated As of LBJ release 2.0.12, it is preferrable to pass a
26    *             {@link LBJ2.jni.GLPKHook} object to the
27    *             {@link LBJ2.infer.ILPInference} constructor.
28   **/
29  public class GLPK extends Inference
30  {

168   /**
169     * Uses the <code>lpx_intopt(LPX*)</code> C routine from the GLPK library
170     * to solve the ILP proglem if it hasn't already been solved.
171    **/
172   protected void infer() throws Exception
173   {
174     if (solver != null) return;
175 
176     constraint.consolidateVariables(variables);
177     indexMap = new HashMap();
178 
179     if (writeStatusFiles)
180       solver = new GLPKHook("GLPKInference" + ID, generateCuts, PRINT_ILP);
181     else solver = new GLPKHook(generateCuts, PRINT_ILP);
182     solver.setMaximize(true);
183 
184     for (Iterator I = variables.values().iterator(); I.hasNext(); )
185     {
186       FirstOrderVariable v = (FirstOrderVariable) I.next();
187       ScoreSet ss = getNormalizer(v.getClassifier()).normalize(v.getScores());
188       Score[] scores = null;
189       if (ss != null) scores = ss.toArray();
190 
191       if (scores == null || scores.length == 0)
192       {
193         System.err.println(
194             "LBJ ERROR: Classifier " + v.getClassifier()
195             + " did not return any scores.  GLPK Inference cannot be "
196             + "performed.");
197         System.exit(1);
198       }
199 
200       int[] indexes = new int[scores.length];
201       double[] coefficients = new double[scores.length];
202       Arrays.fill(coefficients, 1);
203 
204       for (int j = 0; j < scores.length; ++j)
205       {
206         indexes[j] = solver.addBooleanVariable(scores[j].score);
207         indexMap.put(
208             new PropositionalVariable(v.getClassifier(), v.getExample(),
209                                       scores[j].value),
210             new Integer(indexes[j]));
211 
212         if (PRINT_ILP)
213           System.out.println(
214               indexes[j] + "(" + scores[j].score + "): " + v.getClassifier()
215               + "(" + v.getExample() + ") == " + scores[j].value);
216       }
217 
218       solver.addEqualityConstraint(indexes, coefficients, 1);
219     }
220 
221     PropositionalConstraint propositional =
222       ((FirstOrderConstraint) constraint).propositionalize();
223 
224     if (propositional instanceof PropositionalConjunction)
225       propositional =
226         ((PropositionalConjunction) propositional).simplify(true);
227     else propositional = propositional.simplify();
228 
229     if (propositional instanceof PropositionalConstant
230         && !propositional.evaluate())
231     {
232       System.err.println("GLPK ERROR: Unsatisfiable constraints!");
233       solver.addEqualityConstraint(new int[]{ 0 }, new double[]{ 1 }, 2);
234     }
235 
236     topLevel = true;
237     propositional.runVisit(this);
238 
239     if (!solver.solve()) throw new InferenceNotOptimalException(solver, head);
240     int variableIndex = 0;
241 
242     for (Iterator I = variables.values().iterator(); I.hasNext(); )
243     {
244       FirstOrderVariable v = (FirstOrderVariable) I.next();
245       Score[] scores = v.getScores().toArray();
246       for (int j = 0; j < scores.length; ++j, ++variableIndex)
247         if (solver.getBooleanValue(variableIndex))
248           v.setValue(scores[j].value);
249     }
250   }

254     * Retrieves the value of the specified variable as identified by the
255     * classifier and the object that produce that variable.
256     *
257     * @param c  The classifier producing the variable.
258     * @param o  The object from which the variable is produced.
259     * @return   The current value of the requested variable.  If the variable
260     *           does not exist in this inference, the result of the

305   /**
306     * Derived classes override this method to do some type of processing on
307     * constraints of the parameter's type.
308     *
309     * @param c  The constraint to process.
310    **/
311   public void visit(PropositionalDoubleImplication c)
312   {
313     assert topLevel : "GLPK: PropositionalDoubleImplication encountered.";
314     topLevel = false;
315 
316     int[] indexes = new int[2];
317     double[] coefficients = new double[2];
318     double bound = 0;
319 
320     c.left.runVisit(this);
321     indexes[0] = returnIndex;
322     if (returnNegation)
323     {
324       coefficients[0] = -1;
325       --bound;
326     }
327     else coefficients[0] = 1;
328 
329     c.right.runVisit(this);
330     indexes[1] = returnIndex;
331     if (returnNegation)
332     {
333       coefficients[1] = 1;
334       ++bound;
335     }
336     else coefficients[1] = -1;
337 
338     solver.addEqualityConstraint(indexes, coefficients, bound);
339 
340     topLevel = true;
341   }

356   /**
357     * Derived classes override this method to do some type of processing on
358     * constraints of the parameter's type.
359     *
360     * @param c  The constraint to process.
361    **/
362   public void visit(PropositionalConjunction c)
363   {
364     PropositionalConstraint[] children =
365       (PropositionalConstraint[]) c.getChildren();
366 
367     int[] indexes = null;
368     double[] coefficients = null;
369     double bound;
370 
371     if (topLevel)
372     {
373       PropositionalConstraint[] variables =
374         new PropositionalConstraint[children.length];
375       int size = 0;
376       for (int i = 0; i < children.length; ++i)
377       {
378         if (children[i] instanceof PropositionalVariable
379             || children[i] instanceof PropositionalNegation)
380           variables[size++] = children[i];
381         else children[i].runVisit(this);
382       }
383 
384       if (size > 0)
385       {
386         indexes = new int[size];
387         coefficients = new double[size];
388         bound = size;
389 
390         for (int i = 0; i < size; ++i)
391         {
392           variables[i].runVisit(this);
393           indexes[i] = returnIndex;
394           if (returnNegation)
395           {
396             coefficients[i] = -1;
397             --bound;
398           }
399           else coefficients[i] = 1;
400         }
401 
402         solver.addEqualityConstraint(indexes, coefficients, bound);
403       }
404     }
405     else
406     {
407       indexes = new int[children.length + 1];
408       coefficients = new double[children.length + 1];
409       bound = 0;
410 
411       for (int i = 0; i < children.length; ++i)
412       {
413         children[i].runVisit(this);
414         indexes[i] = returnIndex;
415         if (returnNegation)
416         {
417           coefficients[i] = -1;
418           --bound;
419         }
420         else coefficients[i] = 1;
421       }
422 
423       String[] stringIndexes = new String[children.length];
424       for (int i = 0; i < children.length; ++i)
425         stringIndexes[i] = (coefficients[i] < 0 ? "!" : "") + indexes[i];
426       Arrays.sort(stringIndexes);
427       String key = stringIndexes[0];
428       for (int i = 1; i < stringIndexes.length; ++i)
429         key += "&" + stringIndexes[i];
430       Integer I = (Integer) indexMap.get(key);
431 
432       if (I == null)
433       {
434         I = new Integer(createNewVariable(key));
435         indexMap.put(key, I);
436 
437         indexes[children.length] = I.intValue();
438         coefficients[children.length] = -children.length;
439         solver.addGreaterThanConstraint(indexes, coefficients, bound);
440 
441         coefficients[children.length] = -1;
442         solver.addLessThanConstraint(indexes, coefficients,
443                                      bound + children.length - 1);
444       }
445 
446       returnIndex = I.intValue();
447       returnNegation = false;
448     }
449   }

452   /**
453     * Derived classes override this method to do some type of processing on
454     * constraints of the parameter's type.
455     *
456     * @param c  The constraint to process.
457    **/
458   public void visit(PropositionalDisjunction c)
459   {
460     PropositionalConstraint[] children =
461       (PropositionalConstraint[]) c.getChildren();
462 
463     int[] indexes = null;
464     double[] coefficients = null;
465     double bound = 0;
466 
467     if (topLevel)
468     {
469       int subConstraintIndex = -1;
470 
471       for (int i = 0; i < children.length && subConstraintIndex == -1; ++i)
472       {
473         if (children[i] instanceof PropositionalVariable
474             || children[i] instanceof PropositionalNegation)
475           continue;
476         if (children[i] instanceof PropositionalConjunction)
477           subConstraintIndex = i;
478         else if (children[i] instanceof PropositionalAtLeast)
479           subConstraintIndex = i;
480       }
481 
482       if (subConstraintIndex > -1)
483       {
484         PropositionalConstraint[] subChildren =
485           (PropositionalConstraint[])
486           children[subConstraintIndex].getChildren();
487         int multiplier =
488           children[subConstraintIndex] instanceof PropositionalConjunction
489           ? subChildren.length
490           : ((PropositionalAtLeast) children[subConstraintIndex]).getM();
491 
492         indexes = new int[subChildren.length + children.length - 1];
493         coefficients = new double[subChildren.length + children.length - 1];
494         bound = multiplier;
495 
496         topLevel = false;
497 
498         int j = 0;
499         for (int i = 0; i < children.length; ++i)
500         {
501           if (i == subConstraintIndex) continue;
502 
503           children[i].runVisit(this);
504           indexes[j] = returnIndex;
505           if (returnNegation)
506           {
507             coefficients[j] = -multiplier;
508             bound -= multiplier;
509           }
510           else coefficients[j] = multiplier;
511 
512           ++j;
513         }
514 
515         for (int i = 0; i < subChildren.length; ++i, ++j)
516         {
517           subChildren[i].runVisit(this);
518           indexes[j] = returnIndex;
519           if (returnNegation)
520           {
521             coefficients[j] = -1;
522             --bound;
523           }
524           else coefficients[j] = 1;
525         }
526 
527         topLevel = true;
528 
529         solver.addGreaterThanConstraint(indexes, coefficients, bound);
530         return;
531       }
532     }
533 
534     if (topLevel)
535     {
536       indexes = new int[children.length];
537       coefficients = new double[children.length];
538       bound = 1;
539     }
540     else
541     {
542       indexes = new int[children.length + 1];
543       coefficients = new double[children.length + 1];
544     }
545 
546     boolean saveTopLevel = topLevel;
547     topLevel = false;
548     for (int i = 0; i < children.length; ++i)
549     {
550       children[i].runVisit(this);
551       indexes[i] = returnIndex;
552       if (returnNegation)
553       {
554         coefficients[i] = -1;
555         --bound;
556       }
557       else coefficients[i] = 1;
558     }
559 
560     topLevel = saveTopLevel;
561     if (topLevel)
562       solver.addGreaterThanConstraint(indexes, coefficients, bound);
563     else
564     {
565       String[] stringIndexes = new String[children.length];
566       for (int i = 0; i < children.length; ++i)
567         stringIndexes[i] = (coefficients[i] < 0 ? "!" : "") + indexes[i];
568       Arrays.sort(stringIndexes);
569       String key = stringIndexes[0];
570       for (int i = 1; i < stringIndexes.length; ++i)
571         key += "|" + stringIndexes[i];
572       Integer I = (Integer) indexMap.get(key);
573 
574       if (I == null)
575       {
576         I = new Integer(createNewVariable(key));
577         indexMap.put(key, I);
578 
579         indexes[children.length] = I.intValue();
580         coefficients[children.length] = -1;
581         solver.addGreaterThanConstraint(indexes, coefficients, bound);
582 
583         coefficients[children.length] = -children.length;
584         solver.addLessThanConstraint(indexes, coefficients, bound);
585       }
586 
587       returnIndex = I.intValue();
588       returnNegation = false;
589     }
590   }

593   /**
594     * Derived classes override this method to do some type of processing on
595     * constraints of the parameter's type.
596     *
597     * @param c  The constraint to process.
598    **/
599   public void visit(PropositionalAtLeast c)
600   {
601     PropositionalConstraint[] children =
602       (PropositionalConstraint[]) c.getChildren();
603 
604     int[] indexes = null;
605     double[] coefficients = null;
606     double bound = 0;
607 
608     if (topLevel)
609     {
610       indexes = new int[children.length];
611       coefficients = new double[children.length];
612       bound = c.getM();
613     }
614     else
615     {
616       indexes = new int[children.length + 1];
617       coefficients = new double[children.length + 1];
618     }
619 
620     boolean saveTopLevel = topLevel;
621     topLevel = false;
622     for (int i = 0; i < children.length; ++i)
623     {
624       children[i].runVisit(this);
625       indexes[i] = returnIndex;
626       if (returnNegation)
627       {
628         coefficients[i] = -1;
629         --bound;
630       }
631       else coefficients[i] = 1;
632     }
633 
634     topLevel = saveTopLevel;
635     if (topLevel)
636       solver.addGreaterThanConstraint(indexes, coefficients, bound);
637     else
638     {
639       String[] stringIndexes = new String[children.length];
640       for (int i = 0; i < children.length; ++i)
641         stringIndexes[i] = (coefficients[i] < 0 ? "!" : "") + indexes[i];
642       Arrays.sort(stringIndexes);
643       String key = "atl" + c.getM() + "of" + stringIndexes[0];
644       for (int i = 1; i < stringIndexes.length; ++i)
645         key += "&" + stringIndexes[i];
646       Integer I = (Integer) indexMap.get(key);
647 
648       if (I == null)
649       {
650         I = new Integer(createNewVariable(key));
651         indexMap.put(key, I);
652 
653         indexes[children.length] = I.intValue();
654         coefficients[children.length] = -c.getM();
655         solver.addGreaterThanConstraint(indexes, coefficients, bound);
656 
657         coefficients[children.length] = -children.length;
658         solver.addLessThanConstraint(indexes, coefficients,
659                                      bound + c.getM() - 1);
660       }
661 
662       returnIndex = I.intValue();
663       returnNegation = false;
664     }
665   }
