4   /**
5     * Represents either an equality or an inequality between two values, a
6     * classifier application and a value, or two classifier applications.
7     *
8     * @author Nick Rizzolo
9    **/
10  public abstract class FirstOrderEquality extends FirstOrderConstraint
11  {
12    /** <code>true</code> if equality, <code>false</code> if inequality. */
13    protected boolean equality;
14    /**
15      * This object provides the implementation of the method that replaces the
16      * values and variables in an equality given new settings of the
17      * quantification variables; if this member variable is set to
18      * <code>null</code>, it means this <code>FirstOrderEquality</code> is not
19      * nested in a quantification.
20     **/
21    protected EqualityArgumentReplacer replacer;
22    /**
23      * The map that this constraint's variables have been consolidated into, or
24      * <code>null</code> if variable consolidation has not been performed.
25     **/
26    protected java.util.AbstractMap variableMap;
27  
28  
29    /**
30      * Initializing constructor.
31      *
32      * @param e  Indicates whether this is an equality or an inequality.
33     **/
34    public FirstOrderEquality(boolean e) { this(e, null); }
35  
36    /**
37      * This constructor specifies a variable setter for when this equality is
38      * quantified.
39      *
40      * @param e  Indicates whether this is an equality or an inequality.
41      * @param r  An argument replacer.
42     **/
43    public FirstOrderEquality(boolean e, EqualityArgumentReplacer r)
44    {
45      equality = e;
46      replacer = r;
47      variableMap = null;
48    }
49  
50  
51    /**
52      * Returns the children of this constraint in an array.
53      *
54      * @return The children of this constraint in an array.
55     **/
56    public Constraint[] getChildren() { return new FirstOrderConstraint[0]; }
57  }
