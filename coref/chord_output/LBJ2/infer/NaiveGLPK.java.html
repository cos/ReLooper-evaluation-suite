8   /**
9     * Uses the GNU Linear Programming Kit library to perform Integer Linear
10    * Programming over the variables, maximizing the sum of all
11    * learner-object-value triples selected while respecting the constraints.
12    * This code implements the most straight-forward algorithm for translating
13    * FOL constraints to linear inequalities.  First, the FOL constraints are
14    * propositionalized and conjuncted together to arrive at a single
15    * propositional expression representing all the constraints.  Then, each
16    * type of propositional subexpression recursively translates its children,
17    * replacing each with an unnegated variable before translating itself into
18    * two (or one, in the case of negation) linear inequalities.
19    *
20    * <p> This class assumes that the <code>constraint</code> variable inherited
21    * from class <code>Inference</code> is of type
22    * <code>FirstOrderConstraint</code>.
23    *
24    * @author Nick Rizzolo
25    * @deprecated As of LBJ release 2.0.12, it is preferrable to pass a
26    *             {@link LBJ2.jni.GLPKHook} object to the
27    *             {@link LBJ2.infer.ILPInference} constructor.
28   **/
29  public class NaiveGLPK extends GLPK
30  {

32    private static final boolean PRINT_ILP = false;

78    /**
79      * Uses the <code>lpx_intopt(LPX*)</code> C routine from the GLPK library
80      * to solve the ILP proglem if it hasn't already been solved.
81     **/
82    protected void infer() throws Exception
83    {
84      if (solver != null) return;
85  
86      constraint.consolidateVariables(variables);
87      indexMap = new HashMap();
88  
89      if (writeStatusFiles)
90        solver =
91          new GLPKHook("NaiveGLPKInference" + ID, generateCuts, PRINT_ILP);
92      else solver = new GLPKHook(generateCuts, PRINT_ILP);
93      solver.setMaximize(true);
94  
95      for (Iterator I = variables.values().iterator(); I.hasNext(); )
96      {
97        FirstOrderVariable v = (FirstOrderVariable) I.next();
98        ScoreSet ss = getNormalizer(v.getClassifier()).normalize(v.getScores());
99        Score[] scores = null;
100       if (ss != null) scores = ss.toArray();
101 
102       if (scores == null || scores.length == 0)
103       {
104         System.err.println(
105             "LBJ ERROR: Classifier " + v.getClassifier()
106             + " did not return any scores.  GLPK Inference cannot be "
107             + "performed.");
108         System.exit(1);
109       }
110 
111       int[] indexes = new int[scores.length];
112       double[] coefficients = new double[scores.length];
113       Arrays.fill(coefficients, 1);
114 
115       for (int j = 0; j < scores.length; ++j)
116       {
117         indexes[j] = solver.addBooleanVariable(scores[j].score);
118         indexMap.put(
119             new PropositionalVariable(v.getClassifier(), v.getExample(),
120                                       scores[j].value),
121             new Integer(indexes[j]));
122 
123         if (PRINT_ILP)
124         {
125           System.out.println(
126               indexes[j] + "(" + scores[j].score + "): " + v.getClassifier()
127               + "(" + v.getExample() + ") == " + scores[j].value);
128         }
129       }
130 
131       solver.addEqualityConstraint(indexes, coefficients, 1);
132     }
133 
134     PropositionalConstraint propositional =
135       ((FirstOrderConstraint) constraint).propositionalize();
136     propositional = propositional.simplify();
137     propositional.runVisit(this);
138     solver.addEqualityConstraint(
139         new int[]{ returnIndex }, new double[]{ 1 }, 1);
140 
141     if (PRINT_ILP)
142     {
143       StringBuffer buffer = new StringBuffer();
144       solver.write(buffer);
145       System.out.println(buffer);
146     }
147 
148     if (!solver.solve()) throw new InferenceNotOptimalException(solver, head);
149     int variableIndex = 0;
150 
151     for (Iterator I = variables.values().iterator(); I.hasNext(); )
152     {
153       FirstOrderVariable v = (FirstOrderVariable) I.next();
154       Score[] scores = v.getScores().toArray();
155       for (int j = 0; j < scores.length; ++j, ++variableIndex)
156         if (solver.getBooleanValue(variableIndex))
157           v.setValue(scores[j].value);
158     }
159   }

162   /**
163     * Two <code>Inference</code> objects are equal when they have the same
164     * run-time type and store the same head object.  I.e., the <code>==</code>
165     * operator must return <code>true</code> when comparing the two head
166     * objects for this method to return <code>true</code>.
167     *
168     * @param o  The object to compare to this object.
169     * @return   <code>true</code> iff this object equals the argument object
170     *           as defined above.
171    **/
172   public boolean equals(Object o)
173   {
174     if (!(o instanceof NaiveGLPK)) return false;
175     return head == ((NaiveGLPK) o).head;
176   }

185   public void visit(PropositionalDoubleImplication c)
186   {
187     PropositionalConstraint[] children =
188       (PropositionalConstraint[]) c.getChildren();
189     visit(
190         new PropositionalConjunction(
191           new PropositionalImplication(children[0], children[1]),
192           new PropositionalImplication(children[1], children[0])));
193   }

212   /**
213     * Derived classes override this method to do some type of processing on
214     * constraints of the parameter's type.
215     *
216     * @param c  The constraint to process.
217    **/
218   public void visit(PropositionalConjunction c)
219   {
220     PropositionalConstraint[] children =
221       (PropositionalConstraint[]) c.getChildren();
222 
223     int[] indexes = new int[children.length + 1];
224     double[] coefficients = new double[children.length + 1];
225 
226     for (int i = 0; i < children.length; ++i)
227     {
228       children[i].runVisit(this);
229       indexes[i] = returnIndex;
230       coefficients[i] = 1;
231     }
232 
233     indexes[children.length] = Integer.MAX_VALUE;
234     Arrays.sort(indexes);
235     String key = "" + indexes[0];
236     for (int i = 1; i < children.length; ++i) key += "&" + indexes[i];
237     Integer I = (Integer) indexMap.get(key);
238 
239     if (I == null)
240     {
241       I = new Integer(createNewVariable(key));
242       indexMap.put(key, I);
243 
244       indexes[children.length] = I.intValue();
245       coefficients[children.length] = -1;
246       solver.addLessThanConstraint(indexes, coefficients,
247                                    children.length - 1);
248 
249       coefficients = new double[]{ 1, -1 };
250       for (int i = 0; i < children.length; ++i)
251         solver.addGreaterThanConstraint(
252             new int[]{ indexes[i], indexes[children.length] },
253             coefficients, 0);
254     }
255 
256     returnIndex = I.intValue();
257   }

313   public void visit(PropositionalAtLeast c)
314   {
315     PropositionalConstraint[] children =
316       (PropositionalConstraint[]) c.getChildren();
317 
318     int[] indexes = new int[children.length + 1];
319     double[] coefficients = new double[children.length + 1];
320 
321     for (int i = 0; i < children.length; ++i)
322     {
323       children[i].runVisit(this);
324       indexes[i] = returnIndex;
325       coefficients[i] = 1;
326     }
327 
328     indexes[children.length] = Integer.MAX_VALUE;
329     Arrays.sort(indexes);
330     String key = "atl" + c.getM() + "of" + indexes[0];
331     for (int i = 1; i < children.length; ++i)
332       key += "&" + indexes[i];
333     Integer I = (Integer) indexMap.get(key);
334 
335     if (I == null)
336     {
337       I = new Integer(createNewVariable(key));
338       indexMap.put(key, I);
339 
340       indexes[children.length] = I.intValue();
341       coefficients[children.length] = -c.getM();
342       solver.addGreaterThanConstraint(indexes, coefficients, 0);
343 
344       coefficients[children.length] = -children.length;
345       solver.addLessThanConstraint(indexes, coefficients, c.getM() - 1);
346     }
347 
348     returnIndex = I.intValue();
349   }

358   public void visit(PropositionalNegation c)
359   {
360     PropositionalConstraint[] children =
361       (PropositionalConstraint[]) c.getChildren();
362     children[0].runVisit(this);
363 
364     int index = returnIndex;
365     String key = "!" + index;
366     Integer I = (Integer) indexMap.get(key);
367 
368     if (I == null)
369     {
370       I = new Integer(createNewVariable(key));
371       indexMap.put(key, I);
372 
373       int[] indexes = new int[]{ index, I.intValue() };
374       double[] coefficients = new double[]{ 1, 1 };
375       solver.addEqualityConstraint(indexes, coefficients, 1);
376     }
377 
378     returnIndex = I.intValue();
379   }
