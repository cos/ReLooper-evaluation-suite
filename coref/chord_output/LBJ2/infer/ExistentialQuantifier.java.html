6   /**
7     * An existential quantifier states that the constraint must hold for at
8     * least one object from the collection.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class ExistentialQuantifier extends Quantifier
13  {
14    /**
15      * Initializing constructor.
16      *
17      * @param q    The name of the quantification variable.
18      * @param col  The collection of objects to iterate over.
19      * @param con  The constraint being quantified.
20     **/
21    public ExistentialQuantifier(String q, Collection col,
22                                 FirstOrderConstraint con)
23    {
24      super(q, col, con);
25    }
26  
27    /**
28      * This constructor specifies a variable setter for when this quantifier is
29      * itself quantified.
30      *
31      * @param q    The name of the quantification variable.
32      * @param col  The collection of objects to iterate over.
33      * @param con  The constraint being quantified.
34      * @param qar  The variable setter.
35     **/
36    public ExistentialQuantifier(String q, Collection col,
37                                 FirstOrderConstraint con,
38                                 QuantifierArgumentReplacer qar)
39    {
40      super(q, col, con, qar);
41    }
42  
43  
44    /** Determines whether the constraint is satisfied. */
45    public boolean evaluate()
46    {
47      int index = initialize();
48  
49      for (Iterator I = collection.iterator(); I.hasNext(); )
50      {
51        enclosingQuantificationSettings.set(index, I.next());
52        constraint.setQuantificationVariables(enclosingQuantificationSettings);
53  
54        if (constraint.evaluate())
55        {
56          enclosingQuantificationSettings.removeElementAt(index);
57          return true;
58        }
59      }
60  
61      enclosingQuantificationSettings.removeElementAt(index);
62      return false;
63    }
64  
65  
66    /**
67      * This method sets the given quantification variables to the given object
68      * references and evaluates the expressions involving those variables in
69      * this constraint's children.
70      *
71      * @param o  The new object references for the enclosing quantification
72      *           variables, in order of nesting.
73     **/
74    public void setQuantificationVariables(Vector o)
75    {
76      enclosingQuantificationSettings = o;
77  
78      if (replacer != null)
79      {
80        replacer.setQuantificationVariables(o);
81        collection = replacer.getCollection();
82      }
83    }
84  
85  
86    /**
87      * Transforms this first order constraint into a propositional constraint.
88      *
89      * @return The propositionalized constraint.
90     **/
91    public PropositionalConstraint propositionalize()
92    {
93      PropositionalConstraint result = null;
94  
95      int index = initialize();
96      for (Iterator I = collection.iterator(); I.hasNext(); )
97      {
98        enclosingQuantificationSettings.set(index, I.next());
99        constraint.setQuantificationVariables(enclosingQuantificationSettings);
100 
101       if (result == null) result = constraint.propositionalize();
102       else
103         result =
104           new PropositionalDisjunction(result, constraint.propositionalize());
105     }
106 
107     enclosingQuantificationSettings.removeElementAt(index);
108     if (result == null) result = new PropositionalConstant(false);
109     return result;
110   }

113   /**
114     * The hash code of a <code>ExistentialQuantifier</code> is the sum of the
115     * hash codes of its children plus one.
116     *
117     * @return The hash code for this <code>ExistentialQuantifier</code>.
118    **/
119   public int hashCode() { return super.hashCode() + 1; }
120 
121 
122   /**
123     * Two <code>ExistentialQuantifier</code>s are equivalent when their
124     * children are equivalent.
125     *
126     * @return <code>true</code> iff the argument is an equivalent
127     *         <code>ExistentialQuantifier</code>.
128    **/
129   public boolean equals(Object o)
130   {
131     if (!(o instanceof ExistentialQuantifier)) return false;
132     ExistentialQuantifier q = (ExistentialQuantifier) o;
133     return super.equals(q);
134   }

144   public void runVisit(Inference infer) { infer.visit(this); }
