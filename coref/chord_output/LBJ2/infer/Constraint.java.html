6   /**
7     * A constraint is an expression that is either satisified or unsatisfied by
8     * its constituent classifier applications.
9     *
10    * @author Nick Rizzolo
11   **/
12  public abstract class Constraint
13  {
14    /** Determines whether the constraint is satisfied. */
15    abstract public boolean evaluate();
16  
17  
18    /**
19      * Replaces all unquantified variables with the unique copy stored as a
20      * value of the given map; also instantiates all quantified variables and
21      * stores them in the given map.
22      *
23      * @param m  The map in which to find unique copies of the variables.
24     **/
25    abstract public void consolidateVariables(AbstractMap m);
26  
27  
28    /**
29      * Returns the children of this constraint in an array.
30      *
31      * @return The children of this constraint in an array.
32     **/
33    abstract public Constraint[] getChildren();
34  
35  
36    /**
37      * Calls the appropriate <code>visit(&middot;)</code> method of the given
38      * <code>Inference</code> for this <code>Constraint</code>, as per the
39      * visitor pattern.
40      *
41      * @param infer  The inference visiting this constraint.
42     **/
43    abstract public void runVisit(Inference infer);
44  }
