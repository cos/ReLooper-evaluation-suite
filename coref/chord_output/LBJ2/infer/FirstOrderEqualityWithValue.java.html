7   /**
8     * Represents the comparison of a classifier application with a value.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class FirstOrderEqualityWithValue extends FirstOrderEquality
13  {
14    /** The variable on the left of the equality. */
15    protected FirstOrderVariable left;
16    /** The value on the right of the equality. */
17    protected String right;
18  
19  
20    /**
21      * Initializing constructor.
22      *
23      * @param e  Indicates whether this is an equality or an inequality.
24      * @param l  The classifier application.
25      * @param r  The value.
26     **/
27    public FirstOrderEqualityWithValue(boolean e, FirstOrderVariable l,
28                                       String r)
29    {
30      this(e, l, r, null);
31    }
32  
33    /**
34      * This constructor specifies a variable setter for when this equality is
35      * quantified.
36      *
37      * @param e    Indicates whether this is an equality or an inequality.
38      * @param l    The classifier application.
39      * @param r    The value.
40      * @param ear  An argument replacer.
41     **/
42    public FirstOrderEqualityWithValue(boolean e, FirstOrderVariable l,
43                                       String r, EqualityArgumentReplacer ear)
44    {
45      super(e, ear);
46      left = l;
47      right = r;
48    }
49  
50  
51    /**
52      * Replaces all unquantified variables with the unique copy stored as a
53      * value of the given map; also instantiates all quantified variables and
54      * stores them in the given map.
55      *
56      * @param m  The map in which to find unique copies of the variables.
57     **/
58    public void consolidateVariables(java.util.AbstractMap m)
59    {
60      variableMap = m;
61      if (m.containsKey(left)) left = (FirstOrderVariable) m.get(left);
62      else m.put(left, left);
63    }
64  
65  
66    /**
67      * This method sets the given quantification variables to the given object
68      * references and evaluates the expressions involving those variables in
69      * this constraint's <code>FirstOrderEquality</code> children.
70      *
71      * @param o  The new object references for the enclosing quantification
72      *           variables, in order of nesting.
73     **/
74    public void setQuantificationVariables(Vector o)
75    {
76      if (replacer == null)
77      {
78        System.err.println(
79            "LBJ ERROR: Attempting to set quantification variable in "
80            + "FirstOrderEqualityWithValue with no variable setter "
81            + "implementation provided.");
82        System.exit(1);
83      }
84  
85      replacer.setQuantificationVariables(o);
86      if (!replacer.leftConstant)
87      {
88        left =
89          new FirstOrderVariable(left.getClassifier(),
90                                 replacer.getLeftObject());
91        if (variableMap != null && variableMap.containsKey(left))
92          left = (FirstOrderVariable) variableMap.get(left);
93      }
94  
95      if (!replacer.rightConstant) right = replacer.getRightValue();
96    }
97  
98  
99    /** Determines whether the constraint is satisfied. */
100   public boolean evaluate()
101   {
102     return equality == left.getValue().equals(right);
103   }

105 
106   /**
107     * Transforms this first order constraint into a propositional constraint.
108     *
109     * @return The propositionalized constraint.
110    **/
111   public PropositionalConstraint propositionalize()
112   {
113     Score[] leftScores = left.getScores().toArray();
114     boolean found = false;
115     for (int i = 0; i < leftScores.length && !found; ++i)
116       found = leftScores[i].value.equals(right);
117 
118     PropositionalConstraint result = null;
119     if (!found) result = new PropositionalConstant(false);
120     else
121       result = new PropositionalVariable(left.getClassifier(),
122                                          left.getExample(), right);
123 
124     if (!equality) result = new PropositionalNegation(result);
125     return result;
126   }

129   /**
130     * The hash code of a <code>FirstOrderEqualityWithValue</code> is the sum
131     * of the hash codes of its children plus 1.
132     *
133     * @return The hash code for this <code>FirstOrderEqualityWithValue</code>.
134    **/
135   public int hashCode()
136   {
137     if (replacer != null) return replacer.hashCode();
138     return left.hashCode() + right.hashCode() + 1;
139   }

142   /**
143     * Two <code>FirstOrderEqualityWithValue</code>s are equivalent when their
144     * children are equivalent.
145     *
146     * @return <code>true</code> iff the argument is a
147     *         <code>FirstOrderEqualityWithValue</code> involving the same
148     *         children.
149    **/
150   public boolean equals(Object o)
151   {
152     if (!(o instanceof FirstOrderEqualityWithValue)) return false;
153     FirstOrderEqualityWithValue n = (FirstOrderEqualityWithValue) o;
154     return replacer == n.replacer
155            && (replacer != null
156                || replacer == null
157                   && left.equals(n.left) && right.equals(n.right));
158   }
