6   /**
7     * Anonymous inner classes extending this class are instantiated by the code
8     * generated by the LBJ compiler when creating
9     * <code>FirstOrderConstraint</code> representations.  The methods of this
10    * class are used to compute new values for the arguments of a quantified
11    * <code>Quantifier</code>.  Only certain value returning methods are
12    * overridden.  The others will throw
13    * <code>UnsupportedOperationException</code>s.
14    *
15    * @see    LBJ2.infer.FirstOrderConstraint
16    * @see    LBJ2.infer.Quantifier
17    * @see    java.lang.UnsupportedOperationException
18    * @author Nick Rizzolo
19   **/
20  abstract public class QuantifierArgumentReplacer extends ArgumentReplacer
21  {
22    /**
23      * This flag is set if the collection of the quantifier is not quantified.
24     **/
25    public boolean collectionConstant;
26    /**
27      * This flag is set if the bound parameter of an
28      * <code>AtLeastQuantifier</code> or an <code>AtMostQuantifier</code> is
29      * not quantified.
30     **/
31    public boolean boundConstant;
32  
33  
34    /**
35      * Initializing constructor.
36      *
37      * @param c  The context of the corresponding equality, except for
38      *           quantification variables.
39     **/
40    public QuantifierArgumentReplacer(Object[] c)
41    {
42      super(c);
43      collectionConstant = boundConstant = false;
44    }
45  
46    /**
47      * Use this constructor to indicate which of the two arguments of the
48      * equality is in fact not quantified.
49      *
50      * @param c  The context of the corresponding equality, except for
51      *           quantification variables.
52      * @param b  Set to <code>false</code> if the unquantified argument is the
53      *           collection; set to <code>true</code> if the unquantified
54      *           argument is the bound.
55     **/
56    public QuantifierArgumentReplacer(Object[] c, boolean b)
57    {
58      super(c);
59      collectionConstant = !b;
60      boundConstant = b;
61    }
62

64    /**
65      * Computes the new collection.  This method needs to be overridden if it
66      * is to be called, since by default it simply throws an
67      * <code>UnsupportedOperationException</code>.
68     **/
69    public Collection getCollection()
70    {
71      throw new UnsupportedOperationException(
72          "LBJ ERROR: getCollection() not supported.");
73    }

76    /**
77      * Computes the new value of the bound.  This method needs to be overridden

79      * <code>UnsupportedOperationException</code>.
80     **/
81    public int getBound()
82    {
83      throw new UnsupportedOperationException(
84          "LBJ ERROR: getBound() not supported.");
85    }
