6   /**
7     * Represents a first order constraint with an arbitrary number of arguments,
8     * usually assumed to be greater than or equal to 2.
9     *
10    * @author Nick Rizzolo
11   **/
12  public abstract class FirstOrderNAryConstraint extends FirstOrderConstraint
13  {
14    /** The children of the operator. */
15    protected HashSet children;
16  
17  
18    /** Default constructor. */
19    public FirstOrderNAryConstraint() { children = new HashSet(); }
20  
21  
22    /**
23      * Replaces all unquantified variables with the unique copy stored as a
24      * value of the given map; also instantiates all quantified variables and
25      * stores them in the given map.
26      *
27      * @param m  The map in which to find unique copies of the variables.
28     **/
29    public void consolidateVariables(java.util.AbstractMap m)
30    {
31      for (Iterator I = children.iterator(); I.hasNext(); )
32        ((FirstOrderConstraint) I.next()).consolidateVariables(m);
33    }
34  
35  
36    /**
37      * This method sets the given quantification variables to the given object
38      * references and evaluates the expressions involving those variables in
39      * this constraint's <code>FirstOrderEquality</code> children.
40      *
41      * @param o  The new object references for the enclosing quantification
42      *           variables, in order of nesting.
43     **/
44    public void setQuantificationVariables(Vector o)
45    {
46      for (Iterator I = children.iterator(); I.hasNext(); )
47        ((FirstOrderConstraint) I.next()).setQuantificationVariables(o);
48    }
49  
50  
51    /**
52      * Returns the children of this constraint in an array.
53      *
54      * @return The children of this constraint in an array.
55     **/
56    public Constraint[] getChildren()
57    {
58      return (PropositionalConstraint[])
59             children.toArray(new PropositionalConstraint[children.size()]);
60    }
61  
62  
63    /**
64      * Determines whether the given constraint is a term of this constraint.
65      *
66      * @param c  The given constraint.
67      * @return   <code>true</code> iff the given constraint is contained in
68      *           this constraint.
69     **/
70    public boolean contains(FirstOrderConstraint c)
71    {
72      return children.contains(c);
73    }
74  
75  
76    /**
77      * Returns the number of terms in this constraint.
78      *
79      * @return The number of terms in this constraint.
80     **/
81    public int size() { return children.size(); }
82  
83  
84    /**
85      * If the given constraint has the same type as this constraint, its terms
86      * are merged into this constraint; otherwise, it is added as a new term.
87      *
88      * @param c  The constraint to add.
89     **/
90    abstract public void add(FirstOrderConstraint c);
91  }
