6   /**
7     * The <code>PrintAST</code> pass simply prints a text representation of the
8     * parsed AST to standard output.
9     *
10    * @author Nick Rizzolo
11   **/
12  public class PrintAST extends Pass
13  {
14    /** The current amount of indentation to print. */
15    private int indent;
16  
17  
18    /**
19      * Instantiates a pass that runs on an entire <code>AST</code>.
20      *
21      * @param ast  The program to run this pass on.
22     **/
23    public PrintAST(AST ast) { super(ast); }
24  
25  
26    /**
27      * Prints the given text preceeded by the amount of indentation called for
28      * in the <code>indent</code> member variable and followed by a new line.
29      *
30      * @param text The text to print.
31     **/
32    public void indentedPrintln(String text)
33    {
34      for (int i = 0; i < indent; ++i) System.out.print("  ");
35      System.out.println(text);
36    }
37  
38  
39    /**
40      * Prints the given text preceeded by the amount of indentation called for
41      * in the <code>indent</code> member variable and followed by the line
42      * number and byte offset information for the specified
43      * <code>ASTNode</code> and a new line.
44      *
45      * @param text The text to print.
46      * @param node The node for which to print line and byte offset
47      *             information.
48     **/
49    public void indentedPrintln(String text, ASTNode node)
50    {
51      for (int i = 0; i < indent; ++i) System.out.print("  ");
52      System.out.println(text + " (" + (node.line + 1) + ", " + node.byteOffset
53                         + ")");
54    }
55  
56  
57    /**
58      * The default routine for printing a non-terminal AST node is to first
59      * print the name of the AST node's class with line and byte offset
60      * information, and then recursively print its children at indentation
61      * level one higher.
62      *
63      * @param text The text to print.
64      * @param node The node for which to print line and byte offset
65      *             information.
66     **/
67    public void nonTerminal(String text, ASTNode node)
68    {
69      indentedPrintln(text, node);
70      ++indent;
71      runOnChildren(node);
72      --indent;
73    }

82    {
83      indent = 0;
84      nonTerminal("AST", node);
85    }

414   /**
415     * Runs this pass on all nodes of the indicated type.
416     *
417     * @param node The node to process.
418    **/
419   public void run(VariableDeclaration node)
420   {
421     indentedPrintln("VariableDeclaration", node);
422     ++indent;
423     if (node.isFinal) indentedPrintln("final");
424     node.type.runPass(this);
425     node.names.runPass(this);
426 
427     indentedPrintln("ExpressionList", node.initializers);
428     ++indent;
429     ExpressionList.ExpressionListIterator I =
430       node.initializers.listIterator();
431     while (I.hasNext())
432     {
433       Expression i = I.nextItem();
434       if (i == null) indentedPrintln("null");
435       else i.runPass(this);
436     }
437 
438     indent -= 2;
439   }
