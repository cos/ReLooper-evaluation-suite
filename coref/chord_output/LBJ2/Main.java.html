10  /**
11    * LBJ2's command line interface.  Passing a source file to this class will
12    * invoke LBJ2's frontend and optimization passes, resulting in the execution
13    * of the source file's code including creation of java files that implement
14    * the source file's semantics. <p>
15    *
16    * <font size=+2><b>LBJ</b></font> stands for
17    * <font size=+1><b>L</b></font>earning <font size=+1><b>B</b></font>ased
18    * <font size=+1><b>J</b></font>ava.  LBJ2 is a language for building systems
19    * that learn. <p>
20    *
21    * <dl>
22    *   <dt>Usage:</dt>
23    *   <dd>
24    *     <code>java LBJ2.Main [options] &lt;source file&gt;</code>

31    *             <td>
32    *               Compile only: This option tells LBJ2 to translate the given
33    *               source to Java, but not to compile the generated Java
34    *               sources or do any training.
35    *             </td>
36    *           </tr>

41    *               in the specified directory, just like <code>javac</code>'s
42    *               <code>-d</code> command line parameter.
43    *             </td>
44    *           </tr>
45    *           <tr>
46    *             <td valign=top><code>-j &lt;a&gt;</code></td>
47    *             <td>
48    *               Sends the contents of <code>&lt;a&gt;</code> to
49    *               <code>javac</code> as command line arguments while
50    *               compiling.  Don't forget to put quotes around
51    *               <code>&lt;a&gt;</code> if there is more than one such

53    *             </td>

233   /**
234     * The main compiler driver.  This method parses command line options and
235     * then calls all of LBJ2's components.
236     *
237     * @param args           The user's command line arguments are found here.
238     * @exception Exception  An exception is thrown when any error occurs.
239    **/
240   public static void main(String[] args) throws Exception
241   {
242     AST ast = null;
243     try { ast = frontend(ProcessCommandLine(args)); }
244     catch (Exception e)
245     {
246       if ("version".equals(e.getMessage())) System.exit(0);
247 
248       if (e.getMessage() == null
249           || !e.getMessage().equals("Incorrect arguments"))
250         throw e;
251       System.exit(1);
252     }
253 
254     if (ast == null) return;
255       // Happens if --lexerOutput, --parserOutput, or --semanticOutput is
256       // enabled.
257 
258     fileNames = new HashSet();
259     passes = new LinkedList();
260 
261     new SemanticAnalysis(ast).run();
262     Pass.printErrorsAndWarnings();
263 
264     if (generatedSourceDirectory != null)
265     {
266       if (!AST.globalSymbolTable.getPackage().equals(""))
267         generatedSourceDirectory +=
268           File.separator + AST.globalSymbolTable.getPackage()
269                            .replace('.', File.separatorChar);
270     }
271     else if (sourceDirectory != null)
272       generatedSourceDirectory = sourceDirectory;
273 
274     if (classPackageDirectory != null
275         && !AST.globalSymbolTable.getPackage().equals(""))
276       classDirectory =
277         classPackageDirectory + File.separator
278         + AST.globalSymbolTable.getPackage().replace('.', File.separatorChar);
279     else classDirectory = classPackageDirectory;
280 
281     if (clean) passes.add(new Clean(ast));
282     else
283     {
284       passes.add(new RevisionAnalysis(ast));
285       passes.add(new TranslateToJava(ast));
286       if (!compileOnly) passes.add(new Train(ast, trainingOutput));
287     }
288 
289     for (Iterator I = passes.iterator(); I.hasNext() && !Pass.fatalError; )
290     {
291       ((Pass) I.next()).run();
292       Pass.printErrorsAndWarnings();
293     }
294 
295     if (Pass.fatalError) System.exit(1);
296   }

299   /**
300     * Sets all the internal flags that correspond to the specified command
301     * line parameters, and checks the command line for errors.
302     *
303     * @param args           The user's command line arguments are found here.
304     * @exception Exception  An exception is thrown if an error is found.
305     * @return               A stream for the input source file.
306    **/
307   private static FileInputStream ProcessCommandLine(String[] args)
308     throws Exception
309   {
310     if (args.length < 1)
311     {
312       PrintUsage();
313       throw new Exception("Incorrect arguments");
314     }
315 
316     boolean printVersion = false;
317     int index;
318     for (index = 0; index < args.length - 1; ++index)
319     {
320       if (args[index].equals("-t"))
321       {
322         try
323         {
324           trainingOutput = Integer.parseInt(args[++index]);
325           if (trainingOutput < 0) throw new Exception();
326         }
327         catch (Exception e)
328         {
329           PrintUsage();
330           throw
331             new Exception("The -t argument must be followed by a "
332                           + "non-negative integer.");
333         }
334       }
335       else if (args[index].equals("-c")) compileOnly = true;
336       else if (args[index].equals("-d"))
337         classPackageDirectory = args[++index];
338       else if (args[index].equals("-j"))
339         javacArguments += " " + args[++index];
340       else if (args[index].equals("-p")) concurrentTraining = true;
341       else if (args[index].equals("-s")) printSymbols = true;
342       else if (args[index].equals("-v")) printVersion = true;
343       else if (args[index].equals("-w")) warningsDisabled = true;
344       else if (args[index].equals("-x")) clean = true;
345       else if (args[index].equals("-generatedsourcepath")
346                || args[index].equals("-gsp"))
347         generatedSourceDirectory = args[++index];
348       else if (args[index].equals("-sourcepath")) sourcePath = args[++index];
349       else if (args[index].equals("--parserDebug")) parserDebug = true;
350       else if (args[index].equals("--lexerOutput")) printTokens = true;
351       else if (args[index].equals("--parserOutput")) printAST = true;
352       else if (args[index].equals("--semanticOutput")) printSemantic = true;
353       else
354       {
355         PrintUsage();
356         throw new Exception("Unrecognized parameter: " + args[index]);
357       }
358     }
359 
360     if (printVersion || args.length == 1 && args[0].equals("-v"))
361     {
362       System.out.println("Learning Based Java (LBJ) "
363                          + Configuration.packageVersion);
364       System.out.println(
365           "Copyright (C) 2008, Nicholas D. Rizzolo and Dan Roth.");
366       System.out.println("Cognitive Computations Group");
367       System.out.println("University of Illinois at Urbana-Champaign");
368       System.out.println(Configuration.webSite);
369       throw new Exception("version");
370     }
371 
372     if (javacArguments.indexOf("-d ") != -1
373         || javacArguments.indexOf("-sourcepath ") != -1
374         || javacArguments.indexOf("-classpath ") != -1)
375       throw new Exception(
376           "None of the options '-d', '-sourcepath', or '-classpath' should "
377           + "be specified inside LBJ's '-j' option.  Instead, specify '-d' "
378           + "and '-sourcepath' directly as options to LBJ, and specify "
379           + "-classpath to the JVM when executing LBJ.");
380 
381     if (clean
382         && (compileOnly || printTokens || printAST || printSemantic
383             || trainingOutput != 0))
384     {
385       System.err.println(
386           "The -x flag supercedes all other flags except --parserDebug and "
387           + "the path related flags.");
388       compileOnly = printTokens = printAST = printSemantic = false;
389       trainingOutput = 0;
390     }
391 
392     if (index >= args.length)
393       throw new Exception("Error: No input filename specified.");
394 
395     String file = args[index];
396     if (!(file.length() > 4 && file.endsWith(".lbj")))
397     {
398       PrintUsage();
399       throw new Exception("Source file name must end with \".lbj\".");
400     }
401 
402     int lastSlash = file.lastIndexOf(File.separatorChar);
403     if (lastSlash != -1)
404     {
405       sourceDirectory = file.substring(0, lastSlash);
406       sourceFilename = file.substring(lastSlash + 1);
407     }
408     else sourceFilename = file;
409 
410     sourceFileBase = sourceFilename.substring(0, sourceFilename.length() - 4);
411 
412     FileInputStream instream;
413     try { instream = new FileInputStream(file); }
414     catch (FileNotFoundException e)
415     {
416       System.err.println("Error: Unable to open input file " + file + ": "
417                          + e.getMessage());
418       throw e;
419     }
420 
421     return instream;
422   }

425   /**
426     * This method scans and then parses the input.
427     *
428     * @param in             A stream for the source input file.
429     * @exception Exception  Thrown when any error occurs.
430     * @return               The AST that results from parsing.
431    **/
432   private static AST frontend(FileInputStream in) throws Exception
433   {
434     Yylex scanner = new Yylex(in);
435     scanner.sourceFilename = sourceFilename;
436     if (printTokens)
437     {
438       dumpTokenStream(scanner);
439       return null;
440     }
441 
442     AST ast = null;
443     parser LBJ2parser = new parser(scanner);
444     if (parserDebug) ast = (AST) LBJ2parser.debug_parse().value;
445     else ast = (AST) LBJ2parser.parse().value;
446 
447     if (ast == null)
448       throw new InternalError("Parser returned null abstract syntax tree.");
449 
450     AST result = ast;
451     if (printAST)
452     {
453       new PrintAST(ast).run();
454       result = null;
455     }
456 
457     if (printSemantic)
458     {
459       SemanticAnalysis semantic = new SemanticAnalysis(ast);
460       semantic.run();
461       System.out.println("\nGlobal symbol table:");
462       System.out.println("--------------------");
463       ast.symbolTable.print();
464 
465       System.out.println("\nDependor graph:");
466       System.out.println("--------------------");
467       SemanticAnalysis.printDependorGraph();
468 
469       System.out.println("\nInvoked graph:");
470       System.out.println("--------------------");
471       SemanticAnalysis.printInvokedGraph();
472       System.out.println();
473 
474       result = null;
475     }
476 
477     if (printSymbols)
478     {
479       new DeclarationNames(ast).run();
480       result = null;
481     }
482 
483     return result;
484   }

487   /**
488    * Dump the token stream produced by the given scanner to standard output.
489    * Returns when the end-of-file token is returned from the scanner, or if an
490    * exception is thrown by the scanner's next_token() method. <p>
491    *
492    * Tokens are output as follows: the name of the token (as provided by the
493    * array symNames.nameTable[]), followed by a tab, followed by the token's
494    * semantic value (see TokenValue.toString()), followed by a tab, followed
495    * by the line and the byte offset in the file where the token began (the
496    * last two separated by a colon). <p>
497    *
498    * Error tokens are printed specially; an error message is printed with only
499    * the line number listed.
500    *
501    * @param scanner A reference to the JLex generated scanner object.
502    */
503   private static void dumpTokenStream(Yylex scanner)
504   {
505     Symbol t;
506     TokenValue tValue;
507 
508     while (true)
509     {
510       try { t = scanner.next_token(); }
511       catch (IOException e)
512       {
513         System.err.println(e);
514         return;
515       }
516 
517       tValue = (TokenValue)t.value;
518       switch (t.sym)
519       {
520         case sym.EOF: return;
521         case sym.error:
522           System.out.println("Scanner returned error token at "
523                               + tValue.line);
524           break;
525         default:
526           System.out.println(symNames.nameTable[t.sym] + "\t" + tValue
527                               + "\t" + (tValue.line + 1) + ":"
528                               + tValue.byteOffset);
529       }
530     }
531   }
