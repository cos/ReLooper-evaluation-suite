7   /**
8     * Native interface to the GNU Linear Programming Kit (GLPK), designed for
9     * version 4.14.  In order to use this class, GLPK 4.14 must be installed
10    * such that the following statement would work in a <code>C</code> source
11    * file: <br><br>
12    *
13    *   <code>#include &lt;glpk.h&gt;</code> <br><br>
14    *
15    * and such that that <code>C</code> file could be compiled and linked with
16    * <code>-lglpk</code> on the command line.  Furthermore, the
17    * <code>GLPKHook</code> library distributed with LBJ must be accessible in a
18    * location where the JVM will search for it.
19    *
20    * <p> An object of this class represents a single <i>integer</i> linear
21    * programming problem.  All columns added are assumed to represent Boolean
22    * variables (i.e., integer variables that may take either the value 0 or the
23    * value 1).
24   **/
25  public class GLPKHook implements ILPSolver
26  {
27    static { System.loadLibrary("LBJGLPKHook"); }

111     * Creates the problem object with a call to <code>createProblem()</code>.
112     *
113     * @param s  A file name for the "debug" file, in which diagnostic messages
114     *           are written when an optimal solution cannot be found.
115     * @param g  Whether or not to generate cuts.
116     * @param p  Set <code>true</code> to cause a textual representation of the
117     *           problem to be printed to <code>STDOUT</code> when invoking the
118     *           {@link #solve()} method before the solution is carried out.
119     * @see      #createProblem()
120    **/
121   public GLPKHook(String s, boolean g, boolean p)
122   {
123     if (!LBJ2.Configuration.GLPKLinked)
124     {
125       System.err.println(
126           "LBJ ERROR: LBJ has not been configured correctly to invoke GLPK "
127           + "inference.  Install GLPK 4.14 or better and re-configure.");
128       System.exit(1);
129     }
130 
131     debugFileName = s;
132     printProblem = p;
133     generateCuts = g;
134     reset();
135   }

202   /**
203     * Adds a general, multi-valued discrete variable, which is implemented as
204     * a set of Boolean variables, one per value of the discrete variable, with
205     * exactly one of those variables set <code>true</code> at any given time.
206     * Since GLPK does not handle these sets of Boolean variables specially,
207     * this method simply calls {@link #addBooleanVariable(double)} repeatedly
208     * and then adds an equality constraint.
209     *
210     * @param c  The objective function coefficients for the new Boolean
211     *           variables.
212     * @return   The indexes of the newly created variables.
213    **/
214   public int[] addDiscreteVariable(double[] c)
215   {
216     int[] result = new int[c.length];
217     for (int i = 0; i < c.length; ++i) result[i] = addBooleanVariable(c[i]);
218 
219     double[] coefficients = new double[c.length];
220     Arrays.fill(coefficients, 1);
221     addEqualityConstraint(result, coefficients, 1);
222     return result;
223   }

226   /**
227     * Adds a general, multi-valued discrete variable, which is implemented as
228     * a set of Boolean variables, one per value of the discrete variable, with
229     * exactly one of those variables set <code>true</code> at any given time.
230     * Since GLPK does not handle these sets of Boolean variables specially,
231     * this method simply calls {@link #addBooleanVariable(double)} repeatedly
232     * and then adds an equality constraint.
233     *
234     * @param c  An array of {@link LBJ2.classify.Score}s containing the
235     *           objective function coefficients for the new Boolean variables.
236     * @return   The indexes of the newly created variables.
237    **/
238   public int[] addDiscreteVariable(LBJ2.classify.Score[] c)
239   {
240     int[] result = new int[c.length];
241     for (int i = 0; i < c.length; ++i)
242       result[i] = addBooleanVariable(c[i].score);
243 
244     double[] coefficients = new double[c.length];
245     Arrays.fill(coefficients, 1);
246     addEqualityConstraint(result, coefficients, 1);
247     return result;
248   }

252     * Adds a new Boolean variable (an integer variable constrained to take
253     * either the value 0 or the value 1) with the specified coefficient in the
254     * objective function to the problem.
255     *
256     * @param c  The objective function coefficient for the new Boolean
257     *           variable.

259   public int addBooleanVariable(double c)
260   {
261     if (printProblem)
262     {
263       objectiveFunction += " ";
264       if (c >= 0) objectiveFunction += "+";
265       objectiveFunction += c + " x_" + objectiveCoefficients;
266     }
267 
268     addObjectiveCoefficient(c);
269     return objectiveCoefficients++;
270   }

284   /**
285     * Adds a new fixed constraint to the problem.  The two array arguments
286     * must be the same length, as their elements correspond to each other.
287     * Variables whose coefficients are zero need not be mentioned.  Variables
288     * that are mentioned must have previously been added via
289     * {@link #addBooleanVariable(double)} or
290     * {@link #addDiscreteVariable(double[])}.  The resulting constraint has
291     * the form:
292     * <blockquote> <code>x<sub>i</sub> * a = b</code> </blockquote>
293     * where <code>x<sub>i</sub></code> represents the inference variables
294     * whose indexes are contained in the array <code>i</code> and
295     * <code>*</code> represents dot product.
296     *
297     * @param i  The indexes of the variables with non-zero coefficients.
298     * @param a  The coefficients of the variables with the given indexes.
299     * @param b  The new constraint will enforce equality with this constant.
300    **/
301   public void addEqualityConstraint(int[] i, double[] a, double b)
302   {
303     if (printProblem)
304     {
305       String constraint = "";
306 
307       for (int j = 0; j < i.length; ++j)
308       {
309         constraint += " ";
310         if (a[j] >= 0) constraint += "+";
311         constraint += a[j] + " x_" + i[j];
312       }
313 
314       constraint += " = " + b;
315       constraints.add(constraint);
316     }
317 
318     addFixedConstraint(i, a, b);
319   }

460   /**
461     * Simply calls {@link #nativeSolve()}, saving the result in
462     * {@link #solved}.
463    **/
464   public boolean solve() throws Exception
465   {
466     if (printProblem)
467     {
468       StringBuffer buffer = new StringBuffer();
469       write(buffer);
470       System.out.print(buffer);
471     }
472 
473     return (solved = nativeSolve());
474   }

512   /**
513     * Writes the optimization problem that this solver represents into the
514     * specified buffer, assuming the appropriate member methods were called to
515     * remember this information.
516     *
517     * @param buffer The buffer to write in.
518    **/
519   public void write(StringBuffer buffer)
520   {
521     if (maximize) buffer.append("max");
522     else buffer.append("min");
523     buffer.append(objectiveFunction);
524     buffer.append("\n");
525 
526     for (Iterator I = constraints.iterator(); I.hasNext(); )
527     {
528       buffer.append(I.next());
529       buffer.append("\n");
530     }
