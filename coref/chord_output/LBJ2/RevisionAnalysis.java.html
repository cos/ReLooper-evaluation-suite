10  /**
11    * To be run after <code>SemanticAnalysis</code>, this pass determines which
12    * <code>CodeGenerator</code>s need to have their code generated and which
13    * classifiers need to be trained based on the revisions made to the LBJ
14    * source file.
15    *
16    * <p> A hard coded classifier, a constraint, or an inference named
17    * <code>foo</code> needs its code regenerated iff at least one of the
18    * following is true:
19    * <ul>
20    *   <li> The file <code>foo.java</code> does not exist.
21    *   <li>
22    *     Using the comments at the top of <code>foo.java</code>, it is
23    *     determined that the code specifying <code>foo</code> has been revised.
24    * </ul>
25    * If the comments at the top of <code>foo.java</code> do not exist, or if
26    * they don't have the expected form, the file will not be overwritten and an
27    * error will be generated.
28    *
29    * <p> All <code>CodeGenerator</code>s are also labeled as either "affected"
30    * (by a revision) or "unaffected".  An <code>CodeGenerator</code> named
31    * <code>foo</code> is labeled "affected" iff at least one of the following
32    * is true:
33    * <ul>
34    *   <li>
35    *     <code>foo</code> is a hard coded classifier, a constraint, or an
36    *     inference and either:
37    *     <ul>
38    *       <li> its code needed to be regenerated as described above or
39    *       <li> it invokes another "affected" <code>CodeGenerator</code>.
40    *     </ul>
41    *   <li>
42    *     <code>foo</code> is a learning classifier and at least one of its
43    *     label or extractor classifiers is "affected".
44    * </ul>
45    *
46    * <p> A learning classifier named <code>foo</code> needs to have its code
47    * regenerated and retrained iff at least one of the following is true:
48    * <ul>
49    *   <li> The file <code>foo.java</code> does not exist.
50    *   <li>
51    *     Using the comments at the top of <code>foo.java</code>, it is
52    *     determined that the code specifying <code>foo</code> has been revised.
53    *   <li> At least one of its label or extractor classifiers is "affected".
54    * </ul>
55    *
56    * @see    LBJ2.SemanticAnalysis
57    * @author Nick Rizzolo
58   **/
59  public class RevisionAnalysis extends Pass
60  {
61    /** Constant representing the "unaffected" revision status. */
62    public static final Integer UNAFFECTED = new Integer(0);
63    /** Constant representing the "affected" revision status. */
64    public static final Integer AFFECTED = new Integer(1);
65    /** Constant representing the "revised" revision status. */
66    public static final Integer REVISED = new Integer(2);

93    /**
94      * This method reads the comments at the top of the file containing the
95      * code corresponding to the specified code generating node to determine if
96      * the LBJ source describing that code generator has been modified since
97      * the LBJ2 compiler was last executed.
98      *
99      * @param node     The code generating node.
100     * @param convert  Whether or not the code is converted to hexadecimal
101     *                 compressed format.
102     * @return         <code>true</code> iff the associated Java file did not
103     *                 exist or it contained the expected comments and those
104     *                 comments indicate that a revision has taken place.
105    **/
106   private boolean codeRevision(CodeGenerator node, boolean convert)
107   {
108     String name = node.getName();
109 
110     if (node instanceof LearningClassifierExpression)
111     {
112       String lcFilePath = name + ".lc";
113       if (Main.classDirectory != null)
114         lcFilePath = Main.classDirectory + File.separator + lcFilePath;
115 
116       File lcFile = new File(lcFilePath);
117       if (!lcFile.exists()) return true;
118     }
119 
120     name += ".java";
121     if (Main.generatedSourceDirectory != null)
122       name = Main.generatedSourceDirectory + File.separator + name;
123 
124     File javaSource = new File(name);
125     if (!javaSource.exists()) return true;
126 
127     BufferedReader in = null;
128     try { in = new BufferedReader(new FileReader(javaSource)); }
129     catch (Exception e)
130     {
131       System.err.println("Can't open '" + name + "' for input: " + e);
132       System.exit(1);
133     }
134 
135     String line1 = "";
136     String line2 = "";
137     try
138     {
139       line1 = in.readLine();
140       line2 = in.readLine();
141     }
142     catch (Exception e)
143     {
144       System.err.println("Can't read from '" + name + "': " + e);
145       System.exit(1);
146     }
147 
148     try { in.close(); }
149     catch (Exception e)
150     {
151       System.err.println("Can't close file '" + name + "': " + e);
152       System.exit(1);
153     }
154 
155     if (line1 == null || line2 == null || !line2.startsWith("// ")
156         || !TranslateToJava.disclaimer.equals(line1))
157     {
158       reportError(node.getLine(),
159           "The file '" + name + "' does not appear to have been generated by "
160           + "LBJ2, but LBJ2 needs to overwrite it.  Either remove the file, "
161           + "or change the name of the classifier in '" + Main.sourceFilename
162           + "'.");
163       return false;
164     }
165 
166     line2 = line2.substring(3);
167     String expected = null;
168 
169     if (convert)
170     {
171       StringBuffer buffer = node.shallow();
172 
173       PrintStream converter = null;
174       ByteArrayOutputStream converted = new ByteArrayOutputStream();
175       try
176       {
177         converter = new PrintStream(
178                       new GZIPOutputStream(
179                         new HexOutputStream(converted)));
180       }
181       catch (Exception e)
182       {
183         System.err.println("Could not create converter stream.");
184         System.exit(1);
185       }
186 
187       converter.print(buffer.toString());
188       converter.close();
189 
190       expected = converted.toString();
191     }
192     else expected = node.shallow().toString();
193 
194     return !line2.equals(expected);
195   }

198   /**
199     * Recursively propagates the information about which nodes are "affected".
200     *
201     * @param name The name of an affected node.
202    **/
203   private void propagateAffected(String name)
204   {
205     boolean isCompositeGenerator =
206       SemanticAnalysis.representationTable.get(name)
207       instanceof CompositeGenerator;
208     boolean isRevised = revisionStatus.get(name) == REVISED;
209     HashSet dependors = (HashSet) SemanticAnalysis.dependorGraph.get(name);
210 
211     assert dependors != null : "null entry in dependorGraph for " + name;
212 
213     for (Iterator I = dependors.iterator(); I.hasNext(); )
214     {
215       Object dependor = I.next();
216 
217       if (!revisionStatus.containsKey(dependor))
218       {
219         if (isCompositeGenerator && isRevised
220             && SemanticAnalysis.representationTable.get(dependor)
221                instanceof LearningClassifierExpression)
222           revisionStatus.put(dependor, REVISED);
223         else revisionStatus.put(dependor, AFFECTED);
224         propagateAffected((String) dependor);
225       }
226     }
227   }

230   /**
231     * Runs this pass on all nodes of the indicated type.
232     *
233     * @param list The node to process.
234    **/
235   public void run(DeclarationList list)
236   {
237     noChanges = true;
238     if (list.size() == 0) return;
239 
240     runOnChildren(list);
241 
242     noChanges = revisionStatus.size() == 0;
243     String[] revised =
244       (String[]) revisionStatus.keySet().toArray(new String[0]);
245     for (int i = 0; i < revised.length; ++i) propagateAffected(revised[i]);
246 
247     for (Iterator I = SemanticAnalysis.dependorGraph.keySet().iterator();
248          I.hasNext(); )
249     {
250       Object name = I.next();
251       if (!revisionStatus.containsKey(name))
252         revisionStatus.put(name, UNAFFECTED);
253     }
254   }

257   /**
258     * Runs this pass on all nodes of the indicated type.
259     *
260     * @param cn The node to process.
261    **/
262   public void run(ClassifierName cn)
263   {
264     if (cn.referent == cn.name) return;
265     if (codeRevision(cn, false))
266       revisionStatus.put(cn.name.toString(), REVISED);
267   }
