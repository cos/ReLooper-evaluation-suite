76    /**
77      * factorial of n.  Argument is of type double but must be, numerically, an
78      * integer factorial returned as double but is, numerically, should be an
79      * integer numerical rounding may makes this an approximation after n = 21
80     **/
81    public static double factorial(double n){
82      if(n<0 || (n-Math.floor(n))!=0)throw new IllegalArgumentException("\nn must be a positive integer\nIs a Gamma funtion [Fmath.gamma(x)] more appropriate?");
83      double f = 1.0D;
84      double iCount = 2.0D;
85      while(iCount<=n){
86        f*=iCount;
87        iCount += 1.0D;
88      }
89      return f;
90    }

92    /**
93      * log to base e of the factorial of n.  Argument is of type double but
94      * must be, numerically, an integer log[e](factorial) returned as double
95      * numerical rounding may makes this an approximation
96     **/
97    public static double logFactorial(double n){
98      if(n<0 || (n-Math.floor(n))!=0)throw new IllegalArgumentException("\nn must be a positive integer\nIs a Gamma funtion [Fmath.gamma(x)] more appropriate?");
99      double f = 0.0D;
100     double iCount = 2.0D;
101     while(iCount<=n){
102       f+=Math.log(iCount);
103       iCount += 1.0D;
104     }
105     return f;
106   }

108   /** Gamma function, Lanczos approximation (6 terms) */
109   public static double gamma(double x){
110 
111     double xcopy = x;
112     double first = x + lgfGamma + 0.5;
113     double second = lgfCoeff[0];
114     double fg = 0.0D;
115 
116     if(x>=0.0){
117       if(x>=1.0D && x-(int)x==0.0D){
118         fg = factorial(x)/x;
119       }
120       else{
121         first = Math.pow(first, x + 0.5)*Math.exp(-first);
122         for(int i=1; i<=lgfN; i++)second += lgfCoeff[i]/++xcopy;
123         fg = first*Math.sqrt(2.0*Math.PI)*second/x;
124       }
125     }
126     else{
127       fg = -Math.PI/(x*gamma(-x)*Math.sin(Math.PI*x));
128     }
129     return fg;
130   }

132   /**
133     * log to base e of the Gamma function, Lanczos approximation (6 terms).
134     * Retained for backward compatibility.
135    **/
136   public static double logGamma(double x){
137     double xcopy = x;
138     double fg = 0.0D;
139     double first = x + lgfGamma + 0.5;
140     double second = lgfCoeff[0];
141 
142     if(x>=0.0){
143       if(x>=1.0 && x-(int)x==0.0){
144         fg = logFactorial(x)-Math.log(x);
145       }
146       else{
147         first -= (x + 0.5)*Math.log(first);
148         for(int i=1; i<=lgfN; i++)second += lgfCoeff[i]/++xcopy;
149         fg = Math.log(Math.sqrt(2.0*Math.PI)*second/x) - first;
150       }
151     }
152     else{
153       fg = Math.PI/(gamma(1.0D-x)*Math.sin(Math.PI*x));
154 
155       if(fg!=1.0/0.0 && fg!=-1.0/0.0){
156         if(fg<0){
157           throw new IllegalArgumentException("\nThe gamma function is negative");
158         }
159         else{
160           fg = Math.log(fg);
161         }
162       }
163     }
164     return fg;
165   }

167   /**
168     * Incomplete fraction summation used in the method
169     * {@link #regularisedBetaFunction(double,double,double)}.  modified
170     * Lentz's method
171    **/
172   public static double contFract(double a, double b, double x){
173     int maxit = 500;
174     double eps = 3.0e-7;
175     double aplusb = a + b;
176     double aplus1 = a + 1.0D;
177     double aminus1 = a - 1.0D;
178     double c = 1.0D;
179     double d = 1.0D - aplusb*x/aplus1;
180     if(Math.abs(d)<FPMIN)d = FPMIN;
181     d = 1.0D/d;
182     double h = d;
183     double aa = 0.0D;
184     double del = 0.0D;
185     int i=1, i2=0;
186     boolean test=true;
187     while(test){
188       i2=2*i;
189       aa = i*(b-i)*x/((aminus1+i2)*(a+i2));
190       d = 1.0D + aa*d;
191       if(Math.abs(d)<FPMIN)d = FPMIN;
192       c = 1.0D + aa/c;
193       if(Math.abs(c)<FPMIN)c = FPMIN;
194       d = 1.0D/d;
195       h *= d*c;
196       aa = -(a+i)*(aplusb+i)*x/((a+i2)*(aplus1+i2));
197       d = 1.0D + aa*d;
198       if(Math.abs(d)<FPMIN)d = FPMIN;
199       c = 1.0D + aa/c;
200       if(Math.abs(c)<FPMIN)c = FPMIN;
201       d = 1.0D/d;
202       del = d*c;
203       h *= del;
204       i++;
205       if(Math.abs(del-1.0D) < eps)test=false;
206       if(i>maxit){
207         test=false;
208         System.out.println("Maximum number of iterations ("+maxit+") exceeded in Stat.contFract in Stat.incomplete Beta");
209       }
210     }
211     return h;
212   }

214   /**
215     * Regularised Incomplete Beta function.  Continued Fraction approximation
216     * (see Numerical recipies for details of method)
217    **/
218   public static double regularisedBetaFunction(double z, double w, double x){
219     if(x<0.0D || x>1.0D)throw new IllegalArgumentException("Argument x, "+x+", must be lie between 0 and 1 (inclusive)");
220     double ibeta = 0.0D;
221     if(x==0.0D){
222       ibeta=0.0D;
223     }
224     else{
225       if(x==1.0D){
226         ibeta=1.0D;
227       }
228       else{
229         // Term before continued fraction
230         ibeta = Math.exp(logGamma(z+w) - logGamma(z) - logGamma(w) + z*Math.log(x) + w*Math.log(1.0D-x));
231         // Continued fraction
232         if(x < (z+1.0D)/(z+w+2.0D)){
233           ibeta = ibeta*contFract(z, w, x)/z;
234         }
235         else{
236           // Use symmetry relationship
237           ibeta = 1.0D - ibeta*contFract(w, z, 1.0D-x)/w;
238         }
239       }
240     }
241     return ibeta;
242   }

246   /** Returns the Student's t cumulative distribution function probability */
247   public static double studentTcdf(double tValue, int df){
248     double ddf = (double)df;
249     double x = ddf/(ddf+tValue*tValue);
250     return 0.5D*(1.0D + (regularisedBetaFunction(ddf/2.0D, 0.5D, 1) - regularisedBetaFunction(ddf/2.0D, 0.5D, x))*sign(tValue));
251   }

254   /**
255     * Computes the multiplier for the standard error of the mean when finding
256     * a <i>(1 - alpha) * 100%</i> confidence interval.
257     *
258     * @param df     The degrees of freedom.
259     * @param alpha  The fraction of the distribution to leave outside the
260     *               interval.
261     * @return <i>m</i> such that <i>mu +- m s</i> represents a
262     *         <i>(1 - alpha) * 100%</i> confidence interval, where <i>mu</i>
263     *         is the sample mean and <i>s</i> is the sample's standard
264     *         deviation.
265    **/
266   public static double tTable(int df, double alpha)
267   {
268     double c = 1 - alpha / 2.0;
269     double max = 700, min = -700;
270     boolean same = false;
271 
272     while (!same)
273     {
274       double mid = (max + min) / 2.0;
275       if (studentTcdf(mid, df) < c)
276       {
277         same = min == mid;
278         min = mid;
279       }
280       else
281       {
282         same = max == mid;
283         max = mid;
284       }
285     }
286 
287     return (max + min) / 2.0;
288   }
