22    /**
23      * Simply converts the type of the given matrix from <code>double</code> to 
24      *
25      * @param m  The matrix.
26      * @return The type converted matrix.
27     **/
28    public static Double[][] wrapDouble(double[][] m)
29    {
30      Double[][] result = new Double[m.length][];
31  
32      for (int i = 0; i < m.length; ++i)
33        if (m[i] != null)
34        {
35          result[i] = new Double[m[i].length];
36          for (int j = 0; j < m[i].length; ++j)
37            result[i][j] = new Double(m[i][j]);
38        }
39  
40      return result;
41    }

97    /**
98      * Formats the given data into an ASCII table.  Any entry in any row of
99      * <code>data</code> set to <code>null</code> will become a single dash in
100     * the output.  All non-<code>null</code> entries will be rounded to three
101     * significant digits after the decimal place.
102     *
103     * @param columnLabels One label for each column in the output, starting
104     *                     with the column of row labels if one exists.  If
105     *                     <code>null</code>, no column label row will be
106     *                     returned.
107     * @param rowLabels    One label for each non-<code>null</code> row in
108     *                     <code>data</code>.  If <code>null</code>, no row
109     *                     label column will be returned.
110     * @param data         The numerical data.
111     * @return The string respresentation of the data.
112    **/
113   public static String[] tableFormat(String[] columnLabels,
114                                      String[] rowLabels, Double[][] data)
115   {
116     if (data == null || data.length == 0) return new String[0];
117     int dataColumns = 0;
118 
119     for (int i = 0; i < data.length; ++i)
120     {
121       int current = data[i] == null ? 0 : data[i].length;
122       dataColumns = Math.max(dataColumns, current);
123     }
124 
125     if (dataColumns == 0) return new String[0];
126 
127     int[] sigDigits = new int[dataColumns];
128     for (int i = 0; i < dataColumns; ++i)
129       sigDigits[i] = defaultSignificantDigits;
130     return tableFormat(columnLabels, rowLabels, data, sigDigits);
131   }

215   /**
216     * Formats the given data into an ASCII table.  All non-<code>null</code>
217     * entries will be rounded to three significant digits after the decimal
218     * place.
219     *
220     * @param columnLabels One label for each column in the output, starting
221     *                     with the column of row labels if one exists.  If
222     *                     <code>null</code>, no column label row will be
223     *                     returned.
224     * @param rowLabels    One label for each non-<code>null</code> row in
225     *                     <code>data</code>.  If <code>null</code>, no row
226     *                     label column will be returned.
227     * @param data         The numerical data.
228     * @param sigDigits    Significant digits specified on a column by column
229     *                     basis, only for <code>data</code> columns.
230     * @param dashRows     The indexes of rows in <code>data</code> which
231     *                     should be preceded by a row of dashes in the output.
232     * @return The string respresentation of the data.
233    **/
234   public static String[] tableFormat(String[] columnLabels,
235                                      String[] rowLabels, Double[][] data,
236                                      int[] sigDigits, int[] dashRows)
237   {
238     if (data == null || data.length == 0) return new String[0];
239     if (sigDigits == null) return tableFormat(columnLabels, rowLabels, data);
240 
241     int dataColumns = 0;
242 
243     for (int i = 0; i < data.length; ++i)
244     {
245       int current = data[i] == null ? 0 : data[i].length;
246       dataColumns = Math.max(dataColumns, current);
247     }
248 
249     if (dataColumns == 0) return new String[0];
250 
251     int columns = dataColumns;
252     if (rowLabels != null) ++columns;
253 
254     if (sigDigits.length < dataColumns)
255     {
256       int[] temp = new int[dataColumns];
257       System.arraycopy(sigDigits, 0, temp, 0, sigDigits.length);
258       for (int i = sigDigits.length; i < dataColumns; ++i)
259         temp[i] = defaultSignificantDigits;
260       sigDigits = temp;
261     }
262 
263     int dataRows = data.length;
264     int rows = dataRows;
265 
266     if (columnLabels != null)
267     {
268       if (columnLabels.length < columns)
269       {
270         String[] temp = new String[columns];
271         System.arraycopy(columnLabels, 0, temp,
272                          columns - columnLabels.length, columnLabels.length);
273         columnLabels = temp;
274       }
275 
276       for (int i = 0; i < columnLabels.length; ++i)
277         if (columnLabels[i] == null) columnLabels[i] = "";
278       ++rows;
279     }
280 
281     if (rowLabels != null)
282     {
283       String[] temp = new String[dataRows];
284       int length = Math.min(rowLabels.length, dataRows);
285       System.arraycopy(rowLabels, 0, temp, 0, length);
286       rowLabels = temp;
287 
288       for (int i = 0; i < rowLabels.length; ++i)
289         if (rowLabels[i] == null) rowLabels[i] = "";
290     }
291 
292     return asciiTableFormat(columns, dataColumns, rows, dataRows,
293                             columnLabels, rowLabels, data, sigDigits,
294                             dashRows);
295   }

298   /**
299     * Formats the given data into an ASCII table.  Any row of
300     * <code>data</code> set to <code>null</code> will become a row of dashes
301     * in the output.  Any entry in any row of <code>data</code> set to
302     * <code>null</code> will become a single dash in the output.  All non-null
303     * entries will be rounded to three significant digits after the decimal
304     * place.
305     *
306     * @param columns      The total number of columns in the table.
307     * @param dataColumns  The number of columns containing data in the table.
308     * @param rows         The total number of rows in the table.
309     * @param dataRows     The number of rows containing data in the table.
310     * @param columnLabels One label for each column in the output, starting
311     *                     with the column of row labels if one exists.  If
312     *                     <code>null</code>, no column label row will be
313     *                     returned.
314     * @param rowLabels    One label for each non-null row in
315     *                     <code>data</code>.  If <code>null</code>, no row
316     *                     label column will be returned.
317     * @param data         The numerical data.
318     * @param sigDigits    Significant digits specified on a column by column
319     *                     basis, only for <code>data</code> columns.
320     * @param dashRows     The indexes of rows in <code>data</code> which
321     *                     should be preceded by a row of dashes in the output.
322     * @return The string respresentation of the data.
323    **/
324   protected static String[] asciiTableFormat(
325       int columns, int dataColumns, int rows, int dataRows,
326       String[] columnLabels, String[] rowLabels, Double[][] data,
327       int[] sigDigits, int[] dashRows)
328   {
329     String[][] table = new String[rows][columns];
330     int dataRowStart = 0;
331     int dataColumnStart = 0;
332 
333     if (columnLabels != null)
334     {
335       dataRowStart = 1;
336       table[0] = columnLabels;
337     }
338 
339     if (rowLabels != null)
340     {
341       dataColumnStart = 1;
342       for (int i = 0; i < dataRows; ++i)
343         table[dataRowStart + i][0] = rowLabels[i];
344     }
345 
346     for (int i = 0; i < dataRows; ++i)
347       for (int j = 0; j < dataColumns; ++j)
348         table[dataRowStart + i][dataColumnStart + j] =
349           data[i] == null || data[i][j] == null
350             ? "-" : format(data[i][j].doubleValue(), sigDigits[j]);
351 
352     int[] columnWidths = new int[columns];
353 
354     for (int i = 0; i < rows; ++i)
355       for (int j = 0; j < columns; ++j)
356       {
357         int current = 0;
358         columnWidths[j] = Math.max(columnWidths[j], table[i][j].length());
359       }
360 
361     if (columnLabels != null)
362       for (int j = 0; j < columns; ++j)
363         table[0][j] = center(table[0][j], columnWidths[j]);
364     if (rowLabels != null)
365       for (int i = 0; i < dataRows; ++i)
366         table[dataRowStart + i][0] =
367           ljust(table[dataRowStart + i][0], columnWidths[0]);
368 
369     for (int i = 0; i < dataRows; ++i)
370         for (int j = 0; j < dataColumns; ++j)
371           table[dataRowStart + i][dataColumnStart + j] =
372             table[dataRowStart + i][dataColumnStart + j].equals("-")
373               ? center("-", columnWidths[dataColumnStart + j])
374               : rjust(table[dataRowStart + i][dataColumnStart + j],
375                       columnWidths[dataColumnStart + j]);
376 
377     Arrays.sort(dashRows);
378     int d = 0;
379     while (d < dashRows.length && dashRows[d] < 0) ++d;
380 
381     String[] result = new String[rows + dashRows.length - d];
382     String dashes = "";
383     for (int i = 0; i < columnWidths.length - 1; ++i) dashes += "-";
384     for (int j = 0; j < columnWidths.length; ++j)
385       for (int i = 0; i < columnWidths[j]; ++i)
386         dashes += "-";
387     int r = 0;
388 
389     for (int i = 0; i < result.length; ++i)
390       if (d < dashRows.length && (r == dashRows[d] + 1 || r == rows))
391       {
392         result[i] = dashes;
393         ++d;
394       }
395       else
396       {
397         result[i] = table[r][0];
398         for (int j = 1; j < columns; ++j)
399           result[i] += " " + table[r][j];
400         ++r;
401       }
402 
403     return result;
404   }

407   /**
408     * Transposes the given matrix so that the rows become the columns and the
409     * columns become the rows.
410     *
411     * @param m  The matrix to transpose.
412     * @return The transposed matrix.
413    **/
414   public static double[][] transpose(double[][] m)
415   {
416     if (m == null) return null;
417     if (m.length == 0) return new double[0][];
418     int columns = 0;
419     int rows = 0;
420 
421     for (int i = 0; i < m.length; ++i)
422       if (m[i] != null)
423       {
424         rows = Math.max(rows, m[i].length);
425         ++columns;
426       }
427 
428     double[][] result = new double[rows][];
429 
430     for (int i = 0; i < result.length; ++i)
431     {
432       result[i] = new double[columns];
433       int c = 0;
434 
435       for (int j = 0; j < m.length; ++j)
436         if (m[j] != null)
437         {
438           if (m[j].length > i) result[i][c] = m[j][i];
439           ++c;
440         }
441     }
442 
443     return result;
444   }

487   /**
488     * Formats a floating point number so that it is rounded and zero-padded to
489     * the given number of significant digits after the decimal point.
490     *
491     * @param f  The floating point value to format.
492     * @param d  The number of significant digits to round to.  If less than 0,
493     *           this method assumes 0; if greater than 18, this method assumes
494     *           18.
495     * @return   The formatted result.
496    **/
497   protected static String format(double f, int d)
498   {
499     if (d > 18) d = 18;
500     long m = 1;
501     for (int j = 0; j < d; ++j) m *= 10;
502 
503     String unformatted = "" + f;
504     String sign = "";
505 
506     if (unformatted.startsWith("-"))
507     {
508       sign = "-";
509       unformatted = unformatted.substring(1);
510       f *= -1;
511     }
512 
513     int leftDigits = 0;
514     int p = unformatted.indexOf('.');
515     if (p == -1) leftDigits = unformatted.length();
516     else leftDigits = p;
517 
518     String base = unformatted;
519     String exponent = "";
520     int e = unformatted.indexOf('E');
521 
522     if (e != -1)
523     {
524       base = unformatted.substring(0, e);
525       exponent = unformatted.substring(e);
526       f = Double.parseDouble(base);
527     }
528 
529     StringBuffer buffer = new StringBuffer("" + Math.round(f * m));
530     while (buffer.length() < d + leftDigits) buffer.insert(0, '0');
531     if (d > 0) buffer.insert(buffer.length() - d, '.');
532     return sign + buffer + exponent;
533   }

536   /**
537     * Returns a space-padded string of at least the specified width such that
538     * the argument string is left-justified within the returned string.
539     *
540     * @param original The string to justify.
541     * @param width    The width within which the original string should be
542     *                 padded.
543     * @return The padded string.  If the original is longer than the specified
544     *         width, the original is simply returned.
545    **/
546   protected static String ljust(String original, int width)
547   {
548     while (original.length() < width) original += " ";
549     return original;

570   /**
571     * Returns a space-padded string of at least the specified width such that
572     * the argument string is centered within the returned string.
573     *
574     * @param original The string to justify.
575     * @param width    The width within which the original string should be
576     *                 padded.
577     * @return The padded string.  If the original is longer than the specified
578     *         width, the original is simply returned.
579    **/
580   protected static String center(String original, int width)
581   {
582     int toAdd = width - original.length(), i;
583     for (i = 0; i < toAdd / 2; ++i) original = " " + original;
584     for (; i < toAdd; ++i) original += " ";
585     return original;
586   }
