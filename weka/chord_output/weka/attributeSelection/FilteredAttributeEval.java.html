41  /**
42   <!-- globalinfo-start -->
43   * Class for running an arbitrary attribute evaluator on data that has been passed through an 
44   * arbitrary filter (note: filters that alter the order or number of attributes are not allowed). 
45   * Like the evaluator, the structure of the filter is based exclusively on the training data.
46   * <p/>
47   <!-- globalinfo-end -->
48   *
49   <!-- options-start -->
50   * Valid options are: <p/>
51   * 
52   * <pre> -W &lt;evaluator specification&gt;
53   *  Full name of base evaluator to use, followed by evaluator options.
54   *  eg: "weka.attributeSelection.InfoGainAttributeEval -M"</pre>
55   * 
56   * <pre> -F &lt;filter specification&gt;
57   *  Full class name of filter to use, followed
58   *  by filter options.
59   *  eg: "weka.filters.supervised.instance.SpreadSubsample -M 1"</pre>
60   * 
61   <!-- options-end -->
62   *
63   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
64   * @version $Revision: 5563 $
65   */
66  public class FilteredAttributeEval

86    /**
87     * Returns default capabilities of the evaluator.
88     *
89     * @return      the capabilities of this evaluator.
90     */
91    public Capabilities getCapabilities() {
92      Capabilities result;
93  
94      if (getFilter() == null) {
95        result = super.getCapabilities();
96        result.disableAll();
97      } else {
98        result = getFilter().getCapabilities();
99      }
100     
101     // set dependencies
102     for (Capability cap: Capability.values()) {
103       result.enableDependency(cap);
104     }
105     
106     return result;
107   }

143   /**
144    * Parses a given list of options. <p/>
145    *
146    <!-- options-start -->
147    * Valid options are: <p/>
148    * 
149    * <pre> -W &lt;evaluator specification&gt;
150    *  Full name of base evaluator to use, followed by evaluator options.
151    *  eg: "weka.attributeSelection.InfoGainAttributeEval -M"</pre>
152    * 
153    * <pre> -F &lt;filter specification&gt;
154    *  Full class name of filter to use, followed
155    *  by filter options.
156    *  eg: "weka.filters.supervised.instance.SpreadSubsample -M 1"</pre>
157    * 
158    <!-- options-end -->  
159    *
160    * @param options the list of options as an array of strings
161    * @throws Exception if an option is not supported
162    */
163   public void setOptions(String[] options) throws Exception {
164     String evaluator = Utils.getOption('W', options);
165     
166     if (evaluator.length() > 0) { 
167       String[] evaluatorSpec = Utils.splitOptions(evaluator);
168       if (evaluatorSpec.length == 0) {
169         throw new IllegalArgumentException("Invalid evaluator specification string");
170       }
171       
172       String evaluatorName = evaluatorSpec[0];
173       evaluatorSpec[0] = "";
174       setAttributeEvaluator((ASEvaluation)Utils.forName(AttributeEvaluator.class,
175                                                         evaluatorName, evaluatorSpec));
176 
177     } else {      
178       setAttributeEvaluator(new InfoGainAttributeEval());
179     }
180 
181     // Same for filter
182     String filterString = Utils.getOption('F', options);
183     if (filterString.length() > 0) {
184       String [] filterSpec = Utils.splitOptions(filterString);
185       if (filterSpec.length == 0) {
186     throw new IllegalArgumentException("Invalid filter specification string");
187       }
188       String filterName = filterSpec[0];
189       filterSpec[0] = "";
190       setFilter((Filter) Utils.forName(Filter.class, filterName, filterSpec));
191     } else {
192       setFilter(new weka.filters.supervised.instance.SpreadSubsample());
193     }
194   }

198    *
199    * @return an array of strings suitable for passing to setOptions

201   public String[] getOptions() {
202     ArrayList<String> options = new ArrayList<String>();
203     
204     options.add("-W");
205     options.add(getEvaluatorSpec());
206 
207     options.add("-F");
208     options.add(getFilterSpec());
209     
210     return options.toArray(new String[0]);
211   }

215    *
216    * @return a String containing the name of the evalautor + any options
217    */
218   protected String getEvaluatorSpec() {
219     AttributeEvaluator a = m_evaluator;
220     if (a instanceof OptionHandler) {
221       return a.getClass().getName() + " "
222         + Utils.joinOptions(((OptionHandler)a).getOptions());
223     }
224     return a.getClass().getName();
225   }

307   /**
308    * Initializes a filtered attribute evaluator.
309    *
310    * @param data set of instances serving as training data 
311    * @throws Exception if the evaluator has not been 
312    * generated successfully
313    */
314   public void buildEvaluator(Instances data) throws Exception {
315     // can evaluator handle data?
316     getCapabilities().testWithFail(data);
317 
318     // Structure of original
319     Instances original = new Instances(data, 0);
320 
321     m_filter.setInputFormat(data);
322     data = Filter.useFilter(data, m_filter);
323 
324     // Can only proceed if filter has not altered the order or
325     // number of attributes in the data
326     if (data.numAttributes() != original.numAttributes()) {
327       throw new Exception("Filter must not alter the number of "
328                           +"attributes in the data!");
329     }
330 
331     // Check the class index (if set)
332     if (original.classIndex() >= 0) {
333       if (data.classIndex() != original.classIndex()) {
334         throw new Exception("Filter must not change the class attribute!");
335       }
336     }
337 
338     // check the order
339     for (int i = 0; i < original.numAttributes(); i++) {
340       if (!data.attribute(i).name().equals(original.attribute(i).name())) {
341         throw new Exception("Filter must not alter the order of the attributes!");
342       }
343     }
344 
345     // can the evaluator handle this data?
346     ((ASEvaluation)getAttributeEvaluator()).getCapabilities().testWithFail(data);
347     m_filteredInstances = data.stringFreeStructure();
348     
349     ((ASEvaluation)m_evaluator).buildEvaluator(data);
350   }

354    * evaluator.

361     return m_evaluator.evaluateAttribute(attribute);
362   }

364   /**
365    * Describe the attribute evaluator
366    * @return a description of the attribute evaluator as a string
367    */
368   public String toString() {
369     StringBuffer text = new StringBuffer();
370     
371     if (m_filteredInstances == null) {
372       text.append("Filtered attribute evaluator has not been built");
373     } else {
374       text.append("Filtered Attribute Evaluator");
375       text.append("\nFilter: " + getFilterSpec());
376       text.append("\nAttribute evaluator: " + getEvaluatorSpec());
377       text.append("\n\nFiltered header:\n");
378       text.append(m_filteredInstances);
379     }
380     text.append("\n");
381     return text.toString();
382   }
