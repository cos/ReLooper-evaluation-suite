24  package weka.experiment;
25  import java.io.Serializable;
26  
27  /**
28   * Interface for objects able to listen for results obtained
29   * by a ResultProducer
30   *
31   * @author Len Trigg (trigg@cs.waikato.ac.nz)
32   * @version $Revision: 1.7 $
33   */
34  
35  public interface ResultListener extends Serializable {
36  
37    /**
38     * Determines if there are any constraints (imposed by the
39     * destination) on additional result columns to be produced by
40     * resultProducers. Null should be returned if there are NO
41     * constraints, otherwise a list of column names should be
42     * returned as an array of Strings.
43     * @param rp the ResultProducer to which the constraints will apply
44     * @return an array of column names to which resutltProducer's
45     * additional results will be restricted.
46     * @exception Exception if an error occurs
47     */
48    String [] determineColumnConstraints(ResultProducer rp) 
49      throws Exception;
50  
51    /**
52     * Prepare for the results to be received.
53     *
54     * @param rp the ResultProducer that will generate the results
55     * @exception Exception if an error occurs during preprocessing.
56     */
57    void preProcess(ResultProducer rp) throws Exception;
58    
59    /**
60     * Perform any postprocessing. When this method is called, it indicates
61     * that no more results will be sent that need to be grouped together
62     * in any way.
63     *
64     * @param rp the ResultProducer that generated the results
65     * @exception Exception if an error occurs
66     */
67    void postProcess(ResultProducer rp) throws Exception;
68    
69    /**
70     * Accepts results from a ResultProducer.
71     *
72     * @param rp the ResultProducer that generated the results
73     * @param key an array of Objects (Strings or Doubles) that uniquely
74     * identify a result for a given ResultProducer with given compatibilityState
75     * @param result the results stored in an array. The objects stored in
76     * the array may be Strings, Doubles, or null (for the missing value).
77     * @exception Exception if the result could not be accepted.
78     */
79    void acceptResult(ResultProducer rp, Object [] key, Object [] result)
80      throws Exception;
81  
82    /**
83     * Determines whether the results for a specified key must be
84     * generated.
85     *
86     * @param rp the ResultProducer wanting to generate the results
87     * @param key an array of Objects (Strings or Doubles) that uniquely
88     * identify a result for a given ResultProducer with given compatibilityState
89     * @return true if the result should be generated
90     * @exception Exception if it could not be determined if the result 
91     * is needed.
92     */
93    boolean isResultRequired(ResultProducer rp, Object [] key) 
94      throws Exception;
95  
96  }
