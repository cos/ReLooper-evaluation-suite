40  /**
41   <!-- globalinfo-start -->
42   * Outputs the received results in arff format to a Writer. All results must be received before the instances can be written out.
43   * <p/>
44   <!-- globalinfo-end -->
45   *
46   <!-- options-start -->
47   * Valid options are: <p/>
48   * 
49   * <pre> -O &lt;file name&gt;
50   *  The filename where output will be stored. Use - for stdout.
51   *  (default temp file)</pre>
52   * 
53   <!-- options-end -->
54   * 
55   * @author Len Trigg (trigg@cs.waikato.ac.nz)
56   * @version $Revision: 5987 $
57   */
58  public class InstancesResultListener 
59    extends CSVResultListener {
60    
61    /** for serialization */
62    static final long serialVersionUID = -2203808461809311178L;
63  
64    /** Stores the instances created so far, before assigning to a header */
65    protected transient FastVector m_Instances;
66    
67    /** Stores the attribute types for each column */
68    protected transient int [] m_AttributeTypes;
69  
70    /** For lookup of indices given a string value for each nominal attribute */
71    protected transient Hashtable [] m_NominalIndexes;
72  
73    /** Contains strings seen so far for each nominal attribute */ 
74    protected transient FastVector [] m_NominalStrings;
75   
76    /** 
77     * Sets temporary file.
78     */
79    public InstancesResultListener() {
80  
81      File resultsFile;
82      try {
83        resultsFile = File.createTempFile("weka_experiment", ".arff");
84        resultsFile.deleteOnExit();
85      } catch (Exception e) {
86        System.err.println("Cannot create temp file, writing to standard out.");
87        resultsFile = new File("-");
88      }
89      setOutputFile(resultsFile);
90      setOutputFileName("");
91    } 
92  
93    /**
94     * Returns a string describing this result listener
95     * @return a description of the result listener suitable for
96     * displaying in the explorer/experimenter gui
97     */
98    public String globalInfo() {
99      return
100         "Outputs the received results in arff format to "
101       + "a Writer. All results must be received before the instances can be "
102       + "written out.";
103   }

105   /**
106    * Prepare for the results to be received.
107    *
108    * @param rp the ResultProducer that will generate the results
109    * @exception Exception if an error occurs during preprocessing.
110    */
111   public void preProcess(ResultProducer rp) throws Exception {
112 
113     m_RP = rp;
114     if ((m_OutputFile == null) || (m_OutputFile.getName().equals("-"))) {
115       m_Out = new PrintWriter(System.out, true);
116     } else {
117       m_Out = new PrintWriter(
118           new BufferedOutputStream(
119           new FileOutputStream(m_OutputFile)), true);
120     }
121 
122     Object [] keyTypes = m_RP.getKeyTypes();
123     Object [] resultTypes = m_RP.getResultTypes();
124 
125     m_AttributeTypes = new int [keyTypes.length + resultTypes.length];
126     m_NominalIndexes = new Hashtable [m_AttributeTypes.length];
127     m_NominalStrings = new FastVector [m_AttributeTypes.length];
128     m_Instances = new FastVector();
129 
130     for (int i = 0; i < m_AttributeTypes.length; i++) {
131       Object attribute = null;
132       if (i < keyTypes.length) {
133     attribute = keyTypes[i];
134       } else {
135     attribute = resultTypes[i - keyTypes.length];
136       }
137       if (attribute instanceof String) {
138     m_AttributeTypes[i] = Attribute.NOMINAL;
139     m_NominalIndexes[i] = new Hashtable();
140     m_NominalStrings[i] = new FastVector();
141       } else if (attribute instanceof Double) {
142     m_AttributeTypes[i] = Attribute.NUMERIC;
143       } else {
144     throw new Exception("Unknown attribute type in column " + (i + 1));
145       }
146     }
147   }
148   
149   /**
150    * Perform any postprocessing. When this method is called, it indicates
151    * that no more results will be sent that need to be grouped together
152    * in any way.
153    *
154    * @param rp the ResultProducer that generated the results
155    * @exception Exception if an error occurs
156    */
157   public void postProcess(ResultProducer rp) throws Exception {
158 
159     if (m_RP != rp) {
160       throw new Error("Unrecognized ResultProducer sending results!!");
161     }
162     String [] keyNames = m_RP.getKeyNames();
163     String [] resultNames = m_RP.getResultNames();
164     FastVector attribInfo = new FastVector();
165     for (int i = 0; i < m_AttributeTypes.length; i++) {
166       String attribName = "Unknown";
167       if (i < keyNames.length) {
168     attribName = "Key_" + keyNames[i];
169       } else {
170     attribName = resultNames[i - keyNames.length];
171       }
172       
173       switch (m_AttributeTypes[i]) {
174       case Attribute.NOMINAL:
175     if (m_NominalStrings[i].size() > 0) {
176       attribInfo.addElement(new Attribute(attribName,
177                           m_NominalStrings[i]));
178     } else {
179       attribInfo.addElement(new Attribute(attribName, (FastVector)null));
180     }
181     break;
182       case Attribute.NUMERIC:
183     attribInfo.addElement(new Attribute(attribName));
184     break;
185       case Attribute.STRING:
186     attribInfo.addElement(new Attribute(attribName, (FastVector)null));
187     break;
188       default:
189     throw new Exception("Unknown attribute type");
190       }
191     }
192 
193     Instances result = new Instances("InstanceResultListener", attribInfo, 
194                      m_Instances.size());
195     for (int i = 0; i < m_Instances.size(); i++) {
196       result.add((Instance)m_Instances.elementAt(i));
197     }
198 
199     m_Out.println(new Instances(result, 0));
200     for (int i = 0; i < result.numInstances(); i++) {
201       m_Out.println(result.instance(i));
202     }
203     
204     if (!(m_OutputFile == null) && !(m_OutputFile.getName().equals("-"))) {
205       m_Out.close();
206     }
207   }

209   /**
210    * Collects each instance and adjusts the header information.
211    *
212    * @param rp the ResultProducer that generated the result
213    * @param key The key for the results.
214    * @param result The actual results.
215    * @exception Exception if the result could not be accepted.
216    */
217   public void acceptResult(ResultProducer rp, Object[] key, Object[] result) 
218     throws Exception {
219 
220     if (m_RP != rp) {
221       throw new Error("Unrecognized ResultProducer sending results!!");
222     }
223     
224     Instance newInst = new DenseInstance(m_AttributeTypes.length);
225     for(int i = 0; i < m_AttributeTypes.length; i++) {
226       Object val = null;
227       if (i < key.length) {
228     val = key[i];
229       } else {
230     val = result[i - key.length];
231       }
232       if (val == null) {
233     newInst.setValue(i, Utils.missingValue());
234       } else {
235     switch (m_AttributeTypes[i]) {
236     case Attribute.NOMINAL:
237       String str = (String) val;
238       Double index = (Double)m_NominalIndexes[i].get(str);
239       if (index == null) {
240         index = new Double(m_NominalStrings[i].size());
241         m_NominalIndexes[i].put(str, index);
242         m_NominalStrings[i].addElement(str);
243       }
244       newInst.setValue(i, index.doubleValue());
245       break;
246     case Attribute.NUMERIC:
247       double dou = ((Double) val).doubleValue();
248       newInst.setValue(i, (double)dou);
249       break;
250     default:
251       newInst.setValue(i, Utils.missingValue());
252     }
253       }
254     }
255     m_Instances.addElement(newInst);
256   }
