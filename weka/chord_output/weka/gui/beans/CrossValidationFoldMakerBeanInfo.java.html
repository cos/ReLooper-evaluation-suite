23  package weka.gui.beans;
24  
25  import java.beans.*;
26  
27  /**
28   * BeanInfo class for the cross validation fold maker bean
29   *
30   * @author <a href="mailto:mhall@cs.waikato.ac.nz">Mark Hall</a>
31   * @version $Revision: 5017 $
32   */
33  public class CrossValidationFoldMakerBeanInfo 
34    extends AbstractTrainAndTestSetProducerBeanInfo {
35    
36    /**
37     * Return the property descriptors for this bean
38     *
39     * @return a <code>PropertyDescriptor[]</code> value
40     */
41    public PropertyDescriptor[] getPropertyDescriptors() {
42      try {
43        PropertyDescriptor p1;
44        PropertyDescriptor p2;
45        PropertyDescriptor p3;
46        p1 = new PropertyDescriptor("folds", CrossValidationFoldMaker.class);
47        p2 = new PropertyDescriptor("seed", CrossValidationFoldMaker.class);
48        p3 = new PropertyDescriptor("preserveOrder", CrossValidationFoldMaker.class);
49        PropertyDescriptor [] pds = { p1, p2, p3 };
50        return pds;
51      } catch (Exception ex) {
52        ex.printStackTrace();
53      }
54      return null;
55    }
56  
57    /**
58     * Return the bean descriptor for this bean
59     *
60     * @return a <code>BeanDescriptor</code> value
61     */
62    public BeanDescriptor getBeanDescriptor() {
63      return new BeanDescriptor(weka.gui.beans.CrossValidationFoldMaker.class,
64                    CrossValidationFoldMakerCustomizer.class);
65    }
66  }
