23  package weka.gui.beans;
24  
25  import java.beans.*;
26  
27  /**
28   * Bean info class for AbstractDataSource. All beans that extend
29   * AbstractDataSource might want to extend this class
30   *
31   * @author <a href="mailto:mhall@cs.waikato.ac.nz">Mark Hall</a>
32   * @version $Revision: 1.3 $
33   * @since 1.0
34   * @see SimpleBeanInfo
35   */
36  public class AbstractDataSourceBeanInfo extends SimpleBeanInfo {
37  
38    /**
39     * Get the event set descriptors pertinent to data sources
40     *
41     * @return an <code>EventSetDescriptor[]</code> value
42     */
43    public EventSetDescriptor [] getEventSetDescriptors() {
44      try {
45        EventSetDescriptor [] esds =  
46        { new EventSetDescriptor(DataSource.class, 
47                     "dataSet",
48                     DataSourceListener.class,
49                     "acceptDataSet"),
50      new EventSetDescriptor(DataSource.class, 
51                     "instance",
52                     InstanceListener.class,
53                     "acceptInstance")
54           };
55        return esds;
56      } catch (Exception ex) {
57        ex.printStackTrace();
58      }
59      return null;
60    }
61  }
