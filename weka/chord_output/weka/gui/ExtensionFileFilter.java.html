24  package weka.gui;
25  
26  import java.io.File;
27  import java.io.Serializable;
28  import java.io.FilenameFilter;
29  import javax.swing.filechooser.FileFilter;
30  
31  /**
32   * Provides a file filter for FileChoosers that accepts or rejects files
33   * based on their extension. Compatible with both java.io.FilenameFilter and
34   * javax.swing.filechooser.FileFilter (why there are two I have no idea).
35   *
36   * @author Len Trigg (trigg@cs.waikato.ac.nz)
37   * @version $Revision: 1.8 $
38   */
39  public class ExtensionFileFilter
40    extends FileFilter
41    implements FilenameFilter, Serializable {
42  
43    /** The text description of the types of files accepted */
44    protected String m_Description;
45  
46    /** The filename extensions of accepted files */
47    protected String[] m_Extension;
48  
49    /**
50     * Creates the ExtensionFileFilter
51     *
52     * @param extension the extension of accepted files.
53     * @param description a text description of accepted files.
54     */
55    public ExtensionFileFilter(String extension, String description) {
56      m_Extension = new String [1];
57      m_Extension[0] = extension;
58      m_Description = description;
59    }
60  
61    /**
62     * Creates an ExtensionFileFilter that accepts files that have any of
63     * the extensions contained in the supplied array.
64     *
65     * @param extensions an array of acceptable file extensions (as Strings).
66     * @param description a text description of accepted files.
67     */
68    public ExtensionFileFilter(String [] extensions, String description) {
69      m_Extension = extensions;
70      m_Description = description;
71    }
72    
73    /**
74     * Gets the description of accepted files.
75     *
76     * @return the description.
77     */
78    public String getDescription() {
79      
80      return m_Description;
81    }
82    
83    /**
84     * Returns a copy of the acceptable extensions.
85     * 
86     * @return the accepted extensions
87     */
88    public String[] getExtensions() {
89      return (String[]) m_Extension.clone();
90    }
91    
92    /**
93     * Returns true if the supplied file should be accepted (i.e.: if it
94     * has the required extension or is a directory).
95     *
96     * @param file the file of interest.
97     * @return true if the file is accepted by the filter.
98     */
99    public boolean accept(File file) {
100     
101     String name = file.getName().toLowerCase();
102     if (file.isDirectory()) {
103       return true;
104     }
105     for (int i = 0; i < m_Extension.length; i++) {
106       if (name.endsWith(m_Extension[i])) {
107     return true;
108       }
109     }
110     return false;
111   }
112   
113   /**
114    * Returns true if the file in the given directory with the given name
115    * should be accepted.
116    *
117    * @param dir the directory where the file resides.
118    * @param name the name of the file.
119    * @return true if the file is accepted.
120    */
121   public boolean accept(File dir, String name) {
122     return accept(new File(dir, name));
123   }
124 }
