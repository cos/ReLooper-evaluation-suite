38   <!-- globalinfo-start -->
39   * Applies several filters successively. In case all supplied filters are StreamableFilters, it will act as a streamable one, too.
40   * <p/>
41   <!-- globalinfo-end -->
42   *
43   <!-- options-start -->
44   * Valid options are: <p/>
45   * 
46   * <pre> -D
47   *  Turns on output of debugging information.</pre>
48   * 
49   * <pre> -F &lt;classname [options]&gt;
50   *  A filter to apply (can be specified multiple times).</pre>
51   * 
52   <!-- options-end -->
53   *
54   * @author  FracPete (fracpete at waikato dot ac dot nz)
55   * @version $Revision: 5501 $
56   * @see     weka.filters.StreamableFilter
57   */

71    protected boolean m_StreamableChecked = false;

84    /**
85     * Returns an enumeration describing the available options.
86     *
87     * @return        an enumeration of all the available options.
88     */
89    public Enumeration listOptions() {
90      Vector result = new Vector();
91      Enumeration enm = super.listOptions();
92      while (enm.hasMoreElements())
93        result.add(enm.nextElement());
94        
95      result.addElement(new Option(
96                "\tA filter to apply (can be specified multiple times).",
97                "F", 1, "-F <classname [options]>"));
98  
99      return result.elements();
100   }

102   /**
103    * Parses a list of options for this object. <p/>
104    *
105    <!-- options-start -->
106    * Valid options are: <p/>
107    * 
108    * <pre> -D
109    *  Turns on output of debugging information.</pre>
110    * 
111    * <pre> -F &lt;classname [options]&gt;
112    *  A filter to apply (can be specified multiple times).</pre>
113    * 
114    <!-- options-end -->
115    *
116    * @param options     the list of options as an array of strings
117    * @throws Exception  if an option is not supported
118    */
119   public void setOptions(String[] options) throws Exception {
120     String        tmpStr;
121     String        filter;
122     String[]      options2;
123     Vector        filters;
124 
125     super.setOptions(options);
126     
127     filters = new Vector();
128     while ((tmpStr = Utils.getOption("F", options)).length() != 0) {
129       options2    = Utils.splitOptions(tmpStr);
130       filter      = options2[0];
131       options2[0] = "";
132       filters.add(Utils.forName(Filter.class, filter, options2));
133     }
134 
135     // at least one filter
136     if (filters.size() == 0)
137       filters.add(new AllFilter());
138 
139     setFilters((Filter[]) filters.toArray(new Filter[filters.size()]));
140   }
141 
142   /**
143    * Gets the current settings of the filter.
144    *
145    * @return        an array of strings suitable for passing to setOptions
146    */
147   public String[] getOptions() {
148     Vector        result;
149     String[]      options;
150     int           i;
151 
152     result = new Vector();
153 
154     options = super.getOptions();
155     for (i = 0; i < options.length; i++)
156       result.add(options[i]);
157     
158     for (i = 0; i < getFilters().length; i++) {
159       result.add("-F");
160       result.add(getFilterSpec(getFilter(i)));
161     }
162 
163     return (String[]) result.toArray(new String[result.size()]);
164   }

167    * Returns the Capabilities of this filter.

169    * @return            the capabilities of this object
170    * @see               Capabilities

172   public Capabilities getCapabilities() {
173     if (getFilters().length == 0) {
174       Capabilities result = super.getCapabilities();
175       result.disableAll();

177       return result;
178     } else {
179       return getFilters()[0].getCapabilities();
180     }

236   /**
237    * returns the filter classname and the options as one string
238    * 
239    * @param filter  the filter to get the specs for
240    * @return        the classname plus options
241    */
242   protected String getFilterSpec(Filter filter) {
243     String        result;
244 
245     if (filter == null) {
246       result = "";
247     }
248     else {
249       result  = filter.getClass().getName();
250       if (filter instanceof OptionHandler)
251         result += " " 
252           + Utils.joinOptions(((OptionHandler) filter).getOptions());
253     }
254 
255     return result;
256   }

258   /**
259    * tests whether all the enclosed filters are streamable
260    *
261    * @return    true if all the enclosed filters are streamable
262    */
263   public boolean isStreamableFilter() {
264     int           i;
265 
266     if (!m_StreamableChecked) {
267       m_Streamable        = true;
268       m_StreamableChecked = true;
269 
270       for (i = 0; i < getFilters().length; i++) {
271         if (getFilter(i) instanceof MultiFilter)
272           m_Streamable = ((MultiFilter) getFilter(i)).isStreamableFilter();
273         else if (getFilter(i) instanceof StreamableFilter)
274           m_Streamable = true;
275         else
276           m_Streamable = false;
277 
278         if (!m_Streamable)
279           break;
280       }
281 
282       if (getDebug())
283         System.out.println("Streamable: " + m_Streamable);
284     }
285 
286     return m_Streamable;
287   }

289   /**
290    * Returns true if the output format is immediately available after the
291    * input format has been set and not only after all the data has been
292    * seen (see batchFinished()). This method should normally return true
293    * for a stream filter, since the data will be processed in a batch
294    * manner instead (or at least for the second batch of files, see
295    * m_FirstBatchDone).
296    *
297    * @return      true if the output format is immediately available
298    * @see         #batchFinished()
299    * @see         #setInputFormat(Instances)
300    * @see         #m_FirstBatchDone
301    */
302   protected boolean hasImmediateOutputFormat() {
303     return isStreamableFilter();
304   }
305   
306   /**
307    * Determines the output format based on the input format and returns 
308    * this. In case the output format cannot be returned immediately, i.e.,
309    * hasImmediateOutputFormat() returns false, then this method will called
310    * from batchFinished() after the call of preprocess(Instances), in which,
311    * e.g., statistics for the actual processing step can be gathered.
312    *
313    * @param inputFormat     the input format to base the output format on
314    * @return                the output format
315    * @throws Exception      in case the determination goes wrong
316    * @see                   #hasImmediateOutputFormat()
317    * @see                   #batchFinished()
318    * @see                   #preprocess(Instances)
319    */
320   protected Instances determineOutputFormat(Instances inputFormat) throws Exception {
321     Instances   result;
322     int         i;
323     
324     result = getInputFormat();
325     
326     for (i = 0; i < getFilters().length; i++) {
327       if (!isFirstBatchDone())
328     getFilter(i).setInputFormat(result);
329       result = getFilter(i).getOutputFormat();
330     }
331 
332     return result;
333   }

335   /**
336    * processes the given instance (may change the provided instance) and
337    * returns the modified version.
338    *
339    * @param instance    the instance to process
340    * @return            the modified data
341    * @throws Exception  in case the processing goes wrong
342    */
343   protected Instance process(Instance instance) throws Exception {
344     Instance    result;
345     int         i;
346     
347     result = (Instance) instance.copy();
348 
349     for (i = 0; i < getFilters().length; i++) {
350       getFilter(i).input(result);
351       result = getFilter(i).output();
352     }
353 
354     return result;
355   }
356 
357   /**
358    * Processes the given data (may change the provided dataset) and returns
359    * the modified version. This method is called in batchFinished().
360    * This implementation only calls process(Instance) for each instance
361    * in the given dataset.
362    *
363    * @param instances   the data to process
364    * @return            the modified data
365    * @throws Exception  in case the processing goes wrong
366    * @see               #batchFinished()
367    * @see               #process(Instance)
368    */
369   protected Instances process(Instances instances) throws Exception {
370     Instances     result;
371     int           i;
372 
373     result = instances;
374     
375     for (i = 0; i < getFilters().length; i++) {
376       if (!isFirstBatchDone())
377     getFilter(i).setInputFormat(result);
378       result = Filter.useFilter(result, getFilter(i));
379     }
380     
381     return result;
382   }
