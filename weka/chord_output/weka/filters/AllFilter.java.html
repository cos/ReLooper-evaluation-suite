24  package weka.filters;
25  
26  import weka.core.Capabilities;
27  import weka.core.Instance;
28  import weka.core.Instances;
29  import weka.core.RevisionUtils;
30  import weka.core.Capabilities.Capability;
31  
32  /** 
33   * A simple instance filter that passes all instances directly
34   * through. Basically just for testing purposes.
35   *
36   * @author Len Trigg (trigg@cs.waikato.ac.nz)
37   * @version $Revision: 5501 $
38   */
39  public class AllFilter
40    extends Filter
41    implements Sourcable {
42  
43    /** for serialization */
44    static final long serialVersionUID = 5022109283147503266L;
45  
46    /**
47     * Returns a string describing this filter
48     *
49     * @return a description of the filter suitable for
50     * displaying in the explorer/experimenter gui
51     */
52    public String globalInfo() {
53      return "An instance filter that passes all instances through unmodified."
54        + " Primarily for testing purposes.";
55    }
56  
57    /** 
58     * Returns the Capabilities of this filter.
59     *
60     * @return            the capabilities of this object
61     * @see               Capabilities
62     */
63    public Capabilities getCapabilities() {
64      Capabilities result = super.getCapabilities();
65      result.disableAll();
66  
67      // attributes
68      result.enableAllAttributes();
69      result.enable(Capability.MISSING_VALUES);
70      
71      // class
72      result.enableAllClasses();
73      result.enable(Capability.MISSING_CLASS_VALUES);
74      result.enable(Capability.NO_CLASS);
75      
76      return result;
77    }
78  
79    /**
80     * Sets the format of the input instances.
81     *
82     * @param instanceInfo    an Instances object containing the input 
83     *                instance structure (any instances contained 
84     *                in the object are ignored - only the structure 
85     *                is required).
86     * @return true       if the outputFormat may be collected immediately
87     * @throws Exception      if something goes wrong
88     */
89    public boolean setInputFormat(Instances instanceInfo) throws Exception {
90  
91      super.setInputFormat(instanceInfo);
92      setOutputFormat(instanceInfo);
93      return true;
94    }
95  
96  
97    /**
98     * Input an instance for filtering. Ordinarily the instance is processed
99     * and made available for output immediately. Some filters require all
100    * instances be read before producing output.
101    *
102    * @param instance the input instance
103    * @return true if the filtered instance may now be
104    * collected with output().
105    * @throws IllegalStateException if no input format has been defined.
106    */
107   public boolean input(Instance instance) {
108 
109     if (getInputFormat() == null) {
110       throw new IllegalStateException("No input instance format defined");
111     }
112     if (m_NewBatch) {
113       resetQueue();
114       m_NewBatch = false;
115     }
116     push((Instance)instance.copy());
117     return true;
118   }
119   
120   /**
121    * Returns a string that describes the filter as source. The
122    * filter will be contained in a class with the given name (there may
123    * be auxiliary classes),
124    * and will contain two methods with these signatures:
125    * <pre><code>
126    * // converts one row
127    * public static Object[] filter(Object[] i);
128    * // converts a full dataset (first dimension is row index)
129    * public static Object[][] filter(Object[][] i);
130    * </code></pre>
131    * where the array <code>i</code> contains elements that are either
132    * Double, String, with missing values represented as null. The generated
133    * code is public domain and comes with no warranty.
134    *
135    * @param className   the name that should be given to the source class.
136    * @param data    the dataset used for initializing the filter
137    * @return            the object source described by a string
138    * @throws Exception  if the source can't be computed
139    */
140   public String toSource(String className, Instances data) throws Exception {
141     StringBuffer        result;
142     
143     result = new StringBuffer();
144     
145     result.append("class " + className + " {\n");
146     result.append("\n");
147     result.append("  /**\n");
148     result.append("   * filters a single row\n");
149     result.append("   * \n");
150     result.append("   * @param i the row to process\n");
151     result.append("   * @return the processed row\n");
152     result.append("   */\n");
153     result.append("  public static Object[] filter(Object[] i) {\n");
154     result.append("    return i;\n");
155     result.append("  }\n");
156     result.append("\n");
157     result.append("  /**\n");
158     result.append("   * filters multiple rows\n");
159     result.append("   * \n");
160     result.append("   * @param i the rows to process\n");
161     result.append("   * @return the processed rows\n");
162     result.append("   */\n");
163     result.append("  public static Object[][] filter(Object[][] i) {\n");
164     result.append("    return i;\n");
165     result.append("  }\n");
166     result.append("}\n");
167     
168     return result.toString();
169   }
170   
171   /**
172    * Returns the revision string.
173    * 
174    * @return        the revision
175    */
176   public String getRevision() {
177     return RevisionUtils.extract("$Revision: 5501 $");
178   }
179 
180   /**
181    * Main method for testing this class.
182    *
183    * @param argv should contain arguments to the filter: use -h for help
184    */
185   public static void main(String [] argv) {
186     runFilter(new AllFilter(), argv);
187   }
188 }
