49  /** 
50   <!-- globalinfo-start -->
51   * Reduces the dimensionality of the data by projecting it onto a lower dimensional subspace using a random matrix with columns of unit length (i.e. It will reduce the number of attributes in the data while preserving much of its variation like PCA, but at a much less computational cost).<br/>
52   * It first applies the  NominalToBinary filter to convert all attributes to numeric before reducing the dimension. It preserves the class attribute.<br/>
53   * <br/>
54   * For more information, see:<br/>
55   * <br/>
56   * Dmitriy Fradkin, David Madigan: Experiments with random projections for machine learning. In: KDD '03: Proceedings of the ninth ACM SIGKDD international conference on Knowledge discovery and data mining, New York, NY, USA, 517-522, 003.
57   * <p/>
58   <!-- globalinfo-end -->
59   * 
60   <!-- technical-bibtex-start -->
61   * BibTeX:
62   * <pre>
63   * &#64;inproceedings{Fradkin003,
64   *    address = {New York, NY, USA},
65   *    author = {Dmitriy Fradkin and David Madigan},
66   *    booktitle = {KDD '03: Proceedings of the ninth ACM SIGKDD international conference on Knowledge discovery and data mining},
67   *    pages = {517-522},
68   *    publisher = {ACM Press},
69   *    title = {Experiments with random projections for machine learning},
70   *    year = {003}
71   * }
72   * </pre>
73   * <p/>
74   <!-- technical-bibtex-end -->
75   *
76   <!-- options-start -->
77   * Valid options are: <p/>
78   * 
79   * <pre> -N &lt;number&gt;
80   *  The number of dimensions (attributes) the data should be reduced to
81   *  (default 10; exclusive of the class attribute, if it is set).</pre>
82   * 
83   * <pre> -D [SPARSE1|SPARSE2|GAUSSIAN]
84   *  The distribution to use for calculating the random matrix.
85   *  Sparse1 is:
86   *    sqrt(3)*{-1 with prob(1/6), 0 with prob(2/3), +1 with prob(1/6)}
87   *  Sparse2 is:
88   *    {-1 with prob(1/2), +1 with prob(1/2)}
89   * </pre>
90   * 
91   * <pre> -P &lt;percent&gt;
92   *  The percentage of dimensions (attributes) the data should
93   *  be reduced to (exclusive of the class attribute, if it is set). This -N
94   *  option is ignored if this option is present or is greater
95   *  than zero.</pre>
96   * 
97   * <pre> -M
98   *  Replace missing values using the ReplaceMissingValues filter</pre>
99   * 
100  * <pre> -R &lt;num&gt;
101  *  The random seed for the random number generator used for
102  *  calculating the random matrix (default 42).</pre>
103  * 
104  <!-- options-end -->
105  *
106  * @author Ashraf M. Kibriya (amk14@cs.waikato.ac.nz) 
107  * @version $Revision: 5987 $ [1.0 - 22 July 2003 - Initial version (Ashraf M. Kibriya)]
108  */

213    * Parses a given list of options. <p/>

216    * Valid options are: <p/>
217    * 
218    * <pre> -N &lt;number&gt;
219    *  The number of dimensions (attributes) the data should be reduced to
220    *  (default 10; exclusive of the class attribute, if it is set).</pre>
221    * 
222    * <pre> -D [SPARSE1|SPARSE2|GAUSSIAN]
223    *  The distribution to use for calculating the random matrix.
224    *  Sparse1 is:
225    *    sqrt(3)*{-1 with prob(1/6), 0 with prob(2/3), +1 with prob(1/6)}
226    *  Sparse2 is:
227    *    {-1 with prob(1/2), +1 with prob(1/2)}
228    * </pre>
229    * 
230    * <pre> -P &lt;percent&gt;
231    *  The percentage of dimensions (attributes) the data should
232    *  be reduced to (exclusive of the class attribute, if it is set). This -N
233    *  option is ignored if this option is present or is greater
234    *  than zero.</pre>
235    * 
236    * <pre> -M
237    *  Replace missing values using the ReplaceMissingValues filter</pre>
238    * 
239    * <pre> -R &lt;num&gt;
240    *  The random seed for the random number generator used for
241    *  calculating the random matrix (default 42).</pre>
242    * 
243    <!-- options-end -->
244    *
245    * @param options the list of options as an array of strings
246    * @throws Exception if an option is not supported
247    */
248   public void setOptions(String[] options) throws Exception {
249 
250 
251     String mString = Utils.getOption('P', options);
252     if (mString.length() != 0) {
253     setPercent((double) Double.parseDouble(mString)); //setNumberOfAttributes((int) Integer.parseInt(mString));
254     } else {
255         setPercent(0);
256     mString = Utils.getOption('N', options);
257     if (mString.length() != 0) 
258         setNumberOfAttributes(Integer.parseInt(mString));       
259     else        
260         setNumberOfAttributes(10);
261     }    
262     
263     mString = Utils.getOption('R', options);
264     if(mString.length()!=0) {
265     setRandomSeed( Long.parseLong(mString) );
266     }
267 
268     mString = Utils.getOption('D', options);
269     if(mString.length()!=0) {
270     if(mString.equalsIgnoreCase("sparse1"))
271        setDistribution( new SelectedTag(SPARSE1, TAGS_DSTRS_TYPE) );
272     else if(mString.equalsIgnoreCase("sparse2"))
273        setDistribution( new SelectedTag(SPARSE2, TAGS_DSTRS_TYPE) );
274     else if(mString.equalsIgnoreCase("gaussian"))
275        setDistribution( new SelectedTag(GAUSSIAN, TAGS_DSTRS_TYPE) );      
276     }
277 
278     if(Utils.getFlag('M', options))
279     setReplaceMissingValues(true);
280     else
281     setReplaceMissingValues(false);
282 
283 
284    //if(Utils.getFlag('G', options))
285    //    setUseGaussian(true);
286    //else
287    //    setUseGaussian(false);
288     
289   }

291   /**
292    * Gets the current settings of the filter.
293    *
294    * @return an array of strings suitable for passing to setOptions
295    */
296   public String [] getOptions() {
297 
298     String [] options = new String [10];
299     int current = 0;
300 
301     //if (getUseGaussian()) {
302     //  options[current++] = "-G";
303     //}
304 
305     if (getReplaceMissingValues()) {
306       options[current++] = "-M";
307     }
308 
309     if (getPercent() == 0) {
310       options[current++] = "-N";
311       options[current++] = "" + getNumberOfAttributes();
312     }
313     else {
314       options[current++] = "-P";
315       options[current++] = "" + getPercent();
316     }
317     
318     options[current++] = "-R";
319     options[current++] = "" + getRandomSeed();
320     
321     SelectedTag t = getDistribution();
322     options[current++] = "-D";
323     options[current++] = ""+t.getSelectedTag().getReadable();
324 
325 
326     while (current < options.length) {
327       options[current++] = "";
328     }
329 
330     return options;
331   }

565   /**
566    * Sets the format of the input instances.
567    *
568    * @param instanceInfo an Instances object containing the input 
569    * instance structure (any instances contained in the object are 
570    * ignored - only the structure is required).
571    * @return true if the outputFormat may be collected immediately
572    * @throws Exception if the input format can't be set 
573    * successfully
574    */
575   public boolean setInputFormat(Instances instanceInfo) throws Exception {      
576     super.setInputFormat(instanceInfo);
577     /*
578     if (instanceInfo.classIndex() < 0) {
579       throw new UnassignedClassException("No class has been assigned to the instances");
580     }
581     */
582     
583     for(int i=0; i<instanceInfo.numAttributes(); i++) {        
584     if( i!=instanceInfo.classIndex() && instanceInfo.attribute(i).isNominal() ) {
585             if(instanceInfo.classIndex()>=0)
586                 m_ntob = new weka.filters.supervised.attribute.NominalToBinary();
587             else
588                 m_ntob = new weka.filters.unsupervised.attribute.NominalToBinary();
589             
590             break;
591     }
592     }
593 
594     //r.setSeed(m_rndmSeed); //in case the setRandomSeed() is not
595                            //called we better set the seed to its 
596                            //default value of 42.
597     boolean temp=true;
598     if(m_replaceMissing!=null) {
599     m_replaceMissing = new weka.filters.unsupervised.attribute.ReplaceMissingValues();
600     if(m_replaceMissing.setInputFormat(instanceInfo))
601         temp=true;
602     else
603         temp=false;
604     }
605     
606     if(m_ntob!=null) {
607     if(m_ntob.setInputFormat(instanceInfo)) {
608         setOutputFormat();
609         return temp && true;
610     }
611     else { 
612         return false;
613     }
614     }
615     else {
616     setOutputFormat();
617     return temp && true;
618     }
619   }

622   /**
623    * Input an instance for filtering.
624    *
625    * @param instance the input instance
626    * @return true if the filtered instance may now be
627    * collected with output().
628    * @throws IllegalStateException if no input format has been set
629    */
630   public boolean input(Instance instance) throws Exception {
631 
632     Instance newInstance=null;
633 
634     if (getInputFormat()==null) {
635     throw new IllegalStateException("No input instance format defined");
636     }
637     if(m_NewBatch) {
638       resetQueue();
639       //if(ntob!=null) 
640       //      ntob.m_NewBatch=true;
641       m_NewBatch = false;
642     }
643     
644     boolean replaceDone=false;
645     if(m_replaceMissing!=null) {
646     if(m_replaceMissing.input(instance)) {
647         if(m_OutputFormatDefined == false)
648         setOutputFormat();
649         newInstance = m_replaceMissing.output();
650         replaceDone = true;
651     }
652     else
653         return false;;
654     }
655 
656     if(m_ntob!=null) {
657     if(replaceDone==false)
658         newInstance = instance;
659     if(m_ntob.input(newInstance)) {
660         if(m_OutputFormatDefined == false) 
661         setOutputFormat();
662         newInstance = m_ntob.output();
663         newInstance = convertInstance(newInstance);
664         push(newInstance);
665         return true;    
666     }
667     else {
668         return false;
669     }
670     }
671     else {
672     if(replaceDone==false)
673         newInstance = instance;
674     newInstance = convertInstance(newInstance);
675     push(newInstance);
676     return true;
677     }
678   }

681   /**
682    * Signify that this batch of input to the filter is finished.
683    *
684    * @return true if there are instances pending output
685    * @throws NullPointerException if no input structure has been defined,
686    * @throws Exception if there was a problem finishing the batch.
687    */
688   public boolean batchFinished() throws Exception {
689       if (getInputFormat() == null) {
690       throw new NullPointerException("No input instance format defined");
691       }
692       
693       boolean conversionDone=false;
694       if(m_replaceMissing!=null) {
695       if(m_replaceMissing.batchFinished()) {
696           Instance newInstance, instance;
697           
698           while((instance=m_replaceMissing.output())!=null) {
699           if(!m_OutputFormatDefined)
700               setOutputFormat();
701           if(m_ntob!=null) {
702               m_ntob.input(instance);
703           }
704           else {
705               newInstance = convertInstance(instance);
706               push(newInstance);
707           }
708           }
709 
710           if(m_ntob!=null) {
711           if(m_ntob.batchFinished()) {
712               //Instance newInstance, instance;
713               while((instance=m_ntob.output())!=null) {
714               if(!m_OutputFormatDefined)
715                   setOutputFormat();
716               newInstance = convertInstance(instance);
717               push(newInstance);
718               }
719               m_ntob = null;              
720           }
721           }
722           m_replaceMissing = null;
723           conversionDone=true;
724       }
725       }
726 
727       if(conversionDone==false && m_ntob!=null) {
728       if(m_ntob.batchFinished()) {
729           Instance newInstance, instance;
730           while((instance=m_ntob.output())!=null) {
731           if(!m_OutputFormatDefined)
732               setOutputFormat();
733           newInstance = convertInstance(instance);
734           push(newInstance);
735           }
736           m_ntob = null;
737       }
738       }
739       m_OutputFormatDefined=false;
740       return super.batchFinished();
741   }

744   /** Sets the output format */  
745   protected void setOutputFormat() {
746       Instances currentFormat;
747       if(m_ntob!=null) {
748       currentFormat = m_ntob.getOutputFormat();
749       }
750       else 
751       currentFormat = getInputFormat();
752       
753       if(m_percent>0)
754       { m_k = (int) ((getInputFormat().numAttributes()-1)*m_percent); 
755       // System.out.print("numAtts: "+currentFormat.numAttributes());
756       // System.out.print("percent: "+m_percent);
757       // System.out.print("percent*numAtts: "+(currentFormat.numAttributes()*m_percent));
758       // System.out.println("m_k: "+m_k);
759       }
760 
761       Instances newFormat;
762       int newClassIndex=-1;
763       FastVector attributes = new FastVector();
764       for(int i=0; i<m_k; i++) {
765       attributes.addElement( new Attribute("K"+(i+1)) );
766       }
767       if(currentFormat.classIndex()!=-1)  {  //if classindex is set
768       //attributes.removeElementAt(attributes.size()-1);
769       attributes.addElement(currentFormat.attribute(currentFormat.classIndex()));
770       newClassIndex = attributes.size()-1;
771       }
772 
773       newFormat = new Instances(currentFormat.relationName(), attributes, 0);
774       if(newClassIndex!=-1)
775       newFormat.setClassIndex(newClassIndex);
776       m_OutputFormatDefined=true;
777 
778       m_random = new Random();
779       m_random.setSeed(m_rndmSeed);
780 
781       m_rmatrix = new double[m_k][currentFormat.numAttributes()];
782       if(m_distribution==GAUSSIAN) {
783       for(int i=0; i<m_rmatrix.length; i++) 
784           for(int j=0; j<m_rmatrix[i].length; j++) 
785           m_rmatrix[i][j] = m_random.nextGaussian();
786       }
787       else {
788       boolean useDstrWithZero = (m_distribution==SPARSE1);
789       for(int i=0; i<m_rmatrix.length; i++) 
790           for(int j=0; j<m_rmatrix[i].length; j++) 
791           m_rmatrix[i][j] = rndmNum(useDstrWithZero);
792       }
793 
794       setOutputFormat(newFormat);
795   }

797   /**
798    * converts a single instance to the required format
799    *
800    * @param currentInstance     the instance to convert
801    * @return                    the converted instance
802    */
803   protected Instance convertInstance(Instance currentInstance) {
804 
805       Instance newInstance;
806       double vals[] = new double[getOutputFormat().numAttributes()];
807       int classIndex = (m_ntob==null) ? getInputFormat().classIndex():m_ntob.getOutputFormat().classIndex();
808 
809       for(int i = 0; i < m_k; i++) {
810         vals[i] = computeRandomProjection(i,classIndex,currentInstance);
811       }
812       if (classIndex != -1) {
813         vals[m_k] = currentInstance.value(classIndex);
814       }
815 
816       newInstance = new DenseInstance(currentInstance.weight(), vals);
817       newInstance.setDataset(getOutputFormat());
818 
819       return newInstance;
820   }

823   /**
824    * computes one random projection for a given instance (skip missing values)
825    *
826    * @param rpIndex     offset the new random projection attribute
827    * @param classIndex  classIndex of the input instance
828    * @param instance    the instance to convert
829    * @return    the random sum
830    */
831 
832   protected double computeRandomProjection(int rpIndex, int classIndex, Instance instance) {
833 
834     double sum = 0.0;
835     for(int i = 0; i < instance.numValues(); i++) {
836       int index = instance.index(i);
837       if (index != classIndex) {
838         double value = instance.valueSparse(i);
839         if (!Utils.isMissingValue(value)) {
840           sum += m_rmatrix[rpIndex][index] * value;
841         }
842       }
843     }
844     return sum;
845   }

847   private static final int weights[] = {1, 1, 4};
848   private static final int vals[] = {-1, 1, 0};
849   private static final int weights2[] = {1, 1};
850   private static final int vals2[] = {-1, 1};
851   private static final double sqrt3 = Math.sqrt(3);

853   /**
854    * returns a double x such that <br/>
855    *      x = sqrt(3) * { -1 with prob. 1/6, 0 with prob. 2/3, 1 with prob. 1/6 }
856    *      
857    * @param useDstrWithZero
858    * @return the generated number
859    */
860   protected double rndmNum(boolean useDstrWithZero) {
861       if(useDstrWithZero)
862       return sqrt3 * vals[weightedDistribution(weights)];
863       else
864       return vals2[weightedDistribution(weights2)];
865   }

867   /** 
868    * Calculates a weighted distribution
869    * 
870    * @param weights the weights to use
871    * @return
872    */
873   protected int weightedDistribution(int [] weights) {
874       int sum=0; 
875       
876       for(int i=0; i<weights.length; i++) 
877       sum += weights[i];
878       
879       int val = (int)Math.floor(m_random.nextDouble()*sum);
880       
881       for(int i=0; i<weights.length; i++) {
882       val -= weights[i];
883       if(val<0)
884           return i;
885       }
886       return -1;
887   }
