47   <!-- globalinfo-start -->
48   * Merges many values of a nominal attribute into one value.
49   * <p/>
50   <!-- globalinfo-end -->
51   * 
52   <!-- options-start -->
53   * Valid options are: <p/>
54   * 
55   * <pre> -C &lt;col&gt;
56   *  Sets the attribute index
57   *  (default: last)</pre>
58   * 
59   * <pre> -M &lt;label&gt;
60   *  Sets the label of the newly merged classes
61   *  (default: 'merged')</pre>
62   * 
63   * <pre> -R &lt;range&gt;
64   *  Sets the merge range. 'first and 'last' are accepted as well.'
65   *  E.g.: first-5,7,9,20-last
66   *  (default: 1,2)</pre>
67   * 
68   <!-- options-end -->
69   *
70   * @author Kathryn Hempstalk (kah18 at cs.waikato.ac.nz)

127   /**
128    * Parses a given list of options. <p/>
129    * 
130    <!-- options-start -->
131    * Valid options are: <p/>
132    * 
133    * <pre> -C &lt;col&gt;
134    *  Sets the attribute index
135    *  (default: last)</pre>
136    * 
137    * <pre> -M &lt;label&gt;
138    *  Sets the label of the newly merged classes
139    *  (default: 'merged')</pre>
140    * 
141    * <pre> -R &lt;range&gt;
142    *  Sets the merge range. 'first and 'last' are accepted as well.'
143    *  E.g.: first-5,7,9,20-last
144    *  (default: 1,2)</pre>
145    * 
146    <!-- options-end -->
147    *
148    * @param options     the list of options as an array of strings
149    * @throws Exception  if an option is not supported
150    */
151   public void setOptions(String[] options) throws Exception {
152     String  tmpStr;
153     
154     tmpStr = Utils.getOption('C', options);
155     if (tmpStr.length() != 0) {
156       setAttributeIndex(tmpStr);
157     } else {
158       setAttributeIndex("last");
159     }
160 
161     tmpStr = Utils.getOption('L', options);
162     if (tmpStr.length() != 0) {
163       setLabel(tmpStr);
164     } else {
165       setLabel("merged");
166     }
167 
168     tmpStr = Utils.getOption('R', options);
169     if (tmpStr.length() != 0) {
170       setMergeValueRange(tmpStr);
171     } else {
172       setMergeValueRange("1,2");
173     }
174 
175     if (getInputFormat() != null) {
176       setInputFormat(getInputFormat());
177     }
178   }

223   /**
224    * Sets the format of the input instances.
225    *
226    * @param instanceInfo    an Instances object containing the input 
227    *                instance structure (any instances contained 
228    *                in the object are ignored - only the structure 
229    *                is required).
230    * @return            true if the outputFormat may be collected immediately
231    * @throws Exception      if the input format can't be set 
232    *                successfully
233    */
234   public boolean setInputFormat(Instances instanceInfo) throws Exception {
235     super.setInputFormat(instanceInfo);
236     
237     m_AttIndex.setUpper(instanceInfo.numAttributes() - 1);
238 
239     m_MergeRange.setUpper(instanceInfo.attribute(m_AttIndex.getIndex()).numValues() - 1);
240     if ((instanceInfo.classIndex() > -1) && (instanceInfo.classIndex() == m_AttIndex.getIndex())) {
241       throw new Exception("Cannot process class attribute.");
242     }
243     if (!instanceInfo.attribute(m_AttIndex.getIndex()).isNominal()) {
244       throw new UnsupportedAttributeTypeException("Chosen attribute not nominal.");
245     }
246     if (instanceInfo.attribute(m_AttIndex.getIndex()).numValues() < 2) {
247       throw new UnsupportedAttributeTypeException("Chosen attribute has less than " +
248       "two values.");
249     }
250 
251     setOutputFormat();
252     return true;
253   }
254 
255   /**
256    * Set the output format. Takes the current average class values
257    * and m_InputFormat and calls setOutputFormat(Instances) 
258    * appropriately.
259    */
260   private void setOutputFormat() {
261     Instances newData;
262     FastVector newAtts, newVals;
263 
264     // Compute new attributes
265     newAtts = new FastVector(getInputFormat().numAttributes());
266     for (int j = 0; j < getInputFormat().numAttributes(); j++) {
267       Attribute att = getInputFormat().attribute(j);
268       if (j != m_AttIndex.getIndex()) {
269     newAtts.addElement(att.copy());
270       } else {
271 
272     // Compute list of attribute values   
273     newVals = new FastVector(att.numValues() - 1);
274     for (int i = 0; i < att.numValues(); i++) {
275       boolean inMergeList = false;
276 
277       if(att.value(i).equalsIgnoreCase(m_Label)){
278         //don't want to add this one.
279         inMergeList = true;     
280       }else{
281         inMergeList = m_MergeRange.isInRange(i);
282       }
283 
284       if(!inMergeList){
285         //add it.
286         newVals.addElement(att.value(i));
287       }
288     }
289     newVals.addElement(m_Label);
290 
291 
292     newAtts.addElement(new Attribute(att.name(), newVals));
293       }
294     }
295 
296     // Construct new header
297     newData = new Instances(getInputFormat().relationName(), newAtts, 0);
298     newData.setClassIndex(getInputFormat().classIndex());
299     setOutputFormat(newData);
300   }

302   /**
303    * Input an instance for filtering. The instance is processed
304    * and made available for output immediately.
305    *
306    * @param instance    the input instance
307    * @return        true if the filtered instance may now be
308    *            collected with output().
309    * @throws IllegalStateException  if no input format has been set.
310    */
311   public boolean input(Instance instance) {
312     if (getInputFormat() == null) {
313       throw new IllegalStateException("No input instance format defined");
314     }
315     if (m_NewBatch) {
316       resetQueue();
317       m_NewBatch = false;
318     }
319 
320     Attribute att = getInputFormat().attribute(m_AttIndex.getIndex());
321     FastVector newVals = new FastVector(att.numValues() - 1);
322     for (int i = 0; i < att.numValues(); i++) {
323       boolean inMergeList = false;
324 
325       if(att.value(i).equalsIgnoreCase(m_Label)){
326     //don't want to add this one.
327     inMergeList = true;     
328       }else{
329     inMergeList = m_MergeRange.isInRange(i);
330       }
331 
332       if(!inMergeList){
333     //add it.
334     newVals.addElement(att.value(i));
335       }
336     }
337     newVals.addElement(m_Label);
338 
339     Attribute temp = new Attribute(att.name(), newVals);
340 
341     Instance newInstance = (Instance)instance.copy();    
342     if (!newInstance.isMissing(m_AttIndex.getIndex())) {
343       String currValue = newInstance.stringValue(m_AttIndex.getIndex());
344       if(temp.indexOfValue(currValue) == -1)
345     newInstance.setValue(m_AttIndex.getIndex(), temp.indexOfValue(m_Label));
346       else
347     newInstance.setValue(m_AttIndex.getIndex(), temp.indexOfValue(currValue));
348     }
349 
350     push(newInstance);
351     return true;
352   }
