21  package weka.filters.unsupervised.attribute;
22  
23  import weka.core.Instances;
24  import weka.filters.AbstractFilterTest;
25  import weka.filters.Filter;
26  
27  import junit.framework.Test;
28  import junit.framework.TestSuite;
29  
30  /**
31   * Tests RandomSubset. Run from the command line with: <p/>
32   * java weka.filters.unsupervised.attribute.RandomSubsetTest
33   *
34   * @author FracPete (fracpete at waikato dot ac dot nz)
35   * @version $Revision: 1.1 $
36   */
37  public class RandomSubsetTest 
38    extends AbstractFilterTest {
39    
40    /**
41     * Initializes the test.
42     * 
43     * @param name    the name of the test
44     */
45    public RandomSubsetTest(String name) { 
46      super(name);  
47    }
48  
49    /**
50     * Creates a default RandomSubset.
51     * 
52     * @return        the filter
53     */
54    public Filter getFilter() {
55      return new RandomSubset();
56    }
57  
58    /**
59     * Creates a specialized RandomSubset.
60     * 
61     * @param num     the number of attributes
62     * @return        the filter
63     */
64    public Filter getFilter(double num) {
65      RandomSubset result = new RandomSubset();
66      result.setNumAttributes(num);
67      return result;
68    }
69  
70    /**
71     * performs the actual test.
72     * 
73     * @param numSel  the number of attributes to select
74     * @param numOut  the number of attributes that are expected
75     */
76    protected void performTest(double numSel, int numOut) {
77      m_Filter         = getFilter(numSel);
78      Instances icopy  = new Instances(m_Instances);
79      Instances result = null;
80      try {
81        m_Filter.setInputFormat(icopy);
82      } 
83      catch (Exception ex) {
84        ex.printStackTrace();
85        fail("Exception thrown on setInputFormat(): \n" + ex.getMessage());
86      }
87      try {
88        result = Filter.useFilter(icopy, m_Filter);
89        assertNotNull(result);
90      } 
91      catch (Exception ex) {
92        ex.printStackTrace();
93        fail("Exception thrown on useFilter(): \n" + ex.getMessage());
94      }
95  
96      assertEquals(numOut, result.numAttributes());
97      assertEquals(icopy.numInstances(), m_Instances.numInstances());
98    }
99  
100   /**
101    * Tests a percentage.
102    */
103   public void testPercentage() {
104     performTest(0.5, 4);
105   }
106 
107   /**
108    * Tests an absolute number.
109    */
110   public void testAbsolute() {
111     performTest(5, 5);
112   }
113 
114   /**
115    * Returns a test suite.
116    * 
117    * @return        the test suite
118    */
119   public static Test suite() {
120     return new TestSuite(RandomSubsetTest.class);
121   }
122 
123   /**
124    * Runs the test from commandline.
125    * 
126    * @param args    ignored
127    */
128   public static void main(String[] args){
129     junit.textui.TestRunner.run(suite());
130   }
131 }
