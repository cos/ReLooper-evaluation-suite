21  package weka.filters.unsupervised.attribute;
22  
23  import weka.classifiers.meta.FilteredClassifier;
24  import weka.core.Instances;
25  import weka.core.TestInstances;
26  import weka.filters.AbstractFilterTest;
27  import weka.filters.Filter;
28  
29  import junit.framework.Test;
30  import junit.framework.TestSuite;
31  
32  /**
33   * Tests ClusterMembership. Run from the command line with: <p/>
34   * java weka.filters.unsupervised.attribute.ClusterMembershipTest
35   *
36   * @author FracPete (fracpete at waikato dot ac dot nz)
37   * @version $Revision: 1.4 $
38   */
39  public class ClusterMembershipTest 
40    extends AbstractFilterTest {
41    
42    public ClusterMembershipTest(String name) { 
43      super(name);  
44    }
45  
46    /** Need to remove non-nominal/numeric attributes, set class index */
47    protected void setUp() throws Exception {
48      super.setUp();
49      
50      // remove attributes that are not nominal/numeric
51      int i = 0;
52      while (i < m_Instances.numAttributes()) {
53        if (   (    !m_Instances.attribute(i).isNominal()
54                 && !m_Instances.attribute(i).isNumeric() )
55            || m_Instances.attribute(i).isDate() )
56          m_Instances.deleteAttributeAt(i);
57        else
58          i++;
59      }
60  
61      // class index
62      m_Instances.setClassIndex(1);
63    }
64    
65    /** Creates a default ClusterMembership */
66    public Filter getFilter() {
67      ClusterMembership f = new ClusterMembership();
68      return f;
69    }
70  
71    /**
72     * returns the configured FilteredClassifier. Since the base classifier is
73     * determined heuristically, derived tests might need to adjust it.
74     * 
75     * @return the configured FilteredClassifier
76     */
77    protected FilteredClassifier getFilteredClassifier() {
78      FilteredClassifier  result;
79      
80      result = new FilteredClassifier();
81      
82      result.setFilter(getFilter());
83      result.setClassifier(new weka.classifiers.trees.J48());
84      
85      return result;
86    }
87    
88    /**
89     * returns data generated for the FilteredClassifier test
90     * 
91     * @return        the dataset for the FilteredClassifier
92     * @throws Exception  if generation of data fails
93     */
94    protected Instances getFilteredClassifierData() throws Exception{
95      TestInstances   test;
96      Instances       result;
97  
98      test = TestInstances.forCapabilities(m_FilteredClassifier.getCapabilities());
99      test.setClassIndex(TestInstances.CLASS_IS_LAST);
100 
101     result = test.generate();
102     
103     return result;
104   }
105 
106   public void testNominal() {
107     m_Filter = getFilter();
108     m_Instances.setClassIndex(1);
109     Instances result = useFilter();
110     // classes must be still the same
111     assertEquals(m_Instances.numClasses(), result.numClasses());
112     // at least one cluster per label besides class
113     assertTrue(result.numAttributes() >= m_Instances.numClasses() + 1);
114   }
115 
116   public void testNumeric() {
117     m_Filter = getFilter();
118     m_Instances.setClassIndex(2);
119     Instances result = useFilter();
120     // at least one cluster (only one clusterer is generateed) besides class
121     assertTrue(result.numAttributes() >= 1 + 1);
122   }
123 
124   public static Test suite() {
125     return new TestSuite(ClusterMembershipTest.class);
126   }
127 
128   public static void main(String[] args){
129     junit.textui.TestRunner.run(suite());
130   }
131 }
