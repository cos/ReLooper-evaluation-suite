21  package weka.filters.unsupervised.attribute;
22  
23  import weka.core.Instances;
24  import weka.filters.AbstractFilterTest;
25  import weka.filters.Filter;
26  
27  import junit.framework.Test;
28  import junit.framework.TestSuite;
29  import junit.textui.TestRunner;
30  
31  /**
32   * Tests RenameAttribute. Run from the command line with: <p/>
33   * java weka.filters.unsupervised.attribute.RenameAttributeTest
34   *
35   * @author FracPete (fracpete at waikato dot ac dot nz)
36   * @version $Revision: 6107 $
37   */
38  public class RenameAttributeTest 
39    extends AbstractFilterTest {
40  
41    /**
42     * Initializes the test.
43     * 
44     * @param name    the name of the test
45     */
46    public RenameAttributeTest(String name) { 
47      super(name);  
48    }
49  
50    /** 
51     * Creates a default RenameAttribute.
52     * 
53     * @return        the default filter
54     */
55    public Filter getFilter() {
56      return new RenameAttribute();
57    }
58  
59    /** 
60     * Creates a specialized RenameAttribute.
61     * 
62     * @param find    the find expression
63     * @param replace the replace expression
64     * @param all     whether to replace all occurrences
65     * @param range   the range of attributes
66     * @param invert  whether to invert the selection
67     * @return        the filter
68     */
69    public Filter getFilter(String find, String replace, boolean all, String range, boolean invert) {
70      RenameAttribute     result;
71      
72      result = new RenameAttribute();
73      result.setFind(find);
74      result.setReplace(replace);
75      result.setReplaceAll(all);
76      result.setAttributeIndices(range);
77      result.setInvertSelection(invert);
78      
79      return result;
80    }
81  
82    /**
83     * performs the actual test.
84     * 
85     * @return        the generated data
86     */
87    protected Instances performTest() {
88      Instances icopy = new Instances(m_Instances);
89      Instances result = null;
90      try {
91        m_Filter.setInputFormat(icopy);
92      } 
93      catch (Exception ex) {
94        ex.printStackTrace();
95        fail("Exception thrown on setInputFormat(): \n" + ex.getMessage());
96      }
97      try {
98        result = Filter.useFilter(icopy, m_Filter);
99        assertNotNull(result);
100     } 
101     catch (Exception ex) {
102       ex.printStackTrace();
103       fail("Exception thrown on useFilter(): \n" + ex.getMessage());
104     }
105 
106     assertEquals(icopy.numAttributes(), result.numAttributes());
107     assertEquals(icopy.numInstances(), m_Instances.numInstances());
108     
109     return result;
110   }
111 
112   /**
113    * Returns a test suite.
114    * 
115    * @return        the suite
116    */
117   public static Test suite() {
118     return new TestSuite(RenameAttributeTest.class);
119   }
120 
121   /**
122    * Tests to replace only the first occurrence.
123    */
124   public void testReplaceFirst() {
125     Instances   result;
126 
127     m_Filter = getFilter("t", "_", false, "first-last", false);
128     result   = performTest();
129     
130     assertEquals("S_ringAtt1", result.attribute(0).name());
131   }
132 
133   /**
134    * Tests to replace all the occurrences.
135    */
136   public void testReplaceAll() {
137     Instances   result;
138 
139     m_Filter = getFilter("t", "_", true, "first-last", false);
140     result   = performTest();
141     
142     assertEquals("S_ringA__1", result.attribute(0).name());
143   }
144 
145   /**
146    * Tests the inverting of the attribute range.
147    */
148   public void testInvertRange() {
149     Instances   result;
150 
151     m_Filter = getFilter("t", "_", true, "first", true);
152     result   = performTest();
153     
154     assertTrue("The first attribute contains '_'!", (result.attribute(0).name().indexOf("_") == -1));
155   }
156 
157   /**
158    * Tests the use of capturing groups.
159    */
160   public void testGroup() {
161     Instances   result;
162     int     i;
163 
164     m_Filter = getFilter("(.+)(Att)(.+)", "$1$3", true, "first-last", false);
165     result   = performTest();
166     
167     for (i = 0; i < result.numAttributes(); i++)
168       assertTrue(result.attribute(i).name() + " still contains 'Att'", (result.attribute(i).name().indexOf("Att") == -1));
169   }
170   
171   /**
172    * Runs the test from command-line.
173    * 
174    * @param args    ignored
175    */
176   public static void main(String[] args){
177     TestRunner.run(suite());
178   }
179 }
