31  /**
32   * Tests AddValues. Run from the command line with: <p/>
33   * java weka.filters.unsupervised.attribute.AddValuesTest

38  public class AddValuesTest 
39    extends AbstractFilterTest {
40    
41    /**
42     * initializes the test
43     * 
44     * @param name    the name of the test
45     */
46    public AddValuesTest(String name) { 
47      super(name);

50    /**
51     * Creates a default AddValues
52     * 
53     * @return        the configured filter
54     */
55    public Filter getFilter() {
56      AddValues filter = new AddValues();
57      filter.setAttributeIndex("2");
58      return filter;
59    }

61    /**
62     * Creates a specialized AddValues
63     * 
64     * @param sorted  whether to sort
65     * @param labels  the labels to add
66     * @return        the configured filter
67     */
68    public Filter getFilter(boolean sorted, String labels) {
69      AddValues filter = new AddValues();
70      filter.setAttributeIndex("2");
71      filter.setSort(sorted);
72      filter.setLabels(labels);
73      return filter;
74    }
75    
76    /**
77     * Compare two datasets to see if they differ.
78     *
79     * @param data1 one set of instances
80     * @param data2 the other set of instances
81     * @throws Exception if the datasets differ
82     */
83    protected void compareDatasets(Instances data1, Instances data2)
84      throws Exception {
85      
86      if (data1.numAttributes() != data2.numAttributes())
87        throw new Exception("number of attributes has changed");
88      
89      if (!(data2.numInstances() == data1.numInstances()))
90        throw new Exception("number of instances has changed");
91  
92      for (int i = 0; i < data2.numInstances(); i++) {
93        Instance orig = data1.instance(i);
94        Instance copy = data2.instance(i);
95        for (int j = 0; j < orig.numAttributes(); j++) {
96          if (orig.isMissing(j)) {
97            if (!copy.isMissing(j))
98              throw new Exception("instances have changed");
99          }
100         else if (!orig.toString(j).equals(copy.toString(j))) {
101           throw new Exception("instances have changed");
102         }
103         
104         if (orig.weight() != copy.weight())
105           throw new Exception("instance weights have changed");
106       }
107     }
108   }
109 
110   /**
111    * performs the tests with the given filter
112    * 
113    * @param filter  the configured filter to use in the tests
114    */
115   protected void performTest(Filter filter) {
116     m_Filter = filter;
117     Instances icopy = new Instances(m_Instances);
118     Instances result = null;
119     
120     try {
121       m_Filter.setInputFormat(icopy);
122     } 
123     catch (Exception ex) {
124       ex.printStackTrace();
125       fail("Exception thrown on setInputFormat(): \n" + ex.getMessage());
126     }
127     
128     try {
129       result = Filter.useFilter(m_Instances, m_Filter);
130     } 
131     catch (Exception ex) {
132       ex.printStackTrace();
133       fail("Exception thrown on useFilter(): \n" + ex.getMessage());
134     }
135 
136     try {
137       compareDatasets(icopy, result);
138     } 
139     catch (Exception ex) {
140       ex.printStackTrace();
141       fail("Datasets differ: \n" + ex.getMessage());
142     }
143   }

150   }

164   }

166   /**
167    * tests the sorting and the adding of labels
168    */
169   public void testSortAndLabels() {
170     performTest(getFilter(false, "__blah,__blubber"));
171   }
172   
173   /**
174    * tests the filter in conjunction with the FilteredClassifier
175    */
176   public void testFilteredClassifier() {
177     try {
178       Instances data = getFilteredClassifierData();
179 
180       for (int i = 0; i < data.numAttributes(); i++) {
181     if (data.classIndex() == i)
182       continue;
183     if (data.attribute(i).isNominal()) {
184       ((AddValues) m_FilteredClassifier.getFilter()).setAttributeIndex(
185           "" + (i + 1));
186       break;
187     }
188       }
189     }
190     catch (Exception e) {
191       fail("Problem setting up test for FilteredClassifier: " + e.toString());
192     }
193     
194     super.testFilteredClassifier();
195   }
