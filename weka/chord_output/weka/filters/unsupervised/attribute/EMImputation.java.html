45  /** 
46   <!-- globalinfo-start -->
47   * Replaces missing numeric values using Expectation Maximization with a multivariate normal model. Described in " Schafer, J.L. Analysis of Incomplete Multivariate Data, New York: Chapman and Hall, 1997."
48   * <p/>
49   <!-- globalinfo-end -->
50   *
51   <!-- options-start -->
52   * Valid options are: <p/>
53   * 
54   * <pre> -N
55   *  Maximum number of iterations for Expectation 
56   *  Maximization. (-1 = no maximum)</pre>
57   * 
58   * <pre> -E
59   *  Threshold for convergence in Expectation 
60   *  Maximization. If the change in the observed data 
61   *  log-likelihood (posterior density if a ridge prior 
62   *   is being used) across iterations is no more than 
63   *  this value, then convergence is considered to be 
64   *  achieved and the iterative process is ceased. 
65   *  (default = 0.0001)</pre>
66   * 
67   * <pre> -P
68   *  Use a ridge prior instead of the noninformative 
69   *  prior. This helps when the data has a singular 
70   *  covariance matrix.</pre>
71   * 
72   * <pre> -Q
73   *  The ridge parameter for when a ridge prior is 
74   *  used.</pre>
75   * 
76   <!-- options-end -->
77   * 
78   * @author Amri Napolitano 
79   * @version $Revision: 5987 $
80   */

179   /**
180    * Parses a given list of options. <p/>
181    *
182    <!-- options-start -->
183    * Valid options are: <p/>
184    * 
185    * <pre> -N
186    *  Maximum number of iterations for Expectation 
187    *  Maximization. (-1 = no maximum)</pre>
188    * 
189    * <pre> -E
190    *  Threshold for convergence in Expectation 
191    *  Maximization. If the change in the observed data 
192    *  log-likelihood (posterior density if a ridge prior 
193    *   is being used) across iterations is no more than 
194    *  this value, then convergence is considered to be 
195    *  achieved and the iterative process is ceased. 
196    *  (default = 0.0001)</pre>
197    * 
198    * <pre> -P
199    *  Use a ridge prior instead of the noninformative 
200    *  prior. This helps when the data has a singular 
201    *  covariance matrix.</pre>
202    * 
203    * <pre> -Q
204    *  The ridge parameter for when a ridge prior is 
205    *  used.</pre>
206    * 
207    <!-- options-end -->
208    *
209    * @param options the list of options as an array of strings
210    * @throws Exception if an option is not supported
211    */
212   public void setOptions (String[] options)
213   throws Exception {
214     String optionString;
215     
216     // set # EM iterations
217     optionString = Utils.getOption('N', options);
218     if (optionString.length() != 0) {
219       setNumIterations(Integer.parseInt(optionString));
220     }
221     // set log-likelihood EM convergence threshold
222     optionString = Utils.getOption('E', options);
223     if (optionString.length() != 0) {
224       setLogLikelihoodThreshold(Double.valueOf(optionString).doubleValue());
225     }
226     // set whether to use ridge prior
227     setUseRidgePrior(Utils.getFlag('P', options));
228     // set ridge parameter
229     optionString = Utils.getOption('Q', options);
230     if (optionString.length() != 0) {
231       setRidge(Double.valueOf(optionString));
232     }
233   }

236    * Gets the current settings of EMImputation
237    *
238    * @return an array of strings suitable for passing to setOptions()
239    */
240   public String[] getOptions () {
241     
242     String[] options = new String[7];
243     int current = 0;
244     
245     options[current++] = "-N";
246     options[current++] = "" + getNumIterations();
247     
248     options[current++] = "-E";
249     options[current++] = "" + getLogLikelihoodThreshold();
250     
251     options[current++] = "-Q";
252     options[current++] = "" + getRidge();
253     
254     if(getUseRidgePrior()) {
255       options[current++] = "-P";
256     }
257     
258     while(current < options.length) {
259       options[current++] = "";
260     }
261     
262     return  options;
263   }

408   /**
409    * Processes the given data (may change the provided dataset) and returns
410    * the modified version. This method is called in batchFinished().
411    *
412    * @param instances   the data to process
413    * @return            the modified data
414    * @throws Exception  in case the arguments are inappropriate or the processing goes wrong
415    * @see               #batchFinished()
416    */
417   protected Instances process(Instances instances) throws Exception {
418     
419     int numInstances = instances.numInstances();
420     int numAttributes = instances.numAttributes();
421     int classIndex = instances.classIndex();
422     // if only one numeric attribute, don't do procedure
423     if (((classIndex < 0 || instances.classAttribute().isNumeric()) && 
424         numAttributes < 2) || numAttributes < 3) {
425       throw new Exception("Must have 2 or more numeric attributes for EM Imputation");     
426     }
427     
428     // check for only numeric independent attributes
429     for (int i = 0; i < numAttributes; i++) {
430       if (instances.attribute(i).isNumeric() == false && instances.classIndex() != i) {
431         throw new Exception("EM Imputation can only handle numeric attributes");
432       }
433     }
434     
435     // check inputs for ok values
436     if (m_LogLikelihoodThreshold < 0) { // if loglikelihood threshold is negative, convergence would never occur
437       throw new Exception("Log-likelihood threshold must be non-negative.");
438     }
439     if (m_ridgePrior == true && m_ridge <= 0) {
440       throw new Exception("Ridge parameter should be positive.");
441     }
442     
443     // create datasets ready for processing
444     Instances preprocessed, dataWithClass;
445     Instances [] datasets = null;
446     if (isFirstBatchDone() == false) { // model hasn't been built yet
447       // check that n > p + 1
448       if (numInstances < numAttributes + 2) {
449         throw new Exception("EMImputation: Number of instances must be >= number of attributes + 2");
450       }
451       datasets = prepareData(instances, true);
452     } else { // model has been built
453       datasets = prepareData(instances, false);
454     }
455     preprocessed = datasets[0]; // holds data sorted by missing data pattern with attribute about pattern
456     dataWithClass = datasets[1]; // holds data in same sorted order without the class removed
457     
458     // build model if it hasn't been built yet
459     if (isFirstBatchDone() == false) {
460       // get matrix of complete-data sufficient statistics, T, for observed values
461       Matrix t_obs = getTObs(preprocessed);
462       
463       // perform expectation maximization to determine model parameters
464       m_theta = EM(preprocessed, t_obs);
465     }
466 
467     // fill in missing values from model
468     impute(preprocessed, m_theta);
469     
470     // produce final output dataset
471     dataWithClass = postProcessData(preprocessed, dataWithClass);
472     
473     if (dataWithClass.numInstances() != 0) {
474       return dataWithClass;
475     } else {
476       return instances;
477     }
478   }

480   /**
481    * Standardizes a (preprocessed) dataset using m_means and m_stdDevs.
482    * @param data        dataset to standardize
483    */
484   private void standardize(Instances data) {
485     
486     for (int i = 0; i < data.numInstances(); i++) {
487       for (int j = 1; j < m_numAttributes + 1; j++) {
488         if (data.instance(i).isMissing(j) == false) {
489           double value = data.instance(i).value(j);
490             data.instance(i).setValue(j, (value - m_means[j - 1]) / m_stdDevs[j - 1]);
491         }
492       }
493     }
494   }

496   /**
497    * Preprocesses data to be used for EM
498    * @param data            the data to be processed
499    * @return preprocessed   the prepared data
500    * @throws Exception      if processing goes wrong
501    */
502   private Instances prepareMissing(Instances data) throws Exception {
503     
504     // properties of data
505     int numInstances = data.numInstances();
506     int numAttributes = data.numAttributes();
507     
508     Instances preprocessed = new Instances(data, data.numInstances());
509     preprocessed.insertAttributeAt(new Attribute("NumInPattern"), 0);
510     // add instances to preprocessed sorted by missingness pattern
511     int currentPatternStart = 0; // index of 1st instance in current pattern
512     int numInCurrentPattern = 0; // number of instances with current pattern
513     int totalAdded = 0; // total instances added to new dataset
514     boolean [] addedAlready = new boolean[numInstances];
515     boolean [] missing = new boolean[numAttributes]; // missing columns
516     
517     while (totalAdded < numInstances) { // until all instances have been added
518       // set up pattern to find
519       numInCurrentPattern = 0;
520       int numMissing = 0;
521       int newPatternIndex = -1; // index of instance with new pattern
522       for (int i = 0; newPatternIndex == -1 && i < numInstances; i++) {
523         if (!addedAlready[i]) {
524           newPatternIndex = i;
525           addedAlready[i] = true;
526           numInCurrentPattern++;
527           totalAdded++;
528         }
529       }
530       // store new missing pattern
531       for (int j = 0; j < numAttributes; j++) {
532         if (m_unusedAtts[j] == true) {
533           numMissing++;
534           continue;
535         }
536         if (data.instance(newPatternIndex).isMissing(j)) {
537           missing[j] = true;
538           numMissing++;
539         } else {
540           missing[j] = false;
541         }
542       }
543       if (numMissing == numAttributes) { // all attributes are missing - don't use
544         continue;
545       }
546       
547       // create modified instance and add it to the new dataset
548       double [] attributes = new double[numAttributes + 1];
549       for (int j = 0; j < numAttributes; j++) {
550         attributes[j + 1] = data.instance(newPatternIndex).value(j);
551       }
552       Instance newInstance = new DenseInstance(data.instance(newPatternIndex).weight(), attributes);
553       preprocessed.add(newInstance);
554         
555       // go through all remaining instances to find those with same pattern
556       for (int i = 0; i < numInstances; i++) {
557         if (addedAlready[i]) {
558           continue;
559         }
560         // go through all the attributes to see if they match the pattern
561         boolean match = true; // if current instance matches the pattern
562         for (int j = 0; j < numAttributes && match == true; j++) {
563           if (m_unusedAtts[j] == true) {
564             continue;
565           }
566           if (data.instance(i).isMissing(j) != missing[j]) {
567             match = false; // not a matching pattern
568           }
569         }
570         if (match == true) { // if pattern matches
571           // create modified instance and add it to the new dataset
572           attributes = new double[numAttributes + 1];
573           for (int l = 0; l < numAttributes; l++) {
574             attributes[l + 1] = data.instance(i).value(l);
575           }
576           newInstance = new DenseInstance(data.instance(i).weight(), attributes);
577           preprocessed.add(newInstance);
578           
579           // mark current instance added
580           addedAlready[i] = true;
581           
582           // increment number of instances in current pattern and total
583           numInCurrentPattern++;
584           totalAdded++;
585         }
586       } // end iterating through instances remaining in original data
587       
588       // set number of iterations for fully found pattern
589       preprocessed.instance(currentPatternStart).setValue(0, numInCurrentPattern);
590       currentPatternStart += numInCurrentPattern;
591     } // end adding instances to new dataset
592     
593     return preprocessed;
594   }

596   /**
597    * Sets up data for performing imputation. If firstTime is true, it sets some 
598    * initial global parameters (e.g. means/std devs for standardization).
599    * @param originalData    dataset preprocess
600    * @param firstTime       flag for whether or not to set data parameters
601    * @return data           array of preprocessed datasets necessary for other imputation functions
602    * @throws Exception      if processing goes wrong
603    */
604   private Instances [] prepareData(Instances originalData, boolean firstTime) throws Exception {
605     
606     int numAttributes = originalData.numAttributes();
607     int classIndex = originalData.classIndex();
608     
609     // determine attributes that won't be used in the imputation (all missing or all one value or nonnumeric class attribute)
610     if (firstTime == true) {
611       String unusedAttributeIndices = "";
612       m_unusedAtts = new boolean[numAttributes];
613       for (int j = 0; j < numAttributes; j++) {
614         AttributeStats currentStats = originalData.attributeStats(j);
615         if ((classIndex == j && originalData.classAttribute().isNumeric() == false) ||
616             currentStats.distinctCount < 2) {
617           m_unusedAtts[j] = true;
618           unusedAttributeIndices += (j + 2) + ",";
619         } else {
620           m_unusedAtts[j] = false;
621         }
622       }
623       if (unusedAttributeIndices.contentEquals("") == false) {
624         // set up remove filter
625         m_unusedAttributeRemover = new Remove();
626         m_unusedAttributeRemover.setInvertSelection(false);
627         m_unusedAttributeRemover.setAttributeIndices(unusedAttributeIndices);
628       }
629     }
630     
631     // preprocess for working with the missing values
632     Instances preprocessed = prepareMissing(originalData);
633     
634     // remove attributes that won't be used
635     Instances withClass = new Instances(preprocessed, 0, preprocessed.numInstances()); // to store class values
636     if (m_unusedAttributeRemover != null) {
637       m_unusedAttributeRemover.setInputFormat(preprocessed);
638       preprocessed = Filter.useFilter(preprocessed, m_unusedAttributeRemover);
639       preprocessed.setClassIndex(-1);
640     }
641     
642     // if preparing the first time data, set parameters
643     if (firstTime == true) {
644       m_numAttributes = preprocessed.numAttributes() - 1;
645       m_means = new double[m_numAttributes];
646       m_stdDevs = new double[m_numAttributes];
647       for (int i = 1; i < m_numAttributes + 1; i++) {
648         Stats columnStats = preprocessed.attributeStats(i).numericStats;
649         columnStats.calculateDerived();
650         m_means[i - 1] = columnStats.mean;
651         m_stdDevs[i - 1] = columnStats.stdDev;
652       }
653     }
654     
655     // standardize data (to avoid rounding errors)
656     standardize(preprocessed);
657     
658     // return array of datasets
659     Instances [] data = new Instances[2];
660     data[0] = preprocessed;
661     data[1] = withClass;
662     return data;
663   }

665   /**
666    * Postprocesses data to make ready for output
667    * @param imputedData   dataset with final (standardized) imputed values
668    * @param withClass     dataset without class attribute removed
669    * @return withClass    the final imputed, postprocessed data
670    * @throws Exception    if processing goes wrong
671    */
672   private Instances postProcessData(Instances imputedData, Instances withClass) throws Exception {
673     
674     // remove missingPattern attribute from withClass
675     Remove patternRemover = new Remove();
676     patternRemover.setInvertSelection(false);
677     patternRemover.setAttributeIndices("1");
678     patternRemover.setInputFormat(withClass);
679     withClass = Filter.useFilter(withClass, patternRemover);
680     
681     // destandardize imputed values and place in proper spots in data with class
682     for (int i = 0; i < withClass.numInstances(); i++) {
683       int usedAttributes = 0;
684       for (int j = 0; j < withClass.numAttributes(); j++) {
685         if (m_unusedAtts[j] == true) {
686           continue;
687         }
688         if (withClass.instance(i).isMissing(j)) {
689           withClass.instance(i).setValue(j, imputedData.instance(i).value(usedAttributes + 1) 
690                                             * m_stdDevs[usedAttributes] + m_means[usedAttributes]);
691         }
692         usedAttributes++;
693       }
694     }
695     return withClass;
696   }

698   /**
699    * Calculates T_obs (complete data sufficient statistics for the observed values). 
700    * @param data        preprocessed dataset with missing values
701    * @return t          T_obs
702    * @throws Exception  if processing goes wrong
703    */
704   private Matrix getTObs(Instances data) throws Exception {
705     
706     int p = m_numAttributes; // number of columns
707     
708     // matrix of complete-data sufficient statistics for observed values
709     Matrix t = new Matrix(p+1, p+1);
710     
711     // initialize T for observed data (T_obs)
712     int currentPatternStart = 0;
713     while (currentPatternStart < data.numInstances()) {
714       // number of instances in current missingness pattern is held in 1st attribute
715       int numInCurrentPattern = (int) data.instance(currentPatternStart).value(0);
716       t.set(0, 0, t.get(0,0) + numInCurrentPattern);
717       for (int i = 1; i < p + 1; i++) {
718         // if current column is not missing in this pattern
719         if (!data.instance(currentPatternStart).isMissing(i)) {
720           // add values in this column for this pattern
721           for (int k = 0; k < numInCurrentPattern; k++) {
722             t.set(0, i, t.get(0, i) + data.instance(currentPatternStart + k).value(i));
723           }
724           // iterate through columns to add appropriate squares and crossproducts
725           for (int j = i; j < p + 1; j++) {
726             // if this column is also not missing in the current missingness pattern
727             if (!data.instance(currentPatternStart).isMissing(j)) {
728               for (int k = 0; k < numInCurrentPattern; k++) {
729                 t.set(i, j, t.get(i, j) + 
730                       data.instance(currentPatternStart + k).value(i) * 
731                       data.instance(currentPatternStart + k).value(j));
732               } // end iterating through instances in currrent missingness pattern
733             }
734           } // end iterating through column indices (inner loop) 
735         }
736       } // end iterating through column indices (outer loop)
737       
738       // move counter to start of next missingness pattern (or end of dataset)
739       currentPatternStart += numInCurrentPattern;
740     } // end iterating through missingness patterns
741     
742     // copy to symmetric lower triangular portion
743     for (int i = 0; i < p + 1; i++) {
744       for (int j = 1; j < p + 1; j++) {
745         t.set(j, i, t.get(i, j));
746       }
747     }
748     
749     return t;
750   }

752   /**
753    * Performs the expectation maximization (EM) algorithm to find the maximum 
754    * likelihood estimate (or posterior mode if ridge prior is being used)
755    * for the multivariate normal parameters of a dataset with missing values. 
756    * @param data          preprocessed dataset with missing values
757    * @param t_obs         the complete data sufficient statistics for the observed values
758    * @return theta        the maximum likelihood estimate for the parameters of the multivariate normal distribution
759    * @throws Exception    if processing goes wrong
760    */
761   private Matrix EM(Instances data, Matrix t_obs) throws Exception {
762     
763     int p = m_numAttributes; // number of columns
764     Matrix theta = new Matrix(p+1, p+1); // parameter matrix
765     
766     // if numIterations is -1, change to largest int
767     int numIterations = m_numIterations;
768     if (numIterations < 0) {
769       numIterations = Integer.MAX_VALUE;
770     }
771     
772     // starting theta value (means and variances of each column, correlations left at zero)
773     // values are standardized so means are 0 and variances are 1
774     theta.set(0, 0, -1);
775     for (int i = 1; i < data.numAttributes(); i++) {
776       theta.set(0, i, 0); // mu_i
777       theta.set(i, 0, 0);
778       theta.set(i, i, 1); // sigma_ii
779     }
780     
781     double likelihood = logLikelihood(data, theta);
782     double deltaLikelihood = Double.MAX_VALUE;
783     for (int i = 0; i < numIterations && deltaLikelihood > m_LogLikelihoodThreshold; i++) {
784       theta = doEMIteration(data, theta, t_obs);
785       double newLikelihood = logLikelihood(data, theta);
786       deltaLikelihood = newLikelihood - likelihood;
787       likelihood = newLikelihood;
788     }
789     
790     return theta;
791   }

793   /**
794    * Performs one iteration of expectation maximization (EM).
795    * @param data          preprocessed dataset with missing values
796    * @param theta         a matrix containing the starting estimate of the multivariate normal parameters
797    * @param t_obs         the complete data sufficient statistics for the observed values
798    * @return theta        the maximum likelihood estimate (or posterior mode for ridge prior) for the parameters of the multivariate normal distribution at the end of the iteration
799    * @throws Exception    if arguments are inappropriate or processing goes wrong
800    */
801   private Matrix doEMIteration(Instances data, Matrix theta, Matrix t_obs) throws Exception {
802     
803     int p = m_numAttributes; // number of columns
804     Matrix t = t_obs.copy();
805     
806     // go through each pattern
807     int currentPatternStart = 0;
808     while (currentPatternStart < data.numInstances()) {
809       // number of instances in current missingness pattern is held in 1st attribute
810       int numInCurrentPattern = (int) data.instance(currentPatternStart).value(0);
811       // set up binary flags for if column is observed in this pattern
812       boolean [] r = new boolean[p + 1];
813       int [] observedColumns = new int[p + 1]; // indices of observed columns
814       int [] missingColumns = new int[p + 1]; // indices of missing columns
815       int numMissing = 0; // number of missing columns
816       int numObserved = 0; // number of observed columns
817       for (int l = 1; l < p + 1; l++) {
818         if (data.instance(currentPatternStart).isMissing(l)) {
819           missingColumns[numMissing++] = l;
820         } else {
821           observedColumns[numObserved++] = l;
822           r[l] = true;
823         }
824       }
825       observedColumns[numObserved] = -1; // list end indicator
826       missingColumns[numMissing] = -1; // list end indicator
827       
828       for (int j = 1; j < p + 1; j++) {
829         if (r[j] == true && theta.get(j, j) > 0) {
830           theta = swp(theta, j);
831         } else if (r[j] == false && theta.get(j, j) < 0) {
832           theta = rsw(theta, j);
833         }
834       }
835       
836       double [] c = new double[p + 1]; // temporary workspace to hold y_ij^*
837       for (int i = 0; i < numInCurrentPattern; i++) {
838         
839         // calculate y_ij^* values
840         for (int jCounter = 0, j = missingColumns[jCounter]; j != -1; 
841              jCounter++, j = missingColumns[jCounter]) {
842           c[j] = theta.get(0, j);
843           
844           for (int kCounter = 0, k = observedColumns[kCounter]; k != -1;
845                kCounter++, k = observedColumns[kCounter]) {
846             c[j] += theta.get(k, j) * data.instance(currentPatternStart + i).value(k);
847           }
848         }
849         
850         // update t
851         for (int jCounter = 0, j = missingColumns[jCounter]; j != -1; 
852              jCounter++, j = missingColumns[jCounter]) {
853           t.set(0, j, t.get(0, j) + c[j]);
854           t.set(j, 0, t.get(0, j));
855           
856           for (int kCounter = 0, k = observedColumns[kCounter]; k != -1;
857                kCounter++, k = observedColumns[kCounter]) {
858             t.set(k, j, t.get(k, j) + c[j] * data.instance(currentPatternStart + i).value(k));
859             t.set(j, k, t.get(k, j));
860           }
861           
862           for (int kCounter = 0, k = missingColumns[kCounter]; k != -1;
863                kCounter++, k = missingColumns[kCounter]) {
864             if (k >= j) {
865               t.set(k, j, t.get(k, j) + theta.get(k, j) + c[k] * c[j]);
866               t.set(j, k, t.get(k, j));
867             }
868           }
869         }
870       }
871       
872       // move counter to start of next missingness pattern (or end of dataset)
873       currentPatternStart += numInCurrentPattern;
874     } // end iterating through missingness patterns
875     
876     // modify complete data sufficient statistics if using ridge prior
877     if (m_ridgePrior) {
878       double n = data.numInstances();
879       double m = m_ridge;
880       Matrix deltaInv = Matrix.identity(m_numAttributes, m_numAttributes).times(m_ridge); // delta^(-1)
881       // sufficient statistics t1 and t2
882       Matrix t1 = t.getMatrix(1, (int)p, 0, 0);
883       Matrix t2 = t.getMatrix(1, (int)p, 1, (int)p);
884       // modified sufficient statistics t1~ and t2~
885       Matrix t1Tilde, t2Tilde;
886       
887       t1Tilde = t1;
888       
889       t2Tilde = t2.minus(t1.times(t1.transpose()).times(1.0 / n));
890       t2Tilde = t2Tilde.plus(deltaInv).times(n / (n + m + p + 2));
891       t2Tilde.plusEquals(t1Tilde.times(t1Tilde.transpose()).times(1.0 / n));
892       
893       // replace t1 and t2 in t with t1~ and t2~
894       t.setMatrix(1, (int)p, 0, 0, t1Tilde); // replacing t1 in first row
895       t.setMatrix(0, 0, 1, (int)p, t1Tilde.transpose()); // replacing t1' in first column
896       t.setMatrix(1, (int)p, 1, (int)p, t2Tilde); // replacing t2
897     }
898     
899     return swp(t.times(1.0 / data.numInstances()), 0);
900   }

902   /**
903    * Calculates the observed data log-likelihood for theta (or observed data log posterior 
904    * density if ridge prior is being used).
905    * @param data          numeric dataset with instances ordered by missingness pattern and an extra attribute added for keeping track of the patterns
906    * @param theta       a matrix containing the multivariate normal parameters
907    * @return likelihood the observed data log-likelihood for theta (or log posterior density if using a ridge prior)
908    * @throws Exception  if processing goes wrong
909    */
910   private double logLikelihood(Instances data, Matrix theta) throws Exception {
911     
912     int p = m_numAttributes; // number of columns
913     double likelihood = 0.0; // return value
914     double d = 0;
915     
916     double [] c = new double[p + 1]; // temporary storage for mu
917     for (int j = 1; j < p + 1; j++) {
918       c[j] = theta.get(0, j);
919     }
920     
921     // matrix for calculating the log posterior density if necessary
922     Matrix sigma = theta.getMatrix(1, p, 1, p); // covariance matrix from theta
923     
924     // go through each pattern
925     int currentPatternStart = 0;
926     while (currentPatternStart < data.numInstances()) {
927       // number of instances in current missingness pattern is held in 1st attribute
928       int numInCurrentPattern = (int) data.instance(currentPatternStart).value(0);
929       // set up binary flags for if column is observed in this pattern
930       boolean [] r = new boolean[p + 1];
931       int [] observedColumns = new int[p + 1]; // indices of observed columns
932       int [] missingColumns = new int[p + 1]; // indices of missing columns
933       int numMissing = 0; // number of missing columns
934       int numObserved = 0; // number of observed columns
935       for (int l = 1; l < p + 1; l++) {
936         if (data.instance(currentPatternStart).isMissing(l)) {
937           missingColumns[numMissing++] = l;
938         } else {
939           observedColumns[numObserved++] = l;
940           r[l] = true;
941         }
942       }
943       observedColumns[numObserved] = -1; // list end indicator
944       missingColumns[numMissing] = -1; // list end indicator
945       
946       for (int j = 1; j < p + 1; j++) {
947         if (r[j] == true && theta.get(j, j) > 0) {
948           d += Math.log(theta.get(j, j));
949           theta = swp(theta, j);
950         } else if (r[j] == false && theta.get(j, j) < 0) {
951           theta = rsw(theta, j);
952           d -= Math.log(theta.get(j, j));
953         }
954       }
955       
956       Matrix m = new Matrix(p + 1, p + 1);
957       for (int i = 0; i < numInCurrentPattern; i++) {
958         for (int jCounter = 0, j = observedColumns[jCounter]; j != -1; 
959              jCounter++, j = observedColumns[jCounter]) {
960           
961           for (int kCounter = jCounter, k = observedColumns[kCounter]; k != -1;
962                kCounter++, k = observedColumns[kCounter]) {
963             m.set(j, k, m.get(j, k) + (data.instance(currentPatternStart + i).value(j) -
964                   c[j]) * (data.instance(currentPatternStart + i).value(k) - c[k]));
965             m.set(k, j, m.get(j, k));
966           }
967         }
968       }
969       
970       double t = 0;
971       
972       for (int jCounter = 0, j = observedColumns[jCounter]; j != -1; 
973            jCounter++, j = observedColumns[jCounter]) {
974         
975         for (int kCounter = 0, k = observedColumns[kCounter]; k != -1;
976              kCounter++, k = observedColumns[kCounter]) {
977             t -= theta.get(j, k) * m.get(j, k);
978         }
979       }
980       
981       //update likelihood
982       likelihood -= ((double)numInCurrentPattern * d + t) / 2.0;
983       
984       // move counter to start of next missingness pattern (or end of dataset)
985       currentPatternStart += numInCurrentPattern;
986     } // end iterating through missingness patterns
987     
988     // modify to produce log posterior density if ridge prior is being used
989     if (m_ridgePrior) {
990       Matrix deltaInv = Matrix.identity(p, p).times(m_ridge); // delta^(-1)
991       double m = m_ridge;
992       double logPi = 0; // log pi(theta) term to be added to MLE to get log posterior density
993       
994       Matrix M0 = deltaInv;
995       
996       // compute log pi(theta)
997       logPi = Math.log(Math.abs(sigma.det()));
998       logPi *= -0.5 * (m + p + 2);
999       logPi -= 0.5 * sigma.inverse().times(M0).trace();
1000      
1001      // modify result to give log posterior density
1002      likelihood += logPi;
1003    }
1004    
1005    return likelihood;
1006  }

1008  /**
1009   * Performs the imputation. Draws Y_mis(i+1) as the expected value of (Y_mis | Y_obs, theta(i)).
1010   * @param data          preprocessed dataset with missing values
1011   * @param theta         a matrix containing the multivariate normal parameters
1012   * @throws Exception    if processing goes wrong
1013   */
1014  private void impute(Instances data, Matrix theta) throws Exception {
1015    
1016    int p = m_numAttributes; // number of columns
1017    
1018    // go through each pattern
1019    int currentPatternStart = 0;
1020    while (currentPatternStart < data.numInstances()) {
1021      // number of instances in current missingness pattern is held in 1st attribute
1022      int numInCurrentPattern = (int) data.instance(currentPatternStart).value(0);
1023      // set up binary flags for if column is observed in this pattern
1024      boolean [] r = new boolean[p + 1];
1025      int [] observedColumns = new int[p + 1]; // indices of observed columns
1026      int [] missingColumns = new int[p + 1]; // indices of missing columns
1027      int numMissing = 0; // number of missing columns
1028      int numObserved = 0; // number of observed columns
1029      for (int l = 1; l < p + 1; l++) {
1030        if (data.instance(currentPatternStart).isMissing(l)) {
1031          missingColumns[numMissing++] = l;
1032        } else {
1033          observedColumns[numObserved++] = l;
1034          r[l] = true;
1035        }
1036      }
1037      observedColumns[numObserved] = -1; // list end indicator
1038      missingColumns[numMissing] = -1; // list end indicator
1039      
1040      for (int j = 1; j < p + 1; j++) {
1041        if (r[j] == true && theta.get(j, j) > 0) {
1042          theta = swp(theta, j);
1043        } else if (r[j] == false && theta.get(j, j) < 0) {
1044          theta = rsw(theta, j);
1045        }
1046      }
1047      
1048      for (int i = 0; i < numInCurrentPattern; i++) {
1049        for (int jCounter = 0, j = missingColumns[jCounter]; j != -1; 
1050             jCounter++, j = missingColumns[jCounter]) {
1051          // impute the missing values
1052          data.instance(currentPatternStart + i).setValue(j, theta.get(0, j));
1053          for (int kCounter = 0, k = observedColumns[kCounter]; k != -1;
1054               kCounter++, k = observedColumns[kCounter]) {
1055            double y_ij = data.instance(currentPatternStart + i).value(j);
1056            double y_ik = data.instance(currentPatternStart + i).value(k);
1057            double theta_kj = theta.get(k, j);
1058            data.instance(currentPatternStart + i).setValue(j, y_ij + theta_kj * y_ik);
1059          }
1060        }
1061      } // end iterating through instances in pattern
1062      // move counter to start of next missingness pattern (or end of dataset)
1063      currentPatternStart += numInCurrentPattern;
1064    } // end iterating through missingness patterns
1065  }

1067  /**
1068   * Performs the sweep operation on a matrix at the given position. 
1069   * @param g           a matrix
1070   * @param k           the pivot position
1071   * @param dir         the direction to do the sweep in (1 = normal sweep, -1 = reverse sweep)
1072   * @return h          the matrix after being swept on position k
1073   * @throws Exception  if processing goes wrong
1074   */
1075  private static Matrix doSweep(Matrix g, int k, int dir) throws Exception {
1076    
1077    // number of rows/columns
1078    int p = g.getRowDimension();
1079    
1080    // check if k is in range
1081    if (k < 0 || k >= p) {
1082      throw new Exception("Position to be swept on must be within range.");
1083    }
1084    
1085    // check if dir is 1 or -1
1086    if (dir != 1 && dir != -1) {
1087      throw new Exception("Sweep direction must be 1 or -1.");
1088    }
1089    
1090    // result matrix
1091    Matrix h = g.copy();
1092    
1093    // check that pivot value is not zero
1094    double kkValue = g.get(k, k);
1095    if (kkValue == 0) {
1096      throw new Exception("Sweep: Division by zero (pivot value).");
1097    }
1098    
1099    // process elements
1100    for (int i = 0; i < p; i++) {
1101      for (int j = i; j < p; j++) {
1102        if (i == k && j == k) { // pivot position
1103          h.set(i, j, -1.0 / kkValue);
1104        } else if (i == k || j == k) { // value in row or column k
1105          h.set(i, j, (double)dir * g.get(i, j) / kkValue);
1106          h.set(j, i, h.get(i, j)); // copy to symmetric value
1107        } else {
1108          h.set(i, j, g.get(i, j) - g.get(i, k) * g.get(k, j) / kkValue);
1109          h.set(j, i, h.get(i, j)); // copy to symmetric value
1110        }
1111      }
1112    }
1113    
1114    return h;
1115  }

1119   * @param g           a matrix
1120   * @param k           the pivot position
1121   * @return h          the matrix after being swept on position k

1123   */
1124  private static Matrix swp(Matrix g, int k) throws Exception {
1125
