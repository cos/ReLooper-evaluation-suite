21  package weka.filters.unsupervised.instance;
22  
23  import weka.core.Instances;
24  import weka.filters.AbstractFilterTest;
25  import weka.filters.Filter;
26  
27  import junit.framework.Test;
28  import junit.framework.TestSuite;
29  
30  /**
31   * Tests Resample. Run from the command line with:<p>
32   * java weka.filters.unsupervised.instance.ResampleTest
33   *
34   * @author <a href="mailto:len@reeltwo.com">Len Trigg</a>
35   * @version $Revision: 1.3 $
36   */
37  public class ResampleTest extends AbstractFilterTest {
38    
39    public ResampleTest(String name) { super(name);  }
40  
41    /** Creates a default Resample */
42    public Filter getFilter() {
43      Resample f = new Resample();
44      f.setSampleSizePercent(50);
45      return f;
46    }
47  
48    public void testSampleSizePercent() {
49      Instances result = useFilter();
50      assertEquals(m_Instances.numAttributes(), result.numAttributes());
51      assertEquals("Expecting output to be 50% of input",
52                   m_Instances.numInstances() / 2,  result.numInstances());
53  
54      ((Resample)m_Filter).setSampleSizePercent(200);
55      result = useFilter();
56      assertEquals(m_Instances.numAttributes(), result.numAttributes());
57      assertEquals("Expecting output to be 200% of input",
58                   m_Instances.numInstances() * 2,  result.numInstances());
59    }
60  
61    public void testSampleSizePercentNoReplacement() {
62      ((Resample) m_Filter).setSampleSizePercent(20);
63      ((Resample) m_Filter).setNoReplacement(true);
64      Instances result = useFilter();
65      assertEquals(m_Instances.numAttributes(), result.numAttributes());
66      assertEquals("Expecting output to be 20% of input",
67                   (int) (m_Instances.numInstances() * 20.0 / 100),  result.numInstances());
68    }
69  
70    public void testSampleSizePercentNoReplacementInverted() {
71      ((Resample) m_Filter).setSampleSizePercent(20);
72      ((Resample) m_Filter).setNoReplacement(true);
73      ((Resample) m_Filter).setInvertSelection(true);
74      Instances result = useFilter();
75      assertEquals(m_Instances.numAttributes(), result.numAttributes());
76      assertEquals("Expecting output to be 80% of input (20% inverted)",
77                   m_Instances.numInstances() 
78                   - (int) (m_Instances.numInstances() * 20.0 / 100),  result.numInstances());
79    }
80  
81    public static Test suite() {
82      return new TestSuite(ResampleTest.class);
83    }
84  
85    public static void main(String[] args){
86      junit.textui.TestRunner.run(suite());
87    }
88  
89  }
