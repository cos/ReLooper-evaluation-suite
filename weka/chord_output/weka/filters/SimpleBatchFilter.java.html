23  package weka.filters;
24  
25  import weka.core.Instance;
26  import weka.core.Instances;
27  
28  /** 
29   * This filter is a superclass for simple batch filters. 
30   * <p/>
31   * 
32   * <b>General notes:</b><br/>
33   * <ul>
34   *   <li>After adding instances to the filter via input(Instance) one always
35   *   has to call batchFinished() to make them available via output(). </li>
36   *   <li>After the first call of batchFinished() the field m_FirstBatchDone is
37   *   set to <code>true</code>. </li>
38   * </ul>
39   * <p/>
40   * 
41   * <b>Example:</b><br/>
42   * The following code snippet uses the filter <code>SomeFilter</code> on a
43   * dataset that is loaded from <code>filename</code>.
44   * <pre>
45   * import weka.core.*;
46   * import weka.filters.*;
47   * import java.io.*;
48   * ...
49   * SomeFilter filter = new SomeFilter();
50   * // set necessary options for the filter
51   * Instances data = new Instances(
52   *                    new BufferedReader(
53   *                      new FileReader(filename)));
54   * Instances filteredData = Filter.useFilter(data, filter);
55   * </pre>
56   *
57   * <b>Implementation:</b><br/>
58   * Only the following abstract methods need to be implemented:
59   * <ul>
60   *   <li>globalInfo()</li>
61   *   <li>determineOutputFormat(Instances)</li>
62   *   <li>process(Instances)</li>
63   * </ul>
64   * <br/>
65   * And the <b>getCapabilities()</b> method must return what kind of
66   * attributes and classes the filter can handle.
67   * <p/>
68   * 
69   * If more options are necessary, then the following methods need to be
70   * overriden:
71   * <ul>
72   *   <li>listOptions()</li>
73   *   <li>setOptions(String[])</li>
74   *   <li>getOptions()</li>
75   * </ul>
76   * <p/>
77   *
78   * To make the filter available from commandline one must add the following
79   * main method for correct execution (&lt;Filtername&gt; must be replaced
80   * with the actual filter classname):
81   * <pre>
82   *  public static void main(String[] args) {
83   *    runFilter(new &lt;Filtername&gt;(), args);
84   *  }
85   * </pre>
86   * <p/>
87   *
88   * <b>Example implementation:</b><br/>
89   * <pre>
90   * import weka.core.*;
91   * import weka.core.Capabilities.*;
92   * import weka.filters.*;
93   *
94   * public class SimpleBatch
95   *   extends SimpleBatchFilter {
96   *   
97   *   public String globalInfo() {
98   *     return "A simple batch filter that adds an additional attribute 'bla' at the end containing the index of the processed instance.";
99   *   }
100  *     
101  *   public Capabilities getCapabilities() {
102  *     Capabilities result = super.getCapabilities();
103  *     result.enableAllAttributes();
104  *     result.enableAllClasses();
105  *     result.enable(Capability.NO_CLASS);  // filter doesn't need class to be set
106  *     return result;
107  *   }
108  * 
109  *   protected Instances determineOutputFormat(Instances inputFormat) {
110  *     Instances result = new Instances(inputFormat, 0);
111  *     result.insertAttributeAt(new Attribute("bla"), result.numAttributes());
112  *     return result;
113  *   }
114  * 
115  *   protected Instances process(Instances inst) {
116  *     Instances result = new Instances(determineOutputFormat(inst), 0);
117  *     for (int i = 0; i &lt; inst.numInstances(); i++) {
118  *       double[] values = new double[result.numAttributes()];
119  *       for (int n = 0; n &lt; inst.numAttributes(); n++)
120  *         values[n] = inst.instance(i).value(n);
121  *       values[values.length - 1] = i;
122  *       result.add(new Instance(1, values));
123  *     }
124  *     return result;
125  *   }
126  * 
127  *   public static void main(String[] args) {
128  *     runFilter(new SimpleBatch(), args);
129  *   }
130  * }
131  * </pre>
132  * <p/>
133  * 
134  * <b>Options:</b><br/>
135  * Valid filter-specific options are:<p/>
136  *
137  * -D <br/>
138  * Turns on output of debugging information.<p/>
139  *
140  * @author  FracPete (fracpete at waikato dot ac dot nz)
141  * @version $Revision: 5047 $
142  * @see     SimpleStreamFilter 
143  * @see     #input(Instance)
144  * @see     #batchFinished()
145  * @see     #m_FirstBatchDone
146  */
147 public abstract class SimpleBatchFilter
148   extends SimpleFilter {
149 
150   /** for serialization */
151   private static final long serialVersionUID = 8102908673378055114L;
152 
153   /**
154    * returns true if the output format is immediately available after the
155    * input format has been set and not only after all the data has been
156    * seen (see batchFinished())
157    *
158    * @return      true if the output format is immediately available
159    * @see         #batchFinished()
160    * @see         #setInputFormat(Instances)
161    */
162   protected boolean hasImmediateOutputFormat() {
163     return false;
164   }
165 
166   /**
167    * Input an instance for filtering. Filter requires all
168    * training instances be read before producing output (calling the method
169    * batchFinished() makes the data available). If this instance is part of
170    * a new batch, m_NewBatch is set to false.
171    *
172    * @param instance    the input instance
173    * @return            true if the filtered instance may now be
174    *                    collected with output().
175    * @throws  IllegalStateException if no input structure has been defined
176    * @throws Exception  if something goes wrong
177    * @see           #batchFinished()
178    */
179   public boolean input(Instance instance) throws Exception {
180     if (getInputFormat() == null)
181       throw new IllegalStateException("No input instance format defined");
182     
183     if (m_NewBatch) {
184       resetQueue();
185       m_NewBatch = false;
186     }
187 
188     bufferInput(instance);
189     
190     if (isFirstBatchDone()) {
191       Instances inst = new Instances(getInputFormat());
192       inst = process(inst);
193       for (int i = 0; i < inst.numInstances(); i++)
194     push(inst.instance(i));
195       flushInput();
196     }
197 
198     return m_FirstBatchDone;
199   }
200 
201   /**
202    * Signify that this batch of input to the filter is finished. If
203    * the filter requires all instances prior to filtering, output()
204    * may now be called to retrieve the filtered instances. Any
205    * subsequent instances filtered should be filtered based on setting
206    * obtained from the first batch (unless the setInputFormat has been
207    * re-assigned or new options have been set). Sets m_FirstBatchDone
208    * and m_NewBatch to true.
209    *
210    * @return        true if there are instances pending output
211    * @throws IllegalStateException  if no input format has been set. 
212    * @throws Exception  if something goes wrong
213    * @see           #m_NewBatch
214    * @see           #m_FirstBatchDone 
215    */
216   public boolean batchFinished() throws Exception {
217     int         i;
218     Instances   inst;
219     
220     if (getInputFormat() == null)
221       throw new IllegalStateException("No input instance format defined");
222 
223     // get data
224     inst = new Instances(getInputFormat());
225 
226     // if output format hasn't been set yet, do it now
227     if (!hasImmediateOutputFormat() && !isFirstBatchDone())
228       setOutputFormat(determineOutputFormat(new Instances(inst, 0)));
229 
230     // don't do anything in case there are no instances pending.
231     // in case of second batch, they may have already been processed
232     // directly by the input method and added to the output queue
233     if (inst.numInstances() > 0) {
234       // process data
235       inst = process(inst);
236 
237       // clear input queue
238       flushInput();
239 
240       // move it to the output
241       for (i = 0; i < inst.numInstances(); i++)
242     push(inst.instance(i));
243     }
244     
245     m_NewBatch       = true;
246     m_FirstBatchDone = true;
247     
248     return (numPendingOutput() != 0);
249   }
250 }
