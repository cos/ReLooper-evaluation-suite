42  /**
43   * Class for examining the capabilities and finding problems with 
44   * kernels. If you implement an kernels using the WEKA.libraries,
45   * you should run the checks on it to ensure robustness and correct
46   * operation. Passing all the tests of this object does not mean
47   * bugs in the kernels don't exist, but this will help find some
48   * common ones. <p/>
49   * 
50   * Typical usage: <p/>
51   * <code>java weka.classifiers.functions.supportVector.CheckKernel -W kernel_name 
52   * -- kernel_options </code><p/>
53   * 
54   * CheckKernel reports on the following:
55   * <ul>
56   *    <li> Kernel abilities 
57   *      <ul>
58   *         <li> Possible command line options to the kernels </li>
59   *         <li> Whether the kernels can predict nominal, numeric, string, 
60   *              date or relational class attributes. </li>
61   *         <li> Whether the kernels can handle numeric predictor attributes </li>
62   *         <li> Whether the kernels can handle nominal predictor attributes </li>
63   *         <li> Whether the kernels can handle string predictor attributes </li>
64   *         <li> Whether the kernels can handle date predictor attributes </li>
65   *         <li> Whether the kernels can handle relational predictor attributes </li>
66   *         <li> Whether the kernels can handle multi-instance data </li>
67   *         <li> Whether the kernels can handle missing predictor values </li>
68   *         <li> Whether the kernels can handle missing class values </li>
69   *         <li> Whether a nominal kernels only handles 2 class problems </li>
70   *         <li> Whether the kernels can handle instance weights </li>
71   *      </ul>
72   *    </li>
73   *    <li> Correct functioning 
74   *      <ul>
75   *         <li> Correct initialisation during buildKernel (i.e. no result
76   *              changes when buildKernel called repeatedly) </li>
77   *         <li> Whether the kernels alters the data passed to it 
78   *              (number of instances, instance order, instance weights, etc) </li>
79   *      </ul>
80   *    </li>
81   *    <li> Degenerate cases 
82   *      <ul>
83   *         <li> building kernels with zero training instances </li>
84   *         <li> all but one predictor attribute values missing </li>
85   *         <li> all predictor attribute values missing </li>
86   *         <li> all but one class values missing </li>
87   *         <li> all class values missing </li>
88   *      </ul>
89   *    </li>
90   * </ul>
91   * Running CheckKernel with the debug option set will output the 
92   * training and test datasets for any failed tests.<p/>
93   *
94   * The <code>weka.classifiers.AbstractKernelTest</code> uses this
95   * class to test all the kernels. Any changes here, have to be 
96   * checked in that abstract test class, too. <p/>
97   *
98   <!-- options-start -->
99   * Valid options are: <p/>
100  * 
101  * <pre> -D
102  *  Turn on debugging output.</pre>
103  * 
104  * <pre> -S
105  *  Silent mode - prints nothing to stdout.</pre>
106  * 
107  * <pre> -N &lt;num&gt;
108  *  The number of instances in the datasets (default 20).</pre>
109  * 
110  * <pre> -nominal &lt;num&gt;
111  *  The number of nominal attributes (default 2).</pre>
112  * 
113  * <pre> -nominal-values &lt;num&gt;
114  *  The number of values for nominal attributes (default 1).</pre>
115  * 
116  * <pre> -numeric &lt;num&gt;
117  *  The number of numeric attributes (default 1).</pre>
118  * 
119  * <pre> -string &lt;num&gt;
120  *  The number of string attributes (default 1).</pre>
121  * 
122  * <pre> -date &lt;num&gt;
123  *  The number of date attributes (default 1).</pre>
124  * 
125  * <pre> -relational &lt;num&gt;
126  *  The number of relational attributes (default 1).</pre>
127  * 
128  * <pre> -num-instances-relational &lt;num&gt;
129  *  The number of instances in relational/bag attributes (default 10).</pre>
130  * 
131  * <pre> -words &lt;comma-separated-list&gt;
132  *  The words to use in string attributes.</pre>
133  * 
134  * <pre> -word-separators &lt;chars&gt;
135  *  The word separators to use in string attributes.</pre>
136  * 
137  * <pre> -W
138  *  Full name of the kernel analysed.
139  *  eg: weka.classifiers.functions.supportVector.RBFKernel
140  *  (default weka.classifiers.functions.supportVector.RBFKernel)</pre>
141  * 
142  * <pre> 
143  * Options specific to kernel weka.classifiers.functions.supportVector.RBFKernel:
144  * </pre>
145  * 
146  * <pre> -D
147  *  Enables debugging output (if available) to be printed.
148  *  (default: off)</pre>
149  * 
150  * <pre> -no-checks
151  *  Turns off all checks - use with caution!
152  *  (default: checks on)</pre>
153  * 
154  * <pre> -C &lt;num&gt;
155  *  The size of the cache (a prime number), 0 for full cache and 
156  *  -1 to turn it off.
157  *  (default: 250007)</pre>
158  * 
159  * <pre> -G &lt;num&gt;
160  *  The Gamma parameter.
161  *  (default: 0.01)</pre>
162  * 
163  <!-- options-end -->
164  *
165  * Options after -- are passed to the designated kernel.<p/>
166  *
167  * @author Len Trigg (trigg@cs.waikato.ac.nz)
168  * @author FracPete (fracpete at waikato dot ac dot nz)
169  * @version $Revision: 1.3 $
170  * @see TestInstances
171  */

196     while (en.hasMoreElements())
197       result.addElement(en.nextElement());
198     
199     result.addElement(new Option(
200         "\tFull name of the kernel analysed.\n"
201         +"\teg: weka.classifiers.functions.supportVector.RBFKernel\n"
202         + "\t(default weka.classifiers.functions.supportVector.RBFKernel)",
203         "W", 1, "-W"));
204     
205     if ((m_Kernel != null) 
206         && (m_Kernel instanceof OptionHandler)) {
207       result.addElement(new Option("", "", 0, 
208           "\nOptions specific to kernel "
209           + m_Kernel.getClass().getName()
210           + ":"));
211       Enumeration enu = ((OptionHandler)m_Kernel).listOptions();
212       while (enu.hasMoreElements())
213         result.addElement(enu.nextElement());
214     }
215     
216     return result.elements();
217   }

286    *

288    *
289    * @param options the list of options as an array of strings

292   public void setOptions(String[] options) throws Exception {
293     String      tmpStr;
294     
295     super.setOptions(options);
296     
297     tmpStr = Utils.getOption('W', options);
298     if (tmpStr.length() == 0)
299       tmpStr = weka.classifiers.functions.supportVector.RBFKernel.class.getName();
300     setKernel(
301     (Kernel) forName(
302         "weka.classifiers.functions.supportVector", 
303         Kernel.class, 
304         tmpStr, 
305         Utils.partitionOptions(options)));
306   }

308   /**
309    * Gets the current settings of the CheckKernel.
310    *
311    * @return an array of strings suitable for passing to setOptions
312    */
313   public String[] getOptions() {
314     Vector        result;
315     String[]      options;
316     int           i;
317     
318     result = new Vector();
319     
320     options = super.getOptions();
321     for (i = 0; i < options.length; i++)
322       result.add(options[i]);
323     
324     if (getKernel() != null) {
325       result.add("-W");
326       result.add(getKernel().getClass().getName());
327     }
328     
329     if ((m_Kernel != null) && (m_Kernel instanceof OptionHandler))
330       options = ((OptionHandler) m_Kernel).getOptions();
331     else
332       options = new String[0];
333     
334     if (options.length > 0) {
335       result.add("--");
336       for (i = 0; i < options.length; i++)
337         result.add(options[i]);
338     }
339     
340     return (String[]) result.toArray(new String[result.size()]);
341   }

343   /**
344    * Begin the tests, reporting results to System.out
345    */
346   public void doTests() {
347     
348     if (getKernel() == null) {
349       println("\n=== No kernel set ===");
350       return;
351     }
352     println("\n=== Check on kernel: "
353         + getKernel().getClass().getName()
354         + " ===\n");
355     
356     // Start tests
357     m_ClasspathProblems = false;
358     println("--> Checking for interfaces");
359     canTakeOptions();
360     boolean weightedInstancesHandler = weightedInstancesHandler()[0];
361     boolean multiInstanceHandler = multiInstanceHandler()[0];
362     println("--> Kernel tests");
363     declaresSerialVersionUID();
364     testsPerClassType(Attribute.NOMINAL,    weightedInstancesHandler, multiInstanceHandler);
365     testsPerClassType(Attribute.NUMERIC,    weightedInstancesHandler, multiInstanceHandler);
366     testsPerClassType(Attribute.DATE,       weightedInstancesHandler, multiInstanceHandler);
367     testsPerClassType(Attribute.STRING,     weightedInstancesHandler, multiInstanceHandler);
368     testsPerClassType(Attribute.RELATIONAL, weightedInstancesHandler, multiInstanceHandler);
369   }

389   /**
390    * Run a battery of tests for a given class attribute type
391    *
392    * @param classType true if the class attribute should be numeric
393    * @param weighted true if the kernel says it handles weights
394    * @param multiInstance true if the kernel is a multi-instance kernel
395    */
396   protected void testsPerClassType(int classType, 
397                                    boolean weighted,
398                                    boolean multiInstance) {
399     
400     boolean PNom = canPredict(true,  false, false, false, false, multiInstance, classType)[0];
401     boolean PNum = canPredict(false, true,  false, false, false, multiInstance, classType)[0];
402     boolean PStr = canPredict(false, false, true,  false, false, multiInstance, classType)[0];
403     boolean PDat = canPredict(false, false, false, true,  false, multiInstance, classType)[0];
404     boolean PRel;
405     if (!multiInstance)
406       PRel = canPredict(false, false, false, false,  true, multiInstance, classType)[0];
407     else
408       PRel = false;
409 
410     if (PNom || PNum || PStr || PDat || PRel) {
411       if (weighted)
412         instanceWeights(PNom, PNum, PStr, PDat, PRel, multiInstance, classType);
413       
414       if (classType == Attribute.NOMINAL)
415         canHandleNClasses(PNom, PNum, PStr, PDat, PRel, multiInstance, 4);
416 
417       if (!multiInstance) {
418     canHandleClassAsNthAttribute(PNom, PNum, PStr, PDat, PRel, multiInstance, classType, 0);
419     canHandleClassAsNthAttribute(PNom, PNum, PStr, PDat, PRel, multiInstance, classType, 1);
420       }
421       
422       canHandleZeroTraining(PNom, PNum, PStr, PDat, PRel, multiInstance, classType);
423       boolean handleMissingPredictors = canHandleMissing(PNom, PNum, PStr, PDat, PRel, 
424           multiInstance, classType, 
425           true, false, 20)[0];
426       if (handleMissingPredictors)
427         canHandleMissing(PNom, PNum, PStr, PDat, PRel, multiInstance, classType, true, false, 100);
428       
429       boolean handleMissingClass = canHandleMissing(PNom, PNum, PStr, PDat, PRel, 
430           multiInstance, classType, 
431           false, true, 20)[0];
432       if (handleMissingClass)
433         canHandleMissing(PNom, PNum, PStr, PDat, PRel, multiInstance, classType, false, true, 100);
434       
435       correctBuildInitialisation(PNom, PNum, PStr, PDat, PRel, multiInstance, classType);
436       datasetIntegrity(PNom, PNum, PStr, PDat, PRel, multiInstance, classType,
437           handleMissingPredictors, handleMissingClass);
438     }
439   }

442    * Checks whether the scheme can take command line options.
443    *
444    * @return index 0 is true if the kernel can take options
445    */
446   protected boolean[] canTakeOptions() {
447     
448     boolean[] result = new boolean[2];
449     
450     print("options...");
451     if (m_Kernel instanceof OptionHandler) {
452       println("yes");
453       if (m_Debug) {
454         println("\n=== Full report ===");
455         Enumeration enu = ((OptionHandler)m_Kernel).listOptions();
456         while (enu.hasMoreElements()) {
457           Option option = (Option) enu.nextElement();
458           print(option.synopsis() + "\n" 
459               + option.description() + "\n");
460         }
461         println("\n");
462       }
463       result[0] = true;
464     }
465     else {
466       println("no");
467       result[0] = false;
468     }
469     
470     return result;
471   }
