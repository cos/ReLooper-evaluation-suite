330   /** 
331    * takeStep method from Shevade et al.s paper.
332    * parameters correspond to pseudocode from paper.
333    * 
334    * @param i1
335    * @param i2
336    * @param alpha2
337    * @param alpha2Star
338    * @param phi2
339    * @return
340    * @throws Exception
341    */
342   protected int takeStep(int i1, int i2, double alpha2, double alpha2Star, double phi2) throws Exception {
343     //procedure takeStep(i1, i2)
344     //
345     //  if (i1 == i2) 
346     //    return 0 
347     if (i1 == i2) {
348       return 0;
349     }
350     double C1 = m_C * m_data.instance(i1).weight();
351     double C2 = m_C * m_data.instance(i2).weight();
352     //  alpha1, alpha1' = Lagrange multipliers for i1 
353     double alpha1 = m_alpha[i1];
354     double alpha1Star = m_alphaStar[i1];
355 //  double y1 = m_target[i1];
356     // TODO: verify we do not need to recompute m_error[i1] here
357     // TODO: since m_error is only updated for indices in m_I0
358     double phi1 = m_error[i1];
359 //  if ((m_iSet[i1] & I0)==0) {
360 //  phi1 = -SVMOutput(i1) - m_b + m_target[i1];
361 //  m_error[i1] = phi1;
362 //  }
363     //  k11 = kernel(point[i1], point[i1]) 
364     //  k12 = kernel(point[i1], point[i2]) 
365     //  k22 = kernel(point[i2], point[i2]) 
366     //  eta = -2*k12+k11+k22 
367     //  gamma = alpha1-alpha1'+alpha2-alpha2'
368     //
369     double k11 = m_kernel.eval(i1, i1, m_data.instance(i1));
370     double k12 = m_kernel.eval(i1, i2, m_data.instance(i1));
371     double k22 = m_kernel.eval(i2, i2, m_data.instance(i2));
372     double eta = -2 * k12 + k11 + k22;
373     double gamma = alpha1 - alpha1Star + alpha2 - alpha2Star;
374 //  if (eta < 0) {
375     // this may happen due to numeric instability
376     // due to Mercer's condition, this should not happen, hence we give up
377 //  return 0;
378 //  }
379     //  % We assume that eta > 0. Otherwise one has to repeat the complete 
380     //  % reasoning similarly (i.e. compute objective functions at L and H 
381     //  % and decide which one is largest
382     //
383     //  case1 = case2 = case3 = case4 = finished = 0 
384     //  alpha1old = alpha1, 
385     //  alpha1old' = alpha1' 
386     //  alpha2old = alpha2, 
387     //  alpha2old' = alpha2' 
388     //  deltaphi = F1 - F2 
389     //
390     
391     //  while !finished
392     //    % This loop is passed at most three times 
393     //    % Case variables needed to avoid attempting small changes twice 
394     //    if (case1 == 0) &&
395     //       (alpha1 > 0 || (alpha1' == 0 && deltaphi > 0)) && 
396     //       (alpha2 > 0 || (alpha2' == 0 && deltaphi < 0))
397     //        compute L, H (w.r.t. alpha1, alpha2) 
398     //        if (L < H)
399     //          a2 = alpha2 - (deltaphi / eta ) a2 = min(a2, H) a2 = max(L, a2) a1 = alpha1 - (a2 - alpha2) 
400     //          update alpha1, alpha2 if change is larger than some eps 
401     //        else
402     //          finished = 1 
403     //        endif 
404     //      case1 = 1 
405     //    elseif (case2 == 0) &&
406     //           (alpha1 > 0 || (alpha1' == 0 && deltaphi > 2*epsilon)) && 
407     //           (alpha2' > 0 || (alpha2 == 0 && deltaphi > 2*epsilon))
408     //
409     //        compute L, H (w.r.t. alpha1, alpha2') 
410     //        if (L < H)
411     //          a2 = alpha2' + ((deltaphi - 2*epsilon)/eta)) a2 = min(a2, H) a2 = max(L, a2) a1 = alpha1 + (a2-alpha2') 
412     //          update alpha1, alpha2' if change is larger than some eps 
413     //        else
414     //          finished = 1 
415     //        endif 
416     //        case2 = 1 
417     //    elseif (case3 == 0) &&
418     //           (alpha1' > 0 || (alpha1 == 0 && deltaphi < -2*epsilon)) && 
419     //           (alpha2 > 0 || (alpha2' == 0 && deltaphi < -2*epsilon))
420     //         compute L, H (w.r.t. alpha1', alpha2) 
421     //         if (L < H)
422     //           a2 = alpha2 - ((deltaphi + 2*epsilon)/eta) a2 = min(a2, H) a2 = max(L, a2) a1 = alpha1' + (a2 - alpha2) 
423     //           update alpha1', alpha2 if change is larger than some eps 
424     //         else
425     //           finished = 1 
426     //         endif 
427     //         case3 = 1 
428     //    elseif (case4 == 0) &&
429     //           (alpha1' > 0) || (alpha1 == 0 && deltaphi < 0)) && 
430     //           (alpha2' > 0) || (alpha2 == 0 && deltaphi > 0))
431     //         compute L, H (w.r.t. alpha1', alpha2') 
432     //         if (L < H) 
433     //           a2 = alpha2' + deltaphi/eta a2 = min(a2, H) a2 = max(L, a2) a1 = alpha1' - (a2 - alpha2') 
434     //           update alpha1, alpha2' if change is larger than some eps 
435     //         else
436     //           finished = 1 
437     //         endif 
438     //         case4 = 1 
439     //    else
440     //      finished = 1 
441     //    endif 
442     //    update deltaphi 
443     //  endwhile 
444     
445     double alpha1old = alpha1;
446     double alpha1Starold = alpha1Star;
447     double alpha2old = alpha2;
448     double alpha2Starold = alpha2Star;
449     double deltaPhi = phi1 - phi2;
450     
451     if (findOptimalPointOnLine(i1, alpha1, alpha1Star, C1, i2, alpha2, alpha2Star, C2, gamma, eta, deltaPhi)) {
452       
453       alpha1 = m_alpha[i1];
454       alpha1Star = m_alphaStar[i1];
455       alpha2 = m_alpha[i2];
456       alpha2Star = m_alphaStar[i2];
457       
458       //  if changes in alpha('), alpha2(') are larger than some eps
459       //    Update f-cache[i] for i in I.0 using new Lagrange multipliers 
460       //    Store the changes in alpha, alpha' array 
461       //    Update I.0, I.1, I.2, I.3 
462       //    Compute (i.low, b.low) and (i.up, b.up) by applying the conditions mentioned above, using only i1, i2 and indices in I.0 
463       //    return 1 
464       //  else
465       //    return 0
466       //endif endprocedure
467       
468       //        Update error cache using new Lagrange multipliers 
469       double dAlpha1 = alpha1 - alpha1old - (alpha1Star - alpha1Starold);
470       double dAlpha2 = alpha2 - alpha2old - (alpha2Star - alpha2Starold);
471       for (int j = m_I0.getNext(-1); j != -1; j = m_I0.getNext(j)) {
472     if ((j != i1) && (j != i2)) {
473       m_error[j] -= dAlpha1 * m_kernel.eval(i1, j, m_data.instance(i1)) 
474       + dAlpha2 * m_kernel.eval(i2, j, m_data.instance(i2));
475     }
476       }
477       m_error[i1] -= dAlpha1 * k11 + dAlpha2 * k12;
478       m_error[i2] -= dAlpha1 * k12 + dAlpha2 * k22;
479       
480       updateIndexSetFor(i1, C1);
481       updateIndexSetFor(i2, C2);
482       
483       //    Compute (i.low, b.low) and (i.up, b.up) by applying the conditions mentioned above, using only i1, i2 and indices in I.0 
484       m_bUp = Double.MAX_VALUE; 
485       m_bLow = -Double.MAX_VALUE; 
486       for (int j = m_I0.getNext(-1); j != -1; j = m_I0.getNext(j)) {
487     updateBoundaries(j, m_error[j]);
488       }
489       if (!m_I0.contains(i1)) {
490     updateBoundaries(i1, m_error[i1]);
491       }
492       if (!m_I0.contains(i2)) {
493     updateBoundaries(i2, m_error[i2]);
494       }
495       
496       return 1;
497     } 
498     else {
499       return 0;
500     }
501   }

503   /**
504    * updates the index sets I0a, IOb, I1, I2 and I3 for vector i
505    * 
506    * @param i index of vector
507    * @param C capacity for vector i
508    * @throws Exception
509    */
510   protected void updateIndexSetFor(int i, double C) throws Exception {
511     /*
512      m_I0a.delete(i);
513      m_I0b.delete(i);
514      m_I1.delete(i);
515      m_I2.delete(i);
516      m_I3.delete(i);
517      */
518     if (m_alpha[i] == 0 && m_alphaStar[i] == 0) {
519       //m_I1.insert(i);
520       m_iSet[i] = I1;
521       m_I0.delete(i);
522     } else if (m_alpha[i] > 0) {
523       if (m_alpha[i] < C) {
524     if ((m_iSet[i] & I0) == 0) {
525       //m_error[i] = -SVMOutput(i) - m_b + m_target[i];
526       m_I0.insert(i);
527     }
528     //m_I0a.insert(i);
529     m_iSet[i] = I0a;
530       } else { // m_alpha[i] == C
531     //m_I3.insert(i);
532     m_iSet[i] = I3;
533     m_I0.delete(i);
534       }
535     } else {// m_alphaStar[i] > 0 
536       if (m_alphaStar[i] < C) {
537     if ((m_iSet[i] & I0) == 0) {
538       //m_error[i] = -SVMOutput(i) - m_b + m_target[i];
539       m_I0.insert(i);
540     }
541     //m_I0b.insert(i);
542     m_iSet[i] = I0b;
543       } else { // m_alpha[i] == C
544     //m_I2.insert(i);
545     m_iSet[i] = I2;
546     m_I0.delete(i);
547       }
548     }
549   }

551   /**
552    * updates boundaries bLow and bHi and corresponding indexes
553    * 
554    * @param i2 index of vector
555    * @param F2 error of vector i2
556    */
557   protected void updateBoundaries(int i2, double F2) {      
558     int iSet = m_iSet[i2];
559     
560     double FLow = m_bLow;
561     if ((iSet & (I2 | I0b)) > 0) {
562       FLow = F2 + m_epsilon;
563     } else if ((iSet & (I1 | I0a)) > 0) {
564       FLow = F2 - m_epsilon;
565     }
566     if (m_bLow < FLow) {
567       m_bLow = FLow;
568       m_iLow = i2;
569     }
570     double FUp = m_bUp;
571     if ((iSet & (I3 | I0a)) > 0) {
572       FUp = F2 - m_epsilon;
573     } else if ((iSet & (I1 | I0b)) > 0) {
574       FUp = F2 + m_epsilon;
575     }
576     if (m_bUp > FUp) {
577       m_bUp = FUp;
578       m_iUp = i2;
579     }
580   }

582   /** 
583    * parameters correspond to pseudocode from paper.
584    * 
585    * @param i2 index of  candidate
586    * @return
587    * @throws Exception
588    */
589   protected int examineExample(int i2) throws Exception {
590     //procedure examineExample(i2)
591     //
592     //  alpha2, alpha2' = Lagrange multipliers for i2 
593     double alpha2 = m_alpha[i2];
594     double alpha2Star = m_alphaStar[i2];
595     
596     //  if (i2 is in I.0)
597     //    F2 = f-cache[i2] 
598     //  else
599     //    compute F2 = F.i2 and set f-cache[i2] = F2 
600     //    % Update (b.low, i.low) or (b.up, i.up) using (F2, i2)... 
601     //    if (i2 is in I.1)
602     //      if (F2+epsilon < b.up)
603     //        b.up = F2+epsilon, 
604     //        i.up = i2 
605     //      elseif (F2-epsilon > b.low)
606     //        b.low = F2-epsilon, 
607     //        i.low = i2 
608     //      end if 
609     //    elseif ( (i2 is in I.2) && (F2+epsilon > b.low) )
610     //      b.low = F2+epsilon, 
611     //      i.low = i2 
612     //    elseif ( (i2 is in I.3) && (F2-epsilon < b.up) )
613     //      b.up = F2-epsilon, 
614     //      i.up = i2 
615     //    endif 
616     //  endif 
617     
618     int iSet = m_iSet[i2];
619     double F2 = m_error[i2];
620     if (!m_I0.contains(i2)) {
621       F2 = -SVMOutput(i2) - m_b + m_target[i2];
622       m_error[i2] = F2;
623       if (iSet == I1) {
624     if (F2 + m_epsilon < m_bUp) {
625       m_bUp = F2 + m_epsilon;
626       m_iUp = i2;
627     } else if (F2 - m_epsilon > m_bLow) {
628       m_bLow = F2 - m_epsilon;
629       m_iLow = i2;
630     } 
631       } else if ((iSet == I2) && (F2 + m_epsilon > m_bLow)) {
632     m_bLow = F2 + m_epsilon;
633     m_iLow = i2;
634       } else if ((iSet == I3) && (F2 - m_epsilon < m_bUp)) {
635     m_bUp = F2 - m_epsilon;
636     m_iUp = i2;
637       }
638     }
639     
640     //  % Check optimality using current b.low and b.up and, if 
641     //  % violated, find an index i1 to do joint optimization with i2... 
642     //  optimality = 1;
643     //  case 1: i2 is in I.0a
644     //    if (b.low-(F2-epsilon) > 2 * tol)
645     //      optimality = 0;  
646     //      i1 = i.low; 
647     //      % For i2 in I.0a choose the better i1... 
648     //      if ((F2-epsilon)-b.up > b.low-(F2-epsilon))
649     //        i1 = i.up; 
650     //      endif 
651     //    elseif ((F2-epsilon)-b.up > 2 * tol)
652     //      optimality = 0; 
653     //      i1 = i.up; 
654     //      % For i2 in I.0a choose the better i1... 
655     //      if ((b.low-(F2-epsilon) > (F2-epsilon)-b.up)
656     //        i1 = i.low; 
657     //      endif 
658     //    endif 
659     //  case 2: i2 is in I.0b
660     //    if (b.low-(F2+epsilon) > 2 * tol)
661     //      optimality = 0; 
662     //      i1 = i.low; 
663     //      % For i2 in I.0b choose the better i1... 
664     //      if ((F2+epsilon)-b.up > b.low-(F2+epsilon))
665     //        i1 = i.up; 
666     //      endif 
667     //    elseif ((F2+epsilon)-b.up > 2 * tol)
668     //      optimality = 0; 
669     //      i1 = i.up; 
670     //      % For i2 in I.0b choose the better i1... 
671     //      if ((b.low-(F2+epsilon) > (F2+epsilon)-b.up)
672     //        i1 = i.low; 
673     //      endif 
674     //    endif 
675     //  case 3: i2 is in I.1
676     //    if (b.low-(F2+epsilon) > 2 * tol)
677     //      optimality = 0; 
678     //      i1 = i.low; 
679     //      % For i2 in I1 choose the better i1... 
680     //      if ((F2+epsilon)-b.up > b.low-(F2+epsilon)
681     //        i1 = i.up; 
682     //      endif 
683     //    elseif ((F2-epsilon)-b.up > 2 * tol)
684     //      optimality = 0; 
685     //      i1 = i.up; 
686     //      % For i2 in I1 choose the better i1... 
687     //      if (b.low-(F2-epsilon) > (F2-epsilon)-b.up)
688     //        i1 = i.low; 
689     //      endif 
690     //    endif 
691     //  case 4: i2 is in I.2
692     //    if ((F2+epsilon)-b.up > 2*tol)
693     //      optimality = 0, 
694     //      i1 = i.up 
695     //     endif 
696     //  case 5: i2 is in I.3
697     //    if ((b.low-(F2-epsilon) > 2*tol)
698     //      optimality = 0, i1 = i.low 
699     //    endif
700     
701     int i1 = i2;
702     boolean bOptimality = true;
703     //case 1: i2 is in I.0a
704     if (iSet == I0a) {
705       if (m_bLow - (F2 - m_epsilon) > 2 * m_fTolerance) {
706     bOptimality = false;
707     i1 = m_iLow;
708     //% For i2 in I .0 a choose the better i1...
709     if ((F2 - m_epsilon) - m_bUp > m_bLow - (F2 - m_epsilon)) {
710       i1 = m_iUp;
711     }
712       } else if ((F2 - m_epsilon) - m_bUp > 2 * m_fTolerance) {
713     bOptimality = false;
714     i1 = m_iUp;
715     //% For i2 in I.0a choose the better i1... 
716     if (m_bLow - (F2 - m_epsilon) > (F2 - m_epsilon) - m_bUp) {
717       i1 = m_iLow;
718     }
719       }
720     } // case 2: i2 is in I.0b
721     else if (iSet == I0b) {
722       if (m_bLow - (F2 + m_epsilon) > 2 * m_fTolerance) {
723     bOptimality = false;
724     i1 = m_iLow; // % For i2 in I.0b choose the better i1... 
725     if ((F2 + m_epsilon) - m_bUp > m_bLow - (F2 + m_epsilon)) {
726       i1 = m_iUp;
727     }
728       } else if ((F2 + m_epsilon) - m_bUp > 2 * m_fTolerance) {
729     bOptimality = false;
730     i1 = m_iUp; // % For i2 in I.0b choose the better i1... 
731     if (m_bLow - (F2 + m_epsilon) > (F2 + m_epsilon) - m_bUp) {
732       i1 = m_iLow;
733     }
734       }
735     } // case 3: i2 is in I.1
736     else if (iSet == I1) {
737       if (m_bLow - (F2 + m_epsilon) > 2 * m_fTolerance) {
738     bOptimality = false;
739     i1 = m_iLow;
740     //% For i2 in I1 choose the better i1...
741     if ((F2 + m_epsilon) - m_bUp > m_bLow - (F2 + m_epsilon)) {
742       i1 = m_iUp;
743     }
744       } else if ((F2 - m_epsilon) - m_bUp > 2 * m_fTolerance) {
745     bOptimality = false;
746     i1 = m_iUp; // % For i2 in I1 choose the better i1... 
747     if (m_bLow - (F2 - m_epsilon) > (F2 - m_epsilon) - m_bUp) {
748       i1 = m_iLow;
749     }
750       }
751     } //case 4: i2 is in I.2
752     else if (iSet == I2) {
753       if ((F2 + m_epsilon) - m_bUp > 2 * m_fTolerance) {
754     bOptimality = false;
755     i1 = m_iUp;
756       }
757     } //case 5: i2 is in I.3
758     else if (iSet == I3) {
759       if (m_bLow - (F2 - m_epsilon) > 2 * m_fTolerance) {
760     bOptimality = false;
761     i1 = m_iLow;
762       }
763     }
764     // if (optimality == 1) 
765     //    return 0
766     //  if (takeStep(i1, i2))
767     //    return 1 
768     //   else
769     //    return 0 
770     //  endif 
771     //endprocedure
772     if (bOptimality) {
773       return 0;
774     }
775     return takeStep(i1, i2, m_alpha[i2], m_alphaStar[i2], F2);
776   }

778   /** 
779    * initialize various variables before starting the actual optimizer 
780    * 
781    * @param data    data set used for learning
782    * @throws Exception  if something goes wrong
783    */
784   protected void init(Instances data) throws Exception {
785     super.init(data);
786     // from Keerthi's pseudo code:
787     //  set alpha and alpha' to zero for every example set I.1 to contain all the examples 
788     //  Choose any example i from the training set. 
789     //  set b.up = target[i]+epsilon 
790     //  set b.low = target[i]-espilon 
791     //  i.up = i.low = i; 
792     // Initialize sets
793     m_I0 = new SMOset(m_data.numInstances());
794     m_iSet = new int [m_data.numInstances()];
795     for (int i = 0; i < m_nInstances; i++) {
796       m_iSet[i] = I1;
797     }
798     // m_iUp = m_random.nextInt(m_nInstances);
799     m_iUp = 0;
800     m_bUp = m_target[m_iUp] + m_epsilon;
801     m_iLow = m_iUp;
802     m_bLow = m_target[m_iLow] - m_epsilon;
803     //init error cache
804     m_error = new double[m_nInstances];
805     for (int i = 0; i < m_nInstances; i++) {
806       m_error[i] = m_target[i];
807     }
808   }

935   protected void wrapUp() throws Exception {
