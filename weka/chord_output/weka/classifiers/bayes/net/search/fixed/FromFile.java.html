36  /** 
37   <!-- globalinfo-start -->
38   * The FromFile reads the structure of a Bayes net from a file in BIFF format.
39   * <p/>
40   <!-- globalinfo-end -->
41   *
42   <!-- options-start -->
43   * Valid options are: <p/>
44   * 
45   * <pre> -B &lt;BIF File&gt;
46   *  Name of file containing network structure in BIF format
47   * </pre>
48   * 
49   <!-- options-end -->
50   * 
51   * @author Remco Bouckaert
52   * @version $Revision: 1.8 $
53   */
54  public class FromFile 
55      extends SearchAlgorithm {
56    
57      /** for serialization */
58      static final long serialVersionUID = 7334358169507619525L;
59    
60      /** name of file to read structure from **/
61      String m_sBIFFile = "";
62          
63      /**
64       * Returns a string describing this object
65       * @return a description of the classifier suitable for
66       * displaying in the explorer/experimenter gui
67       */
68      public String globalInfo() {
69        return 
70            "The FromFile reads the structure of a Bayes net from a file "
71          + "in BIFF format.";
72      }
73  
74      /**
75       * 
76       * @param bayesNet
77       * @param instances the instances to work with
78       * @throws Exception if attribute from BIF file could not be found
79       */
80      public void buildStructure (BayesNet bayesNet, Instances instances) throws Exception {
81          // read network structure in BIF format
82          BIFReader bifReader = new BIFReader();
83          bifReader.processFile(m_sBIFFile);
84          // copy parent sets
85          for (int iAttribute = 0; iAttribute < instances.numAttributes(); iAttribute++) {
86              int iBIFAttribute = bifReader.getNode(bayesNet.getNodeName(iAttribute));
87              ParentSet bifParentSet = bifReader.getParentSet(iBIFAttribute);
88              for (int iBIFParent = 0; iBIFParent < bifParentSet.getNrOfParents(); iBIFParent++) {
89                  String sParent = bifReader.getNodeName(bifParentSet.getParent(iBIFParent));
90                  int iParent = 0;
91                  while (iParent < instances.numAttributes() && !bayesNet.getNodeName(iParent).equals(sParent)) {
92                      iParent++;
93                  }
94                  if (iParent >= instances.numAttributes()) {
95                      throw new Exception("Could not find attribute " + sParent + " from BIF file in data");
96                  }
97                  bayesNet.getParentSet(iAttribute).addParent(iParent, instances);
98              }
99          }
100     } // buildStructure

102     /**
103      * Set name of network in BIF file to read structure from
104      * 
105      * @param sBIFFile the name of the BIF file
106      */
107     public void setBIFFile(String sBIFFile) {
108         m_sBIFFile = sBIFFile;
109     }
110 
111     /**
112      * Get name of network in BIF file to read structure from
113      * @return BIF file name
114      */
115     public String getBIFFile() {
116         return m_sBIFFile;
117     }
118 
119     /**
120      * Returns an enumeration describing the available options.
121      *
122      * @return an enumeration of all the available options.
123      */
124     public Enumeration listOptions() {
125       Vector newVector = new Vector();
126 
127       newVector.addElement(new Option("\tName of file containing network structure in BIF format\n", 
128                      "B", 1, "-B <BIF File>"));
129 
130           Enumeration en = super.listOptions();
131           while (en.hasMoreElements())
132             newVector.addElement(en.nextElement());
133           
134       return newVector.elements();
135     }
136 
137     /**
138      * Parses a given list of options. <p/>
139      *
140      <!-- options-start -->
141      * Valid options are: <p/>
142      * 
143      * <pre> -B &lt;BIF File&gt;
144      *  Name of file containing network structure in BIF format
145      * </pre>
146      * 
147      <!-- options-end -->
148      *
149      * @param options the list of options as an array of strings
150      * @throws Exception if an option is not supported
151      */
152     public void setOptions(String[] options) throws Exception {
153       setBIFFile( Utils.getOption('B', options));
154           
155           super.setOptions(options);
156     }
157 
158     /**
159      * Gets the current settings of the search algorithm.
160      *
161      * @return an array of strings suitable for passing to setOptions
162      */
163     public String [] getOptions() {
164           String[] superOptions = super.getOptions();
165       String [] options  = new String [2 + superOptions.length];
166       int current = 0;
167 
168           options[current++] = "-B";
169       options[current++] = "" + getBIFFile();
170 
171           // insert options from parent class
172           for (int iOption = 0; iOption < superOptions.length; iOption++) {
173                   options[current++] = superOptions[iOption];
174           }
175 
176       // Fill up rest with empty strings, not nulls!
177       while (current < options.length) {
178         options[current++] = "";
179       }
180       return options;
181     }
182 
183     /**
184      * Returns the revision string.
185      * 
186      * @return      the revision
187      */
188     public String getRevision() {
189       return RevisionUtils.extract("$Revision: 1.8 $");
190     }
