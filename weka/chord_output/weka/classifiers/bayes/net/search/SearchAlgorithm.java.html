72      /**
73       * AddArcMakesSense checks whether adding the arc from iAttributeTail to iAttributeHead
74       * does not already exists and does not introduce a cycle
75       * 
76       * @param bayesNet
77       * @param instances
78       * @param iAttributeHead index of the attribute that becomes head of the arrow
79       * @param iAttributeTail index of the attribute that becomes tail of the arrow
80       * @return true if adding arc is allowed, otherwise false
81       */
82      protected boolean addArcMakesSense(
83          BayesNet bayesNet,
84          Instances instances,
85          int iAttributeHead,
86          int iAttributeTail) {
87          if (iAttributeHead == iAttributeTail) {
88              return false;
89          }
90  
91          // sanity check: arc should not be in parent set already
92          if (isArc(bayesNet, iAttributeHead, iAttributeTail)) {
93                 return false;
94          }
95  
96          // sanity check: arc should not introduce a cycle
97          int nNodes = instances.numAttributes();
98          boolean[] bDone = new boolean[nNodes];
99  
100         for (int iNode = 0; iNode < nNodes; iNode++) {
101             bDone[iNode] = false;
102         }
103 
104         // check for cycles
105         bayesNet.getParentSet(iAttributeHead).addParent(iAttributeTail, instances);
106 
107         for (int iNode = 0; iNode < nNodes; iNode++) {
108 
109             // find a node for which all parents are 'done'
110             boolean bFound = false;
111 
112             for (int iNode2 = 0; !bFound && iNode2 < nNodes; iNode2++) {
113                 if (!bDone[iNode2]) {
114                     boolean bHasNoParents = true;
115 
116                     for (int iParent = 0; iParent < bayesNet.getParentSet(iNode2).getNrOfParents(); iParent++) {
117                         if (!bDone[bayesNet.getParentSet(iNode2).getParent(iParent)]) {
118                             bHasNoParents = false;
119                         }
120                     }
121 
122                     if (bHasNoParents) {
123                         bDone[iNode2] = true;
124                         bFound = true;
125                     }
126                 }
127             }
128 
129             if (!bFound) {
130                 bayesNet.getParentSet(iAttributeHead).deleteLastParent(instances);
131 
132                 return false;
133             }
134         }
135 
136         bayesNet.getParentSet(iAttributeHead).deleteLastParent(instances);
137 
138         return true;
139     } // AddArcMakesCycle

141     /**
142      * reverseArcMakesSense checks whether the arc from iAttributeTail to
143      * iAttributeHead exists and reversing does not introduce a cycle
144      * 
145      * @param bayesNet
146      * @param instances
147      * @param iAttributeHead index of the attribute that is head of the arrow
148      * @param iAttributeTail index of the attribute that is tail of the arrow
149      * @return true if the arc from iAttributeTail to iAttributeHead exists and reversing does not introduce a cycle 
150      */
151     protected boolean reverseArcMakesSense(
152         BayesNet bayesNet,
153         Instances instances,
154         int iAttributeHead,
155         int iAttributeTail) {
156       
157         if (iAttributeHead == iAttributeTail) {
158             return false;
159         }
160 
161         // sanity check: arc should be in parent set already
162         if (!isArc(bayesNet, iAttributeHead, iAttributeTail)) {
163             return false;
164         }
165 
166         // sanity check: arc should not introduce a cycle
167         int nNodes = instances.numAttributes();
168         boolean[] bDone = new boolean[nNodes];
169 
170         for (int iNode = 0; iNode < nNodes; iNode++) {
171             bDone[iNode] = false;
172         }
173 
174         // check for cycles
175         bayesNet.getParentSet(iAttributeTail).addParent(iAttributeHead, instances);
176 
177         for (int iNode = 0; iNode < nNodes; iNode++) {
178 
179             // find a node for which all parents are 'done'
180             boolean bFound = false;
181 
182             for (int iNode2 = 0; !bFound && iNode2 < nNodes; iNode2++) {
183                 if (!bDone[iNode2]) {
184                     ParentSet parentSet = bayesNet.getParentSet(iNode2);
185                     boolean bHasNoParents = true;
186                     for (int iParent = 0; iParent < parentSet.getNrOfParents(); iParent++) {
187                         if (!bDone[parentSet.getParent(iParent)]) {
188                             
189                             // this one has a parent which is not 'done' UNLESS it is the arc to be reversed
190                             if (!(iNode2 == iAttributeHead && parentSet.getParent(iParent) == iAttributeTail)) {
191                                 bHasNoParents = false;
192                             }
193                         }
194                     }
195 
196                     if (bHasNoParents) {
197                         bDone[iNode2] = true;
198                         bFound = true;
199                     }
200                 }
201             }
202 
203             if (!bFound) {
204                 bayesNet.getParentSet(iAttributeTail).deleteLastParent(instances);
205                 return false;
206             }
207         }
208 
209         bayesNet.getParentSet(iAttributeTail).deleteLastParent(instances);
210         return true;
211     } // ReverseArcMakesCycle

213     /**
214      * IsArc checks whether the arc from iAttributeTail to iAttributeHead already exists
215      * 
216      * @param bayesNet
217      * @param iAttributeHead index of the attribute that becomes head of the arrow
218      * @param iAttributeTail index of the attribute that becomes tail of the arrow
219      * @return true if the arc from iAttributeTail to iAttributeHead already exists
220      */
221     protected boolean isArc(BayesNet bayesNet, int iAttributeHead, int iAttributeTail) {
222         for (int iParent = 0; iParent < bayesNet.getParentSet(iAttributeHead).getNrOfParents(); iParent++) {
223             if (bayesNet.getParentSet(iAttributeHead).getParent(iParent) == iAttributeTail) {
224                 return true;
225             }
226         }
227 
228         return false;
229     } // IsArc

267     /**
268      * buildStructure determines the network structure/graph of the network.
269      * The default behavior is creating a network where all nodes have the first
270      * node as its parent (i.e., a BayesNet that behaves like a naive Bayes classifier).
271      * This method can be overridden by derived classes to restrict the class
272      * of network structures that are acceptable.
273      * 
274      * @param bayesNet the network
275      * @param instances the data to use
276      * @throws Exception if something goes wrong
277      */
278     public void buildStructure(BayesNet bayesNet, Instances instances) throws Exception {
279         if (m_bInitAsNaiveBayes) {
280             int iClass = instances.classIndex();
281             // initialize parent sets to have arrow from classifier node to
282             // each of the other nodes
283             for (int iAttribute = 0; iAttribute < instances.numAttributes(); iAttribute++) {
284                 if (iAttribute != iClass) {
285                     bayesNet.getParentSet(iAttribute).addParent(iClass, instances);
286                 }
287             }
288         }
289         search(bayesNet, instances);
290         if (m_bMarkovBlanketClassifier) {
291             doMarkovBlanketCorrection(bayesNet, instances);
292         }
293     } // buildStructure

304     /** 
305      * for each node in the network make sure it is in the
306      * Markov blanket of the classifier node, and if not,
307      * add arrows so that it is. If the node is an ancestor
308      * of the classifier node, add arrow pointing to the classifier
309      * node, otherwise, add arrow pointing to attribute node.
310      * 
311      * @param bayesNet
312      * @param instances
313      */
314     protected void doMarkovBlanketCorrection(BayesNet bayesNet, Instances instances) {
315         // Add class node as parent if it is not in the Markov Boundary
316         int iClass = instances.classIndex();
317         ParentSet ancestors = new ParentSet();
318         int nOldSize = 0;
319         ancestors.addParent(iClass, instances);
320         while (nOldSize != ancestors.getNrOfParents()) {
321             nOldSize = ancestors.getNrOfParents();
322             for (int iNode = 0; iNode < nOldSize; iNode++) {
323                 int iCurrent = ancestors.getParent(iNode);
324                 ParentSet p = bayesNet.getParentSet(iCurrent);
325                 for (int iParent = 0; iParent < p.getNrOfParents(); iParent++) {
326                     if (!ancestors.contains(p.getParent(iParent))) {
327                         ancestors.addParent(p.getParent(iParent), instances);
328                     }
329                 }
330             }
331         }
332         for (int iAttribute = 0; iAttribute < instances.numAttributes(); iAttribute++) {
333             boolean bIsInMarkovBoundary = (iAttribute == iClass)
334                     || bayesNet.getParentSet(iAttribute).contains(iClass)
335                     || bayesNet.getParentSet(iClass).contains(iAttribute);
336             for (int iAttribute2 = 0; !bIsInMarkovBoundary && iAttribute2 < instances.numAttributes(); iAttribute2++) {
337                 bIsInMarkovBoundary =
338                     bayesNet.getParentSet(iAttribute2).contains(iAttribute)
339                         && bayesNet.getParentSet(iAttribute2).contains(iClass);
340             }
341             if (!bIsInMarkovBoundary) {
342                 if (ancestors.contains(iAttribute)) {
343                     if (bayesNet.getParentSet(iClass).getCardinalityOfParents() < 1024) {
344                         bayesNet.getParentSet(iClass).addParent(iAttribute, instances);
345                     } else {
346                         // too bad
347                     }
348                 } else {
349                     bayesNet.getParentSet(iAttribute).addParent(iClass, instances);
350                 }
351             }
352         }
353     } // doMarkovBlanketCorrection
