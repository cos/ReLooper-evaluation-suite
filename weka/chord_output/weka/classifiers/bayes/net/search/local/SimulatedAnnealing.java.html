137     /**
138      * 
139      * @param bayesNet the network
140      * @param instances the data to use
141      * @throws Exception if something goes wrong
142      */
143     public void search (BayesNet bayesNet, Instances instances) throws Exception {
144         m_random = new Random(m_nSeed);
145         
146         // determine base scores
147         double [] fBaseScores = new double [instances.numAttributes()];
148         double fCurrentScore = 0;
149         for (int iAttribute = 0; iAttribute < instances.numAttributes(); iAttribute++) {
150             fBaseScores[iAttribute] = calcNodeScore(iAttribute);
151             fCurrentScore += fBaseScores[iAttribute];
152         }
153 
154         // keep track of best scoring network
155         double fBestScore = fCurrentScore;
156         BayesNet bestBayesNet = new BayesNet();
157         bestBayesNet.m_Instances = instances;
158         bestBayesNet.initStructure();
159         copyParentSets(bestBayesNet, bayesNet);
160 
161         double fTemp = m_fTStart;
162         for (int iRun = 0; iRun < m_nRuns; iRun++) {
163             boolean bRunSucces = false;
164             double fDeltaScore = 0.0;
165             while (!bRunSucces) {
166                 // pick two nodes at random
167                 int iTailNode = Math.abs(m_random.nextInt()) % instances.numAttributes();
168                 int iHeadNode = Math.abs(m_random.nextInt()) % instances.numAttributes();
169                 while (iTailNode == iHeadNode) {
170                     iHeadNode = Math.abs(m_random.nextInt()) % instances.numAttributes();
171                 }
172                 if (isArc(bayesNet, iHeadNode, iTailNode)) {
173                     bRunSucces = true;
174                     // either try a delete
175                     bayesNet.getParentSet(iHeadNode).deleteParent(iTailNode, instances);
176                     double fScore = calcNodeScore(iHeadNode);
177                     fDeltaScore = fScore - fBaseScores[iHeadNode];
178 //System.out.println("Try delete " + iTailNode + "->" + iHeadNode + " dScore = " + fDeltaScore);                    
179                     if (fTemp * Math.log((Math.abs(m_random.nextInt()) % 10000)/10000.0  + 1e-100) < fDeltaScore) {
180 //System.out.println("success!!!");                    
181                         fCurrentScore += fDeltaScore;
182                         fBaseScores[iHeadNode] = fScore;
183                     } else {
184                         // roll back
185                         bayesNet.getParentSet(iHeadNode).addParent(iTailNode, instances);
186                     }
187                 } else {
188                     // try to add an arc
189                     if (addArcMakesSense(bayesNet, instances, iHeadNode, iTailNode)) {
190                         bRunSucces = true;
191                         double fScore = calcScoreWithExtraParent(iHeadNode, iTailNode);
192                         fDeltaScore = fScore - fBaseScores[iHeadNode];
193 //System.out.println("Try add " + iTailNode + "->" + iHeadNode + " dScore = " + fDeltaScore);                    
194                         if (fTemp * Math.log((Math.abs(m_random.nextInt()) % 10000)/10000.0  + 1e-100) < fDeltaScore) {
195 //System.out.println("success!!!");                    
196                             bayesNet.getParentSet(iHeadNode).addParent(iTailNode, instances);
197                             fBaseScores[iHeadNode] = fScore;
198                             fCurrentScore += fDeltaScore;
199                         }
200                     }
201                 }
202             }
203             if (fCurrentScore > fBestScore) {
204                 copyParentSets(bestBayesNet, bayesNet);             
205             }
206             fTemp = fTemp * m_fDelta;
207         }
208 
209         copyParentSets(bayesNet, bestBayesNet);
210     } // buildStructure

212     /** CopyParentSets copies parent sets of source to dest BayesNet
213      * @param dest destination network
214      * @param source source network
215      */
216     void copyParentSets(BayesNet dest, BayesNet source) {
217         int nNodes = source.getNrOfNodes();
218         // clear parent set first
219         for (int iNode = 0; iNode < nNodes; iNode++) {
220             dest.getParentSet(iNode).copy(source.getParentSet(iNode));
221         }       
222     } // CopyParentSets

284     /**
285      * Returns an enumeration describing the available options.
286      *
287      * @return an enumeration of all the available options.
288      */
289     public Enumeration listOptions() {
290         Vector newVector = new Vector(3);
291 
292         newVector.addElement(new Option("\tStart temperature", "A", 1, "-A <float>"));
293         newVector.addElement(new Option("\tNumber of runs", "U", 1, "-U <integer>"));
294         newVector.addElement(new Option("\tDelta temperature", "D", 1, "-D <float>"));
295         newVector.addElement(new Option("\tRandom number seed", "R", 1, "-R <seed>"));
296 
297         Enumeration enu = super.listOptions();
298         while (enu.hasMoreElements()) {
299             newVector.addElement(enu.nextElement());
300         }
301         return newVector.elements();
302     }

304     /**
305      * Parses a given list of options. <p/>
306      *
307      <!-- options-start -->
308      * Valid options are: <p/>
309      * 
310      * <pre> -A &lt;float&gt;
311      *  Start temperature</pre>
312      * 
313      * <pre> -U &lt;integer&gt;
314      *  Number of runs</pre>
315      * 
316      * <pre> -D &lt;float&gt;
317      *  Delta temperature</pre>
318      * 
319      * <pre> -R &lt;seed&gt;
320      *  Random number seed</pre>
321      * 
322      * <pre> -mbc
323      *  Applies a Markov Blanket correction to the network structure, 
324      *  after a network structure is learned. This ensures that all 
325      *  nodes in the network are part of the Markov blanket of the 
326      *  classifier node.</pre>
327      * 
328      * <pre> -S [BAYES|MDL|ENTROPY|AIC|CROSS_CLASSIC|CROSS_BAYES]
329      *  Score type (BAYES, BDeu, MDL, ENTROPY and AIC)</pre>
330      * 
331      <!-- options-end -->
332      *
333      * @param options the list of options as an array of strings
334      * @throws Exception if an option is not supported
335      */
336     public void setOptions(String[] options) throws Exception {
337         String sTStart = Utils.getOption('A', options);
338         if (sTStart.length() != 0) {
339             setTStart(Double.parseDouble(sTStart));
340         }
341         String sRuns = Utils.getOption('U', options);
342         if (sRuns.length() != 0) {
343             setRuns(Integer.parseInt(sRuns));
344         }
345         String sDelta = Utils.getOption('D', options);
346         if (sDelta.length() != 0) {
347             setDelta(Double.parseDouble(sDelta));
348         }
349         String sSeed = Utils.getOption('R', options);
350         if (sSeed.length() != 0) {
351             setSeed(Integer.parseInt(sSeed));
352         }
353         super.setOptions(options);
354     }
355 
356     /**
357      * Gets the current settings of the search algorithm.
358      *
359      * @return an array of strings suitable for passing to setOptions
360      */
361     public String[] getOptions() {
362         String[] superOptions = super.getOptions();
363         String[] options = new String[8 + superOptions.length];
364         int current = 0;
365         options[current++] = "-A";
366         options[current++] = "" + getTStart();
367 
368         options[current++] = "-U";
369         options[current++] = "" + getRuns();
370 
371         options[current++] = "-D";
372         options[current++] = "" + getDelta();
373 
374         options[current++] = "-R";
375         options[current++] = "" + getSeed();
376 
377         // insert options from parent class
378         for (int iOption = 0; iOption < superOptions.length; iOption++) {
379             options[current++] = superOptions[iOption];
380         }
381 
382         // Fill up rest with empty strings, not nulls!
383         while (current < options.length) {
384             options[current++] = "";
385         }
386         return options;
387     }
