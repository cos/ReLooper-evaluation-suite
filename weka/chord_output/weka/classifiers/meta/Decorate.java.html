322 
323       String [] superOptions = super.getOptions();
324       String [] options = new String [superOptions.length + 4];
325       
326       int current = 0;
327       options[current++] = "-E"; options[current++] = "" + getDesiredSize();
328       options[current++] = "-R"; options[current++] = "" + getArtificialSize();
329       
330       System.arraycopy(superOptions, 0, options, current, 
331                        superOptions.length);
332       
333       current += superOptions.length;
334       while (current < options.length) {
335         options[current++] = "";
336       }
337       return options;

474     /**
475      * Build Decorate classifier
476      *
477      * @param data the training data to be used for generating the classifier
478      * @throws Exception if the classifier could not be built successfully
479      */
480     public void buildClassifier(Instances data) throws Exception {
481       if(m_Classifier == null) {
482         throw new Exception("A base classifier has not been specified!");
483       }
484       
485       // can classifier handle the data?
486       getCapabilities().testWithFail(data);
487       
488       // remove instances with missing class
489       data = new Instances(data);
490       data.deleteWithMissingClass();
491   
492     //initialize random number generator
493     if(m_Seed==-1) m_Random = new Random();
494     else m_Random = new Random(m_Seed);
495     
496     int i = 1;//current committee size
497     int numTrials = 1;//number of Decorate iterations 
498     Instances divData = new Instances(data);//local copy of data - diversity data
499     Instances artData = null;//artificial data
500 
501     //compute number of artficial instances to add at each iteration
502     int artSize = (int) (Math.abs(m_ArtSize)*divData.numInstances());
503     if(artSize==0) artSize=1;//atleast add one random example
504     computeStats(data);//Compute training data stats for creating artificial examples
505     
506     //initialize new committee
507     m_Committee = new Vector();
508     Classifier newClassifier = m_Classifier;
509     newClassifier.buildClassifier(divData);
510     m_Committee.add(newClassifier);
511     double eComm = computeError(divData);//compute ensemble error
512     if(m_Debug) System.out.println("Initialize:\tClassifier "+i+" added to ensemble. Ensemble error = "+eComm);
513     
514     //repeat till desired committee size is reached OR the max number of iterations is exceeded 
515     while(i<m_DesiredSize && numTrials<m_NumIterations){
516         //Generate artificial training examples
517         artData = generateArtificialData(artSize, data);
518         
519         //Label artificial examples
520         labelData(artData);
521         addInstances(divData, artData);//Add new artificial data
522         
523         //Build new classifier
524         Classifier tmp[] = AbstractClassifier.makeCopies(m_Classifier,1);
525         newClassifier = tmp[0]; 
526         newClassifier.buildClassifier(divData);
527         //Remove all the artificial data
528         removeInstances(divData, artSize);
529         
530         //Test if the new classifier should be added to the ensemble
531         m_Committee.add(newClassifier);//add new classifier to current committee
532         double currError = computeError(divData);
533         if(currError <= eComm){//adding the new member did not increase the error
534         i++;
535         eComm = currError;
536         if(m_Debug) System.out.println("Iteration: "+(1+numTrials)+"\tClassifier "+i+" added to ensemble. Ensemble error = "+eComm);
537         }else{//reject the current classifier because it increased the ensemble error 
538         m_Committee.removeElementAt(m_Committee.size()-1);//pop the last member
539         }
540         numTrials++;
541     }
542     }

544     /** 
545      * Compute and store statistics required for generating artificial data.
546      *
547      * @param data training instances
548      * @throws Exception if statistics could not be calculated successfully
549      */
550     protected void computeStats(Instances data) throws Exception{
551     int numAttributes = data.numAttributes();
552     m_AttributeStats = new Vector(numAttributes);//use to map attributes to their stats
553     
554     for(int j=0; j<numAttributes; j++){
555         if(data.attribute(j).isNominal()){
556         //Compute the probability of occurence of each distinct value 
557         int []nomCounts = (data.attributeStats(j)).nominalCounts;
558         double []counts = new double[nomCounts.length];
559         if(counts.length < 2) throw new Exception("Nominal attribute has less than two distinct values!"); 
560         //Perform Laplace smoothing
561         for(int i=0; i<counts.length; i++)
562             counts[i] = nomCounts[i] + 1;
563         Utils.normalize(counts);
564         double []stats = new double[counts.length - 1];
565         stats[0] = counts[0];
566         //Calculate cumulative probabilities
567         for(int i=1; i<stats.length; i++)
568             stats[i] = stats[i-1] + counts[i];
569         m_AttributeStats.add(j,stats);
570         }else if(data.attribute(j).isNumeric()){
571         //Get mean and standard deviation from the training data
572         double []stats = new double[2];
573         stats[0] = data.meanOrMode(j);
574         stats[1] = Math.sqrt(data.variance(j));
575         m_AttributeStats.add(j,stats);
576         }else System.err.println("Decorate can only handle numeric and nominal values.");
577     }
578     }

580     /**
581      * Generate artificial training examples.
582      * @param artSize size of examples set to create
583      * @param data training data
584      * @return the set of unlabeled artificial examples
585      */
586     protected Instances generateArtificialData(int artSize, Instances data){
587     int numAttributes = data.numAttributes();
588     Instances artData = new Instances(data, artSize);
589     double []att; 
590     Instance artInstance;
591     
592     for(int i=0; i<artSize; i++){
593         att = new double[numAttributes];
594         for(int j=0; j<numAttributes; j++){
595         if(data.attribute(j).isNominal()){
596             //Select nominal value based on the frequency of occurence in the training data  
597             double []stats = (double [])m_AttributeStats.get(j);
598             att[j] =  (double) selectIndexProbabilistically(stats);
599         }
600         else if(data.attribute(j).isNumeric()){
601             //Generate numeric value from the Guassian distribution 
602             //defined by the mean and std dev of the attribute
603             double []stats = (double [])m_AttributeStats.get(j);
604             att[j] = (m_Random.nextGaussian()*stats[1])+stats[0];
605         }else System.err.println("Decorate can only handle numeric and nominal values.");
606         }
607         artInstance = new DenseInstance(1.0, att);
608         artData.add(artInstance);
609     }
610     return artData;
611     }

614     /** 
615      * Labels the artificially generated data.
616      *
617      * @param artData the artificially generated instances
618      * @throws Exception if instances cannot be labeled successfully 
619      */
620     protected void labelData(Instances artData) throws Exception {
621     Instance curr;
622     double []probs;
623     
624     for(int i=0; i<artData.numInstances(); i++){
625         curr = artData.instance(i);
626         //compute the class membership probs predicted by the current ensemble 
627         probs = distributionForInstance(curr);
628         //select class label inversely proportional to the ensemble predictions
629         curr.setClassValue(inverseLabel(probs));
630     }   
631     }

634     /** 
635      * Select class label such that the probability of selection is
636      * inversely proportional to the ensemble's predictions.
637      *
638      * @param probs class membership probabilities of instance
639      * @return index of class label selected
640      * @throws Exception if instances cannot be labeled successfully 
641      */
642     protected int inverseLabel(double []probs) throws Exception{
643     double []invProbs = new double[probs.length];
644     //Produce probability distribution inversely proportional to the given
645     for(int i=0; i<probs.length; i++){
646         if(probs[i]==0){
647         invProbs[i] = Double.MAX_VALUE/probs.length; 
648         //Account for probability values of 0 - to avoid divide-by-zero errors
649         //Divide by probs.length to make sure normalizing works properly
650         }else{
651         invProbs[i] = 1.0 / probs[i];
652         }
653     }
654     Utils.normalize(invProbs);
655     double []cdf = new double[invProbs.length];
656     //Compute cumulative probabilities 
657     cdf[0] = invProbs[0];
658     for(int i=1; i<invProbs.length; i++){
659         cdf[i] = invProbs[i]+cdf[i-1];
660     }
661     
662     if(Double.isNaN(cdf[invProbs.length-1]))
663         System.err.println("Cumulative class membership probability is NaN!"); 
664     return selectIndexProbabilistically(cdf);
665     }
666     
667     /** 
668      * Given cumulative probabilities select a nominal attribute value index 
669      *
670      * @param cdf array of cumulative probabilities
671      * @return index of attribute selected based on the probability distribution 
672      */
673     protected int selectIndexProbabilistically(double []cdf){
674     double rnd = m_Random.nextDouble();
675     int index = 0;
676     while(index < cdf.length && rnd > cdf[index]){
677         index++;
678     }
679     return index;
680     }

682     /**
683      * Removes a specified number of instances from the given set of instances.
684      *
685      * @param data given instances
686      * @param numRemove number of instances to delete from the given instances
687      */
688     protected void removeInstances(Instances data, int numRemove){
689     int num = data.numInstances();
690     for(int i=num - 1; i>num - 1 - numRemove;i--){
691         data.delete(i);
692     }
693     }

706     /** 
707      * Computes the error in classification on the given data.
708      *
709      * @param data the instances to be classified
710      * @return classification error
711      * @throws Exception if error can not be computed successfully
712      */
713     protected double computeError(Instances data) throws Exception {
714     double error = 0.0;
715     int numInstances = data.numInstances();
716     Instance curr;
717     
718     for(int i=0; i<numInstances; i++){
719         curr = data.instance(i);
720         //Check if the instance has been misclassified
721         if(curr.classValue() != ((int) classifyInstance(curr))) error++;
722     }
723     return (error/numInstances);
724     }

726   /**
727    * Calculates the class membership probabilities for the given test instance.
728    *
729    * @param instance the instance to be classified
730    * @return predicted class probability distribution
731    * @throws Exception if distribution can't be computed successfully
732    */
733   public double[] distributionForInstance(Instance instance) throws Exception {
734       if (instance.classAttribute().isNumeric()) {
735       throw new UnsupportedClassTypeException("Decorate can't handle a numeric class!");
736       }
737       double [] sums = new double [instance.numClasses()], newProbs; 
738       Classifier curr;
739       
740       for (int i = 0; i < m_Committee.size(); i++) {
741       curr = (Classifier) m_Committee.get(i);
742       newProbs = curr.distributionForInstance(instance);
743       for (int j = 0; j < newProbs.length; j++)
744         sums[j] += newProbs[j];
745       }
746       if (Utils.eq(Utils.sum(sums), 0)) {
747       return sums;
748       } else {
749       Utils.normalize(sums);
750       return sums;
751       }
752   }
