23  package weka.classifiers.misc.monotone;
24  
25  import weka.core.RevisionHandler;
26  import weka.core.RevisionUtils;
27  
28  import java.util.Arrays;
29  import java.util.Comparator;
30  
31  /** 
32   * Class for doing multidimensional sorting, using an array of
33   * <code> Comparator. </code>
34   * The goal is to sort an array topologically.  If <code> o1 </code>
35   * and <code> o2 </code> are two objects of the array <code> a, </code>
36   * and for all valid indices <code> i </code> in the array <code> c </code>
37   * if holds that <code> c[i].compare(o1,o2) &lt; 0 </code> then
38   * <code> o1 </code> comes before <code> o2 </code> in the sorted array.
39   * <p>
40   * A typical is the sorting of vectors in an n-dimensional space,
41   * where the ordering is determined by the product ordering.
42   * </p>
43   * <p>
44   * This implementation is part of the master's thesis: "Studie
45   * en implementatie van instantie-gebaseerde algoritmen voor gesuperviseerd
46   * rangschikken", Stijn Lievens, Ghent University, 2004. 
47   * </p>
48   * 
49   * @author Stijn Lievens (stijn.lievens@ugent.be)
50   * @version $Revision: 5922 $
51   */
52  public class MultiDimensionalSort
53    implements RevisionHandler {
54  
55    /** 
56     * Sort an array using different comparators.
57     *
58     * @param a the array to be sorted.  The sorted array is returned 
59     * in the array <code> a </code> itself.
60     * @param c an array holding the different comparators
61     */
62    public static void multiDimensionalSort (Object[] a, Comparator[] c) {
63      multiDimensionalSort(a, 0, a.length, c);
64    }
65  
66    /** 
67     * Sort part of an array using different comparators.
68     * 
69     * @param a the array to be sorted, the indicated part of the array will
70     * be replaced by the sorted elements
71     * @param fromIndex index of the first element to be sorted (inclusive)
72     * @param toIndex index of the last element to be sorted (exclusive)
73     * @param c array holding the different comparators
74     * @throws IllegalArgumentException if <code> fromIndex &gt; toIndex </code>
75     */
76    public static void multiDimensionalSort(Object[] a, int fromIndex,
77        int toIndex, Comparator[] c) 
78      throws IllegalArgumentException {
79      
80      if (fromIndex > toIndex) {
81        throw new IllegalArgumentException
82        ("Illegal range: fromIndex can be at most toIndex");
83      }
84      multiDimensionalSort(a, fromIndex, toIndex, c, 0);
85    }
86  
87    /**
88     * Do the actual sorting in a recursive way.
89     * 
90     * @param a the array to be sorted, the indicated part of the array will
91     * be replaced by the sorted elements
92     * @param fromIndex index of the first element to be sorted (inclusive)
93     * @param toIndex index of the last element to be sorted (exclusive)
94     * @param c array holding the different comparators
95     * @param depth the index of the comparator to use
96     */
97    private static void multiDimensionalSort(Object[] a, int fromIndex,
98        int toIndex, Comparator[] c, int depth) {
99  
100     if (depth == c.length) {
101       return; // maximum depth reached
102     }
103 
104     Comparator comp = c[depth]; // comparator to use
105     Arrays.sort(a, fromIndex, toIndex, comp); // sort part of the array 
106 
107     // look for breakpoints in the array and sort recursively
108     int mark = fromIndex;
109     for (int i = fromIndex + 1; i < toIndex; i++) {
110       if (comp.compare(a[i - 1], a[i]) != 0) {
111     // a[i] is different from a[i-1], breakpoint detected
112     multiDimensionalSort(a, mark, i, c, depth + 1);
113     mark = i;
114       }
115     }
116     // sort the last part 
117     multiDimensionalSort(a, mark, toIndex, c, depth + 1);
118   }
119   
120   /**
121    * Returns the revision string.
122    * 
123    * @return        the revision
124    */
125   public String getRevision() {
126     return RevisionUtils.extract("$Revision: 5922 $");
127   }
128 }
