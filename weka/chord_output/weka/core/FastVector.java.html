23  package weka.core;
24  
25  import java.util.Enumeration;
26  import java.util.Collection;
27  import java.util.ArrayList;
28  
29  /**
30   * Simple extension of ArrayList. Exists for legacy reasons.
31   *
32   * @author Eibe Frank (eibe@cs.waikato.ac.nz)
33   * @version $Revision: 5953 $
34   */
35  @Deprecated 
36  public class FastVector<E> extends ArrayList<E> implements Copyable, RevisionHandler {
37  
38    /** for serialization */
39    private static final long serialVersionUID = -2173635135622930169L;
40  
41    /**
42     * Constructs an empty vector with initial
43     * capacity zero.
44     */
45    public FastVector() {
46      super();
47    }
48  
49    /**
50     * Constructs a vector with the given capacity.
51     *
52     * @param capacity the vector's initial capacity
53     */
54    public FastVector(int capacity) {
55      super(capacity);
56    }
57  
58    /**
59     * Adds an element to this vector. Increases its
60     * capacity if its not large enough.
61     *
62     * @param element the element to add
63     */
64    public final void addElement(E element) {
65      add(element);
66    }
67  
68    /**
69     * Produces a shallow copy of this vector.
70     *
71     * @return the new vector
72     */
73    public final FastVector<E> copy() {
74      return Utils.cast(clone());
75    }
76  
77    /**
78     * Clones the vector and shallow copies all its elements.
79     * The elements have to implement the Copyable interface.
80     * 
81     * @return the new vector
82     */
83    public final FastVector<E> copyElements() {
84  
85      FastVector<E> copy = copy();
86      for (int i = 0; i < size(); i++) {
87        copy.set(i, Utils.<E>cast(((Copyable)get(i)).copy()));
88      }
89      return copy;
90    }
91  
92    /**
93     * Returns the element at the given position.
94     *
95     * @param index the element's index
96     * @return the element with the given index
97     */
98    public final E elementAt(int index) {
99      return get(index);
100   }
101 
102   /**
103    * Returns an enumeration of this vector.
104    *
105    * @return an enumeration of this vector
106    */
107   public final Enumeration elements() {
108     return new WekaEnumeration(this);
109   }
110 
111   /**
112    * Returns an enumeration of this vector, skipping the
113    * element with the given index.
114    *
115    * @param index the element to skip
116    * @return an enumeration of this vector
117    */
118   public final Enumeration elements(int index) {
119     return new WekaEnumeration(this, index);
120   }
121 
122   /**
123    * Returns the first element of the vector.
124    *
125    * @return the first element of the vector
126    */
127   public final  E firstElement() {
128     return get(0);
129   }
130 
131   /**
132    * Inserts an element at the given position.
133    *
134    * @param element the element to be inserted
135    * @param index the element's index
136    */
137   public final void insertElementAt(E element, int index) {
138     add(index, element);
139   }
140 
141   /**
142    * Returns the last element of the vector.
143    *
144    * @return the last element of the vector
145    */
146   public final E lastElement() {
147     return get(size() - 1);
148   }
149 
150   /**
151    * Deletes an element from this vector.
152    *
153    * @param index the index of the element to be deleted
154    */
155   public final void removeElementAt(int index) {
156     remove(index);
157   }
158 
159   /**
160    * Removes all components from this vector and sets its 
161    * size to zero. 
162    */
163   public final void removeAllElements() {
164     clear();
165   }
166 
167   /**
168    * Appends all elements of the supplied vector to this vector.
169    *
170    * @param toAppend the FastVector containing elements to append.
171    */
172   public final void appendElements(Collection<? extends E> toAppend) {
173     addAll(toAppend);
174   }
175 
176   /**
177    * Sets the vector's capacity to the given value.
178    *
179    * @param capacity the new capacity
180    */
181   public final void setCapacity(int capacity) {
182     ensureCapacity(capacity);
183   }
184 
185   /**
186    * Sets the element at the given index.
187    *
188    * @param element the element to be put into the vector
189    * @param index the index at which the element is to be placed
190    */
191   public final void setElementAt(E element, int index) {
192     set(index, element);
193   }
194 
195   /**
196    * Swaps two elements in the vector.
197    *
198    * @param first index of the first element
199    * @param second index of the second element
200    */
201   public final void swap(int first, int second) {
202     
203     E in = get(first);
204     set(first, get(second));
205     set(second, in);
206   }
207   
208   /**
209    * Returns the revision string.
210    * 
211    * @return        the revision
212    */
213   public String getRevision() {
214     return RevisionUtils.extract("$Revision: 5953 $");
215   }
216 }
