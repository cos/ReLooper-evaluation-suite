34  public class FlexibleDecimalFormat
35    extends DecimalFormat
36    implements RevisionHandler {
37  
38    /** for serialization */
39    private static final long serialVersionUID = 110912192794064140L;
40  
41    private DecimalFormat nf = null;
42    private int      digits = 7;
43    private boolean  exp = false; 
44    private int      intDigits = 1;
45    private int      decimalDigits = 0;
46    private int      expIntDigits = 1;          // ??? 
47    private int      expDecimalDigits = 0;      // ???
48    private int      power = 2;
49    private boolean  trailing = false;
50    private boolean  grouping = false;
51    private boolean  sign = false;
52      
53    public FlexibleDecimalFormat ( ) {
54      this( 5 );
55    }
56  
57    public FlexibleDecimalFormat ( int digits ) {
58      if( digits < 1 ) 
59        throw new IllegalArgumentException("digits < 1");
60      this.digits = digits;
61      intDigits = 1;
62    }
63  
64    public FlexibleDecimalFormat ( int digits, boolean trailing ) {
65      this( digits );
66      this.trailing = trailing;
67    }
68  
69    public FlexibleDecimalFormat ( int digits, boolean exp, boolean trailing, 
70                   boolean grouping ) {
71      this.trailing = trailing;
72      this.exp = exp;
73      this.digits = digits;
74      this.grouping = grouping;
75      if( exp ) {
76        this.intDigits = 1;
77        this.decimalDigits = digits - intDigits;
78      }
79      else {
80        this.decimalDigits = decimalDigits;
81        this.intDigits = Math.max( 1, digits - decimalDigits );
82      }
83    }
84  
85    public FlexibleDecimalFormat ( double d ) {
86      newFormat( d );
87    }
88  
89    private void newFormat ( double d ) {
90      if( needExponentialFormat( d ) ) {
91        exp = true;
92        intDigits = 1;
93        expDecimalDigits = decimalDigits( d, true );
94        if( d < 0) sign = true;
95        else sign = false;
96      }
97      else {
98        exp = false;
99        intDigits = Math.max(1, intDigits( d ));
100       decimalDigits = decimalDigits( d, false );
101       if( d < 0.0 ) sign = true;
102       else sign = false;
103     }
104   }
105 
106   public void update ( double d ) {
107     if( Math.abs( intDigits(d) -1 ) > 99 ) power = 3;
108     expIntDigits = 1;
109     expDecimalDigits = Math.max( expDecimalDigits, 
110                  decimalDigits( d, true ));
111     if( d < 0) sign = true;
112     if( needExponentialFormat( d ) || exp ) {
113       exp = true;
114     }
115     else {
116       intDigits = Math.max(intDigits, intDigits( d ));
117       decimalDigits = Math.max(decimalDigits, decimalDigits( d, false ));
118       if( d < 0) sign = true;
119     }
120   }
121 
122   private static int intDigits ( double d ) {
123     return (int) Math.floor( Math.log( Math.abs( d * (1 + 1e-14) ) ) / 
124                  Math.log ( 10 ) ) + 1;
125   }
126     
127   private int decimalDigits ( double d, boolean expo ) {
128     if( d == 0.0 ) return 0;
129     d = Math.abs( d );
130     int e = intDigits( d );
131     if( expo ) {
132       d /= Math.pow(10, e-1);
133       e = 1;
134     }
135     if( e >= digits ) return 0;
136     int iD = Math.max(1, e);  
137     int dD = digits - e;
138     if( !trailing && dD > 0 ) {  // to get rid of trailing zeros
139       FloatingPointFormat f = new 
140     FloatingPointFormat( iD + 1 + dD, dD, true);
141       String dString = f.format( d );
142       while( dD > 0 ) {
143     if( dString.charAt(iD+1+dD-1) == '0' ) {
144       dD--;
145     }
146     else break;
147       }
148     }
149     return dD;
150   }
151     
152   public boolean  needExponentialFormat ( double d ) {
153     if( d == 0.0 ) return false;
154     int e = intDigits( d );
155     if( e > digits + 5 || e < -3 ) return true;
156     else return false;
157   }
158     
159   public void grouping ( boolean grouping ) {
160     this.grouping = grouping;
161   }
162     
163   private static void println ( Object obj ){
164     System.out.println( obj );
165   }
166 
167   private void setFormat ( ) {
168     int dot = 1;
169     if( decimalDigits == 0) dot = 0;
170     if( exp ) 
171       nf = new ExponentialFormat( 1 + expDecimalDigits, power, sign, 
172                   grouping || trailing );
173     else { 
174       int s = sign ? 1 : 0;
175       nf = new FloatingPointFormat( s +intDigits +dot +decimalDigits, 
176                     decimalDigits, grouping || trailing);
177     }
178   }
179     
180   private void setFormat ( double d ) {
181     newFormat( d );
182     setFormat();
183   }
184 
185   public StringBuffer format (double number, StringBuffer toAppendTo,
186                   FieldPosition pos) {
187     if( grouping ) {
188       if( nf == null ) {
189     setFormat();
190       }
191     }
192     else setFormat( number );
193     
194     return toAppendTo.append( nf.format(number) );
195   }
196 
197   public int width () {
198     
199     if( !trailing && !grouping ) 
200       throw new RuntimeException( "flexible width" );
201     
202     return format(0.).length();
203   }
204     
205   public StringBuffer formatString ( String str ) {
206     int w = width();
207     int h = ( w - str.length() ) / 2;
208     StringBuffer text = new StringBuffer();
209     for(int i = 0; i < h; i++ ){
210       text.append( ' ' );
211     }
212     text.append( str );
213     for(int i = 0; i < w - h - str.length(); i++ ){
214       text.append( ' ' );
215     }
216     return text;
217   }

225     return RevisionUtils.extract("$Revision: 5953 $");
226   }
