17  import weka.core.RevisionHandler;
18  import weka.core.RevisionUtils;

20  import java.io.Serializable;

22  /** 
23   * Cholesky Decomposition.
24   * <P>
25   * For a symmetric, positive definite matrix A, the Cholesky decomposition is
26   * an lower triangular matrix L so that A = L*L'.
27   * <P>
28   * If the matrix is not symmetric or positive definite, the constructor
29   * returns a partial decomposition and sets an internal flag that may
30   * be queried by the isSPD() method.
31   * <p/>
32   * Adapted from the <a href="http://math.nist.gov/javanumerics/jama/" target="_blank">JAMA</a> package.
33   *
34   * @author The Mathworks and NIST 
35   * @author Fracpete (fracpete at waikato dot ac dot nz)
36   * @version $Revision: 5953 $
37   */
38  public class CholeskyDecomposition 
39    implements Serializable, RevisionHandler {
40  
41    /** for serialization */
42    private static final long serialVersionUID = -8739775942782694701L;
43  
44    /** 
45     * Array for internal storage of decomposition.
46     * @serial internal array storage.
47     */
48    private double[][] L;
49  
50    /** 
51     * Row and column dimension (square matrix).
52     * @serial matrix dimension.
53     */
54    private int n;
55  
56    /** 
57     * Symmetric and positive definite flag.
58     * @serial is symmetric and positive definite flag.
59     */
60    private boolean isspd;
61  
62    /** 
63     * Cholesky algorithm for symmetric and positive definite matrix.
64     *
65     * @param  Arg   Square, symmetric matrix.
66     */
67    public CholeskyDecomposition(Matrix Arg) {
68      // Initialize.
69      double[][] A = Arg.getArray();
70      n = Arg.getRowDimension();
71      L = new double[n][n];
72      isspd = (Arg.getColumnDimension() == n);
73      // Main loop.
74      for (int j = 0; j < n; j++) {
75        double[] Lrowj = L[j];
76        double d = 0.0;
77        for (int k = 0; k < j; k++) {
78          double[] Lrowk = L[k];
79          double s = 0.0;
80          for (int i = 0; i < k; i++) {
81            s += Lrowk[i]*Lrowj[i];
82          }
83          Lrowj[k] = s = (A[j][k] - s)/L[k][k];
84          d = d + s*s;
85          isspd = isspd & (A[k][j] == A[j][k]); 
86        }
87        d = A[j][j] - d;
88        isspd = isspd & (d > 0.0);
89        L[j][j] = Math.sqrt(Math.max(d,0.0));
90        for (int k = j+1; k < n; k++) {
91          L[j][k] = 0.0;
92        }
93      }
94    }
95  
96    /** 
97     * Is the matrix symmetric and positive definite?
98     * @return     true if A is symmetric and positive definite.
99     */
100   public boolean isSPD() {
101     return isspd;
102   }
103 
104   /** 
105    * Return triangular factor.
106    * @return     L
107    */
108   public Matrix getL() {
109     return new Matrix(L,n,n);
110   }
111 
112   /** 
113    * Solve A*X = B
114    * @param  B   A Matrix with as many rows as A and any number of columns.
115    * @return     X so that L*L'*X = B
116    * @exception  IllegalArgumentException  Matrix row dimensions must agree.
117    * @exception  RuntimeException  Matrix is not symmetric positive definite.
118    */
119   public Matrix solve(Matrix B) {
120     if (B.getRowDimension() != n) {
121       throw new IllegalArgumentException("Matrix row dimensions must agree.");
122     }
123     if (!isspd) {
124       throw new RuntimeException("Matrix is not symmetric positive definite.");
125     }
126 
127     // Copy right hand side.
128     double[][] X = B.getArrayCopy();
129     int nx = B.getColumnDimension();
130 
131     // Solve L*Y = B;
132     for (int k = 0; k < n; k++) {
133       for (int j = 0; j < nx; j++) {
134     for (int i = 0; i < k ; i++) {
135       X[k][j] -= X[i][j]*L[k][i];
136     }
137     X[k][j] /= L[k][k];
138       }
139     }
140 
141     // Solve L'*X = Y;
142     for (int k = n-1; k >= 0; k--) {
143       for (int j = 0; j < nx; j++) {
144     for (int i = k+1; i < n ; i++) {
145       X[k][j] -= X[i][j]*L[i][k];
146     }
147     X[k][j] /= L[k][k];
148       }
149     }
150 
151     return new Matrix(X,n,nx);
152   }

154   /**
155    * Returns the revision string.
156    * 
157    * @return        the revision
158    */
159   public String getRevision() {
160     return RevisionUtils.extract("$Revision: 5953 $");
161   }
162 }
