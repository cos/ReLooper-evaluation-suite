178   /** Sorts the array in place with index changed 
179    *  @param xi   first index
180    *  @param xj   last index
181    *  @param index array that stores all indices
182    */
183   public void  sortWithIndex( int xi, int xj, IntVector index ) {
184     if( xi < xj ) { 
185       double x, f, k;
186       int xm = (int) (xi + xj) / 2; // median index
187       x = Math.min( V[xi],             // median of three
188             Math.max( V[xm], V[xj])); 
189       int i = xi;
190       int j = xj;
191       while( i < j ) {
192     while( V[i] < x && i < xj ) i++;
193     while( V[j] > x && j > xi ) j--;
194     if( i <= j ){
195       swap(i, j);
196       index.swap(i, j);
197       i++;
198       j--;
199     }
200       }
201       sortWithIndex(xi, j, index);
202       sortWithIndex(i, xj, index);
203     }
204   }

231   /** Sets the capacity of the vector
232    *  @param n the capacity.  
233    */
234   public void  setCapacity ( int n ) {
235     if( n == capacity() ) return;
236     double [] oldV = V;
237     int m = Math.min( n, size() );
238     V = new double[ n ];
239     setSize( m );
240     set(0, m-1, oldV, 0);
241   }

315    * Returns the inner product of two DoubleVectors

319   public double  innerProduct(DoubleVector v) {
320     if(size() != v.size()) 
321       throw new IllegalArgumentException("sizes unmatch");
322     double p = 0;
323     for (int i = 0; i < size(); i++) {
324       p += V[i] * v.V[i];

630   /**
631    *  Returns the maximum value of all elements
632    */
633   public double max () {
634     if( size() < 1 ) throw new IllegalArgumentException("zero size");
635     double ma = V[0];
636     if( size() < 2 ) return ma;
637     for( int i = 1; i < size(); i++ ) {
638       if( V[i] > ma ) ma = V[i];
639     }
640     return ma;
641   }

644   /**
645    *  Applies a method to the vector
646    *  @param className the class name
647    *  @param method the method
648    */
649   public DoubleVector map( String className, String method ) {
650     try {
651       Class<?> c = Class.forName( className );
652       Class [] cs = new Class[1]; 
653       cs[ 0 ] = Double.TYPE;
654       Method m = c.getMethod( method, cs );
655       
656       DoubleVector w = new DoubleVector( size() );
657       Object [] obj = new Object[1];
658       for( int i = 0; i < size(); i++ ) {
659     obj[0] = new Double( V[i] );
660     w.set( i, Double.parseDouble(m.invoke( null, obj ).toString()) ); 
661       }
662       return w;
663     }
664     catch ( Exception e ) {
665       e.printStackTrace();
666       System.exit(1);
667     }
668     return null;
669   }

700   /** Convert the DoubleVecor to a string
701    *  @param digits the number of digits after decimal point
702    *  @param trailing true if trailing zeros are to be shown
703    */ 
704   public String  toString( int digits, boolean trailing ) {
705     if( isEmpty() ) return "null vector";
706 
707     StringBuffer text = new StringBuffer();
708     FlexibleDecimalFormat nf = new FlexibleDecimalFormat( digits, 
709                               trailing );
710     nf.grouping( true );
711     for( int i = 0; i < size(); i ++ ) nf.update( V[i] );
712     int count = 0;
713     int width = 80;
714     String number;
715     for( int i = 0; i < size(); i++ ) {
716       number = nf.format(V[i]);
717       count += 1 + number.length();
718       if( count > width-1 ) { 
719     text.append('\n'); 
720     count = 1 + number.length();
721       }
722       text.append( " " + number );
723     }
724     
725     return text.toString();
726   }
