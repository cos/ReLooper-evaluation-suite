50    /** 
51     * sqrt(a^2 + b^2) without under/overflow. 
52     */
53    public static double hypot(double a, double b) {
54      double r;
55      if (Math.abs(a) > Math.abs(b)) {
56        r = b/a;
57        r = Math.abs(a)*Math.sqrt(1+r*r);
58      } else if (b != 0) {
59        r = a/b;
60        r = Math.abs(b)*Math.sqrt(1+r*r);
61      } else {
62        r = 0.0;
63      }
64      return r;
65    }

88    /** 
89     *  Returns the cumulative probability of a normal distribution.
90     *  @param x the quantile
91     *  @param mean the mean of the normal distribution
92     *  @param sd the standard deviation of the normal distribution.
93     */
94    public static double  pnorm( double x, double mean, double sd ) 
95    {
96      if( sd <= 0.0 )
97        throw new IllegalArgumentException("standard deviation <= 0.0");
98      return pnorm( (x - mean) / sd );
99    }

101   /** 
102    *  Returns the cumulative probability of a set of normal distributions
103    *  with different means.
104    *  @param x the vector of quantiles
105    *  @param mean the means of the normal distributions
106    *  @param sd the standard deviation of the normal distribution.
107    *  @return the cumulative probability */
108   public static DoubleVector  pnorm( double x, DoubleVector mean, 
109                                      double sd ) 
110   {
111     DoubleVector p = new DoubleVector( mean.size() );
112         
113     for( int i = 0; i < mean.size(); i++ ) {
114       p.set( i, pnorm(x, mean.get(i), sd) );
115     }
116     return p;
117   }

119   /** Returns the density of the standard normal.
120    *  @param x the quantile
121    *  @return the density
122    */
123   public static double  dnorm( double x ) 
124   {
125     return Math.exp( - x * x / 2. ) * PSI;
126   }

128   /** Returns the density value of a standard normal.
129    *  @param x the quantile
130    *  @param mean the mean of the normal distribution
131    *  @param sd the standard deviation of the normal distribution.
132    *  @return the density */
133   public static double  dnorm( double x, double mean, double sd ) 
134   {
135     if( sd <= 0.0 )
136       throw new IllegalArgumentException("standard deviation <= 0.0");
137     return dnorm( (x - mean) / sd );
138   }

157   /** Returns the log-density of the standard normal.
158    *  @param x the quantile
159    *  @return the density
160    */
161   public static double  dnormLog( double x ) 
162   {
163     return logPSI - x * x / 2.;
164   }

166   /** Returns the log-density value of a standard normal.
167    *  @param x the quantile
168    *  @param mean the mean of the normal distribution
169    *  @param sd the standard deviation of the normal distribution.
170    *  @return the density */
171   public static double  dnormLog( double x, double mean, double sd ) {
172     if( sd <= 0.0 )
173       throw new IllegalArgumentException("standard deviation <= 0.0");
174     return - Math.log(sd) + dnormLog( (x - mean) / sd );
175   }

182    * @return the density */

184                                         double sd )

186     DoubleVector denLog = new DoubleVector( mean.size() );
187         
188     for( int i = 0; i < mean.size(); i++ ) {
189       denLog.set( i, dnormLog(x, mean.get(i), sd) );

194   /** 
195    *  Generates a sample of a normal distribution.
196    *  @param n the size of the sample
197    *  @param mean the mean of the normal distribution
198    *  @param sd the standard deviation of the normal distribution.
199    *  @param random the random stream
200    *  @return the sample
201    */
202   public static DoubleVector rnorm( int n, double mean, double sd, 
203                                     Random random ) 
204   {
205     if( sd < 0.0)
206       throw new IllegalArgumentException("standard deviation < 0.0");
207         
208     if( sd == 0.0 ) return new DoubleVector( n, mean );
209     DoubleVector v = new DoubleVector( n );
210     for( int i = 0; i < n; i++ ) 
211       v.set( i, (random.nextGaussian() + mean) / sd );
212     return v;
213   }

217   /** Returns the cumulative probability of the Chi-squared distribution
218    *  @param x the quantile
219    */
220   public static double  pchisq( double x ) 
221   {
222     double xh = Math.sqrt( x );
223     return pnorm( xh ) - pnorm( -xh );
224   }

226   /** Returns the cumulative probability of the noncentral Chi-squared
227    *  distribution.
228    *  @param x the quantile
229    *  @param ncp the noncentral parameter */
230   public static double  pchisq( double x, double ncp ) 
231   {
232     double mean = Math.sqrt( ncp );
233     double xh = Math.sqrt( x );
234     return pnorm( xh - mean ) - pnorm( -xh - mean );
235   }

237   /** Returns the cumulative probability of a set of noncentral Chi-squared
238    *  distributions.
239    *  @param x the quantile
240    *  @param ncp the noncentral parameters */
241   public static DoubleVector  pchisq( double x, DoubleVector ncp )
242   {
243     int n = ncp.size();
244     DoubleVector p = new DoubleVector( n );
245     double mean;
246     double xh = Math.sqrt( x );
247         
248     for( int i = 0; i < n; i++ ) {
249       mean = Math.sqrt( ncp.get(i) );
250       p.set( i, pnorm( xh - mean ) - pnorm( -xh - mean ) );
251     }
252     return p;
253   }

255   /** Returns the density of the Chi-squared distribution.
256    *  @param x the quantile
257    *  @return the density
258    */
259   public static double  dchisq( double x ) 
260   {
261     if( x == 0.0 ) return Double.POSITIVE_INFINITY;
262     double xh = Math.sqrt( x );
263     return dnorm( xh ) / xh;
264   }
265     
266   /** Returns the density of the noncentral Chi-squared distribution.
267    *  @param x the quantile
268    *  @param ncp the noncentral parameter
269    */
270   public static double  dchisq( double x, double ncp ) 
271   {
272     if( ncp == 0.0 ) return dchisq( x );
273     double xh = Math.sqrt( x );
274     double mean = Math.sqrt( ncp );
275     return (dnorm( xh - mean ) + dnorm( -xh - mean)) / (2 * xh);
276   }
277     
278   /** Returns the density of the noncentral Chi-squared distribution.
279    *  @param x the quantile
280    *  @param ncp the noncentral parameters 
281    */
282   public static DoubleVector  dchisq( double x, DoubleVector ncp )
283   {
284     int n = ncp.size();
285     DoubleVector d = new DoubleVector( n );
286     double xh = Math.sqrt( x );
287     double mean;
288     for( int i = 0; i < n; i++ ) {
289       mean = Math.sqrt( ncp.get(i) );
290       if( ncp.get(i) == 0.0 ) d.set( i, dchisq( x ) );
291       else d.set( i, (dnorm( xh - mean ) + dnorm( -xh - mean)) / 
292                   (2 * xh) );
293     }
294     return d;
295   }

297   /** Returns the log-density of the noncentral Chi-square distribution.
298    *  @param x the quantile
299    *  @return the density
300    */
301   public static double  dchisqLog( double x ) 
302   {
303     if( x == 0.0) return Double.POSITIVE_INFINITY;
304     double xh = Math.sqrt( x );
305     return dnormLog( xh ) - Math.log( xh );
306   }

308   /** Returns the log-density value of a noncentral Chi-square distribution.
309    *  @param x the quantile
310    *  @param ncp the noncentral parameter
311    *  @return the density */
312   public static double  dchisqLog( double x, double ncp ) {
313     if( ncp == 0.0 ) return dchisqLog( x );
314     double xh = Math.sqrt( x );
315     double mean = Math.sqrt( ncp );
316     return Math.log( dnorm( xh - mean ) + dnorm( -xh - mean) ) - 
317     Math.log(2 * xh);
318   }

320   /** Returns the log-density of a set of noncentral Chi-squared
321    *  distributions.
322    *  @param x the quantile
323    *  @param ncp the noncentral parameters */
324   public static DoubleVector  dchisqLog( double x, DoubleVector ncp )
325   {
326     DoubleVector dLog = new DoubleVector( ncp.size() );
327     double xh = Math.sqrt( x );
328     double mean;
329         
330     for( int i = 0; i < ncp.size(); i++ ) {
331       mean = Math.sqrt( ncp.get(i) );
332       if( ncp.get(i) == 0.0 ) dLog.set( i, dchisqLog( x ) );
333       else dLog.set( i, Math.log( dnorm( xh - mean ) + dnorm( -xh - mean) ) - 
334                      Math.log(2 * xh) );
335     }
336     return dLog;
337   }

339   /** 
340    *  Generates a sample of a Chi-square distribution.
341    *  @param n the size of the sample
342    *  @param ncp the noncentral parameter
343    *  @param random the random stream
344    *  @return the sample
345    */
346   public static DoubleVector  rchisq( int n, double ncp, Random random ) 
347   {
348     DoubleVector v = new DoubleVector( n );
349     double mean = Math.sqrt( ncp );
350     double x;
351     for( int i = 0; i < n; i++ ) {
352       x = random.nextGaussian() + mean;
353       v.set( i, x * x );
354     }
355     return v;
356   }
