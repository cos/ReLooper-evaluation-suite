22  /** 
23   * Singular Value Decomposition.
24   * <P>
25   * For an m-by-n matrix A with m &gt;= n, the singular value decomposition is an
26   * m-by-n orthogonal matrix U, an n-by-n diagonal matrix S, and an n-by-n
27   * orthogonal matrix V so that A = U*S*V'.
28   * <P>
29   * The singular values, sigma[k] = S[k][k], are ordered so that sigma[0] &gt;=
30   * sigma[1] &gt;= ... &gt;= sigma[n-1].
31   * <P>
32   * The singular value decompostion always exists, so the constructor will never
33   * fail.  The matrix condition number and the effective numerical rank can be
34   * computed from this decomposition.
35   * <p/>
36   * Adapted from the <a href="http://math.nist.gov/javanumerics/jama/" target="_blank">JAMA</a> package.
37   *
38   * @author The Mathworks and NIST 
39   * @author Fracpete (fracpete at waikato dot ac dot nz)
40   * @version $Revision: 5953 $
41   */
42  public class SingularValueDecomposition 
43    implements Serializable, RevisionHandler {
44  
45    /** for serialization */
46    private static final long serialVersionUID = -8738089610999867951L;
47    
48    /** 
49     * Arrays for internal storage of U and V.
50     * @serial internal storage of U.
51     * @serial internal storage of V.
52     */
53    private double[][] U, V;
54  
55    /** 
56     * Array for internal storage of singular values.
57     * @serial internal storage of singular values.
58     */
59    private double[] s;
60  
61    /** 
62     * Row and column dimensions.
63     * @serial row dimension.
64     * @serial column dimension.
65     */
66    private int m, n;
67  
68    /** 
69     * Construct the singular value decomposition
70     * @param Arg    Rectangular matrix
71     */
72    public SingularValueDecomposition(Matrix Arg) {
73  
74      // Derived from LINPACK code.
75      // Initialize.
76      double[][] A = Arg.getArrayCopy();
77      m = Arg.getRowDimension();
78      n = Arg.getColumnDimension();
79  
80      /* Apparently the failing cases are only a proper subset of (m<n), 
81       so let's not throw error.  Correct fix to come later?
82      if (m<n) {
83        throw new IllegalArgumentException("Jama SVD only works for m >= n"); }
84      */
85  
86      int nu = Math.min(m,n);
87      s = new double [Math.min(m+1,n)];
88      U = new double [m][nu];
89      V = new double [n][n];
90      double[] e = new double [n];
91      double[] work = new double [m];
92      boolean wantu = true;
93      boolean wantv = true;
94  
95      // Reduce A to bidiagonal form, storing the diagonal elements
96      // in s and the super-diagonal elements in e.
97  
98      int nct = Math.min(m-1,n);
99      int nrt = Math.max(0,Math.min(n-2,m));
100     for (int k = 0; k < Math.max(nct,nrt); k++) {
101       if (k < nct) {
102 
103         // Compute the transformation for the k-th column and
104         // place the k-th diagonal in s[k].
105         // Compute 2-norm of k-th column without under/overflow.
106         s[k] = 0;
107         for (int i = k; i < m; i++) {
108           s[k] = Maths.hypot(s[k],A[i][k]);
109         }
110         if (s[k] != 0.0) {
111           if (A[k][k] < 0.0) {
112             s[k] = -s[k];
113           }
114           for (int i = k; i < m; i++) {
115             A[i][k] /= s[k];
116           }
117           A[k][k] += 1.0;
118         }
119         s[k] = -s[k];
120       }
121       for (int j = k+1; j < n; j++) {
122         if ((k < nct) & (s[k] != 0.0))  {
123 
124           // Apply the transformation.
125 
126           double t = 0;
127           for (int i = k; i < m; i++) {
128             t += A[i][k]*A[i][j];
129           }
130           t = -t/A[k][k];
131           for (int i = k; i < m; i++) {
132             A[i][j] += t*A[i][k];
133           }
134         }
135 
136         // Place the k-th row of A into e for the
137         // subsequent calculation of the row transformation.
138 
139         e[j] = A[k][j];
140       }
141       if (wantu & (k < nct)) {
142 
143         // Place the transformation in U for subsequent back
144         // multiplication.
145 
146         for (int i = k; i < m; i++) {
147           U[i][k] = A[i][k];
148         }
149       }
150       if (k < nrt) {
151 
152         // Compute the k-th row transformation and place the
153         // k-th super-diagonal in e[k].
154         // Compute 2-norm without under/overflow.
155         e[k] = 0;
156         for (int i = k+1; i < n; i++) {
157           e[k] = Maths.hypot(e[k],e[i]);
158         }
159         if (e[k] != 0.0) {
160           if (e[k+1] < 0.0) {
161             e[k] = -e[k];
162           }
163           for (int i = k+1; i < n; i++) {
164             e[i] /= e[k];
165           }
166           e[k+1] += 1.0;
167         }
168         e[k] = -e[k];
169         if ((k+1 < m) & (e[k] != 0.0)) {
170 
171           // Apply the transformation.
172 
173           for (int i = k+1; i < m; i++) {
174             work[i] = 0.0;
175           }
176           for (int j = k+1; j < n; j++) {
177             for (int i = k+1; i < m; i++) {
178               work[i] += e[j]*A[i][j];
179             }
180           }
181           for (int j = k+1; j < n; j++) {
182             double t = -e[j]/e[k+1];
183             for (int i = k+1; i < m; i++) {
184               A[i][j] += t*work[i];
185             }
186           }
187         }
188         if (wantv) {
189 
190           // Place the transformation in V for subsequent
191           // back multiplication.
192 
193           for (int i = k+1; i < n; i++) {
194             V[i][k] = e[i];
195           }
196         }
197       }
198     }
199 
200     // Set up the final bidiagonal matrix or order p.
201 
202     int p = Math.min(n,m+1);
203     if (nct < n) {
204       s[nct] = A[nct][nct];
205     }
206     if (m < p) {
207       s[p-1] = 0.0;
208     }
209     if (nrt+1 < p) {
210       e[nrt] = A[nrt][p-1];
211     }
212     e[p-1] = 0.0;
213 
214     // If required, generate U.
215 
216     if (wantu) {
217       for (int j = nct; j < nu; j++) {
218         for (int i = 0; i < m; i++) {
219           U[i][j] = 0.0;
220         }
221         U[j][j] = 1.0;
222       }
223       for (int k = nct-1; k >= 0; k--) {
224         if (s[k] != 0.0) {
225           for (int j = k+1; j < nu; j++) {
226             double t = 0;
227             for (int i = k; i < m; i++) {
228               t += U[i][k]*U[i][j];
229             }
230             t = -t/U[k][k];
231             for (int i = k; i < m; i++) {
232               U[i][j] += t*U[i][k];
233             }
234           }
235           for (int i = k; i < m; i++ ) {
236             U[i][k] = -U[i][k];
237           }
238           U[k][k] = 1.0 + U[k][k];
239           for (int i = 0; i < k-1; i++) {
240             U[i][k] = 0.0;
241           }
242         } else {
243           for (int i = 0; i < m; i++) {
244             U[i][k] = 0.0;
245           }
246           U[k][k] = 1.0;
247         }
248       }
249     }
250 
251     // If required, generate V.
252 
253     if (wantv) {
254       for (int k = n-1; k >= 0; k--) {
255         if ((k < nrt) & (e[k] != 0.0)) {
256           for (int j = k+1; j < nu; j++) {
257             double t = 0;
258             for (int i = k+1; i < n; i++) {
259               t += V[i][k]*V[i][j];
260             }
261             t = -t/V[k+1][k];
262             for (int i = k+1; i < n; i++) {
263               V[i][j] += t*V[i][k];
264             }
265           }
266         }
267         for (int i = 0; i < n; i++) {
268           V[i][k] = 0.0;
269         }
270         V[k][k] = 1.0;
271       }
272     }
273 
274     // Main iteration loop for the singular values.
275 
276     int pp = p-1;
277     int iter = 0;
278     double eps = Math.pow(2.0,-52.0);
279     double tiny = Math.pow(2.0,-966.0);
280     while (p > 0) {
281       int k,kase;
282 
283       // Here is where a test for too many iterations would go.
284 
285       // This section of the program inspects for
286       // negligible elements in the s and e arrays.  On
287       // completion the variables kase and k are set as follows.
288 
289       // kase = 1     if s(p) and e[k-1] are negligible and k<p
290       // kase = 2     if s(k) is negligible and k<p
291       // kase = 3     if e[k-1] is negligible, k<p, and
292       //              s(k), ..., s(p) are not negligible (qr step).
293       // kase = 4     if e(p-1) is negligible (convergence).
294 
295       for (k = p-2; k >= -1; k--) {
296         if (k == -1) {
297           break;
298         }
299         if (Math.abs(e[k]) <=
300           tiny + eps*(Math.abs(s[k]) + Math.abs(s[k+1]))) {
301           e[k] = 0.0;
302           break;
303         }
304       }
305       if (k == p-2) {
306         kase = 4;
307       } else {
308         int ks;
309         for (ks = p-1; ks >= k; ks--) {
310           if (ks == k) {
311             break;
312           }
313           double t = (ks != p ? Math.abs(e[ks]) : 0.) + 
314                                                   (ks != k+1 ? Math.abs(e[ks-1]) : 0.);
315           if (Math.abs(s[ks]) <= tiny + eps*t)  {
316             s[ks] = 0.0;
317             break;
318           }
319         }
320         if (ks == k) {
321           kase = 3;
322         } else if (ks == p-1) {
323           kase = 1;
324         } else {
325           kase = 2;
326           k = ks;
327         }
328       }
329       k++;
330 
331       // Perform the task indicated by kase.
332 
333       switch (kase) {
334 
335         // Deflate negligible s(p).
336 
337         case 1: {
338                   double f = e[p-2];
339                   e[p-2] = 0.0;
340                   for (int j = p-2; j >= k; j--) {
341                     double t = Maths.hypot(s[j],f);
342                     double cs = s[j]/t;
343                     double sn = f/t;
344                     s[j] = t;
345                     if (j != k) {
346                       f = -sn*e[j-1];
347                       e[j-1] = cs*e[j-1];
348                     }
349                     if (wantv) {
350                       for (int i = 0; i < n; i++) {
351                         t = cs*V[i][j] + sn*V[i][p-1];
352                         V[i][p-1] = -sn*V[i][j] + cs*V[i][p-1];
353                         V[i][j] = t;
354                       }
355                     }
356                   }
357         }
358         break;
359 
360         // Split at negligible s(k).
361 
362         case 2: {
363                   double f = e[k-1];
364                   e[k-1] = 0.0;
365                   for (int j = k; j < p; j++) {
366                     double t = Maths.hypot(s[j],f);
367                     double cs = s[j]/t;
368                     double sn = f/t;
369                     s[j] = t;
370                     f = -sn*e[j];
371                     e[j] = cs*e[j];
372                     if (wantu) {
373                       for (int i = 0; i < m; i++) {
374                         t = cs*U[i][j] + sn*U[i][k-1];
375                         U[i][k-1] = -sn*U[i][j] + cs*U[i][k-1];
376                         U[i][j] = t;
377                       }
378                     }
379                   }
380         }
381         break;
382 
383         // Perform one qr step.
384 
385         case 3: {
386 
387                   // Calculate the shift.
388 
389                   double scale = Math.max(Math.max(Math.max(Math.max(
390                             Math.abs(s[p-1]),Math.abs(s[p-2])),Math.abs(e[p-2])), 
391                         Math.abs(s[k])),Math.abs(e[k]));
392                   double sp = s[p-1]/scale;
393                   double spm1 = s[p-2]/scale;
394                   double epm1 = e[p-2]/scale;
395                   double sk = s[k]/scale;
396                   double ek = e[k]/scale;
397                   double b = ((spm1 + sp)*(spm1 - sp) + epm1*epm1)/2.0;
398                   double c = (sp*epm1)*(sp*epm1);
399                   double shift = 0.0;
400                   if ((b != 0.0) | (c != 0.0)) {
401                     shift = Math.sqrt(b*b + c);
402                     if (b < 0.0) {
403                       shift = -shift;
404                     }
405                     shift = c/(b + shift);
406                   }
407                   double f = (sk + sp)*(sk - sp) + shift;
408                   double g = sk*ek;
409 
410                   // Chase zeros.
411 
412                   for (int j = k; j < p-1; j++) {
413                     double t = Maths.hypot(f,g);
414                     double cs = f/t;
415                     double sn = g/t;
416                     if (j != k) {
417                       e[j-1] = t;
418                     }
419                     f = cs*s[j] + sn*e[j];
420                     e[j] = cs*e[j] - sn*s[j];
421                     g = sn*s[j+1];
422                     s[j+1] = cs*s[j+1];
423                     if (wantv) {
424                       for (int i = 0; i < n; i++) {
425                         t = cs*V[i][j] + sn*V[i][j+1];
426                         V[i][j+1] = -sn*V[i][j] + cs*V[i][j+1];
427                         V[i][j] = t;
428                       }
429                     }
430                     t = Maths.hypot(f,g);
431                     cs = f/t;
432                     sn = g/t;
433                     s[j] = t;
434                     f = cs*e[j] + sn*s[j+1];
435                     s[j+1] = -sn*e[j] + cs*s[j+1];
436                     g = sn*e[j+1];
437                     e[j+1] = cs*e[j+1];
438                     if (wantu && (j < m-1)) {
439                       for (int i = 0; i < m; i++) {
440                         t = cs*U[i][j] + sn*U[i][j+1];
441                         U[i][j+1] = -sn*U[i][j] + cs*U[i][j+1];
442                         U[i][j] = t;
443                       }
444                     }
445                   }
446                   e[p-2] = f;
447                   iter = iter + 1;
448         }
449         break;
450 
451         // Convergence.
452 
453         case 4: {
454 
455                   // Make the singular values positive.
456 
457                   if (s[k] <= 0.0) {
458                     s[k] = (s[k] < 0.0 ? -s[k] : 0.0);
459                     if (wantv) {
460                       for (int i = 0; i <= pp; i++) {
461                         V[i][k] = -V[i][k];
462                       }
463                     }
464                   }
465 
466                   // Order the singular values.
467 
468                   while (k < pp) {
469                     if (s[k] >= s[k+1]) {
470                       break;
471                     }
472                     double t = s[k];
473                     s[k] = s[k+1];
474                     s[k+1] = t;
475                     if (wantv && (k < n-1)) {
476                       for (int i = 0; i < n; i++) {
477                         t = V[i][k+1]; V[i][k+1] = V[i][k]; V[i][k] = t;
478                       }
479                     }
480                     if (wantu && (k < m-1)) {
481                       for (int i = 0; i < m; i++) {
482                         t = U[i][k+1]; U[i][k+1] = U[i][k]; U[i][k] = t;
483                       }
484                     }
485                     k++;
486                   }
487                   iter = 0;
488                   p--;
489         }
490         break;
491       }
492     }
493   }

495   /** 
496    * Return the left singular vectors
497    * @return     U
498    */
499   public Matrix getU() {
500     return new Matrix(U,m,Math.min(m+1,n));
501   }
502 
503   /** 
504    * Return the right singular vectors
505    * @return     V
506    */
507   public Matrix getV() {
508     return new Matrix(V,n,n);
509   }
510 
511   /** 
512    * Return the one-dimensional array of singular values
513    * @return     diagonal of S.
514    */
515   public double[] getSingularValues() {
516     return s;
517   }
518 
519   /** 
520    * Return the diagonal matrix of singular values
521    * @return     S
522    */
523   public Matrix getS() {
524     Matrix X = new Matrix(n,n);
525     double[][] S = X.getArray();
526     for (int i = 0; i < n; i++) {
527       for (int j = 0; j < n; j++) {
528         S[i][j] = 0.0;
529       }
530       S[i][i] = this.s[i];
531     }
532     return X;
533   }
534 
535   /** 
536    * Two norm
537    * @return     max(S)
538    */
539   public double norm2() {
540     return s[0];
541   }
542 
543   /** 
544    * Two norm condition number
545    * @return     max(S)/min(S)
546    */
547   public double cond() {
548     return s[0]/s[Math.min(m,n)-1];
549   }
550 
551   /** 
552    * Effective numerical matrix rank
553    * @return     Number of nonnegligible singular values.
554    */
555   public int rank() {
556     double eps = Math.pow(2.0,-52.0);
557     double tol = Math.max(m,n)*s[0]*eps;
558     int r = 0;
559     for (int i = 0; i < s.length; i++) {
560       if (s[i] > tol) {
561         r++;
562       }
563     }
564     return r;
565   }
