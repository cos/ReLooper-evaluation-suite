22  /** 
23   * LU Decomposition.
24   * <P>
25   * For an m-by-n matrix A with m &gt;= n, the LU decomposition is an m-by-n
26   * unit lower triangular matrix L, an n-by-n upper triangular matrix U, and a
27   * permutation vector piv of length m so that A(piv,:) = L*U.  If m &lt; n,
28   * then L is m-by-m and U is m-by-n.
29   * <P>
30   * The LU decompostion with pivoting always exists, even if the matrix is
31   * singular, so the constructor will never fail.  The primary use of the LU
32   * decomposition is in the solution of square systems of simultaneous linear
33   * equations.  This will fail if isNonsingular() returns false.
34   * <p/>
35   * Adapted from the <a href="http://math.nist.gov/javanumerics/jama/" target="_blank">JAMA</a> package.
36   *
37   * @author The Mathworks and NIST 
38   * @author Fracpete (fracpete at waikato dot ac dot nz)
39   * @version $Revision: 5953 $
40   */
41  public class LUDecomposition 
42    implements Serializable, RevisionHandler {
43  
44    /** for serialization */
45    private static final long serialVersionUID = -2731022568037808629L;
46  
47    /** 
48     * Array for internal storage of decomposition.
49     * @serial internal array storage.
50     */
51    private double[][] LU;
52  
53    /** 
54     * Row and column dimensions, and pivot sign.
55     * @serial column dimension.
56     * @serial row dimension.
57     * @serial pivot sign.
58     */
59    private int m, n, pivsign; 
60  
61    /** 
62     * Internal storage of pivot vector.
63     * @serial pivot vector.
64     */
65    private int[] piv;
66  
67    /** 
68     * LU Decomposition
69     * @param  A   Rectangular matrix
70     */
71    public LUDecomposition(Matrix A) {
72  
73      // Use a "left-looking", dot-product, Crout/Doolittle algorithm.
74  
75      LU = A.getArrayCopy();
76      m = A.getRowDimension();
77      n = A.getColumnDimension();
78      piv = new int[m];
79      for (int i = 0; i < m; i++) {
80        piv[i] = i;
81      }
82      pivsign = 1;
83      double[] LUrowi;
84      double[] LUcolj = new double[m];
85  
86      // Outer loop.
87  
88      for (int j = 0; j < n; j++) {
89  
90        // Make a copy of the j-th column to localize references.
91  
92        for (int i = 0; i < m; i++) {
93          LUcolj[i] = LU[i][j];
94        }
95  
96        // Apply previous transformations.
97  
98        for (int i = 0; i < m; i++) {
99          LUrowi = LU[i];
100 
101         // Most of the time is spent in the following dot product.
102 
103         int kmax = Math.min(i,j);
104         double s = 0.0;
105         for (int k = 0; k < kmax; k++) {
106           s += LUrowi[k]*LUcolj[k];
107         }
108 
109         LUrowi[j] = LUcolj[i] -= s;
110       }
111 
112       // Find pivot and exchange if necessary.
113 
114       int p = j;
115       for (int i = j+1; i < m; i++) {
116         if (Math.abs(LUcolj[i]) > Math.abs(LUcolj[p])) {
117           p = i;
118         }
119       }
120       if (p != j) {
121         for (int k = 0; k < n; k++) {
122           double t = LU[p][k]; LU[p][k] = LU[j][k]; LU[j][k] = t;
123         }
124         int k = piv[p]; piv[p] = piv[j]; piv[j] = k;
125         pivsign = -pivsign;
126       }
127 
128       // Compute multipliers.
129 
130       if (j < m & LU[j][j] != 0.0) {
131         for (int i = j+1; i < m; i++) {
132           LU[i][j] /= LU[j][j];
133         }
134       }
135     }
136   }
137 
138   /** 
139    * Is the matrix nonsingular?
140    * @return     true if U, and hence A, is nonsingular.
141    */
142   public boolean isNonsingular() {
143     for (int j = 0; j < n; j++) {
144       if (LU[j][j] == 0)
145         return false;
146     }
147     return true;
148   }
149 
150   /** 
151    * Return lower triangular factor
152    * @return     L
153    */
154   public Matrix getL() {
155     Matrix X = new Matrix(m,n);
156     double[][] L = X.getArray();
157     for (int i = 0; i < m; i++) {
158       for (int j = 0; j < n; j++) {
159         if (i > j) {
160           L[i][j] = LU[i][j];
161         } else if (i == j) {
162           L[i][j] = 1.0;
163         } else {
164           L[i][j] = 0.0;
165         }
166       }
167     }
168     return X;
169   }
170 
171   /** 
172    * Return upper triangular factor
173    * @return     U
174    */
175   public Matrix getU() {
176     Matrix X = new Matrix(n,n);
177     double[][] U = X.getArray();
178     for (int i = 0; i < n; i++) {
179       for (int j = 0; j < n; j++) {
180         if (i <= j) {
181           U[i][j] = LU[i][j];
182         } else {
183           U[i][j] = 0.0;
184         }
185       }
186     }
187     return X;
188   }
189 
190   /** 
191    * Return pivot permutation vector
192    * @return     piv
193    */
194   public int[] getPivot() {
195     int[] p = new int[m];
196     for (int i = 0; i < m; i++) {
197       p[i] = piv[i];
198     }
199     return p;
200   }
201 
202   /** 
203    * Return pivot permutation vector as a one-dimensional double array
204    * @return     (double) piv
205    */
206   public double[] getDoublePivot() {
207     double[] vals = new double[m];
208     for (int i = 0; i < m; i++) {
209       vals[i] = (double) piv[i];
210     }
211     return vals;
212   }
213 
214   /** 
215    * Determinant
216    * @return     det(A)
217    * @exception  IllegalArgumentException  Matrix must be square
218    */
219   public double det() {
220     if (m != n) {
221       throw new IllegalArgumentException("Matrix must be square.");
222     }
223     double d = (double) pivsign;
224     for (int j = 0; j < n; j++) {
225       d *= LU[j][j];
226     }
227     return d;
228   }
229 
230   /** 
231    * Solve A*X = B
232    * @param  B   A Matrix with as many rows as A and any number of columns.
233    * @return     X so that L*U*X = B(piv,:)
234    * @exception  IllegalArgumentException Matrix row dimensions must agree.
235    * @exception  RuntimeException  Matrix is singular.
236    */
237   public Matrix solve(Matrix B) {
238     if (B.getRowDimension() != m) {
239       throw new IllegalArgumentException("Matrix row dimensions must agree.");
240     }
241     if (!this.isNonsingular()) {
242       throw new RuntimeException("Matrix is singular.");
243     }
244 
245     // Copy right hand side with pivoting
246     int nx = B.getColumnDimension();
247     Matrix Xmat = B.getMatrix(piv,0,nx-1);
248     double[][] X = Xmat.getArray();
249 
250     // Solve L*Y = B(piv,:)
251     for (int k = 0; k < n; k++) {
252       for (int i = k+1; i < n; i++) {
253         for (int j = 0; j < nx; j++) {
254           X[i][j] -= X[k][j]*LU[i][k];
255         }
256       }
257     }
258     // Solve U*X = Y;
259     for (int k = n-1; k >= 0; k--) {
260       for (int j = 0; j < nx; j++) {
261         X[k][j] /= LU[k][k];
262       }
263       for (int i = 0; i < k; i++) {
264         for (int j = 0; j < nx; j++) {
265           X[i][j] -= X[k][j]*LU[i][k];
266         }
267       }
268     }
269     return Xmat;
270   }
