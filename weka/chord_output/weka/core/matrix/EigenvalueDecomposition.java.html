22  /** 
23   * Eigenvalues and eigenvectors of a real matrix. 
24   * <P>
25   * If A is symmetric, then A = V*D*V' where the eigenvalue matrix D is diagonal
26   * and the eigenvector matrix V is orthogonal.  I.e. A =
27   * V.times(D.times(V.transpose())) and V.times(V.transpose()) equals the
28   * identity matrix.
29   * <P>
30   * If A is not symmetric, then the eigenvalue matrix D is block diagonal with
31   * the real eigenvalues in 1-by-1 blocks and any complex eigenvalues, lambda +
32   * i*mu, in 2-by-2 blocks, [lambda, mu; -mu, lambda].  The columns of V
33   * represent the eigenvectors in the sense that A*V = V*D, i.e. A.times(V)
34   * equals V.times(D).  The matrix V may be badly conditioned, or even singular,
35   * so the validity of the equation A = V*D*inverse(V) depends upon V.cond().
36   * <p/>
37   * Adapted from the <a href="http://math.nist.gov/javanumerics/jama/" target="_blank">JAMA</a> package.
38   *
39   * @author The Mathworks and NIST 
40   * @author Fracpete (fracpete at waikato dot ac dot nz)
41   * @version $Revision: 5953 $
42   */
43  public class EigenvalueDecomposition 
44    implements Serializable, RevisionHandler {

91    /** 
92     * Symmetric Householder reduction to tridiagonal form.
93     * <p/>
94     * This is derived from the Algol procedures tred2 by Bowdler, Martin,
95     * Reinsch, and Wilkinson, Handbook for Auto. Comp., Vol.ii-Linear Algebra,
96     * and the corresponding Fortran subroutine in EISPACK.
97     */
98    private void tred2() {
99  
100     for (int j = 0; j < n; j++) {
101       d[j] = V[n-1][j];
102     }
103 
104     // Householder reduction to tridiagonal form.
105 
106     for (int i = n-1; i > 0; i--) {
107 
108       // Scale to avoid under/overflow.
109 
110       double scale = 0.0;
111       double h = 0.0;
112       for (int k = 0; k < i; k++) {
113         scale = scale + Math.abs(d[k]);
114       }
115       if (scale == 0.0) {
116         e[i] = d[i-1];
117         for (int j = 0; j < i; j++) {
118           d[j] = V[i-1][j];
119           V[i][j] = 0.0;
120           V[j][i] = 0.0;
121         }
122       } else {
123 
124         // Generate Householder vector.
125 
126         for (int k = 0; k < i; k++) {
127           d[k] /= scale;
128           h += d[k] * d[k];
129         }
130         double f = d[i-1];
131         double g = Math.sqrt(h);
132         if (f > 0) {
133           g = -g;
134         }
135         e[i] = scale * g;
136         h = h - f * g;
137         d[i-1] = f - g;
138         for (int j = 0; j < i; j++) {
139           e[j] = 0.0;
140         }
141 
142         // Apply similarity transformation to remaining columns.
143 
144         for (int j = 0; j < i; j++) {
145           f = d[j];
146           V[j][i] = f;
147           g = e[j] + V[j][j] * f;
148           for (int k = j+1; k <= i-1; k++) {
149             g += V[k][j] * d[k];
150             e[k] += V[k][j] * f;
151           }
152           e[j] = g;
153         }
154         f = 0.0;
155         for (int j = 0; j < i; j++) {
156           e[j] /= h;
157           f += e[j] * d[j];
158         }
159         double hh = f / (h + h);
160         for (int j = 0; j < i; j++) {
161           e[j] -= hh * d[j];
162         }
163         for (int j = 0; j < i; j++) {
164           f = d[j];
165           g = e[j];
166           for (int k = j; k <= i-1; k++) {
167             V[k][j] -= (f * e[k] + g * d[k]);
168           }
169           d[j] = V[i-1][j];
170           V[i][j] = 0.0;
171         }
172       }
173       d[i] = h;
174     }
175 
176     // Accumulate transformations.
177 
178     for (int i = 0; i < n-1; i++) {
179       V[n-1][i] = V[i][i];
180       V[i][i] = 1.0;
181       double h = d[i+1];
182       if (h != 0.0) {
183         for (int k = 0; k <= i; k++) {
184           d[k] = V[k][i+1] / h;
185         }
186         for (int j = 0; j <= i; j++) {
187           double g = 0.0;
188           for (int k = 0; k <= i; k++) {
189             g += V[k][i+1] * V[k][j];
190           }
191           for (int k = 0; k <= i; k++) {
192             V[k][j] -= g * d[k];
193           }
194         }
195       }
196       for (int k = 0; k <= i; k++) {
197         V[k][i+1] = 0.0;
198       }
199     }
200     for (int j = 0; j < n; j++) {
201       d[j] = V[n-1][j];
202       V[n-1][j] = 0.0;
203     }
204     V[n-1][n-1] = 1.0;
205     e[0] = 0.0;
206   }

208   /** 
209    * Symmetric tridiagonal QL algorithm.
210    * <p/>
211    * This is derived from the Algol procedures tql2, by Bowdler, Martin,
212    * Reinsch, and Wilkinson, Handbook for Auto. Comp., Vol.ii-Linear Algebra,
213    * and the corresponding Fortran subroutine in EISPACK.
214    */
215   private void tql2() {
216 
217     for (int i = 1; i < n; i++) {
218       e[i-1] = e[i];
219     }
220     e[n-1] = 0.0;
221 
222     double f = 0.0;
223     double tst1 = 0.0;
224     double eps = Math.pow(2.0,-52.0);
225     for (int l = 0; l < n; l++) {
226 
227       // Find small subdiagonal element
228 
229       tst1 = Math.max(tst1,Math.abs(d[l]) + Math.abs(e[l]));
230       int m = l;
231       while (m < n) {
232         if (Math.abs(e[m]) <= eps*tst1) {
233           break;
234         }
235         m++;
236       }
237 
238       // If m == l, d[l] is an eigenvalue,
239       // otherwise, iterate.
240 
241       if (m > l) {
242         int iter = 0;
243         do {
244           iter = iter + 1;  // (Could check iteration count here.)
245 
246           // Compute implicit shift
247 
248           double g = d[l];
249           double p = (d[l+1] - g) / (2.0 * e[l]);
250           double r = Maths.hypot(p,1.0);
251           if (p < 0) {
252             r = -r;
253           }
254           d[l] = e[l] / (p + r);
255           d[l+1] = e[l] * (p + r);
256           double dl1 = d[l+1];
257           double h = g - d[l];
258           for (int i = l+2; i < n; i++) {
259             d[i] -= h;
260           }
261           f = f + h;
262 
263           // Implicit QL transformation.
264 
265           p = d[m];
266           double c = 1.0;
267           double c2 = c;
268           double c3 = c;
269           double el1 = e[l+1];
270           double s = 0.0;
271           double s2 = 0.0;
272           for (int i = m-1; i >= l; i--) {
273             c3 = c2;
274             c2 = c;
275             s2 = s;
276             g = c * e[i];
277             h = c * p;
278             r = Maths.hypot(p,e[i]);
279             e[i+1] = s * r;
280             s = e[i] / r;
281             c = p / r;
282             p = c * d[i] - s * g;
283             d[i+1] = h + s * (c * g + s * d[i]);
284 
285             // Accumulate transformation.
286 
287             for (int k = 0; k < n; k++) {
288               h = V[k][i+1];
289               V[k][i+1] = s * V[k][i] + c * h;
290               V[k][i] = c * V[k][i] - s * h;
291             }
292           }
293           p = -s * s2 * c3 * el1 * e[l] / dl1;
294           e[l] = s * p;
295           d[l] = c * p;
296 
297           // Check for convergence.
298 
299         } while (Math.abs(e[l]) > eps*tst1);
300       }
301       d[l] = d[l] + f;
302       e[l] = 0.0;
303     }
304 
305     // Sort eigenvalues and corresponding vectors.
306 
307     for (int i = 0; i < n-1; i++) {
308       int k = i;
309       double p = d[i];
310       for (int j = i+1; j < n; j++) {
311         if (d[j] < p) {
312           k = j;
313           p = d[j];
314         }
315       }
316       if (k != i) {
317         d[k] = d[i];
318         d[i] = p;
319         for (int j = 0; j < n; j++) {
320           p = V[j][i];
321           V[j][i] = V[j][k];
322           V[j][k] = p;
323         }
324       }
325     }
326   }

328   /**
329    * Nonsymmetric reduction to Hessenberg form.
330    * <p/>
331    * This is derived from the Algol procedures orthes and ortran, by Martin
332    * and Wilkinson, Handbook for Auto. Comp., Vol.ii-Linear Algebra, and the
333    * corresponding Fortran subroutines in EISPACK.
334    */
335   private void orthes() {
336 
337     int low = 0;
338     int high = n-1;
339 
340     for (int m = low+1; m <= high-1; m++) {
341 
342       // Scale column.
343 
344       double scale = 0.0;
345       for (int i = m; i <= high; i++) {
346         scale = scale + Math.abs(H[i][m-1]);
347       }
348       if (scale != 0.0) {
349 
350         // Compute Householder transformation.
351 
352         double h = 0.0;
353         for (int i = high; i >= m; i--) {
354           ort[i] = H[i][m-1]/scale;
355           h += ort[i] * ort[i];
356         }
357         double g = Math.sqrt(h);
358         if (ort[m] > 0) {
359           g = -g;
360         }
361         h = h - ort[m] * g;
362         ort[m] = ort[m] - g;
363 
364         // Apply Householder similarity transformation
365         // H = (I-u*u'/h)*H*(I-u*u')/h)
366 
367         for (int j = m; j < n; j++) {
368           double f = 0.0;
369           for (int i = high; i >= m; i--) {
370             f += ort[i]*H[i][j];
371           }
372           f = f/h;
373           for (int i = m; i <= high; i++) {
374             H[i][j] -= f*ort[i];
375           }
376         }
377 
378         for (int i = 0; i <= high; i++) {
379           double f = 0.0;
380           for (int j = high; j >= m; j--) {
381             f += ort[j]*H[i][j];
382           }
383           f = f/h;
384           for (int j = m; j <= high; j++) {
385             H[i][j] -= f*ort[j];
386           }
387         }
388         ort[m] = scale*ort[m];
389         H[m][m-1] = scale*g;
390       }
391     }
392 
393     // Accumulate transformations (Algol's ortran).
394 
395     for (int i = 0; i < n; i++) {
396       for (int j = 0; j < n; j++) {
397         V[i][j] = (i == j ? 1.0 : 0.0);
398       }
399     }
400 
401     for (int m = high-1; m >= low+1; m--) {
402       if (H[m][m-1] != 0.0) {
403         for (int i = m+1; i <= high; i++) {
404           ort[i] = H[i][m-1];
405         }
406         for (int j = m; j <= high; j++) {
407           double g = 0.0;
408           for (int i = m; i <= high; i++) {
409             g += ort[i] * V[i][j];
410           }
411           // Double division avoids possible underflow
412           g = (g / ort[m]) / H[m][m-1];
413           for (int i = m; i <= high; i++) {
414             V[i][j] += g * ort[i];
415           }
416         }
417       }
418     }
419   }

422   /** 
423    * Complex scalar division.
424    */
425   private void cdiv(double xr, double xi, double yr, double yi) {
426     double r,d;
427     if (Math.abs(yr) > Math.abs(yi)) {
428       r = yi/yr;
429       d = yr + r*yi;
430       cdivr = (xr + r*xi)/d;
431       cdivi = (xi - r*xr)/d;
432     } else {
433       r = yr/yi;
434       d = yi + r*yr;
435       cdivr = (r*xr + xi)/d;
436       cdivi = (r*xi - xr)/d;
437     }
438   }

441   /**
442    * Nonsymmetric reduction from Hessenberg to real Schur form.
443    * <p/>
444    * This is derived from the Algol procedure hqr2, by Martin and Wilkinson,
445    * Handbook for Auto. Comp., Vol.ii-Linear Algebra, and the corresponding
446    * Fortran subroutine in EISPACK.
447    */
448   private void hqr2() {
449 
450     // Initialize
451 
452     int nn = this.n;
453     int n = nn-1;
454     int low = 0;
455     int high = nn-1;
456     double eps = Math.pow(2.0,-52.0);
457     double exshift = 0.0;
458     double p=0,q=0,r=0,s=0,z=0,t,w,x,y;
459 
460     // Store roots isolated by balanc and compute matrix norm
461 
462     double norm = 0.0;
463     for (int i = 0; i < nn; i++) {
464       if (i < low | i > high) {
465         d[i] = H[i][i];
466         e[i] = 0.0;
467       }
468       for (int j = Math.max(i-1,0); j < nn; j++) {
469         norm = norm + Math.abs(H[i][j]);
470       }
471     }
472 
473     // Outer loop over eigenvalue index
474 
475     int iter = 0;
476     while (n >= low) {
477 
478       // Look for single small sub-diagonal element
479 
480       int l = n;
481       while (l > low) {
482         s = Math.abs(H[l-1][l-1]) + Math.abs(H[l][l]);
483         if (s == 0.0) {
484           s = norm;
485         }
486         if (Math.abs(H[l][l-1]) < eps * s) {
487           break;
488         }
489         l--;
490       }
491 
492       // Check for convergence
493       // One root found
494 
495       if (l == n) {
496         H[n][n] = H[n][n] + exshift;
497         d[n] = H[n][n];
498         e[n] = 0.0;
499         n--;
500         iter = 0;
501 
502         // Two roots found
503 
504       } else if (l == n-1) {
505         w = H[n][n-1] * H[n-1][n];
506         p = (H[n-1][n-1] - H[n][n]) / 2.0;
507         q = p * p + w;
508         z = Math.sqrt(Math.abs(q));
509         H[n][n] = H[n][n] + exshift;
510         H[n-1][n-1] = H[n-1][n-1] + exshift;
511         x = H[n][n];
512 
513         // Real pair
514 
515         if (q >= 0) {
516           if (p >= 0) {
517             z = p + z;
518           } else {
519             z = p - z;
520           }
521           d[n-1] = x + z;
522           d[n] = d[n-1];
523           if (z != 0.0) {
524             d[n] = x - w / z;
525           }
526           e[n-1] = 0.0;
527           e[n] = 0.0;
528           x = H[n][n-1];
529           s = Math.abs(x) + Math.abs(z);
530           p = x / s;
531           q = z / s;
532           r = Math.sqrt(p * p+q * q);
533           p = p / r;
534           q = q / r;
535 
536           // Row modification
537 
538           for (int j = n-1; j < nn; j++) {
539             z = H[n-1][j];
540             H[n-1][j] = q * z + p * H[n][j];
541             H[n][j] = q * H[n][j] - p * z;
542           }
543 
544           // Column modification
545 
546           for (int i = 0; i <= n; i++) {
547             z = H[i][n-1];
548             H[i][n-1] = q * z + p * H[i][n];
549             H[i][n] = q * H[i][n] - p * z;
550           }
551 
552           // Accumulate transformations
553 
554           for (int i = low; i <= high; i++) {
555             z = V[i][n-1];
556             V[i][n-1] = q * z + p * V[i][n];
557             V[i][n] = q * V[i][n] - p * z;
558           }
559 
560           // Complex pair
561 
562         } else {
563           d[n-1] = x + p;
564           d[n] = x + p;
565           e[n-1] = z;
566           e[n] = -z;
567         }
568         n = n - 2;
569         iter = 0;
570 
571         // No convergence yet
572 
573       } else {
574 
575         // Form shift
576 
577         x = H[n][n];
578         y = 0.0;
579         w = 0.0;
580         if (l < n) {
581           y = H[n-1][n-1];
582           w = H[n][n-1] * H[n-1][n];
583         }
584 
585         // Wilkinson's original ad hoc shift
586 
587         if (iter == 10) {
588           exshift += x;
589           for (int i = low; i <= n; i++) {
590             H[i][i] -= x;
591           }
592           s = Math.abs(H[n][n-1]) + Math.abs(H[n-1][n-2]);
593           x = y = 0.75 * s;
594           w = -0.4375 * s * s;
595         }
596 
597         // MATLAB's new ad hoc shift
598 
599         if (iter == 30) {
600           s = (y - x) / 2.0;
601           s = s * s + w;
602           if (s > 0) {
603             s = Math.sqrt(s);
604             if (y < x) {
605               s = -s;
606             }
607             s = x - w / ((y - x) / 2.0 + s);
608             for (int i = low; i <= n; i++) {
609               H[i][i] -= s;
610             }
611             exshift += s;
612             x = y = w = 0.964;
613           }
614         }
615 
616         iter = iter + 1;   // (Could check iteration count here.)
617 
618         // Look for two consecutive small sub-diagonal elements
619 
620         int m = n-2;
621         while (m >= l) {
622           z = H[m][m];
623           r = x - z;
624           s = y - z;
625           p = (r * s - w) / H[m+1][m] + H[m][m+1];
626           q = H[m+1][m+1] - z - r - s;
627           r = H[m+2][m+1];
628           s = Math.abs(p) + Math.abs(q) + Math.abs(r);
629           p = p / s;
630           q = q / s;
631           r = r / s;
632           if (m == l) {
633             break;
634           }
635           if (Math.abs(H[m][m-1]) * (Math.abs(q) + Math.abs(r)) <
636               eps * (Math.abs(p) * (Math.abs(H[m-1][m-1]) + Math.abs(z) +
637                   Math.abs(H[m+1][m+1])))) {
638             break;
639                   }
640           m--;
641         }
642 
643         for (int i = m+2; i <= n; i++) {
644           H[i][i-2] = 0.0;
645           if (i > m+2) {
646             H[i][i-3] = 0.0;
647           }
648         }
649 
650         // Double QR step involving rows l:n and columns m:n
651 
652         for (int k = m; k <= n-1; k++) {
653           boolean notlast = (k != n-1);
654           if (k != m) {
655             p = H[k][k-1];
656             q = H[k+1][k-1];
657             r = (notlast ? H[k+2][k-1] : 0.0);
658             x = Math.abs(p) + Math.abs(q) + Math.abs(r);
659             if (x != 0.0) {
660               p = p / x;
661               q = q / x;
662               r = r / x;
663             }
664           }
665           if (x == 0.0) {
666             break;
667           }
668           s = Math.sqrt(p * p + q * q + r * r);
669           if (p < 0) {
670             s = -s;
671           }
672           if (s != 0) {
673             if (k != m) {
674               H[k][k-1] = -s * x;
675             } else if (l != m) {
676               H[k][k-1] = -H[k][k-1];
677             }
678             p = p + s;
679             x = p / s;
680             y = q / s;
681             z = r / s;
682             q = q / p;
683             r = r / p;
684 
685             // Row modification
686 
687             for (int j = k; j < nn; j++) {
688               p = H[k][j] + q * H[k+1][j];
689               if (notlast) {
690                 p = p + r * H[k+2][j];
691                 H[k+2][j] = H[k+2][j] - p * z;
692               }
693               H[k][j] = H[k][j] - p * x;
694               H[k+1][j] = H[k+1][j] - p * y;
695             }
696 
697             // Column modification
698 
699             for (int i = 0; i <= Math.min(n,k+3); i++) {
700               p = x * H[i][k] + y * H[i][k+1];
701               if (notlast) {
702                 p = p + z * H[i][k+2];
703                 H[i][k+2] = H[i][k+2] - p * r;
704               }
705               H[i][k] = H[i][k] - p;
706               H[i][k+1] = H[i][k+1] - p * q;
707             }
708 
709             // Accumulate transformations
710 
711             for (int i = low; i <= high; i++) {
712               p = x * V[i][k] + y * V[i][k+1];
713               if (notlast) {
714                 p = p + z * V[i][k+2];
715                 V[i][k+2] = V[i][k+2] - p * r;
716               }
717               V[i][k] = V[i][k] - p;
718               V[i][k+1] = V[i][k+1] - p * q;
719             }
720           }  // (s != 0)
721         }  // k loop
722       }  // check convergence
723     }  // while (n >= low)
724 
725     // Backsubstitute to find vectors of upper triangular form
726 
727     if (norm == 0.0) {
728       return;
729     }
730 
731     for (n = nn-1; n >= 0; n--) {
732       p = d[n];
733       q = e[n];
734 
735       // Real vector
736 
737       if (q == 0) {
738         int l = n;
739         H[n][n] = 1.0;
740         for (int i = n-1; i >= 0; i--) {
741           w = H[i][i] - p;
742           r = 0.0;
743           for (int j = l; j <= n; j++) {
744             r = r + H[i][j] * H[j][n];
745           }
746           if (e[i] < 0.0) {
747             z = w;
748             s = r;
749           } else {
750             l = i;
751             if (e[i] == 0.0) {
752               if (w != 0.0) {
753                 H[i][n] = -r / w;
754               } else {
755                 H[i][n] = -r / (eps * norm);
756               }
757 
758               // Solve real equations
759 
760             } else {
761               x = H[i][i+1];
762               y = H[i+1][i];
763               q = (d[i] - p) * (d[i] - p) + e[i] * e[i];
764               t = (x * s - z * r) / q;
765               H[i][n] = t;
766               if (Math.abs(x) > Math.abs(z)) {
767                 H[i+1][n] = (-r - w * t) / x;
768               } else {
769                 H[i+1][n] = (-s - y * t) / z;
770               }
771             }
772 
773             // Overflow control
774 
775             t = Math.abs(H[i][n]);
776             if ((eps * t) * t > 1) {
777               for (int j = i; j <= n; j++) {
778                 H[j][n] = H[j][n] / t;
779               }
780             }
781           }
782         }
783 
784         // Complex vector
785 
786       } else if (q < 0) {
787         int l = n-1;
788 
789         // Last vector component imaginary so matrix is triangular
790 
791         if (Math.abs(H[n][n-1]) > Math.abs(H[n-1][n])) {
792           H[n-1][n-1] = q / H[n][n-1];
793           H[n-1][n] = -(H[n][n] - p) / H[n][n-1];
794         } else {
795           cdiv(0.0,-H[n-1][n],H[n-1][n-1]-p,q);
796           H[n-1][n-1] = cdivr;
797           H[n-1][n] = cdivi;
798         }
799         H[n][n-1] = 0.0;
800         H[n][n] = 1.0;
801         for (int i = n-2; i >= 0; i--) {
802           double ra,sa,vr,vi;
803           ra = 0.0;
804           sa = 0.0;
805           for (int j = l; j <= n; j++) {
806             ra = ra + H[i][j] * H[j][n-1];
807             sa = sa + H[i][j] * H[j][n];
808           }
809           w = H[i][i] - p;
810 
811           if (e[i] < 0.0) {
812             z = w;
813             r = ra;
814             s = sa;
815           } else {
816             l = i;
817             if (e[i] == 0) {
818               cdiv(-ra,-sa,w,q);
819               H[i][n-1] = cdivr;
820               H[i][n] = cdivi;
821             } else {
822 
823               // Solve complex equations
824 
825               x = H[i][i+1];
826               y = H[i+1][i];
827               vr = (d[i] - p) * (d[i] - p) + e[i] * e[i] - q * q;
828               vi = (d[i] - p) * 2.0 * q;
829               if (vr == 0.0 & vi == 0.0) {
830                 vr = eps * norm * (Math.abs(w) + Math.abs(q) +
831                     Math.abs(x) + Math.abs(y) + Math.abs(z));
832               }
833               cdiv(x*r-z*ra+q*sa,x*s-z*sa-q*ra,vr,vi);
834               H[i][n-1] = cdivr;
835               H[i][n] = cdivi;
836               if (Math.abs(x) > (Math.abs(z) + Math.abs(q))) {
837                 H[i+1][n-1] = (-ra - w * H[i][n-1] + q * H[i][n]) / x;
838                 H[i+1][n] = (-sa - w * H[i][n] - q * H[i][n-1]) / x;
839               } else {
840                 cdiv(-r-y*H[i][n-1],-s-y*H[i][n],z,q);
841                 H[i+1][n-1] = cdivr;
842                 H[i+1][n] = cdivi;
843               }
844             }
845 
846             // Overflow control
847 
848             t = Math.max(Math.abs(H[i][n-1]),Math.abs(H[i][n]));
849             if ((eps * t) * t > 1) {
850               for (int j = i; j <= n; j++) {
851                 H[j][n-1] = H[j][n-1] / t;
852                 H[j][n] = H[j][n] / t;
853               }
854             }
855           }
856         }
857       }
858     }
859 
860     // Vectors of isolated roots
861 
862     for (int i = 0; i < nn; i++) {
863       if (i < low | i > high) {
864         for (int j = i; j < nn; j++) {
865           V[i][j] = H[i][j];
866         }
867       }
868     }
869 
870     // Back transformation to get eigenvectors of original matrix
871 
872     for (int j = nn-1; j >= low; j--) {
873       for (int i = low; i <= high; i++) {
874         z = 0.0;
875         for (int k = low; k <= Math.min(j,high); k++) {
876           z = z + V[i][k] * H[k][j];
877         }
878         V[i][j] = z;
879       }
880     }
881   }

884   /** 
885    * Check for symmetry, then construct the eigenvalue decomposition
886    *
887    * @param Arg    Square matrix
888    */
889   public EigenvalueDecomposition(Matrix Arg) {
890     double[][] A = Arg.getArray();
891     n = Arg.getColumnDimension();
892     V = new double[n][n];
893     d = new double[n];
894     e = new double[n];
895 
896     issymmetric = true;
897     for (int j = 0; (j < n) & issymmetric; j++) {
898       for (int i = 0; (i < n) & issymmetric; i++) {
899         issymmetric = (A[i][j] == A[j][i]);
900       }
901     }
902 
903     if (issymmetric) {
904       for (int i = 0; i < n; i++) {
905         for (int j = 0; j < n; j++) {
906           V[i][j] = A[i][j];
907         }
908       }
909 
910       // Tridiagonalize.
911       tred2();
912 
913       // Diagonalize.
914       tql2();
915 
916     } else {
917       H = new double[n][n];
918       ort = new double[n];
919 
920       for (int j = 0; j < n; j++) {
921         for (int i = 0; i < n; i++) {
922           H[i][j] = A[i][j];
923         }
924       }
925 
926       // Reduce to Hessenberg form.
927       orthes();
928 
929       // Reduce Hessenberg to real Schur form.
930       hqr2();
931     }
932   }

959    * Return the block diagonal eigenvalue matrix
960    * @return     D
961    */
962   public Matrix getD() {
963     Matrix X = new Matrix(n,n);
964     double[][] D = X.getArray();
965     for (int i = 0; i < n; i++) {
966       for (int j = 0; j < n; j++) {
967         D[i][j] = 0.0;
968       }
969       D[i][i] = d[i];
970       if (e[i] > 0) {
971         D[i][i+1] = e[i];
972       } else if (e[i] < 0) {
973         D[i][i-1] = e[i];
974       }
975     }
976     return X;
977   }
