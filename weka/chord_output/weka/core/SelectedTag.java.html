23  package weka.core;
24  
25  import java.util.HashSet;
26  
27  /**
28   * Represents a selected value from a finite set of values, where each
29   * value is a Tag (i.e. has some string associated with it). Primarily
30   * used in schemes to select between alternative behaviours,
31   * associating names with the alternative behaviours.
32   *
33   * @author <a href="mailto:len@reeltwo.com">Len Trigg</a> 
34   * @version $Revision: 5953 $
35   */
36  public class SelectedTag
37    implements RevisionHandler {
38    
39    /** The index of the selected tag */
40    protected int m_Selected;
41    
42    /** The set of tags to choose from */
43    protected Tag[] m_Tags;
44    
45    /**
46     * Creates a new <code>SelectedTag</code> instance.
47     *
48     * @param tagID the id of the selected tag.
49     * @param tags an array containing the possible valid Tags.
50     * @throws IllegalArgumentException if the selected tag isn't in the array
51     * of valid values or the IDs/IDStrs are not unique.
52     */
53    public SelectedTag(int tagID, Tag[] tags) {
54      // are IDs unique?
55      HashSet<Integer> ID = new HashSet<Integer>();
56      HashSet<String> IDStr = new HashSet<String>();
57      for (int i = 0; i < tags.length; i++) {
58        ID.add(new Integer(tags[i].getID()));
59        IDStr.add(tags[i].getIDStr());
60      }
61      if (ID.size() != tags.length)
62        throw new IllegalArgumentException("The IDs are not unique!");
63      if (IDStr.size() != tags.length)
64        throw new IllegalArgumentException("The ID strings are not unique!");
65  
66      for (int i = 0; i < tags.length; i++) {
67        if (tags[i].getID() == tagID) {
68      m_Selected = i;
69      m_Tags = tags;
70      return;
71        }
72      }
73      
74      throw new IllegalArgumentException("Selected tag is not valid");
75    }
76    
77    /**
78     * Creates a new <code>SelectedTag</code> instance.
79     *
80     * @param tagText the text of the selected tag (case-insensitive).
81     * @param tags an array containing the possible valid Tags.
82     * @throws IllegalArgumentException if the selected tag isn't in the array
83     * of valid values.
84     */
85    public SelectedTag(String tagText, Tag[] tags) {
86      for (int i = 0; i < tags.length; i++) {
87        if (    tags[i].getReadable().equalsIgnoreCase(tagText)
88         || tags[i].getIDStr().equalsIgnoreCase(tagText) ) {
89          m_Selected = i;
90          m_Tags = tags;
91          return;
92        }
93      }
94      throw new IllegalArgumentException("Selected tag is not valid");
95    }
96    
97    /**
98     * Returns true if this SelectedTag equals another object
99     * 
100    * @param o the object to compare with
101    * @return true if the tags and the selected tag are the same
102    */
103   public boolean equals(Object o) {
104     if ((o == null) || !(o.getClass().equals(this.getClass()))) {
105       return false;
106     }
107     SelectedTag s = (SelectedTag)o;
108     if ((s.getTags() == m_Tags)
109     && (s.getSelectedTag() == m_Tags[m_Selected])) {
110       return true;
111     } else {
112       return false;
113     }
114   }
115   
116   
117   /**
118    * Gets the selected Tag.
119    *
120    * @return the selected Tag.
121    */
122   public Tag getSelectedTag() {
123     return m_Tags[m_Selected];
124   }
125   
126   /**
127    * Gets the set of all valid Tags.
128    *
129    * @return an array containing the valid Tags.
130    */
131   public Tag[] getTags() {
132     return m_Tags;
133   }
134   
135   /**
136    * returns the selected tag in string representation
137    * 
138    * @return the selected tag as string
139    */
140   public String toString() {
141     return getSelectedTag().toString();
142   }
143   
144   /**
145    * Returns the revision string.
146    * 
147    * @return        the revision
148    */
149   public String getRevision() {
150     return RevisionUtils.extract("$Revision: 5953 $");
151   }
152 }
