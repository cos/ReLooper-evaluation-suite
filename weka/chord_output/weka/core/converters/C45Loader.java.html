147   /**
148    * Resets the Loader object and sets the source of the data set to be 
149    * the supplied File object.
150    *
151    * @param file the source file.
152    * @exception IOException if an error occurs
153    */
154   public void setSource(File file) throws IOException {
155     m_structure = null;
156     setRetrieval(NONE);
157 
158     if (file == null) {
159       throw new IOException("Source file object is null!");
160     }
161 
162     String fname = file.getName();
163     String fileStem;
164     String path = file.getParent();
165     if (path != null) {
166       path += File.separator;
167     } else {
168       path = "";
169     }
170     if (fname.indexOf('.') < 0) {
171       fileStem = fname;
172       fname += ".names";
173     } else {
174       fileStem = fname.substring(0, fname.lastIndexOf('.'));
175       fname = fileStem + ".names";
176     }
177     m_fileStem = fileStem;
178     file = new File(path+fname);
179 
180     m_sourceFile = file;
181     try {
182       BufferedReader br = new BufferedReader(new FileReader(file));
183       m_namesReader = br;
184     } catch (FileNotFoundException ex) {
185       throw new IOException("File not found : "+(path+fname));
186     }
187 
188     m_sourceFileData = new File(path+fileStem+".data");
189     try {
190       BufferedReader br = new BufferedReader(new FileReader(m_sourceFileData));
191       m_dataReader = br;
192     } catch (FileNotFoundException ex) {
193       throw new IOException("File not found : "+(path+fname));
194     }
195     m_File = file.getAbsolutePath();
196   }

198   /**
199    * Determines and returns (if possible) the structure (internally the 
200    * header) of the data set as an empty set of instances.
201    *
202    * @return the structure of the data set as an empty set of Instances
203    * @exception IOException if an error occurs
204    */
205   public Instances getStructure() throws IOException {
206     if (m_sourceFile == null) {
207       throw new IOException("No source has beenspecified");
208     }
209 
210     if (m_structure == null) {
211       setSource(m_sourceFile);
212       StreamTokenizer st = new StreamTokenizer(m_namesReader);
213       initTokenizer(st);
214       readHeader(st);
215     }
216     
217     return m_structure;
218   }

220   /**
221    * Return the full data set. If the structure hasn't yet been determined
222    * by a call to getStructure then method should do so before processing
223    * the rest of the data set.
224    *
225    * @return the structure of the data set as an empty set of Instances
226    * @exception IOException if there is no source or parsing fails
227    */
228   public Instances getDataSet() throws IOException {
229     if (m_sourceFile == null) {
230       throw new IOException("No source has been specified");
231     }
232     if (getRetrieval() == INCREMENTAL) {
233       throw new IOException("Cannot mix getting Instances in both incremental and batch modes");
234     }
235     setRetrieval(BATCH);
236     if (m_structure == null) {
237       getStructure();
238     }
239     StreamTokenizer st = new StreamTokenizer(m_dataReader);
240     initTokenizer(st);
241     //    st.ordinaryChar('.');
242     Instances result = new Instances(m_structure);
243     Instance current = getInstance(st);
244 
245     while (current != null) {
246       result.add(current);
247       current = getInstance(st);
248     }
249     try {
250       // close the stream
251       m_dataReader.close();
252       //      reset();
253     } catch (Exception ex) {
254       ex.printStackTrace();
255     }
256     return result;
257   }

259   /**
260    * Read the data set incrementally---get the next instance in the data 
261    * set or returns null if there are no
262    * more instances to get. If the structure hasn't yet been 
263    * determined by a call to getStructure then method should do so before
264    * returning the next instance in the data set.
265    *
266    * If it is not possible to read the data set incrementally (ie. in cases
267    * where the data set structure cannot be fully established before all
268    * instances have been seen) then an exception should be thrown.
269    *
270    * @param structure the dataset header information, will get updated in 
271    * case of string or relational attributes
272    * @return the next instance in the data set as an Instance object or null
273    * if there are no more instances to be read
274    * @exception IOException if there is an error during parsing
275    */
276   public Instance getNextInstance(Instances structure) throws IOException {
277     if (m_sourceFile == null) {
278       throw new IOException("No source has been specified");
279     }
280     
281     if (getRetrieval() == BATCH) {
282       throw new IOException("Cannot mix getting Instances in both incremental and batch modes");
283     }
284     setRetrieval(INCREMENTAL);
285 
286     if (m_structure == null) {
287       getStructure();
288     }
289 
290     StreamTokenizer st = new StreamTokenizer(m_dataReader);
291     initTokenizer(st);
292     //    st.ordinaryChar('.');
293     Instance nextI = getInstance(st);
294     if (nextI != null) {
295       nextI.setDataset(m_structure);
296     }
297     else{
298       try {
299         // close the stream
300         m_dataReader.close();
301         //        reset();
302       } catch (Exception ex) {
303         ex.printStackTrace();
304       }
305     }
306     return nextI;
307   }

309   /**
310    * Reads an instance using the supplied tokenizer.
311    *
312    * @param tokenizer the tokenizer to use
313    * @return an Instance or null if there are no more instances to read
314    * @exception IOException if an error occurs
315    */
316   private Instance getInstance(StreamTokenizer tokenizer) 
317     throws IOException {
318     double [] instance = new double[m_structure.numAttributes()];
319     
320     ConverterUtils.getFirstToken(tokenizer);
321     if (tokenizer.ttype == StreamTokenizer.TT_EOF) {
322       return null;
323     }
324     
325     int counter = 0;
326     for (int i = 0; i < m_numAttribs; i++) {
327       if (i > 0) {
328     ConverterUtils.getToken(tokenizer);
329       }
330 
331       if (!m_ignore[i]) {
332     // Check if value is missing.
333     if  (tokenizer.ttype == '?') {
334       instance[counter++] = Utils.missingValue();
335     } else {
336       String val = tokenizer.sval;
337 
338       if (i == m_numAttribs - 1) {
339         // remove trailing period       
340         if (val.charAt(val.length()-1) == '.') {
341           val = val.substring(0,val.length()-1);
342         }
343       }
344       if (m_structure.attribute(counter).isNominal()) {
345         int index = m_structure.attribute(counter).indexOfValue(val);
346         if (index == -1) {
347           ConverterUtils.errms(tokenizer, "nominal value not declared in "
348                    +"header :"+val+" column "+i);
349         }
350         instance[counter++] = (double)index;
351       } else if (m_structure.attribute(counter).isNumeric()) {
352         try {
353           instance[counter++] = Double.valueOf(val).doubleValue();
354         } catch (NumberFormatException e) {
355           ConverterUtils.errms(tokenizer, "number expected");
356         }
357       } else {
358         System.err.println("Shouldn't get here");
359         System.exit(1);
360       }
361     }
362       }
363     }
364 
365     return new DenseInstance(1.0, instance);
366   }

368   /**
369    * removes the trailing period
370    * 
371    * @param val the string to work on
372    * @return the processed string
373    */
374   private String removeTrailingPeriod(String val) {
375     // remove trailing period
376     if (val.charAt(val.length()-1) == '.') {
377       val = val.substring(0,val.length()-1);
378     }
379     return val;
380   }

382   /**
383    * Reads header (from the names file) using the supplied tokenizer
384    *
385    * @param tokenizer the tokenizer to use
386    * @exception IOException if an error occurs
387    */
388   private void readHeader(StreamTokenizer tokenizer) throws IOException {
389 
390     ArrayList<Attribute> attribDefs = new ArrayList<Attribute>();
391     ArrayList<Integer> ignores = new ArrayList<Integer>();
392     ConverterUtils.getFirstToken(tokenizer);
393     if (tokenizer.ttype == StreamTokenizer.TT_EOF) {
394       ConverterUtils.errms(tokenizer,"premature end of file");
395     }
396 
397     m_numAttribs = 1;
398     // Read the class values
399     ArrayList<String> classVals = new ArrayList<String>();
400     while (tokenizer.ttype != StreamTokenizer.TT_EOL) {
401       String val = tokenizer.sval.trim();
402       
403       if (val.length() > 0) {
404     val = removeTrailingPeriod(val);
405     classVals.add(val);
406       }
407       ConverterUtils.getToken(tokenizer);
408     }
409 
410     // read the attribute names and types
411     int counter = 0;
412     while (tokenizer.ttype != StreamTokenizer.TT_EOF) {
413       ConverterUtils.getFirstToken(tokenizer);
414       if (tokenizer.ttype != StreamTokenizer.TT_EOF) {
415 
416     String attribName = tokenizer.sval;
417 
418     ConverterUtils.getToken(tokenizer);
419     if (tokenizer.ttype == StreamTokenizer.TT_EOL) {
420       ConverterUtils.errms(tokenizer, "premature end of line. Expected "
421                    +"attribute type.");
422     }
423     String temp = tokenizer.sval.toLowerCase().trim();
424     if (temp.startsWith("ignore") || temp.startsWith("label")) {
425       ignores.add(new Integer(counter));
426       counter++;
427     } else if (temp.startsWith("continuous")) {
428       attribDefs.add(new Attribute(attribName));
429       counter++;
430     } else {
431       counter++;
432       // read the values of the attribute
433       ArrayList<String> attribVals = new ArrayList<String>();
434       while (tokenizer.ttype != StreamTokenizer.TT_EOL &&
435          tokenizer.ttype != StreamTokenizer.TT_EOF) {
436         String val = tokenizer.sval.trim();
437 
438         if (val.length() > 0) {
439           val = removeTrailingPeriod(val);
440           attribVals.add(val);
441         }
442         ConverterUtils.getToken(tokenizer);
443       }
444       attribDefs.add(new Attribute(attribName, attribVals));
445     }
446       }
447     }
448 
449     boolean ok = true;
450     int i = -1;
451     if (classVals.size() == 1) {
452       // look to see if this is an attribute name (ala c5 names file style)
453       for (i = 0; i < attribDefs.size(); i++) {
454     if (((Attribute)attribDefs.get(i))
455         .name().compareTo((String)classVals.get(0)) == 0) {
456       ok = false;
457       m_numAttribs--;
458       break;
459     }
460       }
461     }
462 
463     if (ok) {
464       attribDefs.add(new Attribute("Class", classVals));
465     }
466 
467     m_structure = new Instances(m_fileStem, attribDefs, 0);
468 
469     try {
470       if (ok) {
471     m_structure.setClassIndex(m_structure.numAttributes()-1);
472       } else {
473     m_structure.setClassIndex(i);
474       }
475     } catch (Exception ex) {
476       ex.printStackTrace();
477     }
478 
479     m_numAttribs = m_structure.numAttributes() + ignores.size();
480     m_ignore = new boolean[m_numAttribs];
481     for (i = 0; i < ignores.size(); i++) {
482       m_ignore[((Integer)ignores.get(i)).intValue()] = true;
483     }
484   }

486   /**
487    * Initializes the stream tokenizer
488    *
489    * @param tokenizer the tokenizer to initialize
490    */
491   private void initTokenizer(StreamTokenizer tokenizer) {
492     tokenizer.resetSyntax();         
493     tokenizer.whitespaceChars(0, (' '-1));    
494     tokenizer.wordChars('\u00FF','ÿ');
495     tokenizer.whitespaceChars(',',',');
496     tokenizer.whitespaceChars(':',':');
497     //    tokenizer.whitespaceChars('.','.');
498     tokenizer.commentChar('|');
499     tokenizer.whitespaceChars('\t','\t');
500     tokenizer.quoteChar('"');
501     tokenizer.quoteChar('\'');
502     tokenizer.eolIsSignificant(true);
503   }
