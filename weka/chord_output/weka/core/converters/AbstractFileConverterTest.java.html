122   /**
123    * returns the command line options, either for the loader or the saver.
124    * 
125    * @param loader  if true the options for the loader will be returned,
126    *            otherwise the ones for the saver

136   /**
137    * Compare two datasets to see if they differ.
138    *
139    * @param data1 one set of instances
140    * @param data2 the other set of instances
141    * @throws Exception if the datasets differ
142    */
143   protected void compareDatasets(Instances data1, Instances data2)
144     throws Exception {
145     
146     if (m_CheckHeader) {
147       if (!data2.equalHeaders(data1)) {
148     throw new Exception("header has been modified\n" + data2.equalHeadersMsg(data1));
149       }
150     }
151     if (!(data2.numInstances() == data1.numInstances())) {
152       throw new Exception("number of instances has changed");
153     }
154     for (int i = 0; i < data2.numInstances(); i++) {
155       Instance orig = data1.instance(i);
156       Instance copy = data2.instance(i);
157       for (int j = 0; j < orig.numAttributes(); j++) {
158         if (orig.isMissing(j)) {
159           if (!copy.isMissing(j)) {
160             throw new Exception("instances have changed");
161           }
162         } else {
163           if (m_CompareValuesAsString) {
164             if (!orig.toString(j).equals(copy.toString(j))) {
165               throw new Exception("instances have changed");
166             }
167           } else {
168             if (Math.abs(orig.value(j) - copy.value(j)) > m_MaxDiffValues) {
169               throw new Exception("instances have changed");
170             }
171           }
172         }
173         if (Math.abs(orig.weight() - copy.weight()) > m_MaxDiffWeights) {
174           throw new Exception("instance weights have changed");
175         }     
176       }
177     }
178   }

180   /**
181    * Called by JUnit before each test method. This implementation creates
182    * the default loader/saver to test and generates a test set of Instances.
183    *
184    * @throws Exception if an error occurs reading the example instances.
185    */
186   protected void setUp() throws Exception {
187     BufferedWriter  writer;
188     
189     super.setUp();
190     
191     // how to compare datasets
192     // see compareDatasets(Instances,Instances)
193     m_MaxDiffValues         = 0.0;
194     m_MaxDiffWeights        = 0.0;
195     m_CheckHeader           = true;
196     m_CompareValuesAsString = false;
197     
198     m_SourceFilename = getSourceFilename();
199     m_ExportFilename = getExportFilename();
200     
201     // generate source file
202     writer = new BufferedWriter(new FileWriter(m_SourceFilename));
203     writer.write(m_Instances.toString());
204     writer.newLine();
205     writer.flush();
206     writer.close();
207   }
208 
209   /** 
210    * Called by JUnit after each test method.
211    * 
212    * @throws Exception  if fails
213    */
214   protected void tearDown() throws Exception {
215     File    file;
216 
217     // delete temp. files
218     file = new File(m_SourceFilename);
219     if (file.exists())
220       file.delete();
221     file = new File(m_ExportFilename);
222     if (file.exists())
223       file.delete();
224     
225     m_SourceFilename = null;
226     m_ExportFilename = null;
227     
228     super.tearDown();
229   }

231   /**
232    * test the batch saving/loading (via setFile(File)).
233    */
234   public void testBatch() {
235     Instances       data;
236 
237     try {
238       // save
239       m_Saver.setInstances(m_Instances);
240       m_Saver.setFile(new File(m_ExportFilename));
241       m_Saver.writeBatch();
242       
243       // load
244       ((AbstractFileLoader) m_Loader).setFile(new File(m_ExportFilename));
245       data = m_Loader.getDataSet();
246       
247       // compare data
248       try {
249     if (m_Instances.classIndex() != data.classIndex())
250       data.setClassIndex(m_Instances.classIndex());
251     compareDatasets(m_Instances, data);
252       }
253       catch (Exception e) {
254     fail("Incremental load failed (datasets differ): " + e.toString());
255       }
256     }
257     catch (Exception e) {
258       e.printStackTrace();
259       fail("Batch save/load failed: " + e.toString());
260     }
261   }

263   /**
264    * test the incremental loading (via setFile(File)).
265    */
266   public void testIncrementalLoader() {
267     Instance    temp;
268     Instances   data;
269     
270     if (!(getLoader() instanceof IncrementalConverter))
271       return;
272 
273     try {
274       // save
275       m_Saver.setInstances(m_Instances);
276       m_Saver.setFile(new File(m_ExportFilename));
277       m_Saver.writeBatch();
278 
279       // load
280       ((AbstractFileLoader) m_Loader).setFile(new File(m_ExportFilename));
281       data = new Instances(m_Loader.getStructure());
282       do {
283     temp = m_Loader.getNextInstance(data);
284     if (temp != null)
285       data.add(temp);
286       }
287       while (temp != null);
288       
289       // compare data
290       try {
291     if (m_Instances.classIndex() != data.classIndex())
292       data.setClassIndex(m_Instances.classIndex());
293     compareDatasets(m_Instances, data);
294       }
295       catch (Exception e) {
296     fail("Incremental load failed (datasets differ): " + e.toString());
297       }
298     }
299     catch (Exception e) {
300       e.printStackTrace();
301       fail("Incremental load failed: " + e.toString());
302     }
303   }

305   /**
306    * test the incremental save (via setFile(File)).
307    */
308   public void testIncrementalSaver() {
309     int     i;
310     File    file;
311 
312     if (!(getSaver() instanceof IncrementalConverter))
313       return;
314     
315     try {
316       // remove output file if it exists
317       file = new File(m_ExportFilename);
318       if (file.exists())
319     file.delete();
320       
321       // save
322       m_Saver.setFile(new File(m_ExportFilename));
323       m_Saver.setRetrieval(AbstractSaver.INCREMENTAL);
324       m_Saver.setStructure(new Instances(m_Instances, 0));
325       for (i = 0; i < m_Instances.numInstances(); i++) {
326     m_Saver.writeIncremental(m_Instances.instance(i));
327       }
328       m_Saver.writeIncremental(null);
329     }
330     catch (Exception e) {
331       e.printStackTrace();
332       fail("Incremental save failed: " + e.toString());
333     }
334   }
