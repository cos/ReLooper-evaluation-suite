186   /** 
187    * Cancels the incremental saving process and tries to drop the table if 
188    * the write mode is CANCEL.
189    */  
190   public void cancel(){
191   
192       if(getWriteMode() == CANCEL){
193           try{
194               m_DataBaseConnection.update("DROP TABLE "+m_tableName);
195               if(m_DataBaseConnection.tableExists(m_tableName))
196                 System.err.println("Table cannot be dropped.");
197           }catch(Exception ex) {
198               printException(ex);
199           }
200           resetOptions();
201       }
202   }

483   /** 
484    * Writes the structure (header information) to a database by creating a new table.
485    * 
486    * @throws Exception if something goes wrong
487    */
488   private void writeStructure() throws Exception{
489   
490       StringBuffer query = new StringBuffer();
491       Instances structure = getInstances();
492       query.append("CREATE TABLE ");
493       if(m_tabName || m_tableName.equals(""))
494         m_tableName = m_DataBaseConnection.maskKeyword(structure.relationName());
495       if(m_DataBaseConnection.getUpperCase()){
496         m_tableName = m_tableName.toUpperCase();
497         m_createInt = m_createInt.toUpperCase(); 
498         m_createDouble = m_createDouble.toUpperCase(); 
499         m_createText = m_createText.toUpperCase(); 
500         m_createDate = m_createDate.toUpperCase(); 
501       }
502       m_tableName = m_tableName.replaceAll("[^\\w]","_");
503       m_tableName = m_DataBaseConnection.maskKeyword(m_tableName);
504       query.append(m_tableName);
505       if(structure.numAttributes() == 0)
506           throw new Exception("Instances have no attribute.");
507       query.append(" ( ");
508       if(m_id){
509         if(m_DataBaseConnection.getUpperCase())
510               m_idColumn = m_idColumn.toUpperCase();
511         query.append(m_DataBaseConnection.maskKeyword(m_idColumn));
512         query.append(" ");
513         query.append(m_createInt);
514         query.append(" PRIMARY KEY,");
515       }
516       for(int i = 0;i < structure.numAttributes(); i++){
517           Attribute att = structure.attribute(i);
518           String attName = att.name();
519           attName = attName.replaceAll("[^\\w]","_");
520           attName = m_DataBaseConnection.maskKeyword(attName);
521           if(m_DataBaseConnection.getUpperCase())
522               query.append(attName.toUpperCase());
523           else
524               query.append(attName);
525           if(att.isDate())
526               query.append(" " + m_createDate);
527           else{
528               if(att.isNumeric())
529                   query.append(" "+m_createDouble);
530               else
531                   query.append(" "+m_createText);
532           }
533           if(i != structure.numAttributes()-1)
534               query.append(", ");
535       }
536       query.append(" )");
537       //System.out.println(query.toString());
538       m_DataBaseConnection.update(query.toString());
539       m_DataBaseConnection.close();
540       if(!m_DataBaseConnection.tableExists(m_tableName)){
541           throw new IOException("Table cannot be built.");
542       }
543   }

545   /**
546    * inserts the given instance into the table.
547    * 
548    * @param inst the instance to insert
549    * @throws Exception if something goes wrong
550    */
551   private void writeInstance(Instance inst) throws Exception{
552   
553       StringBuffer insert = new StringBuffer();
554       insert.append("INSERT INTO ");
555       insert.append(m_tableName);
556       insert.append(" VALUES ( ");
557       if(m_id){
558         insert.append(m_count);
559         insert.append(", ");
560         m_count++;
561       }
562       for(int j = 0; j < inst.numAttributes(); j++){
563         if(inst.isMissing(j))
564             insert.append("NULL");
565         else{
566             if((inst.attribute(j)).isDate())
567                 insert.append("'" + m_DateFormat.format((long) inst.value(j)) + "'");
568             else if((inst.attribute(j)).isNumeric())
569                 insert.append(inst.value(j));
570             else{
571                 String stringInsert = "'"+inst.stringValue(j)+"'";
572                 if (stringInsert.length() > 2)
573                   stringInsert = stringInsert.replaceAll("''","'");
574                 insert.append(stringInsert);
575             }
576         }
577         if(j != inst.numAttributes()-1)
578             insert.append(", ");
579       }
580       insert.append(" )");
581       //System.out.println(insert.toString());
582       if (m_DataBaseConnection.update(insert.toString()) < 1) {
583         throw new IOException("Tuple cannot be inserted.");
584       }
585       else {
586     m_DataBaseConnection.close();
587       }
588   }

590   /** 
591    * Saves an instances incrementally. Structure has to be set by using the
592    * setStructure() method or setInstances() method. When a structure is set, a table is created. 
593    * 
594    * @param inst the instance to save
595    * @throws IOException throws IOEXception.
596    */  
597   public void writeIncremental(Instance inst) throws IOException{
598   
599       int writeMode = getWriteMode();
600       Instances structure = getInstances();
601       
602       if(m_DataBaseConnection == null)
603            throw new IOException("No database has been set up.");
604       if(getRetrieval() == BATCH)
605           throw new IOException("Batch and incremental saving cannot be mixed.");
606       setRetrieval(INCREMENTAL);
607       
608       try{
609         if(!m_DataBaseConnection.isConnected())
610             connectToDatabase();
611         if(writeMode == WAIT){
612             if(structure == null){
613                 setWriteMode(CANCEL);
614                 if(inst != null)
615                     throw new Exception("Structure(Header Information) has to be set in advance");
616             }
617             else
618                 setWriteMode(STRUCTURE_READY);
619             writeMode = getWriteMode();
620         }
621         if(writeMode == CANCEL){
622           cancel();
623         }
624         if(writeMode == STRUCTURE_READY){
625           setWriteMode(WRITE);
626           writeStructure();
627           writeMode = getWriteMode();
628         }
629         if(writeMode == WRITE){
630           if(structure == null)
631               throw new IOException("No instances information available.");
632           if(inst != null){
633             //write instance 
634             writeInstance(inst);  
635           }
636           else{
637           //close
638               m_DataBaseConnection.disconnectFromDatabase();
639               resetStructure();
640               m_count = 1;
641           }
642         }
643       }catch(Exception ex) {
644             printException(ex);
645        }    
646   }

648   /** 
649    * Writes a Batch of instances.
650    * 
651    * @throws IOException throws IOException
652    */
653   public void writeBatch() throws IOException {
654   
655       Instances instances = getInstances();
656       if(instances == null)
657           throw new IOException("No instances to save");
658       if(getRetrieval() == INCREMENTAL)
659           throw new IOException("Batch and incremental saving cannot be mixed.");
660       if(m_DataBaseConnection == null)
661            throw new IOException("No database has been set up.");
662       setRetrieval(BATCH);
663       try{
664           if(!m_DataBaseConnection.isConnected())
665               connectToDatabase();
666           setWriteMode(WRITE);
667           writeStructure();
668           for(int i = 0; i < instances.numInstances(); i++){
669             writeInstance(instances.instance(i));
670           }
671           m_DataBaseConnection.disconnectFromDatabase();
672           setWriteMode(WAIT);
673           resetStructure();
674           m_count = 1;
675       } catch(Exception ex) {
676             printException(ex);
677        }    
678   }

680   /**
681    * Prints an exception.
682    * 
683    * @param ex the exception to print
684    */  
685   private void printException(Exception ex){
686   
687       System.out.println("\n--- Exception caught ---\n");
688     while (ex != null) {
689         System.out.println("Message:   "
690                                    + ex.getMessage ());
691                 if(ex instanceof SQLException){
692                     System.out.println("SQLState:  "
693                                    + ((SQLException)ex).getSQLState ());
694                     System.out.println("ErrorCode: "
695                                    + ((SQLException)ex).getErrorCode ());
696                     ex = ((SQLException)ex).getNextException();
697                 }
698                 else
699                     ex = null;
700         System.out.println("");
701     }
702       
703   }

705   /** 
706    * Gets the setting.
707    * 
708    * @return the current setting
709    */  
710   public String[] getOptions() {
711     Vector<String> options = new Vector<String>();
712 
713     if ( (getUrl() != null) && (getUrl().length() != 0) ) {
714       options.add("-url");
715       options.add(getUrl());
716     }
717 
718     if ( (getUser() != null) && (getUser().length() != 0) ) {
719       options.add("-user");
720       options.add(getUser());
721     }
722 
723     if ( (getPassword() != null) && (getPassword().length() != 0) ) {
724       options.add("-password");
725       options.add(getPassword());
726     }
727 
728     if ( (m_tableName != null) && (m_tableName.length() != 0) ) {
729       options.add("-T"); 
730       options.add(m_tableName);
731     }
732     
733     if (m_id)
734         options.add("-P");
735 
736     if ( (m_inputFile != null) && (m_inputFile.length() != 0) ) {
737       options.add("-i"); 
738       options.add(m_inputFile);
739     }
740     
741     return (String[]) options.toArray(new String[options.size()]);
742   }

792    * <pre> -url &lt;JDBC URL&gt;
793    *  The JDBC URL to connect to.
794    *  (default: from DatabaseUtils.props file)</pre>
795    * 
796    * <pre> -user &lt;name&gt;
797    *  The user to connect with to the database.
798    *  (default: none)</pre>
799    * 
800    * <pre> -password &lt;password&gt;
801    *  The password to connect with to the database.
802    *  (default: none)</pre>
803    * 
804    * <pre> -T &lt;table name&gt;
805    *  The name of the table.
806    *  (default: the relation name)</pre>
807    * 
808    * <pre> -P
809    *  Add an ID column as primary key. The name is specified
810    *  in the DatabaseUtils file ('idColumn'). The DatabaseLoader
811    *  won't load this column.</pre>
812    * 
813    * <pre> -i &lt;input file name&gt;
814    *  Input file in arff format that should be saved in database.</pre>
815    * 
816    <!-- options-end -->
817    *
818    * @param options the options
819    * @throws Exception if options cannot be set
820    */  
821   public void setOptions(String[] options) throws Exception {
822       
823     String tableString, inputString, tmpStr;
824     
825     resetOptions();
826 
827     tmpStr = Utils.getOption("url", options);
828     if (tmpStr.length() != 0)
829       setUrl(tmpStr);
830 
831     tmpStr = Utils.getOption("user", options);
832     if (tmpStr.length() != 0)
833       setUser(tmpStr);
834 
835     tmpStr = Utils.getOption("password", options);
836     if (tmpStr.length() != 0)
837       setPassword(tmpStr);
838     
839     tableString = Utils.getOption('T', options);
840     
841     inputString = Utils.getOption('i', options);
842     
843     if(tableString.length() != 0){
844         m_tableName = tableString;
845         m_tabName = false;
846     }
847     
848     m_id = Utils.getFlag('P', options);
849     
850     if(inputString.length() != 0){
851         try{
852             m_inputFile = inputString;
853             ArffLoader al = new ArffLoader();
854             File inputFile = new File(inputString);
855             al.setSource(inputFile);
856             setInstances(al.getDataSet());
857             //System.out.println(getInstances());
858             if(tableString.length() == 0)
859                 m_tableName = getInstances().relationName();
860         }catch(Exception ex) {
861             printException(ex);
862             ex.printStackTrace();
863       }    
864     }
865   }
