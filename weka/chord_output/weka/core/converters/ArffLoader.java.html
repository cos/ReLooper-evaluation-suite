77    /**
78     * Reads data from an ARFF file, either in incremental or batch mode. <p/>
79     * 
80     * Typical code for batch usage:
81     * <pre>
82     * BufferedReader reader = new BufferedReader(new FileReader("/some/where/file.arff"));
83     * ArffReader arff = new ArffReader(reader);
84     * Instances data = arff.getData();
85     * data.setClassIndex(data.numAttributes() - 1);
86     * </pre>
87     * 
88     * Typical code for incremental usage:
89     * <pre>
90     * BufferedReader reader = new BufferedReader(new FileReader("/some/where/file.arff"));
91     * ArffReader arff = new ArffReader(reader, 1000);
92     * Instances data = arff.getStructure();
93     * data.setClassIndex(data.numAttributes() - 1);
94     * Instance inst;
95     * while ((inst = arff.readInstance(data)) != null) {
96     *   data.add(inst);
97     * }
98     * </pre>
99     * 
100    * @author  Eibe Frank (eibe@cs.waikato.ac.nz)
101    * @author  Len Trigg (trigg@cs.waikato.ac.nz)
102    * @author  fracpete (fracpete at waikato dot ac dot nz)
103    * @version $Revision: 5987 $
104    */

123     /**
124      * Reads the data completely from the reader. The data can be accessed
125      * via the <code>getData()</code> method.
126      * 
127      * @param reader        the reader to use
128      * @throws IOException  if something goes wrong
129      * @see         #getData()
130      */
131     public ArffReader(Reader reader) throws IOException {
132       m_Tokenizer = new StreamTokenizer(reader);
133       initTokenizer();
134 
135       readHeader(1000);
136       initBuffers();
137       
138       Instance inst;
139       while ((inst = readInstance(m_Data)) != null) {
140         m_Data.add(inst);
141       };
142       
143       compactify();
144     }

165       initBuffers();
166     }

229     /**
230      * Throws error message with line number and last token read.
231      *
232      * @param msg       the error message to be thrown
233      * @throws IOException  containing the error message
234      */
235     protected void errorMessage(String msg) throws IOException {
236       String str = msg + ", read " + m_Tokenizer.toString();
237       if (m_Lines > 0) {
238     int line = Integer.parseInt(str.replaceAll(".* line ", ""));
239     str = str.replaceAll(" line .*", " line " + (m_Lines + line - 1));
240       }
241       throw new IOException(str);
242     }

253     /**
254      * Gets next token, skipping empty lines.
255      *
256      * @throws IOException  if reading the next token fails
257      */
258     protected void getFirstToken() throws IOException {
259       while (m_Tokenizer.nextToken() == StreamTokenizer.TT_EOL) {};
260       
261       if ((m_Tokenizer.ttype == '\'') ||
262     (m_Tokenizer.ttype == '"')) {
263         m_Tokenizer.ttype = StreamTokenizer.TT_WORD;
264       } else if ((m_Tokenizer.ttype == StreamTokenizer.TT_WORD) &&
265            (m_Tokenizer.sval.equals("?"))){
266         m_Tokenizer.ttype = '?';
267       }
268     }
269 
270     /**
271      * Gets index, checking for a premature and of line.
272      *
273      * @throws IOException  if it finds a premature end of line
274      */
275     protected void getIndex() throws IOException {
276       if (m_Tokenizer.nextToken() == StreamTokenizer.TT_EOL) {
277         errorMessage("premature end of line");
278       }
279       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
280         errorMessage("premature end of file");
281       }
282     }

284     /**
285      * Gets token and checks if its end of line.
286      *
287      * @param endOfFileOk   whether EOF is OK
288      * @throws IOException  if it doesn't find an end of line
289      */
290     protected void getLastToken(boolean endOfFileOk) throws IOException {
291       if ((m_Tokenizer.nextToken() != StreamTokenizer.TT_EOL) &&
292     ((m_Tokenizer.ttype != StreamTokenizer.TT_EOF) || !endOfFileOk)) {
293         errorMessage("end of line expected");
294       }
295     }

297     /**
298      * Gets the value of an instance's weight (if one exists)
299      *
300      * @return the value of the instance's weight, or NaN
301      * if no weight has been supplied in the file
302      */
303     protected double getInstanceWeight() throws IOException {
304       double weight = Double.NaN;
305       m_Tokenizer.nextToken();
306       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOL || 
307           m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
308         return weight;
309       }
310       // see if we can read an instance weight
311       //      m_Tokenizer.pushBack();
312       if (m_Tokenizer.ttype == '{') {
313         m_Tokenizer.nextToken();
314         String weightS = m_Tokenizer.sval;
315         // try to parse weight as a double
316         try {
317           weight = Double.parseDouble(weightS);
318         } catch (NumberFormatException e) {
319           // quietly ignore
320           return weight;
321         }
322         // see if we have the closing brace
323         m_Tokenizer.nextToken();
324         if (m_Tokenizer.ttype != '}') {
325           errorMessage("Problem reading instance weight");
326         }
327       }
328       return weight;
329     }

331     /**
332      * Gets next token, checking for a premature and of line.
333      *
334      * @throws IOException  if it finds a premature end of line
335      */
336     protected void getNextToken() throws IOException {
337       if (m_Tokenizer.nextToken() == StreamTokenizer.TT_EOL) {
338         errorMessage("premature end of line");
339       }
340       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
341         errorMessage("premature end of file");
342       } else if ((m_Tokenizer.ttype == '\'') ||
343            (m_Tokenizer.ttype == '"')) {
344         m_Tokenizer.ttype = StreamTokenizer.TT_WORD;
345       } else if ((m_Tokenizer.ttype == StreamTokenizer.TT_WORD) &&
346            (m_Tokenizer.sval.equals("?"))){
347         m_Tokenizer.ttype = '?';
348       }
349     }

351     /**
352      * Initializes the StreamTokenizer used for reading the ARFF file.
353      */
354     protected void initTokenizer(){
355       m_Tokenizer.resetSyntax();         
356       m_Tokenizer.whitespaceChars(0, ' ');    
357       m_Tokenizer.wordChars(' '\u00FF1,'Ã¿');
358       m_Tokenizer.whitespaceChars(',',',');
359       m_Tokenizer.commentChar('%');
360       m_Tokenizer.quoteChar('"');
361       m_Tokenizer.quoteChar('\'');
362       m_Tokenizer.ordinaryChar('{');
363       m_Tokenizer.ordinaryChar('}');
364       m_Tokenizer.eolIsSignificant(true);
365     }

395     /**
396      * Reads a single instance using the tokenizer and returns it. 
397      *
398      * @param structure     the dataset header information, will get updated 
399      *              in case of string or relational attributes
400      * @param flag      if method should test for carriage return after 
401      *              each instance
402      * @return          null if end of file has been reached
403      * @throws IOException  if the information is not read 
404      *              successfully
405      */ 
406     protected Instance getInstance(Instances structure, boolean flag) throws IOException {
407       m_Data = structure;
408       
409       // Check if any attributes have been declared.
410       if (m_Data.numAttributes() == 0) {
411         errorMessage("no header information available");
412       }
413 
414       // Check if end of file reached.
415       getFirstToken();
416       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
417         return null;
418       }
419       
420       // Parse instance
421       if (m_Tokenizer.ttype == '{') {
422         return getInstanceSparse(flag);
423       } else {
424         return getInstanceFull(flag);
425       }
426     }
427 
428     /**
429      * Reads a single instance using the tokenizer and returns it.
430      *
431      * @param flag      if method should test for carriage return after 
432      *              each instance
433      * @return          null if end of file has been reached
434      * @throws IOException  if the information is not read 
435      *              successfully
436      */ 
437     protected Instance getInstanceSparse(boolean flag) throws IOException {
438       int valIndex, numValues = 0, maxIndex = -1;
439       
440       // Get values
441       do {
442         // Get index
443         getIndex();
444         if (m_Tokenizer.ttype == '}') {
445     break;
446         }
447    
448         // Is index valid?
449         try{
450     m_IndicesBuffer[numValues] = Integer.valueOf(m_Tokenizer.sval).intValue();
451         } catch (NumberFormatException e) {
452     errorMessage("index number expected");
453         }
454         if (m_IndicesBuffer[numValues] <= maxIndex) {
455     errorMessage("indices have to be ordered");
456         }
457         if ((m_IndicesBuffer[numValues] < 0) || 
458       (m_IndicesBuffer[numValues] >= m_Data.numAttributes())) {
459     errorMessage("index out of bounds");
460         }
461         maxIndex = m_IndicesBuffer[numValues];
462 
463         // Get value;
464         getNextToken();
465 
466         // Check if value is missing.
467         if  (m_Tokenizer.ttype == '?') {
468           m_ValueBuffer[numValues] = Utils.missingValue();
469         } else {
470 
471     // Check if token is valid.
472     if (m_Tokenizer.ttype != StreamTokenizer.TT_WORD) {
473       errorMessage("not a valid value");
474     }
475           switch (m_Data.attribute(m_IndicesBuffer[numValues]).type()) {
476             case Attribute.NOMINAL:
477               // Check if value appears in header.
478               valIndex = 
479                 m_Data.attribute(m_IndicesBuffer[numValues]).indexOfValue(m_Tokenizer.sval);
480               if (valIndex == -1) {
481                 errorMessage("nominal value not declared in header");
482               }
483               m_ValueBuffer[numValues] = (double)valIndex;
484               break;
485     case Attribute.NUMERIC:
486       // Check if value is really a number.
487       try{
488         m_ValueBuffer[numValues] = Double.valueOf(m_Tokenizer.sval).
489           doubleValue();
490       } catch (NumberFormatException e) {
491         errorMessage("number expected");
492       }
493             break;
494     case Attribute.STRING:
495       m_ValueBuffer[numValues] = 
496         m_Data.attribute(m_IndicesBuffer[numValues]).addStringValue(m_Tokenizer.sval);
497             break;
498           case Attribute.DATE:
499             try {
500               m_ValueBuffer[numValues] = 
501                 m_Data.attribute(m_IndicesBuffer[numValues]).parseDate(m_Tokenizer.sval);
502             } catch (ParseException e) {
503               errorMessage("unparseable date: " + m_Tokenizer.sval);
504             }
505             break;
506           case Attribute.RELATIONAL:
507             try {
508               ArffReader arff = new ArffReader(new StringReader(m_Tokenizer.sval), m_Data.attribute(m_IndicesBuffer[numValues]).relation(), 0);
509               Instances data = arff.getData();
510               m_ValueBuffer[numValues] = m_Data.attribute(m_IndicesBuffer[numValues]).addRelation(data);
511             }
512             catch (Exception e) {
513               throw new IOException(e.toString() + " of line " + getLineNo());
514             }
515             break;
516           default:
517             errorMessage("unknown attribute type in column " + m_IndicesBuffer[numValues]);
518     }
519         }
520         numValues++;
521       } while (true);
522 
523       double weight = 1.0;
524       if (flag) {
525         // check for an instance weight
526         weight = getInstanceWeight();
527         if (!Double.isNaN(weight)) {
528           getLastToken(true);
529         } else {
530           weight = 1.0;
531         }        
532       }
533         
534       // Add instance to dataset
535       double[] tempValues = new double[numValues];
536       int[] tempIndices = new int[numValues];
537       System.arraycopy(m_ValueBuffer, 0, tempValues, 0, numValues);
538       System.arraycopy(m_IndicesBuffer, 0, tempIndices, 0, numValues);
539       Instance inst = new SparseInstance(weight, tempValues, tempIndices, m_Data.numAttributes());
540       inst.setDataset(m_Data);
541       
542       return inst;
543     }

545     /**
546      * Reads a single instance using the tokenizer and returns it.
547      *
548      * @param flag      if method should test for carriage return after 
549      *              each instance
550      * @return          null if end of file has been reached
551      * @throws IOException  if the information is not read 
552      *              successfully
553      */ 
554     protected Instance getInstanceFull(boolean flag) throws IOException {
555       double[] instance = new double[m_Data.numAttributes()];
556       int index;
557       
558       // Get values for all attributes.
559       for (int i = 0; i < m_Data.numAttributes(); i++){
560         // Get next token
561         if (i > 0) {
562     getNextToken();
563         }
564               
565         // Check if value is missing.
566         if  (m_Tokenizer.ttype == '?') {
567     instance[i] = Utils.missingValue();
568         } else {
569 
570     // Check if token is valid.
571     if (m_Tokenizer.ttype != StreamTokenizer.TT_WORD) {
572       errorMessage("not a valid value");
573     }
574           switch (m_Data.attribute(i).type()) {
575           case Attribute.NOMINAL:
576       // Check if value appears in header.
577       index = m_Data.attribute(i).indexOfValue(m_Tokenizer.sval);
578       if (index == -1) {
579         errorMessage("nominal value not declared in header");
580       }
581       instance[i] = (double)index;
582             break;
583     case Attribute.NUMERIC:
584       // Check if value is really a number.
585       try{
586         instance[i] = Double.valueOf(m_Tokenizer.sval).
587           doubleValue();
588       } catch (NumberFormatException e) {
589         errorMessage("number expected");
590       }
591             break;
592     case Attribute.STRING:
593       instance[i] = m_Data.attribute(i).addStringValue(m_Tokenizer.sval);
594             break;
595           case Attribute.DATE:
596             try {
597               instance[i] = m_Data.attribute(i).parseDate(m_Tokenizer.sval);
598             } catch (ParseException e) {
599               errorMessage("unparseable date: " + m_Tokenizer.sval);
600             }
601             break;
602           case Attribute.RELATIONAL:
603             try {
604               ArffReader arff = new ArffReader(new StringReader(m_Tokenizer.sval), m_Data.attribute(i).relation(), 0);
605               Instances data = arff.getData();
606               instance[i] = m_Data.attribute(i).addRelation(data);
607             }
608             catch (Exception e) {
609               throw new IOException(e.toString() + " of line " + getLineNo());
610             }
611             break;
612           default:
613             errorMessage("unknown attribute type in column " + i);
614     }
615         }
616       }
617       
618       double weight = 1.0;
619       if (flag) {
620         // check for an instance weight
621         weight = getInstanceWeight();
622         if (!Double.isNaN(weight)) {
623           getLastToken(true);
624         } else {
625           weight = 1.0;
626         }
627       }
628         
629       // Add instance to dataset
630       Instance inst = new DenseInstance(weight, instance);
631       inst.setDataset(m_Data);
632       
633       return inst;
634     }

636     /**
637      * Reads and stores header of an ARFF file.
638      *
639      * @param capacity      the number of instances to reserve in the data 
640      *              structure
641      * @throws IOException  if the information is not read 
642      *              successfully
643      */ 
644     protected void readHeader(int capacity) throws IOException {
645       m_Lines = 0;
646       String relationName = "";
647       
648       // Get name of relation.
649       getFirstToken();
650       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
651         errorMessage("premature end of file");
652       }
653       if (Instances.ARFF_RELATION.equalsIgnoreCase(m_Tokenizer.sval)) {
654         getNextToken();
655         relationName = m_Tokenizer.sval;
656         getLastToken(false);
657       } else {
658         errorMessage("keyword " + Instances.ARFF_RELATION + " expected");
659       }
660 
661       // Create vectors to hold information temporarily.
662       ArrayList<Attribute> attributes = new ArrayList<Attribute>();
663    
664       // Get attribute declarations.
665       getFirstToken();
666       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
667         errorMessage("premature end of file");
668       }
669 
670       while (Attribute.ARFF_ATTRIBUTE.equalsIgnoreCase(m_Tokenizer.sval)) {
671         attributes = parseAttribute(attributes);
672       }
673 
674       // Check if data part follows. We can't easily check for EOL.
675       if (!Instances.ARFF_DATA.equalsIgnoreCase(m_Tokenizer.sval)) {
676         errorMessage("keyword " + Instances.ARFF_DATA + " expected");
677       }
678       
679       // Check if any attributes have been declared.
680       if (attributes.size() == 0) {
681         errorMessage("no attributes declared");
682       }
683       
684       m_Data = new Instances(relationName, attributes, capacity);
685     }

687     /**
688      * Parses the attribute declaration.
689      *
690      * @param attributes        the current attributes vector
691      * @return          the new attributes vector
692      * @throws IOException  if the information is not read 
693      *              successfully
694      */
695     protected ArrayList<Attribute> parseAttribute(ArrayList<Attribute> attributes) throws IOException {
696       String attributeName;
697       ArrayList<String> attributeValues;
698 
699       // Get attribute name.
700       getNextToken();
701       attributeName = m_Tokenizer.sval;
702       getNextToken();
703       
704       // Check if attribute is nominal.
705       if (m_Tokenizer.ttype == StreamTokenizer.TT_WORD) {
706         
707         // Attribute is real, integer, or string.
708         if (m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_REAL) ||
709             m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_INTEGER) ||
710             m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_NUMERIC)) {
711           attributes.add(new Attribute(attributeName, attributes.size()));
712           readTillEOL();
713         } else if (m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_STRING)) {
714           attributes.add(new Attribute(attributeName, (ArrayList<String>)null,
715                 attributes.size()));
716           readTillEOL();
717         } else if (m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_DATE)) {
718           String format = null;
719           if (m_Tokenizer.nextToken() != StreamTokenizer.TT_EOL) {
720             if ((m_Tokenizer.ttype != StreamTokenizer.TT_WORD) &&
721                 (m_Tokenizer.ttype != '\'') &&
722                 (m_Tokenizer.ttype != '\"')) {
723               errorMessage("not a valid date format");
724             }
725             format = m_Tokenizer.sval;
726             readTillEOL();
727           } else {
728             m_Tokenizer.pushBack();
729           }
730           attributes.add(new Attribute(attributeName, format, attributes.size()));
731           
732         } else if (m_Tokenizer.sval.equalsIgnoreCase(Attribute.ARFF_ATTRIBUTE_RELATIONAL)) {
733           readTillEOL();
734           
735           // Read attributes for subrelation
736           // First, save current set of attributes
737           ArrayList<Attribute> atts = attributes;
738           attributes = new ArrayList<Attribute>();
739           
740           // Now, read attributes until we hit end of declaration of relational value
741           getFirstToken();
742           if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF) {
743             errorMessage("premature end of file");
744           }
745           do {
746             if (Attribute.ARFF_ATTRIBUTE.equalsIgnoreCase(m_Tokenizer.sval)) {
747               attributes = parseAttribute(attributes);
748             } else if (Attribute.ARFF_END_SUBRELATION.equalsIgnoreCase(m_Tokenizer.sval)) {
749               getNextToken();
750               if (!attributeName.equalsIgnoreCase(m_Tokenizer.sval)) {
751                 errorMessage("declaration of subrelation " + attributeName + 
752                       " must be terminated by " + "@end " + attributeName);
753               }
754               break;
755             } else {
756               errorMessage("declaration of subrelation " + attributeName + 
757                     " must be terminated by " + "@end " + attributeName);
758             }
759           } while (true);
760           
761           // Make relation and restore original set of attributes
762           Instances relation = new Instances(attributeName, attributes, 0);
763           attributes = atts;
764           attributes.add(new Attribute(attributeName, relation, attributes.size()));
765         } else {
766           errorMessage("no valid attribute type or invalid "+
767                 "enumeration");
768         }
769       } else {
770         
771         // Attribute is nominal.
772         attributeValues = new ArrayList<String>();
773         m_Tokenizer.pushBack();
774         
775         // Get values for nominal attribute.
776         if (m_Tokenizer.nextToken() != '{') {
777           errorMessage("{ expected at beginning of enumeration");
778         }
779         while (m_Tokenizer.nextToken() != '}') {
780           if (m_Tokenizer.ttype == StreamTokenizer.TT_EOL) {
781             errorMessage("} expected at end of enumeration");
782           } else {
783             attributeValues.add(m_Tokenizer.sval);
784           }
785         }
786         attributes.add(new Attribute(attributeName, attributeValues,
787               attributes.size()));
788       }
789       getLastToken(false);
790       getFirstToken();
791       if (m_Tokenizer.ttype == StreamTokenizer.TT_EOF)
792         errorMessage("premature end of file");
793       
794       return attributes;
795     }
796 
797     /**
798      * Reads and skips all tokens before next end of line token.
799      *
800      * @throws IOException  in case something goes wrong
801      */
802     protected void readTillEOL() throws IOException {
803       while (m_Tokenizer.nextToken() != StreamTokenizer.TT_EOL) {};
804       
805       m_Tokenizer.pushBack();
806     }

961   /**
962    * Determines and returns (if possible) the structure (internally the 
963    * header) of the data set as an empty set of instances.
964    *
965    * @return the structure of the data set as an empty set of Instances
966    * @throws IOException if an error occurs
967    */
968   public Instances getStructure() throws IOException {
969 
970     if (m_sourceReader == null) {
971       throw new IOException("No source has been specified");
972     }
973 
974     if (m_structure == null) {
975       try {
976     m_ArffReader = new ArffReader(m_sourceReader, 1);
977     m_structure  = m_ArffReader.getStructure();
978       } catch (Exception ex) {
979     throw new IOException("Unable to determine structure as arff (Reason: " + ex.toString() + ").");
980       }
981     }
982 
983     return new Instances(m_structure, 0);
984   }

986   /**
987    * Return the full data set. If the structure hasn't yet been determined
988    * by a call to getStructure then method should do so before processing
989    * the rest of the data set.
990    *
991    * @return the structure of the data set as an empty set of Instances
992    * @throws IOException if there is no source or parsing fails
993    */
994   public Instances getDataSet() throws IOException {
995 
996     if (m_sourceReader == null) {
997       throw new IOException("No source has been specified");
998     }
999     if (getRetrieval() == INCREMENTAL) {
1000      throw new IOException("Cannot mix getting Instances in both incremental and batch modes");
1001    }
1002    setRetrieval(BATCH);
1003    if (m_structure == null) {
1004      getStructure();
1005    }
1006
1007    // Read all instances
1008    Instance inst;
1009    while ((inst = m_ArffReader.readInstance(m_structure)) != null)
1010      m_structure.add(inst);
1011    
1012    Instances readIn = new Instances(m_structure);
1013
1014    // close the stream
1015    m_sourceReader.close();
1016    
1017    return readIn;
1018  }

1020  /**
1021   * Read the data set incrementally---get the next instance in the data 
1022   * set or returns null if there are no
1023   * more instances to get. If the structure hasn't yet been 
1024   * determined by a call to getStructure then method should do so before
1025   * returning the next instance in the data set.
1026   *
1027   * @param structure the dataset header information, will get updated in 
1028   * case of string or relational attributes
1029   * @return the next instance in the data set as an Instance object or null
1030   * if there are no more instances to be read
1031   * @throws IOException if there is an error during parsing
1032   */
1033  public Instance getNextInstance(Instances structure) throws IOException {
1034
1035    m_structure = structure;
1036
1037    if (getRetrieval() == BATCH) {
1038      throw new IOException("Cannot mix getting Instances in both incremental and batch modes");
1039    }
1040    setRetrieval(INCREMENTAL);
1041
1042    Instance current = m_ArffReader.readInstance(m_structure);
1043    if (current == null) {
1044      try {
1045        // close the stream
1046        m_sourceReader.close();
1047        //        reset();
1048      } catch (Exception ex) {
1049        ex.printStackTrace();
1050      }
1051    }
1052    return current;
1053  }
