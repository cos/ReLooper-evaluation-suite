23  package weka.core.converters;
24  
25  import weka.core.Capabilities;
26  import weka.core.RevisionUtils;
27  import weka.core.Capabilities.Capability;
28  
29  import java.io.IOException;
30  import java.io.ObjectOutputStream;
31  import java.io.OutputStream;
32  
33  /**
34   <!-- globalinfo-start -->
35   * Serializes the instances to a file with extension bsi.
36   * <p/>
37   <!-- globalinfo-end -->
38   *
39   <!-- options-start -->
40   * Valid options are: <p/>
41   * 
42   * <pre> -i &lt;the input file&gt;
43   * The input file</pre>
44   * 
45   * <pre> -o &lt;the output file&gt;
46   * The output file</pre>
47   * 
48   <!-- options-end -->
49   *
50   * @author Stefan Mutter (mutter@cs.waikato.ac.nz)
51   * @version $Revision: 4912 $
52   * @see Saver
53   */
54  public class SerializedInstancesSaver 
55    extends AbstractFileSaver 
56    implements BatchConverter {
57  
58    /** for serialization. */
59    static final long serialVersionUID = -7717010648500658872L;
60    
61    /** the output stream. */
62    protected ObjectOutputStream m_objectstream;
63    
64    /** Constructor. */  
65    public SerializedInstancesSaver(){
66        resetOptions();
67    }
68      
69    /**
70     * Returns a string describing this Saver.
71     * 
72     * @return a description of the Saver suitable for
73     * displaying in the explorer/experimenter gui
74     */
75    public String globalInfo() {
76      return "Serializes the instances to a file with extension bsi.";
77    }
78   
79    /**
80     * Returns a description of the file type.
81     *
82     * @return a short file description
83     */
84    public String getFileDescription() {
85      return "Binary serialized instances";
86    }
87  
88    /**
89     * Resets the Saver.
90     */
91    public void resetOptions() {
92  
93      super.resetOptions();
94      setFileExtension(".bsi");
95    }
96  
97    /** 
98     * Returns the Capabilities of this saver.
99     *
100    * @return            the capabilities of this object
101    * @see               Capabilities
102    */
103   public Capabilities getCapabilities() {
104     Capabilities result = super.getCapabilities();
105     
106     // attributes
107     result.enableAllAttributes();
108     result.enable(Capability.MISSING_VALUES);
109     
110     // class
111     result.enableAllClasses();
112     result.enable(Capability.MISSING_CLASS_VALUES);
113     result.enable(Capability.NO_CLASS);
114     
115     return result;
116   }
117   
118   /**
119    * Resets the writer, setting writer and objectstream to null.
120    */  
121   public void resetWriter() {
122     super.resetWriter();
123     
124     m_objectstream = null;
125   }
126   
127   /**
128    * Sets the destination output stream.
129    * 
130    * @param output the output stream.
131    * @throws IOException throws an IOException if destination cannot be set
132    */
133   public void setDestination(OutputStream output) throws IOException {
134     super.setDestination(output);
135     
136     m_objectstream = new ObjectOutputStream(output);
137   }
138   
139   /** 
140    * Writes a Batch of instances.
141    * 
142    * @throws IOException throws IOException if saving in batch mode is not possible
143    */
144   public void writeBatch() throws IOException {
145     if(getRetrieval() == INCREMENTAL)
146       throw new IOException("Batch and incremental saving cannot be mixed.");
147     
148     if(getInstances() == null)
149       throw new IOException("No instances to save");
150     
151     setRetrieval(BATCH);
152     
153     if (m_objectstream == null)
154       throw new IOException("No output for serialization.");
155 
156     setWriteMode(WRITE);
157     m_objectstream.writeObject(getInstances());
158     m_objectstream.flush();
159     m_objectstream.close();
160     setWriteMode(WAIT);
161     resetWriter();
162     setWriteMode(CANCEL);
163   }
164   
165   /**
166    * Returns the revision string.
167    * 
168    * @return        the revision
169    */
170   public String getRevision() {
171     return RevisionUtils.extract("$Revision: 4912 $");
172   }
173 
174   /**
175    * Main method.
176    *
177    * @param args should contain the options of a Saver.
178    */
179   public static void main(String[] args) {
180     runFileSaver(new SerializedInstancesSaver(), args);
181   }
182 }
