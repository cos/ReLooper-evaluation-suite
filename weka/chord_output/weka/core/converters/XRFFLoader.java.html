42  /**
43   <!-- globalinfo-start -->
44   * Reads a source that is in the XML version of the ARFF format. It automatically decompresses the data if the extension is '.xrff.gz'.
45   * <p/>
46   <!-- globalinfo-end -->
47   *
48   * @author FracPete (fracpete at waikato dot ac dot nz)
49   * @version $Revision: 5987 $
50   * @see Loader
51   */
52  public class XRFFLoader 
53    extends AbstractFileLoader 
54    implements BatchConverter, URLSourcedLoader {
55  
56    /** for serialization */
57    private static final long serialVersionUID = 3764533621135196582L;
58  
59    /** the file extension */
60    public static String FILE_EXTENSION = XMLInstances.FILE_EXTENSION;
61  
62    /** the extension for compressed files */
63    public static String FILE_EXTENSION_COMPRESSED = FILE_EXTENSION + ".gz";
64  
65    /** the url */
66    protected String m_URL = "http://";
67  
68    /** The reader for the source file. */
69    protected transient Reader m_sourceReader = null;
70  
71    /** the loaded XML document */
72    protected XMLInstances m_XMLInstances;
73    
74    /**
75     * Returns a string describing this Loader
76     * 
77     * @return        a description of the Loader suitable for
78     *            displaying in the explorer/experimenter gui
79     */
80    public String globalInfo() {
81      return 
82          "Reads a source that is in the XML version of the ARFF format. "
83        + "It automatically decompresses the data if the extension is '" 
84        + FILE_EXTENSION_COMPRESSED + "'.";
85    }
86  
87    /**
88     * Get the file extension used for libsvm files
89     *
90     * @return        the file extension
91     */
92    public String getFileExtension() {
93      return FILE_EXTENSION;
94    }
95  
96    /**
97     * Gets all the file extensions used for this type of file
98     *
99     * @return the file extensions
100    */
101   public String[] getFileExtensions() {
102     return new String[]{FILE_EXTENSION, FILE_EXTENSION_COMPRESSED};
103   }
104 
105   /**
106    * Returns a description of the file type.
107    *
108    * @return        a short file description
109    */
110   public String getFileDescription() {
111     return "XRFF data files";
112   }
113 
114   /**
115    * Resets the Loader ready to read a new data set
116    * 
117    * @throws IOException    if something goes wrong
118    */
119   public void reset() throws IOException {
120     m_structure    = null;
121     m_XMLInstances = null;
122 
123     setRetrieval(NONE);
124     
125     if (m_File != null) {
126       setFile(new File(m_File));
127     }
128     else if ((m_URL != null) && !m_URL.equals("http://")) {
129       setURL(m_URL);
130     }
131   }
132 
133   /**
134    * Resets the Loader object and sets the source of the data set to be 
135    * the supplied File object.
136    *
137    * @param file        the source file.
138    * @throws IOException    if an error occurs
139    */
140   public void setSource(File file) throws IOException {
141     m_structure    = null;
142     m_XMLInstances = null;
143     
144     setRetrieval(NONE);
145 
146     if (file == null)
147       throw new IOException("Source file object is null!");
148 
149     try {
150       if (file.getName().endsWith(FILE_EXTENSION_COMPRESSED))
151     setSource(new GZIPInputStream(new FileInputStream(file)));
152       else
153     setSource(new FileInputStream(file));
154     }
155     catch (FileNotFoundException ex) {
156       throw new IOException("File not found");
157     }
158     
159     m_sourceFile = file;
160     m_File       = file.getAbsolutePath();
161   }
162 
163   /**
164    * Resets the Loader object and sets the source of the data set to be 
165    * the supplied url.
166    *
167    * @param url     the source url.
168    * @throws IOException    if an error occurs
169    */
170   public void setSource(URL url) throws IOException {
171     m_structure    = null;
172     m_XMLInstances = null;
173     
174     setRetrieval(NONE);
175     
176     setSource(url.openStream());
177 
178     m_URL = url.toString();
179   }
180   
181   /**
182    * Set the url to load from
183    *
184    * @param url         the url to load from
185    * @throws IOException    if the url can't be set.
186    */
187   public void setURL(String url) throws IOException {
188     m_URL = url;
189     setSource(new URL(url));
190   }
191 
192   /**
193    * Return the current url
194    *
195    * @return the current url
196    */
197   public String retrieveURL() {
198     return m_URL;
199   }
200 
201   /**
202    * Resets the Loader object and sets the source of the data set to be 
203    * the supplied InputStream.
204    *
205    * @param in          the source InputStream.
206    * @throws IOException    if initialization of reader fails.
207    */
208   public void setSource(InputStream in) throws IOException {
209     m_File = (new File(System.getProperty("user.dir"))).getAbsolutePath();
210     m_URL  = "http://";
211 
212     m_sourceReader = new BufferedReader(new InputStreamReader(in));
213   }
214   
215   /**
216    * Determines and returns (if possible) the structure (internally the 
217    * header) of the data set as an empty set of instances.
218    *
219    * @return            the structure of the data set as an empty set 
220    *                of Instances
221    * @throws IOException    if an error occurs
222    */
223   public Instances getStructure() throws IOException {
224     if (m_sourceReader == null)
225       throw new IOException("No source has been specified");
226 
227     if (m_structure == null) {
228       try {
229     m_XMLInstances = new XMLInstances(m_sourceReader);
230     m_structure    = new Instances(m_XMLInstances.getInstances(), 0);
231       }
232       catch (IOException ioe) {
233     // just re-throw it
234     throw ioe;
235       }
236       catch (Exception e) {
237     throw new RuntimeException(e);
238       }
239     }
240 
241     return new Instances(m_structure, 0);
242   }
243   
244   /**
245    * Return the full data set. If the structure hasn't yet been determined
246    * by a call to getStructure then method should do so before processing
247    * the rest of the data set.
248    *
249    * @return            the structure of the data set as an empty 
250    *                set of Instances
251    * @throws IOException    if there is no source or parsing fails
252    */
253   public Instances getDataSet() throws IOException {
254     if (m_sourceReader == null)
255       throw new IOException("No source has been specified");
256     
257     if (getRetrieval() == INCREMENTAL)
258       throw new IOException("Cannot mix getting Instances in both incremental and batch modes");
259 
260     setRetrieval(BATCH);
261     if (m_structure == null)
262       getStructure();
263 
264     try {
265       // close the stream
266       m_sourceReader.close();
267     } catch (Exception ex) {
268     }
269 
270     return m_XMLInstances.getInstances();
271   }
272 
273   /**
274    * XRFFLoader is unable to process a data set incrementally.
275    *
276    * @param structure       ignored
277    * @return            never returns without throwing an exception
278    * @throws IOException    always. XRFFLoader is unable to process a 
279    *                data set incrementally.
280    */
281   public Instance getNextInstance(Instances structure) throws IOException {
282     throw new IOException("XRFFLoader can't read data sets incrementally.");
283   }

285   /**
286    * Returns the revision string.
287    * 
288    * @return        the revision
289    */
290   public String getRevision() {
291     return RevisionUtils.extract("$Revision: 5987 $");
292   }

301   }
