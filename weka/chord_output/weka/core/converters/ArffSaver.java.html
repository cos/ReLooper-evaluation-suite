34  /**
35   * Writes to a destination in arff text format. <p/>
36   *
37   <!-- options-start -->
38   * Valid options are: <p/>
39   * 
40   * <pre> -i &lt;the input file&gt;
41   * The input file</pre>
42   * 
43   * <pre> -o &lt;the output file&gt;
44   * The output file</pre>
45   * 
46   <!-- options-end -->
47   *
48   * @author Stefan Mutter (mutter@cs.waikato.ac.nz)
49   * @version $Revision: 5953 $
50   * @see Saver
51   */
52  public class ArffSaver 
53    extends AbstractFileSaver 
54    implements BatchConverter, IncrementalConverter {
55  
56    /** for serialization */
57    static final long serialVersionUID = 2223634248900042228L;    
58    
59    /** Constructor */  
60    public ArffSaver(){
61    
62        resetOptions();
63    }
64     
65     
66    /**
67     * Returns a string describing this Saver
68     * @return a description of the Saver suitable for
69     * displaying in the explorer/experimenter gui
70     */
71    public String globalInfo() {
72      return "Writes to a destination that is in arff (attribute relation file format) "
73        +"format. ";
74    }
75  
76    
77    /**
78     * Returns a description of the file type.
79     *
80     * @return a short file description
81     */
82    public String getFileDescription() {
83      return "Arff data files";
84    }
85  
86    /**
87     * Resets the Saver 
88     */
89    public void resetOptions() {
90  
91      super.resetOptions();
92      setFileExtension(".arff");
93    }
94  
95    /** 
96     * Returns the Capabilities of this saver.
97     *
98     * @return            the capabilities of this object
99     * @see               Capabilities
100    */
101   public Capabilities getCapabilities() {
102     Capabilities result = super.getCapabilities();
103     
104     // attributes
105     result.enableAllAttributes();
106     result.enable(Capability.MISSING_VALUES);
107     
108     // class
109     result.enableAllClasses();
110     result.enable(Capability.MISSING_CLASS_VALUES);
111     result.enable(Capability.NO_CLASS);
112     
113     return result;
114   }
115   
116   /** Saves an instances incrementally. Structure has to be set by using the
117    * setStructure() method or setInstances() method.
118    * @param inst the instance to save
119    * @throws IOException throws IOEXception if an instance cannot be saved incrementally.
120    */  
121   public void writeIncremental(Instance inst) throws IOException{
122   
123       int writeMode = getWriteMode();
124       Instances structure = getInstances();
125       PrintWriter outW = null;
126       
127       if(getRetrieval() == BATCH || getRetrieval() == NONE)
128           throw new IOException("Batch and incremental saving cannot be mixed.");
129       if(getWriter() != null)
130           outW = new PrintWriter(getWriter());
131           
132       if(writeMode == WAIT){
133         if(structure == null){
134             setWriteMode(CANCEL);
135             if(inst != null)
136                 System.err.println("Structure(Header Information) has to be set in advance");
137         }
138         else
139             setWriteMode(STRUCTURE_READY);
140         writeMode = getWriteMode();
141       }
142       if(writeMode == CANCEL){
143           if(outW != null)
144               outW.close();
145           cancel();
146       }
147       if(writeMode == STRUCTURE_READY){
148           setWriteMode(WRITE);
149           //write header
150           Instances header = new Instances(structure,0);
151           if(retrieveFile() == null || outW == null)
152               System.out.println(header.toString());
153           else{
154               outW.print(header.toString());
155               outW.print("\n");
156               outW.flush();
157           }
158           writeMode = getWriteMode();
159       }
160       if(writeMode == WRITE){
161           if(structure == null)
162               throw new IOException("No instances information available.");
163           if(inst != null){
164           //write instance 
165               if(retrieveFile() == null || outW == null)
166                 System.out.println(inst);
167               else{
168                 outW.println(inst);
169                 m_incrementalCounter++;
170                 //flush every 100 instances
171                 if(m_incrementalCounter > 100){
172                     m_incrementalCounter = 0;
173                     outW.flush();
174                 }
175               }
176           }
177           else{
178           //close
179               if(outW != null){
180                 outW.flush();
181                 outW.close();
182               }
183               m_incrementalCounter = 0;
184               resetStructure();
185               outW = null;
186               resetWriter();
187           }
188       }
189   }
190   
191   /** Writes a Batch of instances
192    * @throws IOException throws IOException if saving in batch mode is not possible
193    */
194   public void writeBatch() throws IOException {
195   
196       if(getInstances() == null)
197           throw new IOException("No instances to save");
198       if(getRetrieval() == INCREMENTAL)
199           throw new IOException("Batch and incremental saving cannot be mixed.");
200       setRetrieval(BATCH);
201       setWriteMode(WRITE);
202       if(retrieveFile() == null && getWriter() == null){
203           System.out.println((getInstances()).toString());
204           setWriteMode(WAIT);
205           return;
206       }
207       PrintWriter outW = new PrintWriter(getWriter());
208       outW.print((getInstances()).toString());
209       outW.flush();
210       outW.close();
211       setWriteMode(WAIT);
212       outW = null;
213       resetWriter();
214       setWriteMode(CANCEL);
215   }

218    * Returns the revision string.
219    * 
220    * @return        the revision
221    */
222   public String getRevision() {
223     return RevisionUtils.extract("$Revision: 5953 $");
224   }

233   }
