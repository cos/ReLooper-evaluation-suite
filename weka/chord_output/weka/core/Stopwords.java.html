611   /**
612    * removes all stopwords
613    */
614   public void clear() {
615     m_Words.clear();
616   }

618   /**
619    * adds the given word to the stopword list (is automatically converted to
620    * lower case and trimmed)
621    *
622    * @param word the word to add
623    */
624   public void add(String word) {
625     if (word.trim().length() > 0)
626       m_Words.add(word.trim().toLowerCase());
627   }
628 
629   /**
630    * removes the word from the stopword list
631    *
632    * @param word the word to remove
633    * @return true if the word was found in the list and then removed
634    */
635   public boolean remove(String word) {
636     return m_Words.remove(word);
637   }

639   /** 
640    * Returns true if the given string is a stop word.
641    * 
642    * @param word the word to test
643    * @return true if the word is a stopword
644    */
645   public boolean is(String word) {
646     return m_Words.contains(word.toLowerCase());
647   }
648 
649   /**
650    * Returns a sorted enumeration over all stored stopwords
651    *
652    * @return the enumeration over all stopwords
653    */
654   public Enumeration elements() {
655     Iterator<String>    iter;
656     Vector<String>      list;
657 
658     iter = m_Words.iterator();
659     list = new Vector<String>();
660 
661     while (iter.hasNext())
662       list.add(iter.next());
663 
664     // sort list
665     Collections.sort(list);
666 
667     return list.elements();
668   }

680   /**

687     read(new BufferedReader(new FileReader(file)));
688   }
689 
690   /**
691    * Generates a new Stopwords object from the reader. The reader is
692    * closed automatically.
693    *
694    * @param reader the reader to get the stopwords from
695    * @throws Exception if reading fails
696    */
697   public void read(BufferedReader reader) throws Exception {
698     String      line;
699 
700     clear();
701     
702     while ((line = reader.readLine()) != null) {
703       line = line.trim();
704       // comment?
705       if (line.startsWith("#"))
706         continue;
707       add(line);
708     }
709 
710     reader.close();
711   }

729   public void write(File file) throws Exception {
730     write(new BufferedWriter(new FileWriter(file)));
731   }
732 
733   /**
734    * Writes the current stopwords to the given writer. The writer is closed
735    * automatically.
736    *
737    * @param writer the writer to get the stopwords from
738    * @throws Exception if writing fails
739    */
740   public void write(BufferedWriter writer) throws Exception {
741     Enumeration   enm;
742 
743     // header
744     writer.write("# generated " + new Date());
745     writer.newLine();
746 
747     enm = elements();
748 
749     while (enm.hasMoreElements()) {
750       writer.write(enm.nextElement().toString());
751       writer.newLine();
752     }
753 
754     writer.flush();
755     writer.close();
756   }

758   /**
759    * returns the current stopwords in a string
760    *
761    * @return the current stopwords
762    */
763   public String toString() {
764     Enumeration   enm;
765     StringBuffer  result;
766 
767     result = new StringBuffer();
768     enm    = elements();
769     while (enm.hasMoreElements()) {
770       result.append(enm.nextElement().toString());
771       if (enm.hasMoreElements())
772         result.append(",");
773     }
774 
775     return result.toString();
776   }

797   /**
798    * Accepts the following parameter: <p/>
799    *
800    * -i file <br/>
801    * loads the stopwords from the given file <p/>
802    *
803    * -o file <br/>
804    * saves the stopwords to the given file <p/>
805    *
806    * -p <br/>
807    * outputs the current stopwords on stdout <p/>
808    *
809    * Any additional parameters are interpreted as words to test as stopwords.
810    * 
811    * @param args commandline parameters
812    * @throws Exception if something goes wrong
813    */
814   public static void main(String[] args) throws Exception {
815     String input = Utils.getOption('i', args);
816     String output = Utils.getOption('o', args);
817     boolean print = Utils.getFlag('p', args);
818 
819     // words to process?
820     Vector<String> words = new Vector<String>();
821     for (int i = 0; i < args.length; i++) {
822       if (args[i].trim().length() > 0)
823         words.add(args[i].trim());
824     }
825     
826     Stopwords stopwords = new Stopwords();
827 
828     // load from file?
829     if (input.length() != 0)
830       stopwords.read(input);
831 
832     // write to file?
833     if (output.length() != 0)
834       stopwords.write(output);
835     
836     // output to stdout?
837     if (print) {
838       System.out.println("\nStopwords:");
839       Enumeration enm = stopwords.elements();
840       int i = 0;
841       while (enm.hasMoreElements()) {
842         System.out.println((i+1) + ". " + enm.nextElement());
843         i++;
844       }
845     }
846 
847     // check words for being a stopword
848     if (words.size() > 0) {
849       System.out.println("\nChecking for stopwords:");
850       for (int i = 0; i < words.size(); i++) {
851         System.out.println(
852             (i+1) + ". " + words.get(i) + ": " 
853             + stopwords.is(words.get(i).toString()));
854       }
855     }
856   }
