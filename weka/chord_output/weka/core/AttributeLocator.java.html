22  package weka.core;
23  
24  import java.io.Serializable;
25  import java.util.Vector;
26  
27  /**
28   * This class locates and records the indices of a certain type of attributes, 
29   * recursively in case of Relational attributes.
30   * 
31   * @author fracpete (fracpete at waikato dot ac dot nz)
32   * @version $Revision: 5953 $
33   * @see Attribute#RELATIONAL
34   */
35  public class AttributeLocator 
36    implements Serializable, Comparable<AttributeLocator>, RevisionHandler {
37    
38    /** for serialization */
39    private static final long serialVersionUID = -2932848827681070345L;
40  
41    /** the attribute indices that may be inspected */
42    protected int[] m_AllowedIndices = null;
43    
44    /** contains the attribute locations, either true or false Boolean objects */
45    protected Vector<Boolean> m_Attributes = null;
46    
47    /** contains the locator locations, either null or a AttributeLocator reference */
48    protected Vector<AttributeLocator> m_Locators = null;
49  
50    /** the type of the attribute */
51    protected int m_Type = -1;
52    
53    /** the referenced data */
54    protected Instances m_Data = null;
55  
56    /** the indices */
57    protected int[] m_Indices = null;
58  
59    /** the indices of locator objects */
60    protected int[] m_LocatorIndices = null;
61    
62    /**
63     * Initializes the AttributeLocator with the given data for the specified
64     * type of attribute. Checks all attributes.
65     * 
66     * @param data    the data to work on
67     * @param type    the type of attribute to locate
68     */
69    public AttributeLocator(Instances data, int type) {
70      this(data, type, 0, data.numAttributes() - 1);
71    }
72    
73    /**
74     * Initializes the AttributeLocator with the given data for the specified
75     * type of attribute. Checks only the given range.
76     * 
77     * @param data    the data to work on
78     * @param type    the type of attribute to locate
79     * @param fromIndex   the first index to inspect (including)
80     * @param toIndex the last index to check (including)
81     */
82    public AttributeLocator(Instances data, int type, int fromIndex, int toIndex) {
83      super();
84  
85      int[] indices = new int[toIndex - fromIndex + 1];
86      for (int i = 0; i < indices.length; i++)
87        indices[i] = fromIndex + i;
88      
89      initialize(data, type, indices);
90    }
91    
92    /**
93     * initializes the AttributeLocator with the given data for the specified
94     * type of attribute. Checks only the given attribute indices.
95     * 
96     * @param data    the data to work on
97     * @param type    the type of attribute to locate
98     * @param indices the attribute indices to check
99     */
100   public AttributeLocator(Instances data, int type, int[] indices) {
101     super();
102 
103     initialize(data, type, indices);
104   }
105   
106   /**
107    * initializes the AttributeLocator
108    * 
109    * @param data    the data to base the search for attributes on
110    * @param type    the type of attribute to look for
111    * @param indices the indices that are allowed to check
112    */
113   protected void initialize(Instances data, int type, int[] indices) {
114     m_Data = new Instances(data, 0);
115     m_Type = type;
116     
117     m_AllowedIndices = new int[indices.length];
118     System.arraycopy(indices, 0, m_AllowedIndices, 0, indices.length);
119     
120     locate();
121 
122     m_Indices        = find(true);
123     m_LocatorIndices = find(false);
124   }
125   
126   /**
127    * returns the type of attribute that is located
128    * 
129    * @return        the type of attribute
130    */
131   public int getType() {
132     return m_Type;
133   }
134   
135   /**
136    * returns the indices that are allowed to check for the attribute type
137    * 
138    * @return        the indices that are checked for the attribute type
139    */
140   public int[] getAllowedIndices() {
141     return m_AllowedIndices;
142   }
143   
144   /**
145    * sets up the structure
146    */
147   protected void locate() {
148     int         i;
149     
150     m_Attributes = new Vector<Boolean>();
151     m_Locators   = new Vector<AttributeLocator>();
152     
153     for (i = 0; i < m_AllowedIndices.length; i++) {
154       if (m_Data.attribute(m_AllowedIndices[i]).type() == Attribute.RELATIONAL)
155     m_Locators.add(new AttributeLocator(m_Data.attribute(m_AllowedIndices[i]).relation(), getType()));
156       else
157     m_Locators.add(null);
158       
159       if (m_Data.attribute(m_AllowedIndices[i]).type() == getType())
160         m_Attributes.add(new Boolean(true));
161       else
162         m_Attributes.add(new Boolean(false));
163     }
164   }
165   
166   /**
167    * returns the underlying data
168    * 
169    * @return      the underlying Instances object
170    */
171   public Instances getData() {
172     return m_Data;
173   }
174   
175   /**
176    * returns the indices of the searched-for attributes (if TRUE) or the indices
177    * of AttributeLocator objects (if FALSE)
178    * 
179    * @param findAtts      if true the indices of attributes are located,
180    *                      otherwise the ones of AttributeLocator objects
181    * @return              the indices of the attributes or the AttributeLocator objects
182    */
183   protected int[] find(boolean findAtts) {
184     int     i;
185     int[]   result;
186     Vector<Integer> indices;
187 
188     // determine locations
189     indices = new Vector<Integer>();
190     if (findAtts) {
191       for (i = 0; i < m_Attributes.size(); i++) {
192     if (((Boolean) m_Attributes.get(i)).booleanValue())
193       indices.add(new Integer(i));
194       }
195     }
196     else {
197       for (i = 0; i < m_Locators.size(); i++) {
198     if (m_Locators.get(i) != null)
199       indices.add(new Integer(i));
200       }
201     }
202     
203     // fill array
204     result = new int[indices.size()];
205     for (i = 0; i < indices.size(); i++)
206       result[i] = ((Integer) indices.get(i)).intValue();
207     
208     return result;
209   }
210 
211   /**
212    * returns actual index in the Instances object.
213    * 
214    * @param index   the index in the m_AllowedIndices array
215    * @return        the actual index in the instances object
216    */
217   public int getActualIndex(int index) {
218     return m_AllowedIndices[index];
219   }
220   
221   /**
222    * Returns the indices of the attributes. These indices are referring
223    * to the m_AllowedIndices array, not the actual indices in the Instances
224    * object.
225    * 
226    * @return    the indices of the attributes
227    * @see   #getActualIndex(int)
228    */
229   public int[] getAttributeIndices() {
230     return m_Indices;
231   }
232   
233   /**
234    * Returns the indices of the AttributeLocator objects.  These indices are 
235    * referring to the m_AllowedIndices array, not the actual indices in the 
236    * Instances object.
237    * 
238    * @return    the indices of the AttributeLocator objects
239    * @see   #getActualIndex(int)
240    */
241   public int[] getLocatorIndices() {
242     return m_LocatorIndices;
243   }
244   
245   /**
246    * Returns the AttributeLocator at the given index. This index refers to
247    * the index of the m_AllowedIndices array, not the actual Instances object.
248    * 
249    * @param index   the index of the locator to retrieve
250    * @return        the AttributeLocator at the given index
251    */
252   public AttributeLocator getLocator(int index) {
253     return (AttributeLocator) m_Locators.get(index);
254   }
255   
256   /**
257    * Compares this object with the specified object for order. Returns a 
258    * negative integer, zero, or a positive integer as this object is less 
259    * than, equal to, or greater than the specified object. Only type and
260    * indices are checked.
261    * 
262    * @param o       the object to compare with
263    * @return        -1 if less than, 0 if equal, +1 if greater than the 
264    *            given object
265    */
266   public int compareTo(AttributeLocator o) {
267     int     result;
268     int     i;
269     
270     result = 0;
271     
272     // 1. check type
273     if (this.getType() < o.getType()) {
274       result = -1;
275     }
276     else if (this.getType() > o.getType()) {
277       result = 1;
278     }
279     else {
280       // 2. check indices
281       if (this.getAllowedIndices().length < o.getAllowedIndices().length) {
282     result = -1;
283       }
284       else if (this.getAllowedIndices().length > o.getAllowedIndices().length) {
285     result = 1;
286       }
287       else {
288     for (i = 0; i < this.getAllowedIndices().length; i++) {
289       if (this.getAllowedIndices()[i] < o.getAllowedIndices()[i]) {
290         result = -1;
291         break;
292       }
293       else if (this.getAllowedIndices()[i] > o.getAllowedIndices()[i]) {
294         result = 1;
295         break;
296       }
297       else {
298         result = 0;
299       }
300     }
301       }
302     }
303     
304     return result;
305   }
306   
307   /**
308    * Indicates whether some other object is "equal to" this one. Only type
309    * and indices are checked.
310    * 
311    * @param o       the AttributeLocator to check for equality
312    * @return        true if the AttributeLocators have the same type and 
313    *            indices
314    */
315   public boolean equals(Object o) {
316     return (compareTo((AttributeLocator) o) == 0);
317   }
318   
319   /**
320    * returns a string representation of this object
321    * 
322    * @return        a string representation
323    */
324   public String toString() {
325     return m_Attributes.toString();
326   }
327   
328   /**
329    * Returns the revision string.
330    * 
331    * @return        the revision
332    */
333   public String getRevision() {
334     return RevisionUtils.extract("$Revision: 5953 $");
335   }
336 }
