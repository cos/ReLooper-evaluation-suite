228   /**
229    * adds the attribute to the XML structure
230    * 
231    * @param parent  the parent node to add the attribute node as child
232    * @param att     the attribute to add
233    */
234   protected void addAttribute(Element parent, Attribute att) {
235     Element     node;
236     Element     child;
237     Element     property;
238     Element     label;
239     String      tmpStr;
240     Enumeration     enm;
241     int         i;
242     
243     node = m_Document.createElement(TAG_ATTRIBUTE);
244     parent.appendChild(node);
245     
246     // XML attributes
247     // name
248     node.setAttribute(ATT_NAME, validContent(att.name()));
249     
250     // type
251     switch (att.type()) {
252       case Attribute.NUMERIC:
253     node.setAttribute(ATT_TYPE, VAL_NUMERIC);
254     break;
255     
256       case Attribute.DATE:
257     node.setAttribute(ATT_TYPE, VAL_DATE);
258     break;
259     
260       case Attribute.NOMINAL:
261     node.setAttribute(ATT_TYPE, VAL_NOMINAL);
262     break;
263     
264       case Attribute.STRING:
265     node.setAttribute(ATT_TYPE, VAL_STRING);
266     break;
267     
268       case Attribute.RELATIONAL:
269     node.setAttribute(ATT_TYPE, VAL_RELATIONAL);
270     break;
271     
272       default:
273     node.setAttribute(ATT_TYPE, "???");
274     }
275     
276     // labels
277     if (att.isNominal()) {
278       child = m_Document.createElement(TAG_LABELS);
279       node.appendChild(child);
280       enm = att.enumerateValues();
281       while (enm.hasMoreElements()) {
282     tmpStr = enm.nextElement().toString();
283     label = m_Document.createElement(TAG_LABEL);
284     child.appendChild(label);
285     label.appendChild(m_Document.createTextNode(validContent(tmpStr)));
286       }
287     }
288     
289     // format
290     if (att.isDate())
291       node.setAttribute(ATT_FORMAT, validContent(att.getDateFormat()));
292     
293     // class
294     if (m_Instances.classIndex() > -1) {
295       if (att == m_Instances.classAttribute())
296     node.setAttribute(ATT_CLASS, VAL_YES);
297     }
298     
299     // add meta-data
300     if ( (att.getMetadata() != null) && (att.getMetadata().size() > 0) ) {
301       child = m_Document.createElement(TAG_METADATA);
302       node.appendChild(child);
303       enm = att.getMetadata().propertyNames();
304       while (enm.hasMoreElements()) {
305     tmpStr = enm.nextElement().toString();
306     property = m_Document.createElement(TAG_PROPERTY);
307     child.appendChild(property);
308     property.setAttribute(ATT_NAME, validContent(tmpStr));
309     property.appendChild(m_Document.createTextNode(validContent(att.getMetadata().getProperty(tmpStr, ""))));
310       }
311     }
312     
313     // relational attribute?
314     if (att.isRelationValued()) {
315       child = m_Document.createElement(TAG_ATTRIBUTES);
316       node.appendChild(child);
317       for (i = 0; i < att.relation().numAttributes(); i++)
318     addAttribute(child, att.relation().attribute(i));
319     }
320   }

322   /**
323    * turns all &lt;, &gt; and &amp;into character entities and returns that 
324    * string. Necessary for TextNodes.
325    * 
326    * @param content string to convert
327    * @return        the valid content string
328    */
329   protected String validContent(String content) {
330     String  result;
331     
332     result = content;
333     
334     // these five entities are recognized by every XML processor
335     // see http://www.xml.com/pub/a/2001/03/14/trxml10.html
336     result = result.replaceAll("&", "&amp;")
337                    .replaceAll("\"", "&quot;")
338                    .replaceAll("'", "&apos;")
339                    .replaceAll("<", "&lt;")
340                    .replaceAll(">", "&gt;");
341     // in addition, replace some other entities as well
342     result = result.replaceAll("\n", "&#10;")
343                    .replaceAll("\r", "&#13;")
344                    .replaceAll("\t", "&#9;");
345     
346     return result;
347   }

349   /**
350    * adds the instance to the XML structure
351    * 
352    * @param parent  the parent node to add the instance node as child
353    * @param inst    the instance to add
354    */
355   protected void addInstance(Element parent, Instance inst) {
356     Element     node;
357     Element     value;
358     Element     child;
359     boolean     sparse;
360     int         i;
361     int         n;
362     int         index;
363     
364     node = m_Document.createElement(TAG_INSTANCE);
365     parent.appendChild(node);
366     
367     // sparse?
368     sparse = (inst instanceof SparseInstance);
369     if (sparse)
370       node.setAttribute(ATT_TYPE, VAL_SPARSE);
371     
372     // weight
373     if (inst.weight() != 1.0)
374       node.setAttribute(ATT_WEIGHT, Utils.doubleToString(inst.weight(), m_Precision));
375     
376     // values
377     for (i = 0; i < inst.numValues(); i++) {
378       index = inst.index(i);
379       
380       value = m_Document.createElement(TAG_VALUE);
381       node.appendChild(value);
382 
383       if (inst.isMissing(index)) {
384     value.setAttribute(ATT_MISSING, VAL_YES);
385       }
386       else {
387     if (inst.attribute(index).isRelationValued()) {
388       child = m_Document.createElement(TAG_INSTANCES);
389       value.appendChild(child);
390       for (n = 0; n < inst.relationalValue(i).numInstances(); n++)
391         addInstance(child, inst.relationalValue(i).instance(n));
392     }
393     else {
394       if (inst.attribute(index).type() == Attribute.NUMERIC)
395         value.appendChild(m_Document.createTextNode(Utils.doubleToString(inst.value(index), m_Precision)));
396       else
397         value.appendChild(m_Document.createTextNode(validContent(inst.stringValue(index))));
398     }
399       }
400       
401       if (sparse)
402     value.setAttribute(ATT_INDEX, "" + (index+1));
403     }
404   }

406   /**
407    * generates the XML structure for the header
408    */
409   protected void headerToXML() {
410     Element root;
411     Element node;
412     Element child;
413     int     i;
414     
415     root = m_Document.getDocumentElement();
416     root.setAttribute(ATT_NAME, validContent(m_Instances.relationName()));
417     root.setAttribute(ATT_VERSION, Version.VERSION);
418     
419     // create "header" node
420     node = m_Document.createElement(TAG_HEADER);
421     root.appendChild(node);
422 
423     // add all attributes
424     child = m_Document.createElement(TAG_ATTRIBUTES);
425     node.appendChild(child);
426     for (i = 0; i < m_Instances.numAttributes(); i++)
427       addAttribute(child, m_Instances.attribute(i));
428   }

435     Element node;
436     Element child;
437     int     i;
438

474   /**
475    * returns the metadata, if any available underneath this node, otherwise
476    * just null
477    * 
478    * @param parent  the attribute node
479    * @return        the metadata, or null if none found
480    * @throws Exception  if generation fails
481    */
482   protected ProtectedProperties createMetadata(Element parent) throws Exception {
483     ProtectedProperties result;
484     Properties      props;
485     Vector      list;
486     Element     node;
487     Element     metanode;
488     int         i;
489     
490     result = null;
491     
492     // find metadata node directly underneath this attribute, but not in
493     // deeper nested attributes (e.g., within relational attributes)
494     metanode = null;
495     list     = getChildTags(parent, TAG_METADATA);
496     if (list.size() > 0)
497       metanode = (Element) list.get(0);
498     
499     // generate properties
500     if (metanode != null) {
501       props = new Properties();
502       list  = getChildTags(metanode, TAG_PROPERTY);
503       for (i = 0; i < list.size(); i++) {
504     node = (Element) list.get(i);
505     props.setProperty(node.getAttribute(ATT_NAME), getContent(node));
506       }
507       result = new ProtectedProperties(props);
508     }
509     
510     return result;
511   }

513   /**
514    * returns the labels listed underneath this (nominal) attribute in a 
515    * ArrayList
516    * 
517    * @param parent  the (nominal) attribute node
518    * @return        the label vector
519    * @throws Exception  if generation fails
520    */
521   protected ArrayList<String> createLabels(Element parent) throws Exception {
522     ArrayList<String>       result;
523     Vector      list;
524     Element     node;
525     Element     labelsnode;
526     int         i;
527     
528     result = new ArrayList<String>();
529     
530     // find labels node directly underneath this attribute, but not in
531     // deeper nested attributes (e.g., within relational attributes)
532     labelsnode = null;
533     list     = getChildTags(parent, TAG_LABELS);
534     if (list.size() > 0)
535       labelsnode = (Element) list.get(0);
536     
537     // retrieve all labels
538     if (labelsnode != null) {
539       list  = getChildTags(labelsnode, TAG_LABEL);
540       for (i = 0; i < list.size(); i++) {
541     node = (Element) list.get(i);
542     result.add(getContent(node));
543       }
544     }
545     
546     return result;
547   }

549   /**
550    * creates an Attribute from the given XML node
551    * 
552    * @param node    the node with the setup
553    * @return        the configured Attribute
554    * @throws Exception  if generation fails, e.g., due to unknown attribute type
555    */
556   protected Attribute createAttribute(Element node) throws Exception {
557     String      typeStr;
558     String      name;
559     int         type;
560     Attribute       result;
561     ArrayList<String>       values;
562     ProtectedProperties metadata;
563     Vector      list;
564     ArrayList<Attribute>        atts;
565     
566     result = null;
567     
568     // name
569     name = node.getAttribute(ATT_NAME);
570 
571     // type
572     typeStr = node.getAttribute(ATT_TYPE);
573     if (typeStr.equals(VAL_NUMERIC))
574       type = Attribute.NUMERIC;
575     else if (typeStr.equals(VAL_DATE))
576       type = Attribute.DATE;
577     else if (typeStr.equals(VAL_NOMINAL))
578       type = Attribute.NOMINAL;
579     else if (typeStr.equals(VAL_STRING))
580       type = Attribute.STRING;
581     else if (typeStr.equals(VAL_RELATIONAL))
582       type = Attribute.RELATIONAL;
583     else
584       throw new Exception(
585       "Attribute type '" + typeStr + "' is not supported!");
586 
587     // metadata
588     metadata = createMetadata(node);
589     
590     switch (type) {
591       case Attribute.NUMERIC:
592     if (metadata == null)
593       result = new Attribute(name);
594     else
595       result = new Attribute(name, metadata);
596     break;
597 
598       case Attribute.DATE:
599     if (metadata == null)
600       result = new Attribute(name, node.getAttribute(ATT_FORMAT));
601     else
602       result = new Attribute(name, node.getAttribute(ATT_FORMAT), metadata);
603     break;
604     
605       case Attribute.NOMINAL:
606     values = createLabels(node);
607     if (metadata == null)
608       result = new Attribute(name, values);
609     else
610       result = new Attribute(name, values, metadata);
611     break;
612     
613       case Attribute.STRING:
614     if (metadata == null)
615       result = new Attribute(name, (ArrayList<String>) null);
616     else
617       result = new Attribute(name, (ArrayList<String>) null, metadata);
618     break;
619     
620       case Attribute.RELATIONAL:
621     list = getChildTags(node, TAG_ATTRIBUTES);
622     node = (Element) list.get(0);
623     atts = createAttributes(node, new int[1]);
624     if (metadata == null)
625       result = new Attribute(name, new Instances(name, atts, 0));
626     else
627       result = new Attribute(name, new Instances(name, atts, 0), metadata);
628     break;
629     }
630     
631     return result;
632   }

634   /**
635    * returns a list of generated attributes
636    * 
637    * @param parent  the attributes node
638    * @param classIndex  array of length 1 to return the class index, if any
639    * @return        the vector with the generated attributes
640    * @throws Exception  if generation fails, e.g., due to unknown attribute type
641    */
642   protected ArrayList<Attribute> createAttributes(Element parent, int[] classIndex) throws Exception {
643     Vector  list;
644     ArrayList<Attribute>    result;
645     int     i;
646     Element node;
647     Attribute   att;
648 
649     result        = new ArrayList<Attribute>();
650     classIndex[0] = -1;
651     
652     list = getChildTags(parent, TAG_ATTRIBUTE);
653     for (i = 0; i < list.size(); i++) {
654       node = (Element) list.get(i);
655       att = createAttribute(node);
656       if (node.getAttribute(ATT_CLASS).equals(VAL_YES))
657     classIndex[0] = i;
658       result.add(att);
659     }
660     
661     return result;
662   }

666    * 
667    * @param header  the data this instance will belong to
668    * @param parent  the instance node
669    * @return        the configured Instance

672   protected Instance createInstance(Instances header, Element parent) throws Exception {
673     Instance    result;
674     Element node;
675     Element child;
676     boolean sparse;
677     int     i;
678     int     index;
679     Vector  list;
680     Vector  subList;
681     double[]    values;
682     String  content;
683     double  weight;
684     Instances   data;
685     
686     result = null;
687 
688     // sparse?
689     sparse = (parent.getAttribute(ATT_TYPE).equals(VAL_SPARSE));
690     values = new double[header.numAttributes()];
691     
692     // weight
693     if (parent.getAttribute(ATT_WEIGHT).length() != 0)
694       weight = Double.parseDouble(parent.getAttribute(ATT_WEIGHT));
695     else
696       weight = 1.0;
697     
698     list = getChildTags(parent, TAG_VALUE);
699     for (i = 0; i < list.size(); i++) {
700       node = (Element) list.get(i);
701       
702       // determine index
703       if (sparse)
704     index = Integer.parseInt(node.getAttribute(ATT_INDEX)) - 1;
705       else
706     index = i;
707 
708       // set value
709       if (node.getAttribute(ATT_MISSING).equals(VAL_YES)) {
710     values[index] = Utils.missingValue();
711       }
712       else {
713     content = getContent(node);
714     switch (header.attribute(index).type()) {
715       case Attribute.NUMERIC:
716         values[index] = Double.parseDouble(content);
717         break;
718         
719       case Attribute.DATE:
720         values[index] = header.attribute(index).parseDate(content);
721         break;
722         
723       case Attribute.NOMINAL:
724         values[index] = header.attribute(index).indexOfValue(content);
725         break;
726         
727       case Attribute.STRING:
728         values[index] = header.attribute(index).addStringValue(content);
729         break;
730         
731       case Attribute.RELATIONAL:
732         subList       = getChildTags(node, TAG_INSTANCES);
733         child         = (Element) subList.get(0);
734         data          = createInstances(header.attribute(index).relation(), child);
735         values[index] = header.attribute(index).addRelation(data);
736         break;
737         
738       default:
739         throw new Exception(
740         "Attribute type " + header.attribute(index).type() 
741         + " is not supported!");  
742     }
743       }
744     }
745     
746     // create instance
747     if (sparse)
748       result = new SparseInstance(weight, values);
749     else
750       result = new DenseInstance(weight, values);
751     
752     return result;
753   }

755   /**
756    * creates Instances from the given XML node
757    * 
758    * @param header  the header of this data
759    * @param parent  the instances node
760    * @return        the generated Instances
761    * @throws Exception  if generation fails, e.g., due to unknown attribute type
762    */
763   protected Instances createInstances(Instances header, Element parent) throws Exception {
764     Instances   result;
765     Vector  list;
766     int     i;
767     
768     result = new Instances(header, 0);
769     
770     list = getChildTags(parent, TAG_INSTANCE);
771     for (i = 0; i < list.size(); i++)
772       result.add(createInstance(result, (Element) list.get(i)));
773     
774     return result;
775   }

777   /**
778    * generates the header from the XML document
779    * 
780    * @return        the generated header
781    * @throws Exception  if generation fails
782    */
783   protected Instances headerFromXML() throws Exception {
784     Instances   result;
785     Element root;
786     Element node;
787     Vector  list;
788     ArrayList<Attribute>    atts;
789     Version version;
790     int[]   classIndex;
791 
792     root = m_Document.getDocumentElement();
793     
794     // check version
795     version = new Version();
796     if (version.isOlder(root.getAttribute(ATT_VERSION)))
797       System.out.println(
798       "WARNING: loading data of version " + root.getAttribute(ATT_VERSION)
799       + " with version " + Version.VERSION);
800     
801     // attributes
802     list       = getChildTags(root, TAG_HEADER);
803     node       = (Element) list.get(0);
804     list       = getChildTags(node, TAG_ATTRIBUTES);
805     node       = (Element) list.get(0);
806     classIndex = new int[1];
807     atts       = createAttributes(node, classIndex);
808 
809     // generate header
810     result = new Instances(root.getAttribute(ATT_NAME), atts, 0);
811     result.setClassIndex(classIndex[0]);
812     
813     return result;
814   }
