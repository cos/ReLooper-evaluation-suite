39  /**
40   * A class for transforming options listed in XML to a regular WEKA command
41   * line string.<p>

49    /** tag for a single option. */
50    public final static String TAG_OPTION = "option";
51  
52    /** tag for a list of options. */
53    public final static String TAG_OPTIONS = "options";
54  
55    /** the name attribute. */
56    public final static String ATT_NAME = "name";
57  
58    /** the type attribute. */
59    public final static String ATT_TYPE = "type";
60  
61    /** the value attribute. */
62    public final static String ATT_VALUE = "value";
63  
64    /** a value of the type attribute. */
65    public final static String VAL_TYPE_FLAG = "flag";
66  
67    /** a value of the type attribute. */
68    public final static String VAL_TYPE_SINGLE = "single";
69  
70    /** a value of the type attribute. */
71    public final static String VAL_TYPE_HYPHENS = "hyphens";
72  
73    /** a value of the type attribute. */
74    public final static String VAL_TYPE_QUOTES = "quotes";
75  
76    /** a value of the type attribute. */
77    public final static String VAL_TYPE_CLASSIFIER = "classifier";
78  
79    /** a value of the type attribute. */
80    public final static String VAL_TYPE_OPTIONHANDLER = "optionhandler";
81  
82    /** the root node. */
83    public final static String ROOT_NODE = TAG_OPTIONS;
84  
85    /** the DTD for the XML file. */
86    public final static String DOCTYPE = 
87      "<!DOCTYPE " + ROOT_NODE + "\n"
88      + "[\n"
89      + "   <!ELEMENT " + TAG_OPTIONS + " (" + TAG_OPTION + ")*>\n"
90      + "   <!ATTLIST " + TAG_OPTIONS + " " + ATT_TYPE + " CDATA \"" + VAL_TYPE_OPTIONHANDLER + "\">\n"
91      + "   <!ATTLIST " + TAG_OPTIONS + " " + ATT_VALUE + " CDATA \"\">\n"
92      + "   <!ELEMENT " + TAG_OPTION + " (#PCDATA | " + TAG_OPTIONS + ")*>\n"
93      + "   <!ATTLIST " + TAG_OPTION + " " + ATT_NAME + " CDATA #REQUIRED>\n"
94      + "   <!ATTLIST " + TAG_OPTION + " " + ATT_TYPE + " (" + VAL_TYPE_FLAG + " | " + VAL_TYPE_SINGLE + " | " + VAL_TYPE_HYPHENS + " | " + VAL_TYPE_QUOTES + ") \"" + VAL_TYPE_SINGLE + "\">\n"
95      + "]\n"
96      + ">";

101   /**

104    * @throws Exception  if the construction of the DocumentBuilder fails
105    * @see       #setValidating(boolean)

107   public XMLOptions() throws Exception {
108     m_XMLDocument = new XMLDocument(); 
109     m_XMLDocument.setRootNode(ROOT_NODE);
110     m_XMLDocument.setDocType(DOCTYPE);

204   /**
205    * pushes any options with type VAL_TYPE_HYPHENS to the end, i.e., the "--" 
206    * are really added at the end.
207    * 
208    * @see       #VAL_TYPE_HYPHENS
209    */
210   protected void fixHyphens() {
211     NodeList    list;
212     Vector<Element> hyphens;
213     int     i;
214     Node    node;
215     Node    tmpNode;
216     boolean isLast;
217 
218     // get all option tags
219     list = m_XMLDocument.findNodes("//" + TAG_OPTION);
220 
221     // get all hyphen tags
222     hyphens = new Vector<Element>();
223     for (i = 0; i < list.getLength(); i++) {
224       if (((Element) list.item(i)).getAttribute(ATT_TYPE).equals(VAL_TYPE_HYPHENS))
225     hyphens.add((Element)list.item(i));
226     }
227 
228     // check all hyphen tags whether they are at the end, if not fix it
229     for (i = 0; i < hyphens.size(); i++) {
230       node = (Node) hyphens.get(i);
231 
232       // at the end?
233       isLast  = true;
234       tmpNode = node;
235       while (tmpNode.getNextSibling() != null) {
236     // normal tag?
237     if (tmpNode.getNextSibling().getNodeType() == Node.ELEMENT_NODE) {
238       isLast = false;
239       break;
240     }
241     tmpNode = tmpNode.getNextSibling();
242       }
243 
244       // move
245       if (!isLast) {
246     tmpNode = node.getParentNode();
247     tmpNode.removeChild(node);
248     tmpNode.appendChild(node);
249       }
250     }
251   }

253   /**
254    * converts the given node into a command line representation and returns it.
255    * 
256    * @param parent  the node to convert to command line
257    * @return        the new command line
258    */
259   protected String toCommandLine(Element parent) {
260     Vector<String>  result;
261     Vector      list;
262     Vector      subList;
263     NodeList        subNodeList;
264     String[]        params;
265     int         i;
266     int         n;
267     String      tmpStr;
268     
269     result = new Vector<String>();
270     
271     // "options" tag
272     if (parent.getNodeName().equals(TAG_OPTIONS)) {
273       // children
274       list = XMLDocument.getChildTags(parent);
275 
276       if (parent.getAttribute(ATT_TYPE).equals(VAL_TYPE_CLASSIFIER)) {
277     System.err.println(
278         "Type '" + VAL_TYPE_CLASSIFIER + "' is deprecated, "
279         + "use '" + VAL_TYPE_OPTIONHANDLER + "' instead!");
280     parent.setAttribute(ATT_TYPE, VAL_TYPE_OPTIONHANDLER);
281       }
282       
283       if (parent.getAttribute(ATT_TYPE).equals(VAL_TYPE_OPTIONHANDLER)) {
284     result.add(parent.getAttribute(ATT_VALUE));
285 
286     // hyphens?
287     if (    (list.size() > 0)
288          && (parent.getParentNode() != null) 
289          && (parent.getParentNode() instanceof Element) 
290          && (((Element) parent.getParentNode()).getNodeName().equals(TAG_OPTION)) 
291          && (((Element) parent.getParentNode()).getAttribute(ATT_TYPE).equals(VAL_TYPE_HYPHENS)) )
292       result.add("--");
293       }
294 
295       // process children
296       for (i = 0; i < list.size(); i++) {
297     tmpStr = toCommandLine((Element) list.get(i));
298     try {
299       params = Utils.splitOptions(tmpStr);
300       for (n = 0; n < params.length; n++)
301         result.add(params[n]);
302     }
303     catch (Exception e) {
304       System.err.println("Error splitting: " + tmpStr);
305       e.printStackTrace();
306     }
307       }
308     }
309     // "option" tag
310     else if (parent.getNodeName().equals(TAG_OPTION)) {
311       subList     = XMLDocument.getChildTags(parent);
312       subNodeList = parent.getChildNodes();
313 
314       result.add("-" + parent.getAttribute(ATT_NAME));
315 
316       // single argument
317       if (parent.getAttribute(ATT_TYPE).equals(VAL_TYPE_SINGLE)) {
318     if (    (subNodeList.getLength() > 0) 
319          && (subNodeList.item(0).getNodeValue().trim().length() > 0) )
320           result.add(subNodeList.item(0).getNodeValue());
321       }
322       // compound argument surrounded by quotes
323       else if (parent.getAttribute(ATT_TYPE).equals(VAL_TYPE_QUOTES)) {
324     result.add(toCommandLine((Element) subList.get(0)));
325       }
326       // classname + further options after "--"
327       else if (parent.getAttribute(ATT_TYPE).equals(VAL_TYPE_HYPHENS)) {
328     tmpStr = toCommandLine((Element) subList.get(0));
329     try {
330       params = Utils.splitOptions(tmpStr);
331       for (n = 0; n < params.length; n++)
332         result.add(params[n]);
333     }
334     catch (Exception e) {
335       System.err.println("Error splitting: " + tmpStr);
336       e.printStackTrace();
337     }
338       }
339     }
340     // other tag
341     else {
342       System.err.println("Unsupported tag '" + parent.getNodeName() + "' - skipped!");
343     }
344     
345     return Utils.joinOptions(result.toArray(new String[result.size()]));
346   }

348   /**
349    * returns the given DOM document as command line.
350    * 
351    * @return        the document as command line
352    * @throws Exception  if anything goes wrong initializing the parsing
353    */
354   public String toCommandLine() throws Exception {
355     return toCommandLine(getDocument().getDocumentElement());
356   }
357 
358   /**
359    * returns the current DOM document as string array.
360    * 
361    * @return        the document as string array
362    * @throws Exception  if anything goes wrong initializing the parsing
363    */
364   public String[] toArray() throws Exception {
365     return Utils.splitOptions(toCommandLine());
366   }

368   /**
369    * returns the object in a string representation (as indented XML output).
370    * 
371    * @return        the object in a string representation
372    */
373   public String toString() {
374     return getXMLDocument().toString();
375   }

386   /**
387    * for testing only. prints the given XML file, the resulting commandline and
388    * the string array.
389    * 
390    * @param args    the commandline options.
391    * @throws Exception  if something goes wrong
392    */
393   public static void main(String[] args) throws Exception {
394     if (args.length > 0) {
395       System.out.println("\nXML:\n\n" + new XMLOptions(args[0]).toString()); 
396 
397       System.out.println("\nCommandline:\n\n" + new XMLOptions(args[0]).toCommandLine());
398 
399       System.out.println("\nString array:\n");
400       String[] options = new XMLOptions(args[0]).toArray();
401       for (int i = 0; i < options.length; i++)
402     System.out.println(options[i]);
403     }
404   }
