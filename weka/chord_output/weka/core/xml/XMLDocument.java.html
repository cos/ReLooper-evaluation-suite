313    * 
314    * @param docType     the DOCTYPE definition (no checking happens!), can be null
315    * @param rootNode    the name of the root node (must correspond to the one 
316    *                given in <code>docType</code>) 
317    * @return        returns the just created DOM document for convenience
318    */
319   public Document newDocument(String docType, String rootNode) {
320     m_Document = getBuilder().newDocument();
321     m_Document.appendChild(m_Document.createElement(rootNode));
322     setDocType(docType);
323     
324     return getDocument();
325   }

327   /**
328    * parses the given XML string (can be XML or a filename) and returns a
329    * DOM Document.
330    * 
331    * @param xml     the xml to parse (if "<?xml" is not found then it is considered a file)
332    * @return        the parsed DOM document
333    * @throws Exception  if something goes wrong with the parsing
334    */
335   public Document read(String xml) throws Exception {
336     if (xml.toLowerCase().indexOf("<?xml") > -1)
337       return read(new ByteArrayInputStream(xml.getBytes()));
338     else
339       return read(new File(xml));
340   }

432   }
433   
434   /**
435    * returns all non tag-children from the given node.
436    * 
437    * @param parent  the node to get the children from
438    * @param name    the name of the tags to return, "" for all
439    * @return        a vector containing all the non-text children
440    */
441   public static Vector<Element> getChildTags(Node parent, String name) {
442     Vector<Element>         result;
443     int            i;
444     NodeList       list;
445     
446     result = new Vector<Element>();
447     
448     list = parent.getChildNodes();
449     for (i = 0; i < list.getLength(); i++) {
450       if (!(list.item(i) instanceof Element))
451     continue;
452       // only tags with a certain name?
453       if (name.length() != 0) {
454     if (!((Element) list.item(i)).getTagName().equals(name))
455       continue;
456       }
457       result.add((Element)list.item(i));
458     }
459     
460     return result;
461   }

463   /**
464    * Returns the specified result of the XPath expression. 
465    * Can return null if an error occurred.
466    * 
467    * @param xpath   the XPath expression to run on the document
468    * @param type    the type of the result
469    * @return        the result
470    */
471   protected Object eval(String xpath, QName type) {
472     Object  result;
473     
474     try {
475       result = m_XPath.evaluate(xpath, m_Document, type);
476     }
477     catch (Exception e) {
478       e.printStackTrace();
479       result = null;
480     }
481     
482     return result;
483   }

538   /**
539    * returns the text between the opening and closing tag of a node
540    * (performs a <code>trim()</code> on the result).
541    * 
542    * @param node    the node to get the text from
543    * @return        the content of the given node
544    */
545   public static String getContent(Element node) {
546     NodeList       list;
547     Node           item;
548     int            i;
549     String         result;
550     
551     result = "";
552     list   = node.getChildNodes();
553     
554     for (i = 0; i < list.getLength(); i++) {
555       item = list.item(i);
556       if (item.getNodeType() == Node.TEXT_NODE)
557     result += item.getNodeValue();
558     }
559     
560     return result.trim();
561   }

563   /**
564    * turns the given node into a XML-stringbuffer according to the depth.
565    * 
566    * @param buf     the stringbuffer so far
567    * @param parent  the current node
568    * @param depth   the current depth
569    * @return        the new XML-stringbuffer
570    */
571   protected StringBuffer toString(StringBuffer buf, Node parent, int depth) {
572     NodeList       list;
573     Node           node;
574     int            i;
575     int            n;
576     String         indent;
577     NamedNodeMap   atts;
578     
579     // build indent
580     indent = "";
581     for (i = 0; i < depth; i++)
582       indent += "   ";
583     
584     if (parent.getNodeType() == Node.TEXT_NODE) {
585       if (!parent.getNodeValue().trim().equals(""))
586     buf.append(indent + parent.getNodeValue().trim() + "\n");
587     }
588     else 
589       if (parent.getNodeType() == Node.COMMENT_NODE) {
590     buf.append(indent + "<!--" + parent.getNodeValue() + "-->\n");
591       }
592       else {
593     buf.append(indent + "<" + parent.getNodeName());
594     // attributes?
595     if (parent.hasAttributes()) {
596       atts = parent.getAttributes();
597       for (n = 0; n < atts.getLength(); n++) {
598         node = atts.item(n);
599         buf.append(" " + node.getNodeName() + "=\"" + node.getNodeValue() + "\"");
600       }
601     }
602     // children?
603     if (parent.hasChildNodes()) {
604       list = parent.getChildNodes();
605       // just a text node?
606       if ( (list.getLength() == 1) && (list.item(0).getNodeType() == Node.TEXT_NODE) ) {
607         buf.append(">");
608         buf.append(list.item(0).getNodeValue().trim());
609         buf.append("</" + parent.getNodeName() + ">\n");
610       }
611       else {
612         buf.append(">\n");
613         for (n = 0; n < list.getLength(); n++) {
614           node = list.item(n);
615           toString(buf, node, depth + 1);
616         }
617         buf.append(indent + "</" + parent.getNodeName() + ">\n");
618       }
619     }
620     else {
621       buf.append("/>\n");
622     }
623       }
624     
625     return buf;
626   }

635   /**
636    * returns the current DOM document as XML-string.
637    * 
638    * @return        the document as XML-string representation
639    */
640   public String toString() {
641     String         header;
642     
643     header = PI + "\n\n";
644     if (getDocType() != null)
645       header += getDocType() + "\n\n";
646     
647     return toString(new StringBuffer(header), getDocument().getDocumentElement(), 0).toString();
648   }

659   /**
660    * for testing only. takes the name of an XML file as first arg, reads that
661    * file, prints it to stdout and if a second filename is given, writes the
662    * parsed document to that again.
663    * 
664    * @param args    the commandline arguments
665    * @throws Exception  if something goes wrong
666    */
667   public static void main(String[] args) throws Exception {
668     XMLDocument     doc;
669     
670     if (args.length > 0) {
671       doc = new XMLDocument();
672       
673       // read
674       doc.read(args[0]);
675       
676       // print to stdout
677       doc.print();
678       
679       // output?
680       if (args.length > 1) {
681     doc.write(args[1]);
682       }
683     }
684   }
