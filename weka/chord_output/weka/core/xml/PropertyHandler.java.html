22  package weka.core.xml;
23  
24  import weka.core.RevisionHandler;
25  import weka.core.RevisionUtils;
26  
27  import java.util.Enumeration;
28  import java.util.HashSet;
29  import java.util.Hashtable;
30  
31  /**
32   * This class stores information about properties to ignore or properties
33   * that are allowed for a certain class.
34   * 
35   * @author FracPete (fracpete at waikato dot ac dot nz)
36   * @version $Revision: 5953 $ 
37   */
38  public class PropertyHandler
39     implements RevisionHandler {
40    
41     /** 
42      * contains display names of properties to ignore in the serialization
43      * process
44      * 
45      * @see #ignored()
46      * @see #addIgnored(String)
47      * @see #removeIgnored(String)
48      * @see #isIgnored(String)
49      */
50    protected Hashtable<Object,HashSet<String>> m_Ignored = null;
51     
52     /**
53      * lists for a class the properties allowed to use for setting and getting.
54      * if a class is not listed, then all get/set-methods are allowed.<br>
55      * Mapping: classname (String) - properties (HashSet, containing the Property-Names)
56      * 
57      * @see #allowed()
58      * @see #addAllowed(Class,String)
59      * @see #removeAllowed(Class,String)
60      * @see #isAllowed(Class,String)
61      */
62    protected Hashtable<Object,HashSet<String>> m_Allowed = null;
63  
64     /**
65      * initializes the handling 
66      */
67     public PropertyHandler() {
68        super();
69  
70        m_Ignored = new Hashtable<Object,HashSet<String>>();
71        m_Allowed = new Hashtable<Object,HashSet<String>>();
72     }
73     
74     /**
75      * returns an enumeration of the stored display names and classes of 
76      * properties to ignore.<br> 
77      * <b>NOTE:</b> String and Class Objects are mixed in this enumeration, depending
78      * whether it is a global property to ignore or just one for a certain class!
79      * 
80      * @return the display names and classes
81      * @see #m_Ignored
82      */
83     public Enumeration ignored() {
84        return m_Ignored.keys();
85     }
86     
87     /**
88      * adds the given display name of a property to the ignore list. Can either
89      * be a complete path (e.g. <code>__root__.options</code>) or only a 
90      * property name (e.g. <code>options</code>). In the latter case it matches 
91      * all occurences of this display name.
92      * 
93      * @param displayName the property to ignore
94      * @see #m_Ignored 
95      */
96     public void addIgnored(String displayName) {
97        HashSet<String>        list;
98        
99        list = new HashSet<String>();
100       list.add(displayName);
101       
102       m_Ignored.put(displayName, list);
103    }
104    
105    /**
106     * adds the given class with the display name of a property to the ignore list.
107     * I.e. this property is only ignored for this class. 
108     * 
109     * @param c the class for which a property is to be ignored
110     * @param displayName the property to ignore
111     * @see #m_Ignored 
112     */
113    public void addIgnored(Class c, String displayName) {
114       HashSet<String>        list;
115       
116       // retrieve list
117       if (m_Ignored.contains(c)) {
118          list = (HashSet<String>) m_Ignored.get(c);
119       }
120       else {
121          list = new HashSet<String>();
122          m_Ignored.put(c, list);
123       }
124       
125       list.add(displayName);
126    }
127    
128    /**
129     * removes the given display name from the ignore list. returns whether the 
130     * removing was succesful, i.e. whether the display name was in the list.
131     * 
132     * @param displayName the property to remove from the ignore list
133     * @return whether the ignore list contained the specified property
134     * @see #m_Ignored
135     */
136    public boolean removeIgnored(String displayName) {
137       return (m_Ignored.remove(displayName) != null);
138    }
139    
140    /**
141     * removes the given display name from the ignore list of the class. 
142     * returns whether the removing was succesful, i.e. whether the display 
143     * name was in the list.
144     * 
145     * @param c the class to remove the property from
146     * @param displayName the property to remove from the ignore list
147     * @return whether the ignore list contained the specified property
148     * @see #m_Ignored
149     */
150    public boolean removeIgnored(Class c, String displayName) {
151       HashSet        list;
152       
153       // retrieve list
154       if (m_Ignored.contains(c))
155          list = (HashSet) m_Ignored.get(c);
156       else
157          list = new HashSet();
158       
159       return list.remove(displayName);
160    }
161    
162    /**
163     * checks whether the given display name is an ignored property
164     * 
165     * @param displayName the property to check whether it is on the ignore list
166     * @return whether the property is in the ignored list
167     * @see #m_Ignored
168     */
169    public boolean isIgnored(String displayName) {
170       return m_Ignored.containsKey(displayName);
171    }
172    
173    /**
174     * checks whether the given display name of a certain class is an ignored 
175     * property. It only checks for this certain class and no derivative classes.
176     * If you also want to check for derivative classes, use 
177     * <code>isIgnored(Object,String)</code>. 
178     * 
179     * @param c the class to check the property for
180     * @param displayName the property to check whether it is on the ignore list
181     * @return whether the property is in the ignored list
182     * @see #m_Ignored
183     * @see #isIgnored(Object, String)
184     */
185    public boolean isIgnored(Class c, String displayName) {
186       HashSet        list;
187       
188       // retrieve list
189       if (m_Ignored.containsKey(c))
190          list = (HashSet) m_Ignored.get(c);
191       else
192          list = new HashSet();
193 
194       return list.contains(displayName);
195    }
196    
197    /**
198     * checks whether the given display name of a given object is an ignored 
199     * property. The object is checked for each stored class whether it is an 
200     * <code>instanceof</code>. If the class is not stored then it will default
201     * to <code>false</code>, since there are no restrictions for this class. 
202     * 
203     * @param o the object to check the property for
204     * @param displayName the property to check whether it is on the ignore list
205     * @return whether the property is in the ignored list
206     * @see #m_Ignored
207     */
208    public boolean isIgnored(Object o, String displayName) {
209       Enumeration    enm;
210       Class          c;
211       Object         element;
212       boolean        result;
213       HashSet        list;
214       
215       result = false;
216       
217       enm = ignored();
218       while (enm.hasMoreElements()) {
219          element = enm.nextElement();
220          
221          // has to be class! not a display name
222          if (!(element instanceof Class))
223             continue;
224          
225          c = (Class) element;
226          
227          // is it an instance of this class?
228          if (c.isInstance(o)) {
229             list   = (HashSet) m_Ignored.get(c);
230             result = list.contains(displayName); 
231             break;
232          }
233       }
234       
235       return result;
236    }
237    
238    /**
239     * returns an enumeration of the classnames for which only certain properties
240     * (display names) are allowed
241     * 
242     * @return the classnames with restriction to properties
243     */
244    public Enumeration allowed() {
245       return m_Allowed.keys();
246    }
247    
248    /**
249     * adds the given property (display name) to the list of allowed properties 
250     * for the specified class.
251     * 
252     * @param c the class to add a property for
253     * @param displayName the property to allow for the class
254     * @see #m_Allowed
255     */
256    public void addAllowed(Class c, String displayName) {
257       HashSet<String>        list;
258       
259       // retrieve list
260       list = (HashSet<String>) m_Allowed.get(c);
261       if (list == null) {
262          list = new HashSet<String>();
263          m_Allowed.put(c, list);
264       }
265       
266       // add property
267       list.add(displayName);
268    }
269    
270    /**
271     * removes the given property (display name) for the specified class from 
272     * the list of allowed properties.
273     * 
274     * @param c the class to remove the property for
275     * @param displayName the property to remove
276     * @return whether the property was found
277     * @see #m_Allowed
278     */
279    public boolean removeAllowed(Class c, String displayName) {
280       boolean           result;
281       HashSet           list;
282       
283       result = false;
284       
285       // retrieve list
286       list = (HashSet) m_Allowed.get(c);
287       
288       // remove property
289       if (list != null)
290          result = list.remove(displayName);
291       
292       return result;
293    }
294    
295    /**
296     * returns whether the given property (display name) is allowed for the
297     * given class. It only checks for this certain class and no derivative classes.
298     * If you also want to check for derivative classes, use 
299     * <code>isAllowed(Object,String)</code>.
300     * 
301     * @param c the class to check the property for
302     * @param displayName the property (display name) to check
303     * @return whether the property is allowed in that context
304     * @see #m_Allowed
305     * @see #isAllowed(Object, String) 
306     */
307    public boolean isAllowed(Class c, String displayName) {
308       boolean        result;
309       HashSet        list;
310       
311       result = true;
312       
313       // retrieve list
314       list = (HashSet) m_Allowed.get(c);
315       
316       // check list
317       if (list != null)
318          result = list.contains(displayName);
319       
320       return result;
321    }
322    
323    /**
324     * returns whether the given property (display name) is allowed for the given
325     * object . The object is checked for each stored class whether it is an 
326     * <code>instanceof</code>. If the class is not stored then it will default
327     * to <code>true</code>, since there are no restrictions for this class.
328     * 
329     * @param o the object to check the property for
330     * @param displayName the property (display name) to check
331     * @return whether the property is allowed in that context 
332     */
333    public boolean isAllowed(Object o, String displayName) {
334       Enumeration    enm;
335       Class          c;
336       boolean        result;
337       HashSet        list;
338       
339       result = true;
340       
341       enm = allowed();
342       while (enm.hasMoreElements()) {
343          c = (Class) enm.nextElement();
344          
345          // is it an instance of this class?
346          if (c.isInstance(o)) {
347             list   = (HashSet) m_Allowed.get(c);
348             result = list.contains(displayName); 
349             break;
350          }
351       }
352       
353       return result;
354    }
355    
356    /**
357     * Returns the revision string.
358     * 
359     * @return       the revision
360     */
361    public String getRevision() {
362      return RevisionUtils.extract("$Revision: 5953 $");
363    }
364 }
