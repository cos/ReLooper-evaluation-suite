54  /**
55   * With this class objects can be serialized to XML instead into a binary 
56   * format. It uses introspection (cf. beans) to retrieve the data from the
57   * given object, i.e. it can only access beans-conform fields automatically.
58   * <p>
59   * The generic approach of writing data as XML can be overriden by adding 
60   * custom methods for reading/writing in a derived class
61   * (cf. <code>m_Properties</code>, <code>m_CustomMethods</code>).<br>
62   * Custom read and write methods must have the same signature (and also be 
63   * <code>public</code>!) as the <code>readFromXML</code> and <code>writeToXML</code>
64   * methods. Methods that apply to the naming rule <code>read + property name</code>
65   * are added automatically to the list of methods by the method 
66   * <code>XMLSerializationMethodHandler.addMethods(...)</code>.  
67   * <p>
68   * Other properties that are not conform the bean set/get-methods have to be 
69   * processed manually in a derived class (cf. <code>readPostProcess(Object)</code>, 
70   * <code>writePostProcess(Object)</code>).
71   * <p>
72   * For a complete XML serialization/deserialization have a look at the 
73   * <code>KOML</code> class.
74   * <p>
75   * If a stored class has a constructor that takes a String to initialize
76   * (e.g. String or Double) then the content of the tag will used for the
77   * constructor, e.g. from 
78   * <pre>&lt;object name="name" class="String" primitive="no"&gt;Smith&lt;/object&gt;</pre>
79   * "Smith" will be used to instantiate a String object as constructor argument.
80   * <p>   
81   * 
82   * @see KOML
83   * @see #fromXML(Document)
84   * @see #toXML(Object)
85   * @see #m_Properties
86   * @see #m_CustomMethods
87   * @see #readPostProcess(Object)
88   * @see #writePostProcess(Object)
89   * @see #readFromXML(Element)
90   * @see #writeToXML(Element, Object, String)
91   * 
92   * @author FracPete (fracpete at waikato dot ac dot nz)
93   * @version $Revision: 5953 $ 
94   */
95  public class XMLSerialization
96     implements RevisionHandler {

201    /**
202     * generates internally a new XML document and clears also the IgnoreList and
203     * the mappings for the Read/Write-Methods
204     * 
205     * @throws Exception if something goes wrong
206     */
207    public void clear() throws Exception {
208       m_Document = new XMLDocument();
209       m_Document.setValidating(true);
210       m_Document.newDocument(DOCTYPE, ROOT_NODE);
211       
212       m_Properties        = new PropertyHandler();
213       m_CustomMethods     = new XMLSerializationMethodHandler(this);
214 
215       m_ClassnameOverride = new Hashtable<Class,String>();
216       // java.io.File is sometimes represented as another class:
217       // - Win32: sun.awt.shell.Win32ShellFolder2 
218       // - Linux: sun.awt.shell.DefaultShellFolder
219       // -> we set it to "java.io.File"
220       m_ClassnameOverride.put(java.io.File.class, java.io.File.class.getName());
221       
222       setVersion(Version.VERSION); 
223       
224       m_CurrentNode = null;
225    }

255    /**
256     * Checks the version in the current Document with the one of the current
257     * release. If the version differ, a warning is printed.
258     */
259    private void checkVersion() {
260       String            versionStr;
261       Version           version;
262       
263       version    = new Version();
264       versionStr = getVersion();
265       if (versionStr.equals(""))
266          System.out.println("WARNING: has no version!");
267       else if (version.isOlder(versionStr))
268          System.out.println("WARNING: loading a newer version (" + versionStr + " > " + Version.VERSION + ")!");
269       else if (version.isNewer(versionStr))
270          System.out.println("NOTE: loading an older version (" + versionStr + " < " + Version.VERSION + ")!");
271    }

273    /**
274     * returns a hashtable with PropertyDescriptors that have "get" and "set" 
275     * methods indexed by the property name.
276     * 
277     * @see java.beans.PropertyDescriptor
278     * @param o the object to retrieve the descriptors from
279     * @return the PropertyDescriptors indexed by name of the property
280     * @throws Exception if the introspection fails
281     */
282    protected Hashtable getDescriptors(Object o) throws Exception {
283       BeanInfo                   info;
284       PropertyDescriptor[]       desc;
285       int                        i;
286       Hashtable<String,PropertyDescriptor>                  result;
287       
288       result = new Hashtable<String,PropertyDescriptor>();
289 
290       info = Introspector.getBeanInfo(o.getClass());
291       desc = info.getPropertyDescriptors();
292       for (i = 0; i < desc.length; i++) {
293          // get AND set method?
294          if ( (desc[i].getReadMethod() != null) && (desc[i].getWriteMethod() != null) ) {
295             // in ignore list, i.e. a general ignore without complete path?
296             if (m_Properties.isIgnored(desc[i].getDisplayName()))
297                continue;
298             
299             // in ignore list of the class?
300             if (m_Properties.isIgnored(o, desc[i].getDisplayName()))
301                continue;
302             
303             // not an allowed property
304             if (!m_Properties.isAllowed(o, desc[i].getDisplayName()))
305                continue;
306             
307             result.put(desc[i].getDisplayName(), desc[i]);
308          }
309       }
310       
311       return result;
312    }

314    /**
315     * returns the path of the "name" attribute from the root down to this node
316     * (including it).
317     * 
318     * @param node the node to get the path for
319     * @return the complete "name" path of this node
320     */
321    protected String getPath(Element node) {
322       String            result;
323       
324       result = node.getAttribute(ATT_NAME);
325 
326       while (node.getParentNode() != node.getOwnerDocument()) {
327          node   = (Element) node.getParentNode(); 
328          result = node.getAttribute(ATT_NAME) + "." + result;
329       }
330       
331       return result;
332    }

348    /**
349     * turns the given string into a boolean, if a positive number is given, 
350     * then zero is considered FALSE, every other number TRUE; the empty string 
351     * is also considered being FALSE
352     * 
353     * @param s the string to turn into a boolean
354     * @return the string as boolean
355     */
356    protected boolean stringToBoolean(String s) {
357       if (s.equals(""))
358          return false;
359       else if (s.equals(VAL_YES))
360          return true;
361       else if (s.equalsIgnoreCase("true"))
362          return true;
363       else if (s.replaceAll("[0-9]*", "").equals(""))
364          return (Integer.parseInt(s) != 0);
365       else
366          return false;
367    }

398    /**
399     * appends a new node to the parent with the given parameters
400     * 
401     * @param parent the parent of this node. if it is <code>null</code> the 
402     *        document root element is used
403     * @param name the name of the node
404     * @param classname the classname for this node
405     * @param primitive whether it is a primitve data type or not (i.e. an object)
406     * @param array the dimensions of the array (0 if not an array)
407     * @param isnull whether it is null
408     * @return the generated node 
409     */
410    protected Element addElement(Element parent, String name, String classname, boolean primitive, int array, boolean isnull) {
411       Element           result;
412 
413       if (parent == null)
414          result = m_Document.getDocument().getDocumentElement();
415       else
416          result = (Element) parent.appendChild(m_Document.getDocument().createElement(TAG_OBJECT));
417       
418       // attributes
419       // mandatory attributes:
420       result.setAttribute(ATT_NAME, name);
421       result.setAttribute(ATT_CLASS, classname);
422       
423       // add following attributes only if necessary, i.e., different from default:
424       if (!booleanToString(primitive).equals(ATT_PRIMITIVE_DEFAULT))
425         result.setAttribute(ATT_PRIMITIVE, booleanToString(primitive));
426 
427       // multi-dimensional array?
428       if (array > 1) {
429         result.setAttribute(ATT_ARRAY, Integer.toString(array));
430       }
431       // backwards compatible: 0 -> no array ("no"), 1 -> 1-dim. array ("yes")
432       else {
433         if (!booleanToString(array == 1).equals(ATT_ARRAY_DEFAULT))
434           result.setAttribute(ATT_ARRAY, booleanToString(array == 1));
435       }
436       
437       if (!booleanToString(isnull).equals(ATT_NULL_DEFAULT))
438         result.setAttribute(ATT_NULL, booleanToString(isnull));
439       
440       return result;
441    }

443    /**
444     * if the class of the given object (or one of its ancestors) is stored in 
445     * the classname override hashtable, then the override name is returned 
446     * otherwise the classname of the given object.
447     * 
448     * @param o          the object to check for overriding its classname
449     * @return           if overridden then the classname stored in the hashtable,
450     *                   otherwise the classname of the given object
451     * @see              #m_ClassnameOverride
452     */
453    protected String overrideClassname(Object o) {
454       Enumeration    enm;
455       String         result;
456       Class          currentCls;
457      
458       result = o.getClass().getName();
459 
460       // check overrides
461       enm    = m_ClassnameOverride.keys();
462       while (enm.hasMoreElements()) {
463          currentCls = (Class) enm.nextElement();
464          if (currentCls.isInstance(o)) {
465            result = (String) m_ClassnameOverride.get(currentCls);
466            break;
467          }
468       }
469       
470       return result;
471    }

473    /**
474     * if the given classname is stored in the classname override hashtable, 
475     * then the override name is returned otherwise the given classname.
476     * <b>Note:</b> in contrast to <code>overrideClassname(Object)</code> does
477     * this method only look for exact name matches. The other method checks
478     * whether the class of the given object is a subclass of any of the stored
479     * overrides.  
480     * 
481     * @param classname  the classname to check for overriding
482     * @return           if overridden then the classname stored in the hashtable,
483     *                   otherwise the given classname
484     * @see              #m_ClassnameOverride
485     * @see              #overrideClassname(Object)
486     */
487    protected String overrideClassname(String classname) {
488       Enumeration    enm;
489       String         result;
490       Class          currentCls;
491      
492       result = classname;
493 
494       // check overrides
495       enm    = m_ClassnameOverride.keys();
496       while (enm.hasMoreElements()) {
497          currentCls = (Class) enm.nextElement();
498          if (currentCls.getName().equals(classname)) {
499            result = (String) m_ClassnameOverride.get(currentCls);
500            break;
501          }
502       }
503       
504       return result;
505    }

524       }

542         trace(new Throwable(), name);

544      m_CurrentNode = parent;

546      node = addElement(parent, name, Boolean.TYPE.getName(), true);
547      node.appendChild(node.getOwnerDocument().createTextNode(new Boolean(o).toString()));
548      
549      return node;
550    }

726    /**
727     * adds the given Object to a DOM structure. 
728     * (only public due to reflection).<br>
729     * <b>Note:</b> <code>overrideClassname(Object)</code> is not invoked in case of
730     * arrays, since the array class could be a superclass, whereas the elements of
731     * the array can be specialized subclasses. In case of an array the method 
732     * <code>overrideClassname(String)</code> is invoked, which searches for an 
733     * exact match of the classname in the override hashtable.
734     * 
735     * @param parent the parent of this object, e.g. the class this object is a member of
736     * @param o the Object to describe in XML
737     * @param name the name of the object
738     * @return the node that was created
739     * @throws Exception if the DOM creation fails
740     * @see #overrideClassname(Object)
741     * @see #overrideClassname(String)
742     * @see #m_ClassnameOverride
743     */
744    public Element writeToXML(Element parent, Object o, String name) throws Exception {
745       String               classname;
746       Element              node;
747       Hashtable            memberlist;
748       Enumeration          enm;
749       Object               member;
750       String               memberName;
751       Method               method;
752       PropertyDescriptor   desc;
753       boolean              primitive;
754       int                  array;
755       int                  i;
756       Object               obj;
757       String               tmpStr;
758 
759       node = null;
760       
761       // for debugging only
762       if (DEBUG)
763          trace(new Throwable(), name);
764 
765       // special handling of null-objects
766       if (o == null) {
767         node = addElement(parent, name, "" + null, false, 0, true);
768         return node;
769       }
770       
771       // used for overriding the classname
772       obj = null;
773       
774       // get information about object
775       array = 0;
776       if (o.getClass().isArray())
777         array = Utils.getArrayDimensions(o);
778       if (array > 0) {
779         classname = Utils.getArrayClass(o.getClass()).getName();
780         primitive = isPrimitiveArray(o.getClass()); 
781       }
782       else {
783          // try to get property descriptor to determine real class
784          // (for primitives the getClass() method returns the corresponding Object-Class!)
785          desc = null;
786          if (parent != null)
787             desc = determineDescriptor(parent.getAttribute(ATT_CLASS), name);
788          
789          if (desc != null)
790             primitive = desc.getPropertyType().isPrimitive(); 
791          else
792             primitive = o.getClass().isPrimitive(); 
793 
794          // for primitives: retrieve primitive type, otherwise the object's real 
795          // class. For non-primitives we can't use the descriptor, since that
796          // might only return an interface as class!
797          if (primitive) {
798             classname = desc.getPropertyType().getName();
799          }
800          else {
801             obj       = o;
802             classname = o.getClass().getName();
803          }
804       }
805       
806       // fix class/primitive if parent is array of primitives, thanks to 
807       // reflection the elements of the array are objects and not primitives!
808       if (    (parent != null) 
809            && (!parent.getAttribute(ATT_ARRAY).equals(""))
810            && (!parent.getAttribute(ATT_ARRAY).equals(VAL_NO))
811            && (stringToBoolean(parent.getAttribute(ATT_PRIMITIVE))) ) {
812          primitive = true;
813          classname = parent.getAttribute(ATT_CLASS);
814          obj       = null;
815       }
816 
817       // perhaps we need to override the classname
818       if (obj != null)
819         classname = overrideClassname(obj);         // for non-arrays
820       else
821         classname = overrideClassname(classname);   // for arrays
822       
823       // create node for current object
824       node = addElement(parent, name, classname, primitive, array);
825       
826       // array? -> save as child with 'name="<index>"'
827       if (array > 0) {
828          for (i = 0; i < Array.getLength(o); i++) {
829             invokeWriteToXML(node, Array.get(o, i), Integer.toString(i));
830          }
831       }
832       // non-array
833       else {
834          // primitive? -> only toString()
835          if (primitive) {
836             node.appendChild(node.getOwnerDocument().createTextNode(o.toString()));
837          }
838          // object
839          else {
840             // process recursively members of this object 
841             memberlist = getDescriptors(o);
842             // if no get/set methods -> we assume it has String-Constructor
843             if (memberlist.size() == 0) {
844               if (!o.toString().equals("")) {
845             tmpStr = o.toString();
846             // these five entities are recognized by every XML processor
847             // see http://www.xml.com/pub/a/2001/03/14/trxml10.html
848             tmpStr = tmpStr.replaceAll("&", "&amp;")
849                            .replaceAll("\"", "&quot;")
850                            .replaceAll("'", "&apos;")
851                            .replaceAll("<", "&lt;")
852                            .replaceAll(">", "&gt;");
853             // in addition, replace some other entities as well
854             tmpStr = tmpStr.replaceAll("\n", "&#10;")
855                            .replaceAll("\r", "&#13;")
856                            .replaceAll("\t", "&#9;");
857             node.appendChild(node.getOwnerDocument().createTextNode(tmpStr));
858               }
859             }
860             else {
861                enm = memberlist.keys();
862                while (enm.hasMoreElements()) {
863                   memberName = enm.nextElement().toString();
864                   
865                   // in ignore list?
866                   if (    (m_Properties.isIgnored(memberName))
867                        || (m_Properties.isIgnored(getPath(node) + "." + memberName))
868                        || (m_Properties.isIgnored(o, getPath(node) + "." + memberName)) )
869                      continue;
870 
871                   // is it allowed?
872                   if (!m_Properties.isAllowed(o, memberName))
873                      continue;
874                   
875                   desc   = (PropertyDescriptor) memberlist.get(memberName);
876                   method = desc.getReadMethod();
877                   member = method.invoke(o, (Object[]) null);
878                   invokeWriteToXML(node, member, memberName);
879                }
880             }
881          }
882       }
883       
884       return node;
885    }

887    /**
888     * either invokes a custom method to write a specific property/class or the standard
889     * method <code>writeToXML(Element,Object,String)</code>
890     * 
891     * @param parent the parent XML node
892     * @param o the object's content will be added as children to the given parent node
893     * @param name the name of the object
894     * @return the node that was created
895     * @throws Exception if invocation or turning into XML fails
896     */
897    protected Element invokeWriteToXML(Element parent, Object o, String name) throws Exception {
898       Method         method;
899       Class[]        methodClasses;
900       Object[]       methodArgs;
901       boolean        array;
902       Element        node;
903       boolean        useDefault;
904       
905       node       = null;
906       method     = null;
907       useDefault = false;
908 
909       m_CurrentNode = parent;
910       
911       // default, if null
912       if (o == null)
913          useDefault = true;
914       
915       try {
916          if (!useDefault) {
917             array = o.getClass().isArray();
918            
919             // display name?
920             if (m_CustomMethods.write().contains(name))
921                method = (Method) m_CustomMethods.write().get(o.getClass());
922             else
923             // class?
924             if ( (!array) && (m_CustomMethods.write().contains(o.getClass())) )
925                method = (Method) m_CustomMethods.write().get(o.getClass());
926             else
927                method = null;
928             
929             useDefault = (method == null);
930          }
931 
932          // custom
933          if (!useDefault) {
934              methodClasses    = new Class[3];
935              methodClasses[0] = Element.class;
936              methodClasses[1] = Object.class;
937              methodClasses[2] = String.class;
938              methodArgs       = new Object[3];
939              methodArgs[0]    = parent;
940              methodArgs[1]    = o;
941              methodArgs[2]    = name;
942              node = (Element) method.invoke(this, methodArgs);
943          }
944          // standard
945          else {
946             node = writeToXML(parent, o, name);
947          }
948       }
949       catch (Exception e) {
950          if (DEBUG)
951             e.printStackTrace();
952          
953          if (m_CurrentNode != null) {
954            System.out.println("Happened near: " + getPath(m_CurrentNode));
955            // print it only once!
956            m_CurrentNode = null;
957          }
958          System.out.println("PROBLEM (write): " + name);
959 
960          throw (Exception) e.fillInStackTrace();
961       }
962       
963       return node;
964    }

1068   /**
1069    * returns an Object representing the primitive described by the given node.
1070    * Here we use a trick to return an object even though its a primitive: by 
1071    * creating a primitive array with reflection of length 1, setting the 
1072    * primtive value as real object and then returning the "object" at 
1073    * position 1 of the array.
1074    * 
1075    * @param node the node to return the value as "primitive" object
1076    * @return the primitive as "pseudo" object
1077    * @throws Exception if the instantiation of the array fails or any of the
1078    *         String conversions fails
1079    */
1080   protected Object getPrimitive(Element node) throws Exception {
1081      Object            result;
1082      Object            tmpResult;
1083      Class             cls;
1084      
1085      cls       = determineClass(node.getAttribute(ATT_CLASS));
1086      tmpResult = Array.newInstance(cls, 1);
1087      
1088      if (cls == Boolean.TYPE)
1089         Array.set(tmpResult, 0, new Boolean(XMLDocument.getContent(node)));
1090      else
1091      if (cls == Byte.TYPE)
1092         Array.set(tmpResult, 0, new Byte(XMLDocument.getContent(node)));
1093      else
1094      if (cls == Character.TYPE)
1095         Array.set(tmpResult, 0, new Character(XMLDocument.getContent(node).charAt(0)));
1096      else
1097      if (cls == Double.TYPE)
1098         Array.set(tmpResult, 0, new Double(XMLDocument.getContent(node)));
1099      else
1100      if (cls == Float.TYPE)
1101         Array.set(tmpResult, 0, new Float(XMLDocument.getContent(node)));
1102      else
1103      if (cls == Integer.TYPE)
1104         Array.set(tmpResult, 0, new Integer(XMLDocument.getContent(node)));
1105      else
1106      if (cls == Long.TYPE)
1107         Array.set(tmpResult, 0, new Long(XMLDocument.getContent(node)));
1108      else
1109      if (cls == Short.TYPE)
1110         Array.set(tmpResult, 0, new Short(XMLDocument.getContent(node)));
1111      else
1112         throw new Exception("Cannot get primitive for class '" + cls.getName() + "'!");
1113      
1114      result = Array.get(tmpResult, 0);
1115      
1116      return result;
1117   }

1255   /**
1256    * adds the specific node to the object via a set method
1257    * 
1258    * @param o            the object to set a property
1259    * @param name         the name of the object for which to set a property
1260    *                     (only for information reasons)
1261    * @param child        the value of the property to add
1262    * @return             the provided object, but augmented by the child
1263    * @throws Exception   if something goes wrong
1264    */
1265   public Object readFromXML(Object o, String name, Element child) throws Exception {
1266      Object               result;
1267      Hashtable            descriptors;
1268      PropertyDescriptor   descriptor;
1269      String               methodName;
1270      Method               method;
1271      Object[]             methodArgs;
1272      Object               tmpResult;
1273      Class                paramClass;
1274     
1275      result      = o;
1276      descriptors = getDescriptors(result);
1277      methodName  = child.getAttribute(ATT_NAME);
1278
1279      // in ignore list?
1280      if (m_Properties.isIgnored(getPath(child)))
1281         return result;
1282      
1283      // in ignore list of class?
1284      if (m_Properties.isIgnored(result, getPath(child)))
1285        return result;
1286      
1287      // is it allowed?
1288      if (!m_Properties.isAllowed(result, methodName))
1289        return result;
1290      
1291      descriptor = (PropertyDescriptor) descriptors.get(methodName);
1292
1293      // unknown property?
1294      if (descriptor == null) {
1295         if (!m_CustomMethods.read().contains(methodName))
1296            System.out.println("WARNING: unknown property '" + name + "." + methodName + "'!");
1297         return result;
1298      }
1299      
1300      method     = descriptor.getWriteMethod();
1301      methodArgs = new Object[1];
1302      tmpResult  = invokeReadFromXML(child);
1303      paramClass = method.getParameterTypes()[0];
1304      
1305      // array?
1306      if (paramClass.isArray()) {
1307         // no data?
1308         if (Array.getLength(tmpResult) == 0)
1309           return result;
1310         methodArgs[0] = (Object[]) tmpResult;
1311      }
1312      // non-array
1313      else {
1314         methodArgs[0] = tmpResult;
1315      }
1316
1317      method.invoke(result, methodArgs);
1318     
1319      return result;
1320   }

1322   /**
1323    * returns an array with the dimensions of the array stored in XML
1324    * @param node the node to determine the dimensions for
1325    * @return the dimensions of the array
1326    */
1327   protected int[] getArrayDimensions(Element node) {
1328     Vector<Element>         children;
1329     Vector<Integer>         tmpVector;      
1330     int[]          tmp;
1331     int[]          result;
1332     int            i;
1333     
1334     // have we reached the innermost dimension?
1335     if (stringToBoolean(node.getAttribute(ATT_ARRAY)))
1336       children = XMLDocument.getChildTags(node);
1337     else
1338       children = null;
1339     
1340     if (children != null) {
1341       tmpVector = new Vector<Integer>();
1342
1343       if (children.size() > 0) {
1344         // are children also arrays?
1345         tmp = getArrayDimensions((Element) children.get(0));
1346         
1347         // further dimensions
1348         if (tmp != null) {
1349           for (i = tmp.length - 1; i >= 0; i--)
1350             tmpVector.add(new Integer(tmp[i]));
1351         }
1352  
1353         // add current dimension
1354         tmpVector.add(0, new Integer(children.size()));
1355       }
1356       else {
1357         tmpVector.add(new Integer(0));
1358       }
1359       
1360       // generate result
1361       result = new int[tmpVector.size()];
1362       for (i = 0; i < result.length; i++)
1363         result[i] = ((Integer) tmpVector.get(tmpVector.size() - i - 1)).intValue();
1364     }
1365     else {
1366       result = null;
1367     }
1368     
1369     return result;
1370   }

1393      Element              child;
1394           
1395      // for debugging only
1396      if (DEBUG)
1397         trace(new Throwable(), node.getAttribute(ATT_NAME));
1398
1399      m_CurrentNode = node;
1400      
1401      result    = null;
1402      
1403      name      = node.getAttribute(ATT_NAME);
1404      classname = node.getAttribute(ATT_CLASS);
1405      primitive = stringToBoolean(node.getAttribute(ATT_PRIMITIVE));
1406      array     = stringToBoolean(node.getAttribute(ATT_ARRAY));
1407      isnull    = stringToBoolean(node.getAttribute(ATT_NULL));
1408
1409      // special handling of null
1410      if (isnull)
1411        return result;
1412
1413      children  = XMLDocument.getChildTags(node);
1414      cls       = determineClass(classname);
1415      
1416      // array
1417      if (array) {
1418         result = Array.newInstance(cls, getArrayDimensions(node));
1419         for (i = 0; i < children.size(); i++) {
1420            child = (Element) children.get(i);
1421            Array.set(result, Integer.parseInt(child.getAttribute(ATT_NAME)), invokeReadFromXML(child));
1422         }
1423      }
1424      // non-array
1425      else {
1426         // primitive/String-constructor
1427         if (children.size() == 0) {
1428            // primitive
1429            if (primitive) {
1430               result = getPrimitive(node);
1431            }
1432            // assumed String-constructor
1433            else {
1434               methodClasses    = new Class[1];
1435               methodClasses[0] = String.class;
1436               methodArgs       = new Object[1];
1437               methodArgs[0]    = XMLDocument.getContent(node);
1438               try {
1439                  constructor   = cls.getConstructor(methodClasses);
1440                  result        = constructor.newInstance(methodArgs);
1441               }
1442               catch (Exception e) {
1443                  // if it's not a class with String constructor, let's try standard constructor
1444                  try {
1445                     result = cls.newInstance();
1446                  }
1447                  catch (Exception e2) {
1448                     // sorry, can't instantiate!
1449                     result = null;
1450                     System.out.println("ERROR: Can't instantiate '" + classname + "'!");
1451                  }
1452               }
1453            }
1454         }
1455         // normal get/set methods
1456         else {
1457            result = cls.newInstance();
1458            for (i = 0; i < children.size(); i++)
1459              result = readFromXML(result, name, (Element) children.get(i));
1460         }
1461      }
1462            
1463      return result;
1464   }

1466   /**
1467    * either invokes a custom method to read a specific property/class or the standard
1468    * method <code>readFromXML(Element)</code>
1469    * 
1470    * @param node the associated XML node
1471    * @return the instance created from the XML description
1472    * @throws Exception if instantiation fails
1473    */
1474   protected Object invokeReadFromXML(Element node) throws Exception {
1475      Method         method;
1476      Class[]        methodClasses;
1477      Object[]       methodArgs;
1478      boolean        array;
1479      boolean        useDefault;
1480
1481      useDefault = false;
1482      method     = null;
1483      m_CurrentNode = node;
1484      
1485      try {
1486         // special handling of null values
1487         if (stringToBoolean(node.getAttribute(ATT_NULL)))
1488           useDefault = true;
1489        
1490         if (!useDefault) {
1491            array = stringToBoolean(node.getAttribute(ATT_ARRAY));
1492           
1493            // display name?
1494            if (m_CustomMethods.read().contains(node.getAttribute(ATT_NAME)))
1495               method = (Method) m_CustomMethods.read().get(node.getAttribute(ATT_NAME));
1496            else
1497            // class name?
1498            if ( (!array) && (m_CustomMethods.read().contains(determineClass(node.getAttribute(ATT_CLASS)))) )
1499               method = (Method) m_CustomMethods.read().get(determineClass(node.getAttribute(ATT_CLASS)));
1500            else
1501               method = null;
1502            
1503            useDefault = (method == null);
1504         }
1505
1506         // custom method
1507         if (!useDefault) {
1508            methodClasses    = new Class[1];
1509            methodClasses[0] = Element.class;
1510            methodArgs       = new Object[1];
1511            methodArgs[0]    = node;
1512            return method.invoke(this, methodArgs);
1513         }
1514         // standard
1515         else {
1516            return readFromXML(node);
1517         }
1518      }
1519      catch (Exception e) {
1520         if (DEBUG)
1521            e.printStackTrace();
1522         
1523         if (m_CurrentNode != null) {
1524           System.out.println("Happened near: " + getPath(m_CurrentNode));
1525           // print it only once!
1526           m_CurrentNode = null;
1527         }
1528         System.out.println("PROBLEM (read): " + node.getAttribute("name"));
1529
1530         throw (Exception) e.fillInStackTrace();
1531      }
1532   }

1563    * @return the XML as object

1566   public Object fromXML(Document document) throws Exception {
1567      if (!document.getDocumentElement().getNodeName().equals(ROOT_NODE))
1568         throw new Exception("Expected '" + ROOT_NODE + "' as root element, but found '" + document.getDocumentElement().getNodeName() + "'!");
1569      m_Document.setDocument(readPreProcess(document));
1570      checkVersion();
1571      return readPostProcess(invokeReadFromXML(m_Document.getDocument().getDocumentElement()));
1572   }
