24  package weka.core;
25  
26  import javax.swing.JOptionPane;
27  
28  /**
29   * A little helper class for Memory management. Very crude, since JDK 1.4 
30   * doesn't offer real Memory Management.<p/>
31   * The memory management can be disabled by using the setEnabled(boolean)
32   * method.
33   *
34   * @author    FracPete (fracpete at waikato dot ac dot nz)
35   * @version   $Revision: 5953 $
36   * @see       #setEnabled(boolean)
37   */
38  public class Memory
39    implements RevisionHandler {
40    
41    /** whether memory management is enabled */
42    protected static boolean m_Enabled = true;
43    
44    /** whether a GUI is present */
45    protected boolean m_UseGUI = false;
46  
47    /** the initial size of the JVM */
48    protected static long m_Initial = Runtime.getRuntime().totalMemory();
49  
50    /** the total memory that is used */
51    protected long m_Total;
52  
53    /** the maximum amount of memory that can be used */
54    protected long m_Max;
55  
56    /** the current runtime variable  */
57    protected Runtime m_Runtime;
58  
59    /**
60     * initializes the memory management without GUI support
61     */
62    public Memory() {
63      this(false);
64    }
65  
66    /**
67     * initializes the memory management
68     * @param useGUI      whether a GUI is present
69     */
70    public Memory(boolean useGUI) {
71      m_UseGUI  = useGUI;
72      m_Runtime = Runtime.getRuntime();
73      m_Max     = m_Runtime.maxMemory();
74      m_Total   = m_Runtime.totalMemory();
75    }
76  
77    /**
78     * returns whether the memory management is enabled
79     * 
80     * @return        true if enabled
81     */
82    public boolean isEnabled() {
83      return m_Enabled;
84    }
85  
86    /**
87     * sets whether the memory management is enabled
88     * 
89     * @param value   true if the management should be enabled
90     */
91    public void setEnabled(boolean value) {
92      m_Enabled = value;
93    }
94  
95    /**
96     * whether to display a dialog in case of a problem (= TRUE) or just print
97     * on stderr (= FALSE)
98     * 
99     * @return        true if the GUI is used
100    */
101   public boolean getUseGUI() {
102     return m_UseGUI;
103   }
104 
105   /**
106    * returns the initial size of the JVM
107    * 
108    * @return        the initial size in bytes
109    */
110   public long getInitial() {
111     return m_Initial;
112   }
113 
114   /**
115    * returns the current memory consumption
116    * 
117    * @return        the current size in bytes
118    */
119   public long getCurrent() {
120     m_Runtime = Runtime.getRuntime();
121     m_Total   = m_Runtime.totalMemory();
122 
123     return m_Total;
124   }
125 
126   /**
127    * returns the maximum amount of memory that can be assigned
128    * 
129    * @return        the maximum size in bytes
130    */
131   public long getMax() {
132     return m_Max;
133   }
134 
135   /**
136    * checks if there's still enough memory left. if ENABLED is true, then
137    * false is returned always
138    * 
139    * @return        true if out of memory (only if management enabled, 
140    *            otherwise always false)
141    */
142   public boolean isOutOfMemory() {
143     if (isEnabled())
144       return ((getMax() - getCurrent()) < (getInitial() + 200000));
145     else
146       return false;
147   }
148 
149   /**
150    * returns the amount of bytes as MB
151    * 
152    * @return        the MB amount
153    */
154   public static double toMegaByte(long bytes) {
155     return (bytes / (double) (1024 * 1024));
156   }
157 
158   /**
159    * prints an error message if OutOfMemory (and if GUI is present a dialog),
160    * otherwise nothing happens. isOutOfMemory() has to be called beforehand,
161    * since it sets all the memory parameters.
162    * @see #isOutOfMemory()
163    * @see #m_Enabled
164    */
165   public void showOutOfMemory() {
166     if (!isEnabled())
167       return;
168       
169     System.gc();
170 
171     String msg =   "Not enough memory. Please load a smaller "  
172                  + "dataset or use larger heap size.\n"
173                  + "- initial JVM size:   " 
174                  + Utils.doubleToString(toMegaByte(m_Initial), 1) + "MB\n"
175                  + "- total memory used:  " 
176                  + Utils.doubleToString(toMegaByte(m_Total), 1) + "MB\n"
177                  + "- max. memory avail.: " 
178                  + Utils.doubleToString(toMegaByte(m_Max), 1) + "MB\n"
179                  + "\n"
180                  + "Note:\n"
181                  + "The Java heap size can be specified with the -Xmx option.\n"
182                  + "E.g., to use 128MB as heap size, the command line looks like this:\n"
183                  + "   java -Xmx128m -classpath ...\n"
184                  + "This does NOT work in the SimpleCLI, the java command refers\n"
185                  + "to the one with which Weka is started.";
186     
187     System.err.println(msg);
188     
189     if (getUseGUI())
190       JOptionPane.showMessageDialog(
191           null, msg, "OutOfMemory", JOptionPane.WARNING_MESSAGE);
192   }
193 
194   /**
195    * stops all the current threads, to make a restart possible
196    */
197   public void stopThreads() {
198     int           i;
199     Thread[]      thGroup;
200     Thread        t;
201 
202     thGroup = new Thread[Thread.activeCount()];
203     Thread.enumerate(thGroup);
204 
205     for (i = 0; i < thGroup.length; i++) {
206       t = thGroup[i];
207       if (t != null) {
208         if (t != Thread.currentThread()) {
209           if (t.getName().startsWith("Thread"))
210             t.stop();
211           else if (t.getName().startsWith("AWT-EventQueue"))
212             t.stop();
213         }
214       }
215     }
216 
217     thGroup = null;
218 
219     System.gc();
220   }
221   
222   /**
223    * Returns the revision string.
224    * 
225    * @return        the revision
226    */
227   public String getRevision() {
228     return RevisionUtils.extract("$Revision: 5953 $");
229   }
230 
231   /**
232    * prints only some statistics
233    *
234    * @param args the commandline arguments - ignored
235    */
236   public static void main(String[] args) {
237     Memory mem = new Memory();
238     System.out.println(
239         "Initial memory: "
240         + Utils.doubleToString(Memory.toMegaByte(mem.getInitial()), 1) + "MB" 
241         + " (" + mem.getInitial() + ")");
242     System.out.println(
243         "Max memory: "
244         + Utils.doubleToString(Memory.toMegaByte(mem.getMax()), 1) + "MB"
245         + " (" + mem.getMax() + ")");
246   }
247 }
