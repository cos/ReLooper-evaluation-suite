23  package weka.core.pmml;
24  
25  import java.io.Serializable;
26  import java.util.ArrayList;
27  
28  import org.w3c.dom.Element;
29  import org.w3c.dom.Node;
30  import org.w3c.dom.NodeList;
31  
32  import weka.core.Attribute;
33  import weka.core.Instances;
34  import weka.core.SerializedObject;
35  
36  /**
37   * Class encapsulating the TransformationDictionary element. Contains
38   * a list of DefineFunctions and DerivedFields (if any).
39   * 
40   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com
41   * @version $Revision 1.0 $
42   */
43  class TransformationDictionary implements Serializable {
44    
45    /** The defined functions (if any) */
46    protected ArrayList<DefineFunction> m_defineFunctions = new ArrayList<DefineFunction>();
47    
48    /** The derived fields (if any) */
49    protected ArrayList<DerivedFieldMetaInfo> m_derivedFields = 
50      new ArrayList<DerivedFieldMetaInfo>();
51    
52    /**
53     * Construct a new TransformationDictionary
54     * 
55     * @param dictionary the Element containing the dictionary
56     * @param dataDictionary the data dictionary as an Instances object
57     * @throws Exception if there is a problem constructing the transformation
58     * dictionary
59     */
60    protected TransformationDictionary(Element dictionary, 
61                                    Instances dataDictionary) throws Exception {
62      
63      // set up incoming field definitions
64  /*    ArrayList<Attribute> incomingFieldDefs = new ArrayList<Attribute>();
65      for (int i = 0; i < dataDictionary.numAttributes(); i++) {
66        incomingFieldDefs.add(dataDictionary.attribute(i));
67      } */
68      
69      // get any derived fields and DefineFunctions
70      NodeList derivedL = dictionary.getChildNodes();
71      for (int i = 0; i < derivedL.getLength(); i++) {
72        Node child = derivedL.item(i);
73        if (child.getNodeType() == Node.ELEMENT_NODE) {
74          String tagName = ((Element)child).getTagName();
75          if (tagName.equals("DerivedField")) {
76            DerivedFieldMetaInfo df = new DerivedFieldMetaInfo((Element)child, null /*incomingFieldDefs*/, null);
77            m_derivedFields.add(df);
78          } else if (tagName.equals("DefineFunction")) {
79            DefineFunction defF = new DefineFunction((Element)child, null);
80            m_defineFunctions.add(defF);
81          }
82        }
83      }
84    }
85    
86    /**
87     * Set the field definitions for the derived fields. Usually called once the
88     * structure of the mining schema + derived fields has been determined. 
89     * Calling this method with an array list of field definitions in the order of 
90     * attributes in the mining schema + derived fields will allow the expressions 
91     * used in the derived fields to access the correct attribute values from the 
92     * incoming instance (also allows for derived fields to reference other
93     * derived fields). This is necessary because construction of the TransformationDictionary uses the
94     * data dictionary to reference fields (the order of fields in the data dictionary
95     * is not guaranteed to be the same as the order in the mining schema).
96     * 
97     * IMPORTANT: for derived field x to be able to reference derived field y, y must
98     * have been declared before x in the PMML file. This is because the process
99     * of constructing an input vector of values to the model proceeds in a linear
100    * left-to-right fashion - so any referenced derived field (e.g. field y),
101    * must have already computed its value when x is evaluated.
102    * 
103    * @param fieldDefs the definition of the incoming fields as an array list of attributes
104    * @throws Exception if a problem occurs
105    */
106   protected void setFieldDefsForDerivedFields(ArrayList<Attribute> fieldDefs) throws Exception {
107     for (int i = 0; i < m_derivedFields.size(); i++) {
108       m_derivedFields.get(i).setFieldDefs(fieldDefs);
109     }
110     
111     // refresh the define functions - force them to pass on their parameter
112     // definitions as field defs to their encapsulated expression. Parameter
113     // defs were not passed on by expressions encapsulated in DefineFunctions
114     // at construction time because the encapsulated expression does not know
115     // whether it is contained in a DefineFunction or a DerivedField. Since
116     // we delay passing on field definitions until all derived fields are 
117     // loaded (in order to allow derived fields to reference other derived fields),
118     // we must tell DefineFunctions to pass on their parameter definitions
119     for (int i = 0; i < m_defineFunctions.size(); i++) {
120       m_defineFunctions.get(i).pushParameterDefs();
121     }
122   }
123   
124   /**
125    * Set the field definitions for the derived fields. Usually called once the
126    * structure of the mining schema has been determined. Calling this method
127    * with an array list of field definitions in the order of attributes in the
128    * mining schema will allow the expressions used in the derived fields to
129    * access the correct attribute values from the incoming instances. This is
130    * necessary because construction of the TransformationDictionary uses the
131    * data dictionary to reference fields (the order of fields in the data dictionary
132    * is not guaranteed to be the same as the order in the mining schema).
133    * 
134    * @param fieldDefs the definition of the incoming fields as an Instances object
135    * @throws Exception if a problem occurs
136    */
137   protected void setFieldDefsForDerivedFields(Instances fieldDefs) throws Exception {
138     ArrayList<Attribute> tempDefs = new ArrayList<Attribute>();
139     for (int i = 0; i < fieldDefs.numAttributes(); i++) {
140       tempDefs.add(fieldDefs.attribute(i));
141     }
142     setFieldDefsForDerivedFields(tempDefs);
143   }
144   
145   protected ArrayList<DerivedFieldMetaInfo> getDerivedFields() {
146     return new ArrayList<DerivedFieldMetaInfo>(m_derivedFields);
147   }
148   
149   /**
150    * Get a named DefineFunction. Returns a deep copy of the
151    * function.
152    * 
153    * @param functionName the name of the function to get
154    * @return the named function or null if it cannot be found
155    * @throws Exception if there is a problem deep copying the function
156    */
157   protected DefineFunction getFunction(String functionName) throws Exception {
158 
159     DefineFunction copy = null;
160     DefineFunction match = null;
161     for (DefineFunction f : m_defineFunctions) {
162       if (f.getName().equals(functionName)) {
163         match = f;
164         //System.err.println("Found a match!!!");
165         break;
166       }
167     }
168     
169     if (match != null) {
170       SerializedObject so = new SerializedObject(match, false);
171       copy = (DefineFunction)so.getObject();
172       //System.err.println(copy);
173     }
174     
175     return copy;
176   }
177   
178   public String toString() {
179     StringBuffer buff = new StringBuffer();
180     
181     buff.append("Transformation dictionary:\n");
182     
183     if (m_derivedFields.size() > 0) {
184       buff.append("derived fields:\n");
185       for (DerivedFieldMetaInfo d : m_derivedFields) {
186         buff.append("" + d.getFieldAsAttribute() + "\n");
187       }
188     }
189     
190     if (m_defineFunctions.size() > 0) {
191       buff.append("\nfunctions:\n");
192       for (DefineFunction f : m_defineFunctions) {
193         buff.append(f.toString("  ") + "\n");
194       }
195     }
196     
197     buff.append("\n");
198     
199     return buff.toString();
200   }
201 }
