23  package weka.core.pmml;
24  
25  import java.io.Serializable;
26  import java.util.ArrayList;
27  import org.w3c.dom.Element;
28  import org.w3c.dom.Node;
29  import org.w3c.dom.NodeList;
30  
31  import weka.core.Attribute;
32  
33  public abstract class Expression  implements Serializable {
34    
35    /**
36     * For serialization
37     */
38    private static final long serialVersionUID = 4448840549804800321L;
39    
40    /** The optype of this Expression */
41    protected FieldMetaInfo.Optype m_opType;
42  
43    /** The field defs */
44    protected ArrayList<Attribute> m_fieldDefs = null;
45    
46    // NOTE - might need to pass in mining schema in order
47    // to determine values for nominal optypes
48    public Expression(FieldMetaInfo.Optype opType, ArrayList<Attribute> fieldDefs) {
49      m_opType = opType;
50      m_fieldDefs = fieldDefs;
51    }
52    
53    /**
54     * Set the field definitions for this Expression to use
55     * 
56     * @param fieldDefs the field definitions to use
57     * @throws Exception if there is a problem setting the field definitions
58     */
59    public void setFieldDefs(ArrayList<Attribute> fieldDefs) throws Exception {
60      m_fieldDefs = fieldDefs;
61    }
62      
63    /**
64     * Get the result of evaluating the expression. In the case
65     * of a continuous optype, a real number is returned; in
66     * the case of a categorical/ordinal optype, the index of the nominal
67     * value is returned as a double.
68     * 
69     * @param incoming the incoming parameter values
70     * @return the result of evaluating the expression
71     * @throws Exception if there is a problem computing the result
72     */
73    public abstract double getResult(double[] incoming) throws Exception;
74    
75    /**
76     * Get the result of evaluating the expression for continuous
77     * optype. Is the same as calling getResult() when the optype
78     * is continuous.
79     * 
80     * @param incoming the incoming parameter values
81     * mining schema
82     * @return the result of evaluating the expression.
83     * @throws Exception if the optype is not continuous.
84     */
85    public double getResultContinuous(double[] incoming) throws Exception {
86      if (!(m_opType == FieldMetaInfo.Optype.CONTINUOUS)) {
87        throw new Exception("[Expression] Can't return continuous result "
88            + "as optype is not continuous");
89      }
90      return getResult(incoming);
91    }
92    
93    /**
94     * Gets the result of evaluating the expression when the
95     * optype is categorical or ordinal as the actual String
96     * value.
97     * 
98     * @param incoming the incoming parameter values 
99     * @return the result of evaluating the expression
100    * @throws Exception if the optype is continuous
101    */
102   public abstract String getResultCategorical(double[] incoming) 
103     throws Exception;
104   
105   /**
106    * Return the structure of the result of applying this Expression
107    * as an Attribute.
108    * 
109    * @return the structure of the result of applying this Expression as an
110    * Attribute.
111    */
112   protected abstract Attribute getOutputDef();

114   /**
115    * Static factory method that returns a subclass of Expression that
116    * encapsulates the type of expression contained in the Element
117    * supplied. Assumes that there is just one expression contained
118    * in the supplied Element.
119    * 
120    * @param container the Node containing the expression
121    * @param opType the optype of the value returned by this Expression.
122    * @param fieldDefs an ArrayList of Attributes for the fields that this
123    * Expression may need to access
124    * Since Expressions are children of either DerivedFields or
125    * DefineFuntions, they will have the same optype as their parent.
126    * @param transDict the TransformationDictionary (may be null if there
127    * is no dictionary)
128    * @return an Expression object or null if there is no known expression in
129    * the container 
130    * @throws Exception for unsupported Expression types 
131    */
132   public static Expression getExpression(Node container, 
133       FieldMetaInfo.Optype opType,
134       ArrayList<Attribute> fieldDefs,
135       TransformationDictionary transDict) throws Exception {
136     
137     // we need to examine children of this Node to find an expression,
138     // not the entire subtree (as would be returned by Element.getElementsByTagName()
139     
140     Expression result = null;
141     String tagName = "";
142     
143     NodeList children = container.getChildNodes();
144     if (children.getLength() == 0) {
145       throw new Exception("[Expression] container has no children!");
146     }
147     
148     // at this level in the tree there should be only one expression type
149     // specified - look for it here.
150     for (int i = 0; i < children.getLength(); i++) {
151       Node child = children.item(i);
152       if (child.getNodeType() == Node.ELEMENT_NODE) {
153         tagName = ((Element)child).getTagName();
154         result = getExpression(tagName, child, opType, fieldDefs, transDict);
155         if (result != null) {
156           break;
157         }
158       }
159     }
160     
161     return result;
162   }
163   
164   /**
165    * Static factory method that returns a subclass of Expression that
166    * encapsulates the type of expression supplied as an argument.
167    * 
168    * @param name the name of the Expression to get
169    * @param expression the Node containing the expression
170    * @param opType the optype of the value returned by this Expression.
171    * @param fieldDefs an ArrayList of Attributes for the fields that this
172    * Expression may need to access
173    * Since Expressions are children of either DerivedFields or
174    * DefineFuntions, they will have the same optype as their parent.
175    * @param transDict the TransformationDictionary (may be null if there
176    * is no dictionary)
177    * @return an Expression object or null if there is no known expression in
178    * the container 
179    * @throws Exception for unsupported Expression types 
180    */
181   public static Expression getExpression(String name, 
182       Node expression,
183       FieldMetaInfo.Optype opType,
184       ArrayList<Attribute> fieldDefs,
185       TransformationDictionary transDict) throws Exception {
186    
187     Expression result = null;
188     
189     if (name.equals("Constant")) {
190       // construct a Constant expression
191       result = new Constant((Element)expression, opType, fieldDefs);
192     } else if (name.equals("FieldRef")) {
193       // construct a FieldRef expression
194       result = new FieldRef((Element)expression, opType, fieldDefs);
195     } else if (name.equals("Apply")) {
196       // construct an Apply expression
197       result = new Apply((Element)expression, opType, fieldDefs, transDict);
198     } else if (name.equals("NormDiscrete")) {
199       result = new NormDiscrete((Element)expression, opType, fieldDefs);
200     } else if (name.equals("NormContinuous")) {
201       result = new NormContinuous((Element)expression, opType, fieldDefs);
202     } else if (name.equals("Discretize")) {
203       result = new Discretize((Element)expression, opType, fieldDefs);
204     } else if (name.equals("MapValues") ||
205         name.equals("Aggregate")) {
206       throw new Exception("[Expression] Unhandled Expression type " + name);
207     }
208     return result;
209   }
210   
211   /**
212    * Return the named attribute from the list of reference fields.
213    * 
214    * @param attName the name of the attribute to retrieve
215    * @return the named attribute (or null if it can't be found).
216    */
217   public Attribute getFieldDef(String attName) {
218     Attribute returnV = null;
219     for (int i = 0; i < m_fieldDefs.size(); i++) {
220       if (m_fieldDefs.get(i).name().equals(attName)) {
221         returnV = m_fieldDefs.get(i);
222         break;
223       }
224     }
225     return returnV;
226   }
227   
228   public int getFieldDefIndex(String attName) {
229     int returnV = -1;
230     for (int i = 0; i < m_fieldDefs.size(); i++) {
231       if (m_fieldDefs.get(i).name().equals(attName)) {
232         returnV = i;
233         break;
234       }
235     }
236     return returnV;
237   }
238   
239   /**
240    * Get the optype of the result of applying this Expression.
241    * 
242    * @return the optype of the result of applying this Expression
243    */
244   public FieldMetaInfo.Optype getOptype() {
245     return m_opType;
246   }
247   
248   public String toString() {
249     return toString("");
250   }
251   
252   public String toString(String pad) {
253     return pad + this.getClass().getName();
254   }
255 }
