144   private static boolean isPMML(Document doc) {
145     NodeList tempL = doc.getElementsByTagName("PMML");
146     if (tempL.getLength() == 0) {
147       return false;
148     }
149     
150     return true;
151   }

153   /**
154    * Read and return a PMML model.
155    *
156    * @param stream the <code>InputStream</code> to read from
157    * @param log the logging object to use (or null if none is to be used)
158    * @return a PMML model
159    * @throws Exception if there is a problem while reading from the stream
160    */
161   public static PMMLModel getPMMLModel(InputStream stream, Logger log) throws Exception {
162     DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
163     DocumentBuilder db = dbf.newDocumentBuilder();
164     Document doc = db.parse(stream);
165     stream.close();
166     doc.getDocumentElement().normalize();
167     if (!isPMML(doc)) {
168       throw new IllegalArgumentException("[PMMLFactory] Source is not a PMML file!!");
169     }
170     
171     //    System.out.println("Root element " + doc.getDocumentElement().getNodeName());
172 
173     Instances dataDictionary = getDataDictionaryAsInstances(doc);
174     TransformationDictionary transDict = getTransformationDictionary(doc, dataDictionary);
175     
176     ModelType modelType = getModelType(doc);
177     if (modelType == ModelType.UNKNOWN_MODEL) {
178       throw new Exception("Unsupported PMML model type");
179     }
180     Element model = getModelElement(doc, modelType);
181 
182     // Construct mining schema and meta data
183     MiningSchema ms = new MiningSchema(model, dataDictionary, transDict);
184 
185     //System.out.println(ms);
186     //System.exit(1);
187     //    Instances miningSchema = getMiningSchemaAsInstances(model, dataDictionary);
188     PMMLModel theModel = getModelInstance(doc, modelType, model, dataDictionary, ms);
189     if (log != null) {
190       theModel.setLog(log);
191     }
192     return theModel;
193   }

195   /**
196    * Get the transformation dictionary (if there is one).
197    * 
198    * @param doc the Document containing the PMML model
199    * @param dataDictionary the data dictionary as an Instances object
200    * @return the transformation dictionary or null if there is none defined in
201    * the Document
202    * @throws Exception if there is a problem getting the transformation
203    * dictionary
204    */
205   protected static TransformationDictionary getTransformationDictionary(Document doc, 
206       Instances dataDictionary) throws Exception {
207     TransformationDictionary transDict = null;
208     
209     NodeList transL = doc.getElementsByTagName("TransformationDictionary");
210     // should be of size 0 or 1
211     if (transL.getLength() > 0) {
212       Node transNode = transL.item(0);
213       if (transNode.getNodeType() == Node.ELEMENT_NODE) {
214         transDict = new TransformationDictionary((Element)transNode, dataDictionary);
215       }
216     }
217     
218     return transDict;
219   }

262   /**
263    * Get an instance of a PMMLModel from the supplied Document
264    *
265    * @param doc the Document holding the pmml
266    * @param modelType the type of model
267    * @param model the Element encapsulating the model part of the Document
268    * @param dataDictionary the data dictionary as an Instances object
269    * @param miningSchema the mining schema
270    * @return a PMMLModel object
271    * @throws Exception if there is a problem constructing the model or
272    * if the model type is not supported
273    */
274   protected static PMMLModel getModelInstance(Document doc, 
275                                               ModelType modelType, Element model,
276                                               Instances dataDictionary,
277                                               MiningSchema miningSchema) throws Exception {
278     PMMLModel pmmlM = null;
279     switch (modelType) {
280     case REGRESSION_MODEL:
281       pmmlM = new Regression(model, dataDictionary, miningSchema);
282       //System.out.println(pmmlM);
283       break;
284     case GENERAL_REGRESSION_MODEL:
285       pmmlM = new GeneralRegression(model, dataDictionary, miningSchema);
286       //System.out.println(pmmlM);
287       break;
288     case NEURAL_NETWORK_MODEL:
289       pmmlM = new NeuralNetwork(model, dataDictionary, miningSchema);
290       break;
291     case TREE_MODEL:
292       pmmlM = new TreeModel(model, dataDictionary, miningSchema);
293       break;
294     case RULESET_MODEL:
295       pmmlM = new RuleSetModel(model, dataDictionary, miningSchema);
296       break;
297     default:
298       throw new Exception("[PMMLFactory] Unknown model type!!");
299     }
300     pmmlM.setPMMLVersion(doc);
301     pmmlM.setCreatorApplication(doc);
302     return pmmlM;
303   }

305   /**
306    * Get the type of model
307    *
308    * @param doc the Document encapsulating the pmml
309    * @return the type of model
310    */
311   protected static ModelType getModelType(Document doc) {
312     NodeList temp = doc.getElementsByTagName("RegressionModel");
313     if (temp.getLength() > 0) {
314       return ModelType.REGRESSION_MODEL;
315     }
316 
317     temp = doc.getElementsByTagName("GeneralRegressionModel");
318     if (temp.getLength() > 0) {
319       return ModelType.GENERAL_REGRESSION_MODEL;
320     }
321     
322     temp = doc.getElementsByTagName("NeuralNetwork");
323     if (temp.getLength() > 0) {
324       return ModelType.NEURAL_NETWORK_MODEL;
325     }
326     
327     temp = doc.getElementsByTagName("TreeModel");
328     if (temp.getLength() > 0) {
329       return ModelType.TREE_MODEL;
330     }
331     
332     temp = doc.getElementsByTagName("RuleSetModel");
333     if (temp.getLength() > 0) {
334       return ModelType.RULESET_MODEL;
335     }
336 
337     return ModelType.UNKNOWN_MODEL;
338   }

340   /**
341    * Get the Element that contains the pmml model
342    *
343    * @param doc the Document encapsulating the pmml
344    * @param modelType the type of model
345    * @throws Exception if the model type is unsupported/unknown
346    */
347   protected static Element getModelElement(Document doc, ModelType modelType) 
348     throws Exception {
349     NodeList temp = null;
350     Element model = null;
351     switch (modelType) {
352     case REGRESSION_MODEL:
353       temp = doc.getElementsByTagName("RegressionModel");
354       break;
355     case GENERAL_REGRESSION_MODEL:
356       temp = doc.getElementsByTagName("GeneralRegressionModel");
357       break;
358     case NEURAL_NETWORK_MODEL:
359       temp = doc.getElementsByTagName("NeuralNetwork");
360       break;
361     case TREE_MODEL:
362       temp = doc.getElementsByTagName("TreeModel");
363       break;
364     case RULESET_MODEL:
365       temp = doc.getElementsByTagName("RuleSetModel");
366       break;
367     default:
368       throw new Exception("[PMMLFactory] unknown/unsupported model type.");
369     }
370 
371     if (temp != null && temp.getLength() > 0) {
372       Node modelNode = temp.item(0);
373       if (modelNode.getNodeType() == Node.ELEMENT_NODE) {
374         model = (Element)modelNode;
375       }
376     }
377 
378     return model;
379   }

381   /**
382    * Get the mining schema as an Instances object
383    *
384    * @param model the Element containing the pmml model
385    * @param dataDictionary the data dictionary as an Instances object
386    * @return the mining schema as an Instances object
387    * @throws Exception if something goes wrong during reading the mining schema
388    * @deprecated Use the MiningSchema class instead
389    */
390   protected static Instances getMiningSchemaAsInstances(Element model,
391                                                         Instances dataDictionary) 
392     throws Exception {
393     ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
394     NodeList fieldList = model.getElementsByTagName("MiningField");
395     int classIndex = -1;
396     int addedCount = 0;
397     for (int i = 0; i < fieldList.getLength(); i++) {
398       Node miningField = fieldList.item(i);
399       if (miningField.getNodeType() == Node.ELEMENT_NODE) {
400         Element miningFieldEl = (Element)miningField;
401         String name = miningFieldEl.getAttribute("name");
402         String usage = miningFieldEl.getAttribute("usageType");
403         // TO-DO: also missing value replacement etc.
404 
405         // find this attribute in the dataDictionary
406         Attribute miningAtt = dataDictionary.attribute(name);
407         if (miningAtt != null) {
408           if (usage.length() == 0 || usage.equals("active") || usage.equals("predicted")) {
409             attInfo.add(miningAtt);
410             addedCount++;
411           }
412           if (usage.equals("predicted")) {
413             classIndex = addedCount - 1;
414           }
415         } else {
416           throw new Exception("Can't find mining field: " + name 
417                               + " in the data dictionary.");
418         }
419       }
420     }
421     
422     Instances insts = new Instances("miningSchema", attInfo, 0);
423     //    System.out.println(insts);
424     if (classIndex != -1) {
425       insts.setClassIndex(classIndex);
426     }
427 
428 
429     return insts;
430   }

432   /**
433    * Get the data dictionary as an Instances object
434    *
435    * @param doc the Document encapsulating the pmml
436    * @return the data dictionary as an Instances object
437    * @throws Exception if there are fields that are not continuous, 
438    * ordinal or categorical in the data dictionary
439    */
440   protected static Instances getDataDictionaryAsInstances(Document doc) 
441     throws Exception {
442 
443     // TO-DO: definition of missing values (see below)
444 
445     ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
446     NodeList dataDictionary = doc.getElementsByTagName("DataField");
447     for (int i = 0; i < dataDictionary.getLength(); i++) {
448       Node dataField = dataDictionary.item(i);
449       if (dataField.getNodeType() == Node.ELEMENT_NODE) {
450         Element dataFieldEl = (Element)dataField;
451         String name = dataFieldEl.getAttribute("name");
452         String type = dataFieldEl.getAttribute("optype");
453         Attribute tempAtt = null;
454         if (name != null && type != null) {
455           if (type.equals("continuous")) {
456             tempAtt = new Attribute(name);
457           } else if (type.equals("categorical") || type.equals("ordinal")) {
458             NodeList valueList = dataFieldEl.getElementsByTagName("Value");
459             if (valueList == null || valueList.getLength() == 0) {
460               // assume that categorical values will be revealed in the actual model.
461               // Create a string attribute for now
462               ArrayList<String> nullV = null;
463               tempAtt = new Attribute(name, (ArrayList<String>)nullV);
464             } else {
465               // add the values (if defined as "valid")
466               ArrayList<String> valueVector = new ArrayList<String>();
467               for (int j = 0; j < valueList.getLength(); j++) {
468                 Node val = valueList.item(j);
469                 if (val.getNodeType() == Node.ELEMENT_NODE) {
470                   // property is optional (default value is "valid")
471                   String property = ((Element)val).getAttribute("property");
472                   if (property == null || property.length() == 0 || property.equals("valid")) {
473                     String value = ((Element)val).getAttribute("value");
474                     valueVector.add(value);
475                   } else {
476                     // Just ignore invalid or missing value definitions for now...
477                     // TO-DO: implement Value meta data with missing/invalid value defs.
478                   }
479                 }
480               }
481               tempAtt = new Attribute(name, valueVector);
482             }
483           } else {
484             throw new Exception("[PMMLFactory] can't handle " + type + "attributes.");
485           }
486           attInfo.add(tempAtt);
487         }
488       }
489     }
490 
491     // TO-DO: check whether certain values are declared to represent
492     // missing or invalid values (applies to both categorical and continuous
493     // attributes
494 
495     // create the Instances structure
496     Instances insts = new Instances("dataDictionary", attInfo, 0);
497     //    System.out.println(insts);
498 
499     return insts;
500   }

502   public static String applyClassifier(PMMLModel model, Instances test) throws Exception {
503     StringBuffer buff = new StringBuffer();
504     if (!(model instanceof PMMLClassifier)) {
505       throw new Exception("PMML model is not a classifier!");
506     }
507 
508     double[] preds = null;
509     PMMLClassifier classifier = (PMMLClassifier)model;
510     for (int i = 0; i < test.numInstances(); i++) {
511       buff.append("Actual: ");
512       Instance temp = test.instance(i);
513       if (temp.classAttribute().isNumeric()) {
514         buff.append(temp.value(temp.classIndex()) + " ");
515       } else {
516         buff.append(temp.classAttribute().value((int)temp.value(temp.classIndex())) + " ");
517       }
518       preds = classifier.distributionForInstance(temp);
519       buff.append(" Predicted: ");
520       for (int j = 0; j < preds.length; j++) {
521         buff.append("" + preds[j] + " ");
522       }
523       buff.append("\n");
524     }
525     return buff.toString();
526   }
