30  /**
31   * Abstract superclass for various types of field meta
32   * data.
33   * 
34   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com
35   * @version $Revision 1.0 $
36   */
37  public abstract class FieldMetaInfo implements Serializable {
38  
39    /**
40     * Inner class for Values
41     */
42    public static class Value implements Serializable {
43      
44      /**
45       * For serialization
46       */
47      private static final long serialVersionUID = -3981030320273649739L;
48  
49      /** The value */
50      protected String m_value;
51      
52      /** 
53       * The display value (might hold a human readable value - e.g.
54       * product name instead of cryptic code).
55       */
56      protected String m_displayValue;
57      
58      /**
59       * Enumerated type for the property. A value
60       * can be valid, invalid or indicate a value
61       * that should be considered as "missing".
62       */
63      public enum Property {
64        VALID ("valid"),
65        INVALID ("invalid"),
66        MISSING ("missing");
67        
68        private final String m_stringVal;
69        
70        Property(String name) {
71          m_stringVal = name;
72        }
73        
74        public String toString() {
75          return m_stringVal;
76        }
77      }
78      protected Property m_property = Property.VALID;
79      
80      /**
81       * Construct a value.
82       * 
83       * @param value the Element containing the value
84       * @throws Exception if there is a problem constucting the value
85       */
86      protected Value(Element value) throws Exception {
87        m_value = value.getAttribute("value");
88        String displayV = value.getAttribute("displayValue");
89        if (displayV != null && displayV.length() > 0) {
90          m_displayValue = displayV;
91        }
92        String property = value.getAttribute("property");
93        if (property != null && property.length() > 0) {
94          for (Property p: Property.values()) {
95            if (p.toString().equals(property)) {
96              m_property = p;
97              break;
98            }
99          }
100       }
101     }
102     
103     public String toString() {
104       String retV = m_value;
105       if (m_displayValue != null) {
106         retV += "(" + m_displayValue + "): " + m_property.toString();
107       }
108       return retV;
109     }
110     
111     public String getValue() {
112       return m_value;
113     }
114     
115     public String getDisplayValue() {
116       return m_displayValue;
117     }
118     
119     public Property getProperty() {
120       return m_property;
121     }
122   }
123   
124   /**
125    * Inner class for an Interval.
126    */
127   public static class Interval implements Serializable {
128     
129     /**
130      * For serialization
131      */
132     private static final long serialVersionUID = -7339790632684638012L;
133 
134     /** The left boundary value */
135     protected double m_leftMargin = Double.NEGATIVE_INFINITY;
136     
137     /** The right boundary value */
138     protected double m_rightMargin = Double.POSITIVE_INFINITY;
139     
140     /**
141      * Enumerated type for the closure.
142      */
143     public enum Closure {
144       OPENCLOSED ("openClosed", "(", "]"),
145       OPENOPEN ("openOpen", "(", ")"),
146       CLOSEDOPEN ("closedOpen", "[", ")"),
147       CLOSEDCLOSED ("closedClosed", "[", "]");
148       
149       private final String m_stringVal;
150       private final String m_left;
151       private final String m_right;
152       
153       Closure(String name, String left, String right) {
154         m_stringVal = name;
155         m_left = left;
156         m_right = right;
157       }
158       
159       public String toString() {
160         return m_stringVal;
161       }
162       
163       public String toString(double leftMargin, double rightMargin) {
164         return m_left + leftMargin + "-" + rightMargin + m_right;
165       }
166     }
167     protected Closure m_closure = Closure.OPENOPEN;
168     
169     /**
170      * Construct an interval.
171      * 
172      * @param interval the Element containing the interval
173      * @throws Exception if there is a problem constructing the interval
174      */
175     protected Interval(Element interval) throws Exception {
176       String leftM = interval.getAttribute("leftMargin");
177       try {
178         m_leftMargin = Double.parseDouble(leftM);
179       } catch (IllegalArgumentException ex) {
180         throw new Exception("[Interval] Can't parse left margin as a number");
181       }
182       
183       String rightM = interval.getAttribute("rightMargin");
184       try {
185         m_rightMargin = Double.parseDouble(rightM);
186       } catch (IllegalArgumentException ex) {
187         throw new Exception("[Interval] Can't parse right margin as a number");
188       }
189       
190       String closure = interval.getAttribute("closure");
191       if (closure == null || closure.length() == 0) {
192         throw new Exception("[Interval] No closure specified!");
193       }
194       for (Closure c : Closure.values()) {
195         if (c.toString().equals(closure)) {
196           m_closure = c;
197           break;
198         }
199       }
200     }
201     
202     /**
203      * Returns true if this interval contains the supplied value.
204      * 
205      * @param value the value to check
206      * @return true if the interval contains the supplied value
207      */
208     public boolean containsValue(double value) {
209       boolean result = false;
210       
211       switch (m_closure) {
212       case OPENCLOSED:
213         if (value > m_leftMargin && value <= m_rightMargin) {
214           result = true;
215         }
216         break;
217       case OPENOPEN:
218         if (value > m_leftMargin && value < m_rightMargin) {
219           result = true;
220         }
221         break;
222       case CLOSEDOPEN:
223         if (value >= m_leftMargin && value < m_rightMargin) {
224           result = true;
225         }
226         break;
227       case CLOSEDCLOSED:
228         if (value >= m_leftMargin && value <= m_rightMargin) {
229           result = true;
230         }
231         break;
232       default:
233         result = false;
234         break;
235       }
236         
237       return result;
238     }
239     
240     public String toString() {
241       return m_closure.toString(m_leftMargin, m_rightMargin);
242     }
243   }

248   /** the name of the field */
249   protected String m_fieldName;
250 
251   /**
252    * Enumerated type for the Optype
253    */
254   public enum Optype {
255     NONE ("none"), 
256     CONTINUOUS ("continuous"), 
257     CATEGORICAL ("categorical"),
258     ORDINAL ("ordinal");
259   
260     private final String m_stringVal;
261     
262     Optype(String name) {
263       m_stringVal = name;
264     }
265   
266     public String toString() {
267       return m_stringVal;
268     }
269   }
270 
271   /** The optype for the target */
272   protected Optype m_optype = Optype.NONE;
273   
274   /**
275    * Get the optype.
276    * 
277    * @return the optype
278    */
279   public Optype getOptype() {
280     return m_optype;
281   }
282   
283   /**
284    * Get the name of this field.
285    * 
286    * @return the name of this field
287    */
288   public String getFieldName() {
289     return m_fieldName;
290   }
291   
292   /**
293    * Construct a new FieldMetaInfo.
294    * 
295    * @param field the Element containing the field
296    */
297   public FieldMetaInfo(Element field) {
298     m_fieldName = field.getAttribute("name");
299     
300     String opType = field.getAttribute("optype");
301     if (opType != null && opType.length() > 0) {
302       for (Optype o : Optype.values()) {
303         if (o.toString().equals(opType)) {
304           m_optype = o;
305           break;
306         }
307       }
308     }
309   }
310   
311   /**
312    * Return this field as an Attribute.
313    * 
314    * @return an Attribute for this field.
315    */
316   public abstract Attribute getFieldAsAttribute();
317 }
