22  package weka.core.pmml;
23  
24  import java.util.ArrayList;
25  
26  import weka.core.Attribute;
27  import weka.core.Utils;
28  
29  /**
30   * Built-in function for min, max, sum, avg, log10,
31   * ln, sqrt, abs, exp, pow, threshold, floor, ceil and round.
32   * 
33   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
34   * @version $Revision 1.0 $
35   */
36  public class BuiltInMath extends Function {
37    
38    /**
39     * For serialization
40     */
41    private static final long serialVersionUID = -8092338695602573652L;
42  
43    /**
44     * Enum for the math functions.
45     */
46    enum MathFunc {
47      MIN ("min") {
48        double eval(double[] args) {
49          return args[Utils.minIndex(args)];
50        }
51        
52        boolean legalNumParams(int num) {
53          return (num > 0);
54        }
55        
56        String[] getParameterNames() {
57          return null; // unbounded number of parameters
58        }
59      },
60      MAX ("max") {
61        double eval(double[] args) {
62          return args[Utils.maxIndex(args)];
63        }
64        
65        boolean legalNumParams(int num) {
66          return (num > 0);
67        }
68        
69        String[] getParameterNames() {
70          return null; // unbounded number of parameters
71        }
72      },
73      SUM ("sum") {
74        double eval(double[] args) {
75          return Utils.sum(args);
76        }
77        
78        boolean legalNumParams(int num) {
79          return (num > 0);
80        }
81        
82        String[] getParameterNames() {
83          return null; // unbounded number of parameters
84        }
85      },
86      AVG ("avg") {
87        double eval(double[] args) {
88          return Utils.mean(args);
89        }
90        
91        boolean legalNumParams(int num) {
92          return (num > 0);
93        }
94        
95        String[] getParameterNames() {
96          return null; // unbounded number of parameters
97        }
98      },
99      LOG10 ("log10") {
100       double eval(double[] args) {
101         return Math.log10(args[0]);
102       }
103       
104       boolean legalNumParams(int num) {
105         return (num == 1);
106       }
107       
108       String[] getParameterNames() {
109         return new String[] {"A"};
110       }
111     },
112     LN ("ln") {
113       double eval(double[] args) {
114         return Math.log(args[0]);
115       }
116       
117       boolean legalNumParams(int num) {
118         return (num == 1);
119       }
120       
121       String[] getParameterNames() {
122         return new String[] {"A"};
123       }
124     },
125     SQRT ("sqrt") {
126       double eval(double[] args) {
127         return Math.sqrt(args[0]);
128       }
129       
130       boolean legalNumParams(int num) {
131         return (num == 1);
132       }
133       
134       String[] getParameterNames() {
135         return new String[] {"A"};
136       }
137     },
138     ABS ("abs") {
139       double eval(double[] args) {
140         return Math.abs(args[0]);
141       }
142       
143       boolean legalNumParams(int num) {
144         return (num == 1);
145       }
146       
147       String[] getParameterNames() {
148         return new String[] {"A"};
149       }
150     },
151     EXP ("exp") {
152       double eval(double[] args) {
153         return Math.exp(args[0]);
154       }
155       
156       boolean legalNumParams(int num) {
157         return (num == 1);
158       }
159       
160       String[] getParameterNames() {
161         return new String[] {"A"};
162       }
163     },
164     POW ("pow") {
165       double eval(double[] args) {
166         return Math.pow(args[0], args[1]);
167       }
168       
169       boolean legalNumParams(int num) {
170         return (num == 2);
171       }
172       
173       String[] getParameterNames() {
174         return new String[] {"A", "B"};
175       }
176     },
177     THRESHOLD ("threshold") {
178       double eval(double[] args) {
179         if (args[0] > args[1]) {
180           return 1.0;
181         } else {
182           return 0.0;
183         }
184       }
185       
186       boolean legalNumParams(int num) {
187         return (num == 2);
188       }
189       
190       String[] getParameterNames() {
191         return new String[] {"A", "B"};
192       }
193     },
194     FLOOR ("floor") {
195       double eval(double[] args) {
196         return Math.floor(args[0]);
197       }
198       
199       boolean legalNumParams(int num) {
200         return (num == 1);
201       }
202       
203       String[] getParameterNames() {
204         return new String[] {"A"};
205       }
206     },
207     CEIL ("ceil") {
208       double eval(double[] args) {
209         return Math.ceil(args[0]);
210       }
211       
212       boolean legalNumParams(int num) {
213         return (num == 1);
214       }
215       
216       String[] getParameterNames() {
217         return new String[] {"A"};
218       }
219     },
220     ROUND ("round") {
221       double eval(double[] args) {
222         return Math.round(args[0]);
223       }
224       
225       boolean legalNumParams(int num) {
226         return (num == 1);
227       }
228       
229       String[] getParameterNames() {
230         return new String[] {"A"};
231       }
232     };
233     
234     abstract double eval(double[] args);
235     abstract boolean legalNumParams(int num);
236     abstract String[] getParameterNames();
237     
238     private final String m_stringVal;
239     
240     MathFunc(String funcName) {
241       m_stringVal = funcName;
242     }
243     
244     public String toString() {
245       return m_stringVal;
246     }
247   }
248   
249   /** The function to apply */
250   protected MathFunc m_func = MathFunc.ABS;
251   
252   /**
253    * Construct a new built-in pmml Math function.
254    * @param func the math function to use
255    */
256   public BuiltInMath(MathFunc func) {
257     m_func = func;
258     m_functionName = m_func.toString();
259   }
260   
261   /**
262    * Set the structure of the parameters that are expected as input by
263    * this function. This must be called before getOutputDef() is called.
264    * 
265    * @param paramDefs the structure of the input parameters
266    * @throws Exception if the number or types of parameters are not acceptable by
267    * this function
268    */
269   public void setParameterDefs(ArrayList<Attribute> paramDefs) throws Exception {
270     m_parameterDefs = paramDefs;
271     
272     if (!m_func.legalNumParams(m_parameterDefs.size())) {
273       throw new Exception("[BuiltInMath] illegal number of parameters for function: " 
274           + m_functionName);
275     }
276   }
277 
278   /**
279    * Get the structure of the result produced by this function.
280    * Subclasses must implement.
281    * 
282    * @return the structure of the result produced by this function.
283    */
284   public Attribute getOutputDef() {
285     return new Attribute("BuiltInMathResult:" + m_func.toString());
286   }
287 
288   /**
289    * Returns an array of the names of the parameters expected
290    * as input by this function. May return null if the function
291    * can accept an unbounded number of arguments.
292    * 
293    * @return an array of the parameter names (or null if the function
294    * can accept any number of arguments).
295    */
296   public String[] getParameterNames() {
297     return m_func.getParameterNames();
298   }
299 
300   /**
301    * Get the result of applying this function.
302    * 
303    * @param incoming the arguments to this function (supplied in order to match that
304    * of the parameter definitions
305    * @return the result of applying this function. When the optype is
306    * categorical or ordinal, an index into the values of the output definition
307    * is returned.
308    * @throws Exception if there is a problem computing the result of this function
309    */
310   public double getResult(double[] incoming) throws Exception {
311     if (m_parameterDefs == null) {
312       throw new Exception("[BuiltInMath] incoming parameter structure has not been set");
313     }
314     
315     if (!m_func.legalNumParams(incoming.length)) {
316       throw new Exception("[BuiltInMath] wrong number of parameters!");
317     }
318     
319     double result = m_func.eval(incoming);
320     
321     return result;
322   }
323   
324   public String toString() {
325     String result = m_func.toString() + "(";
326     for (int i = 0; i < m_parameterDefs.size(); i++) {
327       result += m_parameterDefs.get(i).name();
328       if (i != m_parameterDefs.size() - 1) {
329         result += ", ";
330       } else {
331         result += ")";
332       }
333     }
334     return result;
335   }
336 }
