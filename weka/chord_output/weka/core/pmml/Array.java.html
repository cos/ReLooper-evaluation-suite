40    /**
41     * Utility method to check if an XML element is an array.
42     * 
43     * @param arrayE the XML element to check
44     * @return returns true if the XML element is an array
45     */
46    public static boolean isArray(Element arrayE) {
47      String name = arrayE.getTagName();
48  
49      if (name.equals("Array") || name.equals("NUM-ARRAY") || name.equals("INT-ARRAY")
50          || name.equals("REAL-ARRAY") || name.equals("STRING-ARRAY") 
51          || isSparseArray(arrayE)) {
52        return true;
53      }
54      return false;
55    }
56    
57    /**
58     * Utility method to check if an XML element is a sparse array.
59     * 
60     * @param arrayE the XML element to check.
61     * @return true if the XML element is a sparse array.
62     */
63    private static boolean isSparseArray(Element arrayE) {
64      String name = arrayE.getTagName();
65      
66      if (name.equals("INT-SparseArray") || name.equals("REAL-SparseArray")) {
67        return true;
68      }
69      
70      return false;
71    }

73    /**
74     * Static factory method for creating non-sparse or sparse
75     * array types as needed.
76     * 
77     * @param arrayE the XML element encapsulating the array
78     * @return an appropriate Array type
79     * @throws Exception if there is a problem when constructing the array
80     */
81    public static Array create(Element arrayE) throws Exception {
82      if (!isArray(arrayE)) {
83        throw new Exception("[Array] the supplied element does not contain an array!");
84      }
85      
86      if (isSparseArray(arrayE)) {
87        // TODO: implement sparse array subclass :-)
88      }
89       
90      return new Array(arrayE);
91    }

93    public static enum ArrayType {
94      NUM("NUM-ARRAY"),
95      INT("INT-ARRAY"),
96      REAL("REAL-ARRAY"),
97      STRING("STRING-ARRAY");
98      
99      private final String m_stringVal;
100     
101     ArrayType(String name) {
102       m_stringVal = name;
103     }
104     
105     public String toString() {
106       return m_stringVal;
107     }
108   }

110   /** The values of the array */
111   protected ArrayList<String> m_values = new ArrayList<String>();
112   
113   /** The type of the array */
114   protected ArrayType m_type = ArrayType.NUM;
115   
116   protected Array(Element arrayE) throws Exception {
117     String arrayS = arrayE.getTagName();
118     
119     // get the type of the array
120     if (arrayS.equals("Array")) {
121       String type = arrayE.getAttribute("type"); 
122       if (type.equals("int")) {
123         m_type = ArrayType.INT;
124       } else if (type.equals("real")) {
125         m_type = ArrayType.REAL;
126       } else if (type.equals("string")) {
127         m_type = ArrayType.STRING;
128       }
129     } else {
130       for (ArrayType a : ArrayType.values()) {
131         if (a.toString().equals(arrayS)) {
132           m_type = a;
133           break;
134         }
135       }
136     }
137     // now read the values
138     String contents = arrayE.getChildNodes().item(0).getNodeValue();
139     StringReader sr = new StringReader(contents);
140     StreamTokenizer st = new StreamTokenizer(sr);
141     st.resetSyntax();
142     st.whitespaceChars(0, ' ');
143     st.wordChars(' '\u00FF1,'ÿ');
144     st.whitespaceChars(' ',' ');
145     st.quoteChar('"');
146     st.quoteChar('\'');
147     //m_Tokenizer.eolIsSignificant(true);
148     
149     st.nextToken();
150     while (st.ttype != StreamTokenizer.TT_EOF && 
151         st.ttype != StreamTokenizer.TT_EOL) {
152       m_values.add(st.sval);
153       st.nextToken();
154     }
155   }

224   private void checkInRange(int index) throws Exception {
225     if (index >= m_values.size() || index < 0) {
226       throw new IllegalArgumentException("[Array] index out of range " + index);
227     }
228   }
229   
230   /**
231    * Gets the value at index from the array.
232    * 
233    * @param index the index of the value to get.
234    * @return the value at index in the array as as String.
235    * @throws Exception if index is out of bounds.
236    */
237   public String value(int index) throws Exception {
238     checkInRange(index);
239     
240     return m_values.get(index);
241   }

357   public String toString() {
358     StringBuffer text = new StringBuffer();
359     
360     text.append("[");
361     for (int i = 0; i < m_values.size(); i++) {
362       text.append(m_values.get(i));
363       if (i < m_values.size() - 1) {
364         text.append(",");
365       }
366     }
367     
368     text.append("]");
369     return text.toString();
370   }
