23  package weka.core.pmml;
24  
25  import java.util.ArrayList;
26  
27  import org.w3c.dom.Element;
28  import org.w3c.dom.Node;
29  import org.w3c.dom.NodeList;
30  
31  import weka.core.Attribute;
32  
33  /**
34   * Class encapsulating DefineFunction (used in TransformationDictionary).
35   * 
36   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com
37   * @version $Revision 1.0 $
38   */
39  public class DefineFunction extends Function {
40    
41    /**
42     * For serialization
43     */
44    private static final long serialVersionUID = -1976646917527243888L;
45  
46    /**
47     * Inner class for handling Parameters
48     */
49    protected class ParameterField extends FieldMetaInfo {
50      /**
51       * For serialization
52       */
53      private static final long serialVersionUID = 3918895902507585558L;
54  
55      protected ParameterField(Element field) {
56        super(field);
57      }
58      
59      public Attribute getFieldAsAttribute() {
60        if (m_optype == Optype.CONTINUOUS) {
61          return new Attribute(m_fieldName);
62        }
63        // return a string attribute for categorical/ordinal optypes
64        return new Attribute(m_fieldName, (ArrayList<String>)null);
65      }
66    }
67    
68    /** 
69     * The list of parameters expected by this function. We can use this to do 
70     * some error/type checking when users call setParameterDefs() on us 
71     */
72    protected ArrayList<ParameterField> m_parameters = new ArrayList<ParameterField>();
73    
74    /** The optype for this function */
75    FieldMetaInfo.Optype m_optype = FieldMetaInfo.Optype.NONE;
76    
77    /** The Expression for this function to use */
78    protected Expression m_expression = null;
79    
80    public DefineFunction(Element container, TransformationDictionary transDict) throws Exception {
81      
82      m_functionName = container.getAttribute("name");
83      
84      // get the optype for this function
85      String opType = container.getAttribute("optype");
86      if (opType != null && opType.length() > 0) {
87        for (FieldMetaInfo.Optype o : FieldMetaInfo.Optype.values()) {
88          if (o.toString().equals(opType)) {
89            m_optype = o;
90            break;
91          }
92        }
93      } else {
94        throw new Exception("[DefineFunction] no optype specified!!");
95      }
96      
97      m_parameterDefs = new ArrayList<Attribute>();
98      
99      // get all the parameters
100     NodeList paramL = container.getElementsByTagName("ParameterField");
101     for (int i = 0; i < paramL.getLength(); i++) {
102       Node paramN = paramL.item(i);
103       if (paramN.getNodeType() == Node.ELEMENT_NODE) {
104         ParameterField newP = new ParameterField((Element)paramN);
105         m_parameters.add(newP);
106         
107         // set up default parameter definitions - these will probably get replaced
108         // by more informative ones (i.e. possibly nominal attributes instead of
109         // string attributes) later
110         m_parameterDefs.add(newP.getFieldAsAttribute());
111       }
112     }
113     
114     m_expression = Expression.getExpression(container, m_optype, m_parameterDefs, transDict);
115     
116     // check that the optype of the Expression is compatible with ours
117     if (m_optype == FieldMetaInfo.Optype.CONTINUOUS && 
118         m_expression.getOptype() != m_optype) {
119       throw new Exception("[DefineFunction] optype is continuous but our Expression's optype "
120           + "is not.");
121     }
122     
123     if ((m_optype == FieldMetaInfo.Optype.CATEGORICAL || m_optype == FieldMetaInfo.Optype.ORDINAL) !=
124         (m_expression.getOptype() == FieldMetaInfo.Optype.CATEGORICAL || 
125          m_expression.getOptype() == FieldMetaInfo.Optype.ORDINAL)) {
126       throw new Exception("[DefineFunction] optype is categorical/ordinal but our Expression's optype "
127           + "is not."); 
128     }
129   }
130   
131   public void pushParameterDefs() throws Exception {
132     if (m_parameterDefs == null) {
133       throw new Exception("[DefineFunction] parameter definitions are null! Can't "
134           + "push them to encapsulated expression.");
135     }
136     
137     m_expression.setFieldDefs(m_parameterDefs);
138   }
139 
140   /**
141    * Get the structure of the result produced by this function.
142    * 
143    * @return the structure of the result produced by this function.
144    */
145   public Attribute getOutputDef() {
146     return m_expression.getOutputDef();
147   }
148 
149   /**
150    * Returns an array of the names of the parameters expected
151    * as input by this function. May return null if this function
152    * can take an unbounded number of parameters (i.e. min, max, etc.).
153    * 
154    * @return an array of the parameter names or null if there are an
155    * unbounded number of parameters.
156    */
157   public String[] getParameterNames() {
158     String[] result = new String[m_parameters.size()];
159     for (int i = 0; i < m_parameters.size(); i++) {
160       result[i] = m_parameters.get(i).getFieldName();
161     }
162     
163     return result;
164   }
165 
166   /**
167    * Get the result of applying this function.
168    * 
169    * @param incoming the arguments to this function (supplied in order to match that
170    * of the parameter definitions
171    * @return the result of applying this function. When the optype is
172    * categorical or ordinal, an index into the values of the output definition
173    * is returned.
174    * @throws Exception if there is a problem computing the result of this function
175    */
176   public double getResult(double[] incoming) throws Exception {
177     
178     if (incoming.length != m_parameters.size()) {
179       throw new IllegalArgumentException("[DefineFunction] wrong number of arguments: expected "
180           + m_parameters.size() + ", recieved " + incoming.length);
181     }
182 
183     return m_expression.getResult(incoming);
184   }
185 
186   /**
187    * Set the structure of the parameters that are expected as input by
188    * this function. This must be called before getOutputDef() is called.
189    * 
190    * @param paramDefs the structure of the input parameters
191    * @throws Exception if the number or types of parameters are not acceptable by
192    * this function
193    */
194   public void setParameterDefs(ArrayList<Attribute> paramDefs) throws Exception {
195     if (paramDefs.size() != m_parameters.size()) {
196       throw new Exception("[DefineFunction] number of parameter definitions does not match "
197           + "number of parameters!");
198     }
199     
200     // check these defs against the optypes of the parameters
201     for (int i = 0; i < m_parameters.size(); i++) {
202       if (m_parameters.get(i).getOptype() == FieldMetaInfo.Optype.CONTINUOUS) {
203         if (!paramDefs.get(i).isNumeric()) {
204           throw new Exception("[DefineFunction] parameter "
205               + m_parameters.get(i).getFieldName() + " is continuous, but corresponding "
206               + "supplied parameter def " + paramDefs.get(i).name() + " is not!");
207         }
208       } else {
209         if (!paramDefs.get(i).isNominal() && !paramDefs.get(i).isString()) {
210           throw new Exception("[DefineFunction] parameter "
211               + m_parameters.get(i).getFieldName() + " is categorical/ordinal, but corresponding "
212               + "supplied parameter def " + paramDefs.get(i).name() + " is not!");
213         }
214       }
215     }
216     
217     // now we need to rename these argument definitions to match the names of
218     // the actual parameters
219     ArrayList<Attribute> newParamDefs = new ArrayList<Attribute>();
220     for (int i = 0; i < paramDefs.size(); i++) {
221       Attribute a = paramDefs.get(i);
222       newParamDefs.add(a.copy(m_parameters.get(i).getFieldName()));
223     }
224     
225     m_parameterDefs = newParamDefs;
226     
227     // update the Expression
228     m_expression.setFieldDefs(m_parameterDefs);
229   }
230   
231   public String toString() {
232     return toString("");
233   }
234   
235   public String toString(String pad) {
236     StringBuffer buff = new StringBuffer();
237     
238     buff.append(pad + "DefineFunction (" + m_functionName + "):\n"
239         + pad + "nparameters:\n");
240     
241     for (ParameterField p : m_parameters) {
242       buff.append(pad + p.getFieldAsAttribute() + "\n");
243     }
244     
245     buff.append(pad + "expression:\n" + m_expression.toString(pad + "  "));
246     return buff.toString();
247   }
