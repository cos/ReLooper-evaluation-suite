45    enum Usage {
46      ACTIVE ("active"),
47      PREDICTED ("predicted"),
48      SUPPLEMENTARY ("supplementary"),
49      GROUP ("group"),
50      ORDER ("order");
51      
52      private final String m_stringVal;
53      Usage(String name) {
54        m_stringVal = name;
55      }
56      
57      public String toString() {
58        return m_stringVal;
59      }
60    }
61    
62    /** usage type */
63    Usage m_usageType = Usage.ACTIVE;

65    enum Outlier {
66      ASIS ("asIs"),
67      ASMISSINGVALUES ("asMissingValues"),
68      ASEXTREMEVALUES ("asExtremeValues");
69      
70      private final String m_stringVal;
71      Outlier(String name){
72        m_stringVal = name;
73      }
74      
75      public String toString() {
76        return m_stringVal;
77      }
78    }
79    /** outlier treatmemnt method */
80    protected Outlier m_outlierTreatmentMethod = Outlier.ASIS;
81    
82    /** outlier low value */
83    protected double m_lowValue;
84    /** outlier high value */
85    protected double m_highValue;

87    enum Missing {
88      ASIS ("asIs"),
89      ASMEAN ("asMean"),
90      ASMODE ("asMode"),
91      ASMEDIAN ("asMedian"),
92      ASVALUE ("asValue");
93      
94      private final String m_stringVal;
95      Missing(String name) {
96        m_stringVal = name;
97      }
98      
99      public String toString() {
100       return m_stringVal;
101     }
102   }

104   protected Missing m_missingValueTreatmentMethod = Missing.ASIS;

145   /**
146    * Return a textual representation of this MiningField.
147    * 
148    * @return a String describing this mining field
149    */
150   public String toString() {
151     StringBuffer temp = new StringBuffer();
152     temp.append(m_miningSchemaI.attribute(m_index));
153     temp.append("\n\tusage: " + m_usageType 
154                 + "\n\toutlier treatment: " + m_outlierTreatmentMethod);
155     if (m_outlierTreatmentMethod == Outlier.ASEXTREMEVALUES) {
156       temp.append(" (lowValue = " + m_lowValue + " highValue = " + m_highValue + ")");
157     }
158 
159     temp.append("\n\tmissing value treatment: " 
160                 + m_missingValueTreatmentMethod);
161     if (m_missingValueTreatmentMethod != Missing.ASIS) {
162       temp.append(" (replacementValue = " 
163                   + ((m_missingValueReplacementNominal != null)
164                      ? m_missingValueReplacementNominal
165                      : Utils.doubleToString(m_missingValueReplacementNumeric, 4))
166                   + ")");
167     }
168 
169     return temp.toString();
170   }

209   /**
210    * Apply the missing value treatment method for this field.
211    *
212    * @param value the incoming value to apply the treatment to
213    * @return the value after applying the missing value treatment (if any)
214    * @throws Exception if there is a problem
215    */
216   public double applyMissingValueTreatment(double value) throws Exception {
217     double newVal = value;
218     if (m_missingValueTreatmentMethod != Missing.ASIS && 
219         Utils.isMissingValue(value)) {
220       if (m_missingValueReplacementNominal != null) {
221         Attribute att = m_miningSchemaI.attribute(m_index);
222         int valIndex = att.indexOfValue(m_missingValueReplacementNominal);
223         if (valIndex < 0) {
224           throw new Exception("[MiningSchema] Nominal missing value replacement value doesn't "
225                               + "exist in the mining schema Instances!");
226         }
227         newVal = valIndex;
228       } else {
229         newVal = m_missingValueReplacementNumeric;
230       }
231     }
232     return newVal;
233   }
234 
235   /**
236    * Apply the outlier treatment method for this field.
237    *
238    * @param value the incoming value to apply the treatment to
239    * @return the value after applying the treatment (if any)
240    * @throws Exception if there is a problem
241    */
242   public double applyOutlierTreatment(double value) throws Exception {
243     double newVal = value;
244     if (m_outlierTreatmentMethod != Outlier.ASIS) {
245       if (m_outlierTreatmentMethod == Outlier.ASMISSINGVALUES) {
246         newVal = applyMissingValueTreatment(value);
247       } else {
248         if (value < m_lowValue) {
249           newVal = m_lowValue;
250         } else if (value > m_highValue) {
251           newVal = m_highValue;
252         }
253       }
254     }
255     return newVal;
256   }

263   public Attribute getFieldAsAttribute() {
264     return m_miningSchemaI.attribute(m_index);
265   }
266   /**
267    * Constructs a new MiningFieldMetaInfo object.
268    * 
269    * @param field the Element that contains the field information
270    * @throws Exception if there is a problem during construction
271    */
272   public MiningFieldMetaInfo(Element field) throws Exception {
273     super(field);
274     // m_fieldName = field.getAttribute("name");
275 
276     // get the usage type
277     String usage = field.getAttribute("usageType");
278     for (MiningFieldMetaInfo.Usage u : Usage.values()) {
279       if (u.toString().equals(usage)) {
280         m_usageType = u;
281         break;
282       }
283     }
284     
285     // optype override
286     /*String optype = field.getAttribute("optype");
287     if (optype.length() > 0) {
288       if (optype.equals("continuous")) {
289         m_optypeOverride = FieldMetaInfo.Optype.CONTINUOUS;
290       } else if (optype.equals("categorical")) {
291         m_optypeOverride = FieldMetaInfo.Optype.CATEGORICAL;
292       } else if (optype.equals("ordinal")) {
293         m_optypeOverride = FieldMetaInfo.Optype.ORDINAL;
294       }
295     }*/
296   
297     // importance
298     String importance = field.getAttribute("importance");
299     if (importance.length() > 0) {
300       m_importance = Double.parseDouble(importance);
301     }
302 
303     // outliers
304     String outliers = field.getAttribute("outliers");
305     for (MiningFieldMetaInfo.Outlier o : Outlier.values()) {
306       if (o.toString().equals(outliers)) {
307         m_outlierTreatmentMethod = o;
308         break;
309       }
310     }
311     
312     if (outliers.length() > 0 && m_outlierTreatmentMethod == Outlier.ASEXTREMEVALUES) {
313       // low and high values are required for as extreme values handling
314       String lowValue = field.getAttribute("lowValue");
315       if (lowValue.length() > 0) {
316         m_lowValue = Double.parseDouble(lowValue);
317       } else {
318         throw new Exception("[MiningFieldMetaInfo] as extreme values outlier treatment "
319             + "specified, but no low value defined!");
320       }
321       String highValue = field.getAttribute("highValue");
322       if (highValue.length() > 0) {
323         m_highValue = Double.parseDouble(highValue);
324       } else {
325         throw new Exception("[MiningFieldMetaInfo] as extreme values outlier treatment "
326             + "specified, but no high value defined!");
327       }
328     }
329     
330 
331     // missing values
332     String missingReplacement = field.getAttribute("missingValueReplacement");
333     if (missingReplacement.length() > 0) {
334       // try and parse it as a number
335       try {
336         m_missingValueReplacementNumeric = Double.parseDouble(missingReplacement);
337       } catch (IllegalArgumentException ex) {
338         // must be numeric
339         m_missingValueReplacementNominal = missingReplacement;
340       }
341     
342       // treatment type
343       String missingTreatment = field.getAttribute("missingValueTreatment");
344       for (MiningFieldMetaInfo.Missing m : Missing.values()) {
345         if (m.toString().equals(missingTreatment)) {
346           m_missingValueTreatmentMethod = m;
347           break;
348         }
349       }
350     }
351   }
