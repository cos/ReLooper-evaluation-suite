35  /**
36   * This class encapsulates the mining schema from
37   * a PMML xml file. Specifically, it contains the
38   * fields used in the PMML model as an Instances
39   * object (just the header). It also contains meta
40   * information such as value ranges and how to handle
41   * missing values, outliers etc.
42   *
43   * We also store various other PMML elements here, such as
44   * the TransformationDictionary, DerivedFields and Targets 
45   * (if defined). They are not part of the mining schema per se, but
46   * relate to inputs used by the model and it is convenient to
47   * store them here.
48   *
49   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
50   * @version $Revision: 5987 $
51   */
52  public class MiningSchema implements Serializable {

81    private void getLocalTransformations(Element model) throws Exception {
82      NodeList temp = model.getElementsByTagName("LocalTransformations");
83      
84      if (temp.getLength() > 0) {
85        // should be just one LocalTransformations element
86        Element localT = (Element)temp.item(0);
87        
88        // Set up some field defs to pass in
89        /*ArrayList<Attribute> fieldDefs = new ArrayList<Attribute>();
90        for (int i = 0; i < m_miningSchemaInstancesStructure.numAttributes(); i++) {
91          fieldDefs.add(m_miningSchemaInstancesStructure.attribute(i));
92        } */
93        
94        NodeList localDerivedL = localT.getElementsByTagName("DerivedField");
95        for (int i = 0; i < localDerivedL.getLength(); i++) {
96          Node localDerived = localDerivedL.item(i);
97          if (localDerived.getNodeType() == Node.ELEMENT_NODE) {
98            DerivedFieldMetaInfo d = 
99              new DerivedFieldMetaInfo((Element)localDerived, null /*fieldDefs*/, m_transformationDictionary);
100           m_derivedMeta.add(d);
101         }
102       }
103     }
104   }

106   /**
107    * Constructor for MiningSchema.
108    *
109    * @param model the <code>Element</code> encapsulating the pmml model
110    * @param dataDictionary the data dictionary as an Instances object
111    * @throws Exception if something goes wrong during construction of the
112    * mining schema
113    */
114   public MiningSchema(Element model, 
115                       Instances dataDictionary,
116                       TransformationDictionary transDict) throws Exception {
117     
118     /*// First check for transformation dictionary/local transformations and derived fields.
119     // These are not supported yet.
120     NodeList temp = model.getElementsByTagName("LocalTransformations");
121     if (temp.getLength() > 0) {
122       throw new Exception("[MiningSchema] LocalTransformations "
123           + "are not supported yet.");
124     }*/
125 
126     ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
127     NodeList fieldList = model.getElementsByTagName("MiningField");
128     int classIndex = -1;
129     int addedCount = 0;
130     for (int i = 0; i < fieldList.getLength(); i++) {
131       Node miningField = fieldList.item(i);
132       if (miningField.getNodeType() == Node.ELEMENT_NODE) {
133         Element miningFieldEl = (Element)miningField;
134 
135         MiningFieldMetaInfo mfi = new MiningFieldMetaInfo(miningFieldEl);
136 
137         if (mfi.getUsageType() == MiningFieldMetaInfo.Usage.ACTIVE ||
138             mfi.getUsageType() == MiningFieldMetaInfo.Usage.PREDICTED) {
139 
140           // find this attribute in the dataDictionary
141           Attribute miningAtt = dataDictionary.attribute(mfi.getName());
142           if (miningAtt != null) {
143             mfi.setIndex(addedCount);
144             attInfo.add(miningAtt);
145             addedCount++;
146 
147             if (mfi.getUsageType() == MiningFieldMetaInfo.Usage.PREDICTED) {
148               classIndex = addedCount - 1;
149             }
150 
151             // add to the array list
152             m_miningMeta.add(mfi);
153           } else {
154           throw new Exception("Can't find mining field: " + mfi.getName() 
155                               + " in the data dictionary.");
156           }
157         }
158       }
159     }
160 
161     m_miningSchemaInstancesStructure = new Instances("miningSchema", attInfo, 0);
162     
163     // set these instances on the MiningFieldMetaInfos so that the
164     // toString() method can operate correctly
165     for (MiningFieldMetaInfo m : m_miningMeta) {
166       m.setMiningSchemaInstances(m_miningSchemaInstancesStructure);
167     }
168     
169     m_transformationDictionary = transDict;
170     
171     // Handle transformation dictionary and any local transformations
172     if (m_transformationDictionary != null) {      
173       ArrayList<DerivedFieldMetaInfo> transDerived = transDict.getDerivedFields();
174       m_derivedMeta.addAll(transDerived);
175     }
176     
177     // Get any local transformations
178     getLocalTransformations(model);
179     
180     // Set up the full instances structure: combo of mining schema fields and
181     // all derived fields
182     ArrayList<Attribute> newStructure = new ArrayList<Attribute>();
183     for (MiningFieldMetaInfo m : m_miningMeta) {
184       newStructure.add(m.getFieldAsAttribute());
185     }
186     
187     for (DerivedFieldMetaInfo d : m_derivedMeta) {
188       newStructure.add(d.getFieldAsAttribute());
189     }
190     m_fieldInstancesStructure = new Instances("FieldStructure", newStructure, 0);
191     
192     if (m_transformationDictionary != null) {
193       // first update the field defs for any derived fields in the transformation dictionary
194       // and our complete list of derived fields, now that we have a fixed 
195       // ordering for the mining schema + derived attributes (i.e. could
196       // be different from the order of attributes in the data dictionary that was
197       // used when the transformation dictionary was initially constructed
198       m_transformationDictionary.setFieldDefsForDerivedFields(m_fieldInstancesStructure);
199     }
200 
201     // update the field defs for all our derived fields.
202     for (DerivedFieldMetaInfo d : m_derivedMeta) {
203       d.setFieldDefs(m_fieldInstancesStructure);
204     }
205     
206     if (classIndex != -1) {
207       m_fieldInstancesStructure.setClassIndex(classIndex);
208       m_miningSchemaInstancesStructure.setClassIndex(classIndex);
209     }
210 
211     // do Targets (if any)
212     NodeList targetsList = model.getElementsByTagName("Targets");
213     if (targetsList.getLength() > 0) {
214       if (targetsList.getLength() > 1) {
215         throw new Exception("[MiningSchema] Can only handle a single Target");
216       } else {
217         Node te = targetsList.item(0);
218         if (te.getNodeType() == Node.ELEMENT_NODE) {
219           m_targetMetaInfo = new TargetMetaInfo((Element)te);
220 
221           // fill in any necessary categorical values in the mining schema 
222           // class attribute
223           if (m_fieldInstancesStructure.classIndex() >= 0 && 
224               m_fieldInstancesStructure.classAttribute().isString()) {
225             ArrayList<String> targetVals = m_targetMetaInfo.getValues();
226             if (targetVals.size() > 0) {
227               Attribute classAtt = m_fieldInstancesStructure.classAttribute();
228               for (int i = 0; i < targetVals.size(); i++) {
229                 classAtt.addStringValue(targetVals.get(i));
230               }
231             }
232           }
233         }
234       }
235     }
236   }

238   /**
239    * Apply the missing value treatments (if any) to an incoming instance.
240    *
241    * @param values an array of doubles in order of the fields in the mining schema
242    * that represents the incoming instance (note: use PMMLUtils.instanceToSchema()
243    * to generate this).
244    * @throws Exception if something goes wrong during missing value handling
245    */
246   public void applyMissingValuesTreatment(double[] values) throws Exception {
247     for (int i = 0; i < m_miningMeta.size(); i++) {
248       MiningFieldMetaInfo mfi = m_miningMeta.get(i);
249       values[i] = mfi.applyMissingValueTreatment(values[i]);
250     }
251   }
252 
253   /**
254    * Apply the outlier treatment methods (if any) to an incoming instance.
255    *
256    * @param values an array of doubles in order of the fields in the mining schema
257    * that represents the incoming instance (note: use PMMLUtils.instanceToSchema()
258    * to generate this).
259    * @throws Exception if something goes wrong during outlier treatment handling
260    */
261   public void applyOutlierTreatment(double[] values) throws Exception {
262     for (int i = 0; i < m_miningMeta.size(); i++) {
263       MiningFieldMetaInfo mfi = m_miningMeta.get(i);
264       values[i] = mfi.applyOutlierTreatment(values[i]);
265     }
266   }
267 
268   /**
269    * Apply both missing and outlier treatments to an incoming instance.
270    * @param values an array of doubles in order of the fields in the mining schema
271    * that represents the incoming instance (note: use MappingInfo.instanceToSchema()
272    * to generate this).
273    * @throws Exception if something goes wrong during this process
274    */
275   public void applyMissingAndOutlierTreatments(double[] values) throws Exception {
276     for (int i = 0; i < m_miningMeta.size(); i++) {
277       MiningFieldMetaInfo mfi = m_miningMeta.get(i);
278       values[i] = mfi.applyMissingValueTreatment(values[i]);
279       values[i] = mfi.applyOutlierTreatment(values[i]);
280     }
281   }

332   /**
333    * Method to convert any string attributes in the mining schema
334    * Instances to nominal attributes. This may be necessary if there are
335    * no Value elements defined for categorical fields in the data dictionary.
336    * In this case, elements in the actual model definition will probably reveal
337    * the valid values for categorical fields.
338    */
339   public void convertStringAttsToNominal() {
340     Instances miningSchemaI = getFieldsAsInstances();
341     if (miningSchemaI.checkForStringAttributes()) {
342       ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
343       for (int i = 0; i < miningSchemaI.numAttributes(); i++) {
344         Attribute tempA = miningSchemaI.attribute(i);
345         if (tempA.isString()) {
346           ArrayList<String> valueVector = new ArrayList<String>();
347           for (int j = 0; j < tempA.numValues(); j++) {
348             valueVector.add(tempA.value(j));
349           }
350           Attribute newAtt = new Attribute(tempA.name(), valueVector);
351           attInfo.add(newAtt);
352         } else {
353           attInfo.add(tempA);
354         }
355       }
356       Instances newI = new Instances("miningSchema", attInfo, 0);
357       if (m_fieldInstancesStructure.classIndex() >= 0) {
358         newI.setClassIndex(m_fieldInstancesStructure.classIndex());
359       }
360       m_fieldInstancesStructure = newI;
361 
362       /*      StringToNominal stn = new StringToNominal();
363       stn.setInputFormat(miningSchemaI);
364       Instances newI = Filter.useFilter(miningSchemaI, stn);
365       m_miningSchema = newI; */
366     }
367   }

369   /**
370    * Convert a numeric attribute in the mining schema to nominal.
371    * 
372    * @param index the index of the attribute to convert
373    * @param newVals an ArrayList of the values of the nominal attribute
374    */
375   public void convertNumericAttToNominal(int index, 
376                                          ArrayList<String> newVals) {
377     Instances miningSchemaI = getFieldsAsInstances();
378     if (miningSchemaI.attribute(index).isNominal()) {
379       throw new IllegalArgumentException("[MiningSchema] convertNumericAttToNominal: attribute is "
380                                          + "already nominal!");
381     }
382 
383     ArrayList<String> newValues = new ArrayList<String>();
384     for (int i = 0; i < newVals.size(); i++) {
385       newValues.add(newVals.get(i));
386     }
387 
388     ArrayList<Attribute> attInfo = new ArrayList<Attribute>();
389     for (int i = 0; i < miningSchemaI.numAttributes(); i++) {
390       Attribute tempA = miningSchemaI.attribute(i);
391       if (i == index) {
392         Attribute newAtt = new Attribute(tempA.name(), newValues);
393         attInfo.add(newAtt);
394       } else {
395         attInfo.add(tempA);
396       }
397     }
398 
399     Instances newI = new Instances("miningSchema", attInfo, 0);
400     if (m_fieldInstancesStructure.classIndex() >= 0) {
401       newI.setClassIndex(m_fieldInstancesStructure.classIndex());
402     }
403     m_fieldInstancesStructure = newI;
404   }

414   /**
415    * Get a textual description of the mining schema.
416    *
417    * @return a textual description of the mining schema
418    */
419   public String toString() {
420     StringBuffer temp = new StringBuffer();
421     
422     if (m_transformationDictionary != null) {
423       temp.append(m_transformationDictionary);
424     }
425     
426     temp.append("Mining schema:\n\n");
427     for (MiningFieldMetaInfo m : m_miningMeta) {
428       temp.append(m + "\n");
429     }
430     
431     if (m_derivedMeta.size() > 0) {
432       temp.append("\nDerived fields:\n\n");
433       for (DerivedFieldMetaInfo d : m_derivedMeta) {
434         temp.append(d + "\n");
435       }
436     }
437     temp.append("\n");
438     return temp.toString();
439   }
