23  package weka.core.pmml;
24  
25  import java.io.Serializable;
26  import java.util.ArrayList;
27  
28  import org.w3c.dom.Element;
29  import org.w3c.dom.Node;
30  import org.w3c.dom.NodeList;
31  
32  import weka.core.Attribute;
33  import weka.core.Instances;
34  
35  public class DerivedFieldMetaInfo extends FieldMetaInfo implements Serializable {
36    
37    /** for serialization */
38    
39    /** display name */
40    protected String m_displayName = null;
41    
42    /** 
43     * the list of values (if the field is ordinal) - may be of size zero if none are specified.
44     * If none are specified, we may be able to construct this by querying the Expression in
45     * this derived field 
46     */
47    protected ArrayList<Value> m_values = new ArrayList<Value>();
48    
49    /** the single expression that defines the value of this field */
50    protected Expression m_expression;
51    
52    public DerivedFieldMetaInfo(Element derivedField, ArrayList<Attribute> fieldDefs,
53                                TransformationDictionary transDict) throws Exception {
54      super(derivedField);
55      // m_fieldName = derivedField.getAttribute("name");
56      String displayName = derivedField.getAttribute("displayName");
57      if (displayName != null && displayName.length() > 0) {
58        m_displayName = displayName;
59      }
60      
61      // get any values
62      NodeList valL = derivedField.getElementsByTagName("Value");
63      if (valL.getLength() > 0) {
64        for (int i = 0; i < valL.getLength(); i++) {
65          Node valueN = valL.item(i);
66          if (valueN.getNodeType() == Node.ELEMENT_NODE) {
67            Value v = new Value((Element)valueN);
68            m_values.add(v);
69          }
70        }
71      }
72      
73      // now get the expression
74      m_expression = Expression.getExpression(derivedField, m_optype, fieldDefs, transDict);
75    }
76    
77    /**
78     * Upadate the field definitions for this derived field
79     * 
80     * @param fieldDefs the fields as an ArrayList of Attributes
81     * @throws Exception if there is a problem setting the field definitions
82     */
83    public void setFieldDefs(ArrayList<Attribute> fieldDefs) throws Exception {
84      m_expression.setFieldDefs(fieldDefs);
85    }
86    
87    /**
88     * Upadate the field definitions for this derived field
89     * 
90     * @param fields the fields as an Instances object
91     * @throws Exception if there is a problem setting the field definitions
92     */
93    public void setFieldDefs(Instances fields) throws Exception {
94      ArrayList<Attribute> tempDefs = new ArrayList<Attribute>();
95      for (int i = 0; i < fields.numAttributes(); i++) {
96        tempDefs.add(fields.attribute(i));
97      }
98      setFieldDefs(tempDefs);
99    }
100   
101   /**
102    * Get this derived field as an Attribute.
103    * 
104    * @return an Attribute for this derived field.
105    */
106   public Attribute getFieldAsAttribute() {
107     return m_expression.getOutputDef().copy(m_fieldName);
108   }
109   
110   /**
111    * Get the derived field value for the given incoming vector of
112    * values. Incoming values are assumed to be in the same order
113    * as the attributes supplied in the field definitions ArrayList
114    * used to construct this DerivedField.
115    * 
116    * If the optype of this derived field is continuous, then a real
117    * number is returned. Otherwise, the number returned is the index
118    * of the categorical/ordinal value corresponding to result of computing
119    * the derived field value.
120    * 
121    * @param incoming the incoming parameter values
122    * @return the result of computing the derived value
123    * @throws Exception if there is a problem computing the value
124    */
125   public double getDerivedValue(double[] incoming) throws Exception {
126     return m_expression.getResult(incoming);
127   }
128   
129   public String toString() {
130     StringBuffer buff = new StringBuffer();
131     buff.append(getFieldAsAttribute() + "\nexpression:\n");
132     buff.append(m_expression + "\n");
133     
134     return buff.toString();
135   }
136 }
