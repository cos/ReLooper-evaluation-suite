30  /**
31   * Abstract superclass for PMML built-in and DefineFunctions.
32   * 
33   * @author Mark Hall (mhall{[at]}pentaho{[dot]}com)
34   * @version $Revision 1.0 $
35   */
36  public abstract class Function implements Serializable {
37    
38    /**
39     * For serialization 
40     */
41    private static final long serialVersionUID = -6997738288201933171L;
42    
43    /** The name of this function */
44    protected String m_functionName;
45    
46    /** The structure of the parameters to this function */
47    protected ArrayList<Attribute> m_parameterDefs = null;
48    
49      
50    public String getName() {
51      return m_functionName;
52    }
53    
54    /**
55     * Returns an array of the names of the parameters expected
56     * as input by this function. May return null if this function
57     * can take an unbounded number of parameters (i.e. min, max, etc.).
58     * 
59     * @return an array of the parameter names or null if there are an
60     * unbounded number of parameters.
61     */
62    public abstract String[] getParameterNames();
63    
64    /**
65     * Set the structure of the parameters that are expected as input by
66     * this function. This must be called before getOutputDef() is called.
67     * 
68     * @param paramDefs the structure of the input parameters
69     * @throws Exception if the number or types of parameters are not acceptable by
70     * this function
71     */
72    public abstract void setParameterDefs(ArrayList<Attribute> paramDefs) throws Exception;
73    
74    /**
75     * Get the structure of the result produced by this function.
76     * 
77     * @return the structure of the result produced by this function.
78     */
79    public abstract Attribute getOutputDef();
80    
81    /**
82     * Get the result of applying this function.
83     * 
84     * @param incoming the arguments to this function (supplied in order to match that
85     * of the parameter definitions
86     * @return the result of applying this function. When the optype is
87     * categorical or ordinal, an index into the values of the output definition
88     * is returned.
89     * @throws Exception if there is a problem computing the result of this function
90     */
91    public abstract double getResult(double[] incoming) throws Exception;
92    
93    /**
94     * Get the result of applying this function. Subclasses should overide this
95     * method when they might produce categorical values where the legal set of
96     * values can't be determined apriori (i.e. by just using the input parameter
97     * definitions). An example is the substring function - in this case there
98     * is no way of knowing apriori what all the legal values will be because the
99     * start position and length parameters are not known until the function is
100    * invoked. In this scenario, a client could call getResultCategorical()
101    * repeatedly (in an initialization routine) in order to manually build the
102    * list of legal values and then call this method at processing time, passing
103    * in the pre-computed output structure.
104    * 
105    *  This default implementation ignores the supplied output definition argument
106    *  and simply invokes getResult(incoming).
107    * 
108    * @param incoming the arguments to this function (supplied in order to match that
109    * of the parameter definitions
110    * @param outputDef the output definition to use for looking up the index of
111    * result values (in the case of categorical results)
112    * @return the result of applying this function. When the optype is
113    * categorical or ordinal, an index into the values of the output definition
114    * is returned.
115    * @throws Exception if there is a problem computing the result of this function
116    *
117   public double getResult(double[] incoming, Attribute outputDef) throws Exception {
118     if (outputDef.isString()) {
119       throw new Exception("[Function] outputDef argument must not be a String attribute!");
120     }
121     return getResult(incoming);
122   }*/
123   
124   /**
125    * Get the result of applying this function when the output type categorical.
126    * Will throw an exception for numeric output. If subclasses output definition
127    * is a string attribute (i.e. because all legal values can't be computed apriori),
128    * then the subclass will need to overide this method and return something sensible
129    * in this case.
130    * 
131    * @param incoming the incoming arguments to this function (supplied in order to match
132    * that of the parameter definitions
133    * @return the result of applying this function as a String.
134    * @throws Exception if this method is not applicable because the optype is not
135    * categorical/ordinal
136    *
137   public String getResultCategorical(double[] incoming) throws Exception {
138     if (getOutputDef().isNumeric()) {
139       throw new Exception("[Function] can't return nominal value, output is numeric!!");
140     }
141     
142     if (getOutputDef().isString()) {
143       throw new Exception("[Function] subclass neeeds to overide this method and do "
144           + "something sensible when the output def is a string attribute.");
145     }
146     
147     return getOutputDef().value((int)getResult(incoming));
148   } */

153   /**
154    * Get a built-in PMML Function.
155    * 
156    * @param name the name of the function to get.
157    * @return a built-in Function or null if the named function is not 
158    * known/supported.
159    */
160   public static Function getFunction(String name) {
161     Function result = null;
162     
163     name = name.trim();
164     if (name.equals("+")) {
165       result = new BuiltInArithmetic(BuiltInArithmetic.Operator.ADDITION);
166     } else if (name.equals("-")) {
167       result = new BuiltInArithmetic(BuiltInArithmetic.Operator.SUBTRACTION);
168     } else if (name.equals("*")) {
169       result = new BuiltInArithmetic(BuiltInArithmetic.Operator.MULTIPLICATION);
170     } else if (name.equals("/")) {
171       result = new BuiltInArithmetic(BuiltInArithmetic.Operator.DIVISION);
172     } else if (name.equals(BuiltInMath.MathFunc.MIN.toString())) {
173       result = new BuiltInMath(BuiltInMath.MathFunc.MIN);
174     } else if (name.equals(BuiltInMath.MathFunc.MAX.toString())) {
175       result = new BuiltInMath(BuiltInMath.MathFunc.MAX);
176     } else if (name.equals(BuiltInMath.MathFunc.SUM.toString())) {
177       result = new BuiltInMath(BuiltInMath.MathFunc.SUM);
178     } else if (name.equals(BuiltInMath.MathFunc.AVG.toString())) {
179       result = new BuiltInMath(BuiltInMath.MathFunc.AVG);
180     } else if (name.equals(BuiltInMath.MathFunc.LOG10.toString())) {
181       result = new BuiltInMath(BuiltInMath.MathFunc.LOG10);
182     } else if (name.equals(BuiltInMath.MathFunc.LN.toString())) {
183       result = new BuiltInMath(BuiltInMath.MathFunc.LN);
184     } else if (name.equals(BuiltInMath.MathFunc.SQRT.toString())) {
185       result = new BuiltInMath(BuiltInMath.MathFunc.SQRT);
186     } else if (name.equals(BuiltInMath.MathFunc.ABS.toString())) {
187       result = new BuiltInMath(BuiltInMath.MathFunc.ABS);
188     } else if (name.equals(BuiltInMath.MathFunc.EXP.toString())) {
189       result = new BuiltInMath(BuiltInMath.MathFunc.EXP);
190     } else if (name.equals(BuiltInMath.MathFunc.POW.toString())) {
191       result = new BuiltInMath(BuiltInMath.MathFunc.POW);
192     } else if (name.equals(BuiltInMath.MathFunc.THRESHOLD.toString())) {
193       result = new BuiltInMath(BuiltInMath.MathFunc.THRESHOLD);
194     } else if (name.equals(BuiltInMath.MathFunc.FLOOR.toString())) {
195       result = new BuiltInMath(BuiltInMath.MathFunc.FLOOR);
196     } else if (name.equals(BuiltInMath.MathFunc.CEIL.toString())) {
197       result = new BuiltInMath(BuiltInMath.MathFunc.CEIL);
198     } else if (name.equals(BuiltInMath.MathFunc.ROUND.toString())) {
199       result = new BuiltInMath(BuiltInMath.MathFunc.ROUND);
200     }
201     
202     return result;
203   }

206   /**
207    * Get either a function. Built-in functions are queried first, and then
208    * DefineFunctions in the TransformationDictionary (if any).
209    * 
210    * @param name the name of the function to get.
211    * @param transDict the TransformationDictionary (may be null if there is
212    * no dictionary).
213    * @return the function
214    * @throws Exception if the named function is not known/supported.
215    */
216   public static Function getFunction(String name, TransformationDictionary transDict)
217     throws Exception {
218     
219     Function result = getFunction(name);
220     
221     // try the defined functions in the TransformationDictionary (if any)
222     if (result == null && transDict != null) {
223       result = transDict.getFunction(name);
224     }
225     
226     if (result == null) {
227       throw new Exception("[Function] unknown/unsupported function " + name);
228     }
229     
230     return result;
231   }
232   
233   public String toString() {
234     return toString("");
235   }
236   
237   public String toString(String pad) {
238     return pad + this.getClass().getName();
239   }
