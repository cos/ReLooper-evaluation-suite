310   /**
311    * Generates an ID based on the current settings and returns it. If nothing
312    * can be generated the MISSING_ID string will be returned
313    * 
314    * @return        the ID
315    * @see #MISSING_ID
316    */
317   protected String generateID() {
318     String  result;
319     String[]    authors;
320     String[]    parts;
321     
322     result = m_ID;
323     
324     // try surname of first author and year
325     if (result.length() == 0) {
326       if (exists(Field.AUTHOR) && exists(Field.YEAR)) {
327     authors = getAuthors();
328     if (authors[0].indexOf(",") > -1) {
329       parts = authors[0].split(",");
330       result = parts[0];
331     }
332     else {
333       parts = authors[0].split(" ");
334       if (parts.length == 1)
335         result = parts[0];
336       else
337         result = parts[parts.length - 1];
338     }
339     result += getValue(Field.YEAR);
340     result = result.replaceAll(" ", "");
341       }
342     }
343     
344     // still nothing?
345     if (result.length() == 0)
346       result = MISSING_ID;
347     
348     return result;
349   }

452   /**
453    * Returns a plain-text string representing this technical information.
454    * Note: it only returns a string based on some fields. At least AUTHOR,
455    * YEAR and TITLE are necessary.
456    * 
457    * @return        a string representation of this information
458    */
459   public String toString() {
460     String  result;
461     String[]    authors;
462     int     i;
463     Enumeration<TechnicalInformation>   enm;
464     
465     result  = "";
466     authors = getAuthors();
467     
468     // BOOK
469     if (getType() == Type.BOOK) {
470       for (i = 0; i < authors.length; i++) {
471     if (i > 0)
472       result += ", ";
473     result += authors[i];
474       }
475       if (exists(Field.YEAR))
476     result += " (" + getValue(Field.YEAR) + ").";
477       else
478     result += ".";
479       result += " " + getValue(Field.TITLE) + ".";
480       result += " " + getValue(Field.PUBLISHER);
481       if (exists(Field.ADDRESS))
482     result += ", " + getValue(Field.ADDRESS);
483       result += ".";
484     }
485     // ARTICLE
486     else if (getType() == Type.ARTICLE) {
487       for (i = 0; i < authors.length; i++) {
488     if (i > 0)
489       result += ", ";
490     result += authors[i];
491       }
492       if (exists(Field.YEAR))
493     result += " (" + getValue(Field.YEAR) + ").";
494       else
495     result += ".";
496       result += " " + getValue(Field.TITLE) + ".";
497       
498       // journal
499       if (exists(Field.JOURNAL)) {
500     result += " " + getValue(Field.JOURNAL) + ".";
501 
502     if (exists(Field.VOLUME))
503       result += " " + getValue(Field.VOLUME);
504     if (exists(Field.NUMBER))
505       result += "(" + getValue(Field.NUMBER) + ")";
506     if (exists(Field.PAGES))
507       result += ":" + getValue(Field.PAGES);
508     
509     result += ".";
510       }
511       
512       // other than JOURNAL???
513 
514       // URL
515       if (exists(Field.URL))
516     result += " URL " + getValue(Field.URL) + ".";
517     }
518     // CONFERENCE/INPROCEEDINGS
519     else if ( (getType() == Type.CONFERENCE) || (getType() == Type.INPROCEEDINGS) ) {
520       for (i = 0; i < authors.length; i++) {
521     if (i > 0)
522       result += ", ";
523     result += authors[i];
524       }
525       result += ": " + getValue(Field.TITLE) + ".";
526       result += " In: " + getValue(Field.BOOKTITLE);
527       
528       if (exists(Field.ADDRESS))
529     result += ", " + getValue(Field.ADDRESS);
530       if (exists(Field.PAGES))
531     result += ", " + getValue(Field.PAGES);
532     
533       if (exists(Field.YEAR))
534     result += ", " + getValue(Field.YEAR) + ".";
535       else
536     result += ".";
537     }
538     // INCOLLECTION
539     else if (getType() == Type.INCOLLECTION) {
540       for (i = 0; i < authors.length; i++) {
541     if (i > 0)
542       result += ", ";
543     result += authors[i];
544       }
545       result += ": " + getValue(Field.TITLE) + ".";
546       result += " In ";
547       if (exists(Field.EDITOR))
548     result += getValue(Field.EDITOR) + ", editors, ";
549       result += getValue(Field.BOOKTITLE);
550       
551       if (exists(Field.ADDRESS))
552     result += ", " + getValue(Field.ADDRESS);
553       if (exists(Field.PAGES))
554     result += ", " + getValue(Field.PAGES);
555     
556       if (exists(Field.YEAR))
557     result += ", " + getValue(Field.YEAR) + ".";
558       else
559     result += ".";
560     }
561     // default
562     else {
563       for (i = 0; i < authors.length; i++) {
564     if (i > 0)
565       result += ", ";
566     result += authors[i];
567       }
568       if (exists(Field.YEAR))
569     result += " (" + getValue(Field.YEAR) + ").";
570       else
571     result += ".";
572       result += " " + getValue(Field.TITLE) + ".";
573       if (exists(Field.ADDRESS))
574     result += " " + getValue(Field.ADDRESS) + ".";
575       if (exists(Field.URL))
576     result += " URL " + getValue(Field.URL) + ".";
577     }
578     
579     // additional informations?
580     enm = additional();
581     while (enm.hasMoreElements()) {
582       result += "\n\n" + enm.nextElement().toString();
583     }
584     
585     return result;
586   }

588   /**
589    * Returns a BibTex string representing this technical information. 
590    * Note: this is just a very raw implementation, special characters need to
591    * be escaped manually for LaTeX.
592    * 
593    * @return        the BibTeX representation of this information
594    */
595   public String toBibTex() {
596     String      result;
597     Field       field;
598     Vector<Field>       list;
599     int         i;
600     String      value;
601     
602     result = "@" + getType() + "{" + getID() + "";
603     
604     // sort the fields
605     list = new Vector<Field>();
606     Enumeration<Field> enm  = fields();
607     while (enm.hasMoreElements())
608       list.add(enm.nextElement());
609     Collections.sort(list);
610     
611     // list field=value pairs
612     for (i = 0; i < list.size(); i++) {
613       field = (Field) list.get(i);
614       if (!exists(field))
615     continue;
616       value = getValue(field);
617       value = value.replaceAll("\\~", "\\\\~");
618       result += ",\n   " + field + " = {" + value +  "}";
619     }
620     
621     result += "\n}";
622     
623     // additional informations?
624     Enumeration<TechnicalInformation> enm2 = additional();
625     while (enm2.hasMoreElements()) {
626       result += "\n\n" + ((TechnicalInformation) enm2.nextElement()).toBibTex();
627     }
628     
629     return result;
630   }

635    * @return        the revision
636    */

638     return RevisionUtils.extract("$Revision: 5953 $");

642    * Prints some examples of technical informations if there are no 
643    * commandline options given. Otherwise the information of a given 
644    * TechnicalInformationHandler can be printed. <p/>
645    * 
646    * Valid options are: <p/>
647    * 
648    * -W classname <br/>
649    *  The classname of the TechnicalInformationHandler to print the 
650    *  information for <p/>
651    *  
652    * -bibtex <br/>
653    *  Print the information in BibTeX format <p/>
654    *  
655    * -plaintext <br/>
656    *  Print the information in plain text format <p/>
657    * 
658    * @param args    the commandline options
659    * @throws Exception  if the option parsing fails
660    */
661   public static void main(String[] args) throws Exception {
662     TechnicalInformation    info;
663     TechnicalInformation    additional;
664     String          tmpStr;
665     Class           cls;
666     TechnicalInformationHandler handler;
667     
668     // example from command line
669     if (args.length != 0) {
670       info = null;
671       
672       tmpStr = Utils.getOption('W', args);
673       if (tmpStr.length() != 0) {
674     cls     = Class.forName(tmpStr);
675     handler = (TechnicalInformationHandler) cls.newInstance();
676     info    = handler.getTechnicalInformation();
677       }
678       else {
679     throw new IllegalArgumentException("A classname has to be provided with the -W option!");
680       }
681       
682       if (Utils.getFlag("bibtex", args))
683         System.out.println("\n" + handler.getClass().getName() + ":\n" + info.toBibTex());
684 
685       if (Utils.getFlag("plaintext", args))
686         System.out.println("\n" + handler.getClass().getName() + ":\n" + info.toString());
687     }
688     else {
689       // first example
690       info = new TechnicalInformation(Type.BOOK);
691       info.setValue(Field.AUTHOR, "Ross Quinlan");
692       info.setValue(Field.YEAR, "1993");

701       // second example
