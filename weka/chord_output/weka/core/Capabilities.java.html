33  /**
34   * A class that describes the capabilites (e.g., handling certain types of
35   * attributes, missing values, types of classes, etc.) of a specific classifier.
36   * By default, the classifier is capable of nothing. This ensures that new
37   * features have to be enabled explicitly.
38   * <p/>
39   * 
40   * A common code fragment for making use of the capabilities in a classifier
41   * would be this:
42   * 
43   * <pre>
44   * public void <b>buildClassifier</b>(Instances instances) throws Exception {
45   *   // can the classifier handle the data?
46   *   getCapabilities().<b>testWithFail(instances)</b>;
47   *   ...
48   *   // possible deletion of instances with missing class labels, etc.
49   * </pre>
50   * 
51   * For only testing a single attribute, use this:
52   * 
53   * <pre>
54   *   ...
55   *   Attribute att = instances.attribute(0);
56   *   getCapabilities().<b>testWithFail(att)</b>;
57   *   ...
58   * </pre>
59   * 
60   * Or for testing the class attribute (uses the capabilities that are especially
61   * for the class):
62   * 
63   * <pre>
64   *   ...
65   *   Attribute att = instances.classAttribute();
66   *   getCapabilities().<b>testWithFail(att, <i>true</i>)</b>;
67   *   ...
68   * </pre>
69   * 
70   * @author FracPete (fracpete at waikato dot ac dot nz)
71   * @version $Revision: 5953 $
72   */
73  public class Capabilities implements Cloneable, Serializable, RevisionHandler {

84      /** defines an attribute type */
85      private final static int ATTRIBUTE = 1;

87      /** defines a class type */
88      private final static int CLASS = 2;

91      private final static int ATTRIBUTE_CAPABILITY = 4;

94      private final static int CLASS_CAPABILITY = 8;

96      /** defines a other capability */
97      private final static int OTHER_CAPABILITY = 16;

99      /** enumeration of all capabilities */
100     public enum Capability {
101         // attributes
102         /** can handle nominal attributes */
103         NOMINAL_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Nominal attributes"),
104         /** can handle binary attributes */
105         BINARY_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Binary attributes"),
106         /** can handle unary attributes */
107         UNARY_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Unary attributes"),
108         /** can handle empty nominal attributes */
109         EMPTY_NOMINAL_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Empty nominal attributes"),
110         /** can handle numeric attributes */
111         NUMERIC_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Numeric attributes"),
112         /** can handle date attributes */
113         DATE_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Date attributes"),
114         /** can handle string attributes */
115         STRING_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "String attributes"),
116         /** can handle relational attributes */
117         RELATIONAL_ATTRIBUTES(ATTRIBUTE + ATTRIBUTE_CAPABILITY, "Relational attributes"),
118         /** can handle missing values in attributes */
119         MISSING_VALUES(ATTRIBUTE_CAPABILITY, "Missing values"),
120         // class
121         /** can handle data without class attribute, eg clusterers */
122         NO_CLASS(CLASS_CAPABILITY, "No class"),
123         /** can handle nominal classes */
124         NOMINAL_CLASS(CLASS + CLASS_CAPABILITY, "Nominal class"),
125         /** can handle binary classes */
126         BINARY_CLASS(CLASS + CLASS_CAPABILITY, "Binary class"),
127         /** can handle unary classes */
128         UNARY_CLASS(CLASS + CLASS_CAPABILITY, "Unary class"),
129         /** can handle empty nominal classes */
130         EMPTY_NOMINAL_CLASS(CLASS + CLASS_CAPABILITY, "Empty nominal class"),
131         /** can handle numeric classes */
132         NUMERIC_CLASS(CLASS + CLASS_CAPABILITY, "Numeric class"),
133         /** can handle date classes */
134         DATE_CLASS(CLASS + CLASS_CAPABILITY, "Date class"),
135         /** can handle string classes */
136         STRING_CLASS(CLASS + CLASS_CAPABILITY, "String class"),
137         /** can handle relational classes */
138         RELATIONAL_CLASS(CLASS + CLASS_CAPABILITY, "Relational class"),
139         /** can handle missing values in class attribute */
140         MISSING_CLASS_VALUES(CLASS_CAPABILITY, "Missing class values"),
141         // other
142         /** can handle multi-instance data */
143         ONLY_MULTIINSTANCE(OTHER_CAPABILITY, "Only multi-Instance data");
144 
145         /** the flags for the capabilities */
146         private int m_Flags = 0;
147 
148         /** the display string */
149         private String m_Display;
150 
151         /**
152          * initializes the capability with the given flags
153          * 
154          * @param flags
155          *          "meta-data" for the capability
156          * @param display
157          *          the display string (must be unique!)
158          */
159         private Capability(int flags, String display) {
160             m_Flags = flags;
161             m_Display = display;
162         }
163 
164         /**
165          * returns true if the capability is an attribute
166          * 
167          * @return true if the capability is an attribute
168          */
169         public boolean isAttribute() {
170             return ((m_Flags & ATTRIBUTE) == ATTRIBUTE);
171         }
172 
173         /**
174          * returns true if the capability is a class
175          * 
176          * @return true if the capability is a class
177          */
178         public boolean isClass() {
179             return ((m_Flags & CLASS) == CLASS);
180         }
181 
182         /**
183          * returns true if the capability is an attribute capability
184          * 
185          * @return true if the capability is an attribute capability
186          */
187         public boolean isAttributeCapability() {
188             return ((m_Flags & ATTRIBUTE_CAPABILITY) == ATTRIBUTE_CAPABILITY);
189         }
190 
191         /**
192          * returns true if the capability is a class capability
193          * 
194          * @return true if the capability is a class capability
195          */
196         public boolean isOtherCapability() {
197             return ((m_Flags & OTHER_CAPABILITY) == OTHER_CAPABILITY);
198         }
199 
200         /**
201          * returns true if the capability is a other capability
202          * 
203          * @return true if the capability is a other capability
204          */
205         public boolean isClassCapability() {
206             return ((m_Flags & CLASS_CAPABILITY) == CLASS_CAPABILITY);
207         }
208 
209         /**
210          * returns the display string of the capability
211          * 
212          * @return the display string
213          */
214         public String toString() {
215             return m_Display;
216         }

252     /**
253      * initializes the capabilities for the given owner
254      * 
255      * @param owner
256      *          the object that produced this Capabilities instance
257      */
258     public Capabilities(CapabilitiesHandler owner) {
259         super();
260 
261         setOwner(owner);
262         m_Capabilities = new HashSet<Capability>();
263         m_Dependencies = new HashSet<Capability>();
264 
265         // load properties
266         if (PROPERTIES == null) {
267             try {
268                 PROPERTIES = Utils.readProperties(PROPERTIES_FILE);
269             } catch (Exception e) {
270                 e.printStackTrace();
271                 PROPERTIES = new Properties();
272             }
273         }
274 
275         m_Test = Boolean.parseBoolean(PROPERTIES.getProperty("Test", "true"));
276         m_InstancesTest = Boolean.parseBoolean(PROPERTIES.getProperty("InstancesTest", "true")) && m_Test;
277         m_AttributeTest = Boolean.parseBoolean(PROPERTIES.getProperty("AttributeTest", "true")) && m_Test;
278         m_MissingValuesTest = Boolean.parseBoolean(PROPERTIES.getProperty("MissingValuesTest", "true")) && m_Test;
279         m_MissingClassValuesTest = Boolean.parseBoolean(PROPERTIES.getProperty("MissingClassValuesTest", "true")) && m_Test;
280         m_MinimumNumberInstancesTest = Boolean.parseBoolean(PROPERTIES.getProperty("MinimumNumberInstancesTest", "true"))
281             && m_Test;
282     }

298     /**
299      * retrieves the data from the given Capabilities object
300      * 
301      * @param c
302      *          the capabilities object to initialize with
303      */
304     public void assign(Capabilities c) {
305         for (Capability cap : Capability.values()) {
306             // capability
307             if (c.handles(cap))
308                 enable(cap);
309             else
310                 disable(cap);
311             // dependency
312             if (c.hasDependency(cap))
313                 enableDependency(cap);
314             else
315                 disableDependency(cap);
316         }
317 
318         setMinimumNumberInstances(c.getMinimumNumberInstances());
319     }
320 
321     /**
322      * performs an AND conjunction with the capabilities of the given Capabilities
323      * object and updates itself
324      * 
325      * @param c
326      *          the capabilities to AND with
327      */
328     public void and(Capabilities c) {
329         for (Capability cap : Capability.values()) {
330             // capability
331             if (handles(cap) && c.handles(cap))
332                 m_Capabilities.add(cap);
333             else
334                 m_Capabilities.remove(cap);
335             // dependency
336             if (hasDependency(cap) && c.hasDependency(cap))
337                 m_Dependencies.add(cap);
338             else
339                 m_Dependencies.remove(cap);
340         }
341 
342         // minimum number of instances that both handlers need at least to work
343         if (c.getMinimumNumberInstances() > getMinimumNumberInstances())
344             setMinimumNumberInstances(c.getMinimumNumberInstances());
345     }
346 
347     /**
348      * performs an OR conjunction with the capabilities of the given Capabilities
349      * object and updates itself
350      * 
351      * @param c
352      *          the capabilities to OR with
353      */
354     public void or(Capabilities c) {
355         for (Capability cap : Capability.values()) {
356             // capability
357             if (handles(cap) || c.handles(cap))
358                 m_Capabilities.add(cap);
359             else
360                 m_Capabilities.remove(cap);
361             // dependency
362             if (hasDependency(cap) || c.hasDependency(cap))
363                 m_Dependencies.add(cap);
364             else
365                 m_Dependencies.remove(cap);
366         }
367 
368         if (c.getMinimumNumberInstances() < getMinimumNumberInstances())
369             setMinimumNumberInstances(c.getMinimumNumberInstances());
370     }

372     /**
373      * Returns true if the currently set capabilities support at least all of the
374      * capabiliites of the given Capabilities object (checks only the enum!)
375      * 
376      * @param c
377      *          the capabilities to support at least
378      * @return true if all the requested capabilities are supported
379      */
380     public boolean supports(Capabilities c) {
381         boolean result;
382 
383         result = true;
384 
385         for (Capability cap : Capability.values()) {
386             if (c.handles(cap) && !handles(cap)) {
387                 result = false;
388                 break;
389             }
390         }
391 
392         return result;
393     }
394 
395     /**
396      * Returns true if the currently set capabilities support (or have a
397      * dependency) at least all of the capabilities of the given Capabilities
398      * object (checks only the enum!)
399      * 
400      * @param c
401      *          the capabilities (or dependencies) to support at least
402      * @return true if all the requested capabilities are supported (or at least
403      *         have a dependency)
404      */
405     public boolean supportsMaybe(Capabilities c) {
406         boolean result;
407 
408         result = true;
409 
410         for (Capability cap : Capability.values()) {
411             if (c.handles(cap) && !(handles(cap) || hasDependency(cap))) {
412                 result = false;
413                 break;
414             }
415         }
416 
417         return result;
418     }

477     /**
478      * enables the given capability. Enabling NOMINAL_ATTRIBUTES also enables
479      * BINARY_ATTRIBUTES, UNARY_ATTRIBUTES and EMPTY_NOMINAL_ATTRIBUTES. Enabling
480      * BINARY_ATTRIBUTES also enables UNARY_ATTRIBUTES and
481      * EMPTY_NOMINAL_ATTRIBUTES. Enabling UNARY_ATTRIBUTES also enables
482      * EMPTY_NOMINAL_ATTRIBUTES. But NOMINAL_CLASS only enables BINARY_CLASS,
483      * since normal schemes in Weka don't work with datasets that have only 1
484      * class label (or none).
485      * 
486      * @param c
487      *          the capability to enable
488      */
489     public void enable(Capability c) {
490         // attributes
491         if (c == Capability.NOMINAL_ATTRIBUTES) {
492             enable(Capability.BINARY_ATTRIBUTES);
493         } else if (c == Capability.BINARY_ATTRIBUTES) {
494             enable(Capability.UNARY_ATTRIBUTES);
495         } else if (c == Capability.UNARY_ATTRIBUTES) {
496             enable(Capability.EMPTY_NOMINAL_ATTRIBUTES);
497         }
498         // class
499         else if (c == Capability.NOMINAL_CLASS) {
500             enable(Capability.BINARY_CLASS);
501         }
502 
503         m_Capabilities.add(c);
504     }

506     /**
507      * enables the dependency flag for the given capability Enabling
508      * NOMINAL_ATTRIBUTES also enables BINARY_ATTRIBUTES, UNARY_ATTRIBUTES and
509      * EMPTY_NOMINAL_ATTRIBUTES. Enabling BINARY_ATTRIBUTES also enables
510      * UNARY_ATTRIBUTES and EMPTY_NOMINAL_ATTRIBUTES. Enabling UNARY_ATTRIBUTES
511      * also enables EMPTY_NOMINAL_ATTRIBUTES. But NOMINAL_CLASS only enables
512      * BINARY_CLASS, since normal schemes in Weka don't work with datasets that
513      * have only 1 class label (or none).
514      * 
515      * @param c
516      *          the capability to enable the dependency flag for
517      */
518     public void enableDependency(Capability c) {
519         // attributes
520         if (c == Capability.NOMINAL_ATTRIBUTES) {
521             enableDependency(Capability.BINARY_ATTRIBUTES);
522         } else if (c == Capability.BINARY_ATTRIBUTES) {
523             enableDependency(Capability.UNARY_ATTRIBUTES);
524         } else if (c == Capability.UNARY_ATTRIBUTES) {
525             enableDependency(Capability.EMPTY_NOMINAL_ATTRIBUTES);
526         }
527         // class
528         else if (c == Capability.NOMINAL_CLASS) {
529             enableDependency(Capability.BINARY_CLASS);
530         }
531 
532         m_Dependencies.add(c);
533     }

865     /**
866      * Test the given attribute, whether it can be processed by the handler, given
867      * its capabilities.
868      * 
869      * @param att
870      *          the attribute to test
871      * @param isClass
872      *          whether this attribute is the class attribute
873      * @return true if all the tests succeeded
874      * @see #m_AttributeTest
875      */
876     public boolean test(Attribute att, boolean isClass) {
877         boolean result;
878         Capability cap;
879         Capability capBinary;
880         Capability capUnary;
881         Capability capEmpty;
882         String errorStr;
883 
884         result = true;
885 
886         // shall we test the data?
887         if (!m_AttributeTest)
888             return result;
889 
890         // for exception
891         if (isClass)
892             errorStr = "class";
893         else
894             errorStr = "attributes";
895 
896         switch (att.type()) {
897         case Attribute.NOMINAL:
898             if (isClass) {
899                 cap = Capability.NOMINAL_CLASS;
900                 capBinary = Capability.BINARY_CLASS;
901                 capUnary = Capability.UNARY_CLASS;
902                 capEmpty = Capability.EMPTY_NOMINAL_CLASS;
903             } else {
904                 cap = Capability.NOMINAL_ATTRIBUTES;
905                 capBinary = Capability.BINARY_ATTRIBUTES;
906                 capUnary = Capability.UNARY_ATTRIBUTES;
907                 capEmpty = Capability.EMPTY_NOMINAL_ATTRIBUTES;
908             }
909 
910             if (handles(cap) && (att.numValues() > 2))
911                 break;
912             else if (handles(capBinary) && (att.numValues() == 2))
913                 break;
914             else if (handles(capUnary) && (att.numValues() == 1))
915                 break;
916             else if (handles(capEmpty) && (att.numValues() == 0))
917                 break;
918 
919             if (att.numValues() == 0) {
920                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle empty nominal " + errorStr
921                     + "!"));
922                 result = false;
923             }
924             if (att.numValues() == 1) {
925                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle unary " + errorStr + "!"));
926                 result = false;
927             } else if (att.numValues() == 2) {
928                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle binary " + errorStr + "!"));
929                 result = false;
930             } else {
931                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle multi-valued nominal "
932                     + errorStr + "!"));
933                 result = false;
934             }
935             break;
936 
937         case Attribute.NUMERIC:
938             if (isClass)
939                 cap = Capability.NUMERIC_CLASS;
940             else
941                 cap = Capability.NUMERIC_ATTRIBUTES;
942 
943             if (!handles(cap)) {
944                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle numeric " + errorStr + "!"));
945                 result = false;
946             }
947             break;
948 
949         case Attribute.DATE:
950             if (isClass)
951                 cap = Capability.DATE_CLASS;
952             else
953                 cap = Capability.DATE_ATTRIBUTES;
954 
955             if (!handles(cap)) {
956                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle date " + errorStr + "!"));
957                 result = false;
958             }
959             break;
960 
961         case Attribute.STRING:
962             if (isClass)
963                 cap = Capability.STRING_CLASS;
964             else
965                 cap = Capability.STRING_ATTRIBUTES;
966 
967             if (!handles(cap)) {
968                 m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle string " + errorStr + "!"));
969                 result = false;
970             }
971             break;
972 
973         case Attribute.RELATIONAL:
974             if (isClass)
975                 cap = Capability.RELATIONAL_CLASS;
976             else
977                 cap = Capability.RELATIONAL_ATTRIBUTES;
978 
979             if (!handles(cap)) {
980                 m_FailReason = new UnsupportedAttributeTypeException(
981                     createMessage("Cannot handle relational " + errorStr + "!"));
982                 result = false;
983             }
984             // attributes in the relation of this attribute must be tested
985             // separately with a different Capabilites object
986             break;
987 
988         default:
989             m_FailReason = new UnsupportedAttributeTypeException(createMessage("Cannot handle unknown attribute type '"
990                 + att.type() + "'!"));
991             result = false;
992         }
993 
994         return result;
995     }

1013    /**
1014     * Tests a certain range of attributes of the given data, whether it can be
1015     * processed by the handler, given its capabilities. Classifiers implementing
1016     * the <code>MultiInstanceCapabilitiesHandler</code> interface are checked
1017     * automatically for their multi-instance Capabilities (if no bags, then only
1018     * the bag-structure, otherwise only the first bag).
1019     * 
1020     * @param data
1021     *          the data to test
1022     * @param fromIndex
1023     *          the range of attributes - start (incl.)
1024     * @param toIndex
1025     *          the range of attributes - end (incl.)
1026     * @return true if all the tests succeeded
1027     * @see MultiInstanceCapabilitiesHandler
1028     * @see #m_InstancesTest
1029     * @see #m_MissingValuesTest
1030     * @see #m_MissingClassValuesTest
1031     * @see #m_MinimumNumberInstancesTest
1032     */
1033    public boolean test(Instances data, int fromIndex, int toIndex) {
1034        int i;
1035        int n;
1036        int m;
1037        Attribute att;
1038        Instance inst;
1039        boolean testClass;
1040        Capabilities cap;
1041        boolean missing;
1042        Iterator iter;
1043
1044        // shall we test the data?
1045        if (!m_InstancesTest)
1046            return true;
1047
1048        // no Capabilities? -> warning
1049        if ((m_Capabilities.size() == 0) || ((m_Capabilities.size() == 1) && handles(Capability.NO_CLASS)))
1050            System.err.println(createMessage("No capabilities set!"));
1051
1052        // any attributes?
1053        if (toIndex - fromIndex < 0) {
1054            m_FailReason = new WekaException(createMessage("No attributes!"));
1055            return false;
1056        }
1057
1058        // do wee need to test the class attribute, i.e., is the class attribute
1059        // within the range of attributes?
1060        testClass = (data.classIndex() > -1) && (data.classIndex() >= fromIndex) && (data.classIndex() <= toIndex);
1061
1062        // attributes
1063        for (i = fromIndex; i <= toIndex; i++) {
1064            att = data.attribute(i);
1065
1066            // class is handled separately
1067            if (i == data.classIndex())
1068                continue;
1069
1070            // check attribute types
1071            if (!test(att))
1072                return false;
1073        }
1074
1075        // class
1076        if (!handles(Capability.NO_CLASS) && (data.classIndex() == -1)) {
1077            m_FailReason = new UnassignedClassException(createMessage("Class attribute not set!"));
1078            return false;
1079        }
1080
1081        // special case: no class attribute can be handled
1082        if (handles(Capability.NO_CLASS) && (data.classIndex() > -1)) {
1083            cap = getClassCapabilities();
1084            cap.disable(Capability.NO_CLASS);
1085            iter = cap.capabilities();
1086            if (!iter.hasNext()) {
1087                m_FailReason = new WekaException(createMessage("Cannot handle any class attribute!"));
1088                return false;
1089            }
1090        }
1091
1092        if (testClass && !handles(Capability.NO_CLASS)) {
1093            att = data.classAttribute();
1094            if (!test(att, true))
1095                return false;
1096
1097            // special handling of RELATIONAL class
1098            // TODO: store additional Capabilities for this case
1099
1100            // missing class labels
1101            if (m_MissingClassValuesTest) {
1102                if (!handles(Capability.MISSING_CLASS_VALUES)) {
1103                    for (i = 0; i < data.numInstances(); i++) {
1104                        if (data.instance(i).classIsMissing()) {
1105                            m_FailReason = new WekaException(createMessage("Cannot handle missing class values!"));
1106                            return false;
1107                        }
1108                    }
1109                } else {
1110                    if (m_MinimumNumberInstancesTest) {
1111                        int hasClass = 0;
1112
1113                        for (i = 0; i < data.numInstances(); i++) {
1114                            if (!data.instance(i).classIsMissing())
1115                                hasClass++;
1116                        }
1117
1118                        // not enough instances with class labels?
1119                        if (hasClass < getMinimumNumberInstances()) {
1120                            m_FailReason = new WekaException(
1121                                createMessage("Not enough training instances with class labels (required: "
1122                                    + getMinimumNumberInstances() + ", provided: " + hasClass + ")!"));
1123                            return false;
1124                        }
1125                    }
1126                }
1127            }
1128        }
1129
1130        // missing values
1131        if (m_MissingValuesTest) {
1132            if (!handles(Capability.MISSING_VALUES)) {
1133                missing = false;
1134                for (i = 0; i < data.numInstances(); i++) {
1135                    inst = data.instance(i);
1136
1137                    if (inst instanceof SparseInstance) {
1138                        for (m = 0; m < inst.numValues(); m++) {
1139                            n = inst.index(m);
1140
1141                            // out of scope?
1142                            if (n < fromIndex)
1143                                continue;
1144                            if (n > toIndex)
1145                                break;
1146
1147                            // skip class
1148                            if (n == inst.classIndex())
1149                                continue;
1150
1151                            if (inst.isMissing(n)) {
1152                                missing = true;
1153                                break;
1154                            }
1155                        }
1156                    } else {
1157                        for (n = fromIndex; n <= toIndex; n++) {
1158                            // skip class
1159                            if (n == inst.classIndex())
1160                                continue;
1161
1162                            if (inst.isMissing(n)) {
1163                                missing = true;
1164                                break;
1165                            }
1166                        }
1167                    }
1168
1169                    if (missing) {
1170                        m_FailReason = new NoSupportForMissingValuesException(createMessage("Cannot handle missing values!"));
1171                        return false;
1172                    }
1173                }
1174            }
1175        }
1176
1177        // instances
1178        if (m_MinimumNumberInstancesTest) {
1179            if (data.numInstances() < getMinimumNumberInstances()) {
1180                m_FailReason = new WekaException(createMessage("Not enough training instances (required: "
1181                    + getMinimumNumberInstances() + ", provided: " + data.numInstances() + ")!"));
1182                return false;
1183            }
1184        }
1185
1186        // Multi-Instance? -> check structure (regardless of attribute range!)
1187        if (handles(Capability.ONLY_MULTIINSTANCE)) {
1188            // number of attributes?
1189            if (data.numAttributes() != 3) {
1190                m_FailReason = new WekaException(
1191                    createMessage("Incorrect Multi-Instance format, must be 'bag-id, bag, class'!"));
1192                return false;
1193            }
1194
1195            // type of attributes and position of class?
1196            if (!data.attribute(0).isNominal() || !data.attribute(1).isRelationValued()
1197                || (data.classIndex() != data.numAttributes() - 1)) {
1198                m_FailReason = new WekaException(
1199                    createMessage("Incorrect Multi-Instance format, must be 'NOMINAL att, RELATIONAL att, CLASS att'!"));
1200                return false;
1201            }
1202
1203            // check data immediately
1204            if (getOwner() instanceof MultiInstanceCapabilitiesHandler) {
1205                MultiInstanceCapabilitiesHandler handler = (MultiInstanceCapabilitiesHandler) getOwner();
1206                cap = handler.getMultiInstanceCapabilities();
1207                boolean result;
1208                if (data.numInstances() > 0)
1209                    result = cap.test(data.attribute(1).relation(0));
1210                else
1211                    result = cap.test(data.attribute(1).relation());
1212
1213                if (!result) {
1214                    m_FailReason = cap.m_FailReason;
1215                    return false;
1216                }
1217            }
1218        }
1219
1220        // passed all tests!
1221        return true;
1222    }

1339     * @return the generated source code
1340     */
1341    public String toSource(String objectname, int indent) {
1342        StringBuffer result;
1343        String capsName;
1344        String capName;
1345        String indentStr;
1346        int i;
1347
1348        result = new StringBuffer();
1349
1350        capsName = Capabilities.class.getName();
1351        capName = Capabilities.Capability.class.getName().replaceAll("\\$", ".");
1352
1353        indentStr = "";
1354        for (i = 0; i < indent; i++)
1355            indentStr += " ";
1356

1358        result.append(indentStr + capsName + " " + objectname + " = new " + capsName + "(this);\n");
1359
1360        // capabilities
1361        result.append("\n");
1362        for (Capability cap : Capability.values()) {

1398     * 
1399     * @param data
1400     *          the data to base the capabilities on
1401     * @param multi
1402     *          if true then the structure is checked, too
1403     * @return a data-specific capabilities object
1404     * @throws Exception
1405     *           in case an error occurrs, e.g., an unknown attribute type
1406     */
1407    public static Capabilities forInstances(Instances data, boolean multi) throws Exception {
1408        Capabilities result;
1409        Capabilities multiInstance;
1410        int i;
1411        int n;
1412        int m;
1413        Instance inst;
1414        boolean missing;
1415
1416        result = new Capabilities(null);
1417
1418        // class
1419        if (data.classIndex() == -1) {
1420            result.enable(Capability.NO_CLASS);
1421        } else {
1422            switch (data.classAttribute().type()) {
1423            case Attribute.NOMINAL:
1424                if (data.classAttribute().numValues() == 1)
1425                    result.enable(Capability.UNARY_CLASS);
1426                else if (data.classAttribute().numValues() == 2)
1427                    result.enable(Capability.BINARY_CLASS);
1428                else
1429                    result.enable(Capability.NOMINAL_CLASS);
1430                break;
1431
1432            case Attribute.NUMERIC:
1433                result.enable(Capability.NUMERIC_CLASS);
1434                break;
1435
1436            case Attribute.STRING:
1437                result.enable(Capability.STRING_CLASS);
1438                break;
1439
1440            case Attribute.DATE:
1441                result.enable(Capability.DATE_CLASS);
1442                break;
1443
1444            case Attribute.RELATIONAL:
1445                result.enable(Capability.RELATIONAL_CLASS);
1446                break;
1447
1448            default:
1449                throw new UnsupportedAttributeTypeException("Unknown class attribute type '" + data.classAttribute() + "'!");
1450            }
1451
1452            // missing class values
1453            for (i = 0; i < data.numInstances(); i++) {
1454                if (data.instance(i).classIsMissing()) {
1455                    result.enable(Capability.MISSING_CLASS_VALUES);
1456                    break;
1457                }
1458            }
1459        }
1460
1461        // attributes
1462        for (i = 0; i < data.numAttributes(); i++) {
1463            // skip class
1464            if (i == data.classIndex())
1465                continue;
1466
1467            switch (data.attribute(i).type()) {
1468            case Attribute.NOMINAL:
1469                result.enable(Capability.UNARY_ATTRIBUTES);
1470                if (data.attribute(i).numValues() == 2)
1471                    result.enable(Capability.BINARY_ATTRIBUTES);
1472                else if (data.attribute(i).numValues() > 2)
1473                    result.enable(Capability.NOMINAL_ATTRIBUTES);
1474                break;
1475
1476            case Attribute.NUMERIC:
1477                result.enable(Capability.NUMERIC_ATTRIBUTES);
1478                break;
1479
1480            case Attribute.DATE:
1481                result.enable(Capability.DATE_ATTRIBUTES);
1482                break;
1483
1484            case Attribute.STRING:
1485                result.enable(Capability.STRING_ATTRIBUTES);
1486                break;
1487
1488            case Attribute.RELATIONAL:
1489                result.enable(Capability.RELATIONAL_ATTRIBUTES);
1490                break;
1491
1492            default:
1493                throw new UnsupportedAttributeTypeException("Unknown attribute type '" + data.attribute(i).type() + "'!");
1494            }
1495        }
1496
1497        // missing values
1498        missing = false;
1499        for (i = 0; i < data.numInstances(); i++) {
1500            inst = data.instance(i);
1501
1502            if (inst instanceof SparseInstance) {
1503                for (m = 0; m < inst.numValues(); m++) {
1504                    n = inst.index(m);
1505
1506                    // skip class
1507                    if (n == inst.classIndex())
1508                        continue;
1509
1510                    if (inst.isMissing(n)) {
1511                        missing = true;
1512                        break;
1513                    }
1514                }
1515            } else {
1516                for (n = 0; n < data.numAttributes(); n++) {
1517                    // skip class
1518                    if (n == inst.classIndex())
1519                        continue;
1520
1521                    if (inst.isMissing(n)) {
1522                        missing = true;
1523                        break;
1524                    }
1525                }
1526            }
1527
1528            if (missing) {
1529                result.enable(Capability.MISSING_VALUES);
1530                break;
1531            }
1532        }
1533
1534        // multi-instance data?
1535        if (multi) {
1536            if ((data.numAttributes() == 3) && (data.attribute(0).isNominal()) // bag-id
1537                && (data.attribute(1).isRelationValued()) // bag
1538                && (data.classIndex() == data.numAttributes() - 1)) {
1539                multiInstance = new Capabilities(null);
1540                multiInstance.or(result.getClassCapabilities());
1541                multiInstance.enable(Capability.NOMINAL_ATTRIBUTES);
1542                multiInstance.enable(Capability.RELATIONAL_ATTRIBUTES);
1543                multiInstance.enable(Capability.ONLY_MULTIINSTANCE);
1544                result.assign(multiInstance);
1545            }
1546        }
1547
1548        return result;
1549    }

1551    /**
1552     * loads the given dataset and prints the Capabilities necessary to process
1553     * it.
1554     * <p/>
1555     * 
1556     * Valid parameters:
1557     * <p/>
1558     * 
1559     * -file filename <br/>
1560     * the file to load
1561     * 
1562     * -c index the explicit index of the class attribute (default: none)
1563     * 
1564     * @param args
1565     *          the commandline arguments
1566     * @throws Exception
1567     *           if something goes wrong
1568     */
1569    public static void main(String[] args) throws Exception {
1570        String tmpStr;
1571        String filename;
1572        DataSource source;
1573        Instances data;
1574        int classIndex;
1575        Capabilities cap;
1576        Iterator iter;
1577
1578        if (args.length == 0) {
1579            System.out.println("\nUsage: " + Capabilities.class.getName() + " -file <dataset> [-c <class index>]\n");
1580            return;
1581        }
1582
1583        // get parameters
1584        tmpStr = Utils.getOption("file", args);
1585        if (tmpStr.length() == 0)
1586            throw new Exception("No file provided with option '-file'!");
1587        else
1588            filename = tmpStr;
1589
1590        tmpStr = Utils.getOption("c", args);
1591        if (tmpStr.length() != 0) {
1592            if (tmpStr.equals("first"))
1593                classIndex = 0;
1594            else if (tmpStr.equals("last"))
1595                classIndex = -2; // last
1596            else
1597                classIndex = Integer.parseInt(tmpStr) - 1;
1598        } else {
1599            classIndex = -3; // not set
1600        }
1601
1602        // load data
1603        source = new DataSource(filename);
1604        if (classIndex == -3)
1605            data = source.getDataSet();
1606        else if (classIndex == -2)
1607            data = source.getDataSet(source.getStructure().numAttributes() - 1);
1608        else
1609            data = source.getDataSet(classIndex);
1610
1611        // determine and print capabilities
1612        cap = forInstances(data);
1613        System.out.println("File: " + filename);
1614        System.out.println("Class index: " + ((data.classIndex() == -1) ? "not set" : "" + (data.classIndex() + 1)));
1615        System.out.println("Capabilities:");
1616        iter = cap.capabilities();
1617        while (iter.hasNext())
1618            System.out.println("- " + iter.next());
1619    }
