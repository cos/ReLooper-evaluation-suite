25  /**
26   * This class locates and records the indices of String attributes, 
27   * recursively in case of Relational attributes. The indices are normally
28   * used for copying the Strings from one Instances object to another.
29   * 
30   * @author fracpete (fracpete at waikato dot ac dot nz)
31   * @version $Revision: 5953 $
32   * @see Attribute#STRING
33   * @see Attribute#RELATIONAL
34   */
35  public class StringLocator
36    extends AttributeLocator {
37    
38    /** for serialization */
39    private static final long serialVersionUID = 7805522230268783972L;
40  
41    /**
42     * initializes the StringLocator with the given data
43     * 
44     * @param data    the data to work on
45     */
46    public StringLocator(Instances data) {
47      super(data, Attribute.STRING);
48    }
49    
50    /**
51     * Initializes the StringLocator with the given data. 
52     * Checks only the given range.
53     * 
54     * @param data    the data to work on
55     * @param fromIndex   the first index to inspect (including)
56     * @param toIndex the last index to check (including)
57     */
58    public StringLocator(Instances data, int fromIndex, int toIndex) {
59      super(data, Attribute.STRING, fromIndex, toIndex);
60    }
61    
62    /**
63     * Initializes the AttributeLocator with the given data.
64     * Checks only the specified attribute indices.
65     * 
66     * @param data    the data to work on
67     * @param indices the attribute indices to check
68     */
69    public StringLocator(Instances data, int[] indices) {
70      super(data, Attribute.STRING, indices);
71    }
72  
73    /**
74     * Copies string values contained in the instance copied to a new
75     * dataset. The Instance must already be assigned to a dataset. This
76     * dataset and the destination dataset must have the same structure.
77     *
78     * @param inst        the Instance containing the string values to copy.
79     * @param destDataset     the destination set of Instances
80     * @param strAtts         an AttributeLocator containing the indices of 
81     *                any string attributes in the dataset.  
82     */
83    public static void copyStringValues(Instance inst, Instances destDataset, 
84                                 AttributeLocator strAtts) {
85  
86      if (inst.dataset() == null) {
87        throw new IllegalArgumentException("Instance has no dataset assigned!!");
88      } else if (inst.dataset().numAttributes() != destDataset.numAttributes()) {
89        throw new IllegalArgumentException("Src and Dest differ in # of attributes!!");
90      } 
91      copyStringValues(inst, true, inst.dataset(), strAtts,
92                       destDataset, strAtts);
93    }
94  
95    /**
96     * Takes string values referenced by an Instance and copies them from a
97     * source dataset to a destination dataset. The instance references are
98     * updated to be valid for the destination dataset. The instance may have the 
99     * structure (i.e. number and attribute position) of either dataset (this
100    * affects where references are obtained from). Only works if the number
101    * of string attributes is the same in both indices (implicitly these string
102    * attributes should be semantically same but just with shifted positions).
103    *
104    * @param instance        the instance containing references to strings 
105    *                in the source dataset that will have references 
106    *                updated to be valid for the destination dataset.
107    * @param instSrcCompat   true if the instance structure is the same as 
108    *                the source, or false if it is the same as the 
109    *                destination (i.e. which of the string attribute 
110    *                indices contains the correct locations for this 
111    *                instance).
112    * @param srcDataset      the dataset for which the current instance 
113    *                string references are valid (after any position 
114    *                mapping if needed)
115    * @param srcLoc      an AttributeLocator containing the indices of 
116    *                string attributes in the source datset.
117    * @param destDataset     the dataset for which the current instance 
118    *                string references need to be inserted (after 
119    *                any position mapping if needed)
120    * @param destLoc     an AttributeLocator containing the indices of 
121    *                string attributes in the destination datset.
122    */
123   public static void copyStringValues(Instance instance, boolean instSrcCompat,
124                                   Instances srcDataset, AttributeLocator srcLoc,
125                                   Instances destDataset, AttributeLocator destLoc) {
126     if (srcDataset == destDataset)
127       return;
128     
129     if (srcLoc.getAttributeIndices().length != destLoc.getAttributeIndices().length)
130       throw new IllegalArgumentException("Src and Dest string indices differ in length!!");
131 
132     if (srcLoc.getLocatorIndices().length != destLoc.getLocatorIndices().length)
133       throw new IllegalArgumentException("Src and Dest locator indices differ in length!!");
134 
135     for (int i = 0; i < srcLoc.getAttributeIndices().length; i++) {
136       int instIndex  = instSrcCompat 
137                   ? srcLoc.getActualIndex(srcLoc.getAttributeIndices()[i]) 
138                   : destLoc.getActualIndex(destLoc.getAttributeIndices()[i]);
139       Attribute src  = srcDataset.attribute(srcLoc.getActualIndex(srcLoc.getAttributeIndices()[i]));
140       Attribute dest = destDataset.attribute(destLoc.getActualIndex(destLoc.getAttributeIndices()[i]));
141       if (!instance.isMissing(instIndex)) {
142         int valIndex = dest.addStringValue(src, (int)instance.value(instIndex));
143         instance.setValue(instIndex, (double)valIndex);
144       }
145     }
146     
147     // recurse if necessary
148     int[] srcIndices  = srcLoc.getLocatorIndices();
149     int[] destIndices = destLoc.getLocatorIndices();
150     for (int i = 0; i < srcIndices.length; i++) {
151       int index = instSrcCompat
152                  ? srcLoc.getActualIndex(srcIndices[i])
153                  : destLoc.getActualIndex(destIndices[i]);
154       if (instance.isMissing(index))
155         continue;
156       Instances rel = instSrcCompat
157                  ? instance.relationalValue(index)
158                  : instance.relationalValue(index);
159       AttributeLocator srcStrAttsNew = srcLoc.getLocator(srcIndices[i]);
160       Instances srcDatasetNew = srcStrAttsNew.getData();
161       AttributeLocator destStrAttsNew = destLoc.getLocator(destIndices[i]);
162       Instances destDatasetNew = destStrAttsNew.getData();
163       for (int n = 0; n < rel.numInstances(); n++) {
164         copyStringValues(rel.instance(n), instSrcCompat, srcDatasetNew, srcStrAttsNew, destDatasetNew, destStrAttsNew);
165       }
166     }
167   }
168   
169   /**
170    * Returns the revision string.
171    * 
172    * @return        the revision
173    */
174   public String getRevision() {
175     return RevisionUtils.extract("$Revision: 5953 $");
176   }
177 }
