22  package weka.core;
23  
24  import java.util.Enumeration;
25  import java.util.Vector;
26  
27  /**
28   * Generates Javadoc comments from the TechnicalInformationHandler's data.
29   * Update the BibTex references and the plaintext techincal information. 
30   * <p/>
31   * 
32   <!-- options-start -->
33   * Valid options are: <p/>
34   * 
35   * <pre> -W &lt;classname&gt;
36   *  The class to load.</pre>
37   * 
38   * <pre> -nostars
39   *  Suppresses the '*' in the Javadoc.</pre>
40   * 
41   * <pre> -dir &lt;dir&gt;
42   *  The directory above the package hierarchy of the class.</pre>
43   * 
44   * <pre> -silent
45   *  Suppresses printing in the console.</pre>
46   * 
47   * <pre> -noprolog
48   *  Suppresses the 'BibTex:' prolog in the Javadoc.</pre>
49   * 
50   <!-- options-end -->
51   * 
52   * @author  fracpete (fracpete at waikato dot ac dot nz)
53   * @version $Revision: 5953 $
54   * @see #PLAINTEXT_STARTTAG
55   * @see #PLAINTEXT_ENDTAG
56   * @see #BIBTEX_STARTTAG
57   * @see #BIBTEX_ENDTAG
58   */
59  public class TechnicalInformationHandlerJavadoc 
60    extends Javadoc {
61    
62    /** the start comment tag for inserting the generated BibTex */
63    public final static String PLAINTEXT_STARTTAG = "<!-- technical-plaintext-start -->";
64    
65    /** the end comment tag for inserting the generated BibTex */
66    public final static String PLAINTEXT_ENDTAG = "<!-- technical-plaintext-end -->";
67    
68    /** the start comment tag for inserting the generated BibTex */
69    public final static String BIBTEX_STARTTAG = "<!-- technical-bibtex-start -->";
70    
71    /** the end comment tag for inserting the generated BibTex */
72    public final static String BIBTEX_ENDTAG = "<!-- technical-bibtex-end -->";
73    
74    /** whether to include the "Valid options..." prolog in the Javadoc */
75    protected boolean m_Prolog = true;
76    
77    /**
78     * default constructor 
79     */
80    public TechnicalInformationHandlerJavadoc() {
81      super();
82      
83      m_StartTag    = new String[2];
84      m_EndTag      = new String[2];
85      m_StartTag[0] = PLAINTEXT_STARTTAG;
86      m_EndTag[0]   = PLAINTEXT_ENDTAG;
87      m_StartTag[1] = BIBTEX_STARTTAG;
88      m_EndTag[1]   = BIBTEX_ENDTAG;
89    }
90    
91    /**
92     * Returns an enumeration describing the available options.
93     *
94     * @return an enumeration of all the available options.
95     */
96    public Enumeration listOptions() {
97      Vector<Option>        result;
98      Enumeration   en;
99      
100     result = new Vector<Option>();
101     
102     en = super.listOptions();
103     while (en.hasMoreElements())
104       result.addElement((Option)en.nextElement());
105 
106     result.addElement(new Option(
107         "\tSuppresses the 'BibTex:' prolog in the Javadoc.",
108         "noprolog", 0, "-noprolog"));
109     
110     return result.elements();
111   }
112   
113   /**
114    * Parses a given list of options. 
115    *
116    * @param options the list of options as an array of strings
117    * @throws Exception if an option is not supported
118    */
119   public void setOptions(String[] options) throws Exception {
120     super.setOptions(options);
121 
122     setProlog(!Utils.getFlag("noprolog", options));
123   }
124   
125   /**
126    * Gets the current settings of this object.
127    *
128    * @return an array of strings suitable for passing to setOptions
129    */
130   public String[] getOptions() {
131     Vector<String>        result;
132     String[]      options;
133     int           i;
134     
135     result  = new Vector<String>();
136     
137     options = super.getOptions();
138     for (i = 0; i < options.length; i++)
139       result.add(options[i]);
140 
141     if (!getProlog())
142       result.add("-noprolog");
143     
144     return (String[]) result.toArray(new String[result.size()]);
145   }
146   
147   /**
148    * sets whether to add the "Valid options are..." prolog
149    * 
150    * @param value   true if the prolog is to be used
151    */
152   public void setProlog(boolean value) {
153     m_Prolog = value;
154   }
155   
156   /**
157    * whether "Valid options are..." prolog is included in the Javadoc
158    * 
159    * @return        whether the prolog is currently used
160    */
161   public boolean getProlog() {
162     return m_Prolog;
163   }
164   
165   /**
166    * generates and returns the Javadoc for the specified start/end tag pair.
167    * 
168    * @param index   the index in the start/end tag array
169    * @return        the generated Javadoc
170    * @throws Exception  in case the generation fails
171    */
172   protected String generateJavadoc(int index) throws Exception {
173     String          result;
174     TechnicalInformationHandler handler;
175     
176     result = "";
177     
178     if (!canInstantiateClass())
179       return result;
180     
181     if (!ClassDiscovery.hasInterface(TechnicalInformationHandler.class, getInstance().getClass()))
182     throw new Exception("Class '" + getClassname() + "' is not a TechnicalInformationHandler!");
183 
184     handler = (TechnicalInformationHandler) getInstance();
185     
186     switch (index) {
187       case 0:  // plain text
188     result = toHTML(handler.getTechnicalInformation().toString()) + "\n";
189     break;
190     
191       case 1:  // bibtex
192     // prolog?
193     if (getProlog())
194       result = "BibTeX:\n";
195     result += "<pre>\n";
196     result += toHTML(handler.getTechnicalInformation().toBibTex()).replaceAll("<br/>", "") + "\n";
197     result += "</pre>\n<p/>\n";
198     break;
199     }
200     
201     // stars?
202     if (getUseStars()) 
203       result = indent(result, 1, "* ");
204     
205     return result;
206   }
207   
208   /**
209    * Returns the revision string.
210    * 
211    * @return        the revision
212    */
213   public String getRevision() {
214     return RevisionUtils.extract("$Revision: 5953 $");
215   }
216   
217   /**
218    * Parses the given commandline parameters and generates the Javadoc.
219    * 
220    * @param args    the commandline parameters for the object
221    */
222   public static void main(String[] args) {
223     runJavadoc(new TechnicalInformationHandlerJavadoc(), args);
224   }
225 }
