22  package weka.core.neighboursearch.covertrees;
23  
24  import weka.core.RevisionHandler;
25  import weka.core.RevisionUtils;
26  
27  import java.io.Serializable;
28  import java.util.ArrayList;
29  import java.util.Collection;
30  import java.util.List;
31  
32  /**
33   * Class implementing a stack.
34   * 
35   * @param <T> The type of elements to be stored in 
36   * the stack.
37   * @author Alina Beygelzimer (original C++ code)
38   * @author Sham Kakade (original C++ code)
39   * @author John Langford (original C++ code)
40   * @author Ashraf M. Kibriya (amk14[at-the-rate]cs[dot]waikato[dot]ac[dot]nz) (Java port)
41   * @version $Revision: 5953 $
42   */
43  public class Stack<T>
44    implements Serializable, RevisionHandler {
45  
46    /** for serialization. */
47    private static final long serialVersionUID = 5604056321825539264L;
48    
49    /** The number of elements in the stack. */
50    public int length;
51  
52    /** The elements inside the stack. */
53    public ArrayList<T> elements;
54    
55    /** Constructor. */
56    public Stack() { 
57      length=0; 
58      elements = new ArrayList<T>();
59    }
60  
61    /**
62     * Constructor. 
63     * @param capacity The initial capacity of the stack.
64     */
65    public Stack(int capacity) { 
66      length=0; 
67      elements = new ArrayList<T>(capacity);
68    }
69    
70    /**
71     * Returns the last element in the stack.
72     * @return The last element.
73     */
74    public T last() { 
75      return elements.get(length-1);
76    }
77    
78    /**
79     * Returns the ith element in the stack.
80     * @param i The index of the element to return.
81     * @return The ith element. 
82     */
83    public T element(int i) { 
84      return elements.get(i);
85    }
86    
87    /**
88     * Sets the ith element in the stack. 
89     * @param i The index at which the element is
90     * to be inserted. 
91     * @param e The element to insert. 
92     */
93    public void set(int i, T e) {
94      elements.set(i, e);
95    }
96    
97    /**
98     * Returns a sublist of the elements in the 
99     * stack. 
100    * @param beginIdx The start index of the 
101    * sublist. 
102    * @param uptoLength The length of the 
103    * sublist.
104    * @return The sublist starting from 
105    * beginIdx and of length uptoLength.
106    */
107   public List subList(int beginIdx, int uptoLength) {
108     return elements.subList(beginIdx, uptoLength);
109   }
110   
111   /** Removes all the elements from the stack. */
112   public void clear() {
113     elements.clear();
114     length=0;
115   }
116   
117   /** 
118    * Adds all the given elements in the stack.
119    * @param c The collection of elements to add
120    * in the stack.
121    */ 
122   public void addAll(Collection<? extends T> c) {
123     elements.addAll(c);
124     length = c.size();
125   }
126   
127   /**
128    * Replace all elements in the stack with 
129    * the elements of another given stack.
130    * It first removes all the elements 
131    * currently in the stack, and then adds all
132    * the elements of the provided stack. 
133    * @param s The stack whose elements should 
134    * be put in this stack.
135    */
136   public void replaceAllBy(Stack<T> s) {
137     elements.clear();
138     elements.addAll(s.elements);
139     length = elements.size();
140   }
141   
142   /** 
143    * Pops (removes) the first (last added) 
144    * element in the stack.
145    * @return The poped element. 
146    */
147   public T pop() {
148     length--;
149     return elements.remove(length);    
150   }
151   
152   /**
153    * Pushes the given element to the stack.
154    * @param new_ele The element to be pushed
155    * to the stack. 
156    */
157   public void push(T new_ele) {
158     length++;
159     elements.add(new_ele);
160   }
161   
162   /**
163    * Pushes the given element onto the given 
164    * stack. 
165    * @param v The stack onto push the element.
166    * @param new_ele The element to push.
167    */
168   public void push(Stack<T> v, T new_ele) {
169     length++; 
170     v.elements.add(new_ele);
171   }
172   
173   /**
174    * Returns the revision string.
175    * 
176    * @return        the revision
177    */
178   public String getRevision() {
179     return RevisionUtils.extract("$Revision: 5953 $");
180   }
181 }
