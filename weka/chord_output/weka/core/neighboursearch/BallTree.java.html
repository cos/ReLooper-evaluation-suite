197   /**
198    * Builds the BallTree on the supplied set of 
199    * instances/points (supplied with setInstances(Instances) 
200    * method and referenced by the m_Instances field). This
201    * method should not be called by outside classes. They
202    * should only use setInstances(Instances) method.
203    * 
204    * @throws Exception If no instances are supplied 
205    * (m_Instances is null), or if some other error in the 
206    * supplied BallTreeConstructor occurs while building 
207    * the tree.  
208    */
209   protected void buildTree() throws Exception {
210     if(m_Instances==null)
211       throw new Exception("No instances supplied yet. Have to call " +
212                           "setInstances(instances) with a set of Instances " +
213                           "first.");
214     m_InstList = new int[m_Instances.numInstances()];
215     
216     for(int i=0; i<m_InstList.length; i++) {
217       m_InstList[i] = i;
218     } //end for
219     
220     m_DistanceFunction.setInstances(m_Instances);
221     m_TreeConstructor.setInstances(m_Instances);
222     m_TreeConstructor.setInstanceList(m_InstList);
223     m_TreeConstructor.setEuclideanDistanceFunction(
224                       (EuclideanDistance)m_DistanceFunction);
225     
226     m_Root = m_TreeConstructor.buildTree();
227   }

229   /**
230    * Returns k nearest instances in the current neighbourhood to the supplied
231    * instance. &gt;k instances can be returned if there is more than one instance 
232    * at the kth boundary (i.e. if there are more than 1 instance with the 
233    * same distance as the kth nearest neighbour).
234    * 
235    * @param target  The instance to find the k nearest neighbours for.
236    * @param k       The number of nearest neighbours to find.
237    * @throws Exception  If the neighbours could not be found.
238    * @return The k nearest neighbours of the given target instance 
239    * (&gt;k nearest neighbours, if there are more instances that have same 
240    *  distance as the kth nearest neighbour).
241    */
242   public Instances kNearestNeighbours(Instance target, int k) throws Exception {
243     MyHeap heap = new MyHeap(k);
244 
245     if(m_Stats!=null)
246       m_Stats.searchStart();
247     
248     nearestNeighbours(heap, m_Root, target, k);
249     
250     if(m_Stats!=null)
251       m_Stats.searchFinish();
252 
253     Instances neighbours = new Instances(m_Instances, heap.totalSize());
254     m_Distances = new double[heap.totalSize()];
255     int [] indices = new int[heap.totalSize()];
256     int i=1; MyHeapElement h;
257     while(heap.noOfKthNearest()>0) {
258       h = heap.getKthNearest();
259       indices[indices.length-i] = h.index;
260       m_Distances[indices.length-i] = h.distance;
261       i++;
262     }
263     while(heap.size()>0) {
264       h = heap.get();
265       indices[indices.length-i] = h.index;
266       m_Distances[indices.length-i] = h.distance;
267       i++;
268     }
269     
270     m_DistanceFunction.postProcessDistances(m_Distances);
271     
272     for(i=0; i<indices.length; i++)
273       neighbours.add(m_Instances.instance(indices[i]));
274     
275     return neighbours;  // <---Check this statement
276   }

278   /** 
279    * Does NN search according to Moore's method. 
280    * Should not be used by outside classes. They should instead
281    * use kNearestNeighbours(Instance, int).
282    * P.S.: The distance returned are squared. Need to post process the 
283    * distances. 
284    * @param heap MyHeap object to store/update NNs found during the search.
285    * @param node The BallNode to do the NN search on.
286    * @param target The target instance for which the NNs are required.
287    * @param k The number of NNs to find.
288    * @throws Exception If the structure of the BallTree is not correct, 
289    * or if there is some problem putting NNs in the heap.
290    */
291   protected void nearestNeighbours(MyHeap heap, BallNode node, Instance target,
292                                    int k) throws Exception{
293     double distance = Double.NEGATIVE_INFINITY;
294 
295     if (heap.totalSize() >= k)
296       distance = m_DistanceFunction.distance(target, node.getPivot());
297 
298     // The radius is not squared so need to take sqrt before comparison
299     if (distance > -0.000001
300         && Math.sqrt(heap.peek().distance) < distance - node.getRadius()) {
301       return;
302     } else if (node.m_Left != null && node.m_Right != null) { // if node is not
303                                                               // a leaf
304       if (m_TreeStats != null) {
305         m_TreeStats.incrIntNodeCount();
306       }
307       double leftPivotDist = Math.sqrt(m_DistanceFunction.distance(target,
308           node.m_Left.getPivot(), Double.POSITIVE_INFINITY));
309       double rightPivotDist = Math.sqrt(m_DistanceFunction.distance(target,
310           node.m_Right.getPivot(), Double.POSITIVE_INFINITY));
311       double leftBallDist = leftPivotDist - node.m_Left.getRadius();
312       double rightBallDist = rightPivotDist - node.m_Right.getRadius();
313       // if target is inside both balls then see which center is closer
314       if (leftBallDist < 0 && rightBallDist < 0) {
315         if (leftPivotDist < rightPivotDist) {
316           nearestNeighbours(heap, node.m_Left, target, k);
317           nearestNeighbours(heap, node.m_Right, target, k);
318         } else {
319           nearestNeighbours(heap, node.m_Right, target, k);
320           nearestNeighbours(heap, node.m_Left, target, k);
321         }
322       }
323       // else see which ball is closer (if dist < 0 target is inside a ball, and
324       // hence the ball is closer).
325       else {
326         if (leftBallDist < rightBallDist) {
327           nearestNeighbours(heap, node.m_Left, target, k);
328           nearestNeighbours(heap, node.m_Right, target, k);
329         } else {
330           nearestNeighbours(heap, node.m_Right, target, k);
331           nearestNeighbours(heap, node.m_Left, target, k);
332         }
333       }
334     } else if (node.m_Left != null || node.m_Right != null) { // invalid leaves
335                                                               // assignment
336       throw new Exception("Error: Only one leaf of the built ball tree is " + 
337                           "assigned. Please check code.");
338     } else if (node.m_Left == null && node.m_Right == null) { // if node is a
339                                                               // leaf
340       if (m_TreeStats != null) {
341         m_TreeStats.updatePointCount(node.numInstances());
342         m_TreeStats.incrLeafCount();
343       }
344       for (int i = node.m_Start; i <= node.m_End; i++) {
345         if (target == m_Instances.instance(m_InstList[i])) //for hold-one-out cross-validation
346           continue;
347         if (heap.totalSize() < k) {
348           distance = m_DistanceFunction.distance(target, m_Instances
349               .instance(m_InstList[i]), Double.POSITIVE_INFINITY, m_Stats);
350           heap.put(m_InstList[i], distance);
351         } else {
352           MyHeapElement head = heap.peek();
353           distance = m_DistanceFunction.distance(target, 
354               m_Instances.instance(m_InstList[i]), head.distance, m_Stats);
355           if (distance < head.distance) {
356             heap.putBySubstitute(m_InstList[i], distance);
357           } else if (distance == head.distance) {
358             heap.putKthNearest(m_InstList[i], distance);
359           }
360         }//end else(heap.totalSize())
361       }
362     }//end else if node is a leaf
363   }

365   /**
366    * Returns the nearest instance in the current neighbourhood to the supplied
367    * instance.
368    * 
369    * @param target  The instance to find the nearest neighbour for.
370    * @throws Exception  if the nearest neighbour could not be found.
371    * @return The nearest neighbour of the given target instance.
372    */
373   public Instance nearestNeighbour(Instance target) throws Exception {
374     return kNearestNeighbours(target, 1).instance(0);
375   }

377  /** 
378    * Returns the distances of the k nearest neighbours. The kNearestNeighbours
379    * or nearestNeighbour must always be called before calling this function. If
380    * this function is called before calling either the kNearestNeighbours or 
381    * the nearestNeighbour, then it throws an exception. If, however, any 
382    * one of the two functions is called at any point in the past, then no   
383    * exception is thrown and the distances of NN(s) from the training set for 
384    * the last supplied target instance (to either one of the nearestNeighbour 
385    * functions) is/are returned.
386    *
387    * @return        array containing the distances of the 
388    *                nearestNeighbours. The length and ordering of the 
389    *                array is the same as that of the instances returned 
390    *                by nearestNeighbour functions.
391    * @throws Exception  if called before calling kNearestNeighbours
392    *                or nearestNeighbours.
393    */
394   public double[] getDistances() throws Exception {
395     if(m_Distances==null)
396       throw new Exception("No distances available. Please call either "+
397                           "kNearestNeighbours or nearestNeighbours first.");
398     return m_Distances;    
399   }

564   /**
565    * Parses a given list of options.
566    * 
567    <!-- options-start -->
568    * Valid options are: <p/>
569    * 
570    * <pre> -C &lt;classname and options&gt;
571    *  The construction method to employ. Either TopDown or BottomUp
572    *  (default: weka.core.TopDownConstructor)</pre>
573    * 
574    <!-- options-end --> 
575    * 
576    * @param options     the list of options as an array of strings
577    * @throws Exception  if an option is not supported
578    */
579   public void setOptions(String[] options)
580     throws Exception {
581 
582     super.setOptions(options);
583 
584     String optionString = Utils.getOption('C', options);
585     if(optionString.length() != 0) {
586       String constructorSpec[] = Utils.splitOptions(optionString);
587       if(constructorSpec.length == 0) { 
588         throw new Exception("Invalid BallTreeConstructor specification string."); 
589       }
590       String className = constructorSpec[0];
591       constructorSpec[0] = "";
592 
593       setBallTreeConstructor( (BallTreeConstructor)
594                             Utils.forName( BallTreeConstructor.class, 
595                                            className, constructorSpec) );
596     }
597     else {
598       setBallTreeConstructor(new TopDownConstructor());  
599     }
600   }

602   /**
603    * Gets the current settings of KDtree.
604    * 
605    * @return        an array of strings suitable for passing to setOptions
606    */
607   public String[] getOptions() {
608     Vector<String>  result;
609     String[]        options;
610     int         i;
611     
612     result = new Vector<String>();
613     
614     options = super.getOptions();
615     for (i = 0; i < options.length; i++)
616       result.add(options[i]);
617     
618     result.add("-C");
619     result.add(
620     (m_TreeConstructor.getClass().getName() + " " +
621      Utils.joinOptions(m_TreeConstructor.getOptions())).trim());
622 
623     return result.toArray(new String[result.size()]);
624   }
