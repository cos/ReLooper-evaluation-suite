137   public String[] getOptions() {
138     Vector<String>  result;
139     String[]        options;
140     int         i;
141     
142     result = new Vector<String>();
143     
144     options = super.getOptions();
145     for (i = 0; i < options.length; i++)
146       result.add(options[i]);
147     
148     if (getSkipIdentical())
149       result.add("-S");
150 
151     return result.toArray(new String[result.size()]);
152   }

184   /** 
185    * Returns the nearest instance in the current neighbourhood to the supplied
186    * instance.
187    *  
188    * @param target  The instance to find the nearest neighbour for.
189    * @return        the nearest instance
190    * @throws Exception  if the nearest neighbour could not be found.
191    */
192   public Instance nearestNeighbour(Instance target) throws Exception {
193     return (kNearestNeighbours(target, 1)).instance(0);
194   }
195   
196   /**
197    * Returns k nearest instances in the current neighbourhood to the supplied
198    * instance.
199    *  
200    * @param target  The instance to find the k nearest neighbours for.
201    * @param kNN     The number of nearest neighbours to find.
202    * @return        the k nearest neighbors
203    * @throws Exception  if the neighbours could not be found.
204    */
205   public Instances kNearestNeighbours(Instance target, int kNN) throws Exception {
206   
207     //debug
208     boolean print=false;
209 
210     if(m_Stats!=null)
211       m_Stats.searchStart();
212  
213     MyHeap heap = new MyHeap(kNN);
214     double distance; int firstkNN=0;
215     for(int i=0; i<m_Instances.numInstances(); i++) {
216       if(target == m_Instances.instance(i)) //for hold-one-out cross-validation
217         continue;
218       if(m_Stats!=null) 
219         m_Stats.incrPointCount();
220       if(firstkNN<kNN) {
221         if(print)
222           System.out.println("K(a): "+(heap.size()+heap.noOfKthNearest()));
223         distance = m_DistanceFunction.distance(target, m_Instances.instance(i), Double.POSITIVE_INFINITY, m_Stats);
224         if(distance == 0.0 && m_SkipIdentical)
225           if(i<m_Instances.numInstances()-1)
226             continue;
227           else
228             heap.put(i, distance);
229         heap.put(i, distance);
230         firstkNN++;
231       }
232       else {
233         MyHeapElement temp = heap.peek();
234         if(print)
235           System.out.println("K(b): "+(heap.size()+heap.noOfKthNearest()));
236         distance = m_DistanceFunction.distance(target, m_Instances.instance(i), temp.distance, m_Stats);
237         if(distance == 0.0 && m_SkipIdentical)
238           continue;
239         if(distance < temp.distance) {
240           heap.putBySubstitute(i, distance);
241         }
242         else if(distance == temp.distance) {
243           heap.putKthNearest(i, distance);
244         }
245 
246       }
247     }
248     
249     Instances neighbours = new Instances(m_Instances, (heap.size()+heap.noOfKthNearest()));
250     m_Distances = new double[heap.size()+heap.noOfKthNearest()];
251     int [] indices = new int[heap.size()+heap.noOfKthNearest()];
252     int i=1; MyHeapElement h;
253     while(heap.noOfKthNearest()>0) {
254       h = heap.getKthNearest();
255       indices[indices.length-i] = h.index;
256       m_Distances[indices.length-i] = h.distance;
257       i++;
258     }
259     while(heap.size()>0) {
260       h = heap.get();
261       indices[indices.length-i] = h.index;
262       m_Distances[indices.length-i] = h.distance;
263       i++;
264     }
265     
266     m_DistanceFunction.postProcessDistances(m_Distances);
267     
268     for(int k=0; k<indices.length; k++) {
269       neighbours.add(m_Instances.instance(indices[k]));
270     }
271     
272     if(m_Stats!=null)
273       m_Stats.searchFinish();
274     
275     return neighbours;    
276   }

278   /** 
279    * Returns the distances of the k nearest neighbours. The kNearestNeighbours
280    * or nearestNeighbour must always be called before calling this function. If
281    * this function is called before calling either the kNearestNeighbours or 
282    * the nearestNeighbour, then it throws an exception. If, however, if either
283    * of the nearestNeighbour functions are called at any point in the 
284    * past then no exception is thrown and the distances of the training set from
285    * the last supplied target instance (to either one of the nearestNeighbour 
286    * functions) is/are returned.
287    *
288    * @return        array containing the distances of the 
289    *            nearestNeighbours. The length and ordering of the 
290    *            array is the same as that of the instances returned 
291    *            by nearestNeighbour functions.
292    * @throws Exception  if called before calling kNearestNeighbours
293    *                or nearestNeighbours.
294    */
295   public double[] getDistances() throws Exception {
296     if(m_Distances==null)
297       throw new Exception("No distances available. Please call either "+
298                           "kNearestNeighbours or nearestNeighbours first.");
299     return m_Distances;    
300   }

302   /** 
303    * Sets the instances comprising the current neighbourhood.
304    * 
305    * @param insts   The set of instances on which the nearest neighbour 
306    *            search is carried out. Usually this set is the 
307    *            training set. 
308    * @throws Exception  if setting of instances fails
309    */
310   public void setInstances(Instances insts) throws Exception {
311     m_Instances = insts;
312     m_DistanceFunction.setInstances(insts);
313   }
314   
315   /** 
316    * Updates the LinearNNSearch to cater for the new added instance. This 
317    * implementation only updates the ranges of the DistanceFunction class, 
318    * since our set of instances is passed by reference and should already have 
319    * the newly added instance.
320    * 
321    * @param ins     The instance to add. Usually this is the instance that 
322    *            is added to our neighbourhood i.e. the training 
323    *            instances.
324    * @throws Exception  if the given instances are null
325    */
326   public void update(Instance ins) throws Exception {
327     if(m_Instances==null)
328       throw new Exception("No instances supplied yet. Cannot update without"+
329                           "supplying a set of instances first.");
330     m_DistanceFunction.update(ins);
331   }
332   
333   /** 
334    * Adds the given instance info. This implementation updates the range
335    * datastructures of the DistanceFunction class.
336    * 
337    * @param ins     The instance to add the information of. Usually this is
338    *            the test instance supplied to update the range of 
339    *            attributes in the  distance function.
340    */
341   public void addInstanceInfo(Instance ins) {
342     if(m_Instances!=null)
343       try{ update(ins); }
344       catch(Exception ex) { ex.printStackTrace(); }
345   }
