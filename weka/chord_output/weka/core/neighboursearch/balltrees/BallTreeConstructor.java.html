22  package weka.core.neighboursearch.balltrees;
23  
24  import weka.core.DistanceFunction;
25  import weka.core.EuclideanDistance;
26  import weka.core.Instance;
27  import weka.core.Instances;
28  import weka.core.Option;
29  import weka.core.OptionHandler;
30  import weka.core.RevisionHandler;
31  import weka.core.RevisionUtils;
32  import weka.core.Utils;
33  
34  import java.io.Serializable;
35  import java.util.Enumeration;
36  import java.util.Vector;
37  
38  /**
39   * Abstract class for constructing a BallTree .
40   * 
41   * @author Ashraf M. Kibriya (amk14[at-the-rate]cs[dot]waikato[dot]ac[dot]nz)
42   * @version $Revision: 5953 $
43   */
44  public abstract class BallTreeConstructor 
45    implements OptionHandler, Serializable, RevisionHandler {
46    
47    /** The maximum number of instances allowed in a leaf. */
48    protected int m_MaxInstancesInLeaf=40;
49    
50    /** The maximum relative radius of a leaf node 
51     * (relative to the smallest ball enclosing all the 
52     * data (training) points). */
53    protected double m_MaxRelLeafRadius=0.001;
54    
55    /** Should a parent ball completely enclose the balls
56     * of its two children, or only the points inside
57     * its children. */
58    protected boolean m_FullyContainChildBalls = false;
59    
60    /** The instances on which to build the tree. */
61    protected Instances m_Instances;
62    
63    /** The distance function to use to build the tree. */
64    protected DistanceFunction m_DistanceFunction;
65    
66    /** The number of internal and leaf nodes in the 
67     * built tree. */
68    protected int m_NumNodes;
69    
70    /** The number of leaf nodes in the built tree. */
71    protected int m_NumLeaves;
72    
73    /** The depth of the built tree. */
74    protected int m_MaxDepth;
75    
76    /** The master index array. */
77    protected int[] m_InstList;
78    
79    /**
80     * Creates a new instance of BallTreeConstructor.
81     */
82    public BallTreeConstructor() {
83    }
84    
85    /**
86     *  Builds the ball tree. 
87     * @return The root node of the tree. 
88     * @throws Exception If there is problem building
89     * the tree.
90     */
91    public abstract BallNode buildTree() throws Exception;
92    
93    /**
94     * Adds an instance to the ball tree. 
95     * @param node The root node of the tree.
96     * @param inst The instance to add to the tree.
97     * @return The new master index array after 
98     * adding the instance. 
99     * @throws Exception If there is some problem adding
100    * the given instance to the tree. 
101    */
102   public abstract int[] addInstance(BallNode node, Instance inst) 
103       throws Exception;
104   
105   /**
106    * Returns the tip text for this property.
107    * 
108    * @return tip text for this property suitable for
109    * displaying in the explorer/experimenter gui.
110    */
111   public String maxInstancesInLeafTipText() {
112     return "The maximum number of instances allowed in a leaf.";
113   }
114   
115   /**
116    * Returns the maximum number of instances allowed in a leaf.
117    * @return The maximum number of instances allowed in a leaf.
118    */
119   public int getMaxInstancesInLeaf() {
120     return m_MaxInstancesInLeaf;
121   }
122   
123   /**
124    * Sets the maximum number of instances allowed in a leaf.
125    * @param num The maximum number of instances allowed in 
126    * a leaf.
127    * @throws Exception If the num is < 1. 
128    */ 
129   public void setMaxInstancesInLeaf(int num) throws Exception {
130     if(num<1)
131       throw new Exception("The maximum number of instances in a leaf must " +
132                           "be >=1.");
133     m_MaxInstancesInLeaf = num;
134   }
135 
136   /**
137    * Returns the tip text for this property.
138    * 
139    * @return tip text for this property suitable for
140    * displaying in the explorer/experimenter gui.
141    */
142   public String maxRelativeLeafRadiusTipText() {
143     return "The maximum relative radius allowed for a leaf node. " +
144             "Itis relative to the radius of the smallest ball " +
145             "enclosing all the data points (that were used to " +
146             "build the tree). This smallest ball would be the " +
147             "same as the root node's ball, if ContainChildBalls " +
148             "property is set to false (default).";
149   }
150 
151   /** Returns the maximum relative radius of a leaf node. 
152    * It is relative to the radius of the smallest ball enclosing all 
153    * the data points (that were used to build the tree). This smallest
154    * ball would be the same as the root node's ball, if 
155    * ContainChildBalls property is set to false (default).
156    * @return The maximum relative radius allowed for a leaf.
157    */
158   public double getMaxRelativeLeafRadius() {
159     return m_MaxRelLeafRadius;
160   }
161 
162   /** Sets the maximum relative radius, allowed for a leaf node. The 
163    * radius is relative to the radius of the smallest ball enclosing all 
164    * the data points (that were used to build the tree). This smallest
165    * ball would be the same as the root node's ball, if 
166    * ContainChildBalls property is set to false (default).
167    * @param radius The maximum relative radius allowed for a leaf.
168    * @throws Exception If radius is < 0.0.
169    */
170   public void setMaxRelativeLeafRadius(double radius) throws Exception {
171     if(radius < 0.0)
172       throw new Exception("The radius for the leaves should be >= 0.0");
173     m_MaxRelLeafRadius = radius;
174   }
175 
176   /**
177    * Returns the tip text for this property.
178    * 
179    * @return tip text for this property suitable for
180    * displaying in the explorer/experimenter gui.
181    */
182   public String containChildBallsTipText() {
183     return "Whether to contain fully the child balls.";
184   }
185   
186   /**
187    * Gets whether if a parent ball should completely enclose
188    * its two child balls.
189    * @return true if parent ball is to enclose its child 
190    * balls.
191    */
192   public boolean getContainChildBalls() {
193     return m_FullyContainChildBalls;
194   }
195   
196   /**
197    * Sets whether if a parent ball should completely enclose
198    * its two child balls.
199    * @param containChildBalls Should be tree if parent ball
200    * is to enclose its child balls.
201    */
202   public void setContainChildBalls(boolean containChildBalls) {
203     m_FullyContainChildBalls = containChildBalls;
204   }
205   
206   /**
207    * Sets the instances on which the tree is to be built.
208    * @param inst The instances on which to build the 
209    * ball tree.
210    */
211   public void setInstances(Instances inst) {
212     m_Instances = inst;
213   }
214   
215   /**
216    * Sets the master index array that points to 
217    * instances in m_Instances, so that only this array
218    * is manipulated, and m_Instances is left 
219    * untouched.
220    * @param instList The master index array.
221    */
222   public void setInstanceList(int[] instList) {
223     m_InstList = instList;
224   }
225   
226   /**
227    * Sets the distance function to use to build the 
228    * tree.
229    * @param func The distance function.
230    */
231   public void setEuclideanDistanceFunction(EuclideanDistance func) {
232     m_DistanceFunction = func;
233   }
234   
235   /**
236    * Returns the number of nodes (internal + leaf) 
237    * in the built tree. 
238    * @return The number of nodes in the tree. 
239    */ 
240   public int getNumNodes() {
241     return m_NumNodes;
242   }
243   
244   /**
245    * Returns the number of leaves in the built tree.
246    * @return The number of leaves in the tree.
247    */
248   public int getNumLeaves() {
249     return m_NumLeaves;
250   }
251   
252   /**
253    * Returns the depth of the built tree. 
254    * @return The depth of the tree. 
255    */
256   public int getMaxDepth() {
257     return m_MaxDepth;
258   }
259   
260   /**
261    * Returns an enumeration describing the available options.
262    * 
263    * @return        an enumeration of all the available options.
264    */
265   public Enumeration listOptions() {
266     Vector<Option> newVector = new Vector<Option>();
267     
268     newVector.addElement(new Option(
269     "\tSet maximum number of instances in a leaf node\n"
270     + "\t(default: 40)", 
271     "N", 0, "-N <value>")); 
272     
273     newVector.addElement(new Option(
274     "\tSet internal nodes' radius to the sum \n"
275     + "\tof the child balls radii. So that it \n" 
276     + "contains the child balls.", 
277     "R", 0, "-R")); 
278 
279     return newVector.elements();
280   }
281 
282   /**
283    * Parses a given list of options.
284    * 
285    * @param options     the list of options as an array of strings
286    * @throws Exception  if an option is not supported
287    */
288   public void setOptions(String[] options)
289     throws Exception {
290     
291     String optionString = Utils.getOption('N', options);
292     if(optionString.length() !=  0) {
293       setMaxInstancesInLeaf(Integer.parseInt(optionString));
294     }
295     else {
296       setMaxInstancesInLeaf(40);
297     }
298     
299     setContainChildBalls(Utils.getFlag('R', options));
300   }
301 
302   /**
303    * Gets the current settings.
304    * 
305    * @return        an array of strings suitable for passing to setOptions
306    */
307   public String[] getOptions() {
308     Vector<String>  result;
309     
310     result = new Vector<String>();
311     
312     result.add("-N");
313     result.add("" + getMaxInstancesInLeaf());
314     
315     if (getContainChildBalls())
316       result.add("-R");
317     
318     return result.toArray(new String[result.size()]);
319   }
320   
321   /**
322    * Returns the revision string.
323    * 
324    * @return        the revision
325    */
326   public String getRevision() {
327     return RevisionUtils.extract("$Revision: 5953 $");
328   }
329 }
