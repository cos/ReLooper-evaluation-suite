432   /**
433    * Splits a given point_set into near and far based on the given
434    * scale/level. All points with distance > base^max_scale would be moved
435    * to far set. In other words, all those points that are not covered by the 
436    * next child ball of a point p (ball made of the same point p but of 
437    * smaller radius at the next lower level) are removed from the supplied
438    * current point_set and put into far_set.  
439    * 
440    * @param point_set The supplied set from which all far points 
441    * would be removed.
442    * @param far_set The set in which all far points having distance
443    * > base^max_scale would be put into. 
444    * @param max_scale The given scale based on which the distances
445    * of points are judged to be far or near.   
446    */
447   protected void split(Stack<DistanceNode> point_set,
448       Stack<DistanceNode> far_set, int max_scale) {
449     int new_index = 0;
450     double fmax = dist_of_scale(max_scale);
451     for (int i = 0; i < point_set.length; i++) {
452       DistanceNode n = point_set.element(i);
453       if (n.dist.element(n.dist.length - 1).doubleValue() <= fmax) {
454         point_set.set(new_index++, point_set.element(i));
455       } else
456         far_set.push(point_set.element(i)); // point_set[i]);
457     }
458     List<DistanceNode> l = new java.util.LinkedList<DistanceNode>();
459     for (int i = 0; i < new_index; i++)
460       l.add(point_set.element(i));
461     //removing all and adding only the near points
462     point_set.clear();
463     point_set.addAll(l); // point_set.index=new_index;
464   }

466   /**
467    * Moves all the points in point_set covered by (the ball of) new_point 
468    * into new_point_set, based on the given scale/level.
469    * 
470    * @param point_set The supplied set of instances from which
471    * all points covered by new_point will be removed.
472    * @param new_point_set The set in which all points covered by
473    * new_point will be put into.
474    * @param new_point The given new point.
475    * @param max_scale The scale based on which distances are 
476    * judged (radius of cover ball is calculated).
477    */
478   protected void dist_split(Stack<DistanceNode> point_set,
479       Stack<DistanceNode> new_point_set, 
480       DistanceNode new_point, int max_scale) {
481     int new_index = 0;
482     double fmax = dist_of_scale(max_scale);
483     for (int i = 0; i < point_set.length; i++) {
484       double new_d =  Math.sqrt(m_DistanceFunction.distance(new_point.q(), 
485                point_set.element(i).q(), fmax*fmax));
486       if (new_d <= fmax) {
487         point_set.element(i).dist.push(new_d);
488         new_point_set.push(point_set.element(i));
489       } else
490         point_set.set(new_index++, point_set.element(i));
491     }
492     List<DistanceNode> l = new java.util.LinkedList<DistanceNode>();
493     for (int i = 0; i < new_index; i++)
494       l.add(point_set.element(i));
495     point_set.clear();
496     point_set.addAll(l);
497   }

499   /**
500    * Creates a cover tree recursively using batch insert method. 
501    * 
502    * @param p The index of the instance from which to create the
503    * first node. All other points will be inserted beneath this node
504    * for p.
505    * @param max_scale The current scale/level where the node is to be
506    * created (Also determines the radius of the cover balls created at 
507    * this level).
508    * @param top_scale The max scale in the whole tree.
509    * @param point_set The set of unprocessed points from which child nodes
510    * need to be created.  
511    * @param consumed_set The set of processed points from which child
512    * nodes have already been created. This would be used to find the 
513    * radius of the cover ball of p. 
514    * @return the node of cover tree created with p.
515    */
516   protected CoverTreeNode batch_insert(Integer p, int max_scale, // current
517                                                                  // scale/level
518       int top_scale, // max scale/level for this dataset
519       Stack<DistanceNode> point_set, // set of points that are nearer to p
520                                         // [will also contain returned unused
521                                         // points]
522       Stack<DistanceNode> consumed_set) // to return the set of points that have
523                                         // been used to calc. max_dist to a
524                                         // descendent
525       // Stack<Stack<DistanceNode>> stack) //may not be needed
526       {
527     if (point_set.length == 0) {
528       CoverTreeNode leaf = new_leaf(p);
529       leaf.nodeid = m_NumNodes;
530       m_NumNodes++; // incrementing node count
531       m_NumLeaves++; // incrementing leaves count
532       return leaf;
533     } else {
534       double max_dist = max_set(point_set); // O(|point_set|) the max dist
535       // in point_set to point "p".
536       int next_scale = Math.min(max_scale - 1, get_scale(max_dist));
537       if (next_scale == Integer.MIN_VALUE) { // We have points with distance
538         // 0. if max_dist is 0.
539         Stack<CoverTreeNode> children = new Stack<CoverTreeNode>();
540         CoverTreeNode leaf = new_leaf(p);
541         leaf.nodeid = m_NumNodes;
542         children.push(leaf);
543         m_NumLeaves++;
544         m_NumNodes++; // incrementing node and leaf count
545         while (point_set.length > 0) {
546           DistanceNode tmpnode = point_set.pop();
547           leaf = new_leaf(tmpnode.idx);
548           leaf.nodeid = m_NumNodes;
549           children.push(leaf);
550           m_NumLeaves++;
551           m_NumNodes++; // incrementing node and leaf count
552           consumed_set.push(tmpnode);
553         }
554         CoverTreeNode n = new_node(p); // make a new node out of p and assign
555         // it the children.
556         n.nodeid = m_NumNodes;
557         m_NumNodes++; // incrementing node count
558         n.scale = 100; // A magic number meant to be larger than all scales.
559         n.max_dist = 0; // since all points have distance 0 to p
560         n.num_children = children.length;
561         n.children = children;
562         return n;
563       } else {
564         Stack<DistanceNode> far = new Stack<DistanceNode>();
565         split(point_set, far, max_scale); // O(|point_set|)
566 
567         CoverTreeNode child = batch_insert(p, next_scale, top_scale, point_set,
568             consumed_set);
569 
570         if (point_set.length == 0) { // not creating any node in this
571           // recursive call
572           // push(stack,point_set);
573           point_set.replaceAllBy(far); // point_set=far;
574           return child;
575         } else {
576           CoverTreeNode n = new_node(p);
577           n.nodeid = m_NumNodes;
578           m_NumNodes++; // incrementing node count
579           Stack<CoverTreeNode> children = new Stack<CoverTreeNode>();
580           children.push(child);
581 
582           while (point_set.length != 0) { // O(|point_set| * num_children)
583             Stack<DistanceNode> new_point_set = new Stack<DistanceNode>();
584             Stack<DistanceNode> new_consumed_set = new Stack<DistanceNode>();
585             DistanceNode tmpnode = point_set.pop();
586             double new_dist = tmpnode.dist.last();
587             consumed_set.push(tmpnode);
588 
589             // putting points closer to new_point into new_point_set (and
590             // removing them from point_set)
591             dist_split(point_set, new_point_set, tmpnode, max_scale); // O(|point_saet|)
592             // putting points closer to new_point into new_point_set (and
593             // removing them from far)
594             dist_split(far, new_point_set, tmpnode, max_scale); // O(|far|)
595 
596             CoverTreeNode new_child = batch_insert(tmpnode.idx, next_scale,
597                 top_scale, new_point_set, new_consumed_set);
598             new_child.parent_dist = new_dist;
599 
600             children.push(new_child);
601 
602             // putting the unused points from new_point_set back into
603             // point_set and far
604             double fmax = dist_of_scale(max_scale);
605             tmpnode = null;
606             for (int i = 0; i < new_point_set.length; i++) { // O(|new_point_set|)
607               tmpnode = new_point_set.element(i);
608               tmpnode.dist.pop();
609               if (tmpnode.dist.last() <= fmax)
610                 point_set.push(tmpnode);
611               else
612                 far.push(tmpnode);
613             }
614             // putting the points consumed while recursing for new_point
615             // into consumed_set
616             tmpnode = null;
617             for (int i = 0; i < new_consumed_set.length; i++) { // O(|new_point_set|)
618               tmpnode = new_consumed_set.element(i);
619               tmpnode.dist.pop();
620               consumed_set.push(tmpnode);
621             }
622           }// end while(point_size.size!=0)
623           point_set.replaceAllBy(far); // point_set=far;
624           n.scale = top_scale - max_scale;
625           n.max_dist = max_set(consumed_set);
626           n.num_children = children.length;
627           n.children = children;
628           return n;
629         }// end else if(pointset!=0)
630       }// end else if(next_scale != -214....
631     }// end else if(pointset!=0)
632   }

634   /** 
635    * Builds the tree on the given set of instances.
636    * P.S.: For internal use only. Outside classes 
637    * should call setInstances(). 
638    * @param insts The instances on which to build 
639    * the cover tree.
640    * @throws Exception If the supplied set of 
641    * Instances is empty, or if there are missing
642    * values. 
643    */
644   protected void buildCoverTree(Instances insts) throws Exception {
645     if (insts.numInstances() == 0)
646       throw new Exception(
647       "CoverTree: Empty set of instances. Cannot build tree.");
648     checkMissing(insts);
649     if (m_EuclideanDistance == null)
650       m_DistanceFunction = m_EuclideanDistance = new EuclideanDistance(insts);
651     else
652       m_EuclideanDistance.setInstances(insts);
653     
654     Stack<DistanceNode> point_set = new Stack<DistanceNode>();
655     Stack<DistanceNode> consumed_set = new Stack<DistanceNode>();

673       m_Root = batch_insert(p_idx, get_scale(max_dist), get_scale(max_dist),

679   /**
680    * A class for a heap to store the nearest k neighbours to an instance. 
681    * The heap also takes care of cases where multiple neighbours are the same 
682    * distance away.

694     /**
695      * constructor.
696      * @param maxSize   the maximum size of the heap
697      */
698     public MyHeap(int maxSize) {
699       if((maxSize%2)==0)
700         maxSize++;

716      * @return the first element
717      */
718     public MyHeapElement peek() {

720     }

722     /**
723      * returns the first element and removes it from the heap.
724      * @return the first element
725      * @throws Exception  if no elements in heap
726      */
727     public MyHeapElement get() throws Exception  {
728       if(m_heap[0].index==0)
729         throw new Exception("No elements present in the heap");
730       MyHeapElement r = m_heap[1];
731       m_heap[1] = m_heap[m_heap[0].index];
732       m_heap[0].index--;
733       downheap();
734       return r;
735     }

738      * adds the distance value to the heap.
739      * 
740      * @param d the distance value 
741      * @throws Exception  if the heap gets too large
742      */
743     public void put(double d) throws Exception {
744       if((m_heap[0].index+1)>(m_heap.length-1))
745         throw new Exception("the number of elements cannot exceed the "+
746         "initially set maximum limit");
747       m_heap[0].index++;
748       m_heap[m_heap[0].index] = new MyHeapElement(d);
749       upheap();
750     }

752     /**
753      * Puts an element by substituting it in place of 
754      * the top most element.
755      * 
756      * @param d The distance value.
757      * @throws Exception If distance is smaller than that of the head
758      *         element.
759      */
760     public void putBySubstitute(double d) throws Exception {
761       MyHeapElement head = get();
762       put(d);
763       if(head.distance == m_heap[1].distance) {
764         putKthNearest(head.distance);
765       }
766       else if(head.distance > m_heap[1].distance) {
767         m_KthNearest = null;
768         m_KthNearestSize = 0;
769         initSize = 10;
770       }
771       else if(head.distance < m_heap[1].distance) {
772         throw new Exception("The substituted element is greater than the "+
773         "head element. put() should have been called "+
774         "in place of putBySubstitute()");
775       }
776     }

782     int m_KthNearestSize = 0;
783     
784     /** the initial size of the heap. */
785     int initSize=10;

797     /**
798      * Stores kth nearest elements (if there are 
799      * more than one).
800      * @param d the distance 
801      */
802     public void putKthNearest(double d) {
803       if(m_KthNearest==null) {
804         m_KthNearest = new MyHeapElement[initSize];
805       }
806       if(m_KthNearestSize>=m_KthNearest.length) {
807         initSize += initSize;
808         MyHeapElement temp[] = new MyHeapElement[initSize];
809         System.arraycopy(m_KthNearest, 0, temp, 0, m_KthNearest.length);
810         m_KthNearest = temp;
811       }
812       m_KthNearest[m_KthNearestSize++] = new MyHeapElement(d);
813     }

816      * returns the kth nearest element or null if none there.
817      * 
818      * @return      the kth nearest element
819      */
820     public MyHeapElement getKthNearest() {
821       if(m_KthNearestSize==0)
822         return null;
823       m_KthNearestSize--;
824       return m_KthNearest[m_KthNearestSize];
825     }

827     /** 
828      * performs upheap operation for the heap 
829      * to maintian its properties. 
830      */
831     protected void upheap() {
832       int i = m_heap[0].index;
833       MyHeapElement temp;
834       while( i > 1  && m_heap[i].distance>m_heap[i/2].distance) {
835         temp = m_heap[i];
836         m_heap[i] = m_heap[i/2];
837         i = i/2;
838         m_heap[i] = temp; //this is i/2 done here to avoid another division.
839       }
840     }
841     
842     /** 
843      * performs downheap operation for the heap 
844      * to maintian its properties. 
845      */
846     protected void downheap() {
847       int i = 1;
848       MyHeapElement temp;
849       while( ( (2*i) <= m_heap[0].index &&
850       m_heap[i].distance < m_heap[2*i].distance )
851       ||
852       ( (2*i+1) <= m_heap[0].index &&
853       m_heap[i].distance < m_heap[2*i+1].distance) ) {
854         if((2*i+1)<=m_heap[0].index) {
855           if(m_heap[2*i].distance>m_heap[2*i+1].distance) {
856             temp = m_heap[i];
857             m_heap[i] = m_heap[2*i];
858             i = 2*i;
859             m_heap[i] = temp;
860           }
861           else {
862             temp = m_heap[i];
863             m_heap[i] = m_heap[2*i+1];
864             i = 2*i+1;
865             m_heap[i] = temp;
866           }
867         }
868         else {
869           temp = m_heap[i];
870           m_heap[i] = m_heap[2*i];
871           i = 2*i;
872           m_heap[i] = temp;
873         }
874       }
875     }

1187  /**
1188   * Half-sorts a cover set, so that nodes nearer to the query
1189   * are at the front. 
1190   * @param cover_set The cover set to sort.
1191   */
1192
1193  protected void halfsort(Stack<d_node> cover_set) {
1194    if(cover_set.length <= 1)
1195      return;
1196    int start=0;
1197    int hi = cover_set.length-1;
1198    int right = hi;
1199    int left;
1200    
1201    while (right > start) {
1202      int mid = start + ((hi - start) >> 1);
1203
1204      boolean jumpover = false;
1205      if (compare(mid, start, cover_set) < 0.0)
1206        SWAP(mid, start, cover_set);
1207      if (compare(hi, mid, cover_set) < 0.0)
1208        SWAP(mid, hi, cover_set);
1209      else
1210        jumpover = true;
1211      if (!jumpover && compare(mid, start, cover_set) < 0.0)
1212        SWAP(mid, start, cover_set);
1213      jump_over:
1214      ;
1215
1216      left = start + 1;
1217      right = hi - 1;
1218
1219      do {
1220        while (compare(left, mid, cover_set) < 0.0)
1221          left++;
1222
1223        while (compare(mid, right, cover_set) < 0.0)
1224          right--;
1225
1226        if (left < right) {
1227          SWAP(left, right, cover_set);
1228          if (mid == left)
1229            mid = right;
1230          else if (mid == right)
1231            mid = left;
1232          left++;
1233          right--;
1234        } else if (left == right) {
1235          left++;
1236          right--;
1237          break;
1238        }
1239      } while (left <= right);
1240      hi = right;
1241    }
1242  }

1290      int current_scale, int max_scale, // amk14comment: make sure this gets
1291                                        // passed by reference in Java
1292      Stack<Stack<d_node>> cover_sets, // amk14comment: contains children in
1293                                        // set Q in paper
1294      Stack<d_node> zero_set) // amk14comment: zeroset contains the children at
1295                              // the lowest level i.e. -infinity
1296      throws Exception {

1298    Stack<d_node> cover_set_currentscale = getCoverSet(current_scale,
1299        cover_sets);
1300    for (int i = 0; i < cover_set_currentscale.length; i++) {
1301      parent = cover_set_currentscale.element(i);
1302      CoverTreeNode par = parent.n;
1303      double upper_dist = upper_k.peek().distance + query.max_dist
1304          + query.max_dist; // *upper_bound + query->max_dist + query->max_dist;
1305      if (parent.dist <= upper_dist + par.max_dist) {
1306        CoverTreeNode chi;
1307        if (par == m_Root && par.num_children == 0) // if our tree consists of
1308                                                    // only one root(which is
1309                                                    // also leaf) node
1310          chi = par;
1311        else
1312          chi = par.children.element(0);
1313        if (parent.dist <= upper_dist + chi.max_dist) { // amk14comment: looking
1314                                                        // at child_0 (which is
1315                                                        // the parent itself)
1316          if (chi.num_children > 0) {
1317            if (max_scale < chi.scale) {
1318              max_scale = chi.scale;
1319            }
1320            d_node temp = new d_node(parent.dist, chi);
1321            getCoverSet(chi.scale, cover_sets).push(temp);
1322            if (m_TreeStats != null)
1323              m_TreeStats.incrIntNodeCount();
1324          } else if (parent.dist <= upper_dist) {
1325            d_node temp = new d_node(parent.dist, chi);
1326            zero_set.push(temp);
1327            if (m_TreeStats != null)
1328              m_TreeStats.incrLeafCount();
1329          }
1330        }
1331        for (int c = 1; c < par.num_children; c++) {
1332          chi = par.children.element(c);
1333          double upper_chi = upper_k.peek().distance + chi.max_dist
1334              + query.max_dist + query.max_dist; // *upper_bound + chi.max_dist
1335                                                  // + query.max_dist +
1336                                                  // query.max_dist;
1337          if (shell(parent.dist, chi.parent_dist, upper_chi)) { // amk14comment:parent_query_dist
1338                                                                // -
1339                                                                // child_parent_dist
1340                                                                // <= upper_chi - if child can be 
1341                                                                // inside the shrunk query ball 
1342            // NOT the same as above parent->dist <= upper_dist + chi->max_dist
1343            double d = Math.sqrt(m_DistanceFunction.distance(query.p(),
1344                chi.p(), upper_chi * upper_chi, m_TreeStats));
1345            if (m_TreeStats != null)
1346              m_TreeStats.incrPointCount();
1347            if (d <= upper_chi) { //if child is inside the shrunk query ball
1348              if (d < upper_k.peek().distance) // *upper_bound)
1349                update(upper_k, d);
1350              if (chi.num_children > 0) {
1351                if (max_scale < chi.scale) {
1352                  max_scale = chi.scale;
1353                }
1354                d_node temp = new d_node(d, chi);
1355                getCoverSet(chi.scale, cover_sets).push(temp);
1356                if (m_TreeStats != null)
1357                  m_TreeStats.incrIntNodeCount();
1358              } else if (d <= upper_chi - chi.max_dist) {
1359                d_node temp = new d_node(d, chi);
1360                zero_set.push(temp);
1361                if (m_TreeStats != null)
1362                  m_TreeStats.incrLeafCount();
1363              }
1364            }//end if(d<=upper_chi)
1365          }//end if(shell(parent.dist,...
1366        }//end for(child_1 to n)
1367      }//end if(parent.dist<=upper_dist..
1368    }//end for(covers_sets[current_scale][i])
1369    return max_scale;
1370  }

1372  /**
1373   * Does a brute force NN search on the nodes in the given zero set.
1374   * A zero set might have some nodes added to it that were not k-NNs,
1375   * so need to do a brute-force to pick only the k-NNs (without 
1376   * calculating distances, as each node in the zero set already had 
1377   * its distance calculated to the query, which is stored with the
1378   * node).
1379   *  
1380   * @param k The k in kNN.
1381   * @param query The query. 
1382   * @param zero_set The zero set on which the brute force NN search
1383   * is performed.
1384   * @param upper_k The heap storing distances of k-NNs found during
1385   * the search.
1386   * @param results The returned k-NNs.
1387   * @throws Exception If there is somem problem.
1388   */
1389  protected void brute_nearest(final int k, final CoverTreeNode query,
1390      Stack<d_node> zero_set, MyHeap upper_k, Stack<NeighborList> results)
1391      throws Exception {
1392    if (query.num_children > 0) {
1393      Stack<d_node> new_zero_set = new Stack<d_node>();
1394      CoverTreeNode query_chi = query.children.element(0);
1395      brute_nearest(k, query_chi, zero_set, upper_k, results);
1396      MyHeap new_upper_k = new MyHeap(k);
1397
1398      for (int i = 1; i < query.children.length; i++) {
1399        query_chi = query.children.element(i);
1400        setter(new_upper_k, upper_k.peek().distance + query_chi.parent_dist, k);
1401        copy_zero_set(query_chi, new_upper_k, zero_set, new_zero_set);
1402        brute_nearest(k, query_chi, new_zero_set, new_upper_k, results);
1403      }
1404    } else {
1405      NeighborList temp = new NeighborList(k);
1406      d_node ele;
1407      for (int i = 0; i < zero_set.length; i++) {
1408        ele = zero_set.element(i);
1409        if (ele.dist <= upper_k.peek().distance) {
1410          temp.insertSorted(ele.dist, ele.n.p()); // temp.push(ele.n.p());
1411        }
1412      }
1413      results.push(temp);
1414    }
1415  }

1417  /**
1418   * Performs a recursive k-NN search for a given batch of queries provided in the
1419   * form of a cover tree. P.S.: This function should not be called from outside. 
1420   * Outside classes should use kNearestNeighbours() instead.
1421   *  
1422   * @param k The number of NNs to find.
1423   * @param query_node The node of the query tree to start the search from.
1424   * @param cover_sets The set of sets that contains internal
1425   * nodes that were found to be inside the query ball at previous scales/levels
1426   * (intially there would be just the root node at root level).
1427   * @param zero_set The set that'll contain the leaf nodes that are found to
1428   * be inside the query ball.
1429   * @param current_scale The level/scale to do the search from (this value
1430   * would be used to inspect the cover set in the provided set of cover sets).
1431   * @param max_scale The max scale/level that has so far been inspected.
1432   * @param upper_k The heap containing distances of the best k-NNs found so
1433   * far (initialized to Double.POSITIVE_INFINITY).
1434   * @param results The list of returned k-NNs.
1435   * @throws Exception If there is some problem during the search.
1436   */
1437  protected void internal_batch_nearest_neighbor(final int k, 
1438                        final CoverTreeNode query_node,
1439                        Stack<Stack<d_node>> cover_sets,
1440                        Stack<d_node> zero_set,
1441                        int current_scale,
1442                        int max_scale,
1443                        MyHeap upper_k,
1444                        Stack<NeighborList> results) throws Exception {
1445    if (current_scale > max_scale) { // All remaining points are in the zero set.
1446      brute_nearest(k, query_node, zero_set, upper_k, results);
1447    } else {
1448      // Our query_node has too much scale. Reduce.
1449      if (query_node.scale <= current_scale && query_node.scale != 100) { // amk14comment:if j>=i in paper
1450        CoverTreeNode query_chi;
1451        Stack<d_node> new_zero_set = new Stack<d_node>();
1452        Stack<Stack<d_node>> new_cover_sets = new Stack<Stack<d_node>>();
1453        MyHeap new_upper_k = new MyHeap(k);
1454
1455        for (int i = 1; i < query_node.num_children; i++) { //processing child_1 and onwards
1456          query_chi = query_node.children.element(i);
1457          setter(new_upper_k, upper_k.peek().distance + query_chi.parent_dist, k);
1458          //copy the zero set that satisfy a certain bound to the new zero set
1459          copy_zero_set(query_chi, new_upper_k, zero_set, new_zero_set);
1460          //copy the coversets[current_scale] nodes that satisfy a certain
1461          //bound to the new_cover_sets[current_scale]
1462          copy_cover_sets(query_chi, new_upper_k, cover_sets, new_cover_sets,
1463              current_scale, max_scale);
1464          //search for the query_node child in the nodes nearer to it.
1465          internal_batch_nearest_neighbor(k, query_chi, new_cover_sets,
1466              new_zero_set, current_scale, max_scale, new_upper_k, results);
1467        }
1468        new_cover_sets = null;
1469        new_zero_set = null;
1470        new_upper_k = null;
1471        // now doing child_0 //which is the parent itself, that's why we don't
1472        // need new_zero_set or new_cover_sets
1473        internal_batch_nearest_neighbor(k, query_node.children.element(0),
1474            cover_sets, zero_set, current_scale, max_scale, upper_k, results);
1475      } else { // reduce cover set scale -- amk14comment: if j<i in paper
1476        Stack<d_node> cover_set_i = getCoverSet(current_scale, cover_sets);
1477        // println("sorting");
1478        halfsort(cover_set_i);
1479        max_scale = descend(query_node, upper_k, current_scale, max_scale,
1480            cover_sets, zero_set);
1481        cover_set_i.clear();
1482        current_scale++;
1483        internal_batch_nearest_neighbor(k, query_node, cover_sets, zero_set,
1484            current_scale, max_scale, upper_k, results);
1485      }
1486    }
1487  }

1535   * Performs k-NN serach for a single given query/test Instance.
1536   * 
1537   * @param target The query/test instance.
1538   * @param k Number of k-NNs to find.

1543  protected NeighborList findKNearest(final Instance target, final int k) throws Exception {
1544    Stack<d_node> cover_set_current = new Stack<d_node>(),
1545                   cover_set_next,
1546                   zero_set = new Stack<d_node>();
1547    CoverTreeNode parent, child; d_node par;
1548    MyHeap upper_k = new MyHeap(k);    
1549    double d = Math.sqrt(m_DistanceFunction.distance(m_Root.p(), target, Double.POSITIVE_INFINITY, m_TreeStats)),
1550           upper_bound;
1551    cover_set_current.push(new d_node(d, m_Root));    
1552    setter(upper_k, Double.POSITIVE_INFINITY, k);
1553    this.update(upper_k, d);
1554    //updating stats for the root node
1555    if(m_TreeStats!=null) {
1556        if(m_Root.num_children > 0)
1557          m_TreeStats.incrIntNodeCount();
1558        else
1559          m_TreeStats.incrLeafCount();
1560        m_TreeStats.incrPointCount();
1561    }
1562    
1563    //if root is the only node
1564    if(m_Root.num_children==0) {
1565      NeighborList list = new NeighborList(k);
1566      list.insertSorted(d, m_Root.p());
1567      return list;
1568    }
1569    //else
1570    while(cover_set_current.length>0) {
1571      cover_set_next = new Stack<d_node>();
1572      for(int i=0; i<cover_set_current.length; i++) {
1573    par = cover_set_current.element(i);
1574    parent = par.n;
1575    for(int c=0; c<parent.num_children; c++) {
1576      child = parent.children.element(c);
1577      upper_bound = upper_k.peek().distance;
1578      if(c==0)
1579        d = par.dist;
1580      else {
1581        d = upper_bound + child.max_dist;
1582        d = Math.sqrt(m_DistanceFunction.distance(child.p(), target, d*d, m_TreeStats));
1583          if(m_TreeStats!=null)
1584        m_TreeStats.incrPointCount();
1585      }
1586      if(d <= (upper_bound + child.max_dist)) {
1587        if(c>0 && d < upper_bound) {
1588          update(upper_k, d);
1589        }
1590        if(child.num_children > 0) {
1591          cover_set_next.push(new d_node(d, child));
1592          if(m_TreeStats!=null)
1593        m_TreeStats.incrIntNodeCount();
1594        }
1595        else if (d <= upper_bound){
1596          zero_set.push(new d_node(d, child));
1597          if(m_TreeStats!=null)
1598        m_TreeStats.incrLeafCount();
1599        }
1600      }
1601    } //end for current_set children
1602      } //end for current_set elements
1603      cover_set_current = cover_set_next;
1604    } //end while(curret_set not empty)
1605    
1606    NeighborList list = new NeighborList(k);
1607    d_node tmpnode;
1608    upper_bound = upper_k.peek().distance;      
1609    for(int i=0; i<zero_set.length; i++) {
1610      tmpnode = zero_set.element(i);
1611      if(tmpnode.dist <= upper_bound)
1612    list.insertSorted(tmpnode.dist, tmpnode.n.p());
1613    }
1614    
1615    if(list.currentLength()<=0)
1616      throw new Exception("Error: No neighbour found. This cannot happen");
1617    
1618    return list;
1619  }
