50    /**
51     * A class for a heap to store the nearest k neighbours to an instance. 
52     * The heap also takes care of cases where multiple neighbours are the same 
53     * distance away.
54     * i.e. the minimum size of the heap is k.
55     *
56     * @author Ashraf M. Kibriya (amk14[at-the-rate]cs[dot]waikato[dot]ac[dot]nz)
57     * @version $Revision: 5953 $
58     */
59    protected class MyHeap
60      implements RevisionHandler {
61      
62      /** the heap. */
63      MyHeapElement m_heap[] = null;
64      
65      /**
66       * constructor.
67       * 
68       * @param maxSize       the maximum size of the heap
69       */
70      public MyHeap(int maxSize) {
71        if((maxSize%2)==0)
72          maxSize++;
73        
74        m_heap = new MyHeapElement[maxSize+1];
75        m_heap[0] = new MyHeapElement(0, 0);
76      }
77      
78      /**
79       * returns the size of the heap.
80       * 
81       * @return          the size
82       */
83      public int size() {
84        return m_heap[0].index;
85      }
86      
87      /**
88       * peeks at the first element.
89       * 
90       * @return          the first element
91       */
92      public MyHeapElement peek() {
93        return m_heap[1];
94      }
95      
96      /**
97       * returns the first element and removes it from the heap.
98       * 
99       * @return          the first element
100      * @throws Exception    if no elements in heap
101      */
102     public MyHeapElement get() throws Exception  {
103       if(m_heap[0].index==0)
104         throw new Exception("No elements present in the heap");
105       MyHeapElement r = m_heap[1];
106       m_heap[1] = m_heap[m_heap[0].index];
107       m_heap[0].index--;
108       downheap();
109       return r;
110     }
111     
112     /**
113      * adds the value to the heap.
114      * 
115      * @param i         the index
116      * @param d         the distance
117      * @throws Exception    if the heap gets too large
118      */
119     public void put(int i, double d) throws Exception {
120       if((m_heap[0].index+1)>(m_heap.length-1))
121         throw new Exception("the number of elements cannot exceed the "+
122         "initially set maximum limit");
123       m_heap[0].index++;
124       m_heap[m_heap[0].index] = new MyHeapElement(i, d);
125       upheap();
126     }
127     
128     /**
129      * Puts an element by substituting it in place of 
130      * the top most element.
131      * 
132      * @param i         the index
133      * @param d         the distance
134      * @throws Exception    if distance is smaller than that of the head
135      *              element
136      */
137     public void putBySubstitute(int i, double d) throws Exception {
138       MyHeapElement head = get();
139       put(i, d);
140       //      System.out.println("previous: "+head.distance+" current: "+m_heap[1].distance);
141       if(head.distance == m_heap[1].distance) { //Utils.eq(head.distance, m_heap[1].distance)) {
142         putKthNearest(head.index, head.distance);
143       }
144       else if(head.distance > m_heap[1].distance) { //Utils.gr(head.distance, m_heap[1].distance)) {
145         m_KthNearest = null;
146         m_KthNearestSize = 0;
147         initSize = 10;
148       }
149       else if(head.distance < m_heap[1].distance) {
150         throw new Exception("The substituted element is smaller than the "+
151         "head element. put() should have been called "+
152         "in place of putBySubstitute()");
153       }
154     }
155     
156     /** the kth nearest ones. */
157     MyHeapElement m_KthNearest[] = null;
158     
159     /** The number of kth nearest elements. */
160     int m_KthNearestSize = 0;
161     
162     /** the initial size of the heap. */
163     int initSize=10;
164 
165     /**
166      * returns the number of k nearest.
167      * 
168      * @return the number of k nearest
169      * @see         #m_KthNearestSize
170      */
171     public int noOfKthNearest() {
172       return m_KthNearestSize;
173     }
174     
175     /**
176      * Stores kth nearest elements (if there are 
177      * more than one).
178      * @param i         the index
179      * @param d         the distance
180      */
181     public void putKthNearest(int i,  double d) {
182       if(m_KthNearest==null) {
183         m_KthNearest = new MyHeapElement[initSize];
184       }
185       if(m_KthNearestSize>=m_KthNearest.length) {
186         initSize += initSize;
187         MyHeapElement temp[] = new MyHeapElement[initSize];
188         System.arraycopy(m_KthNearest, 0, temp, 0, m_KthNearest.length);
189         m_KthNearest = temp;
190       }
191       m_KthNearest[m_KthNearestSize++] = new MyHeapElement(i, d);
192     }
193     
194     /**
195      * returns the kth nearest element or null if none there.
196      * 
197      * @return          the kth nearest element
198      */
199     public MyHeapElement getKthNearest() {
200       if(m_KthNearestSize==0)
201         return null;
202       m_KthNearestSize--;
203       return m_KthNearest[m_KthNearestSize];
204     }
205     
206     /** 
207      * performs upheap operation for the heap 
208      * to maintian its properties. 
209      */
210     protected void upheap() {
211       int i = m_heap[0].index;
212       MyHeapElement temp;
213       while( i > 1  && m_heap[i].distance>m_heap[i/2].distance) {
214         temp = m_heap[i];
215         m_heap[i] = m_heap[i/2];
216         i = i/2;
217         m_heap[i] = temp; //this is i/2 done here to avoid another division.
218       }
219     }
220     
221     /** 
222      * performs downheap operation for the heap 
223      * to maintian its properties. 
224      */
225     protected void downheap() {
226       int i = 1;
227       MyHeapElement temp;
228       while( ( (2*i) <= m_heap[0].index &&
229       m_heap[i].distance < m_heap[2*i].distance )
230       ||
231       ( (2*i+1) <= m_heap[0].index &&
232       m_heap[i].distance < m_heap[2*i+1].distance) ) {
233         if((2*i+1)<=m_heap[0].index) {
234           if(m_heap[2*i].distance>m_heap[2*i+1].distance) {
235             temp = m_heap[i];
236             m_heap[i] = m_heap[2*i];
237             i = 2*i;
238             m_heap[i] = temp;
239           }
240           else {
241             temp = m_heap[i];
242             m_heap[i] = m_heap[2*i+1];
243             i = 2*i+1;
244             m_heap[i] = temp;
245           }
246         }
247         else {
248           temp = m_heap[i];
249           m_heap[i] = m_heap[2*i];
250           i = 2*i;
251           m_heap[i] = temp;
252         }
253       }
254     }

258      * 
259      * @return          the total size
260      */

262       return size()+noOfKthNearest();
263     }

363   /**
364    * A class for a linked list to store the nearest k neighbours
365    * to an instance. We use a list so that we can take care of
366    * cases where multiple neighbours are the same distance away.
367    * i.e. the minimum length of the list is k.
368    *
369    * @author Ashraf M. Kibriya (amk14[at-the-rate]cs[dot]waikato[dot]ac[dot]nz)
370    * @version $Revision: 5953 $
371    */ //better to change this into a heap
372   protected class NeighborList
373     implements RevisionHandler {
374 
375     /** The first node in the list. */
376     protected NeighborNode m_First;
377 
378     /** The last node in the list. */
379     protected NeighborNode m_Last;
380 
381     /** The number of nodes to attempt to maintain in the list. */
382     protected int m_Length = 1;
383         
384     /**
385      * Creates the neighborlist with a desired length.
386      *
387      * @param length        the length of list to attempt to maintain
388      */
389     public NeighborList(int length) {
390       m_Length = length;
391     }
392 
393     /**
394      * Gets whether the list is empty.
395      *
396      * @return          true if list is empty
397      */
398     public boolean isEmpty() {
399       return (m_First == null);
400     }
401 
402     /**
403      * Gets the current length of the list.
404      *
405      * @return          the current length of the list
406      */
407     public int currentLength() {
408       int i = 0;
409       NeighborNode current = m_First;
410       while (current != null) {
411         i++;
412         current = current.m_Next;
413       }
414       return i;
415     }
416 
417     /**
418      * Inserts an instance neighbor into the list, maintaining the list
419      * sorted by distance.
420      *
421      * @param distance      the distance to the instance
422      * @param instance      the neighboring instance
423      */
424     public void insertSorted(double distance, Instance instance) {
425       
426       if (isEmpty()) {
427         m_First = m_Last = new NeighborNode(distance, instance);
428       } else {
429         NeighborNode current = m_First;
430         if (distance < m_First.m_Distance) {// Insert at head
431           m_First = new NeighborNode(distance, instance, m_First);
432         } else { // Insert further down the list
433           for( ;(current.m_Next != null) &&
434           (current.m_Next.m_Distance < distance);
435           current = current.m_Next);
436           current.m_Next = new NeighborNode(distance, instance,
437           current.m_Next);
438           if (current.equals(m_Last)) {
439             m_Last = current.m_Next;
440           }
441         }
442         
443         // Trip down the list until we've got k list elements (or more if the
444         // distance to the last elements is the same).
445         int valcount = 0;
446         for(current = m_First; current.m_Next != null;
447         current = current.m_Next) {
448           valcount++;
449           if ((valcount >= m_Length) && (current.m_Distance !=
450           current.m_Next.m_Distance)) {
451             m_Last = current;
452             current.m_Next = null;
453             break;
454           }
455         }
456       }
457     }
458 
459     /**
460      * Prunes the list to contain the k nearest neighbors. If there are
461      * multiple neighbors at the k'th distance, all will be kept.
462      *
463      * @param k         the number of neighbors to keep in the list.
464      */
465     public void pruneToK(int k) {
466       
467       if (isEmpty()) {
468         return;
469       }
470       if (k < 1) {
471         k = 1;
472       }
473       int currentK = 0;
474       double currentDist = m_First.m_Distance;
475       NeighborNode current = m_First;
476       for(; current.m_Next != null; current = current.m_Next) {
477         currentK++;
478         currentDist = current.m_Distance;
479         if ((currentK >= k) && (currentDist != current.m_Next.m_Distance)) {
480           m_Last = current;
481           current.m_Next = null;
482           break;
483         }
484       }
485     }
486     
487     /**
488      * Prints out the contents of the neighborlist.
489      */
490     public void printList() {
491       
492       if (isEmpty()) {
493         System.out.println("Empty list");
494       } else {
495         NeighborNode current = m_First;
496         while (current != null) {
497           System.out.println("Node: instance " + current.m_Instance
498           + ", distance " + current.m_Distance);
499           current = current.m_Next;
500         }
501         System.out.println();
502       }
503     }

598   /**
599    * Parses a given list of options. Valid options are:
600    *
601    <!-- options-start -->
602    <!-- options-end -->
603    *
604    * @param options     the list of options as an array of strings
605    * @throws Exception  if an option is not supported
606    */
607   public void setOptions(String[] options) throws Exception {
608     String nnSearchClass = Utils.getOption('A', options);
609     if(nnSearchClass.length() != 0) {
610       String nnSearchClassSpec[] = Utils.splitOptions(nnSearchClass);
611       if(nnSearchClassSpec.length == 0) { 
612         throw new Exception("Invalid DistanceFunction specification string."); 
613       }
614       String className = nnSearchClassSpec[0];
615       nnSearchClassSpec[0] = "";
616 
617       setDistanceFunction( (DistanceFunction)
618                             Utils.forName( DistanceFunction.class, 
619                                            className, nnSearchClassSpec) );
620     }
621     else {
622       setDistanceFunction(new EuclideanDistance());
623     }
624     
625     setMeasurePerformance(Utils.getFlag('P',options));
626   }

628   /**
629    * Gets the current settings.
630    *
631    * @return        an array of strings suitable for passing to setOptions()
632    */
633   public String [] getOptions() {
634     Vector<String>  result;
635     
636     result = new Vector<String>();
637 
638     result.add("-A");
639     result.add((m_DistanceFunction.getClass().getName() + " " +
640                    Utils.joinOptions(m_DistanceFunction.getOptions())).trim());
641     
642     if(getMeasurePerformance())
643       result.add("-P");
644     
645     return result.toArray(new String[result.size()]);
646   }

834   public static void combSort11(double arrayToSort[], int linkedArray[]) {
835     int switches, j, top, gap;
836     double hold1; int hold2;
837     gap = arrayToSort.length;
838     do {
839       gap=(int)(gap/1.3);
840       switch(gap) {
841         case 0:
842           gap = 1;
843           break;
844         case 9:
845         case 10:
846           gap=11;
847           break;
848         default:
849           break;
850       }
851       switches=0;
852       top = arrayToSort.length-gap;
853       for(int i=0; i<top; i++) {
854         j=i+gap;
855         if(arrayToSort[i] > arrayToSort[j]) {
856           hold1=arrayToSort[i];
857           hold2=linkedArray[i];

863         }//endif
864       }//endfor
865     } while(switches>0 || gap>1);

868   /**
869    * Partitions the instances around a pivot. Used by quicksort and
870    * kthSmallestValue.
871    *
872    * @param arrayToSort     the array of doubles to be sorted
873    * @param linkedArray     the linked array
874    * @param l           the first index of the subset 
875    * @param r           the last index of the subset 
876    * @return            the index of the middle element
877    */
878   protected static int partition(double[] arrayToSort, double[] linkedArray, int l, int r) {
879     double pivot = arrayToSort[(l + r) / 2];
880     double help;
881 
882     while (l < r) {
883       while ((arrayToSort[l] < pivot) && (l < r)) {
884         l++;
885       }
886       while ((arrayToSort[r] > pivot) && (l < r)) {
887         r--;
888       }
889       if (l < r) {
890         help = arrayToSort[l];
891         arrayToSort[l] = arrayToSort[r];
892         arrayToSort[r] = help;
893         help = linkedArray[l];
894         linkedArray[l] = linkedArray[r];
895         linkedArray[r] = help;
896         l++;
897         r--;
898       }
899     }
900     if ((l == r) && (arrayToSort[r] > pivot)) {
901       r--;
902     } 
903 
904     return r;
905   }

907   /**
908    * performs quicksort.
909    * 
910    * @param arrayToSort     the array to sort
911    * @param linkedArray     the linked array
912    * @param left        the first index of the subset 
913    * @param right       the last index of the subset 
914    */
915   public static void quickSort(double[] arrayToSort, double[] linkedArray, int left, int right) {
916     if (left < right) {
917       int middle = partition(arrayToSort, linkedArray, left, right);
918       quickSort(arrayToSort, linkedArray, left, middle);
919       quickSort(arrayToSort, linkedArray, middle + 1, right);
920     }
921   }
