23  package weka.core;
24  
25  import java.io.Serializable;
26  
27  /**
28   * A <code>Tag</code> simply associates a numeric ID with a String description.
29   *
30   * @author <a href="mailto:len@reeltwo.com">Len Trigg</a>
31   * @version $Revision: 5953 $
32   */
33  public class Tag implements Serializable, RevisionHandler {
34  
35    /** for serialization. */
36    private static final long serialVersionUID = 3326379903447135320L;
37  
38    /** The ID */
39    protected int m_ID;
40  
41    /** The unique string for this tag, doesn't have to be numeric */
42    protected String m_IDStr;
43    
44    /** The descriptive text */
45    protected String m_Readable;
46  
47    /**
48     * Creates a new default Tag
49     *
50     */
51    public Tag() {
52      this(0, "A new tag", "A new tag", true);
53    }
54  
55    /**
56     * Creates a new <code>Tag</code> instance.
57     *
58     * @param ident the ID for the new Tag.
59     * @param readable the description for the new Tag.
60     */
61    public Tag(int ident, String readable) {
62      this(ident, "", readable);
63    }
64    
65    /**
66     * Creates a new <code>Tag</code> instance.
67     *
68     * @param ident the ID for the new Tag.
69     * @param identStr the ID string for the new Tag (case-insensitive).
70     * @param readable the description for the new Tag.
71     */
72    public Tag(int ident, String identStr, String readable) {
73      this(ident, identStr, readable, true);
74    }
75  
76    public Tag(int ident, String identStr, String readable, boolean upperCase) {
77      m_ID = ident;
78      if (identStr.length() == 0) {
79        m_IDStr = "" + ident;
80      } else {
81          m_IDStr = identStr;
82        if (upperCase) {
83          m_IDStr = identStr.toUpperCase();
84        } 
85      }
86      m_Readable = readable;
87    }
88  
89    /**
90     * Gets the numeric ID of the Tag.
91     *
92     * @return the ID of the Tag.
93     */
94    public int getID() {
95      return m_ID;
96    }
97  
98    /**
99     * Sets the numeric ID of the Tag.
100    *
101    * @param id the ID of the Tag.
102    */
103   public void setID(int id) {
104     m_ID = id;
105   } 
106 
107   /**
108    * Gets the string ID of the Tag.
109    *
110    * @return the string ID of the Tag.
111    */
112   public String getIDStr() {
113     return m_IDStr;
114   }
115 
116   /**
117    * Sets the string ID of the Tag.
118    *
119    * @param str the string ID of the Tag.
120    */
121   public void setIDStr(String str) {
122     m_IDStr = str;
123   }
124 
125   /**
126    * Gets the string description of the Tag.
127    *
128    * @return the description of the Tag.
129    */
130   public String getReadable() {
131     return m_Readable;
132   }
133 
134   /**
135    * Sets the string description of the Tag.
136    *
137    * @param r the description of the Tag.
138    */
139   public void setReadable(String r) {
140     m_Readable = r;
141   }
142   
143   /**
144    * returns the IDStr
145    * 
146    * @return the IDStr
147    */
148   public String toString() {
149     return m_IDStr;
150   }
151   
152   /**
153    * returns a list that can be used in the listOption methods to list all
154    * the available ID strings, e.g.: &lt;0|1|2&gt; or &lt;what|ever&gt;
155    * 
156    * @param tags the tags to create the list for
157    * @return a list of all ID strings
158    */
159   public static String toOptionList(Tag[] tags) {
160     String  result;
161     int     i;
162     
163     result = "<";
164     for (i = 0; i < tags.length; i++) {
165       if (i > 0)
166     result += "|";
167       result += tags[i];
168     }
169     result += ">";
170     
171     return result;
172   }
173   
174   /**
175    * returns a string that can be used in the listOption methods to list all
176    * the available options, i.e., "\t\tID = Text\n" for each option
177    * 
178    * @param tags the tags to create the string for
179    * @return a string explaining the tags
180    */
181   public static String toOptionSynopsis(Tag[] tags) {
182     String  result;
183     int     i;
184 
185     result = "";
186     for (i = 0; i < tags.length; i++) {
187       result += "\t\t" + tags[i].getIDStr() + " = " + tags[i].getReadable() + "\n";
188     }
189     
190     return result;
191   }
192   
193   /**
194    * Returns the revision string.
195    * 
196    * @return        the revision
197    */
198   public String getRevision() {
199     return RevisionUtils.extract("$Revision: 5953 $");
200   }
201 }
