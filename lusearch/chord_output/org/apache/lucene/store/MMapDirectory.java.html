35  /** File-based {@link Directory} implementation that uses
36   *  mmap for reading, and {@link
37   *  SimpleFSDirectory.SimpleFSIndexOutput} for writing.
38   *
39   * <p><b>NOTE</b>: memory mapping uses up a portion of the
40   * virtual memory address space in your process equal to the
41   * size of the file being mapped.  Before using this class,
42   * be sure your have plenty of virtual address space, e.g. by
43   * using a 64 bit JRE, or a 32 bit JRE with indexes that are
44   * guaranteed to fit within the address space.
45   * On 32 bit platforms also consult {@link #setMaxChunkSize}
46   * if you have problems with mmap failing because of fragmented
47   * address space. If you get an OutOfMemoryException, it is recommended
48   * to reduce the chunk size, until it works.
49   *
50   * <p>Due to <a href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4724038">
51   * this bug</a> in Sun's JRE, MMapDirectory's {@link IndexInput#close}
52   * is unable to close the underlying OS file handle.  Only when GC
53   * finally collects the underlying objects, which could be quite
54   * some time later, will the file handle be closed.
55   *
56   * <p>This will consume additional transient disk usage: on Windows,
57   * attempts to delete or overwrite the files will result in an
58   * exception; on other platforms, which typically have a &quot;delete on
59   * last close&quot; semantics, while such operations will succeed, the bytes
60   * are still consuming space on disk.  For many applications this
61   * limitation is not a problem (e.g. if you have plenty of disk space,
62   * and you don't rely on overwriting files on Windows) but it's still
63   * an important limitation to be aware of.
64   *
65   * <p>This class supplies the workaround mentioned in the bug report
66   * (disabled by default, see {@link #setUseUnmap}), which may fail on
67   * non-Sun JVMs. It forcefully unmaps the buffer on close by using
68   * an undocumented internal cleanup functionality.
69   * {@link #UNMAP_SUPPORTED} is <code>true</code>, if the workaround
70   * can be enabled (with no guarantees).
71   */
72  public class MMapDirectory extends FSDirectory {

98    static final Class[] NO_PARAM_TYPES = new Class[0];
99    static final Object[] NO_PARAMS = new Object[0];
100   
101   private boolean useUnmapHack = false;
102   private int maxBBuf = Constants.JRE_IS_64BIT ? Integer.MAX_VALUE : (256*1024*1024);
103   
104   /**
105    * <code>true</code>, if this platform supports unmapping mmapped files.
106    */
107   public static final boolean UNMAP_SUPPORTED;
108   static {
109     boolean v;
110     try {
111       Class.forName("sun.misc.Cleaner");
112       Class.forName("java.nio.DirectByteBuffer")
113         .getMethod("cleaner", NO_PARAM_TYPES);
114       v = true;
115     } catch (Exception e) {
116       v = false;
117     }
118     UNMAP_SUPPORTED = v;
119   }
120   
121   /**
122    * This method enables the workaround for unmapping the buffers
123    * from address space after closing {@link IndexInput}, that is
124    * mentioned in the bug report. This hack may fail on non-Sun JVMs.
125    * It forcefully unmaps the buffer on close by using
126    * an undocumented internal cleanup functionality.
127    * <p><b>NOTE:</b> Enabling this is completely unsupported
128    * by Java and may lead to JVM crashes if <code>IndexInput</code>
129    * is closed while another thread is still accessing it (SIGSEGV).
130    * @throws IllegalArgumentException if {@link #UNMAP_SUPPORTED}
131    * is <code>false</code> and the workaround cannot be enabled.
132    */
133   public void setUseUnmap(final boolean useUnmapHack) {
134     if (useUnmapHack && !UNMAP_SUPPORTED)
135       throw new IllegalArgumentException("Unmap hack not supported on this platform!");
136     this.useUnmapHack=useUnmapHack;
137   }
138   
139   /**
140    * Returns <code>true</code>, if the unmap workaround is enabled.
141    * @see #setUseUnmap
142    */
143   public boolean getUseUnmap() {
144     return useUnmapHack;
145   }
146   
147   /**
148    * Try to unmap the buffer, this method silently fails if no support
149    * for that in the JVM. On Windows, this leads to the fact,
150    * that mmapped files cannot be modified or deleted.
151    */
152   final void cleanMapping(final ByteBuffer buffer) throws IOException {
153     if (useUnmapHack) {
154       try {
155         AccessController.doPrivileged(new PrivilegedExceptionAction() {
156           public Object run() throws Exception {
157             final Method getCleanerMethod = buffer.getClass()
158               .getMethod("cleaner", NO_PARAM_TYPES);
159             getCleanerMethod.setAccessible(true);
160             final Object cleaner = getCleanerMethod.invoke(buffer, NO_PARAMS);
161             if (cleaner != null) {
162               cleaner.getClass().getMethod("clean", NO_PARAM_TYPES)
163                 .invoke(cleaner, NO_PARAMS);
164             }
165             return null;
166           }
167         });
168       } catch (PrivilegedActionException e) {
169         final IOException ioe = new IOException("unable to unmap the mapped buffer");
170         ioe.initCause(e.getCause());
171         throw ioe;
172       }
173     }
174   }
175   
176   /**
177    * Sets the maximum chunk size (default is {@link Integer#MAX_VALUE} for
178    * 64 bit JVMs and 256 MiBytes for 32 bit JVMs) used for memory mapping.
179    * Especially on 32 bit platform, the address space can be very fragmented,
180    * so large index files cannot be mapped.
181    * Using a lower chunk size makes the directory implementation a little
182    * bit slower (as the correct chunk must be resolved on each seek)
183    * but the chance is higher that mmap does not fail. On 64 bit
184    * Java platforms, this parameter should always be {@link Integer#MAX_VALUE},
185    * as the address space is big enough.
186    */
187   public void setMaxChunkSize(final int maxBBuf) {
188     if (maxBBuf<=0)
189       throw new IllegalArgumentException("Maximum chunk size for mmap must be >0");
190     this.maxBBuf=maxBBuf;
191   }
192   
193   /**
194    * Returns the current mmap chunk size.
195    * @see #setMaxChunkSize
196    */
197   public int getMaxChunkSize() {
198     return maxBBuf;
199   }

201   private class MMapIndexInput extends IndexInput {
202 
203     private ByteBuffer buffer;
204     private final long length;
205     private boolean isClone = false;
206 
207     private MMapIndexInput(RandomAccessFile raf) throws IOException {
208         this.length = raf.length();
209         this.buffer = raf.getChannel().map(MapMode.READ_ONLY, 0, length);
210     }
211 
212     public byte readByte() throws IOException {
213       try {
214         return buffer.get();
215       } catch (BufferUnderflowException e) {
216         throw new IOException("read past EOF");
217       }
218     }
219 
220     public void readBytes(byte[] b, int offset, int len) throws IOException {
221       try {
222         buffer.get(b, offset, len);
223       } catch (BufferUnderflowException e) {
224         throw new IOException("read past EOF");
225       }
226     }
227 
228     public long getFilePointer() {
229       return buffer.position();
230     }
231 
232     public void seek(long pos) throws IOException {
233       buffer.position((int)pos);
234     }
235 
236     public long length() {
237       return length;
238     }
239 
240     public Object clone() {
241       MMapIndexInput clone = (MMapIndexInput)super.clone();
242       clone.isClone = true;
243       clone.buffer = buffer.duplicate();
244       return clone;
245     }
246 
247     public void close() throws IOException {
248       if (isClone || buffer == null) return;
249       // unmap the buffer (if enabled) and at least unset it for GC
250       try {
251         cleanMapping(buffer);
252       } finally {
253         buffer = null;
254       }
255     }
256   }

258   // Because Java's ByteBuffer uses an int to address the
259   // values, it's necessary to access a file >
260   // Integer.MAX_VALUE in size using multiple byte buffers.
261   private class MultiMMapIndexInput extends IndexInput {
262   
263     private ByteBuffer[] buffers;
264     private int[] bufSizes; // keep here, ByteBuffer.size() method is optional
265   
266     private final long length;
267   
268     private int curBufIndex;
269     private final int maxBufSize;
270   
271     private ByteBuffer curBuf; // redundant for speed: buffers[curBufIndex]
272     private int curAvail; // redundant for speed: (bufSizes[curBufIndex] - curBuf.position())
273   
274     private boolean isClone = false;
275     
276     public MultiMMapIndexInput(RandomAccessFile raf, int maxBufSize)
277       throws IOException {
278       this.length = raf.length();
279       this.maxBufSize = maxBufSize;
280       
281       if (maxBufSize <= 0)
282         throw new IllegalArgumentException("Non positive maxBufSize: "
283                                            + maxBufSize);
284       
285       if ((length / maxBufSize) > Integer.MAX_VALUE)
286         throw new IllegalArgumentException
287           ("RandomAccessFile too big for maximum buffer size: "
288            + raf.toString());
289       
290       int nrBuffers = (int) (length / maxBufSize);
291       if (((long) nrBuffers * maxBufSize) < length) nrBuffers++;
292       
293       this.buffers = new ByteBuffer[nrBuffers];
294       this.bufSizes = new int[nrBuffers];
295       
296       long bufferStart = 0;
297       FileChannel rafc = raf.getChannel();
298       for (int bufNr = 0; bufNr < nrBuffers; bufNr++) { 
299         int bufSize = (length > (bufferStart + maxBufSize))
300           ? maxBufSize
301           : (int) (length - bufferStart);
302         this.buffers[bufNr] = rafc.map(MapMode.READ_ONLY,bufferStart,bufSize);
303         this.bufSizes[bufNr] = bufSize;
304         bufferStart += bufSize;
305       }
306       seek(0L);
307     }
308   
309     public byte readByte() throws IOException {
310       // Performance might be improved by reading ahead into an array of
311       // e.g. 128 bytes and readByte() from there.
312       if (curAvail == 0) {
313         curBufIndex++;
314         if (curBufIndex >= buffers.length)
315           throw new IOException("read past EOF");
316         curBuf = buffers[curBufIndex];
317         curBuf.position(0);
318         curAvail = bufSizes[curBufIndex];
319       }
320       curAvail--;
321       return curBuf.get();
322     }
323   
324     public void readBytes(byte[] b, int offset, int len) throws IOException {
325       while (len > curAvail) {
326         curBuf.get(b, offset, curAvail);
327         len -= curAvail;
328         offset += curAvail;
329         curBufIndex++;
330         if (curBufIndex >= buffers.length)
331           throw new IOException("read past EOF");
332         curBuf = buffers[curBufIndex];
333         curBuf.position(0);
334         curAvail = bufSizes[curBufIndex];
335       }
336       curBuf.get(b, offset, len);
337       curAvail -= len;
338     }
339   
340     public long getFilePointer() {
341       return ((long) curBufIndex * maxBufSize) + curBuf.position();
342     }
343   
344     public void seek(long pos) throws IOException {
345       curBufIndex = (int) (pos / maxBufSize);
346       curBuf = buffers[curBufIndex];
347       int bufOffset = (int) (pos - ((long) curBufIndex * maxBufSize));
348       curBuf.position(bufOffset);
349       curAvail = bufSizes[curBufIndex] - bufOffset;
350     }
351   
352     public long length() {
353       return length;
354     }
355   
356     public Object clone() {
357       MultiMMapIndexInput clone = (MultiMMapIndexInput)super.clone();
358       clone.isClone = true;
359       clone.buffers = new ByteBuffer[buffers.length];
360       // No need to clone bufSizes.
361       // Since most clones will use only one buffer, duplicate() could also be
362       // done lazy in clones, e.g. when adapting curBuf.
363       for (int bufNr = 0; bufNr < buffers.length; bufNr++) {
364         clone.buffers[bufNr] = buffers[bufNr].duplicate();
365       }
366       try {
367         clone.seek(getFilePointer());
368       } catch(IOException ioe) {
369         RuntimeException newException = new RuntimeException(ioe);
370         newException.initCause(ioe);
371         throw newException;
372       };
373       return clone;
374     }
375   
376     public void close() throws IOException {
377       if (isClone || buffers == null) return;
378       try {
379         for (int bufNr = 0; bufNr < buffers.length; bufNr++) {
380           // unmap the buffer (if enabled) and at least unset it for GC
381           try {
382             cleanMapping(buffers[bufNr]);
383           } finally {
384             buffers[bufNr] = null;
385           }
386         }
387       } finally {
388         buffers = null;
389       }
390     }
391   }

393   /** Creates an IndexInput for the file with the given name. */
394   public IndexInput openInput(String name, int bufferSize) throws IOException {
395     ensureOpen();
396     File f =  new File(getFile(), name);
397     RandomAccessFile raf = new RandomAccessFile(f, "r");
398     try {
399       return (raf.length() <= (long) maxBBuf)
400              ? (IndexInput) new MMapIndexInput(raf)
401              : (IndexInput) new MultiMMapIndexInput(raf, maxBBuf);
402     } finally {
403       raf.close();
404     }
405   }

409     initOutput(name);
410     return new SimpleFSDirectory.SimpleFSIndexOutput(new File(directory, name));
