22  /**
23   * A memory-resident {@link IndexInput} implementation.
24   * 
25   * @version $Id: RAMInputStream.java 632120 2008-02-28 21:13:59Z mikemccand $
26   */
27  
28  class RAMInputStream extends IndexInput implements Cloneable {
29    static final int BUFFER_SIZE = RAMOutputStream.BUFFER_SIZE;
30  
31    private RAMFile file;
32    private long length;
33  
34    private byte[] currentBuffer;
35    private int currentBufferIndex;
36    
37    private int bufferPosition;
38    private long bufferStart;
39    private int bufferLength;
40  
41    RAMInputStream(RAMFile f) throws IOException {
42      file = f;
43      length = file.length;
44      if (length/BUFFER_SIZE >= Integer.MAX_VALUE) {
45        throw new IOException("Too large RAMFile! "+length); 
46      }
47  
48      // make sure that we switch to the
49      // first needed buffer lazily
50      currentBufferIndex = -1;
51      currentBuffer = null;
52    }
53  
54    public void close() {
55      // nothing to do here
56    }
57  
58    public long length() {
59      return length;
60    }
61  
62    public byte readByte() throws IOException {
63      if (bufferPosition >= bufferLength) {
64        currentBufferIndex++;
65        switchCurrentBuffer(true);
66      }
67      return currentBuffer[bufferPosition++];
68    }
69  
70    public void readBytes(byte[] b, int offset, int len) throws IOException {
71      while (len > 0) {
72        if (bufferPosition >= bufferLength) {
73          currentBufferIndex++;
74          switchCurrentBuffer(true);
75        }
76  
77        int remainInBuffer = bufferLength - bufferPosition;
78        int bytesToCopy = len < remainInBuffer ? len : remainInBuffer;
79        System.arraycopy(currentBuffer, bufferPosition, b, offset, bytesToCopy);
80        offset += bytesToCopy;
81        len -= bytesToCopy;
82        bufferPosition += bytesToCopy;
83      }
84    }
85  
86    private final void switchCurrentBuffer(boolean enforceEOF) throws IOException {
87      if (currentBufferIndex >= file.numBuffers()) {
88        // end of file reached, no more buffers left
89        if (enforceEOF)
90          throw new IOException("Read past EOF");
91        else {
92          // Force EOF if a read takes place at this position
93          currentBufferIndex--;
94          bufferPosition = BUFFER_SIZE;
95        }
96      } else {
97        currentBuffer = (byte[]) file.getBuffer(currentBufferIndex);
98        bufferPosition = 0;
99        bufferStart = (long) BUFFER_SIZE * (long) currentBufferIndex;
100       long buflen = length - bufferStart;
101       bufferLength = buflen > BUFFER_SIZE ? BUFFER_SIZE : (int) buflen;
102     }
103   }
104 
105   public long getFilePointer() {
106     return currentBufferIndex < 0 ? 0 : bufferStart + bufferPosition;
107   }
108 
109   public void seek(long pos) throws IOException {
110     if (currentBuffer==null || pos < bufferStart || pos >= bufferStart + BUFFER_SIZE) {
111       currentBufferIndex = (int) (pos / BUFFER_SIZE);
112       switchCurrentBuffer(false);
113     }
114     bufferPosition = (int) (pos % BUFFER_SIZE);
115   }
116 }
