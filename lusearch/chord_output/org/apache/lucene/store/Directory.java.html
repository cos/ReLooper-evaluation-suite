20  import java.io.IOException;
21  
22  import org.apache.lucene.index.IndexFileNameFilter;
23  
24  /** A Directory is a flat list of files.  Files may be written once, when they
25   * are created.  Once a file is created it may only be opened for read, or
26   * deleted.  Random access is permitted both when reading and writing.
27   *
28   * <p> Java's i/o APIs not used directly, but rather all i/o is
29   * through this API.  This permits things such as: <ul>
30   * <li> implementation of RAM-based indices;
31   * <li> implementation indices stored in a database, via JDBC;
32   * <li> implementation of an index as a single file;
33   * </ul>
34   *
35   * Directory locking is implemented by an instance of {@link
36   * LockFactory}, and can be changed for each Directory
37   * instance using {@link #setLockFactory}.
38   *
39   */
40  public abstract class Directory {
41  
42    volatile protected boolean isOpen = true;
43  
44    /** Holds the LockFactory instance (implements locking for
45     * this Directory instance). */
46    protected LockFactory lockFactory;
47  
48    /** List the files in the directory.
49     * 
50     *  @deprecated For some Directory implementations ({@link
51     *  FSDirectory}, and its subclasses), this method
52     *  silently filters its results to include only index
53     *  files.  Please use {@link #listAll} instead, which
54     *  does no filtering. */
55    public abstract String[] list()
56         throws IOException;
57  
58    /** Returns an array of strings, one for each file in the
59     *  directory.  Unlike {@link #list} this method does no
60     *  filtering of the contents in a directory, and it will
61     *  never return null (throws IOException instead).
62     *
63     *  Currently this method simply falls back to {@link
64     *  #list} for Directory impls outside of Lucene's core &
65     *  contrib, but in 3.0 that method will be removed and
66     *  this method will become abstract. */
67    public String[] listAll()
68      throws IOException
69    {
70      return list();
71    }
72  
73    /** Returns true iff a file with the given name exists. */
74    public abstract boolean fileExists(String name)
75         throws IOException;
76  
77    /** Returns the time the named file was last modified. */
78    public abstract long fileModified(String name)
79         throws IOException;
80  
81    /** Set the modified time of an existing file to now. */
82    public abstract void touchFile(String name)
83         throws IOException;
84  
85    /** Removes an existing file in the directory. */
86    public abstract void deleteFile(String name)
87         throws IOException;
88  
89    /** Renames an existing file in the directory.
90     * If a file already exists with the new name, then it is replaced.
91     * This replacement is not guaranteed to be atomic.
92     * @deprecated 
93     */
94    public abstract void renameFile(String from, String to)
95         throws IOException;
96  
97    /** Returns the length of a file in the directory. */
98    public abstract long fileLength(String name)
99         throws IOException;
100 
101 
102   /** Creates a new, empty file in the directory with the given name.
103       Returns a stream writing this file. */
104   public abstract IndexOutput createOutput(String name) throws IOException;
105 
106   /** Ensure that any writes to this file are moved to
107    *  stable storage.  Lucene uses this to properly commit
108    *  changes to the index, to prevent a machine/OS crash
109    *  from corrupting the index. */
110   public void sync(String name) throws IOException {}
111 
112   /** Returns a stream reading an existing file. */
113   public abstract IndexInput openInput(String name)
114     throws IOException;
115 
116   /** Returns a stream reading an existing file, with the
117    * specified read buffer size.  The particular Directory
118    * implementation may ignore the buffer size.  Currently
119    * the only Directory implementations that respect this
120    * parameter are {@link FSDirectory} and {@link
121    * org.apache.lucene.index.CompoundFileReader}.
122   */
123   public IndexInput openInput(String name, int bufferSize) throws IOException {
124     return openInput(name);
125   }
126 
127   /** Construct a {@link Lock}.
128    * @param name the name of the lock file
129    */
130   public Lock makeLock(String name) {
131       return lockFactory.makeLock(name);
132   }
133   /**
134    * Attempt to clear (forcefully unlock and remove) the
135    * specified lock.  Only call this at a time when you are
136    * certain this lock is no longer in use.
137    * @param name name of the lock to be cleared.
138    */
139   public void clearLock(String name) throws IOException {
140     if (lockFactory != null) {
141       lockFactory.clearLock(name);
142     }
143   }
144 
145   /** Closes the store. */
146   public abstract void close()
147        throws IOException;
148 
149   /**
150    * Set the LockFactory that this Directory instance should
151    * use for its locking implementation.  Each * instance of
152    * LockFactory should only be used for one directory (ie,
153    * do not share a single instance across multiple
154    * Directories).
155    *
156    * @param lockFactory instance of {@link LockFactory}.
157    */
158   public void setLockFactory(LockFactory lockFactory) {
159       this.lockFactory = lockFactory;
160       lockFactory.setLockPrefix(this.getLockID());
161   }
162 
163   /**
164    * Get the LockFactory that this Directory instance is
165    * using for its locking implementation.  Note that this
166    * may be null for Directory implementations that provide
167    * their own locking implementation.
168    */
169   public LockFactory getLockFactory() {
170       return this.lockFactory;
171   }
172 
173   /**
174    * Return a string identifier that uniquely differentiates
175    * this Directory instance from other Directory instances.
176    * This ID should be the same if two Directory instances
177    * (even in different JVMs and/or on different machines)
178    * are considered "the same index".  This is how locking
179    * "scopes" to the right index.
180    */
181   public String getLockID() {
182       return this.toString();
183   }
184 
185   /**
186    * Copy contents of a directory src to a directory dest.
187    * If a file in src already exists in dest then the
188    * one in dest will be blindly overwritten.
189    *
190    * <p><b>NOTE:</b> the source directory cannot change
191    * while this method is running.  Otherwise the results
192    * are undefined and you could easily hit a
193    * FileNotFoundException.
194    *
195    * <p><b>NOTE:</b> this method only copies files that look
196    * like index files (ie, have extensions matching the
197    * known extensions of index files).
198    *
199    * @param src source directory
200    * @param dest destination directory
201    * @param closeDirSrc if <code>true</code>, call {@link #close()} method on source directory
202    * @throws IOException
203    */
204   public static void copy(Directory src, Directory dest, boolean closeDirSrc) throws IOException {
205     final String[] files = src.listAll();
206 
207     IndexFileNameFilter filter = IndexFileNameFilter.getFilter();
208 
209     byte[] buf = new byte[BufferedIndexOutput.BUFFER_SIZE];
210     for (int i = 0; i < files.length; i++) {
211 
212       if (!filter.accept(null, files[i]))
213         continue;
214 
215       IndexOutput os = null;
216       IndexInput is = null;
217       try {
218         // create file in dest directory
219         os = dest.createOutput(files[i]);
220         // read current file
221         is = src.openInput(files[i]);
222         // and copy to dest directory
223         long len = is.length();
224         long readCount = 0;
225         while (readCount < len) {
226           int toRead = readCount + BufferedIndexOutput.BUFFER_SIZE > len ? (int)(len - readCount) : BufferedIndexOutput.BUFFER_SIZE;
227           is.readBytes(buf, 0, toRead);
228           os.writeBytes(buf, toRead);
229           readCount += toRead;
230         }
231       } finally {
232         // graceful cleanup
233         try {
234           if (os != null)
235             os.close();
236         } finally {
237           if (is != null)
238             is.close();
239         }
240       }
241     }
242     if(closeDirSrc)
243       src.close();
244   }
245 
246   /**
247    * @throws AlreadyClosedException if this Directory is closed
248    */
249   protected final void ensureOpen() throws AlreadyClosedException {
250     if (!isOpen)
251       throw new AlreadyClosedException("this Directory is closed");
252   }
