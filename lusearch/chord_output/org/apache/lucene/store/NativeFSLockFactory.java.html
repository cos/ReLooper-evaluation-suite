20  import java.nio.channels.FileChannel;
21  import java.nio.channels.FileLock;
22  import java.io.File;
23  import java.io.RandomAccessFile;
24  import java.io.IOException;
25  import java.util.HashSet;
26  import java.util.Random;
27  
28  /**
29   * <p>Implements {@link LockFactory} using native OS file
30   * locks.  Note that because this LockFactory relies on
31   * java.nio.* APIs for locking, any problems with those APIs
32   * will cause locking to fail.  Specifically, on certain NFS
33   * environments the java.nio.* locks will fail (the lock can
34   * incorrectly be double acquired) whereas {@link
35   * SimpleFSLockFactory} worked perfectly in those same
36   * environments.  For NFS based access to an index, it's
37   * recommended that you try {@link SimpleFSLockFactory}
38   * first and work around the one limitation that a lock file
39   * could be left when the JVM exits abnormally.</p>
40   *
41   * <p>The primary benefit of {@link NativeFSLockFactory} is
42   * that lock files will be properly removed (by the OS) if
43   * the JVM has an abnormal exit.</p>
44   * 
45   * <p>Note that, unlike {@link SimpleFSLockFactory}, the existence of
46   * leftover lock files in the filesystem on exiting the JVM
47   * is fine because the OS will free the locks held against
48   * these files even though the files still remain.</p>
49   *
50   * <p>If you suspect that this or any other LockFactory is
51   * not working properly in your environment, you can easily
52   * test it by using {@link VerifyingLockFactory}, {@link
53   * LockVerifyServer} and {@link LockStressTest}.</p>
54   *
55   * @see LockFactory
56   */
57  
58  public class NativeFSLockFactory extends FSLockFactory {
59  
60    private volatile boolean tested = false;
61  
62    // Simple test to verify locking system is "working".  On
63    // NFS, if it's misconfigured, you can hit long (35
64    // second) timeouts which cause Lock.obtain to take far
65    // too long (it assumes the obtain() call takes zero
66    // time). 
67    private synchronized void acquireTestLock() {
68      if (tested) return;
69      tested = true;
70      
71      // Ensure that lockDir exists and is a directory.
72      if (!lockDir.exists()) {
73        if (!lockDir.mkdirs())
74          throw new RuntimeException("Cannot create directory: " +
75                                lockDir.getAbsolutePath());
76      } else if (!lockDir.isDirectory()) {
77        throw new RuntimeException("Found regular file where directory expected: " + 
78                              lockDir.getAbsolutePath());
79      }
80  
81      String randomLockName = "lucene-" + Long.toString(new Random().nextInt(), Character.MAX_RADIX) + "-test.lock";
82      
83      Lock l = makeLock(randomLockName);
84      try {
85        l.obtain();
86        l.release();
87      } catch (IOException e) {
88        RuntimeException e2 = new RuntimeException("Failed to acquire random test lock; please verify filesystem for lock directory '" + lockDir + "' supports locking");
89        e2.initCause(e);
90        throw e2;
91      }    
92    }
93  
94    /**
95     * Create a NativeFSLockFactory instance, with null (unset)
96     * lock directory. When you pass this factory to a {@link FSDirectory}
97     * subclass, the lock directory is automatically set to the
98     * directory itsself. Be sure to create one instance for each directory
99     * your create!
100    */
101   public NativeFSLockFactory() throws IOException {
102     this((File) null);
103   }
104 
105   /**
106    * Create a NativeFSLockFactory instance, storing lock
107    * files into the specified lockDirName:
108    *
109    * @param lockDirName where lock files are created.
110    */
111   public NativeFSLockFactory(String lockDirName) throws IOException {
112     this(new File(lockDirName));
113   }
114 
115   /**
116    * Create a NativeFSLockFactory instance, storing lock
117    * files into the specified lockDir:
118    * 
119    * @param lockDir where lock files are created.
120    */
121   public NativeFSLockFactory(File lockDir) throws IOException {
122     setLockDir(lockDir);
123   }
124 
125   public synchronized Lock makeLock(String lockName) {
126     acquireTestLock();
127     if (lockPrefix != null)
128       lockName = lockPrefix + "-" + lockName;
129     return new NativeFSLock(lockDir, lockName);
130   }
131 
132   public void clearLock(String lockName) throws IOException {
133     // Note that this isn't strictly required anymore
134     // because the existence of these files does not mean
135     // they are locked, but, still do this in case people
136     // really want to see the files go away:
137     if (lockDir.exists()) {
138       if (lockPrefix != null) {
139         lockName = lockPrefix + "-" + lockName;
140       }
141       File lockFile = new File(lockDir, lockName);
142       if (lockFile.exists() && !lockFile.delete()) {
143         throw new IOException("Cannot delete " + lockFile);
144       }
145     }
146   }
147 };
148 
149 class NativeFSLock extends Lock {
150 
151   private RandomAccessFile f;
152   private FileChannel channel;
153   private FileLock lock;
154   private File path;
155   private File lockDir;
156 
157   /*
158    * The javadocs for FileChannel state that you should have
159    * a single instance of a FileChannel (per JVM) for all
160    * locking against a given file.  To ensure this, we have
161    * a single (static) HashSet that contains the file paths
162    * of all currently locked locks.  This protects against
163    * possible cases where different Directory instances in
164    * one JVM (each with their own NativeFSLockFactory
165    * instance) have set the same lock dir and lock prefix.
166    */
167   private static HashSet LOCK_HELD = new HashSet();
168 
169   public NativeFSLock(File lockDir, String lockFileName) {
170     this.lockDir = lockDir;
171     path = new File(lockDir, lockFileName);
172   }
173 
174   private synchronized boolean lockExists() {
175     return lock != null;
176   }
177 
178   public synchronized boolean obtain() throws IOException {
179 
180     if (lockExists()) {
181       // Our instance is already locked:
182       return false;
183     }
184 
185     // Ensure that lockDir exists and is a directory.
186     if (!lockDir.exists()) {
187       if (!lockDir.mkdirs())
188         throw new IOException("Cannot create directory: " +
189                               lockDir.getAbsolutePath());
190     } else if (!lockDir.isDirectory()) {
191       throw new IOException("Found regular file where directory expected: " + 
192                             lockDir.getAbsolutePath());
193     }
194 
195     String canonicalPath = path.getCanonicalPath();
196 
197     boolean markedHeld = false;
198 
199     try {
200 
201       // Make sure nobody else in-process has this lock held
202       // already, and, mark it held if not:
203 
204       synchronized(LOCK_HELD) {
205         if (LOCK_HELD.contains(canonicalPath)) {
206           // Someone else in this JVM already has the lock:
207           return false;
208         } else {
209           // This "reserves" the fact that we are the one
210           // thread trying to obtain this lock, so we own
211           // the only instance of a channel against this
212           // file:
213           LOCK_HELD.add(canonicalPath);
214           markedHeld = true;
215         }
216       }
217 
218       try {
219         f = new RandomAccessFile(path, "rw");
220       } catch (IOException e) {
221         // On Windows, we can get intermittent "Access
222         // Denied" here.  So, we treat this as failure to
223         // acquire the lock, but, store the reason in case
224         // there is in fact a real error case.
225         failureReason = e;
226         f = null;
227       }
228 
229       if (f != null) {
230         try {
231           channel = f.getChannel();
232           try {
233             lock = channel.tryLock();
234           } catch (IOException e) {
235             // At least on OS X, we will sometimes get an
236             // intermittent "Permission Denied" IOException,
237             // which seems to simply mean "you failed to get
238             // the lock".  But other IOExceptions could be
239             // "permanent" (eg, locking is not supported via
240             // the filesystem).  So, we record the failure
241             // reason here; the timeout obtain (usually the
242             // one calling us) will use this as "root cause"
243             // if it fails to get the lock.
244             failureReason = e;
245           } finally {
246             if (lock == null) {
247               try {
248                 channel.close();
249               } finally {
250                 channel = null;
251               }
252             }
253           }
254         } finally {
255           if (channel == null) {
256             try {
257               f.close();
258             } finally {
259               f = null;
260             }
261           }
262         }
263       }
264 
265     } finally {
266       if (markedHeld && !lockExists()) {
267         synchronized(LOCK_HELD) {
268           if (LOCK_HELD.contains(canonicalPath)) {
269             LOCK_HELD.remove(canonicalPath);
270           }
271         }
272       }
273     }
274     return lockExists();
275   }
276 
277   public synchronized void release() throws IOException {
278     if (lockExists()) {
279       try {
280         lock.release();
281       } finally {
282         lock = null;
283         try {
284           channel.close();
285         } finally {
286           channel = null;
287           try {
288             f.close();
289           } finally {
290             f = null;
291             synchronized(LOCK_HELD) {
292               LOCK_HELD.remove(path.getCanonicalPath());
293             }
294           }
295         }
296       }
297       if (!path.delete())
298         throw new LockReleaseFailedException("failed to delete " + path);
299     }
300   }
301 
302   public synchronized boolean isLocked() {
303     // The test for is isLocked is not directly possible with native file locks:
304     
305     // First a shortcut, if a lock reference in this instance is available
306     if (lockExists()) return true;
307     
308     // Look if lock file is present; if not, there can definitely be no lock!
309     if (!path.exists()) return false;
310     
311     // Try to obtain and release (if was locked) the lock
312     try {
313       boolean obtained = obtain();
314       if (obtained) release();
315       return !obtained;
316     } catch (IOException ioe) {
317       return false;
318     }    
319   }
320 
321   public String toString() {
322     return "NativeFSLock@" + path;
323   }
324 }
