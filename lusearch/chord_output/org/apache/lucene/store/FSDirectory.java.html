37  /**
38   * <a name="subclasses"/>
39   * Base class for Directory implementations that store index
40   * files in the file system.  There are currently three core
41   * subclasses:
42   *
43   * <ul>
44   *
45   *  <li> {@link SimpleFSDirectory} is a straightforward
46   *       implementation using java.io.RandomAccessFile.
47   *       However, it has poor concurrent performance
48   *       (multiple threads will bottleneck) as it
49   *       synchronizes when multiple threads read from the
50   *       same file.
51   *
52   *  <li> {@link NIOFSDirectory} uses java.nio's
53   *       FileChannel's positional io when reading to avoid
54   *       synchronization when reading from the same file.
55   *       Unfortunately, due to a Windows-only <a
56   *       href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6265734">Sun
57   *       JRE bug</a> this is a poor choice for Windows, but
58   *       on all other platforms this is the preferred
59   *       choice.
60   *
61   *  <li> {@link MMapDirectory} uses memory-mapped IO when
62   *       reading. This is a good choice if you have plenty
63   *       of virtual memory relative to your index size, eg
64   *       if you are running on a 64 bit JRE, or you are
65   *       running on a 32 bit JRE but your index sizes are
66   *       small enough to fit into the virtual memory space.
67   *       Java has currently the limitation of not being able to
68   *       unmap files from user code. The files are unmapped, when GC
69   *       releases the byte buffers. Due to
70   *       <a href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4724038">
71   *       this bug</a> in Sun's JRE, MMapDirectory's {@link IndexInput#close}
72   *       is unable to close the underlying OS file handle. Only when
73   *       GC finally collects the underlying objects, which could be
74   *       quite some time later, will the file handle be closed.
75   *       This will consume additional transient disk usage: on Windows,
76   *       attempts to delete or overwrite the files will result in an
77   *       exception; on other platforms, which typically have a &quot;delete on
78   *       last close&quot; semantics, while such operations will succeed, the bytes
79   *       are still consuming space on disk.  For many applications this
80   *       limitation is not a problem (e.g. if you have plenty of disk space,
81   *       and you don't rely on overwriting files on Windows) but it's still
82   *       an important limitation to be aware of. This class supplies a
83   *       (possibly dangerous) workaround mentioned in the bug report,
84   *       which may fail on non-Sun JVMs.
85   * </ul>
86   *
87   * Unfortunately, because of system peculiarities, there is
88   * no single overall best implementation.  Therefore, we've
89   * added the {@link #open} method, to allow Lucene to choose
90   * the best FSDirectory implementation given your
91   * environment, and the known limitations of each
92   * implementation.  For users who have no reason to prefer a
93   * specific implementation, it's best to simply use {@link
94   * #open}.  For all others, you should instantiate the
95   * desired implementation directly.
96   *
97   * <p>The locking implementation is by default {@link
98   * NativeFSLockFactory}, but can be changed by
99   * passing in a custom {@link LockFactory} instance.
100  * The deprecated <code>getDirectory</code> methods default to use
101  * {@link SimpleFSLockFactory} for backwards compatibility.
102  * The system properties 
103  * <code>org.apache.lucene.store.FSDirectoryLockFactoryClass</code>
104  * and <code>org.apache.lucene.FSDirectory.class</code>
105  * are deprecated and only used by the deprecated
106  * <code>getDirectory</code> methods. The system property
107  * <code>org.apache.lucene.lockDir</code> is ignored completely,
108  * If you really want to store locks
109  * elsewhere, you can create your own {@link
110  * SimpleFSLockFactory} (or {@link NativeFSLockFactory},
111  * etc.) passing in your preferred lock directory.
112  *
113  * <p><em>In 3.0 this class will become abstract.</em>
114  *
115  * @see Directory
116  */

248   /** Returns the directory instance for the named location.
249    *
250    * @deprecated Use {@link #open(File, LockFactory)}
251    *
252    * @param file the path to the directory.
253    * @param lockFactory instance of {@link LockFactory} providing the
254    *        locking implementation.
255    * @return the FSDirectory for the named file.  */
256   public static FSDirectory getDirectory(File file, LockFactory lockFactory)
257     throws IOException
258   {
259     file = getCanonicalPath(file);
260 
261     FSDirectory dir;
262     synchronized (DIRECTORIES) {
263       dir = (FSDirectory)DIRECTORIES.get(file);
264       if (dir == null) {
265         try {
266           dir = (FSDirectory)IMPL.newInstance();
267         } catch (Exception e) {
268           throw new RuntimeException("cannot load FSDirectory class: " + e.toString(), e);
269         }
270         dir.init(file, lockFactory);
271         DIRECTORIES.put(file, dir);
272       } else {
273         // Catch the case where a Directory is pulled from the cache, but has a
274         // different LockFactory instance.
275         if (lockFactory != null && lockFactory != dir.getLockFactory()) {
276           throw new IOException("Directory was previously created with a different LockFactory instance; please pass null as the lockFactory instance and use setLockFactory to change it");
277         }
278         dir.checked = false;
279       }
280     }
281     synchronized (dir) {
282       dir.refCount++;
283     }
284     return dir;
285   }

324   private void create() throws IOException {
325     if (directory.exists()) {
326       String[] files = directory.list(IndexFileNameFilter.getFilter());            // clear old files
327       if (files == null)
328         throw new IOException("cannot read directory " + directory.getAbsolutePath() + ": list() returned null");
329       for (int i = 0; i < files.length; i++) {
330         File file = new File(directory, files[i]);
331         if (!file.delete())
332           throw new IOException("Cannot delete " + file);
333       }
334     }
335     lockFactory.clearLock(IndexWriter.WRITE_LOCK_NAME);
336   }

391    *  best implementation given the current environment.
392    *  The directory returned uses the {@link NativeFSLockFactory}.
393    *
394    *  <p>Currently this returns {@link NIOFSDirectory}
395    *  on non-Windows JREs and {@link SimpleFSDirectory}
396    *  on Windows.

398    * <p><b>NOTE</b>: this method may suddenly change which
399    * implementation is returned from release to release, in
400    * the event that higher performance defaults become

402    * your application, please instantiate it directly,
403    * instead. On 64 bit systems, it may also good to
404    * return {@link MMapDirectory}, but this is disabled
405    * because of officially missing unmap support in Java.

414   /** Just like {@link #open(File)}, but allows you to
415    *  also specify a custom {@link LockFactory}. */
416   public static FSDirectory open(File path, LockFactory lockFactory) throws IOException {
417     /* For testing:
418     MMapDirectory dir=new MMapDirectory(path, lockFactory);
419     dir.setUseUnmap(true);
420     return dir;
421     */
422 
423     if (Constants.WINDOWS) {
424       return new SimpleFSDirectory(path, lockFactory);
425     } else {
426       return new NIOFSDirectory(path, lockFactory);
427     }
428   }

430   /* will move to ctor, when reflection is removed in 3.0 */
431   private void init(File path, LockFactory lockFactory) throws IOException {
432 
433     // Set up lockFactory with cascaded defaults: if an instance was passed in,
434     // use that; else if locks are disabled, use NoLockFactory; else if the
435     // system property org.apache.lucene.store.FSDirectoryLockFactoryClass is set,
436     // instantiate that; else, use SimpleFSLockFactory:
437 
438     directory = path;
439 
440     if (directory.exists() && !directory.isDirectory())
441       throw new NoSuchDirectoryException("file '" + directory + "' exists but is not a directory");
442 
443     if (lockFactory == null) {
444 
445       if (disableLocks) {
446         // Locks are disabled:
447         lockFactory = NoLockFactory.getNoLockFactory();
448       } else {
449         String lockClassName = System.getProperty("org.apache.lucene.store.FSDirectoryLockFactoryClass");
450 
451         if (lockClassName != null && !lockClassName.equals("")) {
452           Class c;
453 
454           try {
455             c = Class.forName(lockClassName);
456           } catch (ClassNotFoundException e) {
457             throw new IOException("unable to find LockClass " + lockClassName);
458           }
459 
460           try {
461             lockFactory = (LockFactory) c.newInstance();          
462           } catch (IllegalAccessException e) {
463             throw new IOException("IllegalAccessException when instantiating LockClass " + lockClassName);
464           } catch (InstantiationException e) {
465             throw new IOException("InstantiationException when instantiating LockClass " + lockClassName);
466           } catch (ClassCastException e) {
467             throw new IOException("unable to cast LockClass " + lockClassName + " instance to a LockFactory");
468           }
469         } else {
470           // Our default lock is SimpleFSLockFactory;
471           // default lockDir is our index directory:
472           lockFactory = new SimpleFSLockFactory();
473         }
474       }
475     }
476 
477     setLockFactory(lockFactory);
478     
479     // for filesystem based LockFactory, delete the lockPrefix, if the locks are placed
480     // in index dir. If no index dir is given, set ourselves
481     if (lockFactory instanceof FSLockFactory) {
482       final FSLockFactory lf = (FSLockFactory) lockFactory;
483       final File dir = lf.getLockDir();
484       // if the lock factory has no lockDir set, use the this directory as lockDir
485       if (dir == null) {
486         lf.setLockDir(this.directory);
487         lf.setLockPrefix(null);
488       } else if (dir.getCanonicalPath().equals(this.directory.getCanonicalPath())) {
489         lf.setLockPrefix(null);
490       }
491     }
492   }

494   /** Lists all files (not subdirectories) in the
495    *  directory.  This method never returns null (throws
496    *  {@link IOException} instead).
497    *
498    *  @throws NoSuchDirectoryException if the directory
499    *   does not exist, or does exist but is not a
500    *   directory.
501    *  @throws IOException if list() returns null */
502   public static String[] listAll(File dir) throws IOException {
503     if (!dir.exists())
504       throw new NoSuchDirectoryException("directory '" + dir + "' does not exist");
505     else if (!dir.isDirectory())
506       throw new NoSuchDirectoryException("file '" + dir + "' exists but is not a directory");
507 
508     // Exclude subdirs
509     String[] result = dir.list(new FilenameFilter() {
510         public boolean accept(File dir, String file) {
511           return !new File(dir, file).isDirectory();
512         }
513       });
514 
515     if (result == null)
516       throw new IOException("directory '" + dir + "' exists and is a directory, but cannot be listed: list() returned null");
517 
518     return result;
519   }

576   /** Renames an existing file in the directory. 
577    * Warning: This is not atomic.
578    * @deprecated 
579    */
580   public synchronized void renameFile(String from, String to)
581       throws IOException {
582     ensureOpen();
583     File old = new File(directory, from);
584     File nu = new File(directory, to);
585 
586     /* This is not atomic.  If the program crashes between the call to
587        delete() and the call to renameTo() then we're screwed, but I've
588        been unable to figure out how else to do this... */
589 
590     if (nu.exists())
591       if (!nu.delete())
592         throw new IOException("Cannot delete " + nu);
593 
594     // Rename the old file to the new one. Unfortunately, the renameTo()
595     // method does not work reliably under some JVMs.  Therefore, if the
596     // rename fails, we manually rename by copying the old file to the new one
597     if (!old.renameTo(nu)) {
598       java.io.InputStream in = null;
599       java.io.OutputStream out = null;
600       try {
601         in = new FileInputStream(old);
602         out = new FileOutputStream(nu);
603         // see if the buffer needs to be initialized. Initialization is
604         // only done on-demand since many VM's will never run into the renameTo
605         // bug and hence shouldn't waste 1K of mem for no reason.
606         if (buffer == null) {
607           buffer = new byte[1024];
608         }
609         int len;
610         while ((len = in.read(buffer)) >= 0) {
611           out.write(buffer, 0, len);
612         }
613 
614         // delete the old file.
615         old.delete();
616       }
617       catch (IOException ioe) {
618         IOException newExc = new IOException("Cannot rename " + old + " to " + nu);
619         newExc.initCause(ioe);
620         throw newExc;
621       }
622       finally {
623         try {
624           if (in != null) {
625             try {
626               in.close();
627             } catch (IOException e) {
628               throw new RuntimeException("Cannot close input stream: " + e.toString(), e);
629             }
630           }
631         } finally {
632           if (out != null) {
633             try {
634               out.close();
635             } catch (IOException e) {
636               throw new RuntimeException("Cannot close output stream: " + e.toString(), e);
637             }
638           }
639         }
640       }
641     }
642   }

651   public void sync(String name) throws IOException {
652     ensureOpen();
653     File fullFile = new File(directory, name);
654     boolean success = false;
655     int retryCount = 0;
656     IOException exc = null;
657     while(!success && retryCount < 5) {
658       retryCount++;
659       RandomAccessFile file = null;
660       try {
661         try {
662           file = new RandomAccessFile(fullFile, "rw");
663           file.getFD().sync();
664           success = true;
665         } finally {
666           if (file != null)
667             file.close();
668         }
669       } catch (IOException ioe) {
670         if (exc == null)
671           exc = ioe;
672         try {
673           // Pause 5 msec
674           Thread.sleep(5);
675         } catch (InterruptedException ie) {
676           // In 3.0 we will change this to throw
677           // InterruptedException instead
678           Thread.currentThread().interrupt();
679           throw new RuntimeException(ie);
680         }
681       }
682     }
683     if (!success)
684       // Throw original exception
685       throw exc;
686   }

708   public String getLockID() {
709     ensureOpen();
710     String dirName;                               // name to be hashed
711     try {
712       dirName = directory.getCanonicalPath();
713     } catch (IOException e) {
714       throw new RuntimeException(e.toString(), e);
715     }
716 
717     byte digest[];
718     synchronized (DIGESTER) {
719       digest = DIGESTER.digest(dirName.getBytes());
720     }
721     StringBuffer buf = new StringBuffer();
722     buf.append("lucene-");
723     for (int i = 0; i < digest.length; i++) {
724       int b = digest[i];
725       buf.append(HEX_DIGITS[(b >> 4) & 0xf]);
726       buf.append(HEX_DIGITS[b & 0xf]);
727     }
728 
729     return buf.toString();
730   }

763   /**
764    * Sets the maximum number of bytes read at once from the
765    * underlying file during {@link IndexInput#readBytes}.
766    * The default value is {@link #DEFAULT_READ_CHUNK_SIZE};
767    *
768    * <p> This was introduced due to <a
769    * href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6478546">Sun
770    * JVM Bug 6478546</a>, which throws an incorrect
771    * OutOfMemoryError when attempting to read too many bytes
772    * at once.  It only happens on 32bit JVMs with a large
773    * maximum heap size.</p>
774    *
775    * <p>Changes to this value will not impact any
776    * already-opened {@link IndexInput}s.  You should call
777    * this before attempting to open an index on the
778    * directory.</p>
779    *
780    * <p> <b>NOTE</b>: This value should be as large as
781    * possible to reduce any possible performance impact.  If
782    * you still encounter an incorrect OutOfMemoryError,
783    * trying lowering the chunk size.</p>
784    */
785   public final void setReadChunkSize(int chunkSize) {
786     // LUCENE-1566
787     if (chunkSize <= 0) {
788       throw new IllegalArgumentException("chunkSize must be positive");
789     }
790     if (!Constants.JRE_IS_64BIT) {
791       this.chunkSize = chunkSize;
792     }
793   }
