20  import java.util.ArrayList;
21  import java.io.Serializable;
22  
23  class RAMFile implements Serializable {
24  
25    private static final long serialVersionUID = 1l;
26  
27    private ArrayList buffers = new ArrayList();
28    long length;
29    RAMDirectory directory;
30    long sizeInBytes;
31  
32    // This is publicly modifiable via Directory.touchFile(), so direct access not supported
33    private long lastModified = System.currentTimeMillis();
34  
35    // File used as buffer, in no RAMDirectory
36    RAMFile() {}
37    
38    RAMFile(RAMDirectory directory) {
39      this.directory = directory;
40    }
41  
42    // For non-stream access from thread that might be concurrent with writing
43    synchronized long getLength() {
44      return length;
45    }
46  
47    synchronized void setLength(long length) {
48      this.length = length;
49    }
50  
51    // For non-stream access from thread that might be concurrent with writing
52    synchronized long getLastModified() {
53      return lastModified;
54    }
55  
56    synchronized void setLastModified(long lastModified) {
57      this.lastModified = lastModified;
58    }
59  
60    final byte[] addBuffer(int size) {
61      byte[] buffer = newBuffer(size);
62      synchronized(this) {
63        buffers.add(buffer);
64        sizeInBytes += size;
65      }
66  
67      if (directory != null) {
68        synchronized(directory) {
69          directory.sizeInBytes += size;
70        }
71      }
72      return buffer;
73    }
74  
75    final synchronized byte[] getBuffer(int index) {
76      return (byte[]) buffers.get(index);
77    }
78  
79    final synchronized int numBuffers() {
80      return buffers.size();
81    }
82  
83    /**
84     * Expert: allocate a new buffer. 
85     * Subclasses can allocate differently. 
86     * @param size size of allocated buffer.
87     * @return allocated buffer.
88     */
89    byte[] newBuffer(int size) {
90      return new byte[size];
91    }
92  
93    synchronized long getSizeInBytes() {
94      synchronized(directory) {
95        return sizeInBytes;
96      }
97    }
98    
99  }
