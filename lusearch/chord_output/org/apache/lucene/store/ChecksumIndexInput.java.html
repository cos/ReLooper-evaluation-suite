20  import java.io.IOException;
21  import java.util.zip.CRC32;
22  import java.util.zip.Checksum;
23  
24  /** Writes bytes through to a primary IndexOutput, computing
25   *  checksum as it goes. Note that you cannot use seek(). */
26  public class ChecksumIndexInput extends IndexInput {
27    IndexInput main;
28    Checksum digest;
29  
30    public ChecksumIndexInput(IndexInput main) {
31      this.main = main;
32      digest = new CRC32();
33    }
34  
35    public byte readByte() throws IOException {
36      final byte b = main.readByte();
37      digest.update(b);
38      return b;
39    }
40  
41    public void readBytes(byte[] b, int offset, int len)
42      throws IOException {
43      main.readBytes(b, offset, len);
44      digest.update(b, offset, len);
45    }
46  
47    
48    public long getChecksum() {
49      return digest.getValue();
50    }
51  
52    public void close() throws IOException {
53      main.close();
54    }
55  
56    public long getFilePointer() {
57      return main.getFilePointer();
58    }
59  
60    public void seek(long pos) {
61      throw new RuntimeException("not allowed");
62    }
63  
64    public long length() {
65      return main.length();
66    }
67  }
