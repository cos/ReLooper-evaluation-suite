20  import java.io.File;
21  import java.io.IOException;
22  import java.nio.ByteBuffer;
23  import java.nio.channels.FileChannel;
24  
25  /**
26   * An {@link FSDirectory} implementation that uses
27   * java.nio's FileChannel's positional read, which allows
28   * multiple threads to read from the same file without
29   * synchronizing.
30   *
31   * <p>This class only uses FileChannel when reading; writing
32   * is achieved with {@link SimpleFSDirectory.SimpleFSIndexOutput}.
33   * 
34   * <p><b>NOTE</b>: NIOFSDirectory is not recommended on Windows because of a bug
35   * in how FileChannel.read is implemented in Sun's JRE.
36   * Inside of the implementation the position is apparently
37   * synchronized.  See <a
38   * href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6265734">here</a>
39   * for details.
40   */
41  public class NIOFSDirectory extends FSDirectory {
42  
43    /** Create a new NIOFSDirectory for the named location.
44     * 
45     * @param path the path of the directory
46     * @param lockFactory the lock factory to use, or null for the default
47     * ({@link NativeFSLockFactory});
48     * @throws IOException
49     */
50    public NIOFSDirectory(File path, LockFactory lockFactory) throws IOException {
51      super(path, lockFactory);
52    }
53  
54    /** Create a new NIOFSDirectory for the named location and {@link NativeFSLockFactory}.
55     *
56     * @param path the path of the directory
57     * @throws IOException
58     */
59    public NIOFSDirectory(File path) throws IOException {
60      super(path, null);
61    }
62  
63    // back compatibility so FSDirectory can instantiate via reflection
64    /** @deprecated */
65    NIOFSDirectory() {}
66  
67    /** Creates an IndexInput for the file with the given name. */
68    public IndexInput openInput(String name, int bufferSize) throws IOException {
69      ensureOpen();
70      return new NIOFSIndexInput(new File(getFile(), name), bufferSize, getReadChunkSize());
71    }
72  
73    /** Creates an IndexOutput for the file with the given name. */
74    public IndexOutput createOutput(String name) throws IOException {
75      initOutput(name);
76      return new SimpleFSDirectory.SimpleFSIndexOutput(new File(directory, name));
77    }
78  
79    protected static class NIOFSIndexInput extends SimpleFSDirectory.SimpleFSIndexInput {
80  
81      private ByteBuffer byteBuf; // wraps the buffer for NIO
82  
83      private byte[] otherBuffer;
84      private ByteBuffer otherByteBuf;
85  
86      final FileChannel channel;
87  
88      /** @deprecated Please use ctor taking chunkSize */
89      public NIOFSIndexInput(File path, int bufferSize) throws IOException {
90        this(path, bufferSize, FSDirectory.DEFAULT_READ_CHUNK_SIZE);
91      }
92      
93      public NIOFSIndexInput(File path, int bufferSize, int chunkSize) throws IOException {
94        super(path, bufferSize, chunkSize);
95        channel = file.getChannel();
96      }
97  
98      protected void newBuffer(byte[] newBuffer) {
99        super.newBuffer(newBuffer);
100       byteBuf = ByteBuffer.wrap(newBuffer);
101     }
102 
103     public void close() throws IOException {
104       if (!isClone && file.isOpen) {
105         // Close the channel & file
106         try {
107           channel.close();
108         } finally {
109           file.close();
110         }
111       }
112     }
113 
114     protected void readInternal(byte[] b, int offset, int len) throws IOException {
115 
116       final ByteBuffer bb;
117 
118       // Determine the ByteBuffer we should use
119       if (b == buffer && 0 == offset) {
120         // Use our own pre-wrapped byteBuf:
121         assert byteBuf != null;
122         byteBuf.clear();
123         byteBuf.limit(len);
124         bb = byteBuf;
125       } else {
126         if (offset == 0) {
127           if (otherBuffer != b) {
128             // Now wrap this other buffer; with compound
129             // file, we are repeatedly called with its
130             // buffer, so we wrap it once and then re-use it
131             // on subsequent calls
132             otherBuffer = b;
133             otherByteBuf = ByteBuffer.wrap(b);
134           } else
135             otherByteBuf.clear();
136           otherByteBuf.limit(len);
137           bb = otherByteBuf;
138         } else {
139           // Always wrap when offset != 0
140           bb = ByteBuffer.wrap(b, offset, len);
141         }
142       }
143 
144       int readOffset = bb.position();
145       int readLength = bb.limit() - readOffset;
146       assert readLength == len;
147 
148       long pos = getFilePointer();
149 
150       try {
151         while (readLength > 0) {
152           final int limit;
153           if (readLength > chunkSize) {
154             // LUCENE-1566 - work around JVM Bug by breaking
155             // very large reads into chunks
156             limit = readOffset + chunkSize;
157           } else {
158             limit = readOffset + readLength;
159           }
160           bb.limit(limit);
161           int i = channel.read(bb, pos);
162           if (i == -1) {
163             throw new IOException("read past EOF");
164           }
165           pos += i;
166           readOffset += i;
167           readLength -= i;
168         }
169       } catch (OutOfMemoryError e) {
170         // propagate OOM up and add a hint for 32bit VM Users hitting the bug
171         // with a large chunk size in the fast path.
172         final OutOfMemoryError outOfMemoryError = new OutOfMemoryError(
173               "OutOfMemoryError likely caused by the Sun VM Bug described in "
174               + "https://issues.apache.org/jira/browse/LUCENE-1566; try calling FSDirectory.setReadChunkSize "
175               + "with a a value smaller than the current chunk size (" + chunkSize + ")");
176         outOfMemoryError.initCause(e);
177         throw outOfMemoryError;
178       }
179     }
180   }
181 }
