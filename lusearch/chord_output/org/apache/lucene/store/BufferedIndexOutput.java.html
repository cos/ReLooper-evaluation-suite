20  import java.io.IOException;
21  
22  /** Base implementation class for buffered {@link IndexOutput}. */
23  public abstract class BufferedIndexOutput extends IndexOutput {
24    static final int BUFFER_SIZE = 16384;
25  
26    private final byte[] buffer = new byte[BUFFER_SIZE];
27    private long bufferStart = 0;           // position in file of buffer
28    private int bufferPosition = 0;         // position in buffer
29  
30    /** Writes a single byte.
31     * @see IndexInput#readByte()
32     */
33    public void writeByte(byte b) throws IOException {
34      if (bufferPosition >= BUFFER_SIZE)
35        flush();
36      buffer[bufferPosition++] = b;
37    }
38  
39    /** Writes an array of bytes.
40     * @param b the bytes to write
41     * @param length the number of bytes to write
42     * @see IndexInput#readBytes(byte[],int,int)
43     */
44    public void writeBytes(byte[] b, int offset, int length) throws IOException {
45      int bytesLeft = BUFFER_SIZE - bufferPosition;
46      // is there enough space in the buffer?
47      if (bytesLeft >= length) {
48        // we add the data to the end of the buffer
49        System.arraycopy(b, offset, buffer, bufferPosition, length);
50        bufferPosition += length;
51        // if the buffer is full, flush it
52        if (BUFFER_SIZE - bufferPosition == 0)
53          flush();
54      } else {
55        // is data larger then buffer?
56        if (length > BUFFER_SIZE) {
57          // we flush the buffer
58          if (bufferPosition > 0)
59            flush();
60          // and write data at once
61          flushBuffer(b, offset, length);
62          bufferStart += length;
63        } else {
64          // we fill/flush the buffer (until the input is written)
65          int pos = 0; // position in the input data
66          int pieceLength;
67          while (pos < length) {
68            pieceLength = (length - pos < bytesLeft) ? length - pos : bytesLeft;
69            System.arraycopy(b, pos + offset, buffer, bufferPosition, pieceLength);
70            pos += pieceLength;
71            bufferPosition += pieceLength;
72            // if the buffer is full, flush it
73            bytesLeft = BUFFER_SIZE - bufferPosition;
74            if (bytesLeft == 0) {
75              flush();
76              bytesLeft = BUFFER_SIZE;
77            }
78          }
79        }
80      }
81    }
82  
83    /** Forces any buffered output to be written. */
84    public void flush() throws IOException {
85      flushBuffer(buffer, bufferPosition);
86      bufferStart += bufferPosition;
87      bufferPosition = 0;
88    }
89  
90    /** Expert: implements buffer write.  Writes bytes at the current position in
91     * the output.
92     * @param b the bytes to write
93     * @param len the number of bytes to write
94     */
95    private void flushBuffer(byte[] b, int len) throws IOException {
96      flushBuffer(b, 0, len);
97    }
98  
99    /** Expert: implements buffer write.  Writes bytes at the current position in
100    * the output.
101    * @param b the bytes to write
102    * @param offset the offset in the byte array
103    * @param len the number of bytes to write
104    */
105   protected abstract void flushBuffer(byte[] b, int offset, int len) throws IOException;
106   
107   /** Closes this stream to further operations. */
108   public void close() throws IOException {
109     flush();
110   }
111 
112   /** Returns the current position in this file, where the next write will
113    * occur.
114    * @see #seek(long)
115    */
116   public long getFilePointer() {
117     return bufferStart + bufferPosition;
118   }
119 
120   /** Sets current position in this file, where the next write will occur.
121    * @see #getFilePointer()
122    */
123   public void seek(long pos) throws IOException {
124     flush();
125     bufferStart = pos;
126   }
127 
128   /** The number of bytes in the file. */
129   public abstract long length() throws IOException;
130 
131 
132 }
