20  import java.io.IOException;
21  import java.util.Map;
22  import java.util.HashMap;
23  
24  /** Abstract base class for input from a file in a {@link Directory}.  A
25   * random-access input stream.  Used for all Lucene index input operations.
26   * @see Directory
27   */
28  public abstract class IndexInput implements Cloneable {
29    private byte[] bytes;                           // used by readString()
30    private char[] chars;                           // used by readModifiedUTF8String()
31    private boolean preUTF8Strings;                 // true if we are reading old (modified UTF8) string format
32  
33    /** Reads and returns a single byte.
34     * @see IndexOutput#writeByte(byte)
35     */
36    public abstract byte readByte() throws IOException;
37  
38    /** Reads a specified number of bytes into an array at the specified offset.
39     * @param b the array to read bytes into
40     * @param offset the offset in the array to start storing bytes
41     * @param len the number of bytes to read
42     * @see IndexOutput#writeBytes(byte[],int)
43     */
44    public abstract void readBytes(byte[] b, int offset, int len)
45      throws IOException;
46  
47    /** Reads a specified number of bytes into an array at the
48     * specified offset with control over whether the read
49     * should be buffered (callers who have their own buffer
50     * should pass in "false" for useBuffer).  Currently only
51     * {@link BufferedIndexInput} respects this parameter.
52     * @param b the array to read bytes into
53     * @param offset the offset in the array to start storing bytes
54     * @param len the number of bytes to read
55     * @param useBuffer set to false if the caller will handle
56     * buffering.
57     * @see IndexOutput#writeBytes(byte[],int)
58     */
59    public void readBytes(byte[] b, int offset, int len, boolean useBuffer)
60      throws IOException
61    {
62      // Default to ignoring useBuffer entirely
63      readBytes(b, offset, len);
64    }
65  
66    /** Reads four bytes and returns an int.
67     * @see IndexOutput#writeInt(int)
68     */
69    public int readInt() throws IOException {
70      return ((readByte() & 0xFF) << 24) | ((readByte() & 0xFF) << 16)
71           | ((readByte() & 0xFF) <<  8) |  (readByte() & 0xFF);
72    }
73  
74    /** Reads an int stored in variable-length format.  Reads between one and
75     * five bytes.  Smaller values take fewer bytes.  Negative numbers are not
76     * supported.
77     * @see IndexOutput#writeVInt(int)
78     */
79    public int readVInt() throws IOException {
80      byte b = readByte();
81      int i = b & 0x7F;
82      for (int shift = 7; (b & 0x80) != 0; shift += 7) {
83        b = readByte();
84        i |= (b & 0x7F) << shift;
85      }
86      return i;
87    }
88  
89    /** Reads eight bytes and returns a long.
90     * @see IndexOutput#writeLong(long)
91     */
92    public long readLong() throws IOException {
93      return (((long)readInt()) << 32) | (readInt() & 0xFFFFFFFFL);
94    }
95  
96    /** Reads a long stored in variable-length format.  Reads between one and
97     * nine bytes.  Smaller values take fewer bytes.  Negative numbers are not
98     * supported. */
99    public long readVLong() throws IOException {
100     byte b = readByte();
101     long i = b & 0x7F;
102     for (int shift = 7; (b & 0x80) != 0; shift += 7) {
103       b = readByte();
104       i |= (b & 0x7FL) << shift;
105     }
106     return i;
107   }
108 
109   /** Call this if readString should read characters stored
110    *  in the old modified UTF8 format (length in java chars
111    *  and java's modified UTF8 encoding).  This is used for
112    *  indices written pre-2.4 See LUCENE-510 for details. */
113   public void setModifiedUTF8StringsMode() {
114     preUTF8Strings = true;
115   }
116 
117   /** Reads a string.
118    * @see IndexOutput#writeString(String)
119    */
120   public String readString() throws IOException {
121     if (preUTF8Strings)
122       return readModifiedUTF8String();
123     int length = readVInt();
124     if (bytes == null || length > bytes.length)
125       bytes = new byte[(int) (length*1.25)];
126     readBytes(bytes, 0, length);
127     return new String(bytes, 0, length, "UTF-8");
128   }
129 
130   private String readModifiedUTF8String() throws IOException {
131     int length = readVInt();
132     if (chars == null || length > chars.length)
133       chars = new char[length];
134     readChars(chars, 0, length);
135     return new String(chars, 0, length);
136   }
137 
138   /** Reads Lucene's old "modified UTF-8" encoded
139    *  characters into an array.
140    * @param buffer the array to read characters into
141    * @param start the offset in the array to start storing characters
142    * @param length the number of characters to read
143    * @see IndexOutput#writeChars(String,int,int)
144    * @deprecated -- please use readString or readBytes
145    *                instead, and construct the string
146    *                from those utf8 bytes
147    */
148   public void readChars(char[] buffer, int start, int length)
149        throws IOException {
150     final int end = start + length;
151     for (int i = start; i < end; i++) {
152       byte b = readByte();
153       if ((b & 0x80) == 0)
154     buffer[i] = (char)(b & 0x7F);
155       else if ((b & 0xE0) != 0xE0) {
156     buffer[i] = (char)(((b & 0x1F) << 6)
157          | (readByte() & 0x3F));
158       } else
159     buffer[i] = (char)(((b & 0x0F) << 12)
160         | ((readByte() & 0x3F) << 6)
161             |  (readByte() & 0x3F));
162     }
163   }
164 
165   /**
166    * Expert
167    * 
168    * Similar to {@link #readChars(char[], int, int)} but does not do any conversion operations on the bytes it is reading in.  It still
169    * has to invoke {@link #readByte()} just as {@link #readChars(char[], int, int)} does, but it does not need a buffer to store anything
170    * and it does not have to do any of the bitwise operations, since we don't actually care what is in the byte except to determine
171    * how many more bytes to read
172    * @param length The number of chars to read
173    * @deprecated this method operates on old "modified utf8" encoded
174    *             strings
175    */
176   public void skipChars(int length) throws IOException{
177     for (int i = 0; i < length; i++) {
178       byte b = readByte();
179       if ((b & 0x80) == 0){
180         //do nothing, we only need one byte
181       }
182       else if ((b & 0xE0) != 0xE0) {
183         readByte();//read an additional byte
184       } else{      
185         //read two additional bytes.
186         readByte();
187         readByte();
188       }
189     }
190   }

193   /** Closes the stream to further operations. */
194   public abstract void close() throws IOException;
195 
196   /** Returns the current position in this file, where the next read will
197    * occur.
198    * @see #seek(long)
199    */
200   public abstract long getFilePointer();
201 
202   /** Sets current position in this file, where the next read will occur.
203    * @see #getFilePointer()
204    */
205   public abstract void seek(long pos) throws IOException;
206 
207   /** The number of bytes in the file. */
208   public abstract long length();
209 
210   /** Returns a clone of this stream.
211    *
212    * <p>Clones of a stream access the same data, and are positioned at the same
213    * point as the stream they were cloned from.
214    *
215    * <p>Expert: Subclasses must ensure that clones may be positioned at
216    * different points in the input from each other and from the stream they
217    * were cloned from.
218    */
219   public Object clone() {
220     IndexInput clone = null;
221     try {
222       clone = (IndexInput)super.clone();
223     } catch (CloneNotSupportedException e) {}
224 
225     clone.bytes = null;
226     clone.chars = null;
227 
228     return clone;
229   }
230 
231   // returns Map<String, String>
232   public Map readStringStringMap() throws IOException {
233     final Map map = new HashMap();
234     final int count = readInt();
235     for(int i=0;i<count;i++) {
236       final String key = readString();
237       final String val = readString();
238       map.put(key, val);
239     }
240 
241     return map;
242   }
