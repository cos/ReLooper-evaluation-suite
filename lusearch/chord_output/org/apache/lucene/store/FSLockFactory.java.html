20  import java.io.File;
21  
22  /**
23   * Base class for file system based locking implementation.
24   */
25  
26  public abstract class FSLockFactory extends LockFactory {
27  
28    /**
29     * Directory for the lock files.
30     */
31    protected File lockDir = null;
32  
33    /**
34     * Set the lock directory. This method can be only called
35     * once to initialize the lock directory. It is used by {@link FSDirectory}
36     * to set the lock directory to itsself.
37     * Subclasses can also use this method to set the directory
38     * in the constructor.
39     */
40    protected final void setLockDir(File lockDir) {
41      if (this.lockDir != null)
42        throw new IllegalStateException("You can set the lock directory for this factory only once.");
43      this.lockDir = lockDir;
44    }
45    
46    /**
47     * Retrieve the lock directory.
48     */
49    public File getLockDir() {
50      return lockDir;
51    }
52  
53  }
