20  import java.io.IOException;
21  
22  /**
23   * A memory-resident {@link IndexOutput} implementation.
24   * 
25   * @version $Id: RAMOutputStream.java 691694 2008-09-03 17:34:29Z mikemccand $
26   */
27  
28  public class RAMOutputStream extends IndexOutput {
29    static final int BUFFER_SIZE = 1024;
30  
31    private RAMFile file;
32  
33    private byte[] currentBuffer;
34    private int currentBufferIndex;
35    
36    private int bufferPosition;
37    private long bufferStart;
38    private int bufferLength;
39  
40    /** Construct an empty output buffer. */
41    public RAMOutputStream() {
42      this(new RAMFile());
43    }
44  
45    RAMOutputStream(RAMFile f) {
46      file = f;
47  
48      // make sure that we switch to the
49      // first needed buffer lazily
50      currentBufferIndex = -1;
51      currentBuffer = null;
52    }
53  
54    /** Copy the current contents of this buffer to the named output. */
55    public void writeTo(IndexOutput out) throws IOException {
56      flush();
57      final long end = file.length;
58      long pos = 0;
59      int buffer = 0;
60      while (pos < end) {
61        int length = BUFFER_SIZE;
62        long nextPos = pos + length;
63        if (nextPos > end) {                        // at the last buffer
64          length = (int)(end - pos);
65        }
66        out.writeBytes((byte[])file.getBuffer(buffer++), length);
67        pos = nextPos;
68      }
69    }
70  
71    /** Resets this to an empty buffer. */
72    public void reset() {
73      try {
74        seek(0);
75      } catch (IOException e) {                     // should never happen
76        throw new RuntimeException(e.toString());
77      }
78  
79      file.setLength(0);
80    }
81  
82    public void close() throws IOException {
83      flush();
84    }
85  
86    public void seek(long pos) throws IOException {
87      // set the file length in case we seek back
88      // and flush() has not been called yet
89      setFileLength();
90      if (pos < bufferStart || pos >= bufferStart + bufferLength) {
91        currentBufferIndex = (int) (pos / BUFFER_SIZE);
92        switchCurrentBuffer();
93      }
94  
95      bufferPosition = (int) (pos % BUFFER_SIZE);
96    }
97  
98    public long length() {
99      return file.length;
100   }
101 
102   public void writeByte(byte b) throws IOException {
103     if (bufferPosition == bufferLength) {
104       currentBufferIndex++;
105       switchCurrentBuffer();
106     }
107     currentBuffer[bufferPosition++] = b;
108   }
109 
110   public void writeBytes(byte[] b, int offset, int len) throws IOException {
111     assert b != null;
112     while (len > 0) {
113       if (bufferPosition ==  bufferLength) {
114         currentBufferIndex++;
115         switchCurrentBuffer();
116       }
117 
118       int remainInBuffer = currentBuffer.length - bufferPosition;
119       int bytesToCopy = len < remainInBuffer ? len : remainInBuffer;
120       System.arraycopy(b, offset, currentBuffer, bufferPosition, bytesToCopy);
121       offset += bytesToCopy;
122       len -= bytesToCopy;
123       bufferPosition += bytesToCopy;
124     }
125   }
126 
127   private final void switchCurrentBuffer() throws IOException {
128     if (currentBufferIndex == file.numBuffers()) {
129       currentBuffer = file.addBuffer(BUFFER_SIZE);
130     } else {
131       currentBuffer = (byte[]) file.getBuffer(currentBufferIndex);
132     }
133     bufferPosition = 0;
134     bufferStart = (long) BUFFER_SIZE * (long) currentBufferIndex;
135     bufferLength = currentBuffer.length;
136   }
137 
138   private void setFileLength() {
139     long pointer = bufferStart + bufferPosition;
140     if (pointer > file.length) {
141       file.setLength(pointer);
142     }
143   }
144 
145   public void flush() throws IOException {
146     file.setLastModified(System.currentTimeMillis());
147     setFileLength();
148   }
149 
150   public long getFilePointer() {
151     return currentBufferIndex < 0 ? 0 : bufferStart + bufferPosition;
152   }
153 
154   /** Returns byte usage of all buffers. */
155   public long sizeInBytes() {
156     return file.numBuffers() * BUFFER_SIZE;
157   }
158 }
