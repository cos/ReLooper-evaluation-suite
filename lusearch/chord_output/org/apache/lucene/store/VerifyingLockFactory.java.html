20  import java.net.Socket;
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.OutputStream;
24  
25  /**
26   * A {@link LockFactory} that wraps another {@link
27   * LockFactory} and verifies that each lock obtain/release
28   * is "correct" (never results in two processes holding the
29   * lock at the same time).  It does this by contacting an
30   * external server ({@link LockVerifyServer}) to assert that
31   * at most one process holds the lock at a time.  To use
32   * this, you should also run {@link LockVerifyServer} on the
33   * host & port matching what you pass to the constructor.
34   *
35   * @see LockVerifyServer
36   * @see LockStressTest
37   */
38  
39  public class VerifyingLockFactory extends LockFactory {
40  
41    LockFactory lf;
42    byte id;
43    String host;
44    int port;
45  
46    private class CheckedLock extends Lock {
47      private Lock lock;
48  
49      public CheckedLock(Lock lock) {
50        this.lock = lock;
51      }
52  
53      private void verify(byte message) {
54        try {
55          Socket s = new Socket(host, port);
56          OutputStream out = s.getOutputStream();
57          out.write(id);
58          out.write(message);
59          InputStream in = s.getInputStream();
60          int result = in.read();
61          in.close();
62          out.close();
63          s.close();
64          if (result != 0)
65            throw new RuntimeException("lock was double acquired");
66        } catch (Exception e) {
67          throw new RuntimeException(e);
68        }
69      }
70  
71      public synchronized boolean obtain(long lockWaitTimeout)
72        throws LockObtainFailedException, IOException {
73        boolean obtained = lock.obtain(lockWaitTimeout);
74        if (obtained)
75          verify((byte) 1);
76        return obtained;
77      }
78  
79      public synchronized boolean obtain()
80        throws LockObtainFailedException, IOException {
81        return lock.obtain();
82      }
83  
84      public synchronized boolean isLocked() {
85        return lock.isLocked();
86      }
87  
88      public synchronized void release() throws IOException {
89        if (isLocked()) {
90          verify((byte) 0);
91          lock.release();
92        }
93      }
94    }
95  
96    /**
97     * @param id should be a unique id across all clients
98     * @param lf the LockFactory that we are testing
99     * @param host host or IP where {@link LockVerifyServer}
100             is running
101    * @param port the port {@link LockVerifyServer} is
102             listening on
103   */
104   public VerifyingLockFactory(byte id, LockFactory lf, String host, int port) throws IOException {
105     this.id = id;
106     this.lf = lf;
107     this.host = host;
108     this.port = port;
109   }
110 
111   public synchronized Lock makeLock(String lockName) {
112     return new CheckedLock(lf.makeLock(lockName));
113   }
114 
115   public synchronized void clearLock(String lockName)
116     throws IOException {
117     lf.clearLock(lockName);
118   }
119 }
