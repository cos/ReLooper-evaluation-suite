20  import java.io.File;
21  import java.io.IOException;
22  import java.io.RandomAccessFile;
23  
24  /** A straightforward implementation of {@link FSDirectory}
25   *  using java.io.RandomAccessFile.  However, this class has
26   *  poor concurrent performance (multiple threads will
27   *  bottleneck) as it synchronizes when multiple threads
28   *  read from the same file.  It's usually better to use
29   *  {@link NIOFSDirectory} or {@link MMapDirectory} instead. */
30  public class SimpleFSDirectory extends FSDirectory {
31      
32    /** Create a new SimpleFSDirectory for the named location.
33     *
34     * @param path the path of the directory
35     * @param lockFactory the lock factory to use, or null for the default
36     * ({@link NativeFSLockFactory});
37     * @throws IOException
38     */
39    public SimpleFSDirectory(File path, LockFactory lockFactory) throws IOException {
40      super(path, lockFactory);
41    }
42    
43    /** Create a new SimpleFSDirectory for the named location and {@link NativeFSLockFactory}.
44     *
45     * @param path the path of the directory
46     * @throws IOException
47     */
48    public SimpleFSDirectory(File path) throws IOException {
49      super(path, null);
50    }
51  
52    // back compatibility so FSDirectory can instantiate via reflection
53    /** @deprecated */
54    SimpleFSDirectory() {}
55  
56    /** Creates an IndexOutput for the file with the given name. */
57    public IndexOutput createOutput(String name) throws IOException {
58      initOutput(name);
59      return new SimpleFSIndexOutput(new File(directory, name));
60    }
61  
62    /** Creates an IndexInput for the file with the given name. */
63    public IndexInput openInput(String name, int bufferSize) throws IOException {
64      ensureOpen();
65      return new SimpleFSIndexInput(new File(directory, name), bufferSize, getReadChunkSize());
66    }
67  
68    protected static class SimpleFSIndexInput extends BufferedIndexInput {
69    
70      protected static class Descriptor extends RandomAccessFile {
71        // remember if the file is open, so that we don't try to close it
72        // more than once
73        protected volatile boolean isOpen;
74        long position;
75        final long length;
76        
77        public Descriptor(File file, String mode) throws IOException {
78          super(file, mode);
79          isOpen=true;
80          length=length();
81        }
82    
83        public void close() throws IOException {
84          if (isOpen) {
85            isOpen=false;
86            super.close();
87          }
88        }
89      }
90    
91      protected final Descriptor file;
92      boolean isClone;
93      //  LUCENE-1566 - maximum read length on a 32bit JVM to prevent incorrect OOM 
94      protected final int chunkSize;
95  
96      /** @deprecated Please use ctor taking chunkSize */
97      public SimpleFSIndexInput(File path) throws IOException {
98        this(path, BufferedIndexInput.BUFFER_SIZE, SimpleFSDirectory.DEFAULT_READ_CHUNK_SIZE);
99      }
100   
101     /** @deprecated Please use ctor taking chunkSize */
102     public SimpleFSIndexInput(File path, int bufferSize) throws IOException {
103       this(path, bufferSize, SimpleFSDirectory.DEFAULT_READ_CHUNK_SIZE);
104     }
105     
106     public SimpleFSIndexInput(File path, int bufferSize, int chunkSize) throws IOException {
107       super(bufferSize);
108       file = new Descriptor(path, "r");
109       this.chunkSize = chunkSize;
110     }
111   
112     /** IndexInput methods */
113     protected void readInternal(byte[] b, int offset, int len)
114          throws IOException {
115       synchronized (file) {
116         long position = getFilePointer();
117         if (position != file.position) {
118           file.seek(position);
119           file.position = position;
120         }
121         int total = 0;
122 
123         try {
124           do {
125             final int readLength;
126             if (total + chunkSize > len) {
127               readLength = len - total;
128             } else {
129               // LUCENE-1566 - work around JVM Bug by breaking very large reads into chunks
130               readLength = chunkSize;
131             }
132             final int i = file.read(b, offset + total, readLength);
133             if (i == -1) {
134               throw new IOException("read past EOF");
135             }
136             file.position += i;
137             total += i;
138           } while (total < len);
139         } catch (OutOfMemoryError e) {
140           // propagate OOM up and add a hint for 32bit VM Users hitting the bug
141           // with a large chunk size in the fast path.
142           final OutOfMemoryError outOfMemoryError = new OutOfMemoryError(
143               "OutOfMemoryError likely caused by the Sun VM Bug described in "
144               + "https://issues.apache.org/jira/browse/LUCENE-1566; try calling FSDirectory.setReadChunkSize "
145               + "with a a value smaller than the current chunks size (" + chunkSize + ")");
146           outOfMemoryError.initCause(e);
147           throw outOfMemoryError;
148         }
149       }
150     }
151   
152     public void close() throws IOException {
153       // only close the file if this is not a clone
154       if (!isClone) file.close();
155     }
156   
157     protected void seekInternal(long position) {
158     }
159   
160     public long length() {
161       return file.length;
162     }
163   
164     public Object clone() {
165       SimpleFSIndexInput clone = (SimpleFSIndexInput)super.clone();
166       clone.isClone = true;
167       return clone;
168     }
169   
170     /** Method used for testing. Returns true if the underlying
171      *  file descriptor is valid.
172      */
173     boolean isFDValid() throws IOException {
174       return file.getFD().valid();
175     }
176   }
177 
178   protected static class SimpleFSIndexOutput extends BufferedIndexOutput {
179     RandomAccessFile file = null;
180   
181     // remember if the file is open, so that we don't try to close it
182     // more than once
183     private volatile boolean isOpen;
184 
185     public SimpleFSIndexOutput(File path) throws IOException {
186       file = new RandomAccessFile(path, "rw");
187       isOpen = true;
188     }
189   
190     /** output methods: */
191     public void flushBuffer(byte[] b, int offset, int size) throws IOException {
192       file.write(b, offset, size);
193     }
194     public void close() throws IOException {
195       // only close the file if it has not been closed yet
196       if (isOpen) {
197         boolean success = false;
198         try {
199           super.close();
200           success = true;
201         } finally {
202           isOpen = false;
203           if (!success) {
204             try {
205               file.close();
206             } catch (Throwable t) {
207               // Suppress so we don't mask original exception
208             }
209           } else
210             file.close();
211         }
212       }
213     }
214   
215     /** Random-access methods */
216     public void seek(long pos) throws IOException {
217       super.seek(pos);
218       file.seek(pos);
219     }
220     public long length() throws IOException {
221       return file.length();
222     }
223     public void setLength(long length) throws IOException {
224       file.setLength(length);
225     }
226   }
227 }
