96    public OpenBitSetIterator(long[] bits, int numWords) {
97      arr = bits;
98      words = numWords;

101   // 64 bit shifts
102   private void shift() {
103     if ((int)word ==0) {wordShift +=32; word = word >>>32; }
104     if ((word & 0x0000FFFF) == 0) { wordShift +=16; word >>>=16; }
105     if ((word & 0x000000FF) == 0) { wordShift +=8; word >>>=8; }
106     indexArray = bitlist[(int)word & 0xff];
107   }
108 
109   /***** alternate shift implementations
110   // 32 bit shifts, but a long shift needed at the end
111   private void shift2() {
112     int y = (int)word;
113     if (y==0) {wordShift +=32; y = (int)(word >>>32); }
114     if ((y & 0x0000FFFF) == 0) { wordShift +=16; y>>>=16; }
115     if ((y & 0x000000FF) == 0) { wordShift +=8; y>>>=8; }
116     indexArray = bitlist[y & 0xff];
117     word >>>= (wordShift +1);
118   }
119 
120   private void shift3() {
121     int lower = (int)word;
122     int lowByte = lower & 0xff;
123     if (lowByte != 0) {
124       indexArray=bitlist[lowByte];
125       return;
126     }
127     shift();
128   }

131   /** @deprecated use {@link #nextDoc()} instead. */
132   public boolean next() {
133     return nextDoc() != NO_MORE_DOCS;
134   }
135 
136   public int nextDoc() {
137     if (indexArray == 0) {
138       if (word != 0) {
139         word >>>= 8;
140         wordShift += 8;
141       }
142 
143       while (word == 0) {
144         if (++i >= words) {
145           return curDocId = NO_MORE_DOCS;
146         }
147         word = arr[i];
148         wordShift = -1; // loop invariant code motion should move this
149       }
150 
151       // after the first time, should I go with a linear search, or
152       // stick with the binary search in shift?
153       shift();
154     }
155 
156     int bitIndex = (indexArray & 0x0f) + wordShift;
157     indexArray >>>= 4;
158     // should i<<6 be cached as a separate variable?
159     // it would only save one cycle in the best circumstances.
160     return curDocId = (i<<6) + bitIndex;
161   }
162   
163   /** @deprecated use {@link #advance(int)} instead. */
164   public boolean skipTo(int target) {
165     return advance(target) != NO_MORE_DOCS;
166   }
167 
168   public int advance(int target) {
169     indexArray = 0;
170     i = target >> 6;
171     if (i >= words) {
172       word = 0; // setup so next() will also return -1
173       return curDocId = NO_MORE_DOCS;
174     }
175     wordShift = target & 0x3f;
176     word = arr[i] >>> wordShift;
177     if (word != 0) {
178       wordShift--; // compensate for 1 based arrIndex
179     } else {
180       while (word == 0) {
181         if (++i >= words) {
182           return curDocId = NO_MORE_DOCS;
183         }
184         word = arr[i];
185       }
186       wordShift = -1;
187     }
188 
189     shift();
190 
191     int bitIndex = (indexArray & 0x0f) + wordShift;
192     indexArray >>>= 4;
193     // should i<<6 be cached as a separate variable?
194     // it would only save one cycle in the best circumstances.
195     return curDocId = (i<<6) + bitIndex;
196   }
