20  import java.util.BitSet;
21  
22  import org.apache.lucene.search.DocIdSet;
23  import org.apache.lucene.search.DocIdSetIterator;
24  
25  
26  /** Simple DocIdSet and DocIdSetIterator backed by a BitSet */
27  public class DocIdBitSet extends DocIdSet {
28    private BitSet bitSet;
29      
30    public DocIdBitSet(BitSet bitSet) {
31      this.bitSet = bitSet;
32    }
33  
34    public DocIdSetIterator iterator() {
35      return new DocIdBitSetIterator(bitSet);
36    }
37  
38    /** This DocIdSet implementation is cacheable. */
39    public boolean isCacheable() {
40      return true;
41    }
42    
43    /**
44     * Returns the underlying BitSet. 
45     */
46    public BitSet getBitSet() {
47      return this.bitSet;
48    }
49    
50    private static class DocIdBitSetIterator extends DocIdSetIterator {
51      private int docId;
52      private BitSet bitSet;
53      
54      DocIdBitSetIterator(BitSet bitSet) {
55        this.bitSet = bitSet;
56        this.docId = -1;
57      }
58      
59      /** @deprecated use {@link #docID()} instead. */
60      public int doc() {
61        assert docId != -1;
62        return docId;
63      }
64      
65      public int docID() {
66        return docId;
67      }
68      
69      /** @deprecated use {@link #nextDoc()} instead. */
70      public boolean next() {
71        // (docId + 1) on next line requires -1 initial value for docNr:
72        return nextDoc() != NO_MORE_DOCS;
73      }
74      
75      public int nextDoc() {
76        // (docId + 1) on next line requires -1 initial value for docNr:
77        int d = bitSet.nextSetBit(docId + 1);
78        // -1 returned by BitSet.nextSetBit() when exhausted
79        docId = d == -1 ? NO_MORE_DOCS : d;
80        return docId;
81      }
82    
83      /** @deprecated use {@link #advance(int)} instead. */
84      public boolean skipTo(int skipDocNr) {
85        return advance(skipDocNr) != NO_MORE_DOCS;
86      }
87    
88      public int advance(int target) {
89        int d = bitSet.nextSetBit(target);
90        // -1 returned by BitSet.nextSetBit() when exhausted
91        docId = d == -1 ? NO_MORE_DOCS : d;
92        return docId;
93      }
94    }
95  }
