21  import java.io.Serializable;
22  
23  /**
24   * Use by certain classes to match version compatibility
25   * across releases of Lucene.
26   * 
27   * <p><b>WARNING</b>: When changing the version parameter
28   * that you supply to components in Lucene, do not simply
29   * change the version at search-time, but instead also adjust
30   * your indexing code to match, and re-index.
31   */
32  public final class Version extends Parameter implements Serializable {
33  
34    /**
35     * <p><b>WARNING</b>: if you use this setting, and then
36     * upgrade to a newer release of Lucene, sizable changes
37     * may happen.  If backwards compatibility is important
38     * then you should instead explicitly specify an actual
39     * version.
40     * <p>
41     * If you use this constant then you may need to
42     * <b>re-index all of your documents</b> when upgrading
43     * Lucene, as the way text is indexed may have changed.
44     * Additionally, you may need to <b>re-test your entire
45     * application</b> to ensure it behaves as expected, as
46     * some defaults may have changed and may break functionality
47     * in your application.
48     * @deprecated Use an actual version instead.
49     */
50    public static final Version LUCENE_CURRENT = new Version("LUCENE_CURRENT", 0);
51    
52    /** Match settings and bugs in Lucene's 2.0 release. */
53    public static final Version LUCENE_20 = new Version("LUCENE_20", 2000);
54  
55    /** Match settings and bugs in Lucene's 2.1 release. */
56    public static final Version LUCENE_21 = new Version("LUCENE_21", 2100);
57  
58    /** Match settings and bugs in Lucene's 2.2 release. */
59    public static final Version LUCENE_22 = new Version("LUCENE_22", 2200);
60  
61    /** Match settings and bugs in Lucene's 2.3 release. */
62    public static final Version LUCENE_23 = new Version("LUCENE_23", 2300);
63  
64    /** Match settings and bugs in Lucene's 2.4 release. */
65    public static final Version LUCENE_24 = new Version("LUCENE_24", 2400);
66  
67    /** Match settings and bugs in Lucene's 2.9 release. 
68     *  <p>
69     *  Use this to get the latest &amp; greatest settings, bug
70     *  fixes, etc, for Lucene.
71     */
72    public static final Version LUCENE_29 = new Version("LUCENE_29", 2900);
73  
74    private final int v;
75  
76    public Version(String name, int v) {
77      super(name);
78      this.v = v;
79    }
80  
81    public boolean onOrAfter(Version other) {
82      return v == 0 || v >= other.v;
83    }
84  }
