26  /** An "open" BitSet implementation that allows direct access to the array of words
27   * storing the bits.
28   * <p/>
29   * Unlike java.util.bitset, the fact that bits are packed into an array of longs
30   * is part of the interface.  This allows efficient implementation of other algorithms
31   * by someone other than the author.  It also allows one to efficiently implement
32   * alternate serialization or interchange formats.
33   * <p/>
34   * <code>OpenBitSet</code> is faster than <code>java.util.BitSet</code> in most operations
35   * and *much* faster at calculating cardinality of sets and results of set operations.
36   * It can also handle sets of larger cardinality (up to 64 * 2**32-1)
37   * <p/>
38   * The goals of <code>OpenBitSet</code> are the fastest implementation possible, and
39   * maximum code reuse.  Extra safety and encapsulation
40   * may always be built on top, but if that's built in, the cost can never be removed (and
41   * hence people re-implement their own version in order to get better performance).
42   * If you want a "safe", totally encapsulated (and slower and limited) BitSet
43   * class, use <code>java.util.BitSet</code>.
44   * <p/>

97    /** Constructs an OpenBitSet from an existing long[].
98     * <br/>
99     * The first 64 bits are in long[0],
100    * with bit index 0 at the least significant bit, and bit index 63 at the most significant.
101    * Given a bit index,
102    * the word containing it is long[index/64], and it is at bit number index%64 within that word.
103    * <p>
104    * numWords are the number of elements in the array that contain
105    * set bits (non-zero longs).
106    * numWords should be &lt= bits.length, and
107    * any existing words in the array at position &gt= numWords should be zero.
108    *
109    */
110   public OpenBitSet(long[] bits, int numWords) {
111     this.bits = bits;
112     this.wlen = numWords;
113   }

152   /** Returns true or false for the specified bit index. */
153   public boolean get(int index) {
154     int i = index >> 6;               // div 64
155     // signed shift will keep a negative index and force an
156     // array-index-out-of-bounds-exception, removing the need for an explicit check.
157     if (i>=bits.length) return false;
158 
159     int bit = index & 0x3f;           // mod 64
160     long bitmask = 1L << bit;
161     return (bits[i] & bitmask) != 0;
162   }

260   /** Sets a range of bits, expanding the set size if necessary
261    *
262    * @param startIndex lower index
263    * @param endIndex one-past the last bit to set
264    */
265   public void set(long startIndex, long endIndex) {
266     if (endIndex <= startIndex) return;
267 
268     int startWord = (int)(startIndex>>6);
269 
270     // since endIndex is one past the end, this is index of the last
271     // word to be changed.
272     int endWord   = expandingWordNum(endIndex-1);
273 
274     long startmask = -1L << startIndex;
275     long endmask = -1L >>> -endIndex;  // 64-(endIndex&0x3f) is the same as -endIndex due to wrap
276 
277     if (startWord == endWord) {
278       bits[startWord] |= (startmask & endmask);
279       return;
280     }
281 
282     bits[startWord] |= startmask;
283     Arrays.fill(bits, startWord+1, endWord, -1L);
284     bits[endWord] |= endmask;
285   }

289   protected int expandingWordNum(long index) {
290     int wordNum = (int)(index >> 6);
291     if (wordNum>=wlen) {
292       ensureCapacity(index+1);
293       wlen = wordNum+1;
294     }
295     return wordNum;
296   }

335   /** Clears a range of bits.  Clearing past the end does not change the size of the set.
336    *
337    * @param startIndex lower index
338    * @param endIndex one-past the last bit to clear
339    */
340   public void clear(int startIndex, int endIndex) {
341     if (endIndex <= startIndex) return;
342 
343     int startWord = (startIndex>>6);
344     if (startWord >= wlen) return;
345 
346     // since endIndex is one past the end, this is index of the last
347     // word to be changed.
348     int endWord   = ((endIndex-1)>>6);
349 
350     long startmask = -1L << startIndex;
351     long endmask = -1L >>> -endIndex;  // 64-(endIndex&0x3f) is the same as -endIndex due to wrap
352 
353     // invert masks since we are clearing
354     startmask = ~startmask;
355     endmask = ~endmask;
356 
357     if (startWord == endWord) {
358       bits[startWord] &= (startmask | endmask);
359       return;
360     }
361 
362     bits[startWord] &= startmask;
363 
364     int middle = Math.min(wlen, endWord);
365     Arrays.fill(bits, startWord+1, middle, 0L);
366     if (endWord < wlen) {
367       bits[endWord] &= endmask;
368     }
369   }

484   /** Flips a range of bits, expanding the set size if necessary
485    *
486    * @param startIndex lower index
487    * @param endIndex one-past the last bit to flip
488    */
489   public void flip(long startIndex, long endIndex) {
490     if (endIndex <= startIndex) return;
491     int startWord = (int)(startIndex>>6);
492 
493     // since endIndex is one past the end, this is index of the last
494     // word to be changed.
495     int endWord   = expandingWordNum(endIndex-1);
496 
497     /*** Grrr, java shifting wraps around so -1L>>>64 == -1
498      * for that reason, make sure not to use endmask if the bits to flip will
499      * be zero in the last word (redefine endWord to be the last changed...)
500     long startmask = -1L << (startIndex & 0x3f);     // example: 11111...111000
501     long endmask = -1L >>> (64-(endIndex & 0x3f));   // example: 00111...111111
502     ***/
503 
504     long startmask = -1L << startIndex;
505     long endmask = -1L >>> -endIndex;  // 64-(endIndex&0x3f) is the same as -endIndex due to wrap
506 
507     if (startWord == endWord) {
508       bits[startWord] ^= (startmask & endmask);
509       return;
510     }
511 
512     bits[startWord] ^= startmask;
513 
514     for (int i=startWord+1; i<endWord; i++) {
515       bits[i] = ~bits[i];
516     }
517 
518     bits[endWord] ^= endmask;
519   }

596   /** Returns the index of the first set bit starting at the index specified.
597    *  -1 is returned if there are no more set bits.
598    */
599   public int nextSetBit(int index) {
600     int i = index>>6;
601     if (i>=wlen) return -1;
602     int subIndex = index & 0x3f;      // index within the word
603     long word = bits[i] >> subIndex;  // skip all the bits to the right of index
604 
605     if (word!=0) {
606       return (i<<6) + subIndex + BitUtil.ntz(word);
607     }
608 
609     while(++i < wlen) {
610       word = bits[i];
611       if (word!=0) return (i<<6) + BitUtil.ntz(word);
612     }
613 
614     return -1;
615   }
616 
617   /** Returns the index of the first set bit starting at the index specified.
618    *  -1 is returned if there are no more set bits.
619    */
620   public long nextSetBit(long index) {
621     int i = (int)(index>>>6);
622     if (i>=wlen) return -1;
623     int subIndex = (int)index & 0x3f; // index within the word
624     long word = bits[i] >>> subIndex;  // skip all the bits to the right of index
625 
626     if (word!=0) {
627       return (((long)i)<<6) + (subIndex + BitUtil.ntz(word));
628     }
629 
630     while(++i < wlen) {
631       word = bits[i];
632       if (word!=0) return (((long)i)<<6) + BitUtil.ntz(word);
633     }
634 
635     return -1;
636   }

642     try {
643       OpenBitSet obs = (OpenBitSet)super.clone();
644       obs.bits = (long[]) obs.bits.clone();  // hopefully an array clone is as fast(er) than arraycopy
645       return obs;
646     } catch (CloneNotSupportedException e) {
647       throw new RuntimeException(e);
648     }
649   }

651   /** this = this AND other */
652   public void intersect(OpenBitSet other) {
653     int newLen= Math.min(this.wlen,other.wlen);
654     long[] thisArr = this.bits;
655     long[] otherArr = other.bits;
656     // testing against zero can be more efficient
657     int pos=newLen;
658     while(--pos>=0) {
659       thisArr[pos] &= otherArr[pos];
660     }
661     if (this.wlen > newLen) {
662       // fill zeros from the new shorter length to the old length
663       Arrays.fill(bits,newLen,this.wlen,0);
664     }
665     this.wlen = newLen;
666   }

668   /** this = this OR other */
669   public void union(OpenBitSet other) {
670     int newLen = Math.max(wlen,other.wlen);
671     ensureCapacityWords(newLen);
672 
673     long[] thisArr = this.bits;
674     long[] otherArr = other.bits;
675     int pos=Math.min(wlen,other.wlen);
676     while(--pos>=0) {
677       thisArr[pos] |= otherArr[pos];
678     }
679     if (this.wlen < newLen) {
680       System.arraycopy(otherArr, this.wlen, thisArr, this.wlen, newLen-this.wlen);
681     }
682     this.wlen = newLen;
683   }

686   /** Remove all elements set in other. this = this AND_NOT other */
687   public void remove(OpenBitSet other) {
688     int idx = Math.min(wlen,other.wlen);
689     long[] thisArr = this.bits;
690     long[] otherArr = other.bits;
691     while(--idx>=0) {
692       thisArr[idx] &= ~otherArr[idx];
693     }
694   }

696   /** this = this XOR other */
697   public void xor(OpenBitSet other) {
698     int newLen = Math.max(wlen,other.wlen);
699     ensureCapacityWords(newLen);
700 
701     long[] thisArr = this.bits;
702     long[] otherArr = other.bits;
703     int pos=Math.min(wlen,other.wlen);
704     while(--pos>=0) {
705       thisArr[pos] ^= otherArr[pos];
706     }
707     if (this.wlen < newLen) {
708       System.arraycopy(otherArr, this.wlen, thisArr, this.wlen, newLen-this.wlen);
709     }
710     this.wlen = newLen;
711   }

731   /** returns true if the sets have any elements in common */
732   public boolean intersects(OpenBitSet other) {
733     int pos = Math.min(this.wlen, other.wlen);
734     long[] thisArr = this.bits;
735     long[] otherArr = other.bits;
736     while (--pos>=0) {
737       if ((thisArr[pos] & otherArr[pos])!=0) return true;
738     }
739     return false;
740   }

760   /** Lowers numWords, the number of words in use,
761    * by checking for trailing zero words.
762    */
763   public void trimTrailingZeros() {
764     int idx = wlen-1;
765     while (idx>=0 && bits[idx]==0) idx--;
766     wlen = idx+1;
767   }

769   /** returns the number of 64 bit words it would take to hold numBits */
770   public static int bits2words(long numBits) {
771    return (int)(((numBits-1)>>>6)+1);
772   }

775   /** returns true if both sets have the same bits set */
776   public boolean equals(Object o) {
777     if (this == o) return true;
778     if (!(o instanceof OpenBitSet)) return false;
779     OpenBitSet a;
780     OpenBitSet b = (OpenBitSet)o;
781     // make a the larger set.
782     if (b.wlen > this.wlen) {
783       a = b; b=this;
784     } else {
785       a=this;
786     }
787 
788     // check for any set bits out of the range of b
789     for (int i=a.wlen-1; i>=b.wlen; i--) {
790       if (a.bits[i]!=0) return false;
791     }
792 
793     for (int i=b.wlen-1; i>=0; i--) {
794       if (a.bits[i] != b.bits[i]) return false;
795     }
796 
797     return true;
798   }

801   public int hashCode() {
802       long h = 0x98761234;  // something non-zero for length==0
803       for (int i = bits.length; --i>=0;) {
804       h ^= bits[i];
805       h = (h << 1) | (h >>> 63); // rotate left
806     }
807     return (int)((h>>32) ^ h);  // fold leftmost bits into right
808   }
