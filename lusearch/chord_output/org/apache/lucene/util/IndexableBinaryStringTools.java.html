20  import java.nio.CharBuffer;
21  import java.nio.ByteBuffer;
22  
23  /**
24   * Provides support for converting byte sequences to Strings and back again.
25   * The resulting Strings preserve the original byte sequences' sort order.
26   * 
27   * The Strings are constructed using a Base 8000h encoding of the original
28   * binary data - each char of an encoded String represents a 15-bit chunk
29   * from the byte sequence.  Base 8000h was chosen because it allows for all
30   * lower 15 bits of char to be used without restriction; the surrogate range 
31   * [U+D8000-U+DFFF] does not represent valid chars, and would require
32   * complicated handling to avoid them and allow use of char's high bit.
33   * 
34   * Although unset bits are used as padding in the final char, the original
35   * byte sequence could contain trailing bytes with no set bits (null bytes):
36   * padding is indistinguishable from valid information.  To overcome this
37   * problem, a char is appended, indicating the number of encoded bytes in the
38   * final content char.
39   * 
40   * This class's operations are defined over CharBuffers and ByteBuffers, to
41   * allow for wrapped arrays to be reused, reducing memory allocation costs for
42   * repeated operations.  Note that this class calls array() and arrayOffset()
43   * on the CharBuffers and ByteBuffers it uses, so only wrapped arrays may be
44   * used.  This class interprets the arrayOffset() and limit() values returned by
45   * its input buffers as beginning and end+1 positions on the wrapped array,
46   * respectively; similarly, on the output buffer, arrayOffset() is the first
47   * position written to, and limit() is set to one past the final output array
48   * position.
49   */
50  public class IndexableBinaryStringTools {
51  
52    private static final CodingCase[] CODING_CASES = {
53      // CodingCase(int initialShift, int finalShift)
54      new CodingCase( 7, 1   ),
55      // CodingCase(int initialShift, int middleShift, int finalShift)
56      new CodingCase(14, 6, 2),
57      new CodingCase(13, 5, 3),
58      new CodingCase(12, 4, 4),
59      new CodingCase(11, 3, 5),
60      new CodingCase(10, 2, 6),
61      new CodingCase( 9, 1, 7),
62      new CodingCase( 8, 0   )
63    };
64  
65    // Export only static methods
66    private IndexableBinaryStringTools() {}

68    /**
69     * Returns the number of chars required to encode the given byte sequence.
70     * 
71     * @param original The byte sequence to be encoded.  Must be backed by an array.
72     * @return The number of chars required to encode the given byte sequence
73     * @throws IllegalArgumentException If the given ByteBuffer is not backed by an array
74     */
75    public static int getEncodedLength(ByteBuffer original) 
76      throws IllegalArgumentException {
77      if (original.hasArray()) {
78        // Use long for intermediaries to protect against overflow
79        long length = (long)(original.limit() - original.arrayOffset());
80        return (int)((length * 8L + 14L) / 15L) + 1;
81      } else {
82        throw new IllegalArgumentException("original argument must have a backing array");
83      }
84    }

86    /**
87     * Returns the number of bytes required to decode the given char sequence.
88     * 
89     * @param encoded The char sequence to be encoded.  Must be backed by an array.
90     * @return The number of bytes required to decode the given char sequence
91     * @throws IllegalArgumentException If the given CharBuffer is not backed by an array
92     */
93    public static int getDecodedLength(CharBuffer encoded) 
94      throws IllegalArgumentException {
95      if (encoded.hasArray()) {
96        int numChars = encoded.limit() - encoded.arrayOffset() - 1;
97        if (numChars <= 0) {
98          return 0;
99        } else {
100         int numFullBytesInFinalChar = encoded.charAt(encoded.limit() - 1);
101         int numEncodedChars = numChars - 1;
102         return (numEncodedChars * 15 + 7) / 8 + numFullBytesInFinalChar;
103       }
104     } else {
105       throw new IllegalArgumentException("encoded argument must have a backing array");
106     }
107   }

109   /**
110    * Encodes the input byte sequence into the output char sequence.  Before
111    * calling this method, ensure that the output CharBuffer has sufficient
112    * capacity by calling {@link #getEncodedLength(java.nio.ByteBuffer)}.
113    * 
114    * @param input The byte sequence to encode
115    * @param output Where the char sequence encoding result will go.  The limit
116    *  is set to one past the position of the final char.
117    * @throws IllegalArgumentException If either the input or the output buffer
118    *  is not backed by an array
119    */
120   public static void encode(ByteBuffer input, CharBuffer output) {
121     if (input.hasArray() && output.hasArray()) {
122       byte[] inputArray = input.array();
123       int inputOffset = input.arrayOffset();
124       int inputLength = input.limit() - inputOffset; 
125       char[] outputArray = output.array();
126       int outputOffset = output.arrayOffset();
127       int outputLength = getEncodedLength(input);
128       output.limit(outputOffset + outputLength); // Set output final pos + 1
129       output.position(0);
130       if (inputLength > 0) {
131         int inputByteNum = inputOffset;
132         int caseNum = 0;
133         int outputCharNum = outputOffset;
134         CodingCase codingCase;
135         for ( ; inputByteNum + CODING_CASES[caseNum].numBytes <= inputLength ;
136               ++outputCharNum                                                 ) {
137           codingCase = CODING_CASES[caseNum];
138           if (2 == codingCase.numBytes) {
139             outputArray[outputCharNum]
140               = (char)(((inputArray[inputByteNum] & 0xFF) << codingCase.initialShift)
141                        + (((inputArray[inputByteNum + 1] & 0xFF) >>> codingCase.finalShift)
142                           & codingCase.finalMask)
143                        & (short)0x7FFF);
144           } else { // numBytes is 3
145             outputArray[outputCharNum] 
146               = (char)(((inputArray[inputByteNum] & 0xFF) << codingCase.initialShift)
147                        + ((inputArray[inputByteNum + 1] & 0xFF) << codingCase.middleShift)
148                        + (((inputArray[inputByteNum + 2] & 0xFF) >>> codingCase.finalShift) 
149                           & codingCase.finalMask)
150                        & (short)0x7FFF);          
151           }
152           inputByteNum += codingCase.advanceBytes;          
153           if (++caseNum == CODING_CASES.length) {
154             caseNum = 0;
155           }
156         }
157         // Produce final char (if any) and trailing count chars.
158         codingCase = CODING_CASES[caseNum];
159         
160         if (inputByteNum + 1 < inputLength) { // codingCase.numBytes must be 3
161           outputArray[outputCharNum++] 
162             = (char)((((inputArray[inputByteNum] & 0xFF) << codingCase.initialShift)
163                       + ((inputArray[inputByteNum + 1] & 0xFF) << codingCase.middleShift))
164                      & (short)0x7FFF);
165           // Add trailing char containing the number of full bytes in final char
166           outputArray[outputCharNum++] = (char)1;
167         } else if (inputByteNum < inputLength) {
168           outputArray[outputCharNum++] 
169             = (char)(((inputArray[inputByteNum] & 0xFF) << codingCase.initialShift)
170                      & (short)0x7FFF);
171           // Add trailing char containing the number of full bytes in final char
172           outputArray[outputCharNum++] = caseNum == 0 ? (char)1 : (char)0;
173         } else { // No left over bits - last char is completely filled.
174           // Add trailing char containing the number of full bytes in final char
175           outputArray[outputCharNum++] = (char)1;
176         }
177       }
178     } else {
179       throw new IllegalArgumentException("Arguments must have backing arrays");
180     }
181   }

183   /**
184    * Decodes the input char sequence into the output byte sequence.  Before
185    * calling this method, ensure that the output ByteBuffer has sufficient
186    * capacity by calling {@link #getDecodedLength(java.nio.CharBuffer)}.
187    * 
188    * @param input The char sequence to decode
189    * @param output Where the byte sequence decoding result will go.  The limit
190    *  is set to one past the position of the final char.
191    * @throws IllegalArgumentException If either the input or the output buffer
192    *  is not backed by an array
193    */
194   public static void decode(CharBuffer input, ByteBuffer output) {
195     if (input.hasArray() && output.hasArray()) {
196       int numInputChars = input.limit() - input.arrayOffset() - 1;
197       int numOutputBytes = getDecodedLength(input);
198       output.limit(numOutputBytes + output.arrayOffset()); // Set output final pos + 1
199       output.position(0);
200       byte[] outputArray = output.array();
201       char[] inputArray = input.array();
202       if (numOutputBytes > 0) {
203         int caseNum = 0;
204         int outputByteNum = output.arrayOffset();
205         int inputCharNum = input.arrayOffset();
206         short inputChar;
207         CodingCase codingCase;
208         for ( ; inputCharNum < numInputChars - 1 ; ++inputCharNum) {
209           codingCase = CODING_CASES[caseNum];
210           inputChar = (short)inputArray[inputCharNum];
211           if (2 == codingCase.numBytes) {
212             if (0 == caseNum) {
213               outputArray[outputByteNum] = (byte)(inputChar >>> codingCase.initialShift);
214             } else {
215               outputArray[outputByteNum] += (byte)(inputChar >>> codingCase.initialShift);
216             }
217             outputArray[outputByteNum + 1] = (byte)((inputChar & codingCase.finalMask) 
218                                                     << codingCase.finalShift);
219           } else { // numBytes is 3
220             outputArray[outputByteNum] += (byte)(inputChar >>> codingCase.initialShift);
221             outputArray[outputByteNum + 1] = (byte)((inputChar & codingCase.middleMask)
222                                                     >>> codingCase.middleShift);
223             outputArray[outputByteNum + 2] = (byte)((inputChar & codingCase.finalMask) 
224                                                     << codingCase.finalShift);
225           }
226           outputByteNum += codingCase.advanceBytes;
227           if (++caseNum == CODING_CASES.length) {
228             caseNum = 0;
229           }
230         }
231         // Handle final char
232         inputChar = (short)inputArray[inputCharNum];
233         codingCase = CODING_CASES[caseNum];
234         if (0 == caseNum) {
235           outputArray[outputByteNum] = 0;
236         }
237         outputArray[outputByteNum] += (byte)(inputChar >>> codingCase.initialShift);
238         int bytesLeft = numOutputBytes - outputByteNum;
239         if (bytesLeft > 1) {
240           if (2 == codingCase.numBytes) {
241             outputArray[outputByteNum + 1] = (byte)((inputChar & codingCase.finalMask) 
242                                                     >>> codingCase.finalShift);
243           } else { // numBytes is 3
244             outputArray[outputByteNum + 1] = (byte)((inputChar & codingCase.middleMask)
245                                                     >>> codingCase.middleShift);
246             if (bytesLeft > 2) {
247               outputArray[outputByteNum + 2] = (byte)((inputChar & codingCase.finalMask) 
248                                                       << codingCase.finalShift);
249             }
250           }
251         }
252       }
253     } else {
254       throw new IllegalArgumentException("Arguments must have backing arrays");
255     }
256   }

258   /**
259    * Decodes the given char sequence, which must have been encoded by
260    * {@link #encode(java.nio.ByteBuffer)} or 
261    * {@link #encode(java.nio.ByteBuffer, java.nio.CharBuffer)}.
262    * 
263    * @param input The char sequence to decode
264    * @return A byte sequence containing the decoding result.  The limit
265    *  is set to one past the position of the final char.
266    * @throws IllegalArgumentException If the input buffer is not backed by an
267    *  array
268    */
269   public static ByteBuffer decode(CharBuffer input) {
270     byte[] outputArray = new byte[getDecodedLength(input)];
271     ByteBuffer output = ByteBuffer.wrap(outputArray);
272     decode(input, output);
273     return output;
274   }
275 
276   /**
277    * Encodes the input byte sequence.
278    * 
279    * @param input The byte sequence to encode
280    * @return A char sequence containing the encoding result.  The limit is set
281    *  to one past the position of the final char.
282    * @throws IllegalArgumentException If the input buffer is not backed by an
283    *  array
284    */
285   public static CharBuffer encode(ByteBuffer input) {
286     char[] outputArray = new char[getEncodedLength(input)];
287     CharBuffer output = CharBuffer.wrap(outputArray);
288     encode(input, output);
289     return output;
290   }
291   
292   static class CodingCase {
293     int numBytes, initialShift, middleShift, finalShift, advanceBytes = 2;
294     short middleMask, finalMask;
295 
296     CodingCase(int initialShift, int middleShift, int finalShift) {
297       this.numBytes = 3;
298       this.initialShift = initialShift;
299       this.middleShift = middleShift;
300       this.finalShift = finalShift;
301       this.finalMask = (short)((short)0xFF >>> finalShift);
302       this.middleMask = (short)((short)0xFF << middleShift);
303     }
304 
305     CodingCase(int initialShift, int finalShift) {
306       this.numBytes = 2;
307       this.initialShift = initialShift;
308       this.finalShift = finalShift;
309       this.finalMask = (short)((short)0xFF >>> finalShift);
310       if (finalShift != 0) {
311         advanceBytes = 1; 
312       }
313     }
314   }
