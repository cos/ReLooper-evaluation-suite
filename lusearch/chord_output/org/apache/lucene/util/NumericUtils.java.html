25  /**
26   * This is a helper class to generate prefix-encoded representations for numerical values
27   * and supplies converters to represent float/double values as sortable integers/longs.
28   *
29   * <p>To quickly execute range queries in Apache Lucene, a range is divided recursively
30   * into multiple intervals for searching: The center of the range is searched only with
31   * the lowest possible precision in the trie, while the boundaries are matched
32   * more exactly. This reduces the number of terms dramatically.
33   *
34   * <p>This class generates terms to achieve this: First the numerical integer values need to
35   * be converted to strings. For that integer values (32 bit or 64 bit) are made unsigned
36   * and the bits are converted to ASCII chars with each 7 bit. The resulting string is
37   * sortable like the original integer value. Each value is also prefixed
38   * (in the first char) by the <code>shift</code> value (number of bits removed) used
39   * during encoding.
40   *
41   * <p>To also index floating point numbers, this class supplies two methods to convert them
42   * to integer values by changing their bit layout: {@link #doubleToSortableLong},
43   * {@link #floatToSortableInt}. You will have no precision loss by
44   * converting floating point numbers to integers and back (only that the integer form
45   * is not usable). Other data types like dates can easily converted to longs or ints (e.g.
46   * date to long: {@link java.util.Date#getTime}).
47   *
48   * <p>For easy usage, the trie algorithm is implemented for indexing inside
49   * {@link NumericTokenStream} that can index <code>int</code>, <code>long</code>,
50   * <code>float</code>, and <code>double</code>. For querying,
51   * {@link NumericRangeQuery} and {@link NumericRangeFilter} implement the query part
52   * for the same data types.
53   *
54   * <p>This class can also be used, to generate lexicographically sortable (according
55   * {@link String#compareTo(String)}) representations of numeric data types for other
56   * usages (e.g. sorting).
57   *
58   * <p><font color="red"><b>NOTE:</b> This API is experimental and
59   * might change in incompatible ways in the next release.</font>
60   *
61   * @since 2.9
62   */
63  public final class NumericUtils {

74     * Expert: Longs are stored at lower precision by shifting off lower bits. The shift count is
75     * stored as <code>SHIFT_START_LONG+shift</code> in the first character

77    public static final char SHIFT_START_LONG = (char)0x20;

79    /**

81     * for encoding <code>long</code> values.
82     * @see #longToPrefixCoded(long,int,char[])
83     */
84    public static final int BUF_SIZE_LONG = 63/7 + 2;

93     * Expert: The maximum term length (used for <code>char[]</code> buffer size)
94     * for encoding <code>int</code> values.
95     * @see #intToPrefixCoded(int,int,char[])
96     */
97    public static final int BUF_SIZE_INT = 31/7 + 2;

99    /**
100    * Expert: Returns prefix coded bits after reducing the precision by <code>shift</code> bits.
101    * This is method is used by {@link NumericTokenStream}.
102    * @param val the numeric value
103    * @param shift how many bits to strip from the right
104    * @param buffer that will contain the encoded chars, must be at least of {@link #BUF_SIZE_LONG}
105    * length
106    * @return number of chars written to buffer
107    */
108   public static int longToPrefixCoded(final long val, final int shift, final char[] buffer) {
109     if (shift>63 || shift<0)
110       throw new IllegalArgumentException("Illegal shift value, must be 0..63");
111     int nChars = (63-shift)/7 + 1, len = nChars+1;
112     buffer[0] = (char)(SHIFT_START_LONG + shift);
113     long sortableBits = val ^ 0x8000000000000000L;
114     sortableBits >>>= shift;
115     while (nChars>=1) {
116       // Store 7 bits per character for good efficiency when UTF-8 encoding.
117       // The whole number is right-justified so that lucene can prefix-encode
118       // the terms more efficiently.
119       buffer[nChars--] = (char)(sortableBits & 0x7f);
120       sortableBits >>>= 7;
121     }
122     return len;
123   }

128    * @param val the numeric value
129    * @param shift how many bits to strip from the right

131   public static String longToPrefixCoded(final long val, final int shift) {
132     final char[] buffer = new char[BUF_SIZE_LONG];
133     final int len = longToPrefixCoded(val, shift, buffer);

195   /**
196    * Returns a long from prefixCoded characters.
197    * Rightmost bits will be zero for lower precision codes.
198    * This method can be used to decode e.g. a stored field.
199    * @throws NumberFormatException if the supplied string is
200    * not correctly prefix encoded.
201    * @see #longToPrefixCoded(long)
202    */
203   public static long prefixCodedToLong(final String prefixCoded) {
204     final int shift = prefixCoded.charAt(0)-SHIFT_START_LONG;
205     if (shift>63 || shift<0)
206       throw new NumberFormatException("Invalid shift value in prefixCoded string (is encoded value really a LONG?)");
207     long sortableBits = 0L;
208     for (int i=1, len=prefixCoded.length(); i<len; i++) {
209       sortableBits <<= 7;
210       final char ch = prefixCoded.charAt(i);
211       if (ch>0x7f) {
212         throw new NumberFormatException(
213           "Invalid prefixCoded numerical value representation (char "+
214           Integer.toHexString((int)ch)+" at position "+i+" is invalid)"
215         );
216       }
217       sortableBits |= (long)ch;
218     }
219     return (sortableBits << shift) ^ 0x8000000000000000L;
220   }
221 
222   /**
223    * Returns an int from prefixCoded characters.
224    * Rightmost bits will be zero for lower precision codes.
225    * This method can be used to decode e.g. a stored field.
226    * @throws NumberFormatException if the supplied string is
227    * not correctly prefix encoded.
228    * @see #intToPrefixCoded(int)
229    */
230   public static int prefixCodedToInt(final String prefixCoded) {
231     final int shift = prefixCoded.charAt(0)-SHIFT_START_INT;
232     if (shift>31 || shift<0)
233       throw new NumberFormatException("Invalid shift value in prefixCoded string (is encoded value really an INT?)");
234     int sortableBits = 0;
235     for (int i=1, len=prefixCoded.length(); i<len; i++) {
236       sortableBits <<= 7;
237       final char ch = prefixCoded.charAt(i);
238       if (ch>0x7f) {
239         throw new NumberFormatException(
240           "Invalid prefixCoded numerical value representation (char "+
241           Integer.toHexString((int)ch)+" at position "+i+" is invalid)"
242         );
243       }
244       sortableBits |= (int)ch;
245     }
246     return (sortableBits << shift) ^ 0x80000000;
247   }

249   /**
250    * Converts a <code>double</code> value to a sortable signed <code>long</code>.
251    * The value is converted by getting their IEEE 754 floating-point &quot;double format&quot;
252    * bit layout and then some bits are swapped, to be able to compare the result as long.
253    * By this the precision is not reduced, but the value can easily used as a long.
254    * @see #sortableLongToDouble
255    */
256   public static long doubleToSortableLong(double val) {
257     long f = Double.doubleToRawLongBits(val);
258     if (f<0) f ^= 0x7fffffffffffffffL;
259     return f;
260   }

341    * You may implement a builder that adds clauses to a

344    * method.

346    */
347   public static void splitIntRange(final IntRangeBuilder builder,

350     splitRange(builder, 32, precisionStep, (long)minBound, (long)maxBound);
351   }

353   /** This helper does the splitting for both 32 and 64 bit. */
354   private static void splitRange(
355     final Object builder, final int valSize,
356     final int precisionStep, long minBound, long maxBound
357   ) {
358     if (precisionStep < 1)
359       throw new IllegalArgumentException("precisionStep must be >=1");
360     if (minBound > maxBound) return;
361     for (int shift=0; ; shift += precisionStep) {
362       // calculate new bounds for inner precision
363       final long diff = 1L << (shift+precisionStep),
364         mask = ((1L<<precisionStep) - 1L) << shift;
365       final boolean
366         hasLower = (minBound & mask) != 0L,
367         hasUpper = (maxBound & mask) != mask;
368       final long
369         nextMinBound = (hasLower ? (minBound + diff) : minBound) & ~mask,
370         nextMaxBound = (hasUpper ? (maxBound - diff) : maxBound) & ~mask;
371 
372       if (shift+precisionStep>=valSize || nextMinBound>nextMaxBound) {
373         // We are in the lowest precision or the next precision is not available.
374         addRange(builder, valSize, minBound, maxBound, shift);
375         // exit the split recursion loop
376         break;
377       }
378       
379       if (hasLower)
380         addRange(builder, valSize, minBound, minBound | mask, shift);
381       if (hasUpper)
382         addRange(builder, valSize, maxBound & ~mask, maxBound, shift);
383       
384       // recurse to next precision
385       minBound = nextMinBound;
386       maxBound = nextMaxBound;
387     }
388   }
389   
390   /** Helper that delegates to correct range builder */
391   private static void addRange(
392     final Object builder, final int valSize,
393     long minBound, long maxBound,
394     final int shift
395   ) {
396     // for the max bound set all lower bits (that were shifted away):
397     // this is important for testing or other usages of the splitted range
398     // (e.g. to reconstruct the full range). The prefixEncoding will remove
399     // the bits anyway, so they do not hurt!
400     maxBound |= (1L << shift) - 1L;
401     // delegate to correct range builder
402     switch(valSize) {
403       case 64:
404         ((LongRangeBuilder)builder).addRange(minBound, maxBound, shift);
405         break;
406       case 32:
407         ((IntRangeBuilder)builder).addRange((int)minBound, (int)maxBound, shift);
408         break;
409       default:
410         // Should not happen!
411         throw new IllegalArgumentException("valSize must be 32 or 64.");
412     }
413   }
