20  import java.util.IdentityHashMap;
21  import java.util.Map;
22  
23  /**
24   * An average, best guess, MemoryModel that should work okay on most systems.
25   * 
26   */
27  public class AverageGuessMemoryModel extends MemoryModel {
28    // best guess primitive sizes
29    private final Map sizes = new IdentityHashMap() {
30      {
31        put(boolean.class, new Integer(1));
32        put(byte.class, new Integer(1));
33        put(char.class, new Integer(2));
34        put(short.class, new Integer(2));
35        put(int.class, new Integer(4));
36        put(float.class, new Integer(4));
37        put(double.class, new Integer(8));
38        put(long.class, new Integer(8));
39      }
40    };
41  
42    /*
43     * (non-Javadoc)
44     * 
45     * @see org.apache.lucene.util.MemoryModel#getArraySize()
46     */
47    public int getArraySize() {
48      return 16;
49    }
50  
51    /*
52     * (non-Javadoc)
53     * 
54     * @see org.apache.lucene.util.MemoryModel#getClassSize()
55     */
56    public int getClassSize() {
57      return 8;
58    }
59  
60    /* (non-Javadoc)
61     * @see org.apache.lucene.util.MemoryModel#getPrimitiveSize(java.lang.Class)
62     */
63    public int getPrimitiveSize(Class clazz) {
64      return ((Integer) sizes.get(clazz)).intValue();
65    }
66  
67    /* (non-Javadoc)
68     * @see org.apache.lucene.util.MemoryModel#getReferenceSize()
69     */
70    public int getReferenceSize() {
71      return 4;
72    }
73  
74  }
