20  import java.io.IOException;
21  import java.util.BitSet;
22  
23  import org.apache.lucene.search.DocIdSet;
24  import org.apache.lucene.search.DocIdSetIterator;
25  
26  /**
27   * Stores and iterate on sorted integers in compressed form in RAM. <br>
28   * The code for compressing the differences between ascending integers was
29   * borrowed from {@link org.apache.lucene.store.IndexInput} and
30   * {@link org.apache.lucene.store.IndexOutput}.
31   * <p>
32   * <b>NOTE:</b> this class assumes the stored integers are doc Ids (hence why it
33   * extends {@link DocIdSet}). Therefore its {@link #iterator()} assumes {@link
34   * DocIdSetIterator#NO_MORE_DOCS} can be used as sentinel. If you intent to use
35   * this value, then make sure it's not used during search flow.
36   */
37  public class SortedVIntList extends DocIdSet {
38    /** When a BitSet has fewer than 1 in BITS2VINTLIST_SIZE bits set,
39     * a SortedVIntList representing the index numbers of the set bits
40     * will be smaller than that BitSet.
41     */
42    final static int BITS2VINTLIST_SIZE = 8;
43  
44    private int size;
45    private byte[] bytes;
46    private int lastBytePos;
47      
48    /**
49     *  Create a SortedVIntList from all elements of an array of integers.
50     *
51     * @param  sortedInts  A sorted array of non negative integers.
52     */
53    public SortedVIntList(int[] sortedInts) {
54      this(sortedInts, sortedInts.length);
55    }
56  
57    /**
58     * Create a SortedVIntList from an array of integers.
59     * @param  sortedInts  An array of sorted non negative integers.
60     * @param  inputSize   The number of integers to be used from the array.
61     */
62    public SortedVIntList(int[] sortedInts, int inputSize) {
63      SortedVIntListBuilder builder = new SortedVIntListBuilder();
64      for (int i = 0; i < inputSize; i++) {
65        builder.addInt(sortedInts[i]);
66      }
67      builder.done();
68    }
69  
70    /**
71     * Create a SortedVIntList from a BitSet.
72     * @param  bits  A bit set representing a set of integers.
73     */
74    public SortedVIntList(BitSet bits) {
75      SortedVIntListBuilder builder = new SortedVIntListBuilder();
76      int nextInt = bits.nextSetBit(0);
77      while (nextInt != -1) {
78        builder.addInt(nextInt);
79        nextInt = bits.nextSetBit(nextInt + 1);
80      }
81      builder.done();
82    }
83  
84    /**
85     * Create a SortedVIntList from an OpenBitSet.
86     * @param  bits  A bit set representing a set of integers.
87     */
88    public SortedVIntList(OpenBitSet bits) {
89      SortedVIntListBuilder builder = new SortedVIntListBuilder();
90      int nextInt = bits.nextSetBit(0);
91      while (nextInt != -1) {
92        builder.addInt(nextInt);
93        nextInt = bits.nextSetBit(nextInt + 1);
94      }
95      builder.done();
96    }
97  
98    /**
99     * Create a SortedVIntList.
100    * @param  docIdSetIterator  An iterator providing document numbers as a set of integers.
101    *                  This DocIdSetIterator is iterated completely when this constructor
102    *                  is called and it must provide the integers in non
103    *                  decreasing order.
104    */
105   public SortedVIntList(DocIdSetIterator docIdSetIterator) throws IOException {
106     SortedVIntListBuilder builder = new SortedVIntListBuilder();
107     int doc;
108     while ((doc = docIdSetIterator.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
109       builder.addInt(doc);
110     }
111     builder.done();
112   }
113 
114 
115   private class SortedVIntListBuilder {
116     private int lastInt = 0;
117     
118     SortedVIntListBuilder() {
119       initBytes();
120       lastInt = 0;
121     }
122 
123     void addInt(int nextInt) {
124       int diff = nextInt - lastInt;
125       if (diff < 0) {
126         throw new IllegalArgumentException(
127             "Input not sorted or first element negative.");
128       }
129   
130       if ((lastBytePos + MAX_BYTES_PER_INT) > bytes.length) {
131         // biggest possible int does not fit
132         resizeBytes((bytes.length * 2) + MAX_BYTES_PER_INT);
133       }
134   
135       // See org.apache.lucene.store.IndexOutput.writeVInt()
136       while ((diff & ~VB1) != 0) { // The high bit of the next byte needs to be set.
137         bytes[lastBytePos++] = (byte) ((diff & VB1) | ~VB1);
138         diff >>>= BIT_SHIFT;
139       }
140       bytes[lastBytePos++] = (byte) diff; // Last byte, high bit not set.
141       size++;
142       lastInt = nextInt;
143     }
144     
145     void done() {
146       resizeBytes(lastBytePos);
147     }
148   }
149 
150 
151   private void initBytes() {
152     size = 0;
153     bytes = new byte[128]; // initial byte size
154     lastBytePos = 0;
155   }
156 
157   private void resizeBytes(int newSize) {
158     if (newSize != bytes.length) {
159       byte[] newBytes = new byte[newSize];
160       System.arraycopy(bytes, 0, newBytes, 0, lastBytePos);
161       bytes = newBytes;
162     }
163   }
164 
165   private static final int VB1 = 0x7F;
166   private static final int BIT_SHIFT = 7;
167   private final int MAX_BYTES_PER_INT = (31 / BIT_SHIFT) + 1;
168 
169   /**
170    * @return    The total number of sorted integers.
171    */
172   public int size() {
173     return size;
174   }
175 
176   /**
177    * @return The size of the byte array storing the compressed sorted integers.
178    */
179   public int getByteSize() {
180     return bytes.length;
181   }
182 
183   /** This DocIdSet implementation is cacheable. */
184   public boolean isCacheable() {
185     return true;
186   }
187 
188   /**
189    * @return    An iterator over the sorted integers.
190    */
191   public DocIdSetIterator iterator() {
192     return new DocIdSetIterator() {
193       int bytePos = 0;
194       int lastInt = 0;
195       int doc = -1;
196       
197       private void advance() {
198         // See org.apache.lucene.store.IndexInput.readVInt()
199         byte b = bytes[bytePos++];
200         lastInt += b & VB1;
201         for (int s = BIT_SHIFT; (b & ~VB1) != 0; s += BIT_SHIFT) {
202           b = bytes[bytePos++];
203           lastInt += (b & VB1) << s;
204         }
205       }
206       
207       /** @deprecated use {@link #docID()} instead. */
208       public int doc() {return lastInt;}
209       
210       public int docID() {
211         return doc;
212       }
213       
214       /** @deprecated use {@link #nextDoc()} instead. */
215       public boolean next() {
216         return nextDoc() != NO_MORE_DOCS;
217       }
218 
219       public int nextDoc() {
220         if (bytePos >= lastBytePos) {
221           doc = NO_MORE_DOCS;
222         } else {
223           advance();
224           doc = lastInt;
225         }
226         return doc;
227       }
228       
229       /** @deprecated use {@link #advance(int)} instead. */
230       public boolean skipTo(int docNr) {
231         return advance(docNr) != NO_MORE_DOCS;
232       }
233       
234       public int advance(int target) {
235         while (bytePos < lastBytePos) {
236           advance();
237           if (lastInt >= target) {
238             return doc = lastInt;
239           }
240         }
241         return doc = NO_MORE_DOCS;
242       }
243       
244     };
245   }
246 }
