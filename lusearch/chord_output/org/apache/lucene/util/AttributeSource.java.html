32  /**
33   * An AttributeSource contains a list of different {@link AttributeImpl}s,
34   * and methods to add and get them. There can only be a single instance
35   * of an attribute in the same AttributeSource instance. This is ensured
36   * by passing in the actual type of the Attribute (Class&lt;Attribute&gt;) to 
37   * the {@link #addAttribute(Class)}, which then checks if an instance of
38   * that type is already present. If yes, it returns the instance, otherwise
39   * it creates a new instance and returns it.
40   */
41  public class AttributeSource {
42    /**
43     * An AttributeFactory creates instances of {@link AttributeImpl}s.
44     */
45    public static abstract class AttributeFactory {
46      /**
47       * returns an {@link AttributeImpl} for the supplied {@link Attribute} interface class.
48       * <p>Signature for Java 1.5: <code>public AttributeImpl createAttributeInstance(Class%lt;? extends Attribute&gt; attClass)</code>
49       */
50      public abstract AttributeImpl createAttributeInstance(Class attClass);
51      
52      /**
53       * This is the default factory that creates {@link AttributeImpl}s using the
54       * class name of the supplied {@link Attribute} interface class by appending <code>Impl</code> to it.
55       */
56      public static final AttributeFactory DEFAULT_ATTRIBUTE_FACTORY = new DefaultAttributeFactory();
57      
58      private static final class DefaultAttributeFactory extends AttributeFactory {
59        private static final WeakHashMap/*<Class<? extends Attribute>, WeakReference<Class<? extends AttributeImpl>>>*/ attClassImplMap = new WeakHashMap();
60        
61        private DefaultAttributeFactory() {}
62      
63        public AttributeImpl createAttributeInstance(Class attClass) {
64          try {
65            return (AttributeImpl) getClassForInterface(attClass).newInstance();
66          } catch (InstantiationException e) {
67            throw new IllegalArgumentException("Could not instantiate implementing class for " + attClass.getName());
68          } catch (IllegalAccessException e) {
69            throw new IllegalArgumentException("Could not instantiate implementing class for " + attClass.getName());
70          }
71        }
72        
73        private static Class getClassForInterface(Class attClass) {
74          synchronized(attClassImplMap) {
75            final WeakReference ref = (WeakReference) attClassImplMap.get(attClass);
76            Class clazz = (ref == null) ? null : ((Class) ref.get());
77            if (clazz == null) {
78              try {
79                attClassImplMap.put(attClass, new WeakReference(
80                  clazz = Class.forName(attClass.getName() + "Impl", true, attClass.getClassLoader())
81                ));
82              } catch (ClassNotFoundException e) {
83                throw new IllegalArgumentException("Could not find implementing class for " + attClass.getName());
84              }
85            }
86            return clazz;
87          }
88        }
89      }
90    }

113     this.attributes = input.attributes;
114     this.attributeImpls = input.attributeImpls;
115     this.factory = input.factory;

134   /** Returns a new iterator that iterates the attribute classes
135    * in the same order they were added in.
136    * <p>Signature for Java 1.5: <code>public Iterator&lt;Class&lt;? extends Attribute&gt;&gt; getAttributeClassesIterator()</code>
137    */
138   public Iterator getAttributeClassesIterator() {
139     return Collections.unmodifiableSet(attributes.keySet()).iterator();
140   }
141   
142   /** Returns a new iterator that iterates all unique Attribute implementations.
143    * This iterator may contain less entries that {@link #getAttributeClassesIterator},
144    * if one instance implements more than one Attribute interface.
145    * <p>Signature for Java 1.5: <code>public Iterator&lt;AttributeImpl&gt; getAttributeImplsIterator()</code>
146    */
147   public Iterator getAttributeImplsIterator() {
148     if (hasAttributes()) {
149       if (currentState == null) {
150         computeCurrentState();
151       }
152       final State initState = currentState;
153       return new Iterator() {
154         private State state = initState;
155       
156         public void remove() {
157           throw new UnsupportedOperationException();
158         }
159         
160         public Object next() {
161           if (state == null)
162             throw new NoSuchElementException();
163           final AttributeImpl att = state.attribute;
164           state = state.next;
165           return att;
166         }
167         
168         public boolean hasNext() {
169           return state != null;
170         }
171       };
172     } else {
173       return Collections.EMPTY_SET.iterator();
174     }
175   }

177   /** a cache that stores all interfaces for known implementation classes for performance (slow reflection) */
178   private static final WeakHashMap/*<Class<? extends AttributeImpl>,LinkedList<WeakReference<Class<? extends Attribute>>>>*/
179     knownImplClasses = new WeakHashMap();
180   
181   /** Adds a custom AttributeImpl instance with one or more Attribute interfaces. */
182   public void addAttributeImpl(final AttributeImpl att) {
183     final Class clazz = att.getClass();
184     if (attributeImpls.containsKey(clazz)) return;
185     LinkedList foundInterfaces;
186     synchronized(knownImplClasses) {
187       foundInterfaces = (LinkedList) knownImplClasses.get(clazz);
188       if (foundInterfaces == null) {
189         // we have a strong reference to the class instance holding all interfaces in the list (parameter "att"),
190         // so all WeakReferences are never evicted by GC
191         knownImplClasses.put(clazz, foundInterfaces=new LinkedList());
192         // find all interfaces that this attribute instance implements
193         // and that extend the Attribute interface
194         Class actClazz = clazz;
195         do {
196           Class[] interfaces = actClazz.getInterfaces();
197           for (int i = 0; i < interfaces.length; i++) {
198             final Class curInterface = interfaces[i];
199             if (curInterface != Attribute.class && Attribute.class.isAssignableFrom(curInterface)) {
200               foundInterfaces.add(new WeakReference(curInterface));
201             }
202           }
203           actClazz = actClazz.getSuperclass();
204         } while (actClazz != null);
205       }
206     }
207     
208     // add all interfaces of this AttributeImpl to the maps
209     for (Iterator it = foundInterfaces.iterator(); it.hasNext(); ) {
210       final WeakReference curInterfaceRef = (WeakReference) it.next();
211       final Class curInterface = (Class) curInterfaceRef.get();
212       assert (curInterface != null) :
213         "We have a strong reference on the class holding the interfaces, so they should never get evicted";
214       // Attribute is a superclass of this interface
215       if (!attributes.containsKey(curInterface)) {
216         // invalidate state to force recomputation in captureState()
217         this.currentState = null;
218         attributes.put(curInterface, att);
219         attributeImpls.put(clazz, att);
220       }
221     }
222   }

224   /**
225    * The caller must pass in a Class&lt;? extends Attribute&gt; value.
226    * This method first checks if an instance of that class is 
227    * already in this AttributeSource and returns it. Otherwise a
228    * new instance is created, added to this AttributeSource and returned. 
229    * <p>Signature for Java 1.5: <code>public &lt;T extends Attribute&gt; T addAttribute(Class&lt;T&gt;)</code>
230    */
231   public Attribute addAttribute(Class attClass) {
232     final Attribute att = (Attribute) attributes.get(attClass);
233     if (att == null) {
234       if (!(attClass.isInterface() && Attribute.class.isAssignableFrom(attClass))) {
235         throw new IllegalArgumentException(
236           "addAttribute() only accepts an interface that extends Attribute, but " +
237           attClass.getName() + " does not fulfil this contract."
238         );
239       }
240       final AttributeImpl attImpl = this.factory.createAttributeInstance(attClass);
241       addAttributeImpl(attImpl);
242       return attImpl;
243     } else {
244       return att;
245     }
246   }

248   /** Returns true, iff this AttributeSource has any attributes */
249   public boolean hasAttributes() {
250     return !this.attributes.isEmpty();
251   }

253   /**
254    * The caller must pass in a Class&lt;? extends Attribute&gt; value. 
255    * Returns true, iff this AttributeSource contains the passed-in Attribute.
256    * <p>Signature for Java 1.5: <code>public boolean hasAttribute(Class&lt;? extends Attribute&gt;)</code>
257    */
258   public boolean hasAttribute(Class attClass) {
259     return this.attributes.containsKey(attClass);
260   }
261 
262   /**
263    * The caller must pass in a Class&lt;? extends Attribute&gt; value. 
264    * Returns the instance of the passed in Attribute contained in this AttributeSource
265    * <p>Signature for Java 1.5: <code>public &lt;T extends Attribute&gt; T getAttribute(Class&lt;T&gt;)</code>
266    * 
267    * @throws IllegalArgumentException if this AttributeSource does not contain the
268    *         Attribute. It is recommended to always use {@link #addAttribute} even in consumers
269    *         of TokenStreams, because you cannot know if a specific TokenStream really uses
270    *         a specific Attribute. {@link #addAttribute} will automatically make the attribute
271    *         available. If you want to only use the attribute, if it is available (to optimize
272    *         consuming), use {@link #hasAttribute}.
273    */
274   public Attribute getAttribute(Class attClass) {
275     final Attribute att = (Attribute) this.attributes.get(attClass);
276     if (att == null) {
277       throw new IllegalArgumentException("This AttributeSource does not have the attribute '" + attClass.getName() + "'.");
278     }
279 
280     return att;
281   }

283   /**
284    * This class holds the state of an AttributeSource.
285    * @see #captureState
286    * @see #restoreState
287    */
288   public static final class State implements Cloneable {
289     private AttributeImpl attribute;
290     private State next;
291     
292     public Object clone() {
293       State clone = new State();
294       clone.attribute = (AttributeImpl) attribute.clone();
295       
296       if (next != null) {
297         clone.next = (State) next.clone();
298       }
299       
300       return clone;
301     }
302   }
303   
304   private State currentState = null;
305   
306   private void computeCurrentState() {
307     currentState = new State();
308     State c = currentState;
309     Iterator it = attributeImpls.values().iterator();
310     c.attribute = (AttributeImpl) it.next();
311     while (it.hasNext()) {
312       c.next = new State();
313       c = c.next;
314       c.attribute = (AttributeImpl) it.next();
315     }        
316   }

318   /**
319    * Resets all Attributes in this AttributeSource by calling
320    * {@link AttributeImpl#clear()} on each Attribute implementation.
321    */
322   public void clearAttributes() {
323     if (hasAttributes()) {
324       if (currentState == null) {
325         computeCurrentState();
326       }
327       for (State state = currentState; state != null; state = state.next) {
328         state.attribute.clear();
329       }
330     }
331   }
332   
333   /**
334    * Captures the state of all Attributes. The return value can be passed to
335    * {@link #restoreState} to restore the state of this or another AttributeSource.
336    */
337   public State captureState() {
338     if (!hasAttributes()) {
339       return null;
340     }
341       
342     if (currentState == null) {
343       computeCurrentState();
344     }
345     return (State) this.currentState.clone();
346   }
347   
348   /**
349    * Restores this state by copying the values of all attribute implementations
350    * that this state contains into the attributes implementations of the targetStream.
351    * The targetStream must contain a corresponding instance for each argument
352    * contained in this state (e.g. it is not possible to restore the state of
353    * an AttributeSource containing a TermAttribute into a AttributeSource using
354    * a Token instance as implementation).
355    * <p>
356    * Note that this method does not affect attributes of the targetStream
357    * that are not contained in this state. In other words, if for example
358    * the targetStream contains an OffsetAttribute, but this state doesn't, then
359    * the value of the OffsetAttribute remains unchanged. It might be desirable to
360    * reset its value to the default, in which case the caller should first
361    * call {@link TokenStream#clearAttributes()} on the targetStream.   
362    */
363   public void restoreState(State state) {
364     if (state == null)  return;
365     
366     do {
367       AttributeImpl targetImpl = (AttributeImpl) attributeImpls.get(state.attribute.getClass());
368       if (targetImpl == null)
369         throw new IllegalArgumentException("State contains an AttributeImpl that is not in this AttributeSource");
370       state.attribute.copyTo(targetImpl);
371       state = state.next;
372     } while (state != null);
373   }

375   public int hashCode() {
376     int code = 0;
377     if (hasAttributes()) {
378       if (currentState == null) {
379         computeCurrentState();
380       }
381       for (State state = currentState; state != null; state = state.next) {
382         code = code * 31 + state.attribute.hashCode();
383       }
384     }
385     
386     return code;
387   }

389   public boolean equals(Object obj) {
390     if (obj == this) {
391       return true;
392     }
393 
394     if (obj instanceof AttributeSource) {
395       AttributeSource other = (AttributeSource) obj;  
396     
397       if (hasAttributes()) {
398         if (!other.hasAttributes()) {
399           return false;
400         }
401         
402         if (this.attributeImpls.size() != other.attributeImpls.size()) {
403           return false;
404         }
405   
406         // it is only equal if all attribute impls are the same in the same order
407         if (this.currentState == null) {
408           this.computeCurrentState();
409         }
410         State thisState = this.currentState;
411         if (other.currentState == null) {
412           other.computeCurrentState();
413         }
414         State otherState = other.currentState;
415         while (thisState != null && otherState != null) {
416           if (otherState.attribute.getClass() != thisState.attribute.getClass() || !otherState.attribute.equals(thisState.attribute)) {
417             return false;
418           }
419           thisState = thisState.next;
420           otherState = otherState.next;
421         }
422         return true;
423       } else {
424         return !other.hasAttributes();
425       }
426     } else
427       return false;
428   }

430   public String toString() {
431     StringBuffer sb = new StringBuffer();
432     sb.append('(');
433     
434     if (hasAttributes()) {
435       if (currentState == null) {
436         computeCurrentState();
437       }
438       for (State state = currentState; state != null; state = state.next) {
439         if (state != currentState) sb.append(',');
440         sb.append(state.attribute.toString());
441       }
442     }
443     sb.append(')');
444     return sb.toString();
445   }

447   /**
448    * Performs a clone of all {@link AttributeImpl} instances returned in a new
449    * AttributeSource instance. This method can be used to e.g. create another TokenStream
450    * with exactly the same attributes (using {@link #AttributeSource(AttributeSource)})
451    */
452   public AttributeSource cloneAttributes() {
453     AttributeSource clone = new AttributeSource(this.factory);
454     
455     // first clone the impls
456     if (hasAttributes()) {
457       if (currentState == null) {
458         computeCurrentState();
459       }
460       for (State state = currentState; state != null; state = state.next) {
461         clone.attributeImpls.put(state.attribute.getClass(), state.attribute.clone());
462       }
463     }
464     
465     // now the interfaces
466     Iterator/*<Entry<Class<Attribute>, AttributeImpl>>*/ attIt = this.attributes.entrySet().iterator(); 
467     while (attIt.hasNext()) {
468       Entry/*<Class<Attribute>, AttributeImpl>*/ entry = (Entry/*<Class<Attribute>, AttributeImpl>*/) attIt.next();
469       clone.attributes.put(entry.getKey(), clone.attributeImpls.get(entry.getValue().getClass()));
470     }
471     
472     return clone;
473   }
