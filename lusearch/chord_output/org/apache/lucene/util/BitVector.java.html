20  import java.io.IOException;
21  
22  import org.apache.lucene.store.Directory;
23  import org.apache.lucene.store.IndexInput;
24  import org.apache.lucene.store.IndexOutput;

26  /** Optimized implementation of a vector of bits.  This is more-or-less like
27    java.util.BitSet, but also includes the following:
28    <ul>
29    <li>a count() method, which efficiently computes the number of one bits;</li>
30    <li>optimized read from and write to disk;</li>
31    <li>inlinable get() method;</li>
32    <li>store and load, as bit set or d-gaps, depending on sparseness;</li> 
33    </ul>
34  
35  
36    @version $Id: BitVector.java 765649 2009-04-16 14:29:26Z mikemccand $
37    */
38  public final class BitVector implements Cloneable {
39  
40    private byte[] bits;
41    private int size;
42    private int count = -1;
43  
44    /** Constructs a vector capable of holding <code>n</code> bits. */
45    public BitVector(int n) {
46      size = n;
47      bits = new byte[(size >> 3) + 1];
48    }
49    
50    BitVector(byte[] bits, int size) {
51      this.bits = bits;
52      this.size = size;
53    }
54    
55    public Object clone() {
56      byte[] copyBits = new byte[bits.length];
57      System.arraycopy(bits, 0, copyBits, 0, bits.length);
58      return new BitVector(copyBits, size);
59    }
60    
61    /** Sets the value of <code>bit</code> to one. */
62    public final void set(int bit) {
63      if (bit >= size) {
64        throw new ArrayIndexOutOfBoundsException(bit);
65      }
66      bits[bit >> 3] |= 1 << (bit & 7);
67      count = -1;
68    }
69  
70    /** Sets the value of <code>bit</code> to true, and
71     *  returns true if bit was already set */
72    public final boolean getAndSet(int bit) {
73      if (bit >= size) {
74        throw new ArrayIndexOutOfBoundsException(bit);
75      }
76      final int pos = bit >> 3;
77      final int v = bits[pos];
78      final int flag = 1 << (bit & 7);
79      if ((flag & v) != 0)
80        return true;
81      else {
82        bits[pos] = (byte) (v | flag);
83        if (count != -1)
84          count++;
85        return false;
86      }
87    }
88  
89    /** Sets the value of <code>bit</code> to zero. */
90    public final void clear(int bit) {
91      if (bit >= size) {
92        throw new ArrayIndexOutOfBoundsException(bit);
93      }
94      bits[bit >> 3] &= ~(1 << (bit & 7));
95      count = -1;
96    }
97  
98    /** Returns <code>true</code> if <code>bit</code> is one and
99      <code>false</code> if it is zero. */
100   public final boolean get(int bit) {
101     assert bit >= 0 && bit < size: "bit " + bit + " is out of bounds 0.." + (size-1);
102     return (bits[bit >> 3] & (1 << (bit & 7))) != 0;
103   }
104 
105   /** Returns the number of bits in this vector.  This is also one greater than
106     the number of the largest valid bit number. */
107   public final int size() {
108     return size;
109   }
110 
111   /** Returns the total number of one bits in this vector.  This is efficiently
112     computed and cached, so that, if the vector is not changed, no
113     recomputation is done for repeated calls. */
114   public final int count() {
115     // if the vector has been modified
116     if (count == -1) {
117       int c = 0;
118       int end = bits.length;
119       for (int i = 0; i < end; i++)
120         c += BYTE_COUNTS[bits[i] & 0xFF];     // sum bits per byte
121       count = c;
122     }
123     return count;
124   }
125 
126   private static final byte[] BYTE_COUNTS = {     // table of bits/byte
127     0, 1, 1, 2, 1, 2, 2, 3, 1, 2, 2, 3, 2, 3, 3, 4,
128     1, 2, 2, 3, 2, 3, 3, 4, 2, 3, 3, 4, 3, 4, 4, 5,
129     1, 2, 2, 3, 2, 3, 3, 4, 2, 3, 3, 4, 3, 4, 4, 5,
130     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
131     1, 2, 2, 3, 2, 3, 3, 4, 2, 3, 3, 4, 3, 4, 4, 5,
132     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
133     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
134     3, 4, 4, 5, 4, 5, 5, 6, 4, 5, 5, 6, 5, 6, 6, 7,
135     1, 2, 2, 3, 2, 3, 3, 4, 2, 3, 3, 4, 3, 4, 4, 5,
136     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
137     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
138     3, 4, 4, 5, 4, 5, 5, 6, 4, 5, 5, 6, 5, 6, 6, 7,
139     2, 3, 3, 4, 3, 4, 4, 5, 3, 4, 4, 5, 4, 5, 5, 6,
140     3, 4, 4, 5, 4, 5, 5, 6, 4, 5, 5, 6, 5, 6, 6, 7,
141     3, 4, 4, 5, 4, 5, 5, 6, 4, 5, 5, 6, 5, 6, 6, 7,
142     4, 5, 5, 6, 5, 6, 6, 7, 5, 6, 6, 7, 6, 7, 7, 8
143   };

146   /** Writes this vector to the file <code>name</code> in Directory
147     <code>d</code>, in a format that can be read by the constructor {@link
148     #BitVector(Directory, String)}.  */
149   public final void write(Directory d, String name) throws IOException {
150     IndexOutput output = d.createOutput(name);
151     try {
152       if (isSparse()) { 
153         writeDgaps(output); // sparse bit-set more efficiently saved as d-gaps.
154       } else {
155         writeBits(output);
156       }
157     } finally {
158       output.close();
159     }
160   }
161      
162   /** Write as a bit set */
163   private void writeBits(IndexOutput output) throws IOException {
164     output.writeInt(size());        // write size
165     output.writeInt(count());       // write count
166     output.writeBytes(bits, bits.length);
167   }
168   
169   /** Write as a d-gaps list */
170   private void writeDgaps(IndexOutput output) throws IOException {
171     output.writeInt(-1);            // mark using d-gaps                         
172     output.writeInt(size());        // write size
173     output.writeInt(count());       // write count
174     int last=0;
175     int n = count();
176     int m = bits.length;
177     for (int i=0; i<m && n>0; i++) {
178       if (bits[i]!=0) {
179         output.writeVInt(i-last);
180         output.writeByte(bits[i]);
181         last = i;
182         n -= BYTE_COUNTS[bits[i] & 0xFF];
183       }
184     }
185   }
186 
187   /** Indicates if the bit vector is sparse and should be saved as a d-gaps list, or dense, and should be saved as a bit set. */
188   private boolean isSparse() {
189     // note: order of comparisons below set to favor smaller values (no binary range search.)
190     // note: adding 4 because we start with ((int) -1) to indicate d-gaps format.
191     // note: we write the d-gap for the byte number, and the byte (bits[i]) itself, therefore
192     //       multiplying count by (8+8) or (8+16) or (8+24) etc.:
193     //       - first 8 for writing bits[i] (1 byte vs. 1 bit), and 
194     //       - second part for writing the byte-number d-gap as vint. 
195     // note: factor is for read/write of byte-arrays being faster than vints.  
196     int factor = 10;  
197     if (bits.length < (1<< 7)) return factor * (4 + (8+ 8)*count()) < size();
198     if (bits.length < (1<<14)) return factor * (4 + (8+16)*count()) < size();
199     if (bits.length < (1<<21)) return factor * (4 + (8+24)*count()) < size();
200     if (bits.length < (1<<28)) return factor * (4 + (8+32)*count()) < size();
201     return                            factor * (4 + (8+40)*count()) < size();
202   }
203 
204   /** Constructs a bit vector from the file <code>name</code> in Directory
205     <code>d</code>, as written by the {@link #write} method.
206     */
207   public BitVector(Directory d, String name) throws IOException {
208     IndexInput input = d.openInput(name);
209     try {
210       size = input.readInt();       // read size
211       if (size == -1) {
212         readDgaps(input);
213       } else {
214         readBits(input);
215       }
216     } finally {
217       input.close();
218     }
219   }
220 
221   /** Read as a bit set */
222   private void readBits(IndexInput input) throws IOException {
223     count = input.readInt();        // read count
224     bits = new byte[(size >> 3) + 1];     // allocate bits
225     input.readBytes(bits, 0, bits.length);
226   }
227 
228   /** read as a d-gaps list */ 
229   private void readDgaps(IndexInput input) throws IOException {
230     size = input.readInt();       // (re)read size
231     count = input.readInt();        // read count
232     bits = new byte[(size >> 3) + 1];     // allocate bits
233     int last=0;
234     int n = count();
235     while (n>0) {
236       last += input.readVInt();
237       bits[last] = input.readByte();
238       n -= BYTE_COUNTS[bits[last] & 0xFF];
239     }          
240   }

242   /**
243    * Retrieve a subset of this BitVector.
244    * 
245    * @param start
246    *            starting index, inclusive
247    * @param end
248    *            ending index, exclusive
249    * @return subset
250    */
251   public BitVector subset(int start, int end) {
252     if (start < 0 || end > size() || end < start)
253       throw new IndexOutOfBoundsException();
254     // Special case -- return empty vector is start == end
255     if (end == start) return new BitVector(0);
256     byte[] bits = new byte[((end - start - 1) >>> 3) + 1];
257     int s = start >>> 3;
258     for (int i = 0; i < bits.length; i++) {
259       int cur = 0xFF & this.bits[i + s];
260       int next = i + s + 1 >= this.bits.length ? 0 : 0xFF & this.bits[i + s + 1];
261       bits[i] = (byte) ((cur >>> (start & 7)) | ((next << (8 - (start & 7)))));
262     }
263     int bitsToClear = (bits.length * 8 - (end - start)) % 8;
264     bits[bits.length - 1] &= ~(0xFF << (8 - bitsToClear));
265     return new BitVector(bits, end - start);
266   }
