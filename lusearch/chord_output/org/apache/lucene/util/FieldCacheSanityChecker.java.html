31  /** 
32   * Provides methods for sanity checking that entries in the FieldCache 
33   * are not wasteful or inconsistent.
34   * </p>
35   * <p>
36   * Lucene 2.9 Introduced numerous enhancements into how the FieldCache 
37   * is used by the low levels of Lucene searching (for Sorting and 
38   * ValueSourceQueries) to improve both the speed for Sorting, as well 
39   * as reopening of IndexReaders.  But these changes have shifted the 
40   * usage of FieldCache from "top level" IndexReaders (frequently a 
41   * MultiReader or DirectoryReader) down to the leaf level SegmentReaders.  
42   * As a result, existing applications that directly access the FieldCache 
43   * may find RAM usage increase significantly when upgrading to 2.9 or 
44   * Later.  This class provides an API for these applications (or their 
45   * Unit tests) to check at run time if the FieldCache contains "insane" 
46   * usages of the FieldCache.
47   * </p>
48   * <p>
49   * <b>EXPERIMENTAL API:</b> This API is considered extremely advanced and 
50   * experimental.  It may be removed or altered w/o warning in future releases 
51   * of Lucene.
52   * </p>
53   * @see FieldCache
54   * @see FieldCacheSanityChecker.Insanity
55   * @see FieldCacheSanityChecker.InsanityType
56   */
57  public final class FieldCacheSanityChecker {
58  
59    private RamUsageEstimator ramCalc = null;
60    public FieldCacheSanityChecker() {
61      /* NOOP */
62    }
63    /**
64     * If set, will be used to estimate size for all CacheEntry objects 
65     * dealt with.
66     */
67    public void setRamUsageEstimator(RamUsageEstimator r) {
68      ramCalc = r;
69    }

73     * Quick and dirty convenience method
74     * @see #check
75     */
76    public static Insanity[] checkSanity(FieldCache cache) {
77      return checkSanity(cache.getCacheEntries());
78    }

80    /** 
81     * Quick and dirty convenience method that instantiates an instance with 
82     * "good defaults" and uses it to test the CacheEntry[]
83     * @see #check
84     */
85    public static Insanity[] checkSanity(CacheEntry[] cacheEntries) {
86      FieldCacheSanityChecker sanityChecker = new FieldCacheSanityChecker();
87      // doesn't check for interned
88      sanityChecker.setRamUsageEstimator(new RamUsageEstimator(false));
89      return sanityChecker.check(cacheEntries);
90    }

93    /**
94     * Tests a CacheEntry[] for indication of "insane" cache usage.
95     * <p>
96     * <B>NOTE:</b>FieldCache CreationPlaceholder objects are ignored.
97     * (:TODO: is this a bad idea? are we masking a real problem?)
98     * </p>
99     */
100   public Insanity[] check(CacheEntry[] cacheEntries) {
101     if (null == cacheEntries || 0 == cacheEntries.length) 
102       return new Insanity[0];
103 
104     if (null != ramCalc) {
105       for (int i = 0; i < cacheEntries.length; i++) {
106         cacheEntries[i].estimateSize(ramCalc);
107       }
108     }
109 
110     // the indirect mapping lets MapOfSet dedup identical valIds for us
111     //
112     // maps the (valId) identityhashCode of cache values to 
113     // sets of CacheEntry instances
114     final MapOfSets valIdToItems = new MapOfSets(new HashMap(17));
115     // maps ReaderField keys to Sets of ValueIds
116     final MapOfSets readerFieldToValIds = new MapOfSets(new HashMap(17));
117     //
118 
119     // any keys that we know result in more then one valId
120     final Set valMismatchKeys = new HashSet();
121 
122     // iterate over all the cacheEntries to get the mappings we'll need
123     for (int i = 0; i < cacheEntries.length; i++) {
124       final CacheEntry item = cacheEntries[i];
125       final Object val = item.getValue();
126 
127       if (val instanceof FieldCache.CreationPlaceholder)
128         continue;
129 
130       final ReaderField rf = new ReaderField(item.getReaderKey(), 
131                                             item.getFieldName());
132 
133       final Integer valId = new Integer(System.identityHashCode(val));
134 
135       // indirect mapping, so the MapOfSet will dedup identical valIds for us
136       valIdToItems.put(valId, item);
137       if (1 < readerFieldToValIds.put(rf, valId)) {
138         valMismatchKeys.add(rf);
139       }
140     }
141 
142     final List insanity = new ArrayList(valMismatchKeys.size() * 3);
143 
144     insanity.addAll(checkValueMismatch(valIdToItems, 
145                                        readerFieldToValIds, 
146                                        valMismatchKeys));
147     insanity.addAll(checkSubreaders(valIdToItems, 
148                                     readerFieldToValIds));
149                     
150     return (Insanity[]) insanity.toArray(new Insanity[insanity.size()]);
151   }
152 
153   /** 
154    * Internal helper method used by check that iterates over 
155    * valMismatchKeys and generates a Collection of Insanity 
156    * instances accordingly.  The MapOfSets are used to populate 
157    * the Insanity objects. 
158    * @see InsanityType#VALUEMISMATCH
159    */
160   private Collection checkValueMismatch(MapOfSets valIdToItems,
161                                         MapOfSets readerFieldToValIds,
162                                         Set valMismatchKeys) {
163 
164     final List insanity = new ArrayList(valMismatchKeys.size() * 3);
165 
166     if (! valMismatchKeys.isEmpty() ) { 
167       // we have multiple values for some ReaderFields
168 
169       final Map rfMap = readerFieldToValIds.getMap();
170       final Map valMap = valIdToItems.getMap();
171       final Iterator mismatchIter = valMismatchKeys.iterator();
172       while (mismatchIter.hasNext()) {
173         final ReaderField rf = (ReaderField)mismatchIter.next();
174         final List badEntries = new ArrayList(valMismatchKeys.size() * 2);
175         final Iterator valIter = ((Set)rfMap.get(rf)).iterator();
176         while (valIter.hasNext()) {
177           Iterator entriesIter = ((Set)valMap.get(valIter.next())).iterator();
178           while (entriesIter.hasNext()) {
179             badEntries.add(entriesIter.next());
180           }
181         }
182 
183         CacheEntry[] badness = new CacheEntry[badEntries.size()];
184         badness = (CacheEntry[]) badEntries.toArray(badness);
185 
186         insanity.add(new Insanity(InsanityType.VALUEMISMATCH,
187                                   "Multiple distinct value objects for " + 
188                                   rf.toString(), badness));
189       }
190     }
191     return insanity;
192   }

194   /** 
195    * Internal helper method used by check that iterates over 
196    * the keys of readerFieldToValIds and generates a Collection 
197    * of Insanity instances whenever two (or more) ReaderField instances are 
198    * found that have an ancestry relationships.  
199    *
200    * @see InsanityType#SUBREADER
201    */
202   private Collection checkSubreaders(MapOfSets valIdToItems,
203                                      MapOfSets readerFieldToValIds) {
204 
205     final List insanity = new ArrayList(23);
206 
207     Map badChildren = new HashMap(17);
208     MapOfSets badKids = new MapOfSets(badChildren); // wrapper
209 
210     Map viToItemSets = valIdToItems.getMap();
211     Map rfToValIdSets = readerFieldToValIds.getMap();
212 
213     Set seen = new HashSet(17);
214 
215     Set readerFields = rfToValIdSets.keySet();
216     Iterator rfIter = readerFields.iterator();
217     while (rfIter.hasNext()) {
218       ReaderField rf = (ReaderField) rfIter.next();
219       
220       if (seen.contains(rf)) continue;
221 
222       List kids = getAllDecendentReaderKeys(rf.readerKey);
223       for (int i = 0; i < kids.size(); i++) {
224         ReaderField kid = new ReaderField(kids.get(i), rf.fieldName);
225         
226         if (badChildren.containsKey(kid)) {
227           // we've already process this kid as RF and found other problems
228           // track those problems as our own
229           badKids.put(rf, kid);
230           badKids.putAll(rf, (Collection)badChildren.get(kid));
231           badChildren.remove(kid);
232           
233         } else if (rfToValIdSets.containsKey(kid)) {
234           // we have cache entries for the kid
235           badKids.put(rf, kid);
236         }
237         seen.add(kid);
238       }
239       seen.add(rf);
240     }
241 
242     // every mapping in badKids represents an Insanity
243     Iterator parentsIter = badChildren.keySet().iterator();
244     while (parentsIter.hasNext()) {
245       ReaderField parent = (ReaderField) parentsIter.next();
246       Set kids = (Set) badChildren.get(parent);
247 
248       List badEntries = new ArrayList(kids.size() * 2);
249 
250       // put parent entr(ies) in first
251       {
252         Iterator valIter =((Set)rfToValIdSets.get(parent)).iterator();
253         while (valIter.hasNext()) {
254           badEntries.addAll((Set)viToItemSets.get(valIter.next()));
255         }
256       }
257 
258       // now the entries for the descendants
259       Iterator kidsIter = kids.iterator();
260       while (kidsIter.hasNext()) {
261         ReaderField kid = (ReaderField) kidsIter.next();
262         Iterator valIter =((Set)rfToValIdSets.get(kid)).iterator();
263         while (valIter.hasNext()) {
264           badEntries.addAll((Set)viToItemSets.get(valIter.next()));
265         }
266       }
267 
268       CacheEntry[] badness = new CacheEntry[badEntries.size()];
269       badness = (CacheEntry[]) badEntries.toArray(badness);
270 
271       insanity.add(new Insanity(InsanityType.SUBREADER,
272                                 "Found caches for decendents of " + 
273                                 parent.toString(),
274                                 badness));
275     }
276 
277     return insanity;
278 
279   }

281   /**
282    * Checks if the seed is an IndexReader, and if so will walk
283    * the hierarchy of subReaders building up a list of the objects 
284    * returned by obj.getFieldCacheKey()
285    */
286   private List getAllDecendentReaderKeys(Object seed) {
287     List all = new ArrayList(17); // will grow as we iter
288     all.add(seed);
289     for (int i = 0; i < all.size(); i++) {
290       Object obj = all.get(i);
291       if (obj instanceof IndexReader) {
292         IndexReader[] subs = ((IndexReader)obj).getSequentialSubReaders();
293         for (int j = 0; (null != subs) && (j < subs.length); j++) {
294           all.add(subs[j].getFieldCacheKey());
295         }
296       }
297       
298     }
299     // need to skip the first, because it was the seed
300     return all.subList(1, all.size());
301   }

303   /**
304    * Simple pair object for using "readerKey + fieldName" a Map key
305    */
306   private final static class ReaderField {
307     public final Object readerKey;
308     public final String fieldName;
309     public ReaderField(Object readerKey, String fieldName) {
310       this.readerKey = readerKey;
311       this.fieldName = fieldName;
312     }
313     public int hashCode() {
314       return System.identityHashCode(readerKey) * fieldName.hashCode();
315     }
316     public boolean equals(Object that) {
317       if (! (that instanceof ReaderField)) return false;
318 
319       ReaderField other = (ReaderField) that;
320       return (this.readerKey == other.readerKey &&
321               this.fieldName.equals(other.fieldName));
322     }
323     public String toString() {
324       return readerKey.toString() + "+" + fieldName;
325     }
326   }

328   /**
329    * Simple container for a collection of related CacheEntry objects that 
330    * in conjunction with each other represent some "insane" usage of the 
331    * FieldCache.
332    */
333   public final static class Insanity {
334     private final InsanityType type;
335     private final String msg;
336     private final CacheEntry[] entries;
337     public Insanity(InsanityType type, String msg, CacheEntry[] entries) {
338       if (null == type) {
339         throw new IllegalArgumentException
340           ("Insanity requires non-null InsanityType");
341       }
342       if (null == entries || 0 == entries.length) {
343         throw new IllegalArgumentException
344           ("Insanity requires non-null/non-empty CacheEntry[]");
345       }
346       this.type = type;
347       this.msg = msg;
348       this.entries = entries;
349       
350     }
351     /**
352      * Type of insane behavior this object represents
353      */
354     public InsanityType getType() { return type; }
355     /**
356      * Description of hte insane behavior
357      */
358     public String getMsg() { return msg; }
359     /**
360      * CacheEntry objects which suggest a problem
361      */
362     public CacheEntry[] getCacheEntries() { return entries; }
363     /**
364      * Multi-Line representation of this Insanity object, starting with 
365      * the Type and Msg, followed by each CacheEntry.toString() on it's 
366      * own line prefaced by a tab character
367      */
368     public String toString() {
369       StringBuffer buf = new StringBuffer();
370       buf.append(getType()).append(": ");
371 
372       String m = getMsg();
373       if (null != m) buf.append(m);
374 
375       buf.append('\n');
376 
377       CacheEntry[] ce = getCacheEntries();
378       for (int i = 0; i < ce.length; i++) {
379         buf.append('\t').append(ce[i].toString()).append('\n');
380       }
381 
382       return buf.toString();
383     }
384   }

386   /**
387    * An Enumeration of the different types of "insane" behavior that 
388    * may be detected in a FieldCache.
389    *
390    * @see InsanityType#SUBREADER
391    * @see InsanityType#VALUEMISMATCH
392    * @see InsanityType#EXPECTED
393    */
394   public final static class InsanityType {
395     private final String label;
396     private InsanityType(final String label) {
397       this.label = label;
398     }
399     public String toString() { return label; }
400 
401     /** 
402      * Indicates an overlap in cache usage on a given field 
403      * in sub/super readers.
404      */
405     public final static InsanityType SUBREADER 
406       = new InsanityType("SUBREADER");
407 
408     /** 
409      * <p>
410      * Indicates entries have the same reader+fieldname but 
411      * different cached values.  This can happen if different datatypes, 
412      * or parsers are used -- and while it's not necessarily a bug 
413      * it's typically an indication of a possible problem.
414      * </p>
415      * <p>
416      * <bPNOTE:</b> Only the reader, fieldname, and cached value are actually 
417      * tested -- if two cache entries have different parsers or datatypes but 
418      * the cached values are the same Object (== not just equal()) this method 
419      * does not consider that a red flag.  This allows for subtle variations 
420      * in the way a Parser is specified (null vs DEFAULT_LONG_PARSER, etc...)
421      * </p>
422      */
423     public final static InsanityType VALUEMISMATCH 
424       = new InsanityType("VALUEMISMATCH");
425 
426     /** 
427      * Indicates an expected bit of "insanity".  This may be useful for 
428      * clients that wish to preserve/log information about insane usage 
429      * but indicate that it was expected. 
430      */
431     public final static InsanityType EXPECTED
432       = new InsanityType("EXPECTED");
433   }
