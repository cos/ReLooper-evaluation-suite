20  import org.apache.lucene.LucenePackage;
21  
22  /**
23   * Some useful constants.
24   *
25   *
26   * @version $Id: Constants.java 909831 2010-02-13 14:38:36Z uschindler $
27   **/
28  
29  public final class Constants {
30    private Constants() {}              // can't construct
31  
32    /** The value of <tt>System.getProperty("java.version")<tt>. **/
33    public static final String JAVA_VERSION = System.getProperty("java.version");
34    /** True iff this is Java version 1.1. */
35    public static final boolean JAVA_1_1 = JAVA_VERSION.startsWith("1.1.");
36    /** True iff this is Java version 1.2. */
37    public static final boolean JAVA_1_2 = JAVA_VERSION.startsWith("1.2.");
38    /** True iff this is Java version 1.3. */
39    public static final boolean JAVA_1_3 = JAVA_VERSION.startsWith("1.3.");
40   
41    /** The value of <tt>System.getProperty("os.name")<tt>. **/
42    public static final String OS_NAME = System.getProperty("os.name");
43    /** True iff running on Linux. */
44    public static final boolean LINUX = OS_NAME.startsWith("Linux");
45    /** True iff running on Windows. */
46    public static final boolean WINDOWS = OS_NAME.startsWith("Windows");
47    /** True iff running on SunOS. */
48    public static final boolean SUN_OS = OS_NAME.startsWith("SunOS");
49  
50    public static final String OS_ARCH = System.getProperty("os.arch");
51    public static final String OS_VERSION = System.getProperty("os.version");
52    public static final String JAVA_VENDOR = System.getProperty("java.vendor");
53  
54    // NOTE: this logic may not be correct; if you know of a
55    // more reliable approach please raise it on java-dev!
56    public static final boolean JRE_IS_64BIT;
57    static {
58      String x = System.getProperty("sun.arch.data.model");
59      if (x != null) {
60        JRE_IS_64BIT = x.indexOf("64") != -1;
61      } else {
62        if (OS_ARCH != null && OS_ARCH.indexOf("64") != -1) {
63          JRE_IS_64BIT = true;
64        } else {
65          JRE_IS_64BIT = false;
66        }
67      }
68    }
69  
70    // this method prevents inlining the final version constant in compiled classes,
71    // see: http://www.javaworld.com/community/node/3400
72    private static String ident(final String s) {
73      return s.toString();
74    }
75     
76    public static final String LUCENE_MAIN_VERSION = ident("2.9.2");
77  
78    public static final String LUCENE_VERSION;
79    static {
80      Package pkg = LucenePackage.get();
81      String v = (pkg == null) ? null : pkg.getImplementationVersion();
82      if (v == null) {
83        v = LUCENE_MAIN_VERSION + "-dev";
84      } else if (v.indexOf(LUCENE_MAIN_VERSION) == -1) {
85        v = v + " [" + LUCENE_MAIN_VERSION + "]";
86      }
87      LUCENE_VERSION = ident(v);
88    }
89  }
