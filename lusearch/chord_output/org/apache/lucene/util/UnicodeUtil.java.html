75    public static final class UTF8Result {
76      public byte[] result = new byte[10];
77      public int length;
78  
79      public void setLength(int newLength) {
80        if (result.length < newLength) {
81          byte[] newArray = new byte[(int) (1.5*newLength)];
82          System.arraycopy(result, 0, newArray, 0, length);
83          result = newArray;
84        }
85        length = newLength;
86      }
87    }

89    public static final class UTF16Result {
90      public char[] result = new char[10];
91      public int[] offsets = new int[10];
92      public int length;
93  
94      public void setLength(int newLength) {
95        if (result.length < newLength) {
96          char[] newArray = new char[(int) (1.5*newLength)];
97          System.arraycopy(result, 0, newArray, 0, length);
98          result = newArray;
99        }
100       length = newLength;
101     }
102 
103     public void copyText(UTF16Result other) {
104       setLength(other.length);
105       System.arraycopy(other.result, 0, result, 0, length);
106     }

109   /** Encode characters from a char[] source, starting at
110    *  offset and stopping when the character 0xffff is seen.
111    *  Returns the number of bytes written to bytesOut. */
112   public static void UTF16toUTF8(final char[] source, final int offset, UTF8Result result) {
113 
114     int upto = 0;
115     int i = offset;
116     byte[] out = result.result;
117 
118     while(true) {
119       
120       final int code = (int) source[i++];
121 
122       if (upto+4 > out.length) {
123         byte[] newOut = new byte[2*out.length];
124         assert newOut.length >= upto+4;
125         System.arraycopy(out, 0, newOut, 0, upto);
126         result.result = out = newOut;
127       }
128       if (code < 0x80)
129         out[upto++] = (byte) code;
130       else if (code < 0x800) {
131         out[upto++] = (byte) (0xC0 | (code >> 6));
132         out[upto++] = (byte)(0x80 | (code & 0x3F));
133       } else if (code < 0xD800 || code > 0xDFFF) {
134         if (code == 0xffff)
135           // END
136           break;
137         out[upto++] = (byte)(0xE0 | (code >> 12));
138         out[upto++] = (byte)(0x80 | ((code >> 6) & 0x3F));
139         out[upto++] = (byte)(0x80 | (code & 0x3F));
140       } else {
141         // surrogate pair
142         // confirm valid high surrogate
143         if (code < 0xDC00 && source[i] != 0xffff) {
144           int utf32 = (int) source[i];
145           // confirm valid low surrogate and write pair
146           if (utf32 >= 0xDC00 && utf32 <= 0xDFFF) { 
147             utf32 = ((code - 0xD7C0) << 10) + (utf32 & 0x3FF);
148             i++;
149             out[upto++] = (byte)(0xF0 | (utf32 >> 18));
150             out[upto++] = (byte)(0x80 | ((utf32 >> 12) & 0x3F));
151             out[upto++] = (byte)(0x80 | ((utf32 >> 6) & 0x3F));
152             out[upto++] = (byte)(0x80 | (utf32 & 0x3F));
153             continue;
154           }
155         }
156         // replace unpaired surrogate or out-of-order low surrogate
157         // with substitution character
158         out[upto++] = (byte) 0xEF;
159         out[upto++] = (byte) 0xBF;
160         out[upto++] = (byte) 0xBD;
161       }
162     }
163     //assert matches(source, offset, i-offset-1, out, upto);
164     result.length = upto;
165   }

167   /** Encode characters from a char[] source, starting at
168    *  offset for length chars.  Returns the number of bytes
169    *  written to bytesOut. */
170   public static void UTF16toUTF8(final char[] source, final int offset, final int length, UTF8Result result) {
171 
172     int upto = 0;
173     int i = offset;
174     final int end = offset + length;
175     byte[] out = result.result;
176 
177     while(i < end) {
178       
179       final int code = (int) source[i++];
180 
181       if (upto+4 > out.length) {
182         byte[] newOut = new byte[2*out.length];
183         assert newOut.length >= upto+4;
184         System.arraycopy(out, 0, newOut, 0, upto);
185         result.result = out = newOut;
186       }
187       if (code < 0x80)

191         out[upto++] = (byte)(0x80 | (code & 0x3F));
192       } else if (code < 0xD800 || code > 0xDFFF) {
193         out[upto++] = (byte)(0xE0 | (code >> 12));

201           // confirm valid low surrogate and write pair

223   /** Encode characters from this String, starting at offset
224    *  for length characters.  Returns the number of bytes
225    *  written to bytesOut. */
226   public static void UTF16toUTF8(final String s, final int offset, final int length, UTF8Result result) {
227     final int end = offset + length;
228 
229     byte[] out = result.result;
230 
231     int upto = 0;
232     for(int i=offset;i<end;i++) {
233       final int code = (int) s.charAt(i);
234 
235       if (upto+4 > out.length) {
236         byte[] newOut = new byte[2*out.length];
237         assert newOut.length >= upto+4;
238         System.arraycopy(out, 0, newOut, 0, upto);
239         result.result = out = newOut;
240       }
241       if (code < 0x80)
242         out[upto++] = (byte) code;
243       else if (code < 0x800) {
244         out[upto++] = (byte) (0xC0 | (code >> 6));
245         out[upto++] = (byte)(0x80 | (code & 0x3F));
246       } else if (code < 0xD800 || code > 0xDFFF) {
247         out[upto++] = (byte)(0xE0 | (code >> 12));
248         out[upto++] = (byte)(0x80 | ((code >> 6) & 0x3F));
249         out[upto++] = (byte)(0x80 | (code & 0x3F));
250       } else {
251         // surrogate pair
252         // confirm valid high surrogate
253         if (code < 0xDC00 && (i < end-1)) {
254           int utf32 = (int) s.charAt(i+1);
255           // confirm valid low surrogate and write pair
256           if (utf32 >= 0xDC00 && utf32 <= 0xDFFF) { 
257             utf32 = ((code - 0xD7C0) << 10) + (utf32 & 0x3FF);
258             i++;
259             out[upto++] = (byte)(0xF0 | (utf32 >> 18));
260             out[upto++] = (byte)(0x80 | ((utf32 >> 12) & 0x3F));
261             out[upto++] = (byte)(0x80 | ((utf32 >> 6) & 0x3F));
262             out[upto++] = (byte)(0x80 | (utf32 & 0x3F));
263             continue;
264           }
265         }
266         // replace unpaired surrogate or out-of-order low surrogate
267         // with substitution character
268         out[upto++] = (byte) 0xEF;
269         out[upto++] = (byte) 0xBF;
270         out[upto++] = (byte) 0xBD;
271       }
272     }
273     //assert matches(s, offset, length, out, upto);
274     result.length = upto;
275   }

277   /** Convert UTF8 bytes into UTF16 characters.  If offset
278    *  is non-zero, conversion starts at that starting point
279    *  in utf8, re-using the results from the previous call
280    *  up until offset. */
281   public static void UTF8toUTF16(final byte[] utf8, final int offset, final int length, final UTF16Result result) {
282 
283     final int end = offset + length;
284     char[] out = result.result;
285     if (result.offsets.length <= end) {
286       int[] newOffsets = new int[2*end];
287       System.arraycopy(result.offsets, 0, newOffsets, 0, result.offsets.length);
288       result.offsets  = newOffsets;
289     }
290     final int[] offsets = result.offsets;
291 
292     // If incremental decoding fell in the middle of a
293     // single unicode character, rollback to its start:
294     int upto = offset;
295     while(offsets[upto] == -1)
296       upto--;
297 
298     int outUpto = offsets[upto];
299 
300     // Pre-allocate for worst case 1-for-1
301     if (outUpto+length >= out.length) {
302       char[] newOut = new char[2*(outUpto+length)];
303       System.arraycopy(out, 0, newOut, 0, outUpto);
304       result.result = out = newOut;
305     }
306 
307     while (upto < end) {
308 
309       final int b = utf8[upto]&0xff;
310       final int ch;
311 
312       offsets[upto++] = outUpto;
313 
314       if (b < 0xc0) {
315         assert b < 0x80;
316         ch = b;
317       } else if (b < 0xe0) {
318         ch = ((b&0x1f)<<6) + (utf8[upto]&0x3f);
319         offsets[upto++] = -1;
320       } else if (b < 0xf0) {
321         ch = ((b&0xf)<<12) + ((utf8[upto]&0x3f)<<6) + (utf8[upto+1]&0x3f);
322         offsets[upto++] = -1;
323         offsets[upto++] = -1;
324       } else {
325         assert b < 0xf8;
326         ch = ((b&0x7)<<18) + ((utf8[upto]&0x3f)<<12) + ((utf8[upto+1]&0x3f)<<6) + (utf8[upto+2]&0x3f);
327         offsets[upto++] = -1;
328         offsets[upto++] = -1;
329         offsets[upto++] = -1;
330       }
331 
332       if (ch <= UNI_MAX_BMP) {
333         // target is a character <= 0xFFFF
334         out[outUpto++] = (char) ch;
335       } else {
336         // target is a character in range 0xFFFF - 0x10FFFF
337         final int chHalf = ch - HALF_BASE;
338         out[outUpto++] = (char) ((chHalf >> HALF_SHIFT) + UNI_SUR_HIGH_START);
339         out[outUpto++] = (char) ((chHalf & HALF_MASK) + UNI_SUR_LOW_START);
340       }
341     }
342 
343     offsets[upto] = outUpto;
344     result.length = outUpto;
345   }
