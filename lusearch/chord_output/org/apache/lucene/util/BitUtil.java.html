26    /** Returns the number of bits set in the long */
27    public static int pop(long x) {
28    /* Hacker's Delight 32 bit pop function:
29     * http://www.hackersdelight.org/HDcode/newCode/pop_arrayHS.cc
30     *
31    int pop(unsigned x) {
32       x = x - ((x >> 1) & 0x55555555);
33       x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
34       x = (x + (x >> 4)) & 0x0F0F0F0F;
35       x = x + (x >> 8);
36       x = x + (x >> 16);
37       return x & 0x0000003F;
38      }
39    ***/
40  
41      // 64 bit java version of the C function from above
42      x = x - ((x >>> 1) & 0x5555555555555555L);
43      x = (x & 0x3333333333333333L) + ((x >>>2 ) & 0x3333333333333333L);
44      x = (x + (x >>> 4)) & 0x0F0F0F0F0F0F0F0FL;
45      x = x + (x >>> 8);
46      x = x + (x >>> 16);
47      x = x + (x >>> 32);
48      return ((int)x) & 0x7F;
49    }

51    /*** Returns the number of set bits in an array of longs. */
52    public static long pop_array(long A[], int wordOffset, int numWords) {
53      /*
54      * Robert Harley and David Seal's bit counting algorithm, as documented
55      * in the revisions of Hacker's Delight
56      * http://www.hackersdelight.org/revisions.pdf
57      * http://www.hackersdelight.org/HDcode/newCode/pop_arrayHS.cc
58      *
59      * This function was adapted to Java, and extended to use 64 bit words.
60      * if only we had access to wider registers like SSE from java...
61      *
62      * This function can be transformed to compute the popcount of other functions
63      * on bitsets via something like this:
64      * sed 's/A\[\([^]]*\)\]/\(A[\1] \& B[\1]\)/g'
65      *
66      */
67      int n = wordOffset+numWords;
68      long tot=0, tot8=0;
69      long ones=0, twos=0, fours=0;
70  
71      int i;
72      for (i = wordOffset; i <= n - 8; i+=8) {
73        /***  C macro from Hacker's Delight
74         #define CSA(h,l, a,b,c) \
75         {unsigned u = a ^ b; unsigned v = c; \
76         h = (a & b) | (u & v); l = u ^ v;}
77         ***/
78  
79        long twosA,twosB,foursA,foursB,eights;
80  
81        // CSA(twosA, ones, ones, A[i], A[i+1])
82        {
83          long b=A[i], c=A[i+1];
84          long u=ones ^ b;
85          twosA=(ones & b)|( u & c);
86          ones=u^c;
87        }
88        // CSA(twosB, ones, ones, A[i+2], A[i+3])
89        {
90          long b=A[i+2], c=A[i+3];
91          long u=ones^b;
92          twosB =(ones&b)|(u&c);
93          ones=u^c;
94        }
95        //CSA(foursA, twos, twos, twosA, twosB)
96        {
97          long u=twos^twosA;
98          foursA=(twos&twosA)|(u&twosB);
99          twos=u^twosB;
100       }
101       //CSA(twosA, ones, ones, A[i+4], A[i+5])
102       {
103         long b=A[i+4], c=A[i+5];
104         long u=ones^b;
105         twosA=(ones&b)|(u&c);
106         ones=u^c;
107       }
108       // CSA(twosB, ones, ones, A[i+6], A[i+7])
109       {
110         long b=A[i+6], c=A[i+7];
111         long u=ones^b;
112         twosB=(ones&b)|(u&c);
113         ones=u^c;
114       }
115       //CSA(foursB, twos, twos, twosA, twosB)
116       {
117         long u=twos^twosA;
118         foursB=(twos&twosA)|(u&twosB);
119         twos=u^twosB;
120       }
121 
122       //CSA(eights, fours, fours, foursA, foursB)
123       {
124         long u=fours^foursA;
125         eights=(fours&foursA)|(u&foursB);
126         fours=u^foursB;
127       }
128       tot8 += pop(eights);
129     }
130 
131     // handle trailing words in a binary-search manner...
132     // derived from the loop above by setting specific elements to 0.
133     // the original method in Hackers Delight used a simple for loop:
134     //   for (i = i; i < n; i++)      // Add in the last elements
135     //  tot = tot + pop(A[i]);
136 
137     if (i<=n-4) {
138       long twosA, twosB, foursA, eights;
139       {
140         long b=A[i], c=A[i+1];
141         long u=ones ^ b;
142         twosA=(ones & b)|( u & c);
143         ones=u^c;
144       }
145       {
146         long b=A[i+2], c=A[i+3];
147         long u=ones^b;
148         twosB =(ones&b)|(u&c);
149         ones=u^c;
150       }
151       {
152         long u=twos^twosA;
153         foursA=(twos&twosA)|(u&twosB);
154         twos=u^twosB;
155       }
156       eights=fours&foursA;
157       fours=fours^foursA;
158 
159       tot8 += pop(eights);
160       i+=4;
161     }
162 
163     if (i<=n-2) {
164       long b=A[i], c=A[i+1];
165       long u=ones ^ b;
166       long twosA=(ones & b)|( u & c);
167       ones=u^c;
168 
169       long foursA=twos&twosA;
170       twos=twos^twosA;
171 
172       long eights=fours&foursA;
173       fours=fours^foursA;
174 
175       tot8 += pop(eights);
176       i+=2;
177     }
178 
179     if (i<n) {
180       tot += pop(A[i]);
181     }
182 
183     tot += (pop(fours)<<2)
184             + (pop(twos)<<1)
185             + pop(ones)
186             + (tot8<<3);
187 
188     return tot;
189   }

194   public static long pop_intersect(long A[], long B[], int wordOffset, int numWords) {
195     // generated from pop_array via sed 's/A\[\([^]]*\)\]/\(A[\1] \& B[\1]\)/g'
196     int n = wordOffset+numWords;
197     long tot=0, tot8=0;
198     long ones=0, twos=0, fours=0;
199 
200     int i;
201     for (i = wordOffset; i <= n - 8; i+=8) {
202       long twosA,twosB,foursA,foursB,eights;
203 
204       // CSA(twosA, ones, ones, (A[i] & B[i]), (A[i+1] & B[i+1]))
205       {
206         long b=(A[i] & B[i]), c=(A[i+1] & B[i+1]);
207         long u=ones ^ b;
208         twosA=(ones & b)|( u & c);
209         ones=u^c;
210       }
211       // CSA(twosB, ones, ones, (A[i+2] & B[i+2]), (A[i+3] & B[i+3]))
212       {
213         long b=(A[i+2] & B[i+2]), c=(A[i+3] & B[i+3]);
214         long u=ones^b;
215         twosB =(ones&b)|(u&c);
216         ones=u^c;
217       }
218       //CSA(foursA, twos, twos, twosA, twosB)
219       {
220         long u=twos^twosA;
221         foursA=(twos&twosA)|(u&twosB);
222         twos=u^twosB;
223       }
224       //CSA(twosA, ones, ones, (A[i+4] & B[i+4]), (A[i+5] & B[i+5]))
225       {
226         long b=(A[i+4] & B[i+4]), c=(A[i+5] & B[i+5]);
227         long u=ones^b;
228         twosA=(ones&b)|(u&c);
229         ones=u^c;
230       }
231       // CSA(twosB, ones, ones, (A[i+6] & B[i+6]), (A[i+7] & B[i+7]))
232       {
233         long b=(A[i+6] & B[i+6]), c=(A[i+7] & B[i+7]);
234         long u=ones^b;
235         twosB=(ones&b)|(u&c);
236         ones=u^c;
237       }
238       //CSA(foursB, twos, twos, twosA, twosB)
239       {
240         long u=twos^twosA;
241         foursB=(twos&twosA)|(u&twosB);
242         twos=u^twosB;
243       }
244 
245       //CSA(eights, fours, fours, foursA, foursB)
246       {
247         long u=fours^foursA;
248         eights=(fours&foursA)|(u&foursB);
249         fours=u^foursB;
250       }
251       tot8 += pop(eights);
252     }
253 
254 
255     if (i<=n-4) {
256       long twosA, twosB, foursA, eights;
257       {
258         long b=(A[i] & B[i]), c=(A[i+1] & B[i+1]);
259         long u=ones ^ b;
260         twosA=(ones & b)|( u & c);
261         ones=u^c;
262       }
263       {
264         long b=(A[i+2] & B[i+2]), c=(A[i+3] & B[i+3]);
265         long u=ones^b;
266         twosB =(ones&b)|(u&c);
267         ones=u^c;
268       }
269       {
270         long u=twos^twosA;
271         foursA=(twos&twosA)|(u&twosB);
272         twos=u^twosB;
273       }
274       eights=fours&foursA;
275       fours=fours^foursA;
276 
277       tot8 += pop(eights);
278       i+=4;
279     }
280 
281     if (i<=n-2) {
282       long b=(A[i] & B[i]), c=(A[i+1] & B[i+1]);
283       long u=ones ^ b;
284       long twosA=(ones & b)|( u & c);
285       ones=u^c;
286 
287       long foursA=twos&twosA;
288       twos=twos^twosA;
289 
290       long eights=fours&foursA;
291       fours=fours^foursA;
292 
293       tot8 += pop(eights);
294       i+=2;
295     }
296 
297     if (i<n) {
298       tot += pop((A[i] & B[i]));
299     }
300 
301     tot += (pop(fours)<<2)
302             + (pop(twos)<<1)
303             + pop(ones)
304             + (tot8<<3);
305 
306     return tot;
307   }

604         ones=u^c;
605       }
606       //CSA(foursB, twos, twos, twosA, twosB)
607       {
608         long u=twos^twosA;
609         foursB=(twos&twosA)|(u&twosB);
610         twos=u^twosB;
611       }
612 
613       //CSA(eights, fours, fours, foursA, foursB)
614       {
615         long u=fours^foursA;
616         eights=(fours&foursA)|(u&foursB);
617         fours=u^foursB;
618       }
619       tot8 += pop(eights);
620     }
621 
622 
623     if (i<=n-4) {
624       long twosA, twosB, foursA, eights;
625       {
626         long b=(A[i] ^ B[i]), c=(A[i+1] ^ B[i+1]);
627         long u=ones ^ b;
628         twosA=(ones & b)|( u & c);
629         ones=u^c;
630       }
631       {
632         long b=(A[i+2] ^ B[i+2]), c=(A[i+3] ^ B[i+3]);
633         long u=ones^b;
634         twosB =(ones&b)|(u&c);
635         ones=u^c;
636       }
637       {
638         long u=twos^twosA;
639         foursA=(twos&twosA)|(u&twosB);
640         twos=u^twosB;
641       }
642       eights=fours&foursA;
643       fours=fours^foursA;
644 
645       tot8 += pop(eights);
646       i+=4;
647     }
648 
649     if (i<=n-2) {
650       long b=(A[i] ^ B[i]), c=(A[i+1] ^ B[i+1]);
651       long u=ones ^ b;
652       long twosA=(ones & b)|( u & c);
653       ones=u^c;
654 
655       long foursA=twos&twosA;
656       twos=twos^twosA;
657 
658       long eights=fours&foursA;
659       fours=fours^foursA;
660 
661       tot8 += pop(eights);
662       i+=2;
663     }
664 
665     if (i<n) {
666       tot += pop((A[i] ^ B[i]));
667     }
668 
669     tot += (pop(fours)<<2)
670             + (pop(twos)<<1)
671             + pop(ones)
672             + (tot8<<3);
673 
674     return tot;
675   }

691   /** Returns number of trailing zeros in a 64 bit long value. */
692   public static int ntz(long val) {
693     // A full binary search to determine the low byte was slower than
694     // a linear search for nextSetBit().  This is most likely because
695     // the implementation of nextSetBit() shifts bits to the right, increasing
696     // the probability that the first non-zero byte is in the rhs.
697     //
698     // This implementation does a single binary search at the top level only
699     // so that all other bit shifting can be done on ints instead of longs to
700     // remain friendly to 32 bit architectures.  In addition, the case of a
701     // non-zero first byte is checked for first because it is the most common
702     // in dense bit arrays.
703 
704     int lower = (int)val;
705     int lowByte = lower & 0xff;
706     if (lowByte != 0) return ntzTable[lowByte];
707 
708     if (lower!=0) {
709       lowByte = (lower>>>8) & 0xff;
710       if (lowByte != 0) return ntzTable[lowByte] + 8;
711       lowByte = (lower>>>16) & 0xff;
712       if (lowByte != 0) return ntzTable[lowByte] + 16;
713       // no need to mask off low byte for the last byte in the 32 bit word
714       // no need to check for zero on the last byte either.
715       return ntzTable[lower>>>24] + 24;
716     } else {
717       // grab upper 32 bits
718       int upper=(int)(val>>32);
719       lowByte = upper & 0xff;
720       if (lowByte != 0) return ntzTable[lowByte] + 32;
721       lowByte = (upper>>>8) & 0xff;
722       if (lowByte != 0) return ntzTable[lowByte] + 40;
723       lowByte = (upper>>>16) & 0xff;
724       if (lowByte != 0) return ntzTable[lowByte] + 48;
725       // no need to mask off low byte for the last byte in the 32 bit word
726       // no need to check for zero on the last byte either.
727       return ntzTable[upper>>>24] + 56;
728     }
729   }

752   public static int ntz2(long x) {
753    int n = 0;
754    int y = (int)x;
755    if (y==0) {n+=32; y = (int)(x>>>32); }   // the only 64 bit shift necessary
756    if ((y & 0x0000FFFF) == 0) { n+=16; y>>>=16; }
757    if ((y & 0x000000FF) == 0) { n+=8; y>>>=8; }

762    * <br/> This is an alternate implementation of ntz()
763    */
764   public static int ntz3(long x) {
765    // another implementation taken from Hackers Delight, extended to 64 bits
766    // and converted to Java.
767    // Many 32 bit ntz algorithms are at http://www.hackersdelight.org/HDcode/ntz.cc
768    int n = 1;
769 
770    // do the first step as a long, all others as ints.
771    int y = (int)x;
772    if (y==0) {n+=32; y = (int)(x>>>32); }
773    if ((y & 0x0000FFFF) == 0) { n+=16; y>>>=16; }
774    if ((y & 0x000000FF) == 0) { n+=8; y>>>=8; }
775    if ((y & 0x0000000F) == 0) { n+=4; y>>>=4; }
776    if ((y & 0x00000003) == 0) { n+=2; y>>>=2; }
777    return n - (y & 1);
