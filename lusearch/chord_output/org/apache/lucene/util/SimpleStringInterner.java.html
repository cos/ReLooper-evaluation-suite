20  /**
21   * Simple lockless and memory barrier free String intern cache that is guaranteed
22   * to return the same String instance as String.intern() does.
23   */
24  public class SimpleStringInterner extends StringInterner {
25  
26    private static class Entry {
27      final private String str;
28      final private int hash;
29      private Entry next;
30      private Entry(String str, int hash, Entry next) {
31        this.str = str;
32        this.hash = hash;
33        this.next = next;
34      }
35    }
36  
37    private final Entry[] cache;
38    private final int maxChainLength;
39  
40    /**
41     * @param tableSize  Size of the hash table, should be a power of two.
42     * @param maxChainLength  Maximum length of each bucket, after which the oldest item inserted is dropped.
43     */
44    public SimpleStringInterner(int tableSize, int maxChainLength) {
45      cache = new Entry[Math.max(1,BitUtil.nextHighestPowerOfTwo(tableSize))];
46      this.maxChainLength = Math.max(2,maxChainLength);
47    }
48  
49    // @Override
50    public String intern(String s) {
51      int h = s.hashCode();
52      // In the future, it may be worth augmenting the string hash
53      // if the lower bits need better distribution.
54      int slot = h & (cache.length-1);
55  
56      Entry first = this.cache[slot];
57      Entry nextToLast = null;
58  
59      int chainLength = 0;
60  
61      for(Entry e=first; e!=null; e=e.next) {
62        if (e.hash == h && (e.str == s || e.str.compareTo(s)==0)) {
63        // if (e.str == s || (e.hash == h && e.str.compareTo(s)==0)) {
64          return e.str;
65        }
66  
67        chainLength++;
68        if (e.next != null) {
69          nextToLast = e;
70        }
71      }
72  
73      // insertion-order cache: add new entry at head
74      s = s.intern();
75      this.cache[slot] = new Entry(s, h, first);
76      if (chainLength >= maxChainLength) {
77        // prune last entry
78        nextToLast.next = null;
79      }
80      return s;
81    }
82  }
