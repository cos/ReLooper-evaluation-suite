20  import java.util.BitSet;
21  import java.io.IOException;
22  import org.apache.lucene.index.IndexReader;
23  import org.apache.lucene.util.DocIdBitSet;
24  
25  /** 
26   *  Abstract base class for restricting which documents may
27   *  be returned during searching.
28   *  <p>
29   *  <b>Note:</b> In Lucene 3.0 {@link #bits(IndexReader)} will be removed
30   *  and {@link #getDocIdSet(IndexReader)} will be defined as abstract.
31   *  All implementing classes must therefore implement {@link #getDocIdSet(IndexReader)}
32   *  in order to work with Lucene 3.0.
33   */
34  public abstract class Filter implements java.io.Serializable {
35    
36    /**
37     * @return A BitSet with true for documents which should be permitted in
38     * search results, and false for those that should not.
39     *
40     * <p><b>NOTE:</b> See {@link #getDocIdSet(IndexReader)} for
41     * handling of multi-segment indexes (which applies to
42     * this method as well).
43  
44     * @deprecated Use {@link #getDocIdSet(IndexReader)} instead.
45     */
46    public BitSet bits(IndexReader reader) throws IOException {
47      throw new UnsupportedOperationException();
48    }
49  
50    /**
51     * Creates a {@link DocIdSet} enumerating the documents that should be
52     * permitted in search results. <b>NOTE:</b> null can be
53     * returned if no documents are accepted by this Filter.
54     * <p>
55     * Note: This method will be called once per segment in
56     * the index during searching.  The returned {@link DocIdSet}
57     * must refer to document IDs for that segment, not for
58     * the top-level reader.
59     * 
60     * @param reader a {@link IndexReader} instance opened on the index currently
61     *         searched on. Note, it is likely that the provided reader does not
62     *         represent the whole underlying index i.e. if the index has more than
63     *         one segment the given reader only represents a single segment.
64     *          
65     * @return a DocIdSet that provides the documents which should be permitted or
66     *         prohibited in search results. <b>NOTE:</b> null can be returned if
67     *         no documents will be accepted by this Filter.
68     * 
69     * @see DocIdBitSet
70     */
71    public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
72      return new DocIdBitSet(bits(reader));
73    }
74  }
