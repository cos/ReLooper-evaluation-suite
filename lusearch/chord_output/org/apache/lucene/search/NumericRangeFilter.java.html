24  /**
25   * A {@link Filter} that only accepts numeric values within
26   * a specified range. To use this, you must first index the
27   * numeric values using {@link NumericField} (expert: {@link
28   * NumericTokenStream}).
29   *
30   * <p>You create a new NumericRangeFilter with the static
31   * factory methods, eg:
32   *
33   * <pre>
34   * Filter f = NumericRangeFilter.newFloatRange("weight",
35   *                                             new Float(0.3f), new Float(0.10f),
36   *                                             true, true);
37   * </pre>
38   *
39   * accepts all documents whose float valued "weight" field
40   * ranges from 0.3 to 0.10, inclusive.
41   * See {@link NumericRangeQuery} for details on how Lucene
42   * indexes and searches numeric valued fields.
43   *
44   * <p><font color="red"><b>NOTE:</b> This API is experimental and
45   * might change in incompatible ways in the next
46   * release.</font>
47   *
48   * @since 2.9
49   **/
50  public final class NumericRangeFilter extends MultiTermQueryWrapperFilter {
51  
52    private NumericRangeFilter(final NumericRangeQuery query) {
53      super(query);
54    }
55    
56    /**
57     * Factory that creates a <code>NumericRangeFilter</code>, that filters a <code>long</code>
58     * range using the given <a href="NumericRangeQuery.html#precisionStepDesc"><code>precisionStep</code></a>.
59     * You can have half-open ranges (which are in fact &lt;/&le; or &gt;/&ge; queries)
60     * by setting the min or max value to <code>null</code>. By setting inclusive to false, it will
61     * match all documents excluding the bounds, with inclusive on, the boundaries are hits, too.
62     */
63    public static NumericRangeFilter newLongRange(final String field, final int precisionStep,
64      Long min, Long max, final boolean minInclusive, final boolean maxInclusive
65    ) {
66      return new NumericRangeFilter(
67        NumericRangeQuery.newLongRange(field, precisionStep, min, max, minInclusive, maxInclusive)
68      );
69    }
70

75     * by setting the min or max value to <code>null</code>. By setting inclusive to false, it will
