53    /** Expert: Stores term text values and document ordering data. */
54    public static class StringIndex {
55        
56      public int binarySearchLookup(String key) {
57        // this special case is the reason that Arrays.binarySearch() isn't useful.
58        if (key == null)
59          return 0;
60        
61        int low = 1;
62        int high = lookup.length-1;
63  
64        while (low <= high) {
65          int mid = (low + high) >>> 1;
66          int cmp = lookup[mid].compareTo(key);
67  
68          if (cmp < 0)
69            low = mid + 1;
70          else if (cmp > 0)
71            high = mid - 1;
72          else
73            return mid; // key found
74        }
75        return -(low + 1);  // key not found.
76      }
77      
78      /** All the term values, in natural order. */
79      public final String[] lookup;
80  
81      /** For each document, an index into the lookup array. */
82      public final int[] order;
83  
84      /** Creates one of these objects */
85      public StringIndex (int[] values, String[] lookup) {
86        this.order = values;
87        this.lookup = lookup;
88      }
89    }

520   /**
521    * EXPERT: A unique Identifier/Description for each item in the FieldCache. 
522    * Can be useful for logging/debugging.
523    * <p>
524    * <b>EXPERIMENTAL API:</b> This API is considered extremely advanced 
525    * and experimental.  It may be removed or altered w/o warning in future 
526    * releases 
527    * of Lucene.
528    * </p>
529    */
530   public static abstract class CacheEntry {
531     public abstract Object getReaderKey();
532     public abstract String getFieldName();
533     public abstract Class getCacheType();
534     public abstract Object getCustom();
535     public abstract Object getValue();
536     private String size = null;
537     protected final void setEstimatedSize(String size) {
538       this.size = size;
539     }
540     /** 
541      * @see #estimateSize(RamUsageEstimator)
542      */
543     public void estimateSize() {
544       estimateSize(new RamUsageEstimator(false)); // doesn't check for interned
545     }
546     /** 
547      * Computes (and stores) the estimated size of the cache Value 
548      * @see #getEstimatedSize
549      */
550     public void estimateSize(RamUsageEstimator ramCalc) {
551       long size = ramCalc.estimateRamUsage(getValue());
552       setEstimatedSize(RamUsageEstimator.humanReadableUnits
553                        (size, new DecimalFormat("0.#")));
554                         
555     }
556     /**
557      * The most recently estimated size of the value, null unless 
558      * estimateSize has been called.
559      */
560     public final String getEstimatedSize() {
561       return size;
562     }
563     
564     
565     public String toString() {
566       StringBuffer b = new StringBuffer();
567       b.append("'").append(getReaderKey()).append("'=>");
568       b.append("'").append(getFieldName()).append("',");
569       b.append(getCacheType()).append(",").append(getCustom());
570       b.append("=>").append(getValue().getClass().getName()).append("#");
571       b.append(System.identityHashCode(getValue()));
572       
573       String s = getEstimatedSize();
574       if(null != s) {
575         b.append(" (size =~ ").append(s).append(')');
576       }
577 
578       return b.toString();
579     }

581   }
