27  /**
28   * Expert: A hit queue for sorting by hits by terms in more than one field.
29   * Uses <code>FieldCache.DEFAULT</code> for maintaining internal term lookup tables.
30   *
31   * <p>Created: Dec 8, 2003 12:56:03 PM
32   *
33   * @since   lucene 1.4
34   * @version $Id: FieldSortedHitQueue.java 811070 2009-09-03 18:31:41Z hossman $
35   * @see Searcher#search(Query,Filter,int,Sort)
36   * @see FieldCache
37   * @deprecated see {@link FieldValueHitQueue}
38   */
39  public class FieldSortedHitQueue
40  extends PriorityQueue {
41  
42    /**
43     * Creates a hit queue sorted by the given list of fields.
44     * @param reader  Index to use.
45     * @param fields Fieldable names, in priority order (highest priority first).  Cannot be <code>null</code> or empty.
46     * @param size  The number of hits to retain.  Must be greater than zero.
47     * @throws IOException
48     */
49    public FieldSortedHitQueue (IndexReader reader, SortField[] fields, int size)
50    throws IOException {
51      final int n = fields.length;
52      comparators = new ScoreDocComparator[n];
53      this.fields = new SortField[n];
54      for (int i=0; i<n; ++i) {
55        String fieldname = fields[i].getField();
56        comparators[i] = getCachedComparator (reader, fieldname, fields[i].getType(), fields[i].getParser(), fields[i].getLocale(), fields[i].getFactory());
57        // new SortField instances must only be created when auto-detection is in use
58        if (fields[i].getType() == SortField.AUTO) {
59          if (comparators[i].sortType() == SortField.STRING) {
60            this.fields[i] = new SortField (fieldname, fields[i].getLocale(), fields[i].getReverse());
61          } else {
62            this.fields[i] = new SortField (fieldname, comparators[i].sortType(), fields[i].getReverse());
63          }
64        } else {
65          assert comparators[i].sortType() == fields[i].getType();
66          this.fields[i] = fields[i];
67        }
68      }
69      initialize (size);
70    }

82    /** returns the maximum score encountered by elements inserted via insert()
83     */
84    public float getMaxScore() {
85      return maxscore;

88    // Update maxscore.
89    private final void updateMaxScore(FieldDoc fdoc) {
90      maxscore = Math.max(maxscore, fdoc.score);
91    }
92    
93    // The signature of this method takes a FieldDoc in order to avoid
94    // the unneeded cast to retrieve the score.
95    // inherit javadoc
96    public boolean insert(FieldDoc fdoc) {
97      updateMaxScore(fdoc);
98      return super.insert(fdoc);
99    }

101   // This overrides PriorityQueue.insert() so that insert(FieldDoc) that
102   // keeps track of the score isn't accidentally bypassed.  
103   // inherit javadoc
104   public boolean insert(Object fdoc) {
105     return insert((FieldDoc)fdoc);
106   }

108   // This overrides PriorityQueue.insertWithOverflow() so that
109   // updateMaxScore(FieldDoc) that keeps track of the score isn't accidentally
110   // bypassed.
111   public Object insertWithOverflow(Object element) {
112     updateMaxScore((FieldDoc) element);
113     return super.insertWithOverflow(element);
114   }

116   /**
117    * Returns whether <code>a</code> is less relevant than <code>b</code>.
118    * @param a ScoreDoc
119    * @param b ScoreDoc
120    * @return <code>true</code> if document <code>a</code> should be sorted after document <code>b</code>.
121    */
122   protected boolean lessThan (final Object a, final Object b) {
123     final ScoreDoc docA = (ScoreDoc) a;
124     final ScoreDoc docB = (ScoreDoc) b;
125 
126     // run comparators
127     final int n = comparators.length;
128     int c = 0;
129     for (int i=0; i<n && c==0; ++i) {
130       c = (fields[i].reverse) ? comparators[i].compare (docB, docA)
131                               : comparators[i].compare (docA, docB);
132     }
133     // avoid random sort order that could lead to duplicates (bug #31241):
134     if (c == 0)
135       return docA.doc > docB.doc;
136     return c > 0;
137   }

140   /**
141    * Given a FieldDoc object, stores the values used
142    * to sort the given document.  These values are not the raw
143    * values out of the index, but the internal representation
144    * of them.  This is so the given search hit can be collated
145    * by a MultiSearcher with other search hits.
146    * @param  doc  The FieldDoc to store sort values into.
147    * @return  The same FieldDoc passed in.
148    * @see Searchable#search(Weight,Filter,int,Sort)
149    */
150   FieldDoc fillFields (final FieldDoc doc) {
151     final int n = comparators.length;
152     final Comparable[] fields = new Comparable[n];
153     for (int i=0; i<n; ++i)
154       fields[i] = comparators[i].sortValue(doc);
155     doc.fields = fields;
156     //if (maxscore > 1.0f) doc.score /= maxscore;   // normalize scores
157     return doc;
158   }

161   /** Returns the SortFields being used by this hit queue. */
162   SortField[] getFields() {
163     return fields;
164   }

166   static ScoreDocComparator getCachedComparator (IndexReader reader, String field, int type, FieldCache.Parser parser, Locale locale, SortComparatorSource factory)
167   throws IOException {
168     if (type == SortField.DOC) return ScoreDocComparator.INDEXORDER;
169     if (type == SortField.SCORE) return ScoreDocComparator.RELEVANCE;
170     FieldCacheImpl.Entry entry = (factory != null)
171       ? new FieldCacheImpl.Entry (field, factory)
172       : ( (parser != null)
173         ? new FieldCacheImpl.Entry (field, type, parser)
174         : new FieldCacheImpl.Entry (field, type, locale)
175       );
176     return (ScoreDocComparator)Comparators.get(reader, entry);
177   }
178 
179   /** Internal cache of comparators. Similar to FieldCache, only
180    *  caches comparators instead of term values. */
181   static final FieldCacheImpl.Cache Comparators = new FieldCacheImpl.Cache() {
182 
183     protected Object createValue(IndexReader reader, FieldCacheImpl.Entry entryKey)
184         throws IOException {
185       FieldCacheImpl.Entry entry = (FieldCacheImpl.Entry) entryKey;
186       String fieldname = entry.field;
187       int type = entry.type;
188       Locale locale = entry.locale;
189       FieldCache.Parser parser = null;
190       SortComparatorSource factory = null;
191       if (entry.custom instanceof SortComparatorSource) {
192         factory = (SortComparatorSource) entry.custom;
193       } else {
194         parser = (FieldCache.Parser) entry.custom;
195       }
196       ScoreDocComparator comparator;
197       switch (type) {
198         case SortField.AUTO:
199           comparator = comparatorAuto (reader, fieldname);
200           break;
201         case SortField.INT:
202           comparator = comparatorInt (reader, fieldname, (FieldCache.IntParser)parser);
203           break;
204         case SortField.FLOAT:
205           comparator = comparatorFloat (reader, fieldname, (FieldCache.FloatParser)parser);
206           break;
207         case SortField.LONG:
208           comparator = comparatorLong(reader, fieldname, (FieldCache.LongParser)parser);
209           break;
210         case SortField.DOUBLE:
211           comparator = comparatorDouble(reader, fieldname, (FieldCache.DoubleParser)parser);
212           break;
213         case SortField.SHORT:
214           comparator = comparatorShort(reader, fieldname, (FieldCache.ShortParser)parser);
215           break;
216         case SortField.BYTE:
217           comparator = comparatorByte(reader, fieldname, (FieldCache.ByteParser)parser);
218           break;
219         case SortField.STRING:
220           if (locale != null) comparator = comparatorStringLocale (reader, fieldname, locale);
221           else comparator = comparatorString (reader, fieldname);
222           break;
223         case SortField.CUSTOM:
224           comparator = factory.newComparator (reader, fieldname);
225           break;
226         default:
227           throw new RuntimeException ("unknown field type: "+type);
228       }
229       return comparator;
230     }
231   };

233    /**
234    * Returns a comparator for sorting hits according to a field containing bytes.
235    * @param reader  Index to use.
236    * @param fieldname  Fieldable containing integer values.
237    * @return  Comparator for sorting hits.
238    * @throws IOException If an error occurs reading the index.
239    */
240   static ScoreDocComparator comparatorByte(final IndexReader reader, final String fieldname, final FieldCache.ByteParser parser)
241   throws IOException {
242     final String field = fieldname.intern();
243     final byte[] fieldOrder = FieldCache.DEFAULT.getBytes(reader, field, parser);
244     return new ScoreDocComparator() {
245 
246       public final int compare (final ScoreDoc i, final ScoreDoc j) {
247         final int fi = fieldOrder[i.doc];
248         final int fj = fieldOrder[j.doc];
249         if (fi < fj) return -1;
250         if (fi > fj) return 1;
251         return 0;
252       }
253 
254       public Comparable sortValue (final ScoreDoc i) {
255         return new Byte(fieldOrder[i.doc]);
256       }
257 
258       public int sortType() {
259         return SortField.BYTE;
260       }
261     };
262   }

451   /**
452    * Returns a comparator for sorting hits according to a field containing strings.
453    * @param reader  Index to use.
454    * @param fieldname  Fieldable containing string values.
455    * @return  Comparator for sorting hits.
456    * @throws IOException If an error occurs reading the index.
457    */
458   static ScoreDocComparator comparatorStringLocale (final IndexReader reader, final String fieldname, final Locale locale)
459   throws IOException {
460     final Collator collator = Collator.getInstance (locale);
461     final String field = fieldname.intern();
462     final String[] index = FieldCache.DEFAULT.getStrings (reader, field);
463     return new ScoreDocComparator() {
464 
465         public final int compare(final ScoreDoc i, final ScoreDoc j) {
466             String is = index[i.doc];
467             String js = index[j.doc];
468             if (is == js) {
469                 return 0;
470             } else if (is == null) {
471                 return -1;
472             } else if (js == null) {
473                 return 1;
474             } else {
475                 return collator.compare(is, js);
476             }
477         }
478 
479       public Comparable sortValue (final ScoreDoc i) {
480         return index[i.doc];
481       }
482 
483       public int sortType() {
484         return SortField.STRING;
485       }
486     };
487   }

489   /**
490    * Returns a comparator for sorting hits according to values in the given field.
491    * The terms in the field are looked at to determine whether they contain integers,
492    * floats or strings.  Once the type is determined, one of the other static methods
493    * in this class is called to get the comparator.
494    * @param reader  Index to use.
495    * @param fieldname  Fieldable containing values.
496    * @return  Comparator for sorting hits.
497    * @throws IOException If an error occurs reading the index.
498    */
499   static ScoreDocComparator comparatorAuto (final IndexReader reader, final String fieldname)
500   throws IOException {
501     final String field = fieldname.intern();
502     Object lookupArray = FieldCache.DEFAULT.getAuto (reader, field);
503     if (lookupArray instanceof FieldCache.StringIndex) {
504       return comparatorString (reader, field);
505     } else if (lookupArray instanceof int[]) {
506       return comparatorInt (reader, field, null);
507     } else if (lookupArray instanceof long[]) {
508       return comparatorLong (reader, field, null);
509     } else if (lookupArray instanceof float[]) {
510       return comparatorFloat (reader, field, null);
511     } else if (lookupArray instanceof String[]) {
512       return comparatorString (reader, field);
513     } else {
514       throw new RuntimeException ("unknown data type in field '"+field+"'");
515     }
516   }
