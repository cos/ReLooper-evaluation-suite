19  import org.apache.lucene.index.IndexReader;
20  import org.apache.lucene.search.spans.SpanQuery;
21  import org.apache.lucene.search.spans.Spans;
22  import org.apache.lucene.util.OpenBitSet;
23  
24  import java.io.IOException;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  /**
29   * Constrains search results to only match those which also match a provided
30   * query. Also provides position information about where each document matches
31   * at the cost of extra space compared with the QueryWrapperFilter.
32   * There is an added cost to this above what is stored in a {@link QueryWrapperFilter}.  Namely,
33   * the position information for each matching document is stored.
34   * <p/>
35   * This filter does not cache.  See the {@link org.apache.lucene.search.CachingSpanFilter} for a wrapper that
36   * caches.
37   *
38   *
39   * @version $Id:$
40   */
41  public class SpanQueryFilter extends SpanFilter {
42    protected SpanQuery query;
43  
44    protected SpanQueryFilter()
45    {
46      
47    }
48  
49    /** Constructs a filter which only matches documents matching
50     * <code>query</code>.
51     * @param query The {@link org.apache.lucene.search.spans.SpanQuery} to use as the basis for the Filter.
52     */
53    public SpanQueryFilter(SpanQuery query) {
54      this.query = query;
55    }
56  
57    public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
58      SpanFilterResult result = bitSpans(reader);
59      return result.getDocIdSet();
60    }
61  
62    public SpanFilterResult bitSpans(IndexReader reader) throws IOException {
63  
64      final OpenBitSet bits = new OpenBitSet(reader.maxDoc());
65      Spans spans = query.getSpans(reader);
66      List tmp = new ArrayList(20);
67      int currentDoc = -1;
68      SpanFilterResult.PositionInfo currentInfo = null;
69      while (spans.next())
70      {
71        int doc = spans.doc();
72        bits.set(doc);
73        if (currentDoc != doc)
74        {
75          currentInfo = new SpanFilterResult.PositionInfo(doc);
76          tmp.add(currentInfo);
77          currentDoc = doc;
78        }
79        currentInfo.addPosition(spans.start(), spans.end());
80      }
81      return new SpanFilterResult(bits, tmp);
82    }
83  
84  
85    public SpanQuery getQuery() {
86      return query;
87    }
88  
89    public String toString() {
90      return "SpanQueryFilter(" + query + ")";
91    }
92  
93    public boolean equals(Object o) {
94      return o instanceof SpanQueryFilter && this.query.equals(((SpanQueryFilter) o).query);
95    }
96  
97    public int hashCode() {
98      return query.hashCode() ^ 0x923F64B9;
99    }
100 }
