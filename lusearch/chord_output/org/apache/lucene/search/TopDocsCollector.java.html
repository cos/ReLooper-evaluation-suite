20  import org.apache.lucene.util.PriorityQueue;
21  
22  /**
23   * A base class for all collectors that return a {@link TopDocs} output. This
24   * collector allows easy extension by providing a single constructor which
25   * accepts a {@link PriorityQueue} as well as protected members for that
26   * priority queue and a counter of the number of total hits.<br>
27   * Extending classes can override {@link #topDocs(int, int)} and
28   * {@link #getTotalHits()} in order to provide their own implementation.
29   */
30  public abstract class TopDocsCollector extends Collector {
31  
32    // This is used in case topDocs() is called with illegal parameters, or there
33    // simply aren't (enough) results.
34    protected static final TopDocs EMPTY_TOPDOCS = new TopDocs(0, new ScoreDoc[0], Float.NaN);
35    
36    /**
37     * The priority queue which holds the top documents. Note that different
38     * implementations of PriorityQueue give different meaning to 'top documents'.
39     * HitQueue for example aggregates the top scoring documents, while other PQ
40     * implementations may hold documents sorted by other criteria.
41     */
42    protected PriorityQueue pq;
43  
44    /** The total number of documents that the collector encountered. */
45    protected int totalHits;
46    
47    protected TopDocsCollector(PriorityQueue pq) {
48      this.pq = pq;
49    }
50    
51    /**
52     * Populates the results array with the ScoreDoc instaces. This can be
53     * overridden in case a different ScoreDoc type should be returned.
54     */
55    protected void populateResults(ScoreDoc[] results, int howMany) {
56      for (int i = howMany - 1; i >= 0; i--) { 
57        results[i] = (ScoreDoc) pq.pop();
58      }
59    }
60  
61    /**
62     * Returns a {@link TopDocs} instance containing the given results. If
63     * <code>results</code> is null it means there are no results to return,
64     * either because there were 0 calls to collect() or because the arguments to
65     * topDocs were invalid.
66     */
67    protected TopDocs newTopDocs(ScoreDoc[] results, int start) {
68      return results == null ? EMPTY_TOPDOCS : new TopDocs(totalHits, results);
69    }
70    
71    /** The total number of documents that matched this query. */
72    public int getTotalHits() {
73      return totalHits;
74    }
75    
76    /** Returns the top docs that were collected by this collector. */
77    public final TopDocs topDocs() {
78      // In case pq was populated with sentinel values, there might be less
79      // results than pq.size(). Therefore return all results until either
80      // pq.size() or totalHits.
81      return topDocs(0, totalHits < pq.size() ? totalHits : pq.size());
82    }
83  
84    /**
85     * Returns the documents in the rage [start .. pq.size()) that were collected
86     * by this collector. Note that if start >= pq.size(), an empty TopDocs is
87     * returned.<br>
88     * This method is convenient to call if the application always asks for the
89     * last results, starting from the last 'page'.<br>
90     * <b>NOTE:</b> you cannot call this method more than once for each search
91     * execution. If you need to call it more than once, passing each time a
92     * different <code>start</code>, you should call {@link #topDocs()} and work
93     * with the returned {@link TopDocs} object, which will contain all the
94     * results this search execution collected.
95     */
96    public final TopDocs topDocs(int start) {
97      // In case pq was populated with sentinel values, there might be less
98      // results than pq.size(). Therefore return all results until either
99      // pq.size() or totalHits.
100     return topDocs(start, totalHits < pq.size() ? totalHits : pq.size());
101   }
102 
103   /**
104    * Returns the documents in the rage [start .. start+howMany) that were
105    * collected by this collector. Note that if start >= pq.size(), an empty
106    * TopDocs is returned, and if pq.size() - start &lt; howMany, then only the
107    * available documents in [start .. pq.size()) are returned.<br>
108    * This method is useful to call in case pagination of search results is
109    * allowed by the search application, as well as it attempts to optimize the
110    * memory used by allocating only as much as requested by howMany.<br>
111    * <b>NOTE:</b> you cannot call this method more than once for each search
112    * execution. If you need to call it more than once, passing each time a
113    * different range, you should call {@link #topDocs()} and work with the
114    * returned {@link TopDocs} object, which will contain all the results this
115    * search execution collected.
116    */
117   public final TopDocs topDocs(int start, int howMany) {
118     
119     // In case pq was populated with sentinel values, there might be less
120     // results than pq.size(). Therefore return all results until either
121     // pq.size() or totalHits.
122     int size = totalHits < pq.size() ? totalHits : pq.size();
123 
124     // Don't bother to throw an exception, just return an empty TopDocs in case
125     // the parameters are invalid or out of range.
126     if (start < 0 || start >= size || howMany <= 0) {
127       return newTopDocs(null, start);
128     }
129 
130     // We know that start < pqsize, so just fix howMany. 
131     howMany = Math.min(size - start, howMany);
132     ScoreDoc[] results = new ScoreDoc[howMany];
133 
134     // pq's pop() returns the 'least' element in the queue, therefore need
135     // to discard the first ones, until we reach the requested range.
136     // Note that this loop will usually not be executed, since the common usage
137     // should be that the caller asks for the last howMany results. However it's
138     // needed here for completeness.
139     for (int i = pq.size() - start - howMany; i > 0; i--) { pq.pop(); }
140     
141     // Get the requested results from pq.
142     populateResults(results, howMany);
143     
144     return newTopDocs(results, start);
145   }
146 
147 }
