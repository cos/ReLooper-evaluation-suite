20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.util.ToStringUtils;
22  
23  import java.io.IOException;
24  import java.util.Set;

27  /**
28   * A query that applies a filter to the results of another query.
29   *
30   * <p>Note: the bits are retrieved from the filter each time this
31   * query is used in a search - use a CachingWrapperFilter to avoid
32   * regenerating the bits every time.
33   *
34   * <p>Created: Apr 20, 2004 8:58:29 AM
35   *
36   * @since   1.4
37   * @version $Id: FilteredQuery.java 807821 2009-08-25 21:55:49Z mikemccand $
38   * @see     CachingWrapperFilter
39   */
40  public class FilteredQuery
41  extends Query {
42  
43    Query query;
44    Filter filter;
45  
46    /**
47     * Constructs a new query which applies a filter to the results of the original query.
48     * Filter.getDocIdSet() will be called every time this query is used in a search.
49     * @param query  Query to be filtered, cannot be <code>null</code>.
50     * @param filter Filter to apply to query results, cannot be <code>null</code>.
51     */
52    public FilteredQuery (Query query, Filter filter) {
53      this.query = query;
54      this.filter = filter;
55    }
56  
57    /**
58     * Returns a Weight that applies the filter to the enclosed query's Weight.
59     * This is accomplished by overriding the Scorer returned by the Weight.
60     */
61    public Weight createWeight(final Searcher searcher) throws IOException {
62      final Weight weight = query.createWeight (searcher);
63      final Similarity similarity = query.getSimilarity(searcher);
64      return new Weight() {
65        private float value;
66          
67        // pass these methods through to enclosed query's weight
68        public float getValue() { return value; }
69        public float sumOfSquaredWeights() throws IOException { 
70          return weight.sumOfSquaredWeights() * getBoost() * getBoost(); 
71        }
72        public void normalize (float v) { 
73          weight.normalize(v);
74          value = weight.getValue() * getBoost();
75        }
76        public Explanation explain (IndexReader ir, int i) throws IOException {
77          Explanation inner = weight.explain (ir, i);
78          if (getBoost()!=1) {
79            Explanation preBoost = inner;
80            inner = new Explanation(inner.getValue()*getBoost(),"product of:");
81            inner.addDetail(new Explanation(getBoost(),"boost"));
82            inner.addDetail(preBoost);
83          }
84          Filter f = FilteredQuery.this.filter;
85          DocIdSet docIdSet = f.getDocIdSet(ir);
86          DocIdSetIterator docIdSetIterator = docIdSet == null ? DocIdSet.EMPTY_DOCIDSET.iterator() : docIdSet.iterator();
87          if (docIdSetIterator == null) {
88            docIdSetIterator = DocIdSet.EMPTY_DOCIDSET.iterator();
89          }
90          if (docIdSetIterator.advance(i) == i) {
91            return inner;
92          } else {
93            Explanation result = new Explanation
94              (0.0f, "failure to match filter: " + f.toString());
95            result.addDetail(inner);
96            return result;
97          }
98        }
99  
100       // return this query
101       public Query getQuery() { return FilteredQuery.this; }
102 
103       // return a filtering scorer
104       public Scorer scorer(IndexReader indexReader, boolean scoreDocsInOrder, boolean topScorer)
105           throws IOException {
106         final Scorer scorer = weight.scorer(indexReader, true, false);
107         if (scorer == null) {
108           return null;
109         }
110         DocIdSet docIdSet = filter.getDocIdSet(indexReader);
111         if (docIdSet == null) {
112           return null;
113         }
114         final DocIdSetIterator docIdSetIterator = docIdSet.iterator();
115         if (docIdSetIterator == null) {
116           return null;
117         }
118 
119         return new Scorer(similarity) {
120 
121           private int doc = -1;
122           
123           private int advanceToCommon(int scorerDoc, int disiDoc) throws IOException {
124             while (scorerDoc != disiDoc) {
125               if (scorerDoc < disiDoc) {
126                 scorerDoc = scorer.advance(disiDoc);
127               } else {
128                 disiDoc = docIdSetIterator.advance(scorerDoc);
129               }
130             }
131             return scorerDoc;
132           }
133 
134           /** @deprecated use {@link #nextDoc()} instead. */
135           public boolean next() throws IOException {
136             return nextDoc() != NO_MORE_DOCS;
137           }
138 
139           public int nextDoc() throws IOException {
140             int scorerDoc, disiDoc;
141             return doc = (disiDoc = docIdSetIterator.nextDoc()) != NO_MORE_DOCS
142                 && (scorerDoc = scorer.nextDoc()) != NO_MORE_DOCS
143                 && advanceToCommon(scorerDoc, disiDoc) != NO_MORE_DOCS ? scorer.docID() : NO_MORE_DOCS;
144           }
145           
146           /** @deprecated use {@link #docID()} instead. */
147           public int doc() { return scorer.doc(); }
148           public int docID() { return doc; }
149           
150           /** @deprecated use {@link #advance(int)} instead. */
151           public boolean skipTo(int i) throws IOException {
152             return advance(i) != NO_MORE_DOCS;
153           }
154           
155           public int advance(int target) throws IOException {
156             int disiDoc, scorerDoc;
157             return doc = (disiDoc = docIdSetIterator.advance(target)) != NO_MORE_DOCS
158                 && (scorerDoc = scorer.advance(disiDoc)) != NO_MORE_DOCS 
159                 && advanceToCommon(scorerDoc, disiDoc) != NO_MORE_DOCS ? scorer.docID() : NO_MORE_DOCS;
160           }
161 
162           public float score() throws IOException { return getBoost() * scorer.score(); }
163 
164           // add an explanation about whether the document was filtered
165           public Explanation explain (int i) throws IOException {
166             Explanation exp = scorer.explain(i);
167             
168             if (docIdSetIterator.advance(i) == i) {
169               exp.setDescription ("allowed by filter: "+exp.getDescription());
170               exp.setValue(getBoost() * exp.getValue());
171             } else {
172               exp.setDescription ("removed by filter: "+exp.getDescription());
173               exp.setValue(0.0f);
174             }
175             return exp;
176           }
177         };
178       }
179     };
180   }
181 
182   /** Rewrites the wrapped query. */
183   public Query rewrite(IndexReader reader) throws IOException {
184     Query rewritten = query.rewrite(reader);
185     if (rewritten != query) {
186       FilteredQuery clone = (FilteredQuery)this.clone();
187       clone.query = rewritten;
188       return clone;
189     } else {
190       return this;
191     }
192   }
193 
194   public Query getQuery() {
195     return query;
196   }
197 
198   public Filter getFilter() {
199     return filter;
200   }
201 
202   // inherit javadoc
203   public void extractTerms(Set terms) {
204       getQuery().extractTerms(terms);
205   }
206 
207   /** Prints a user-readable version of this query. */
208   public String toString (String s) {
209     StringBuffer buffer = new StringBuffer();
210     buffer.append("filtered(");
211     buffer.append(query.toString(s));
212     buffer.append(")->");
213     buffer.append(filter);
214     buffer.append(ToStringUtils.boost(getBoost()));
215     return buffer.toString();
216   }
217 
218   /** Returns true iff <code>o</code> is equal to this. */
219   public boolean equals(Object o) {
220     if (o instanceof FilteredQuery) {
221       FilteredQuery fq = (FilteredQuery) o;
222       return (query.equals(fq.query) && filter.equals(fq.filter) && getBoost()==fq.getBoost());
223     }
224     return false;
225   }
226 
227   /** Returns a hash code value for this object. */
228   public int hashCode() {
229     return query.hashCode() ^ filter.hashCode() + Float.floatToRawIntBits(getBoost());
230   }
231 }
