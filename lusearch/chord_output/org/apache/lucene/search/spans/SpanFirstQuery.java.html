20  import java.io.IOException;
21  
22  import java.util.Collection;
23  import java.util.Set;
24  import java.util.ArrayList;
25  
26  import org.apache.lucene.index.IndexReader;
27  import org.apache.lucene.search.Query;
28  import org.apache.lucene.util.ToStringUtils;
29  
30  /** Matches spans near the beginning of a field. */
31  public class SpanFirstQuery extends SpanQuery implements Cloneable {
32    private SpanQuery match;
33    private int end;
34  
35    /** Construct a SpanFirstQuery matching spans in <code>match</code> whose end
36     * position is less than or equal to <code>end</code>. */
37    public SpanFirstQuery(SpanQuery match, int end) {
38      this.match = match;
39      this.end = end;
40    }
41  
42    /** Return the SpanQuery whose matches are filtered. */
43    public SpanQuery getMatch() { return match; }
44  
45    /** Return the maximum end position permitted in a match. */
46    public int getEnd() { return end; }
47  
48    public String getField() { return match.getField(); }
49  
50    /** Returns a collection of all terms matched by this query.
51     * @deprecated use extractTerms instead
52     * @see #extractTerms(Set)
53     */
54    public Collection getTerms() { return match.getTerms(); }
55  
56    public String toString(String field) {
57      StringBuffer buffer = new StringBuffer();
58      buffer.append("spanFirst(");
59      buffer.append(match.toString(field));
60      buffer.append(", ");
61      buffer.append(end);
62      buffer.append(")");
63      buffer.append(ToStringUtils.boost(getBoost()));
64      return buffer.toString();
65    }
66    
67    public Object clone() {
68      SpanFirstQuery spanFirstQuery = new SpanFirstQuery((SpanQuery) match.clone(), end);
69      spanFirstQuery.setBoost(getBoost());
70      return spanFirstQuery;
71    }
72    
73    public void extractTerms(Set terms) {
74          match.extractTerms(terms);
75    }
76  
77    public Spans getSpans(final IndexReader reader) throws IOException {
78      return new Spans() {
79          private Spans spans = match.getSpans(reader);
80  
81          public boolean next() throws IOException {
82            while (spans.next()) {                  // scan to next match
83              if (end() <= end)
84                return true;
85            }
86            return false;
87          }
88  
89          public boolean skipTo(int target) throws IOException {
90            if (!spans.skipTo(target))
91              return false;
92  
93            return spans.end() <= end || next();
94  
95          }
96  
97          public int doc() { return spans.doc(); }
98          public int start() { return spans.start(); }
99          public int end() { return spans.end(); }
100 
101       // TODO: Remove warning after API has been finalized
102       public Collection/*<byte[]>*/ getPayload() throws IOException {
103         ArrayList result = null;
104         if (spans.isPayloadAvailable()) {
105           result = new ArrayList(spans.getPayload());
106         }
107         return result;//TODO: any way to avoid the new construction?
108       }
109 
110       // TODO: Remove warning after API has been finalized
111      public boolean isPayloadAvailable() {
112         return spans.isPayloadAvailable();
113       }
114 
115       public String toString() {
116           return "spans(" + SpanFirstQuery.this.toString() + ")";
117         }
118 
119       };
120   }
121 
122   public Query rewrite(IndexReader reader) throws IOException {
123     SpanFirstQuery clone = null;
124 
125     SpanQuery rewritten = (SpanQuery) match.rewrite(reader);
126     if (rewritten != match) {
127       clone = (SpanFirstQuery) this.clone();
128       clone.match = rewritten;
129     }
130 
131     if (clone != null) {
132       return clone;                        // some clauses rewrote
133     } else {
134       return this;                         // no clauses rewrote
135     }
136   }
137 
138   public boolean equals(Object o) {
139     if (this == o) return true;
140     if (!(o instanceof SpanFirstQuery)) return false;
141 
142     SpanFirstQuery other = (SpanFirstQuery)o;
143     return this.end == other.end
144          && this.match.equals(other.match)
145          && this.getBoost() == other.getBoost();
146   }
147 
148   public int hashCode() {
149     int h = match.hashCode();
150     h ^= (h << 8) | (h >>> 25);  // reversible
151     h ^= Float.floatToRawIntBits(getBoost()) ^ end;
152     return h;
153   }
154 
155 
156 }
