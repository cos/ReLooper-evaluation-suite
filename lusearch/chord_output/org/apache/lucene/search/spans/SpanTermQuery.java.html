20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.index.Term;
22  import org.apache.lucene.util.ToStringUtils;
23  
24  import java.io.IOException;
25  import java.util.ArrayList;
26  import java.util.Collection;
27  import java.util.Set;
28  
29  /** Matches spans containing a term. */
30  public class SpanTermQuery extends SpanQuery {
31    protected Term term;
32  
33    /** Construct a SpanTermQuery matching the named term's spans. */
34    public SpanTermQuery(Term term) { this.term = term; }
35  
36    /** Return the term whose spans are matched. */
37    public Term getTerm() { return term; }
38  
39    public String getField() { return term.field(); }
40    
41    /** Returns a collection of all terms matched by this query.
42     * @deprecated use extractTerms instead
43     * @see #extractTerms(Set)
44     */
45    public Collection getTerms() {
46      Collection terms = new ArrayList();
47      terms.add(term);
48      return terms;
49    }
50    public void extractTerms(Set terms) {
51        terms.add(term);
52    }
53  
54    public String toString(String field) {
55      StringBuffer buffer = new StringBuffer();
56      if (term.field().equals(field))
57        buffer.append(term.text());
58      else
59        buffer.append(term.toString());
60      buffer.append(ToStringUtils.boost(getBoost()));
61      return buffer.toString();
62    }
63  
64    public int hashCode() {
65      final int prime = 31;
66      int result = super.hashCode();
67      result = prime * result + ((term == null) ? 0 : term.hashCode());
68      return result;
69    }
70  
71    public boolean equals(Object obj) {
72      if (this == obj)
73        return true;
74      if (!super.equals(obj))
75        return false;
76      if (getClass() != obj.getClass())
77        return false;
78      SpanTermQuery other = (SpanTermQuery) obj;
79      if (term == null) {
80        if (other.term != null)
81          return false;
82      } else if (!term.equals(other.term))
83        return false;
84      return true;
85    }
86  
87    public Spans getSpans(final IndexReader reader) throws IOException {
88      return new TermSpans(reader.termPositions(term), term);
89    }
90  
91  }
