21  import org.apache.lucene.util.PriorityQueue;

30  /**
31   * Similar to {@link NearSpansOrdered}, but for the unordered case.
32   * 
33   * Expert:
34   * Only public for subclassing.  Most implementations should not need this class
35   */
36  public class NearSpansUnordered extends Spans {
37    private SpanNearQuery query;
38  
39    private List ordered = new ArrayList();         // spans in query order
40    private Spans[] subSpans;  
41    private int slop;                               // from query
42  
43    private SpansCell first;                        // linked list of spans
44    private SpansCell last;                         // sorted by doc only
45  
46    private int totalLength;                        // sum of current lengths
47  
48    private CellQueue queue;                        // sorted queue of spans
49    private SpansCell max;                          // max element in queue
50  
51    private boolean more = true;                    // true iff not done
52    private boolean firstTime = true;               // true before first next()
53  
54    private class CellQueue extends PriorityQueue {
55      public CellQueue(int size) {
56        initialize(size);
57      }
58      
59      protected final boolean lessThan(Object o1, Object o2) {
60        SpansCell spans1 = (SpansCell)o1;
61        SpansCell spans2 = (SpansCell)o2;
62        if (spans1.doc() == spans2.doc()) {
63          return NearSpansOrdered.docSpansOrdered(spans1, spans2);
64        } else {
65          return spans1.doc() < spans2.doc();
66        }
67      }
68    }
69  
70  
71    /** Wraps a Spans, and can be used to form a linked list. */
72    private class SpansCell extends Spans {
73      private Spans spans;
74      private SpansCell next;
75      private int length = -1;
76      private int index;
77  
78      public SpansCell(Spans spans, int index) {
79        this.spans = spans;
80        this.index = index;
81      }
82  
83      public boolean next() throws IOException {
84        return adjust(spans.next());
85      }
86  
87      public boolean skipTo(int target) throws IOException {
88        return adjust(spans.skipTo(target));
89      }
90      
91      private boolean adjust(boolean condition) {
92        if (length != -1) {
93          totalLength -= length;  // subtract old length
94        }
95        if (condition) {
96          length = end() - start(); 
97          totalLength += length; // add new length
98  
99          if (max == null || doc() > max.doc()
100             || (doc() == max.doc()) && (end() > max.end())) {
101           max = this;
102         }
103       }
104       more = condition;
105       return condition;
106     }
107 
108     public int doc() { return spans.doc(); }
109     public int start() { return spans.start(); }
110     public int end() { return spans.end(); }
111                     // TODO: Remove warning after API has been finalized
112     public Collection/*<byte[]>*/ getPayload() throws IOException {
113       return new ArrayList(spans.getPayload());
114     }
115 
116     // TODO: Remove warning after API has been finalized
117    public boolean isPayloadAvailable() {
118       return spans.isPayloadAvailable();
119     }
120 
121     public String toString() { return spans.toString() + "#" + index; }
122   }
123

125   public NearSpansUnordered(SpanNearQuery query, IndexReader reader)
126     throws IOException {
127     this.query = query;
128     this.slop = query.getSlop();
129 
130     SpanQuery[] clauses = query.getClauses();
131     queue = new CellQueue(clauses.length);
132     subSpans = new Spans[clauses.length];    
133     for (int i = 0; i < clauses.length; i++) {
134       SpansCell cell =
135         new SpansCell(clauses[i].getSpans(reader), i);
136       ordered.add(cell);
137       subSpans[i] = cell.spans;
138     }
139   }
140   public Spans[] getSubSpans() {
141       return subSpans;
142   }
143   public boolean next() throws IOException {
144     if (firstTime) {
145       initList(true);
146       listToQueue(); // initialize queue
147       firstTime = false;
148     } else if (more) {
149       if (min().next()) { // trigger further scanning
150         queue.adjustTop(); // maintain queue
151       } else {
152         more = false;
153       }
154     }
155 
156     while (more) {
157 
158       boolean queueStale = false;
159 
160       if (min().doc() != max.doc()) {             // maintain list
161         queueToList();
162         queueStale = true;
163       }
164 
165       // skip to doc w/ all clauses
166 
167       while (more && first.doc() < last.doc()) {
168         more = first.skipTo(last.doc());          // skip first upto last
169         firstToLast();                            // and move it to the end
170         queueStale = true;
171       }
172 
173       if (!more) return false;
174 
175       // found doc w/ all clauses
176 
177       if (queueStale) {                           // maintain the queue
178         listToQueue();
179         queueStale = false;
180       }
181 
182       if (atMatch()) {
183         return true;
184       }
185       
186       more = min().next();
187       if (more) {
188         queue.adjustTop();                      // maintain queue
189       }
190     }
191     return false;                                 // no more matches
192   }
193 
194   public boolean skipTo(int target) throws IOException {
195     if (firstTime) {                              // initialize
196       initList(false);
197       for (SpansCell cell = first; more && cell!=null; cell=cell.next) {
198         more = cell.skipTo(target);               // skip all
199       }
200       if (more) {
201         listToQueue();
202       }
203       firstTime = false;
204     } else {                                      // normal case
205       while (more && min().doc() < target) {      // skip as needed
206         if (min().skipTo(target)) {
207           queue.adjustTop();
208         } else {
209           more = false;
210         }
211       }
212     }
213     return more && (atMatch() ||  next());
214   }
215 
216   private SpansCell min() { return (SpansCell)queue.top(); }
217 
218   public int doc() { return min().doc(); }
219   public int start() { return min().start(); }
220   public int end() { return max.end(); }
221 
222   // TODO: Remove warning after API has been finalized
223   /**
224    * WARNING: The List is not necessarily in order of the the positions
225    * @return Collection of <code>byte[]</code> payloads
226    * @throws IOException
227    */
228   public Collection/*<byte[]>*/ getPayload() throws IOException {
229     Set/*<byte[]*/ matchPayload = new HashSet();
230     for (SpansCell cell = first; cell != null; cell = cell.next) {
231       if (cell.isPayloadAvailable()) {
232         matchPayload.addAll(cell.getPayload());
233       }
234     }
235     return matchPayload;
236   }
237 
238   // TODO: Remove warning after API has been finalized
239   public boolean isPayloadAvailable() {
240     SpansCell pointer = min();
241     while (pointer != null) {
242       if (pointer.isPayloadAvailable()) {
243         return true;
244       }
245       pointer = pointer.next;
246     }
247 
248     return false;
249   }
250 
251   public String toString() {
252     return getClass().getName() + "("+query.toString()+")@"+
253       (firstTime?"START":(more?(doc()+":"+start()+"-"+end()):"END"));
254   }
255 
256   private void initList(boolean next) throws IOException {
257     for (int i = 0; more && i < ordered.size(); i++) {
258       SpansCell cell = (SpansCell)ordered.get(i);
259       if (next)
260         more = cell.next();                       // move to first entry
261       if (more) {
262         addToList(cell);                          // add to list
263       }
264     }
265   }
266 
267   private void addToList(SpansCell cell) throws IOException {
268     if (last != null) {           // add next to end of list
269       last.next = cell;
270     } else
271       first = cell;
272     last = cell;
273     cell.next = null;
274   }
275 
276   private void firstToLast() {
277     last.next = first;            // move first to end of list
278     last = first;
279     first = first.next;
280     last.next = null;
281   }
282 
283   private void queueToList() throws IOException {
284     last = first = null;
285     while (queue.top() != null) {
286       addToList((SpansCell)queue.pop());
287     }
288   }
289   
290   private void listToQueue() {
291     queue.clear(); // rebuild queue
292     for (SpansCell cell = first; cell != null; cell = cell.next) {
293       queue.put(cell);                      // add to queue from list
294     }
295   }
296 
297   private boolean atMatch() {
298     return (min().doc() == max.doc())
299         && ((max.end() - min().start() - totalLength) <= slop);
300   }
301 }
