20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.search.Query;
22  import org.apache.lucene.util.ToStringUtils;
23  
24  import java.io.IOException;
25  import java.util.ArrayList;
26  import java.util.Collection;
27  import java.util.Set;
28  
29  /** Removes matches which overlap with another SpanQuery. */
30  public class SpanNotQuery extends SpanQuery implements Cloneable {
31    private SpanQuery include;
32    private SpanQuery exclude;
33  
34    /** Construct a SpanNotQuery matching spans from <code>include</code> which
35     * have no overlap with spans from <code>exclude</code>.*/
36    public SpanNotQuery(SpanQuery include, SpanQuery exclude) {
37      this.include = include;
38      this.exclude = exclude;
39  
40      if (!include.getField().equals(exclude.getField()))
41        throw new IllegalArgumentException("Clauses must have same field.");
42    }
43  
44    /** Return the SpanQuery whose matches are filtered. */
45    public SpanQuery getInclude() { return include; }
46  
47    /** Return the SpanQuery whose matches must not overlap those returned. */
48    public SpanQuery getExclude() { return exclude; }
49  
50    public String getField() { return include.getField(); }
51  
52    /** Returns a collection of all terms matched by this query.
53     * @deprecated use extractTerms instead
54     * @see #extractTerms(Set)
55     */
56    public Collection getTerms() { return include.getTerms(); }
57    
58    public void extractTerms(Set terms) { include.extractTerms(terms); }
59  
60    public String toString(String field) {
61      StringBuffer buffer = new StringBuffer();
62      buffer.append("spanNot(");
63      buffer.append(include.toString(field));
64      buffer.append(", ");
65      buffer.append(exclude.toString(field));
66      buffer.append(")");
67      buffer.append(ToStringUtils.boost(getBoost()));
68      return buffer.toString();
69    }
70  
71    public Object clone() {
72      SpanNotQuery spanNotQuery = new SpanNotQuery((SpanQuery)include.clone(),(SpanQuery) exclude.clone());
73      spanNotQuery.setBoost(getBoost());
74      return  spanNotQuery;
75    }
76  
77    public Spans getSpans(final IndexReader reader) throws IOException {
78      return new Spans() {
79          private Spans includeSpans = include.getSpans(reader);
80          private boolean moreInclude = true;
81  
82          private Spans excludeSpans = exclude.getSpans(reader);
83          private boolean moreExclude = excludeSpans.next();
84  
85          public boolean next() throws IOException {
86            if (moreInclude)                        // move to next include
87              moreInclude = includeSpans.next();
88  
89            while (moreInclude && moreExclude) {
90  
91              if (includeSpans.doc() > excludeSpans.doc()) // skip exclude
92                moreExclude = excludeSpans.skipTo(includeSpans.doc());
93  
94              while (moreExclude                    // while exclude is before
95                     && includeSpans.doc() == excludeSpans.doc()
96                     && excludeSpans.end() <= includeSpans.start()) {
97                moreExclude = excludeSpans.next();  // increment exclude
98              }
99  
100             if (!moreExclude                      // if no intersection
101                 || includeSpans.doc() != excludeSpans.doc()
102                 || includeSpans.end() <= excludeSpans.start())
103               break;                              // we found a match
104 
105             moreInclude = includeSpans.next();    // intersected: keep scanning
106           }
107           return moreInclude;
108         }
109 
110         public boolean skipTo(int target) throws IOException {
111           if (moreInclude)                        // skip include
112             moreInclude = includeSpans.skipTo(target);
113 
114           if (!moreInclude)
115             return false;
116 
117           if (moreExclude                         // skip exclude
118               && includeSpans.doc() > excludeSpans.doc())
119             moreExclude = excludeSpans.skipTo(includeSpans.doc());
120 
121           while (moreExclude                      // while exclude is before
122                  && includeSpans.doc() == excludeSpans.doc()
123                  && excludeSpans.end() <= includeSpans.start()) {
124             moreExclude = excludeSpans.next();    // increment exclude
125           }
126 
127           if (!moreExclude                      // if no intersection
128                 || includeSpans.doc() != excludeSpans.doc()
129                 || includeSpans.end() <= excludeSpans.start())
130             return true;                          // we found a match
131 
132           return next();                          // scan to next match
133         }
134 
135         public int doc() { return includeSpans.doc(); }
136         public int start() { return includeSpans.start(); }
137         public int end() { return includeSpans.end(); }
138 
139       // TODO: Remove warning after API has been finalized
140       public Collection/*<byte[]>*/ getPayload() throws IOException {
141         ArrayList result = null;
142         if (includeSpans.isPayloadAvailable()) {
143           result = new ArrayList(includeSpans.getPayload());
144         }
145         return result;
146       }
147 
148       // TODO: Remove warning after API has been finalized
149      public boolean isPayloadAvailable() {
150         return includeSpans.isPayloadAvailable();
151       }
152 
153       public String toString() {
154           return "spans(" + SpanNotQuery.this.toString() + ")";
155         }
156 
157       };
158   }
159 
160   public Query rewrite(IndexReader reader) throws IOException {
161     SpanNotQuery clone = null;
162 
163     SpanQuery rewrittenInclude = (SpanQuery) include.rewrite(reader);
164     if (rewrittenInclude != include) {
165       clone = (SpanNotQuery) this.clone();
166       clone.include = rewrittenInclude;
167     }
168     SpanQuery rewrittenExclude = (SpanQuery) exclude.rewrite(reader);
169     if (rewrittenExclude != exclude) {
170       if (clone == null) clone = (SpanNotQuery) this.clone();
171       clone.exclude = rewrittenExclude;
172     }
173 
174     if (clone != null) {
175       return clone;                        // some clauses rewrote
176     } else {
177       return this;                         // no clauses rewrote
178     }
179   }
180 
181     /** Returns true iff <code>o</code> is equal to this. */
182   public boolean equals(Object o) {
183     if (this == o) return true;
184     if (!(o instanceof SpanNotQuery)) return false;
185 
186     SpanNotQuery other = (SpanNotQuery)o;
187     return this.include.equals(other.include)
188             && this.exclude.equals(other.exclude)
189             && this.getBoost() == other.getBoost();
190   }
191 
192   public int hashCode() {
193     int h = include.hashCode();
194     h = (h<<1) | (h >>> 31);  // rotate left
195     h ^= exclude.hashCode();
196     h = (h<<1) | (h >>> 31);  // rotate left
197     h ^= Float.floatToRawIntBits(getBoost());
198     return h;
199   }
