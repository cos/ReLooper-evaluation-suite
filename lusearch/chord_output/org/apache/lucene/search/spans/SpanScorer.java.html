20  import java.io.IOException;
21  
22  import org.apache.lucene.search.Explanation;
23  import org.apache.lucene.search.Weight;
24  import org.apache.lucene.search.Scorer;
25  import org.apache.lucene.search.Similarity;
26  
27  /**
28   * Public for extension only.
29   */
30  public class SpanScorer extends Scorer {
31    protected Spans spans;
32    protected Weight weight;
33    protected byte[] norms;
34    protected float value;
35  
36    /** @deprecated not needed anymore */
37    protected boolean firstTime = true;
38    protected boolean more = true;
39  
40    protected int doc;
41    protected float freq;
42  
43    protected SpanScorer(Spans spans, Weight weight, Similarity similarity, byte[] norms)
44    throws IOException {
45      super(similarity);
46      this.spans = spans;
47      this.norms = norms;
48      this.weight = weight;
49      this.value = weight.getValue();
50      if (this.spans.next()) {
51        doc = -1;
52      } else {
53        doc = NO_MORE_DOCS;
54        more = false;
55      }
56    }
57  
58    /** @deprecated use {@link #nextDoc()} instead. */
59    public boolean next() throws IOException {
60      return nextDoc() != NO_MORE_DOCS;
61    }
62  
63    public int nextDoc() throws IOException {
64      if (!setFreqCurrentDoc()) {
65        doc = NO_MORE_DOCS;
66      }
67      return doc;
68    }
69  
70    /** @deprecated use {@link #advance(int)} instead. */
71    public boolean skipTo(int target) throws IOException {
72      return advance(target) != NO_MORE_DOCS;
73    }
74  
75    public int advance(int target) throws IOException {
76      if (!more) {
77        return doc = NO_MORE_DOCS;
78      }
79      if (spans.doc() < target) { // setFreqCurrentDoc() leaves spans.doc() ahead
80        more = spans.skipTo(target);
81      }
82      if (!setFreqCurrentDoc()) {
83        doc = NO_MORE_DOCS;
84      }
85      return doc;
86    }
87    
88    protected boolean setFreqCurrentDoc() throws IOException {
89      if (!more) {
90        return false;
91      }
92      doc = spans.doc();
93      freq = 0.0f;
94      do {
95        int matchLength = spans.end() - spans.start();
96        freq += getSimilarity().sloppyFreq(matchLength);
97        more = spans.next();
98      } while (more && (doc == spans.doc()));
99      return true;
100   }
101 
102   /** @deprecated use {@link #docID()} instead. */
103   public int doc() { return doc; }
104   
105   public int docID() { return doc; }
106 
107   public float score() throws IOException {
108     float raw = getSimilarity().tf(freq) * value; // raw score
109     return norms == null? raw : raw * Similarity.decodeNorm(norms[doc]); // normalize
110   }
111 
112   public Explanation explain(final int doc) throws IOException {
113     Explanation tfExplanation = new Explanation();
114 
115     int expDoc = advance(doc);
116 
117     float phraseFreq = (expDoc == doc) ? freq : 0.0f;
118     tfExplanation.setValue(getSimilarity().tf(phraseFreq));
119     tfExplanation.setDescription("tf(phraseFreq=" + phraseFreq + ")");
120 
121     return tfExplanation;
122   }
123 
124 }
