32  /** A Spans that is formed from the ordered subspans of a SpanNearQuery
33   * where the subspans do not overlap and have a maximum slop between them.
34   * <p>
35   * The formed spans only contains minimum slop matches.<br>
36   * The matching slop is computed from the distance(s) between
37   * the non overlapping matching Spans.<br>
38   * Successive matches are always formed from the successive Spans
39   * of the SpanNearQuery.
40   * <p>
41   * The formed spans may contain overlaps when the slop is at least 1.
42   * For example, when querying using
43   * <pre>t1 t2 t3</pre>
44   * with slop at least 1, the fragment:
45   * <pre>t1 t2 t1 t3 t2 t3</pre>
46   * matches twice:
47   * <pre>t1 t2 .. t3      </pre>
48   * <pre>      t1 .. t2 t3</pre>
49   *
50   *
51   * Expert:
52   * Only public for subclassing.  Most implementations should not need this class
53   */
54  public class NearSpansOrdered extends Spans {
55    private final int allowedSlop;
56    private boolean firstTime = true;
57    private boolean more = false;
58  
59    /** The spans in the same order as the SpanNearQuery */
60    private final Spans[] subSpans;
61  
62    /** Indicates that all subSpans have same doc() */
63    private boolean inSameDoc = false;
64  
65    private int matchDoc = -1;
66    private int matchStart = -1;
67    private int matchEnd = -1;
68    private List/*<byte[]>*/ matchPayload;
69  
70    private final Spans[] subSpansByDoc;
71    private final Comparator spanDocComparator = new Comparator() {
72      public int compare(Object o1, Object o2) {
73        return ((Spans)o1).doc() - ((Spans)o2).doc();
74      }
75    };
76    
77    private SpanNearQuery query;
78    private boolean collectPayloads = true;
79    
80    public NearSpansOrdered(SpanNearQuery spanNearQuery, IndexReader reader) throws IOException {
81      this(spanNearQuery, reader, true);
82    }
83  
84    public NearSpansOrdered(SpanNearQuery spanNearQuery, IndexReader reader, boolean collectPayloads)
85    throws IOException {
86      if (spanNearQuery.getClauses().length < 2) {
87        throw new IllegalArgumentException("Less than 2 clauses: "
88                                           + spanNearQuery);
89      }
90      this.collectPayloads = collectPayloads;
91      allowedSlop = spanNearQuery.getSlop();
92      SpanQuery[] clauses = spanNearQuery.getClauses();
93      subSpans = new Spans[clauses.length];
94      matchPayload = new LinkedList();
95      subSpansByDoc = new Spans[clauses.length];
96      for (int i = 0; i < clauses.length; i++) {
97        subSpans[i] = clauses[i].getSpans(reader);
98        subSpansByDoc[i] = subSpans[i]; // used in toSameDoc()
99      }
100     query = spanNearQuery; // kept for toString() only.
101   }

104   public int doc() { return matchDoc; }
105 
106   // inherit javadocs
107   public int start() { return matchStart; }
108 
109   // inherit javadocs
110   public int end() { return matchEnd; }
111   
112   public Spans[] getSubSpans() {
113       return subSpans;
114   }

116   // TODO: Remove warning after API has been finalized
117   // TODO: Would be nice to be able to lazy load payloads
118   public Collection/*<byte[]>*/ getPayload() throws IOException {
119     return matchPayload;
120   }

122   // TODO: Remove warning after API has been finalized
123  public boolean isPayloadAvailable() {
124     return matchPayload.isEmpty() == false;
125   }

127   // inherit javadocs
128   public boolean next() throws IOException {
129     if (firstTime) {
130       firstTime = false;
131       for (int i = 0; i < subSpans.length; i++) {
132         if (! subSpans[i].next()) {
133           more = false;
134           return false;
135         }
136       }
137       more = true;
138     }
139     if(collectPayloads) {
140       matchPayload.clear();
141     }
142     return advanceAfterOrdered();
143   }
144 
145   // inherit javadocs
146   public boolean skipTo(int target) throws IOException {
147     if (firstTime) {
148       firstTime = false;
149       for (int i = 0; i < subSpans.length; i++) {
150         if (! subSpans[i].skipTo(target)) {
151           more = false;
152           return false;
153         }
154       }
155       more = true;
156     } else if (more && (subSpans[0].doc() < target)) {
157       if (subSpans[0].skipTo(target)) {
158         inSameDoc = false;
159       } else {
160         more = false;
161         return false;
162       }
163     }
164     if(collectPayloads) {
165       matchPayload.clear();
166     }
167     return advanceAfterOrdered();
168   }
169   
170   /** Advances the subSpans to just after an ordered match with a minimum slop
171    * that is smaller than the slop allowed by the SpanNearQuery.
172    * @return true iff there is such a match.
173    */
174   private boolean advanceAfterOrdered() throws IOException {
175     while (more && (inSameDoc || toSameDoc())) {
176       if (stretchToOrder() && shrinkToAfterShortestMatch()) {
177         return true;
178       }
179     }
180     return false; // no more matches
181   }
182 
183 
184   /** Advance the subSpans to the same document */
185   private boolean toSameDoc() throws IOException {
186     Arrays.sort(subSpansByDoc, spanDocComparator);
187     int firstIndex = 0;
188     int maxDoc = subSpansByDoc[subSpansByDoc.length - 1].doc();
189     while (subSpansByDoc[firstIndex].doc() != maxDoc) {
190       if (! subSpansByDoc[firstIndex].skipTo(maxDoc)) {
191         more = false;
192         inSameDoc = false;
193         return false;
194       }
195       maxDoc = subSpansByDoc[firstIndex].doc();
196       if (++firstIndex == subSpansByDoc.length) {
197         firstIndex = 0;
198       }
199     }
200     for (int i = 0; i < subSpansByDoc.length; i++) {
201       assert (subSpansByDoc[i].doc() == maxDoc)
202              : " NearSpansOrdered.toSameDoc() spans " + subSpansByDoc[0]
203                                  + "\n at doc " + subSpansByDoc[i].doc()
204                                  + ", but should be at " + maxDoc;
205     }
206     inSameDoc = true;
207     return true;
208   }
209   
210   /** Check whether two Spans in the same document are ordered.
211    * @param spans1 
212    * @param spans2 
213    * @return true iff spans1 starts before spans2
214    *              or the spans start at the same position,
215    *              and spans1 ends before spans2.
216    */
217   static final boolean docSpansOrdered(Spans spans1, Spans spans2) {
218     assert spans1.doc() == spans2.doc() : "doc1 " + spans1.doc() + " != doc2 " + spans2.doc();
219     int start1 = spans1.start();
220     int start2 = spans2.start();
221     /* Do not call docSpansOrdered(int,int,int,int) to avoid invoking .end() : */
222     return (start1 == start2) ? (spans1.end() < spans2.end()) : (start1 < start2);
223   }
224 
225   /** Like {@link #docSpansOrdered(Spans,Spans)}, but use the spans
226    * starts and ends as parameters.
227    */
228   private static final boolean docSpansOrdered(int start1, int end1, int start2, int end2) {
229     return (start1 == start2) ? (end1 < end2) : (start1 < start2);
230   }
231 
232   /** Order the subSpans within the same document by advancing all later spans
233    * after the previous one.
234    */
235   private boolean stretchToOrder() throws IOException {
236     matchDoc = subSpans[0].doc();
237     for (int i = 1; inSameDoc && (i < subSpans.length); i++) {
238       while (! docSpansOrdered(subSpans[i-1], subSpans[i])) {
239         if (! subSpans[i].next()) {
240           inSameDoc = false;
241           more = false;
242           break;
243         } else if (matchDoc != subSpans[i].doc()) {
244           inSameDoc = false;
245           break;
246         }
247       }
248     }
249     return inSameDoc;
250   }
251 
252   /** The subSpans are ordered in the same doc, so there is a possible match.
253    * Compute the slop while making the match as short as possible by advancing
254    * all subSpans except the last one in reverse order.
255    */
256   private boolean shrinkToAfterShortestMatch() throws IOException {
257     matchStart = subSpans[subSpans.length - 1].start();
258     matchEnd = subSpans[subSpans.length - 1].end();
259     Set possibleMatchPayloads = new HashSet();
260     if (subSpans[subSpans.length - 1].isPayloadAvailable()) {
261       possibleMatchPayloads.addAll(subSpans[subSpans.length - 1].getPayload());
262     }
263 
264     Collection possiblePayload = null;
265     
266     int matchSlop = 0;
267     int lastStart = matchStart;
268     int lastEnd = matchEnd;
269     for (int i = subSpans.length - 2; i >= 0; i--) {
270       Spans prevSpans = subSpans[i];
271       if (collectPayloads && prevSpans.isPayloadAvailable()) {
272         Collection payload = prevSpans.getPayload();
273         possiblePayload = new ArrayList(payload.size());
274         possiblePayload.addAll(payload);
275       }
276       
277       int prevStart = prevSpans.start();
278       int prevEnd = prevSpans.end();
279       while (true) { // Advance prevSpans until after (lastStart, lastEnd)
280         if (! prevSpans.next()) {
281           inSameDoc = false;
282           more = false;
283           break; // Check remaining subSpans for final match.
284         } else if (matchDoc != prevSpans.doc()) {
285           inSameDoc = false; // The last subSpans is not advanced here.
286           break; // Check remaining subSpans for last match in this document.
287         } else {
288           int ppStart = prevSpans.start();
289           int ppEnd = prevSpans.end(); // Cannot avoid invoking .end()
290           if (! docSpansOrdered(ppStart, ppEnd, lastStart, lastEnd)) {
291             break; // Check remaining subSpans.
292           } else { // prevSpans still before (lastStart, lastEnd)
293             prevStart = ppStart;
294             prevEnd = ppEnd;
295             if (collectPayloads && prevSpans.isPayloadAvailable()) {
296               Collection payload = prevSpans.getPayload();
297               possiblePayload = new ArrayList(payload.size());
298               possiblePayload.addAll(payload);
299             }
300           }
301         }
302       }
303 
304       if (collectPayloads && possiblePayload != null) {
305         possibleMatchPayloads.addAll(possiblePayload);
306       }
307       
308       assert prevStart <= matchStart;
309       if (matchStart > prevEnd) { // Only non overlapping spans add to slop.
310         matchSlop += (matchStart - prevEnd);
311       }
312 
313       /* Do not break on (matchSlop > allowedSlop) here to make sure
314        * that subSpans[0] is advanced after the match, if any.
315        */
316       matchStart = prevStart;
317       lastStart = prevStart;
318       lastEnd = prevEnd;
319     }
320     
321     boolean match = matchSlop <= allowedSlop;
322     
323     if(collectPayloads && match && possibleMatchPayloads.size() > 0) {
324       matchPayload.addAll(possibleMatchPayloads);
325     }
326 
327     return match; // ordered and allowed slop
328   }

330   public String toString() {
331     return getClass().getName() + "("+query.toString()+")@"+
332       (firstTime?"START":(more?(doc()+":"+start()+"-"+end()):"END"));
333   }
