20  import java.io.IOException;
21  import java.util.Collection;
22  import java.util.Set;
23  
24  import org.apache.lucene.index.IndexReader;
25  import org.apache.lucene.search.Query;
26  import org.apache.lucene.search.Weight;
27  import org.apache.lucene.search.Searcher;
28  import org.apache.lucene.search.Similarity;
29  import org.apache.lucene.util.ToStringUtils;

31  /**
32   * <p>Wrapper to allow {@link SpanQuery} objects participate in composite 
33   * single-field SpanQueries by 'lying' about their search field. That is, 
34   * the masked SpanQuery will function as normal, 
35   * but {@link SpanQuery#getField()} simply hands back the value supplied 
36   * in this class's constructor.</p>
37   * 
38   * <p>This can be used to support Queries like {@link SpanNearQuery} or 
39   * {@link SpanOrQuery} across different fields, which is not ordinarily 
40   * permitted.</p>
41   * 
42   * <p>This can be useful for denormalized relational data: for example, when 
43   * indexing a document with conceptually many 'children': </p>
44   * 
45   * <pre>
46   *  teacherid: 1
47   *  studentfirstname: james
48   *  studentsurname: jones
49   *  
50   *  teacherid: 2
51   *  studenfirstname: james
52   *  studentsurname: smith
53   *  studentfirstname: sally
54   *  studentsurname: jones
55   * </pre>
56   * 
57   * <p>a SpanNearQuery with a slop of 0 can be applied across two 
58   * {@link SpanTermQuery} objects as follows:
59   * <pre>
60   *    SpanQuery q1  = new SpanTermQuery(new Term("studentfirstname", "james"));
61   *    SpanQuery q2  = new SpanTermQuery(new Term("studentsurname", "jones"));
62   *    SpanQuery q2m new FieldMaskingSpanQuery(q2, "studentfirstname");
63   *    Query q = new SpanNearQuery(new SpanQuery[]{q1, q2m}, -1, false);
64   * </pre>
65   * to search for 'studentfirstname:james studentsurname:jones' and find 
66   * teacherid 1 without matching teacherid 2 (which has a 'james' in position 0 
67   * and 'jones' in position 1). </p>
68   * 
69   * <p>Note: as {@link #getField()} returns the masked field, scoring will be 
70   * done using the norms of the field name supplied. This may lead to unexpected
71   * scoring behaviour.</p>
72   */
73  public class FieldMaskingSpanQuery extends SpanQuery {
74    private SpanQuery maskedQuery;
75    private String field;
76      
77    public FieldMaskingSpanQuery(SpanQuery maskedQuery, String maskedField) {
78      this.maskedQuery = maskedQuery;
79      this.field = maskedField;
80    }
81  
82    public String getField() {
83      return field;
84    }
85  
86    public SpanQuery getMaskedQuery() {
87      return maskedQuery;
88    }
89  
90    // :NOTE: getBoost and setBoost are not proxied to the maskedQuery
91    // ...this is done to be more consistent with things like SpanFirstQuery
92    
93    public Spans getSpans(IndexReader reader) throws IOException {
94      return maskedQuery.getSpans(reader);
95    }
96  
97    /** @deprecated use {@link #extractTerms(Set)} instead. */
98    public Collection getTerms() {
99      return maskedQuery.getTerms();
100   }
101   
102   public void extractTerms(Set terms) {
103     maskedQuery.extractTerms(terms);
104   }  
105 
106   public Weight createWeight(Searcher searcher) throws IOException {
107     return maskedQuery.createWeight(searcher);
108   }
109 
110   public Similarity getSimilarity(Searcher searcher) {
111     return maskedQuery.getSimilarity(searcher);
112   }
113 
114   public Query rewrite(IndexReader reader) throws IOException {
115     FieldMaskingSpanQuery clone = null;
116 
117     SpanQuery rewritten = (SpanQuery) maskedQuery.rewrite(reader);
118     if (rewritten != maskedQuery) {
119       clone = (FieldMaskingSpanQuery) this.clone();
120       clone.maskedQuery = rewritten;
121     }
122 
123     if (clone != null) {
124       return clone;
125     } else {
126       return this;
127     }
128   }
129 
130   public String toString(String field) {
131     StringBuffer buffer = new StringBuffer();
132     buffer.append("mask(");
133     buffer.append(maskedQuery.toString(field));
134     buffer.append(")");
135     buffer.append(ToStringUtils.boost(getBoost()));
136     buffer.append(" as ");
137     buffer.append(this.field);
138     return buffer.toString();
139   }
140   
141   public boolean equals(Object o) {
142     if (!(o instanceof FieldMaskingSpanQuery))
143       return false;
144     FieldMaskingSpanQuery other = (FieldMaskingSpanQuery) o;
145     return (this.getField().equals(other.getField())
146             && (this.getBoost() == other.getBoost())
147             && this.getMaskedQuery().equals(other.getMaskedQuery()));
148 
149   }
150   
151   public int hashCode() {
152     return getMaskedQuery().hashCode()
153       ^ getField().hashCode()
154       ^ Float.floatToRawIntBits(getBoost());
155   }
156 }
