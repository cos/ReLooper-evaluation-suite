20  import java.io.IOException;
21  import java.util.Set;
22  import java.util.ArrayList;
23  
24  import org.apache.lucene.index.Term;
25  import org.apache.lucene.index.TermPositions;

30  /** A Query that matches documents containing a particular sequence of terms.
31   * A PhraseQuery is built by QueryParser for input like <code>"new york"</code>.
32   * 
33   * <p>This query may be combined with other terms or queries with a {@link BooleanQuery}.
34   */
35  public class PhraseQuery extends Query {
36    private String field;
37    private ArrayList terms = new ArrayList(4);
38    private ArrayList positions = new ArrayList(4);
39    private int maxPosition = 0;
40    private int slop = 0;
41  
42    /** Constructs an empty phrase query. */
43    public PhraseQuery() {}
44  
45    /** Sets the number of other words permitted between words in query phrase.
46      If zero, then this is an exact phrase search.  For larger values this works
47      like a <code>WITHIN</code> or <code>NEAR</code> operator.
48  
49      <p>The slop is in fact an edit-distance, where the units correspond to
50      moves of terms in the query phrase out of position.  For example, to switch
51      the order of two words requires two moves (the first move places the words
52      atop one another), so to permit re-orderings of phrases, the slop must be
53      at least two.
54  
55      <p>More exact matches are scored higher than sloppier matches, thus search
56      results are sorted by exactness.
57  
58      <p>The slop is zero by default, requiring exact matches.*/
59    public void setSlop(int s) { slop = s; }
60    /** Returns the slop.  See setSlop(). */
61    public int getSlop() { return slop; }
62  
63    /**
64     * Adds a term to the end of the query phrase.
65     * The relative position of the term is the one immediately after the last term added.
66     */
67    public void add(Term term) {
68      int position = 0;
69      if(positions.size() > 0)
70          position = ((Integer) positions.get(positions.size()-1)).intValue() + 1;
71  
72      add(term, position);
73    }
74  
75    /**
76     * Adds a term to the end of the query phrase.
77     * The relative position of the term within the phrase is specified explicitly.
78     * This allows e.g. phrases with more than one term at the same position
79     * or phrases with gaps (e.g. in connection with stopwords).
80     * 
81     * @param term
82     * @param position
83     */
84    public void add(Term term, int position) {
85        if (terms.size() == 0)
86            field = term.field();
87        else if (term.field() != field)
88            throw new IllegalArgumentException("All phrase terms must be in the same field: " + term);
89  
90        terms.add(term);
91        positions.add(new Integer(position));
92        if (position > maxPosition) maxPosition = position;
93    }
94  
95    /** Returns the set of terms in this phrase. */
96    public Term[] getTerms() {
97      return (Term[])terms.toArray(new Term[0]);
98    }
99  
100   /**
101    * Returns the relative positions of terms in this phrase.
102    */
103   public int[] getPositions() {
104       int[] result = new int[positions.size()];
105       for(int i = 0; i < positions.size(); i++)
106           result[i] = ((Integer) positions.get(i)).intValue();
107       return result;
108   }
109 
110   private class PhraseWeight extends Weight {
111     private Similarity similarity;
112     private float value;
113     private float idf;
114     private float queryNorm;
115     private float queryWeight;
116     private IDFExplanation idfExp;
117 
118     public PhraseWeight(Searcher searcher)
119       throws IOException {
120       this.similarity = getSimilarity(searcher);
121 
122       idfExp = similarity.idfExplain(terms, searcher);
123       idf = idfExp.getIdf();
124     }
125 
126     public String toString() { return "weight(" + PhraseQuery.this + ")"; }
127 
128     public Query getQuery() { return PhraseQuery.this; }
129     public float getValue() { return value; }
130 
131     public float sumOfSquaredWeights() {
132       queryWeight = idf * getBoost();             // compute query weight
133       return queryWeight * queryWeight;           // square it
134     }
135 
136     public void normalize(float queryNorm) {
137       this.queryNorm = queryNorm;
138       queryWeight *= queryNorm;                   // normalize query weight
139       value = queryWeight * idf;                  // idf for document 
140     }
141 
142     public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer) throws IOException {
143       if (terms.size() == 0)              // optimize zero-term case
144         return null;
145 
146       TermPositions[] tps = new TermPositions[terms.size()];
147       for (int i = 0; i < terms.size(); i++) {
148         TermPositions p = reader.termPositions((Term)terms.get(i));
149         if (p == null)
150           return null;
151         tps[i] = p;
152       }
153 
154       if (slop == 0)                  // optimize exact case
155         return new ExactPhraseScorer(this, tps, getPositions(), similarity,
156                                      reader.norms(field));
157       else
158         return
159           new SloppyPhraseScorer(this, tps, getPositions(), similarity, slop,
160                                  reader.norms(field));
161 
162     }
163 
164     public Explanation explain(IndexReader reader, int doc)
165       throws IOException {
166 
167       Explanation result = new Explanation();
168       result.setDescription("weight("+getQuery()+" in "+doc+"), product of:");
169 
170       StringBuffer docFreqs = new StringBuffer();
171       StringBuffer query = new StringBuffer();
172       query.append('\"');
173       docFreqs.append(idfExp.explain());
174       for (int i = 0; i < terms.size(); i++) {
175         if (i != 0) {
176           query.append(" ");
177         }
178 
179         Term term = (Term)terms.get(i);
180 
181         query.append(term.text());
182       }
183       query.append('\"');
184 
185       Explanation idfExpl =
186         new Explanation(idf, "idf(" + field + ":" + docFreqs + ")");
187 
188       // explain query weight
189       Explanation queryExpl = new Explanation();
190       queryExpl.setDescription("queryWeight(" + getQuery() + "), product of:");
191 
192       Explanation boostExpl = new Explanation(getBoost(), "boost");
193       if (getBoost() != 1.0f)
194         queryExpl.addDetail(boostExpl);
195       queryExpl.addDetail(idfExpl);
196 
197       Explanation queryNormExpl = new Explanation(queryNorm,"queryNorm");
198       queryExpl.addDetail(queryNormExpl);
199 
200       queryExpl.setValue(boostExpl.getValue() *
201                          idfExpl.getValue() *
202                          queryNormExpl.getValue());
203 
204       result.addDetail(queryExpl);
205 
206       // explain field weight
207       Explanation fieldExpl = new Explanation();
208       fieldExpl.setDescription("fieldWeight("+field+":"+query+" in "+doc+
209                                "), product of:");
210 
211       Scorer scorer = scorer(reader, true, false);
212       if (scorer == null) {
213         return new Explanation(0.0f, "no matching docs");
214       }
215       Explanation tfExpl = scorer.explain(doc);
216       fieldExpl.addDetail(tfExpl);
217       fieldExpl.addDetail(idfExpl);
218 
219       Explanation fieldNormExpl = new Explanation();
220       byte[] fieldNorms = reader.norms(field);
221       float fieldNorm =
222         fieldNorms!=null ? Similarity.decodeNorm(fieldNorms[doc]) : 1.0f;
223       fieldNormExpl.setValue(fieldNorm);
224       fieldNormExpl.setDescription("fieldNorm(field="+field+", doc="+doc+")");
225       fieldExpl.addDetail(fieldNormExpl);
226 
227       fieldExpl.setValue(tfExpl.getValue() *
228                          idfExpl.getValue() *
229                          fieldNormExpl.getValue());
230 
231       result.addDetail(fieldExpl);
232 
233       // combine them
234       result.setValue(queryExpl.getValue() * fieldExpl.getValue());
235 
236       if (queryExpl.getValue() == 1.0f)
237         return fieldExpl;
238 
239       return result;
240     }
241   }
242 
243   public Weight createWeight(Searcher searcher) throws IOException {
244     if (terms.size() == 1) {              // optimize one-term case
245       Term term = (Term)terms.get(0);
246       Query termQuery = new TermQuery(term);
247       termQuery.setBoost(getBoost());
248       return termQuery.createWeight(searcher);
249     }
250     return new PhraseWeight(searcher);
251   }
252 
253   /**
254    * @see org.apache.lucene.search.Query#extractTerms(java.util.Set)
255    */
256   public void extractTerms(Set queryTerms) {
257     queryTerms.addAll(terms);
258   }

260   /** Prints a user-readable version of this query. */
261   public String toString(String f) {
262     StringBuffer buffer = new StringBuffer();
263     if (field != null && !field.equals(f)) {
264       buffer.append(field);
265       buffer.append(":");
266     }
267 
268     buffer.append("\"");
269     String[] pieces = new String[maxPosition + 1];
270     for (int i = 0; i < terms.size(); i++) {
271       int pos = ((Integer)positions.get(i)).intValue();
272       String s = pieces[pos];
273       if (s == null) {
274         s = ((Term)terms.get(i)).text();
275       } else {
276         s = s + "|" + ((Term)terms.get(i)).text();
277       }
278       pieces[pos] = s;
279     }
280     for (int i = 0; i < pieces.length; i++) {
281       if (i > 0) {
282         buffer.append(' ');
283       }
284       String s = pieces[i];
285       if (s == null) {
286         buffer.append('?');
287       } else {
288         buffer.append(s);
289       }
290     }
291     buffer.append("\"");
292 
293     if (slop != 0) {
294       buffer.append("~");
295       buffer.append(slop);
296     }
297 
298     buffer.append(ToStringUtils.boost(getBoost()));
299 
300     return buffer.toString();
301   }

303   /** Returns true iff <code>o</code> is equal to this. */
304   public boolean equals(Object o) {
305     if (!(o instanceof PhraseQuery))
306       return false;
307     PhraseQuery other = (PhraseQuery)o;
308     return (this.getBoost() == other.getBoost())
309       && (this.slop == other.slop)
310       &&  this.terms.equals(other.terms)
311       && this.positions.equals(other.positions);
312   }
313 
314   /** Returns a hash code value for this object.*/
315   public int hashCode() {
316     return Float.floatToIntBits(getBoost())
317       ^ slop
318       ^ terms.hashCode()
319       ^ positions.hashCode();
320   }
