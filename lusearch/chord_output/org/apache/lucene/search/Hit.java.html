18  package org.apache.lucene.search;
19  
20  import java.io.IOException;
21  
22  import org.apache.lucene.document.Document;
23  import org.apache.lucene.index.CorruptIndexException;
24  
25  /**
26   * Wrapper used by {@link HitIterator} to provide a lazily loaded hit
27   * from {@link Hits}.
28   *
29   * @deprecated Use {@link TopScoreDocCollector} and {@link TopDocs} instead. Hits will be removed in Lucene 3.0.
30   */
31  public class Hit implements java.io.Serializable {
32  
33    private Document doc = null;
34  
35    private boolean resolved = false;
36  
37    private Hits hits = null;
38    private int hitNumber;
39  
40    /**
41     * Constructed from {@link HitIterator}
42     * @param hits Hits returned from a search
43     * @param hitNumber Hit index in Hits
44     */
45    Hit(Hits hits, int hitNumber) {
46      this.hits = hits;
47      this.hitNumber = hitNumber;
48    }
49  
50    /**
51     * Returns document for this hit.
52     *
53     * @see Hits#doc(int)
54     * @throws CorruptIndexException if the index is corrupt
55     * @throws IOException if there is a low-level IO error
56     */
57    public Document getDocument() throws CorruptIndexException, IOException {
58      if (!resolved) fetchTheHit();
59      return doc;
60    }
61  
62    /**
63     * Returns score for this hit.
64     *
65     * @see Hits#score(int)
66     */
67    public float getScore() throws IOException {
68      return hits.score(hitNumber);
69    }
70  
71    /**
72     * Returns id for this hit.
73     *
74     * @see Hits#id(int)
75     */
76    public int getId() throws IOException {
77      return hits.id(hitNumber);
78    }
79  
80    private void fetchTheHit() throws CorruptIndexException, IOException {
81      doc = hits.doc(hitNumber);
82      resolved = true;
83    }
84  
85    // provide some of the Document style interface (the simple stuff)
86  
87    /**
88     * Returns the boost factor for this hit on any field of the underlying document.
89     *
90     * @see Document#getBoost()
91     * @throws CorruptIndexException if the index is corrupt
92     * @throws IOException if there is a low-level IO error
93     */
94    public float getBoost() throws CorruptIndexException, IOException {
95      return getDocument().getBoost();
96    }
97  
98    /**
99     * Returns the string value of the field with the given name if any exist in
100    * this document, or null.  If multiple fields exist with this name, this
101    * method returns the first value added. If only binary fields with this name
102    * exist, returns null.
103    *
104    * @see Document#get(String)
105    * @throws CorruptIndexException if the index is corrupt
106    * @throws IOException if there is a low-level IO error
107    */
108   public String get(String name) throws CorruptIndexException, IOException {
109     return getDocument().get(name);
110   }
111 
112   /**
113    * Prints the parameters to be used to discover the promised result.
114    */
115   public String toString() {
116     StringBuffer buffer = new StringBuffer();
117     buffer.append("Hit<");
118     buffer.append(hits.toString());
119     buffer.append(" [");
120     buffer.append(hitNumber);
121     buffer.append("] ");
122     if (resolved) {
123         buffer.append("resolved");
124     } else {
125         buffer.append("unresolved");
126     }
127     buffer.append(">");
128     return buffer.toString();
129   }
130 
131 
132 }
