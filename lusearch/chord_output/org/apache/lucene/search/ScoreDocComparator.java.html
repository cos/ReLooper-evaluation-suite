21  /**
22   * Expert: Compares two ScoreDoc objects for sorting.
23   *
24   * <p>Created: Feb 3, 2004 9:00:16 AM 
25   *
26   * @since   lucene 1.4
27   * @version $Id: ScoreDocComparator.java 738219 2009-01-27 20:15:21Z mikemccand $
28   * @deprecated use {@link FieldComparator}
29   */
30  public interface ScoreDocComparator {
31  
32      /** Special comparator for sorting hits according to computed relevance (document score). */
33      static final ScoreDocComparator RELEVANCE = new ScoreDocComparator() {
34          public int compare (ScoreDoc i, ScoreDoc j) {
35              if (i.score > j.score) return -1;
36              if (i.score < j.score) return 1;
37              return 0;
38          }
39          public Comparable sortValue (ScoreDoc i) {
40              return new Float (i.score);
41          }
42          public int sortType() {
43              return SortField.SCORE;
44          }
45      };
46  
47      /** Special comparator for sorting hits according to index order (document number). */
48      static final ScoreDocComparator INDEXORDER = new ScoreDocComparator() {
49          public int compare (ScoreDoc i, ScoreDoc j) {
50              if (i.doc < j.doc) return -1;
51              if (i.doc > j.doc) return 1;
52              return 0;
53          }
54          public Comparable sortValue (ScoreDoc i) {
55              return new Integer (i.doc);
56          }
57          public int sortType() {
58              return SortField.DOC;
59          }
60      };
61  
62      /**
63       * Compares two ScoreDoc objects and returns a result indicating their
64       * sort order.
65       * @param i First ScoreDoc
66       * @param j Second ScoreDoc
67       * @return a negative integer if <code>i</code> should come before <code>j</code><br>
68       *         a positive integer if <code>i</code> should come after <code>j</code><br>
69       *         <code>0</code> if they are equal
70       * @see java.util.Comparator
71       */
72      int compare (ScoreDoc i, ScoreDoc j);
73  
74      /**
75       * Returns the value used to sort the given document.  The
76       * object returned must implement the java.io.Serializable
77       * interface.  This is used by multisearchers to determine how
78       * to collate results from their searchers.
79       * @see FieldDoc
80       * @param i Document
81       * @return Serializable object
82       */
83      Comparable sortValue (ScoreDoc i);
84  
85      /**
86       * Returns the type of sort.  Should return <code>SortField.SCORE</code>,
87       * <code>SortField.DOC</code>, <code>SortField.STRING</code>,
88       * <code>SortField.INTEGER</code>, <code>SortField.FLOAT</code> or
89       * <code>SortField.CUSTOM</code>.  It is not valid to return
90       * <code>SortField.AUTO</code>.
91       * This is used by multisearchers to determine how to collate results
92       * from their searchers.
93       * @return One of the constants in SortField.
94       * @see SortField
95       */
96      int sortType();
97  }
