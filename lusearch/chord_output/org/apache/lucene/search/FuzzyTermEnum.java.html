20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.index.Term;

23  import java.io.IOException;

25  /** Subclass of FilteredTermEnum for enumerating all terms that are similar
26   * to the specified filter term.
27   *
28   * <p>Term enumerations are always ordered by Term.compareTo().  Each term in
29   * the enumeration is greater than all that precede it.
30   */
31  public final class FuzzyTermEnum extends FilteredTermEnum {
32  
33    /* This should be somewhere around the average long word.
34     * If it is longer, we waste time and space. If it is shorter, we waste a
35     * little bit of time growing the array as we encounter longer words.
36     */
37    private static final int TYPICAL_LONGEST_WORD_IN_INDEX = 19;
38  
39    /* Allows us save time required to create a new array
40     * every time similarity is called.
41     */
42    private int[][] d;
43  
44    private float similarity;
45    private boolean endEnum = false;
46  
47    private Term searchTerm = null;
48    private final String field;
49    private final String text;
50    private final String prefix;
51  
52    private final float minimumSimilarity;
53    private final float scale_factor;
54    private final int[] maxDistances = new int[TYPICAL_LONGEST_WORD_IN_INDEX];
55  
56    /**
57     * Creates a FuzzyTermEnum with an empty prefix and a minSimilarity of 0.5f.
58     * <p>
59     * After calling the constructor the enumeration is already pointing to the first 
60     * valid term if such a term exists. 
61     * 
62     * @param reader
63     * @param term
64     * @throws IOException
65     * @see #FuzzyTermEnum(IndexReader, Term, float, int)
66     */
67    public FuzzyTermEnum(IndexReader reader, Term term) throws IOException {
68      this(reader, term, FuzzyQuery.defaultMinSimilarity, FuzzyQuery.defaultPrefixLength);
69    }
70      
71    /**
72     * Creates a FuzzyTermEnum with an empty prefix.
73     * <p>
74     * After calling the constructor the enumeration is already pointing to the first 
75     * valid term if such a term exists. 
76     * 
77     * @param reader
78     * @param term
79     * @param minSimilarity
80     * @throws IOException
81     * @see #FuzzyTermEnum(IndexReader, Term, float, int)
82     */
83    public FuzzyTermEnum(IndexReader reader, Term term, float minSimilarity) throws IOException {
84      this(reader, term, minSimilarity, FuzzyQuery.defaultPrefixLength);
85    }
86      
87    /**
88     * Constructor for enumeration of all terms from specified <code>reader</code> which share a prefix of
89     * length <code>prefixLength</code> with <code>term</code> and which have a fuzzy similarity &gt;
90     * <code>minSimilarity</code>.
91     * <p>
92     * After calling the constructor the enumeration is already pointing to the first 
93     * valid term if such a term exists. 
94     * 
95     * @param reader Delivers terms.
96     * @param term Pattern term.
97     * @param minSimilarity Minimum required similarity for terms from the reader. Default value is 0.5f.
98     * @param prefixLength Length of required common prefix. Default value is 0.
99     * @throws IOException
100    */
101   public FuzzyTermEnum(IndexReader reader, Term term, final float minSimilarity, final int prefixLength) throws IOException {
102     super();
103     
104     if (minSimilarity >= 1.0f)
105       throw new IllegalArgumentException("minimumSimilarity cannot be greater than or equal to 1");
106     else if (minSimilarity < 0.0f)
107       throw new IllegalArgumentException("minimumSimilarity cannot be less than 0");
108     if(prefixLength < 0)
109       throw new IllegalArgumentException("prefixLength cannot be less than 0");
110 
111     this.minimumSimilarity = minSimilarity;
112     this.scale_factor = 1.0f / (1.0f - minimumSimilarity);
113     this.searchTerm = term;
114     this.field = searchTerm.field();
115 
116     //The prefix could be longer than the word.
117     //It's kind of silly though.  It means we must match the entire word.
118     final int fullSearchTermLength = searchTerm.text().length();
119     final int realPrefixLength = prefixLength > fullSearchTermLength ? fullSearchTermLength : prefixLength;
120 
121     this.text = searchTerm.text().substring(realPrefixLength);
122     this.prefix = searchTerm.text().substring(0, realPrefixLength);
123 
124     initializeMaxDistances();
125     this.d = initDistanceArray();
126 
127     setEnum(reader.terms(new Term(searchTerm.field(), prefix)));
128   }
129 
130   /**
131    * The termCompare method in FuzzyTermEnum uses Levenshtein distance to 
132    * calculate the distance between the given term and the comparing term. 
133    */
134   protected final boolean termCompare(Term term) {
135     if (field == term.field() && term.text().startsWith(prefix)) {
136         final String target = term.text().substring(prefix.length());
137         this.similarity = similarity(target);
138         return (similarity > minimumSimilarity);
139     }
140     endEnum = true;
141     return false;
142   }
143   
144   public final float difference() {
145     return (float)((similarity - minimumSimilarity) * scale_factor);
146   }
147   
148   public final boolean endEnum() {
149     return endEnum;
150   }

152   /******************************
153    * Compute Levenshtein distance
154    ******************************/
155   
156   /**
157    * Finds and returns the smallest of three integers 
158    */
159   private static final int min(int a, int b, int c) {
160     final int t = (a < b) ? a : b;
161     return (t < c) ? t : c;
162   }
163 
164   private final int[][] initDistanceArray(){
165     return new int[this.text.length() + 1][TYPICAL_LONGEST_WORD_IN_INDEX];
166   }
167 
168   /**
169    * <p>Similarity returns a number that is 1.0f or less (including negative numbers)
170    * based on how similar the Term is compared to a target term.  It returns
171    * exactly 0.0f when
172    * <pre>
173    *    editDistance &lt; maximumEditDistance</pre>
174    * Otherwise it returns:
175    * <pre>
176    *    1 - (editDistance / length)</pre>
177    * where length is the length of the shortest term (text or target) including a
178    * prefix that are identical and editDistance is the Levenshtein distance for
179    * the two words.</p>
180    *
181    * <p>Embedded within this algorithm is a fail-fast Levenshtein distance
182    * algorithm.  The fail-fast algorithm differs from the standard Levenshtein
183    * distance algorithm in that it is aborted if it is discovered that the
184    * minimum distance between the words is greater than some threshold.
185    *
186    * <p>To calculate the maximum distance threshold we use the following formula:
187    * <pre>
188    *     (1 - minimumSimilarity) * length</pre>
189    * where length is the shortest term including any prefix that is not part of the
190    * similarity comparison.  This formula was derived by solving for what maximum value
191    * of distance returns false for the following statements:
192    * <pre>
193    *   similarity = 1 - ((float)distance / (float) (prefixLength + Math.min(textlen, targetlen)));
194    *   return (similarity > minimumSimilarity);</pre>
195    * where distance is the Levenshtein distance for the two words.
196    * </p>
197    * <p>Levenshtein distance (also known as edit distance) is a measure of similarity
198    * between two strings where the distance is measured as the number of character
199    * deletions, insertions or substitutions required to transform one string to
200    * the other string.
201    * @param target the target word or phrase
202    * @return the similarity,  0.0 or less indicates that it matches less than the required
203    * threshold and 1.0 indicates that the text and target are identical
204    */
205   private float similarity(final String target) {
206     final int m = target.length();
207     final int n = text.length();
208     if (n == 0)  {
209       //we don't have anything to compare.  That means if we just add
210       //the letters for m we get the new word
211       return prefix.length() == 0 ? 0.0f : 1.0f - ((float) m / prefix.length());
212     }
213     if (m == 0) {
214       return prefix.length() == 0 ? 0.0f : 1.0f - ((float) n / prefix.length());
215     }
216 
217     final int maxDistance = getMaxDistance(m);
218 
219     if (maxDistance < Math.abs(m-n)) {
220       //just adding the characters of m to n or vice-versa results in
221       //too many edits
222       //for example "pre" length is 3 and "prefixes" length is 8.  We can see that
223       //given this optimal circumstance, the edit distance cannot be less than 5.
224       //which is 8-3 or more precisely Math.abs(3-8).
225       //if our maximum edit distance is 4, then we can discard this word
226       //without looking at it.
227       return 0.0f;
228     }
229 
230     //let's make sure we have enough room in our array to do the distance calculations.
231     if (d[0].length <= m) {
232       growDistanceArray(m);
233     }
234 
235     // init matrix d
236     for (int i = 0; i <= n; i++) d[i][0] = i;
237     for (int j = 0; j <= m; j++) d[0][j] = j;
238     
239     // start computing edit distance
240     for (int i = 1; i <= n; i++) {
241       int bestPossibleEditDistance = m;
242       final char s_i = text.charAt(i - 1);
243       for (int j = 1; j <= m; j++) {
244         if (s_i != target.charAt(j-1)) {
245             d[i][j] = min(d[i-1][j], d[i][j-1], d[i-1][j-1])+1;
246         }
247         else {
248           d[i][j] = min(d[i-1][j]+1, d[i][j-1]+1, d[i-1][j-1]);
249         }
250         bestPossibleEditDistance = Math.min(bestPossibleEditDistance, d[i][j]);
251       }
252 
253       //After calculating row i, the best possible edit distance
254       //can be found by found by finding the smallest value in a given column.
255       //If the bestPossibleEditDistance is greater than the max distance, abort.
256 
257       if (i > maxDistance && bestPossibleEditDistance > maxDistance) {  //equal is okay, but not greater
258         //the closest the target can be to the text is just too far away.
259         //this target is leaving the party early.
260         return 0.0f;
261       }
262     }
263 
264     // this will return less than 0.0 when the edit distance is
265     // greater than the number of characters in the shorter word.
266     // but this was the formula that was previously used in FuzzyTermEnum,
267     // so it has not been changed (even though minimumSimilarity must be
268     // greater than 0.0)
269     return 1.0f - ((float)d[n][m] / (float) (prefix.length() + Math.min(n, m)));
270   }
271 
272   /**
273    * Grow the second dimension of the array, so that we can calculate the
274    * Levenshtein difference.
275    */
276   private void growDistanceArray(int m) {
277     for (int i = 0; i < d.length; i++) {
278       d[i] = new int[m+1];
279     }
280   }
281 
282   /**
283    * The max Distance is the maximum Levenshtein distance for the text
284    * compared to some other value that results in score that is
285    * better than the minimum similarity.
286    * @param m the length of the "other value"
287    * @return the maximum levenshtein distance that we care about
288    */
289   private final int getMaxDistance(int m) {
290     return (m < maxDistances.length) ? maxDistances[m] : calculateMaxDistance(m);
291   }
292 
293   private void initializeMaxDistances() {
294     for (int i = 0; i < maxDistances.length; i++) {
295       maxDistances[i] = calculateMaxDistance(i);
296     }
297   }
298   
299   private int calculateMaxDistance(int m) {
300     return (int) ((1-minimumSimilarity) * (Math.min(text.length(), m) + prefix.length()));
301   }
302 
303   public void close() throws IOException {
304     super.close();  //call super.close() and let the garbage collector do its work.
305   }
