43    /** Sort by document score (relevancy).  Sort values are Float and higher
44     * values are at the front. */

47    /** Sort by document number (index order).  Sort values are Integer and lower
48     * values are at the front. */

51    /** Guess type of sort based on field contents.  A regular expression is used
52     * to look at the first term indexed for the field and determine if it
53     * represents an integer number, a floating point number, or just arbitrary
54     * string characters.
55     * @deprecated Please specify the exact type, instead.
56     *  Especially, guessing does <b>not</b> work with the new
57     *  {@link NumericField} type.
58     */

65    /** Sort using term values as encoded Integers.  Sort values are Integer and
66     * lower values are at the front. */

74     * lower values are at the front. */
75    public static final int LONG = 6;
76  
77    /** Sort using term values as encoded Doubles.  Sort values are Double and
78     * lower values are at the front. */
79    public static final int DOUBLE = 7;
80  
81    /** Sort using term values as encoded Shorts.  Sort values are Short and

94     * value (using String.compareTo) for all comparisons.
95     * This is typically slower than {@link #STRING}, which

178   /** Creates a sort, possibly in reverse, by terms in the given field, parsed
179    * to numeric values using a custom {@link FieldCache.Parser}.
180    * @param field  Name of field to sort by.  Must not be null.
181    * @param parser Instance of a {@link FieldCache.Parser},
182    *  which must subclass one of the existing numeric
183    *  parsers from {@link FieldCache}. Sort type is inferred
184    *  by testing which numeric parser the parser subclasses.
185    * @param reverse True if natural order should be reversed.
186    * @throws IllegalArgumentException if the parser fails to
187    *  subclass an existing numeric parser, or field is null
188    */
189   public SortField (String field, FieldCache.Parser parser, boolean reverse) {
190     if (parser instanceof FieldCache.IntParser) initFieldType(field, INT);
191     else if (parser instanceof FieldCache.FloatParser) initFieldType(field, FLOAT);
192     else if (parser instanceof FieldCache.ShortParser) initFieldType(field, SHORT);
193     else if (parser instanceof FieldCache.ByteParser) initFieldType(field, BYTE);
194     else if (parser instanceof FieldCache.LongParser) initFieldType(field, LONG);
195     else if (parser instanceof FieldCache.DoubleParser) initFieldType(field, DOUBLE);
196     else
197       throw new IllegalArgumentException("Parser instance does not subclass existing numeric parser from FieldCache (got " + parser + ")");
198 
199     this.reverse = reverse;
200     this.parser = parser;
201   }

268   // Sets field & type, and ensures field is not NULL unless
269   // type is SCORE or DOC
270   private void initFieldType(String field, int type) {
271     this.type = type;
272     if (field == null) {
273       if (type != SCORE && type != DOC)
274         throw new IllegalArgumentException("field can only be null when type is SCORE or DOC");
275     } else {
276       this.field = StringHelper.intern(field);
277     }
278   }

349   public String toString() {
350     StringBuffer buffer = new StringBuffer();
351     switch (type) {
352       case SCORE:
353         buffer.append("<score>");
354         break;
355 
356       case DOC:
357         buffer.append("<doc>");
358         break;
359 
360       case AUTO:
361         buffer.append("<auto: \"").append(field).append("\">");
362         break;
363 
364       case STRING:
365         buffer.append("<string: \"").append(field).append("\">");
366         break;
367 
368       case STRING_VAL:
369         buffer.append("<string_val: \"").append(field).append("\">");
370         break;
371 
372       case BYTE:
373         buffer.append("<byte: \"").append(field).append("\">");
374         break;
375 
376       case SHORT:
377         buffer.append("<short: \"").append(field).append("\">");
378         break;
379 
380       case INT:
381         buffer.append("<int: \"").append(field).append("\">");
382         break;
383 
384       case LONG:
385         buffer.append("<long: \"").append(field).append("\">");
386         break;
387 
388       case FLOAT:
389         buffer.append("<float: \"").append(field).append("\">");
390         break;
391 
392       case DOUBLE:
393         buffer.append("<double: \"").append(field).append("\">");
394         break;
395 
396       case CUSTOM:
397         buffer.append("<custom:\"").append(field).append("\": ").append(factory).append('>');
398         break;
399 
400       default:
401         buffer.append("<???: \"").append(field).append("\">");
402         break;
403     }
404 
405     if (locale != null) buffer.append('(').append(locale).append(')');
406     if (parser != null) buffer.append('(').append(parser).append(')');
407     if (reverse) buffer.append('!');
408 
409     return buffer.toString();
410   }

412   /** Returns true if <code>o</code> is equal to this.  If a
413    *  {@link SortComparatorSource} (deprecated) or {@link
414    *  FieldCache.Parser} was provided, it must properly
415    *  implement equals (unless a singleton is always used). */
416   public boolean equals(Object o) {
417     if (this == o) return true;
418     if (!(o instanceof SortField)) return false;
419     final SortField other = (SortField)o;
420     return (
421       other.field == this.field // field is always interned
422       && other.type == this.type
423       && other.reverse == this.reverse
424       && (other.locale == null ? this.locale == null : other.locale.equals(this.locale))
425       && (other.factory == null ? this.factory == null : other.factory.equals(this.factory))
426       && (other.comparatorSource == null ? this.comparatorSource == null : other.comparatorSource.equals(this.comparatorSource))
427       && (other.parser == null ? this.parser == null : other.parser.equals(this.parser))
428     );
429   }

431   /** Returns true if <code>o</code> is equal to this.  If a
432    *  {@link SortComparatorSource} (deprecated) or {@link
433    *  FieldCache.Parser} was provided, it must properly
434    *  implement hashCode (unless a singleton is always
435    *  used). */
436   public int hashCode() {
437     int hash=type^0x346565dd + Boolean.valueOf(reverse).hashCode()^0xaf5998bb;
438     if (field != null) hash += field.hashCode()^0xff5685dd;
439     if (locale != null) hash += locale.hashCode()^0x08150815;
440     if (factory != null) hash += factory.hashCode()^0x34987555;
441     if (comparatorSource != null) hash += comparatorSource.hashCode();
442     if (parser != null) hash += parser.hashCode()^0x3aaf56ff;
443     return hash;
444   }

453   /** Returns the {@link FieldComparator} to use for
454    * sorting.
455    *
456    * <b>NOTE:</b> This API is experimental and might change in
457    * incompatible ways in the next release.
458    *
459    * @param numHits number of top hits the queue will store
460    * @param sortPos position of this SortField within {@link
461    *   Sort}.  The comparator is primary if sortPos==0,
462    *   secondary if sortPos==1, etc.  Some comparators can
463    *   optimize themselves when they are the primary sort.
464    * @return {@link FieldComparator} to use when sorting
465    */
466   public FieldComparator getComparator(final int numHits, final int sortPos) throws IOException {
467 
468     if (locale != null) {
469       // TODO: it'd be nice to allow FieldCache.getStringIndex
470       // to optionally accept a Locale so sorting could then use
471       // the faster StringComparator impls
472       return new FieldComparator.StringComparatorLocale(numHits, field, locale);
473     }
474 
475     switch (type) {
476     case SortField.SCORE:
477       return new FieldComparator.RelevanceComparator(numHits);
478 
479     case SortField.DOC:
480       return new FieldComparator.DocComparator(numHits);
481 
482     case SortField.INT:
483       return new FieldComparator.IntComparator(numHits, field, parser);
484 
485     case SortField.FLOAT:
486       return new FieldComparator.FloatComparator(numHits, field, parser);
487 
488     case SortField.LONG:
489       return new FieldComparator.LongComparator(numHits, field, parser);
490 
491     case SortField.DOUBLE:
492       return new FieldComparator.DoubleComparator(numHits, field, parser);
493 
494     case SortField.BYTE:
495       return new FieldComparator.ByteComparator(numHits, field, parser);
496 
497     case SortField.SHORT:
498       return new FieldComparator.ShortComparator(numHits, field, parser);
499 
500     case SortField.CUSTOM:
501       assert factory == null && comparatorSource != null;
502       return comparatorSource.newComparator(field, numHits, sortPos, reverse);
503 
504     case SortField.STRING:
505       return new FieldComparator.StringOrdValComparator(numHits, field, sortPos, reverse);
506 
507     case SortField.STRING_VAL:
508       return new FieldComparator.StringValComparator(numHits, field);
509         
510     default:
511       throw new IllegalStateException("Illegal sort type: " + type);
512     }
513   }

515   /**
516    * Attempts to detect the given field type for an IndexReader.
517    * @deprecated
518    */
519   static int detectFieldType(IndexReader reader, String fieldKey) throws IOException {
520     String field = StringHelper.intern(fieldKey);
521     TermEnum enumerator = reader.terms(new Term(field));
522     try {
523       Term term = enumerator.term();
524       if (term == null) {
525         throw new RuntimeException("no terms in field " + field + " - cannot determine sort type");
526       }
527       int ret = 0;
528       if (term.field() == field) {
529         String termtext = term.text().trim();
530 
531         try {
532           Integer.parseInt (termtext);
533           ret = SortField.INT;
534         } catch (NumberFormatException nfe1) {
535           try {
536             Long.parseLong(termtext);
537             ret = SortField.LONG;
538           } catch (NumberFormatException nfe2) {
539             try {
540               Float.parseFloat (termtext);
541               ret = SortField.FLOAT;
542             } catch (NumberFormatException nfe3) {
543               ret = SortField.STRING;
544             }
545           }
546         }         
547       } else {
548         throw new RuntimeException("field \"" + field + "\" does not appear to be indexed");
549       }
550       return ret;
551     } finally {
552       enumerator.close();
553     }
554   }
