20  import java.io.IOException;
21  
22  import org.apache.lucene.index.IndexReader;
23  
24  /**
25   * <p>Expert: Collectors are primarily meant to be used to
26   * gather raw results from a search, and implement sorting
27   * or custom result filtering, collation, etc. </p>
28   *
29   * <p>As of 2.9, this class replaces the deprecated
30   * HitCollector, and offers an API for efficient collection
31   * of hits across sequential {@link IndexReader}s. {@link
32   * IndexSearcher} advances the collector through each of the
33   * sub readers, in an arbitrary order. This results in a
34   * higher performance means of collection.</p>
35   *
36   * <p>Lucene's core collectors are derived from Collector.
37   * Likely your application can use one of these classes, or
38   * subclass {@link TopDocsCollector}, instead of
39   * implementing Collector directly:
40   *
41   * <ul>
42   *      
43   *   <li>{@link TopDocsCollector} is an abstract base class
44   *   that assumes you will retrieve the top N docs,
45   *   according to some criteria, after collection is
46   *   done.  </li>
47   *
48   *   <li>{@link TopScoreDocCollector} is a concrete subclass
49   *   {@link TopDocsCollector} and sorts according to score +
50   *   docID.  This is used internally by the {@link
51   *   IndexSearcher} search methods that do not take an
52   *   explicit {@link Sort}. It is likely the most frequently
53   *   used collector.</li>
54   *
55   *   <li>{@link TopFieldCollector} subclasses {@link
56   *   TopDocsCollector} and sorts according to a specified
57   *   {@link Sort} object (sort by field).  This is used
58   *   internally by the {@link IndexSearcher} search methods
59   *   that take an explicit {@link Sort}.
60   *
61   *   <li>{@link TimeLimitingCollector}, which wraps any other
62   *   Collector and aborts the search if it's taken too much
63   *   time, will subclass Collector in 3.0 (presently it
64   *   subclasses the deprecated HitCollector).</li>
65   *
66   *   <li>{@link PositiveScoresOnlyCollector} wraps any other
67   *   Collector and prevents collection of hits whose score
68   *   is &lt;= 0.0</li>
69   *
70   * </ul>
71   *
72   * <p>Collector decouples the score from the collected doc:
73   * the score computation is skipped entirely if it's not
74   * needed.  Collectors that do need the score should
75   * implement the {@link #setScorer} method, to hold onto the
76   * passed {@link Scorer} instance, and call {@link
77   * Scorer#score()} within the collect method to compute the
78   * current hit's score.  If your collector may request the
79   * score for a single hit multiple times, you should use
80   * {@link ScoreCachingWrappingScorer}. </p>
81   * 
82   * <p><b>NOTE:</b> The doc that is passed to the collect
83   * method is relative to the current reader. If your
84   * collector needs to resolve this to the docID space of the
85   * Multi*Reader, you must re-base it by recording the
86   * docBase from the most recent setNextReader call.  Here's
87   * a simple example showing how to collect docIDs into a
88   * BitSet:</p>
89   * 
90   * <pre>
91   * Searcher searcher = new IndexSearcher(indexReader);
92   * final BitSet bits = new BitSet(indexReader.maxDoc());
93   * searcher.search(query, new Collector() {
94   *   private int docBase;
95   * 
96   *   <em>// ignore scorer</em>
97   *   public void setScorer(Scorer scorer) {
98   *   }
99   *
100  *   <em>// accept docs out of order (for a BitSet it doesn't matter)</em>
101  *   public boolean acceptsDocsOutOfOrder() {
102  *     return true;
103  *   }
104  * 
105  *   public void collect(int doc) {
106  *     bits.set(doc + docBase);
107  *   }
108  * 
109  *   public void setNextReader(IndexReader reader, int docBase) {
110  *     this.docBase = docBase;
111  *   }
112  * });
113  * </pre>
114  *
115  * <p>Not all collectors will need to rebase the docID.  For
116  * example, a collector that simply counts the total number
117  * of hits would skip it.</p>
118  * 
119  * <p><b>NOTE:</b> Prior to 2.9, Lucene silently filtered
120  * out hits with score <= 0.  As of 2.9, the core Collectors
121  * no longer do that.  It's very unusual to have such hits
122  * (a negative query boost, or function query returning
123  * negative custom scores, could cause it to happen).  If
124  * you need that behavior, use {@link
125  * PositiveScoresOnlyCollector}.</p>
126  *
127  * <p><b>NOTE:</b> This API is experimental and might change
128  * in incompatible ways in the next release.</p>
129  * 
130  * @since 2.9
131  */
132 public abstract class Collector {
133   
134   /**
135    * Called before successive calls to {@link #collect(int)}. Implementations
136    * that need the score of the current document (passed-in to
137    * {@link #collect(int)}), should save the passed-in Scorer and call
138    * scorer.score() when needed.
139    */
140   public abstract void setScorer(Scorer scorer) throws IOException;
141   
142   /**
143    * Called once for every document matching a query, with the unbased document
144    * number.
145    * 
146    * <p>
147    * Note: This is called in an inner search loop. For good search performance,
148    * implementations of this method should not call {@link Searcher#doc(int)} or
149    * {@link org.apache.lucene.index.IndexReader#document(int)} on every hit.
150    * Doing so can slow searches by an order of magnitude or more.
151    */
152   public abstract void collect(int doc) throws IOException;
153 
154   /**
155    * Called before collecting from each IndexReader. All doc ids in
156    * {@link #collect(int)} will correspond to reader.
157    * 
158    * Add docBase to the current IndexReaders internal document id to re-base ids
159    * in {@link #collect(int)}.
160    * 
161    * @param reader
162    *          next IndexReader
163    * @param docBase
164    */
165   public abstract void setNextReader(IndexReader reader, int docBase) throws IOException;
166 
167   /**
168    * Return <code>true</code> if this collector does not
169    * require the matching docIDs to be delivered in int sort
170    * order (smallest to largest) to {@link #collect}.
171    *
172    * <p> Most Lucene Query implementations will visit
173    * matching docIDs in order.  However, some queries
174    * (currently limited to certain cases of {@link
175    * BooleanQuery}) can achieve faster searching if the
176    * <code>Collector</code> allows them to deliver the
177    * docIDs out of order.</p>
178    *
179    * <p> Many collectors don't mind getting docIDs out of
180    * order, so it's important to return <code>true</code>
181    * here.
182    */
183   public abstract boolean acceptsDocsOutOfOrder();
184   
185 }
