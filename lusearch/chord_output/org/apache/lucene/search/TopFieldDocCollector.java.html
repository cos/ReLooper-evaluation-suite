20  import java.io.IOException;
21  
22  import org.apache.lucene.index.IndexReader;
23  
24  /** A {@link HitCollector} implementation that collects the top-sorting
25   * documents, returning them as a {@link TopFieldDocs}.  This is used by {@link
26   * IndexSearcher} to implement {@link TopFieldDocs}-based search.
27   *
28   * <p>This may be extended, overriding the collect method to, e.g.,
29   * conditionally invoke <code>super()</code> in order to filter which
30   * documents are collected.
31   *
32   * @deprecated Please use {@link TopFieldCollector} instead.
33   */
34  public class TopFieldDocCollector extends TopDocCollector {
35  
36    private FieldDoc reusableFD;
37  
38    /** Construct to collect a given number of hits.
39     * @param reader the index to be searched
40     * @param sort the sort criteria
41     * @param numHits the maximum number of hits to collect
42     */
43    public TopFieldDocCollector(IndexReader reader, Sort sort, int numHits)
44      throws IOException {
45      super(new FieldSortedHitQueue(reader, sort.fields, numHits));
46    }
47  
48    // javadoc inherited
49    public void collect(int doc, float score) {
50      if (score > 0.0f) {
51        totalHits++;
52        if (reusableFD == null)
53          reusableFD = new FieldDoc(doc, score);
54        else {
55          // Whereas TopScoreDocCollector can skip this if the
56          // score is not competitive, we cannot because the
57          // comparators in the FieldSortedHitQueue.lessThan
58          // aren't in general congruent with "higher score
59          // wins"
60          reusableFD.score = score;
61          reusableFD.doc = doc;
62        }
63        reusableFD = (FieldDoc) hq.insertWithOverflow(reusableFD);
64      }
65    }
66  
67    // javadoc inherited
68    public TopDocs topDocs() {
69      FieldSortedHitQueue fshq = (FieldSortedHitQueue)hq;
70      ScoreDoc[] scoreDocs = new ScoreDoc[fshq.size()];
71      for (int i = fshq.size()-1; i >= 0; i--)      // put docs in array
72        scoreDocs[i] = fshq.fillFields ((FieldDoc) fshq.pop());
73  
74      return new TopFieldDocs(totalHits, scoreDocs,
75                              fshq.getFields(), fshq.getMaxScore());
76    }
77  }
