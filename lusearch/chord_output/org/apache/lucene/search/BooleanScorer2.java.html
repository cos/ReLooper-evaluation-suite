27  /** An alternative to BooleanScorer that also allows a minimum number
28   * of optional scorers that should match.
29   * <br>Implements skipTo(), and has no limitations on the numbers of added scorers.
30   * <br>Uses ConjunctionScorer, DisjunctionScorer, ReqOptScorer and ReqExclScorer.
31   */
32  class BooleanScorer2 extends Scorer {
33    
34    private final List requiredScorers;
35    private final List optionalScorers;
36    private final List prohibitedScorers;
37  
38    private class Coordinator {
39      float[] coordFactors = null;
40      int maxCoord = 0; // to be increased for each non prohibited scorer
41      int nrMatchers; // to be increased by score() of match counting scorers.
42      
43      void init() { // use after all scorers have been added.
44        coordFactors = new float[maxCoord + 1];
45        Similarity sim = getSimilarity();
46        for (int i = 0; i <= maxCoord; i++) {
47          coordFactors[i] = sim.coord(i, maxCoord);
48        }
49      }
50      
51    }
52  
53    private final Coordinator coordinator;
54  
55    /** The scorer to which all scoring will be delegated,
56     * except for computing and using the coordination factor.
57     */
58    private final Scorer countingSumScorer;
59  
60    /** The number of optionalScorers that need to match (if there are any) */
61    private final int minNrShouldMatch;
62    
63    private int doc = -1;
64  
65    /**
66     * Creates a {@link Scorer} with the given similarity and lists of required,
67     * prohibited and optional scorers. In no required scorers are added, at least
68     * one of the optional scorers will have to match during the search.
69     * 
70     * @param similarity
71     *          The similarity to be used.
72     * @param minNrShouldMatch
73     *          The minimum number of optional added scorers that should match
74     *          during the search. In case no required scorers are added, at least
75     *          one of the optional scorers will have to match during the search.
76     * @param required
77     *          the list of required scorers.
78     * @param prohibited
79     *          the list of prohibited scorers.
80     * @param optional
81     *          the list of optional scorers.
82     */
83    public BooleanScorer2(Similarity similarity, int minNrShouldMatch,
84        List required, List prohibited, List optional) throws IOException {
85      super(similarity);
86      if (minNrShouldMatch < 0) {
87        throw new IllegalArgumentException("Minimum number of optional scorers should not be negative");
88      }
89      coordinator = new Coordinator();
90      this.minNrShouldMatch = minNrShouldMatch;
91  
92      optionalScorers = optional;
93      coordinator.maxCoord += optional.size();
94  
95      requiredScorers = required;
96      coordinator.maxCoord += required.size();
97      
98      prohibitedScorers = prohibited;
99      
100     coordinator.init();
101     countingSumScorer = makeCountingSumScorer();
102   }
103   
104   /** Count a scorer as a single match. */
105   private class SingleMatchScorer extends Scorer {
106     private Scorer scorer;
107     private int lastScoredDoc = -1;
108     // Save the score of lastScoredDoc, so that we don't compute it more than
109     // once in score().
110     private float lastDocScore = Float.NaN;
111 
112     SingleMatchScorer(Scorer scorer) {
113       super(scorer.getSimilarity());
114       this.scorer = scorer;
115     }
116     public float score() throws IOException {
117       int doc = docID();
118       if (doc >= lastScoredDoc) {
119         if (doc > lastScoredDoc) {
120           lastDocScore = scorer.score();
121           lastScoredDoc = doc;
122         }
123         coordinator.nrMatchers++;
124       }
125       return lastDocScore;
126     }
127     /** @deprecated use {@link #docID()} instead. */
128     public int doc() {
129       return scorer.doc();
130     }
131     public int docID() {
132       return scorer.docID();
133     }
134     /** @deprecated use {@link #nextDoc()} instead. */
135     public boolean next() throws IOException {
136       return scorer.nextDoc() != NO_MORE_DOCS;
137     }
138     public int nextDoc() throws IOException {
139       return scorer.nextDoc();
140     }
141     /** @deprecated use {@link #advance(int)} instead. */
142     public boolean skipTo(int docNr) throws IOException {
143       return scorer.advance(docNr) != NO_MORE_DOCS;
144     }
145     public int advance(int target) throws IOException {
146       return scorer.advance(target);
147     }
148     public Explanation explain(int docNr) throws IOException {
149       return scorer.explain(docNr);
150     }
151   }

153   private Scorer countingDisjunctionSumScorer(final List scorers,
154       int minNrShouldMatch) throws IOException {
155     // each scorer from the list counted as a single matcher
156     return new DisjunctionSumScorer(scorers, minNrShouldMatch) {
157       private int lastScoredDoc = -1;
158       // Save the score of lastScoredDoc, so that we don't compute it more than
159       // once in score().
160       private float lastDocScore = Float.NaN;
161       public float score() throws IOException {
162         int doc = docID();
163         if (doc >= lastScoredDoc) {
164           if (doc > lastScoredDoc) {
165             lastDocScore = super.score();
166             lastScoredDoc = doc;
167           }
168           coordinator.nrMatchers += super.nrMatchers;
169         }
170         return lastDocScore;
171       }
172     };
173   }
174 
175   private static final Similarity defaultSimilarity = Similarity.getDefault();
176 
177   private Scorer countingConjunctionSumScorer(List requiredScorers) throws IOException {
178     // each scorer from the list counted as a single matcher
179     final int requiredNrMatchers = requiredScorers.size();
180     return new ConjunctionScorer(defaultSimilarity, requiredScorers) {
181       private int lastScoredDoc = -1;
182       // Save the score of lastScoredDoc, so that we don't compute it more than
183       // once in score().
184       private float lastDocScore = Float.NaN;
185       public float score() throws IOException {
186         int doc = docID();
187         if (doc >= lastScoredDoc) {
188           if (doc > lastScoredDoc) {
189             lastDocScore = super.score();
190             lastScoredDoc = doc;
191           }
192           coordinator.nrMatchers += requiredNrMatchers;
193         }
194         // All scorers match, so defaultSimilarity super.score() always has 1 as
195         // the coordination factor.
196         // Therefore the sum of the scores of the requiredScorers
197         // is used as score.
198         return lastDocScore;
199       }
200     };
201   }
202 
203   private Scorer dualConjunctionSumScorer(Scorer req1, Scorer req2) throws IOException { // non counting.
204     return new ConjunctionScorer(defaultSimilarity, new Scorer[]{req1, req2});
205     // All scorers match, so defaultSimilarity always has 1 as
206     // the coordination factor.
207     // Therefore the sum of the scores of two scorers
208     // is used as score.
209   }

211   /** Returns the scorer to be used for match counting and score summing.
212    * Uses requiredScorers, optionalScorers and prohibitedScorers.
213    */
214   private Scorer makeCountingSumScorer() throws IOException { // each scorer counted as a single matcher
215     return (requiredScorers.size() == 0)
216           ? makeCountingSumScorerNoReq()
217           : makeCountingSumScorerSomeReq();
218   }
219 
220   private Scorer makeCountingSumScorerNoReq() throws IOException { // No required scorers
221     // minNrShouldMatch optional scorers are required, but at least 1
222     int nrOptRequired = (minNrShouldMatch < 1) ? 1 : minNrShouldMatch;
223     Scorer requiredCountingSumScorer;
224     if (optionalScorers.size() > nrOptRequired)
225       requiredCountingSumScorer = countingDisjunctionSumScorer(optionalScorers, nrOptRequired);
226     else if (optionalScorers.size() == 1)
227       requiredCountingSumScorer = new SingleMatchScorer((Scorer) optionalScorers.get(0));
228     else
229       requiredCountingSumScorer = countingConjunctionSumScorer(optionalScorers);
230     return addProhibitedScorers(requiredCountingSumScorer);
231   }
232 
233   private Scorer makeCountingSumScorerSomeReq() throws IOException { // At least one required scorer.
234     if (optionalScorers.size() == minNrShouldMatch) { // all optional scorers also required.
235       ArrayList allReq = new ArrayList(requiredScorers);
236       allReq.addAll(optionalScorers);
237       return addProhibitedScorers(countingConjunctionSumScorer(allReq));
238     } else { // optionalScorers.size() > minNrShouldMatch, and at least one required scorer
239       Scorer requiredCountingSumScorer =
240             requiredScorers.size() == 1
241             ? new SingleMatchScorer((Scorer) requiredScorers.get(0))
242             : countingConjunctionSumScorer(requiredScorers);
243       if (minNrShouldMatch > 0) { // use a required disjunction scorer over the optional scorers
244         return addProhibitedScorers( 
245                       dualConjunctionSumScorer( // non counting
246                               requiredCountingSumScorer,
247                               countingDisjunctionSumScorer(
248                                       optionalScorers,
249                                       minNrShouldMatch)));
250       } else { // minNrShouldMatch == 0
251         return new ReqOptSumScorer(
252                       addProhibitedScorers(requiredCountingSumScorer),
253                       optionalScorers.size() == 1
254                         ? new SingleMatchScorer((Scorer) optionalScorers.get(0))
255                         // require 1 in combined, optional scorer.
256                         : countingDisjunctionSumScorer(optionalScorers, 1));
257       }
258     }
259   }
260   
261   /** Returns the scorer to be used for match counting and score summing.
262    * Uses the given required scorer and the prohibitedScorers.
263    * @param requiredCountingSumScorer A required scorer already built.
264    */
265   private Scorer addProhibitedScorers(Scorer requiredCountingSumScorer) throws IOException
266   {
267     return (prohibitedScorers.size() == 0)
268           ? requiredCountingSumScorer // no prohibited
269           : new ReqExclScorer(requiredCountingSumScorer,
270                               ((prohibitedScorers.size() == 1)
271                                 ? (Scorer) prohibitedScorers.get(0)
272                                 : new DisjunctionSumScorer(prohibitedScorers)));
273   }

275   /** Scores and collects all matching documents.
276    * @param hc The collector to which all matching documents are passed through
277    * {@link HitCollector#collect(int, float)}.
278    * <br>When this method is used the {@link #explain(int)} method should not be used.
279    * @deprecated use {@link #score(Collector)} instead.
280    */
281   public void score(HitCollector hc) throws IOException {
282     score(new HitCollectorWrapper(hc));
283   }
284 
285   /** Scores and collects all matching documents.
286    * @param collector The collector to which all matching documents are passed through.
287    * <br>When this method is used the {@link #explain(int)} method should not be used.
288    */
289   public void score(Collector collector) throws IOException {
290     collector.setScorer(this);
291     while ((doc = countingSumScorer.nextDoc()) != NO_MORE_DOCS) {
292       collector.collect(doc);
293     }
294   }
295 
296   /** Expert: Collects matching documents in a range.
297    * <br>Note that {@link #next()} must be called once before this method is
298    * called for the first time.
299    * @param hc The collector to which all matching documents are passed through
300    * {@link HitCollector#collect(int, float)}.
301    * @param max Do not score documents past this.
302    * @return true if more matching documents may remain.
303    * @deprecated use {@link #score(Collector, int, int)} instead.
304    */
305   protected boolean score(HitCollector hc, int max) throws IOException {
306     return score(new HitCollectorWrapper(hc), max, docID());
307   }
308   
309   protected boolean score(Collector collector, int max, int firstDocID) throws IOException {
310     doc = firstDocID;
311     collector.setScorer(this);
312     while (doc < max) {
313       collector.collect(doc);
314       doc = countingSumScorer.nextDoc();
315     }
316     return doc != NO_MORE_DOCS;
317   }

323     return doc;

327   public boolean next() throws IOException {
328     return nextDoc() != NO_MORE_DOCS;
329   }

331   public int nextDoc() throws IOException {
332     return doc = countingSumScorer.nextDoc();
333   }
334   
335   public float score() throws IOException {
336     coordinator.nrMatchers = 0;
337     float sum = countingSumScorer.score();
338     return sum * coordinator.coordFactors[coordinator.nrMatchers];
339   }

341   /** @deprecated use {@link #advance(int)} instead. */
342   public boolean skipTo(int target) throws IOException {
343     return advance(target) != NO_MORE_DOCS;
344   }
345 
346   public int advance(int target) throws IOException {
347     return doc = countingSumScorer.advance(target);
348   }
