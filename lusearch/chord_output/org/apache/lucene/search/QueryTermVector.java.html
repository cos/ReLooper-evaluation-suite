29  import org.apache.lucene.analysis.Analyzer;
30  import org.apache.lucene.analysis.TokenStream;
31  import org.apache.lucene.analysis.tokenattributes.TermAttribute;
32  import org.apache.lucene.index.TermFreqVector;

38  public class QueryTermVector implements TermFreqVector {
39    private String [] terms = new String[0];
40    private int [] termFreqs = new int[0];
41  
42    public String getField() { return null;  }
43  
44    /**
45     * 
46     * @param queryTerms The original list of terms from the query, can contain duplicates
47     */ 
48    public QueryTermVector(String [] queryTerms) {
49  
50      processTerms(queryTerms);
51    }
52  
53    public QueryTermVector(String queryString, Analyzer analyzer) {    
54      if (analyzer != null)
55      {
56        TokenStream stream = analyzer.tokenStream("", new StringReader(queryString));
57        if (stream != null)
58        {
59          List terms = new ArrayList();
60          try {
61            boolean hasMoreTokens = false;
62            
63            stream.reset(); 
64            TermAttribute termAtt = (TermAttribute) stream.addAttribute(TermAttribute.class);
65  
66            hasMoreTokens = stream.incrementToken();
67            while (hasMoreTokens) {
68              terms.add(termAtt.term());
69              hasMoreTokens = stream.incrementToken();
70            }
71            processTerms((String[])terms.toArray(new String[terms.size()]));
72          } catch (IOException e) {
73          }
74        }
75      }                                                              
76    }
77    
78    private void processTerms(String[] queryTerms) {
79      if (queryTerms != null) {
80        Arrays.sort(queryTerms);
81        Map tmpSet = new HashMap(queryTerms.length);
82        //filter out duplicates
83        List tmpList = new ArrayList(queryTerms.length);
84        List tmpFreqs = new ArrayList(queryTerms.length);
85        int j = 0;
86        for (int i = 0; i < queryTerms.length; i++) {
87          String term = queryTerms[i];
88          Integer position = (Integer)tmpSet.get(term);
89          if (position == null) {
90            tmpSet.put(term, new Integer(j++));
91            tmpList.add(term);
92            tmpFreqs.add(new Integer(1));
93          }       
94          else {
95            Integer integer = (Integer)tmpFreqs.get(position.intValue());
96            tmpFreqs.set(position.intValue(), new Integer(integer.intValue() + 1));          
97          }
98        }
99        terms = (String[])tmpList.toArray(terms);
100       //termFreqs = (int[])tmpFreqs.toArray(termFreqs);
101       termFreqs = new int[tmpFreqs.size()];
102       int i = 0;
103       for (Iterator iter = tmpFreqs.iterator(); iter.hasNext();) {
104         Integer integer = (Integer) iter.next();
105         termFreqs[i++] = integer.intValue();
106       }
107     }
108   }
109   
110   public final String toString() {
111         StringBuffer sb = new StringBuffer();
112         sb.append('{');
113         for (int i=0; i<terms.length; i++) {
114             if (i>0) sb.append(", ");
115             sb.append(terms[i]).append('/').append(termFreqs[i]);
116         }
117         sb.append('}');
118         return sb.toString();
119     }
120   
121 
122   public int size() {
123     return terms.length;
124   }
125 
126   public String[] getTerms() {
127     return terms;
128   }
129 
130   public int[] getTermFrequencies() {
131     return termFreqs;
132   }
133 
134   public int indexOf(String term) {
135     int res = Arrays.binarySearch(terms, term);
136         return res >= 0 ? res : -1;
137   }
138 
139   public int[] indexesOf(String[] terms, int start, int len) {
140     int res[] = new int[len];
141 
142     for (int i=0; i < len; i++) {
143         res[i] = indexOf(terms[i]);
144     }
145     return res;                  
146   }

148 }
