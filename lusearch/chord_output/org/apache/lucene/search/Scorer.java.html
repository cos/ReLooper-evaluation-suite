20  import java.io.IOException;
21  
22  /**
23   * Expert: Common scoring functionality for different types of queries.
24   *
25   * <p>
26   * A <code>Scorer</code> iterates over documents matching a
27   * query in increasing order of doc Id.
28   * </p>
29   * <p>
30   * Document scores are computed using a given <code>Similarity</code>
31   * implementation.
32   * </p>
33   *
34   * <p><b>NOTE</b>: The values Float.Nan,
35   * Float.NEGATIVE_INFINITY and Float.POSITIVE_INFINITY are
36   * not valid scores.  Certain collectors (eg {@link
37   * TopScoreDocCollector}) will not properly collect hits
38   * with these scores.
39   *
40   * @see BooleanQuery#setAllowDocsOutOfOrder
41   */
42  public abstract class Scorer extends DocIdSetIterator {
43    private Similarity similarity;
44  
45    /** Constructs a Scorer.
46     * @param similarity The <code>Similarity</code> implementation used by this scorer.
47     */
48    protected Scorer(Similarity similarity) {
49      this.similarity = similarity;
50    }
51  
52    /** Returns the Similarity implementation used by this scorer. */
53    public Similarity getSimilarity() {
54      return this.similarity;
55    }
56  
57    /** Scores and collects all matching documents.
58     * @param hc The collector to which all matching documents are passed through
59     * {@link HitCollector#collect(int, float)}.
60     * <br>When this method is used the {@link #explain(int)} method should not be used.
61     * @deprecated use {@link #score(Collector)} instead.
62     */
63    public void score(HitCollector hc) throws IOException {
64      score(new HitCollectorWrapper(hc));
65    }
66    
67    /** Scores and collects all matching documents.
68     * @param collector The collector to which all matching documents are passed.
69     * <br>When this method is used the {@link #explain(int)} method should not be used.
70     */
71    public void score(Collector collector) throws IOException {
72      collector.setScorer(this);
73      int doc;
74      while ((doc = nextDoc()) != NO_MORE_DOCS) {
75        collector.collect(doc);
76      }
77    }
78  
79    /** Expert: Collects matching documents in a range.  Hook for optimization.
80     * Note that {@link #next()} must be called once before this method is called
81     * for the first time.
82     * @param hc The collector to which all matching documents are passed through
83     * {@link HitCollector#collect(int, float)}.
84     * @param max Do not score documents past this.
85     * @return true if more matching documents may remain.
86     * @deprecated use {@link #score(Collector, int, int)} instead.
87     */
88    protected boolean score(HitCollector hc, int max) throws IOException {
89      return score(new HitCollectorWrapper(hc), max, docID());
90    }
91    
92    /**
93     * Expert: Collects matching documents in a range. Hook for optimization.
94     * Note, <code>firstDocID</code> is added to ensure that {@link #nextDoc()}
95     * was called before this method.
96     * 
97     * @param collector
98     *          The collector to which all matching documents are passed.
99     * @param max
100    *          Do not score documents past this.
101    * @param firstDocID
102    *          The first document ID (ensures {@link #nextDoc()} is called before
103    *          this method.
104    * @return true if more matching documents may remain.
105    */
106   protected boolean score(Collector collector, int max, int firstDocID) throws IOException {
107     collector.setScorer(this);
108     int doc = firstDocID;
109     while (doc < max) {
110       collector.collect(doc);
111       doc = nextDoc();
112     }
113     return doc != NO_MORE_DOCS;
114   }
115   
116   /** Returns the score of the current document matching the query.
117    * Initially invalid, until {@link #next()} or {@link #skipTo(int)}
118    * is called the first time, or when called from within
119    * {@link Collector#collect}.
120    */
121   public abstract float score() throws IOException;
122 
123   /** Returns an explanation of the score for a document.
124    * <br>When this method is used, the {@link #next()}, {@link #skipTo(int)} and
125    * {@link #score(HitCollector)} methods should not be used.
126    * @param doc The document number for the explanation.
127    *
128    * @deprecated Please use {@link IndexSearcher#explain}
129    * or {@link Weight#explain} instead.
130    */
131   public Explanation explain(int doc) throws IOException {
132     throw new UnsupportedOperationException();
133   }
134 
135 }
