20  import java.util.List;
21  import java.util.Iterator;
22  import java.io.IOException;
23  
24  import org.apache.lucene.util.ScorerDocQueue;
25  
26  /** A Scorer for OR like queries, counterpart of <code>ConjunctionScorer</code>.
27   * This Scorer implements {@link Scorer#skipTo(int)} and uses skipTo() on the given Scorers. 
28   * TODO: Implement score(HitCollector, int).
29   */
30  class DisjunctionSumScorer extends Scorer {
31    /** The number of subscorers. */ 
32    private final int nrScorers;
33    
34    /** The subscorers. */
35    protected final List subScorers;
36    
37    /** The minimum number of scorers that should match. */
38    private final int minimumNrMatchers;
39    
40    /** The scorerDocQueue contains all subscorers ordered by their current doc(),
41     * with the minimum at the top.
42     * <br>The scorerDocQueue is initialized the first time next() or skipTo() is called.
43     * <br>An exhausted scorer is immediately removed from the scorerDocQueue.
44     * <br>If less than the minimumNrMatchers scorers
45     * remain in the scorerDocQueue next() and skipTo() return false.
46     * <p>
47     * After each to call to next() or skipTo()
48     * <code>currentSumScore</code> is the total score of the current matching doc,
49     * <code>nrMatchers</code> is the number of matching scorers,
50     * and all scorers are after the matching doc, or are exhausted.
51     */
52    private ScorerDocQueue scorerDocQueue;
53    
54    /** The document number of the current match. */
55    private int currentDoc = -1;
56  
57    /** The number of subscorers that provide the current match. */
58    protected int nrMatchers = -1;
59  
60    private float currentScore = Float.NaN;
61    
62    /** Construct a <code>DisjunctionScorer</code>.
63     * @param subScorers A collection of at least two subscorers.
64     * @param minimumNrMatchers The positive minimum number of subscorers that should
65     * match to match this query.
66     * <br>When <code>minimumNrMatchers</code> is bigger than
67     * the number of <code>subScorers</code>,
68     * no matches will be produced.
69     * <br>When minimumNrMatchers equals the number of subScorers,
70     * it more efficient to use <code>ConjunctionScorer</code>.
71     */
72    public DisjunctionSumScorer( List subScorers, int minimumNrMatchers) throws IOException {
73      super(null);
74      
75      nrScorers = subScorers.size();
76  
77      if (minimumNrMatchers <= 0) {
78        throw new IllegalArgumentException("Minimum nr of matchers must be positive");
79      }
80      if (nrScorers <= 1) {
81        throw new IllegalArgumentException("There must be at least 2 subScorers");
82      }
83  
84      this.minimumNrMatchers = minimumNrMatchers;
85      this.subScorers = subScorers;
86  
87      initScorerDocQueue();
88    }
89    
90    /** Construct a <code>DisjunctionScorer</code>, using one as the minimum number
91     * of matching subscorers.
92     */
93    public DisjunctionSumScorer(List subScorers) throws IOException {
94      this(subScorers, 1);
95    }
96  
97    /** Called the first time next() or skipTo() is called to
98     * initialize <code>scorerDocQueue</code>.
99     */
100   private void initScorerDocQueue() throws IOException {
101     Iterator si = subScorers.iterator();
102     scorerDocQueue = new ScorerDocQueue(nrScorers);
103     while (si.hasNext()) {
104       Scorer se = (Scorer) si.next();
105       if (se.nextDoc() != NO_MORE_DOCS) { // doc() method will be used in scorerDocQueue.
106         scorerDocQueue.insert(se);
107       }
108     }
109   }
110 
111   /** Scores and collects all matching documents.
112    * @param hc The collector to which all matching documents are passed through
113    * {@link HitCollector#collect(int, float)}.
114    * <br>When this method is used the {@link #explain(int)} method should not be used.
115    * @deprecated use {@link #score(Collector)} instead.
116    */
117   public void score(HitCollector hc) throws IOException {
118     score(new HitCollectorWrapper(hc));
119   }
120   
121   /** Scores and collects all matching documents.
122    * @param collector The collector to which all matching documents are passed through.
123    * <br>When this method is used the {@link #explain(int)} method should not be used.
124    */
125   public void score(Collector collector) throws IOException {
126     collector.setScorer(this);
127     while (nextDoc() != NO_MORE_DOCS) {
128       collector.collect(currentDoc);
129     }
130   }
131 
132   /** Expert: Collects matching documents in a range.  Hook for optimization.
133    * Note that {@link #next()} must be called once before this method is called
134    * for the first time.
135    * @param hc The collector to which all matching documents are passed through
136    * {@link HitCollector#collect(int, float)}.
137    * @param max Do not score documents past this.
138    * @return true if more matching documents may remain.
139    * @deprecated use {@link #score(Collector, int, int)} instead.
140    */
141   protected boolean score(HitCollector hc, int max) throws IOException {
142     return score(new HitCollectorWrapper(hc), max, docID());
143   }
144   
145   /** Expert: Collects matching documents in a range.  Hook for optimization.
146    * Note that {@link #next()} must be called once before this method is called
147    * for the first time.
148    * @param collector The collector to which all matching documents are passed through.
149    * @param max Do not score documents past this.
150    * @return true if more matching documents may remain.
151    */
152   protected boolean score(Collector collector, int max, int firstDocID) throws IOException {
153     // firstDocID is ignored since nextDoc() sets 'currentDoc'
154     collector.setScorer(this);
155     while (currentDoc < max) {
156       collector.collect(currentDoc);
157       if (nextDoc() == NO_MORE_DOCS) {
158         return false;
159       }
160     }
161     return true;
162   }
163 
164   /** @deprecated use {@link #nextDoc()} instead. */
165   public boolean next() throws IOException {
166     return nextDoc() != NO_MORE_DOCS;
167   }
168 
169   public int nextDoc() throws IOException {
170     if (scorerDocQueue.size() < minimumNrMatchers || !advanceAfterCurrent()) {
171       currentDoc = NO_MORE_DOCS;
172     }
173     return currentDoc;
174   }
175 
176   /** Advance all subscorers after the current document determined by the
177    * top of the <code>scorerDocQueue</code>.
178    * Repeat until at least the minimum number of subscorers match on the same
179    * document and all subscorers are after that document or are exhausted.
180    * <br>On entry the <code>scorerDocQueue</code> has at least <code>minimumNrMatchers</code>
181    * available. At least the scorer with the minimum document number will be advanced.
182    * @return true iff there is a match.
183    * <br>In case there is a match, </code>currentDoc</code>, </code>currentSumScore</code>,
184    * and </code>nrMatchers</code> describe the match.
185    *
186    * TODO: Investigate whether it is possible to use skipTo() when
187    * the minimum number of matchers is bigger than one, ie. try and use the
188    * character of ConjunctionScorer for the minimum number of matchers.
189    * Also delay calling score() on the sub scorers until the minimum number of
190    * matchers is reached.
191    * <br>For this, a Scorer array with minimumNrMatchers elements might
192    * hold Scorers at currentDoc that are temporarily popped from scorerQueue.
193    */
194   protected boolean advanceAfterCurrent() throws IOException {
195     do { // repeat until minimum nr of matchers
196       currentDoc = scorerDocQueue.topDoc();
197       currentScore = scorerDocQueue.topScore();
198       nrMatchers = 1;
199       do { // Until all subscorers are after currentDoc
200         if (!scorerDocQueue.topNextAndAdjustElsePop()) {
201           if (scorerDocQueue.size() == 0) {
202             break; // nothing more to advance, check for last match.
203           }
204         }
205         if (scorerDocQueue.topDoc() != currentDoc) {
206           break; // All remaining subscorers are after currentDoc.
207         }
208         currentScore += scorerDocQueue.topScore();
209         nrMatchers++;
210       } while (true);
211       
212       if (nrMatchers >= minimumNrMatchers) {
213         return true;
214       } else if (scorerDocQueue.size() < minimumNrMatchers) {
215         return false;
216       }
217     } while (true);
218   }
219   
220   /** Returns the score of the current document matching the query.
221    * Initially invalid, until {@link #next()} is called the first time.
222    */
223   public float score() throws IOException { return currentScore; }
224    
225   /** @deprecated use {@link #docID()} instead. */
226   public int doc() { return currentDoc; }
227 
228   public int docID() {
229     return currentDoc;
230   }
231   
232   /** Returns the number of subscorers matching the current document.
233    * Initially invalid, until {@link #next()} is called the first time.
234    */
235   public int nrMatchers() {
236     return nrMatchers;
237   }
238 
239   /**
240    * Skips to the first match beyond the current whose document number is
241    * greater than or equal to a given target. <br>
242    * When this method is used the {@link #explain(int)} method should not be
243    * used. <br>
244    * The implementation uses the skipTo() method on the subscorers.
245    * 
246    * @param target
247    *          The target document number.
248    * @return true iff there is such a match.
249    * @deprecated use {@link #advance(int)} instead.
250    */
251   public boolean skipTo(int target) throws IOException {
252     return advance(target) != NO_MORE_DOCS;
253   }
254 
255   /**
256    * Advances to the first match beyond the current whose document number is
257    * greater than or equal to a given target. <br>
258    * When this method is used the {@link #explain(int)} method should not be
259    * used. <br>
260    * The implementation uses the skipTo() method on the subscorers.
261    * 
262    * @param target
263    *          The target document number.
264    * @return the document whose number is greater than or equal to the given
265    *         target, or -1 if none exist.
266    */
267   public int advance(int target) throws IOException {
268     if (scorerDocQueue.size() < minimumNrMatchers) {
269       return currentDoc = NO_MORE_DOCS;
270     }
271     if (target <= currentDoc) {
272       return currentDoc;
273     }
274     do {
275       if (scorerDocQueue.topDoc() >= target) {
276         return advanceAfterCurrent() ? currentDoc : (currentDoc = NO_MORE_DOCS);
277       } else if (!scorerDocQueue.topSkipToAndAdjustElsePop(target)) {
278         if (scorerDocQueue.size() < minimumNrMatchers) {
279           return currentDoc = NO_MORE_DOCS;
280         }
281       }
282     } while (true);
283   }
284   
285   /** @return An explanation for the score of a given document. */
286   public Explanation explain(int doc) throws IOException {
287     Explanation res = new Explanation();
288     Iterator ssi = subScorers.iterator();
289     float sumScore = 0.0f;
290     int nrMatches = 0;
291     while (ssi.hasNext()) {
292       Explanation es = ((Scorer) ssi.next()).explain(doc);
293       if (es.getValue() > 0.0f) { // indicates match
294         sumScore += es.getValue();
295         nrMatches++;
296       }
297       res.addDetail(es);
298     }
299     if (nrMatchers >= minimumNrMatchers) {
300       res.setValue(sumScore);
301       res.setDescription("sum over at least " + minimumNrMatchers
302                          + " of " + subScorers.size() + ":");
303     } else {
304       res.setValue(0.0f);
305       res.setDescription(nrMatches + " match(es) but at least "
306                          + minimumNrMatchers + " of "
307                          + subScorers.size() + " needed");
308     }
309     return res;
310   }
