27  /** A Query that matches documents matching boolean combinations of other
28    * queries, e.g. {@link TermQuery}s, {@link PhraseQuery}s or other
29    * BooleanQuerys.
30    */
31  public class BooleanQuery extends Query {

33    private static int maxClauseCount = 1024;

35    /** Thrown when an attempt is made to add more than {@link
36     * #getMaxClauseCount()} clauses. This typically happens if
37     * a PrefixQuery, FuzzyQuery, WildcardQuery, or TermRangeQuery 
38     * is expanded to many terms during search. 
39     */
40    public static class TooManyClauses extends RuntimeException {
41      public TooManyClauses() {}
42      public String getMessage() {
43        return "maxClauseCount is set to " + maxClauseCount;
44      }
45    }

47    /** Return the maximum number of clauses permitted, 1024 by default.
48     * Attempts to add more than the permitted number of clauses cause {@link
49     * TooManyClauses} to be thrown.
50     * @see #setMaxClauseCount(int)
51     */
52    public static int getMaxClauseCount() { return maxClauseCount; }

54    /** 
55     * Set the maximum number of clauses permitted per BooleanQuery.
56     * Default value is 1024.
57     */
58    public static void setMaxClauseCount(int maxClauseCount) {
59      if (maxClauseCount < 1)
60        throw new IllegalArgumentException("maxClauseCount must be >= 1");
61      BooleanQuery.maxClauseCount = maxClauseCount;
62    }

91    public Similarity getSimilarity(Searcher searcher) {
92      Similarity result = super.getSimilarity(searcher);
93      if (disableCoord) {                           // disable coord as requested
94        result = new SimilarityDelegator(result) {
95            public float coord(int overlap, int maxOverlap) {
96              return 1.0f;
97            }
98          };
99      }
100     return result;

103   /**
104    * Specifies a minimum number of the optional BooleanClauses
105    * which must be satisfied.
106    *
107    * <p>
108    * By default no optional clauses are necessary for a match
109    * (unless there are no required clauses).  If this method is used,
110    * then the specified number of clauses is required.
111    * </p>
112    * <p>
113    * Use of this method is totally independent of specifying that
114    * any specific clauses are required (or prohibited).  This number will
115    * only be compared against the number of matching optional clauses.
116    * </p>
117    * <p>
118    * EXPERT NOTE: Using this method may force collecting docs in order,
119    * regardless of whether setAllowDocsOutOfOrder(true) has been called.
120    * </p>
121    *
122    * @param min the number of optional clauses that must match
123    * @see #setAllowDocsOutOfOrder
124    */
125   public void setMinimumNumberShouldMatch(int min) {
126     this.minNrShouldMatch = min;
127   }

166   /**
167    * Expert: the Weight for BooleanQuery, used to
168    * normalize, score and explain these queries.
169    *
170    * <p>NOTE: this API and implementation is subject to
171    * change suddenly in the next release.</p>
172    */
173   protected class BooleanWeight extends Weight {
174     /** The Similarity implementation. */
175     protected Similarity similarity;
176     protected ArrayList weights;
177 
178     public BooleanWeight(Searcher searcher)
179       throws IOException {
180       this.similarity = getSimilarity(searcher);
181       weights = new ArrayList(clauses.size());
182       for (int i = 0 ; i < clauses.size(); i++) {
183         BooleanClause c = (BooleanClause)clauses.get(i);
184         weights.add(c.getQuery().createWeight(searcher));
185       }
186     }
187 
188     public Query getQuery() { return BooleanQuery.this; }
189     public float getValue() { return getBoost(); }
190 
191     public float sumOfSquaredWeights() throws IOException {
192       float sum = 0.0f;
193       for (int i = 0 ; i < weights.size(); i++) {
194         BooleanClause c = (BooleanClause)clauses.get(i);
195         Weight w = (Weight)weights.get(i);
196         // call sumOfSquaredWeights for all clauses in case of side effects
197         float s = w.sumOfSquaredWeights();         // sum sub weights
198         if (!c.isProhibited())
199           // only add to sum for non-prohibited clauses
200           sum += s;
201       }
202 
203       sum *= getBoost() * getBoost();             // boost each sub-weight
204 
205       return sum ;
206     }

209     public void normalize(float norm) {
210       norm *= getBoost();                         // incorporate boost
211       for (Iterator iter = weights.iterator(); iter.hasNext();) {
212         Weight w = (Weight) iter.next();
213         // normalize all clauses, (even if prohibited in case of side affects)
214         w.normalize(norm);
215       }
216     }

218     public Explanation explain(IndexReader reader, int doc)
219       throws IOException {
220       final int minShouldMatch =
221         BooleanQuery.this.getMinimumNumberShouldMatch();
222       ComplexExplanation sumExpl = new ComplexExplanation();
223       sumExpl.setDescription("sum of:");
224       int coord = 0;
225       int maxCoord = 0;
226       float sum = 0.0f;
227       boolean fail = false;
228       int shouldMatchCount = 0;
229       for (Iterator wIter = weights.iterator(), cIter = clauses.iterator(); wIter.hasNext();) {
230         Weight w = (Weight) wIter.next();
231         BooleanClause c = (BooleanClause) cIter.next();
232         if (w.scorer(reader, true, true) == null) {
233           continue;
234         }
235         Explanation e = w.explain(reader, doc);
236         if (!c.isProhibited()) maxCoord++;
237         if (e.isMatch()) {
238           if (!c.isProhibited()) {
239             sumExpl.addDetail(e);
240             sum += e.getValue();
241             coord++;
242           } else {
243             Explanation r =
244               new Explanation(0.0f, "match on prohibited clause (" + c.getQuery().toString() + ")");
245             r.addDetail(e);
246             sumExpl.addDetail(r);
247             fail = true;
248           }
249           if (c.getOccur() == Occur.SHOULD)
250             shouldMatchCount++;
251         } else if (c.isRequired()) {
252           Explanation r = new Explanation(0.0f, "no match on required clause (" + c.getQuery().toString() + ")");
253           r.addDetail(e);
254           sumExpl.addDetail(r);
255           fail = true;
256         }
257       }
258       if (fail) {
259         sumExpl.setMatch(Boolean.FALSE);
260         sumExpl.setValue(0.0f);
261         sumExpl.setDescription
262           ("Failure to meet condition(s) of required/prohibited clause(s)");
263         return sumExpl;
264       } else if (shouldMatchCount < minShouldMatch) {
265         sumExpl.setMatch(Boolean.FALSE);
266         sumExpl.setValue(0.0f);
267         sumExpl.setDescription("Failure to match minimum number "+
268                                "of optional clauses: " + minShouldMatch);
269         return sumExpl;
270       }
271       
272       sumExpl.setMatch(0 < coord ? Boolean.TRUE : Boolean.FALSE);
273       sumExpl.setValue(sum);
274       
275       float coordFactor = similarity.coord(coord, maxCoord);
276       if (coordFactor == 1.0f)                      // coord is no-op
277         return sumExpl;                             // eliminate wrapper
278       else {
279         ComplexExplanation result = new ComplexExplanation(sumExpl.isMatch(),
280                                                            sum*coordFactor,
281                                                            "product of:");
282         result.addDetail(sumExpl);
283         result.addDetail(new Explanation(coordFactor,
284                                          "coord("+coord+"/"+maxCoord+")"));
285         return result;
286       }
287     }

289     public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer)
290         throws IOException {
291       List required = new ArrayList();
292       List prohibited = new ArrayList();
293       List optional = new ArrayList();
294       for (Iterator wIter = weights.iterator(), cIter = clauses.iterator(); wIter.hasNext();) {
295         Weight w = (Weight) wIter.next();
296         BooleanClause c = (BooleanClause) cIter.next();
297         Scorer subScorer = w.scorer(reader, true, false);
298         if (subScorer == null) {
299           if (c.isRequired()) {
300             return null;
301           }
302         } else if (c.isRequired()) {
303           required.add(subScorer);
304         } else if (c.isProhibited()) {
305           prohibited.add(subScorer);
306         } else {
307           optional.add(subScorer);
308         }
309       }
310       
311       // Check if we can return a BooleanScorer
312       scoreDocsInOrder |= !allowDocsOutOfOrder; // until it is removed, factor in the static setting.
313       if (!scoreDocsInOrder && topScorer && required.size() == 0 && prohibited.size() < 32) {
314         return new BooleanScorer(similarity, minNrShouldMatch, optional, prohibited);
315       }
316       
317       if (required.size() == 0 && optional.size() == 0) {
318         // no required and optional clauses.
319         return null;
320       } else if (optional.size() < minNrShouldMatch) {
321         // either >1 req scorer, or there are 0 req scorers and at least 1
322         // optional scorer. Therefore if there are not enough optional scorers
323         // no documents will be matched by the query
324         return null;
325       }
326       
327       // Return a BooleanScorer2
328       return new BooleanScorer2(similarity, minNrShouldMatch, required, prohibited, optional);
329     }

331     public boolean scoresDocsOutOfOrder() {
332       int numProhibited = 0;
333       for (Iterator cIter = clauses.iterator(); cIter.hasNext();) {
334         BooleanClause c = (BooleanClause) cIter.next();
335         if (c.isRequired()) {
336           return false; // BS2 (in-order) will be used by scorer()
337         } else if (c.isProhibited()) {
338           ++numProhibited;
339         }
340       }
341       
342       if (numProhibited > 32) { // cannot use BS
343         return false;
344       }
345       
346       // scorer() will return an out-of-order scorer if requested.
347       return true;
348     }

353    * Whether hit docs may be collected out of docid order.
354    * 
355    * @deprecated this will not be needed anymore, as
356    *             {@link Weight#scoresDocsOutOfOrder()} is used.
357    */
358   private static boolean allowDocsOutOfOrder = true;
359 
360   /**
361    * Expert: Indicates whether hit docs may be collected out of docid order.
362    * 
363    * <p>
364    * Background: although the contract of the Scorer class requires that
365    * documents be iterated in order of doc id, this was not true in early
366    * versions of Lucene. Many pieces of functionality in the current Lucene code
367    * base have undefined behavior if this contract is not upheld, but in some
368    * specific simple cases may be faster. (For example: disjunction queries with
369    * less than 32 prohibited clauses; This setting has no effect for other
370    * queries.)
371    * </p>
372    * 
373    * <p>
374    * Specifics: By setting this option to true, docid N might be scored for a
375    * single segment before docid N-1. Across multiple segments, docs may be
376    * scored out of order regardless of this setting - it only applies to scoring
377    * a single segment.
378    * 
379    * Being static, this setting is system wide.
380    * </p>
381    * 
382    * @deprecated this is not needed anymore, as
383    *             {@link Weight#scoresDocsOutOfOrder()} is used.
384    */
385   public static void setAllowDocsOutOfOrder(boolean allow) {
386     allowDocsOutOfOrder = allow;
387   }

418   public Query rewrite(IndexReader reader) throws IOException {
419     if (minNrShouldMatch == 0 && clauses.size() == 1) {                    // optimize 1-clause queries
420       BooleanClause c = (BooleanClause)clauses.get(0);
421       if (!c.isProhibited()) {            // just return clause
422 
423         Query query = c.getQuery().rewrite(reader);    // rewrite first
424 
425         if (getBoost() != 1.0f) {                 // incorporate boost
426           if (query == c.getQuery())                   // if rewrite was no-op
427             query = (Query)query.clone();         // then clone before boost
428           query.setBoost(getBoost() * query.getBoost());
429         }
430 
431         return query;
432       }
433     }
434 
435     BooleanQuery clone = null;                    // recursively rewrite
436     for (int i = 0 ; i < clauses.size(); i++) {
437       BooleanClause c = (BooleanClause)clauses.get(i);
438       Query query = c.getQuery().rewrite(reader);
439       if (query != c.getQuery()) {                     // clause rewrote: must clone
440         if (clone == null)
441           clone = (BooleanQuery)this.clone();
442         clone.clauses.set(i, new BooleanClause(query, c.getOccur()));
443       }
444     }
445     if (clone != null) {
446       return clone;                               // some clauses rewrote
447     } else
448       return this;                                // no clauses rewrote
449   }

465   /** Prints a user-readable version of this query. */
466   public String toString(String field) {
467     StringBuffer buffer = new StringBuffer();
468     boolean needParens=(getBoost() != 1.0) || (getMinimumNumberShouldMatch()>0) ;
469     if (needParens) {
470       buffer.append("(");
471     }
472 
473     for (int i = 0 ; i < clauses.size(); i++) {
474       BooleanClause c = (BooleanClause)clauses.get(i);
475       if (c.isProhibited())
476         buffer.append("-");
477       else if (c.isRequired())
478         buffer.append("+");
479 
480       Query subQuery = c.getQuery();
481       if (subQuery != null) {
482         if (subQuery instanceof BooleanQuery) {   // wrap sub-bools in parens
483           buffer.append("(");
484           buffer.append(subQuery.toString(field));
485           buffer.append(")");
486         } else {
487           buffer.append(subQuery.toString(field));
488         }
489       } else {
490         buffer.append("null");
491       }
492 
493       if (i != clauses.size()-1)
494         buffer.append(" ");
495     }
496 
497     if (needParens) {
498       buffer.append(")");
499     }
500 
501     if (getMinimumNumberShouldMatch()>0) {
502       buffer.append('~');
503       buffer.append(getMinimumNumberShouldMatch());
504     }
505 
506     if (getBoost() != 1.0f)
507     {
508       buffer.append(ToStringUtils.boost(getBoost()));
509     }
510 
511     return buffer.toString();
512   }

514   /** Returns true iff <code>o</code> is equal to this. */
515   public boolean equals(Object o) {
516     if (!(o instanceof BooleanQuery))
517       return false;
518     BooleanQuery other = (BooleanQuery)o;
519     return (this.getBoost() == other.getBoost())
520         && this.clauses.equals(other.clauses)
521         && this.getMinimumNumberShouldMatch() == other.getMinimumNumberShouldMatch()
522         && this.disableCoord == other.disableCoord;
523   }

525   /** Returns a hash code value for this object.*/
526   public int hashCode() {
527     return Float.floatToIntBits(getBoost()) ^ clauses.hashCode()
528       + getMinimumNumberShouldMatch() + (disableCoord ? 17:0);
529   }
