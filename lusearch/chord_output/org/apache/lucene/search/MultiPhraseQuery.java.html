29  /**
30   * MultiPhraseQuery is a generalized version of PhraseQuery, with an added
31   * method {@link #add(Term[])}.
32   * To use this class, to search for the phrase "Microsoft app*" first use
33   * add(Term) on the term "Microsoft", then find all terms that have "app" as
34   * prefix using IndexReader.terms(Term), and use MultiPhraseQuery.add(Term[]
35   * terms) to add them to the query.
36   *
37   * @version 1.0
38   */
39  public class MultiPhraseQuery extends Query {
40    private String field;
41    private ArrayList termArrays = new ArrayList();
42    private ArrayList positions = new ArrayList();
43  
44    private int slop = 0;
45  
46    /** Sets the phrase slop for this query.
47     * @see PhraseQuery#setSlop(int)
48     */
49    public void setSlop(int s) { slop = s; }
50  
51    /** Sets the phrase slop for this query.
52     * @see PhraseQuery#getSlop()
53     */
54    public int getSlop() { return slop; }
55  
56    /** Add a single term at the next position in the phrase.
57     * @see PhraseQuery#add(Term)
58     */
59    public void add(Term term) { add(new Term[]{term}); }
60  
61    /** Add multiple terms at the next position in the phrase.  Any of the terms
62     * may match.
63     *
64     * @see PhraseQuery#add(Term)
65     */
66    public void add(Term[] terms) {
67      int position = 0;
68      if (positions.size() > 0)
69        position = ((Integer) positions.get(positions.size()-1)).intValue() + 1;
70  
71      add(terms, position);
72    }
73  
74    /**
75     * Allows to specify the relative position of terms within the phrase.
76     * 
77     * @see PhraseQuery#add(Term, int)
78     * @param terms
79     * @param position
80     */
81    public void add(Term[] terms, int position) {
82      if (termArrays.size() == 0)
83        field = terms[0].field();
84  
85      for (int i = 0; i < terms.length; i++) {
86        if (terms[i].field() != field) {
87          throw new IllegalArgumentException(
88              "All phrase terms must be in the same field (" + field + "): "
89                  + terms[i]);
90        }
91      }
92  
93      termArrays.add(terms);
94      positions.add(new Integer(position));
95    }
96  
97    /**
98     * Returns a List<Term[]> of the terms in the multiphrase.
99     * Do not modify the List or its contents.
100    */
101   public List getTermArrays() {
102       return Collections.unmodifiableList(termArrays);
103   }
104 
105   /**
106    * Returns the relative positions of terms in this phrase.
107    */
108   public int[] getPositions() {
109     int[] result = new int[positions.size()];
110     for (int i = 0; i < positions.size(); i++)
111       result[i] = ((Integer) positions.get(i)).intValue();
112     return result;
113   }
114 
115   // inherit javadoc
116   public void extractTerms(Set terms) {
117     for (Iterator iter = termArrays.iterator(); iter.hasNext();) {
118       Term[] arr = (Term[])iter.next();
119       for (int i=0; i<arr.length; i++) {
120         terms.add(arr[i]);
121       }
122     }
123   }

126   private class MultiPhraseWeight extends Weight {
127     private Similarity similarity;
128     private float value;
129     private float idf;
130     private float queryNorm;
131     private float queryWeight;
132 
133     public MultiPhraseWeight(Searcher searcher)
134       throws IOException {
135       this.similarity = getSimilarity(searcher);
136 
137       // compute idf
138       Iterator i = termArrays.iterator();
139       while (i.hasNext()) {
140         Term[] terms = (Term[])i.next();
141         for (int j=0; j<terms.length; j++) {
142           idf += getSimilarity(searcher).idf(terms[j], searcher);
143         }
144       }
145     }
146 
147     public Query getQuery() { return MultiPhraseQuery.this; }
148     public float getValue() { return value; }
149 
150     public float sumOfSquaredWeights() {
151       queryWeight = idf * getBoost();             // compute query weight
152       return queryWeight * queryWeight;           // square it
153     }
154 
155     public void normalize(float queryNorm) {
156       this.queryNorm = queryNorm;
157       queryWeight *= queryNorm;                   // normalize query weight
158       value = queryWeight * idf;                  // idf for document 
159     }
160 
161     public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer) throws IOException {
162       if (termArrays.size() == 0)                  // optimize zero-term case
163         return null;
164 
165       TermPositions[] tps = new TermPositions[termArrays.size()];
166       for (int i=0; i<tps.length; i++) {
167         Term[] terms = (Term[])termArrays.get(i);
168 
169         TermPositions p;
170         if (terms.length > 1)
171           p = new MultipleTermPositions(reader, terms);
172         else
173           p = reader.termPositions(terms[0]);
174 
175         if (p == null)
176           return null;
177 
178         tps[i] = p;
179       }
180 
181       if (slop == 0)
182         return new ExactPhraseScorer(this, tps, getPositions(), similarity,
183                                      reader.norms(field));
184       else
185         return new SloppyPhraseScorer(this, tps, getPositions(), similarity,
186                                       slop, reader.norms(field));
187     }
188 
189     public Explanation explain(IndexReader reader, int doc)
190       throws IOException {
191       ComplexExplanation result = new ComplexExplanation();
192       result.setDescription("weight("+getQuery()+" in "+doc+"), product of:");
193 
194       Explanation idfExpl = new Explanation(idf, "idf("+getQuery()+")");
195 
196       // explain query weight
197       Explanation queryExpl = new Explanation();
198       queryExpl.setDescription("queryWeight(" + getQuery() + "), product of:");
199 
200       Explanation boostExpl = new Explanation(getBoost(), "boost");
201       if (getBoost() != 1.0f)
202         queryExpl.addDetail(boostExpl);
203 
204       queryExpl.addDetail(idfExpl);
205 
206       Explanation queryNormExpl = new Explanation(queryNorm,"queryNorm");
207       queryExpl.addDetail(queryNormExpl);
208 
209       queryExpl.setValue(boostExpl.getValue() *
210                          idfExpl.getValue() *
211                          queryNormExpl.getValue());
212 
213       result.addDetail(queryExpl);
214 
215       // explain field weight
216       ComplexExplanation fieldExpl = new ComplexExplanation();
217       fieldExpl.setDescription("fieldWeight("+getQuery()+" in "+doc+
218                                "), product of:");
219 
220       Scorer scorer = scorer(reader, true, false);
221       if (scorer == null) {
222         return new Explanation(0.0f, "no matching docs");
223       }
224       Explanation tfExpl = scorer.explain(doc);
225       fieldExpl.addDetail(tfExpl);
226       fieldExpl.addDetail(idfExpl);
227 
228       Explanation fieldNormExpl = new Explanation();
229       byte[] fieldNorms = reader.norms(field);
230       float fieldNorm =
231         fieldNorms!=null ? Similarity.decodeNorm(fieldNorms[doc]) : 1.0f;
232       fieldNormExpl.setValue(fieldNorm);
233       fieldNormExpl.setDescription("fieldNorm(field="+field+", doc="+doc+")");
234       fieldExpl.addDetail(fieldNormExpl);
235 
236       fieldExpl.setMatch(Boolean.valueOf(tfExpl.isMatch()));
237       fieldExpl.setValue(tfExpl.getValue() *
238                          idfExpl.getValue() *
239                          fieldNormExpl.getValue());
240 
241       result.addDetail(fieldExpl);
242       result.setMatch(fieldExpl.getMatch());
243 
244       // combine them
245       result.setValue(queryExpl.getValue() * fieldExpl.getValue());
246 
247       if (queryExpl.getValue() == 1.0f)
248         return fieldExpl;
249 
250       return result;
251     }
252   }

254   public Query rewrite(IndexReader reader) {
255     if (termArrays.size() == 1) {                 // optimize one-term case
256       Term[] terms = (Term[])termArrays.get(0);
257       BooleanQuery boq = new BooleanQuery(true);
258       for (int i=0; i<terms.length; i++) {
259         boq.add(new TermQuery(terms[i]), BooleanClause.Occur.SHOULD);
260       }
261       boq.setBoost(getBoost());
262       return boq;
263     } else {
264       return this;
265     }
266   }
267 
268   public Weight createWeight(Searcher searcher) throws IOException {
269     return new MultiPhraseWeight(searcher);
270   }

272   /** Prints a user-readable version of this query. */
273   public final String toString(String f) {
274     StringBuffer buffer = new StringBuffer();
275     if (!field.equals(f)) {
276       buffer.append(field);
277       buffer.append(":");
278     }
279 
280     buffer.append("\"");
281     Iterator i = termArrays.iterator();
282     while (i.hasNext()) {
283       Term[] terms = (Term[])i.next();
284       if (terms.length > 1) {
285         buffer.append("(");
286         for (int j = 0; j < terms.length; j++) {
287           buffer.append(terms[j].text());
288           if (j < terms.length-1)
289             buffer.append(" ");
290         }
291         buffer.append(")");
292       } else {
293         buffer.append(terms[0].text());
294       }
295       if (i.hasNext())
296         buffer.append(" ");
297     }
298     buffer.append("\"");
299 
300     if (slop != 0) {
301       buffer.append("~");
302       buffer.append(slop);
303     }
304 
305     buffer.append(ToStringUtils.boost(getBoost()));
306 
307     return buffer.toString();
308   }

311   /** Returns true if <code>o</code> is equal to this. */
312   public boolean equals(Object o) {
313     if (!(o instanceof MultiPhraseQuery)) return false;
314     MultiPhraseQuery other = (MultiPhraseQuery)o;
315     return this.getBoost() == other.getBoost()
316       && this.slop == other.slop
317       && termArraysEquals(this.termArrays, other.termArrays)
318       && this.positions.equals(other.positions);
319   }

321   /** Returns a hash code value for this object.*/
322   public int hashCode() {
323     return Float.floatToIntBits(getBoost())
324       ^ slop
325       ^ termArraysHashCode()
326       ^ positions.hashCode()
327       ^ 0x4AC65113;
328   }

330   // Breakout calculation of the termArrays hashcode
331   private int termArraysHashCode() {
332     int hashCode = 1;
333     Iterator iterator = termArrays.iterator();
334     while (iterator.hasNext()) {
335       Term[] termArray = (Term[]) iterator.next();
336       hashCode = 31 * hashCode
337           + (termArray == null ? 0 : arraysHashCode(termArray));
338     }
339     return hashCode;
340   }
341 
342   private int arraysHashCode(Term[] termArray) {
343       if (termArray == null)
344           return 0;
345 
346       int result = 1;
347 
348       for (int i = 0; i < termArray.length; i++) {
349         Term term = termArray[i];
350         result = 31 * result + (term == null ? 0 : term.hashCode());
351       }
352 
353       return result;
354   }
355 
356   // Breakout calculation of the termArrays equals
357   private boolean termArraysEquals(List termArrays1, List termArrays2) {
358     if (termArrays1.size() != termArrays2.size()) {
359       return false;
360     }
361     ListIterator iterator1 = termArrays1.listIterator();
362     ListIterator iterator2 = termArrays2.listIterator();
363     while (iterator1.hasNext()) {
364       Term[] termArray1 = (Term[]) iterator1.next();
365       Term[] termArray2 = (Term[]) iterator2.next();
366       if (!(termArray1 == null ? termArray2 == null : Arrays.equals(termArray1,
367           termArray2))) {
368         return false;
369       }
370     }
371     return true;
372   }
