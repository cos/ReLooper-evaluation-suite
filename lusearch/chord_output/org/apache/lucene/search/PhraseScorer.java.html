20  import java.io.IOException;
21  
22  import org.apache.lucene.index.TermPositions;
23  
24  /** Expert: Scoring functionality for phrase queries.
25   * <br>A document is considered matching if it contains the phrase-query terms  
26   * at "valid" positions. What "valid positions" are
27   * depends on the type of the phrase query: for an exact phrase query terms are required 
28   * to appear in adjacent locations, while for a sloppy phrase query some distance between 
29   * the terms is allowed. The abstract method {@link #phraseFreq()} of extending classes
30   * is invoked for each document containing all the phrase query terms, in order to 
31   * compute the frequency of the phrase query in that document. A non zero frequency
32   * means a match. 
33   */
34  abstract class PhraseScorer extends Scorer {
35    private Weight weight;
36    protected byte[] norms;
37    protected float value;
38  
39    private boolean firstTime = true;
40    private boolean more = true;
41    protected PhraseQueue pq;
42    protected PhrasePositions first, last;
43  
44    private float freq; //phrase frequency in current doc as computed by phraseFreq().
45  
46    PhraseScorer(Weight weight, TermPositions[] tps, int[] offsets,
47        Similarity similarity, byte[] norms) {
48      super(similarity);
49      this.norms = norms;
50      this.weight = weight;
51      this.value = weight.getValue();
52  
53      // convert tps to a list of phrase positions.
54      // note: phrase-position differs from term-position in that its position
55      // reflects the phrase offset: pp.pos = tp.pos - offset.
56      // this allows to easily identify a matching (exact) phrase 
57      // when all PhrasePositions have exactly the same position.
58      for (int i = 0; i < tps.length; i++) {
59        PhrasePositions pp = new PhrasePositions(tps[i], offsets[i]);
60        if (last != null) {             // add next to end of list
61          last.next = pp;
62        } else {
63          first = pp;
64        }
65        last = pp;
66      }
67  
68      pq = new PhraseQueue(tps.length);             // construct empty pq
69      first.doc = -1;
70    }
71  
72    /** @deprecated use {@link #docID()} instead. */
73    public int doc() { return first.doc; }
74    
75    public int docID() { return first.doc; }
76  
77    /** @deprecated use {@link #nextDoc()} instead. */
78    public boolean next() throws IOException {
79      return nextDoc() != NO_MORE_DOCS;
80    }
81  
82    public int nextDoc() throws IOException {
83      if (firstTime) {
84        init();
85        firstTime = false;
86      } else if (more) {
87        more = last.next();                         // trigger further scanning
88      }
89      if (!doNext()) {
90        first.doc = NO_MORE_DOCS;
91      }
92      return first.doc;
93    }
94    
95    // next without initial increment
96    private boolean doNext() throws IOException {
97      while (more) {
98        while (more && first.doc < last.doc) {      // find doc w/ all the terms
99          more = first.skipTo(last.doc);            // skip first upto last
100         firstToLast();                            // and move it to the end
101       }
102 
103       if (more) {
104         // found a doc with all of the terms
105         freq = phraseFreq();                      // check for phrase
106         if (freq == 0.0f)                         // no match
107           more = last.next();                     // trigger further scanning
108         else
109           return true;                            // found a match
110       }
111     }
112     return false;                                 // no more matches
113   }
114 
115   public float score() throws IOException {
116     //System.out.println("scoring " + first.doc);
117     float raw = getSimilarity().tf(freq) * value; // raw score
118     return norms == null ? raw : raw * Similarity.decodeNorm(norms[first.doc]); // normalize
119   }
120 
121   /** @deprecated use {@link #advance(int)} instead. */
122   public boolean skipTo(int target) throws IOException {
123     return advance(target) != NO_MORE_DOCS;
124   }
125 
126   public int advance(int target) throws IOException {
127     firstTime = false;
128     for (PhrasePositions pp = first; more && pp != null; pp = pp.next) {
129       more = pp.skipTo(target);
130     }
131     if (more) {
132       sort();                                     // re-sort
133     }
134     if (!doNext()) {
135       first.doc = NO_MORE_DOCS;
136     }
137     return first.doc;
138   }
139   
140   /**
141    * For a document containing all the phrase query terms, compute the
142    * frequency of the phrase in that document. 
143    * A non zero frequency means a match.
144    * <br>Note, that containing all phrase terms does not guarantee a match - they have to be found in matching locations.  
145    * @return frequency of the phrase in current doc, 0 if not found. 
146    */
147   protected abstract float phraseFreq() throws IOException;
148 
149   private void init() throws IOException {
150     for (PhrasePositions pp = first; more && pp != null; pp = pp.next) {
151       more = pp.next();
152     }
153     if (more) {
154       sort();
155     }
156   }
157   
158   private void sort() {
159     pq.clear();
160     for (PhrasePositions pp = first; pp != null; pp = pp.next) {
161       pq.add(pp);
162     }
163     pqToList();
164   }
165 
166   protected final void pqToList() {
167     last = first = null;
168     while (pq.top() != null) {
169       PhrasePositions pp = (PhrasePositions) pq.pop();
170       if (last != null) {             // add next to end of list
171         last.next = pp;
172       } else
173         first = pp;
174       last = pp;
175       pp.next = null;
176     }
177   }
178 
179   protected final void firstToLast() {
180     last.next = first;            // move first to end of list
181     last = first;
182     first = first.next;
183     last.next = null;
184   }
185 
186   public Explanation explain(final int doc) throws IOException {
187     Explanation tfExplanation = new Explanation();
188 
189     int d = advance(doc);
190     float phraseFreq = (d == doc) ? freq : 0.0f;
191     tfExplanation.setValue(getSimilarity().tf(phraseFreq));
192     tfExplanation.setDescription("tf(phraseFreq=" + phraseFreq + ")");
193 
194     return tfExplanation;
195   }
196 
197   public String toString() { return "scorer(" + weight + ")"; }
198 
199 }
