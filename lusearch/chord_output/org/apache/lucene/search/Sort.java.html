20  import java.io.Serializable;
21  import java.util.Arrays;
22  
23  
24  /**
25   * Encapsulates sort criteria for returned hits.
26   *
27   * <p>The fields used to determine sort order must be carefully chosen.
28   * Documents must contain a single term in such a field,
29   * and the value of the term should indicate the document's relative position in
30   * a given sort order.  The field must be indexed, but should not be tokenized,
31   * and does not need to be stored (unless you happen to want it back with the
32   * rest of your document data).  In other words:
33   *
34   * <p><code>document.add (new Field ("byNumber", Integer.toString(x), Field.Store.NO, Field.Index.NOT_ANALYZED));</code></p>
35   * 
36   *
37   * <p><h3>Valid Types of Values</h3>
38   *
39   * <p>There are four possible kinds of term values which may be put into
40   * sorting fields: Integers, Longs, Floats, or Strings.  Unless
41   * {@link SortField SortField} objects are specified, the type of value
42   * in the field is determined by parsing the first term in the field.
43   *
44   * <p>Integer term values should contain only digits and an optional
45   * preceding negative sign.  Values must be base 10 and in the range
46   * <code>Integer.MIN_VALUE</code> and <code>Integer.MAX_VALUE</code> inclusive.
47   * Documents which should appear first in the sort
48   * should have low value integers, later documents high values
49   * (i.e. the documents should be numbered <code>1..n</code> where
50   * <code>1</code> is the first and <code>n</code> the last).
51   *
52   * <p>Long term values should contain only digits and an optional
53   * preceding negative sign.  Values must be base 10 and in the range
54   * <code>Long.MIN_VALUE</code> and <code>Long.MAX_VALUE</code> inclusive.
55   * Documents which should appear first in the sort
56   * should have low value integers, later documents high values.
57   * 
58   * <p>Float term values should conform to values accepted by
59   * {@link Float Float.valueOf(String)} (except that <code>NaN</code>
60   * and <code>Infinity</code> are not supported).
61   * Documents which should appear first in the sort
62   * should have low values, later documents high values.
63   *
64   * <p>String term values can contain any valid String, but should
65   * not be tokenized.  The values are sorted according to their
66   * {@link Comparable natural order}.  Note that using this type
67   * of term value has higher memory requirements than the other
68   * two types.
69   *
70   * <p><h3>Object Reuse</h3>
71   *
72   * <p>One of these objects can be
73   * used multiple times and the sort order changed between usages.
74   *
75   * <p>This class is thread safe.
76   *
77   * <p><h3>Memory Usage</h3>
78   *
79   * <p>Sorting uses of caches of term values maintained by the
80   * internal HitQueue(s).  The cache is static and contains an integer
81   * or float array of length <code>IndexReader.maxDoc()</code> for each field
82   * name for which a sort is performed.  In other words, the size of the
83   * cache in bytes is:
84   *
85   * <p><code>4 * IndexReader.maxDoc() * (# of different fields actually used to sort)</code>
86   *
87   * <p>For String fields, the cache is larger: in addition to the
88   * above array, the value of every term in the field is kept in memory.
89   * If there are many unique terms in the field, this could
90   * be quite large.
91   *
92   * <p>Note that the size of the cache is not affected by how many
93   * fields are in the index and <i>might</i> be used to sort - only by
94   * the ones actually used to sort a result set.
95   *
96   * <p>Created: Feb 12, 2004 10:53:57 AM
97   *
98   * @since   lucene 1.4
99   * @version $Id: Sort.java 795179 2009-07-17 18:23:30Z mikemccand $
100  */
101 public class Sort
102 implements Serializable {
103 
104   /**
105    * Represents sorting by computed relevance. Using this sort criteria returns
106    * the same results as calling
107    * {@link Searcher#search(Query) Searcher#search()}without a sort criteria,
108    * only with slightly more overhead.
109    */
110   public static final Sort RELEVANCE = new Sort();
111 
112   /** Represents sorting by index order. */
113   public static final Sort INDEXORDER = new Sort(SortField.FIELD_DOC);
114 
115   // internal representation of the sort criteria
116   SortField[] fields;
117 
118   /**
119    * Sorts by computed relevance. This is the same sort criteria as calling
120    * {@link Searcher#search(Query) Searcher#search()}without a sort criteria,
121    * only with slightly more overhead.
122    */
123   public Sort() {
124     this(SortField.FIELD_SCORE);
125   }
126 
127   /**
128    * Sorts by the terms in <code>field</code> then by index order (document
129    * number). The type of value in <code>field</code> is determined
130    * automatically.
131    * 
132    * @see SortField#AUTO
133    * @deprecated Please specify the type explicitly by
134    * first creating a {@link SortField} and then use {@link
135    * #Sort(SortField)}
136    */
137   public Sort(String field) {
138     setSort(field, false);
139   }
140 
141   /**
142    * Sorts possibly in reverse by the terms in <code>field</code> then by
143    * index order (document number). The type of value in <code>field</code> is
144    * determined automatically.
145    * 
146    * @see SortField#AUTO
147    * @deprecated Please specify the type explicitly by
148    * first creating a {@link SortField} and then use {@link
149    * #Sort(SortField)}
150    */
151   public Sort(String field, boolean reverse) {
152     setSort(field, reverse);
153   }
154 
155   /**
156    * Sorts in succession by the terms in each field. The type of value in
157    * <code>field</code> is determined automatically.
158    * 
159    * @see SortField#AUTO
160    * @deprecated Please specify the type explicitly by
161    * first creating {@link SortField}s and then use {@link
162    * #Sort(SortField[])}
163    */
164   public Sort(String[] fields) {
165     setSort(fields);
166   }
167 
168   /** Sorts by the criteria in the given SortField. */
169   public Sort(SortField field) {
170     setSort(field);
171   }
172 
173   /** Sorts in succession by the criteria in each SortField. */
174   public Sort(SortField[] fields) {
175     setSort(fields);
176   }
177 
178   /**
179    * Sets the sort to the terms in <code>field</code> then by index order
180    * (document number).
181    * @deprecated Please specify the type explicitly by
182    * first creating a {@link SortField} and then use {@link
183    * #setSort(SortField)}
184    */
185   public final void setSort(String field) {
186     setSort(field, false);
187   }
188 
189   /**
190    * Sets the sort to the terms in <code>field</code> possibly in reverse,
191    * then by index order (document number).
192    * @deprecated Please specify the type explicitly by
193    * first creating a {@link SortField} and then use {@link
194    * #setSort(SortField)}
195    */
196   public void setSort(String field, boolean reverse) {
197     fields = new SortField[] { new SortField(field, SortField.AUTO, reverse) };
198   }
199 
200   /** Sets the sort to the terms in each field in succession.
201    * @deprecated Please specify the type explicitly by
202    * first creating {@link SortField}s and then use {@link
203    * #setSort(SortField[])} */
204   public void setSort(String[] fieldnames) {
205     final int n = fieldnames.length;
206     SortField[] nfields = new SortField[n];
207     for (int i = 0; i < n; ++i) {
208       nfields[i] = new SortField(fieldnames[i], SortField.AUTO);
209     }
210     fields = nfields;
211   }
212 
213   /** Sets the sort to the given criteria. */
214   public void setSort(SortField field) {
215     this.fields = new SortField[] { field };
216   }
217 
218   /** Sets the sort to the given criteria in succession. */
219   public void setSort(SortField[] fields) {
220     this.fields = fields;
221   }
222   
223   /**
224    * Representation of the sort criteria.
225    * @return Array of SortField objects used in this sort criteria
226    */
227   public SortField[] getSort() {
228     return fields;
229   }
230 
231   public String toString() {
232     StringBuffer buffer = new StringBuffer();
233 
234     for (int i = 0; i < fields.length; i++) {
235       buffer.append(fields[i].toString());
236       if ((i+1) < fields.length)
237         buffer.append(',');
238     }
239 
240     return buffer.toString();
241   }
242 
243   /** Returns true if <code>o</code> is equal to this. */
244   public boolean equals(Object o) {
245     if (this == o) return true;
246     if (!(o instanceof Sort)) return false;
247     final Sort other = (Sort)o;
248     return Arrays.equals(this.fields, other.fields);
249   }
250 
251   /** Returns a hash code value for this object. */
252   public int hashCode() {
253     // TODO in Java 1.5: switch to Arrays.hashCode().  The 
254     // Java 1.4 workaround below calculates the same hashCode
255     // as Java 1.5's new Arrays.hashCode()
256     return 0x45aaf665 + Arrays.asList(fields).hashCode();
257   }
258 }
