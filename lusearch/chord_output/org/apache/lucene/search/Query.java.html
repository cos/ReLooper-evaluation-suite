28  /** The abstract base class for queries.
29      <p>Instantiable subclasses are:
30      <ul>
31      <li> {@link TermQuery}
32      <li> {@link MultiTermQuery}
33      <li> {@link BooleanQuery}
34      <li> {@link WildcardQuery}
35      <li> {@link PhraseQuery}
36      <li> {@link PrefixQuery}
37      <li> {@link MultiPhraseQuery}
38      <li> {@link FuzzyQuery}
39      <li> {@link TermRangeQuery}
40      <li> {@link NumericRangeQuery}
41      <li> {@link org.apache.lucene.search.spans.SpanQuery}
42      </ul>
43      <p>A parser for queries is contained in:
44      <ul>
45      <li>{@link org.apache.lucene.queryParser.QueryParser QueryParser}
46      </ul>
47  */
48  public abstract class Query implements java.io.Serializable, Cloneable {
49    private float boost = 1.0f;                     // query boost factor
50  
51    /** Sets the boost for this query clause to <code>b</code>.  Documents
52     * matching this clause will (in addition to the normal weightings) have
53     * their score multiplied by <code>b</code>.
54     */
55    public void setBoost(float b) { boost = b; }
56  
57    /** Gets the boost for this clause.  Documents matching
58     * this clause will (in addition to the normal weightings) have their score
59     * multiplied by <code>b</code>.   The boost is 1.0 by default.
60     */
61    public float getBoost() { return boost; }
62  
63    /** Prints a query to a string, with <code>field</code> assumed to be the 
64     * default field and omitted.
65     * <p>The representation used is one that is supposed to be readable
66     * by {@link org.apache.lucene.queryParser.QueryParser QueryParser}. However,
67     * there are the following limitations:
68     * <ul>
69     *  <li>If the query was created by the parser, the printed
70     *  representation may not be exactly what was parsed. For example,
71     *  characters that need to be escaped will be represented without
72     *  the required backslash.</li>
73     * <li>Some of the more complicated queries (e.g. span queries)
74     *  don't have a representation that can be parsed by QueryParser.</li>
75     * </ul>
76     */
77    public abstract String toString(String field);
78  
79    /** Prints a query to a string. */
80    public String toString() {
81      return toString("");
82    }
83  
84    /**
85     * Expert: Constructs an appropriate Weight implementation for this query.
86     * 
87     * <p>
88     * Only implemented by primitive queries, which re-write to themselves.
89     */
90    public Weight createWeight(Searcher searcher) throws IOException {
91      throw new UnsupportedOperationException();
92    }
93  
94    /**
95     * Expert: Constructs and initializes a Weight for a top-level query.
96     */
97    public Weight weight(Searcher searcher) throws IOException {
98      Query query = searcher.rewrite(this);
99      Weight weight = query.createWeight(searcher);
100     float sum = weight.sumOfSquaredWeights();
101     float norm = getSimilarity(searcher).queryNorm(sum);
102     if (Float.isInfinite(norm) || Float.isNaN(norm))
103       norm = 1.0f;
104     weight.normalize(norm);
105     return weight;
106   }

109   /** Expert: called to re-write queries into primitive queries. For example,
110    * a PrefixQuery will be rewritten into a BooleanQuery that consists
111    * of TermQuerys.
112    */
113   public Query rewrite(IndexReader reader) throws IOException {
114     return this;
115   }

118   /** Expert: called when re-writing queries under MultiSearcher.
119    *
120    * Create a single query suitable for use by all subsearchers (in 1-1
121    * correspondence with queries). This is an optimization of the OR of
122    * all queries. We handle the common optimization cases of equal
123    * queries and overlapping clauses of boolean OR queries (as generated
124    * by MultiTermQuery.rewrite()).
125    * Be careful overriding this method as queries[0] determines which
126    * method will be called and is not necessarily of the same type as
127    * the other queries.
128   */
129   public Query combine(Query[] queries) {
130     HashSet uniques = new HashSet();
131     for (int i = 0; i < queries.length; i++) {
132       Query query = queries[i];
133       BooleanClause[] clauses = null;
134       // check if we can split the query into clauses
135       boolean splittable = (query instanceof BooleanQuery);
136       if(splittable){
137         BooleanQuery bq = (BooleanQuery) query;
138         splittable = bq.isCoordDisabled();
139         clauses = bq.getClauses();
140         for (int j = 0; splittable && j < clauses.length; j++) {
141           splittable = (clauses[j].getOccur() == BooleanClause.Occur.SHOULD);
142         }
143       }
144       if(splittable){
145         for (int j = 0; j < clauses.length; j++) {
146           uniques.add(clauses[j].getQuery());
147         }
148       } else {
149         uniques.add(query);
150       }
151     }
152     // optimization: if we have just one query, just return it
153     if(uniques.size() == 1){
154         return (Query)uniques.iterator().next();
155     }
156     Iterator it = uniques.iterator();
157     BooleanQuery result = new BooleanQuery(true);
158     while (it.hasNext())
159       result.add((Query) it.next(), BooleanClause.Occur.SHOULD);
160     return result;
161   }

165    * Expert: adds all terms occurring in this query to the terms set. Only
166    * works if this query is in its {@link #rewrite rewritten} form.
167    * 
168    * @throws UnsupportedOperationException if this query is not yet rewritten
169    */
170   public void extractTerms(Set terms) {
171     // needs to be implemented by query subclasses
172     throw new UnsupportedOperationException();
173   }

177   /** Expert: merges the clauses of a set of BooleanQuery's into a single
178    * BooleanQuery.
179    *
180    *<p>A utility for use by {@link #combine(Query[])} implementations.
181    */
182   public static Query mergeBooleanQueries(BooleanQuery[] queries) {
183     HashSet allClauses = new HashSet();
184     for (int i = 0; i < queries.length; i++) {
185       BooleanClause[] clauses = queries[i].getClauses();
186       for (int j = 0; j < clauses.length; j++) {
187         allClauses.add(clauses[j]);
188       }
189     }
190 
191     boolean coordDisabled =
192       queries.length==0? false : queries[0].isCoordDisabled();
193     BooleanQuery result = new BooleanQuery(coordDisabled);
194     Iterator i = allClauses.iterator();
195     while (i.hasNext()) {
196       result.add((BooleanClause)i.next());
197     }
198     return result;
199   }

202   /** Expert: Returns the Similarity implementation to be used for this query.
203    * Subclasses may override this method to specify their own Similarity
204    * implementation, perhaps one that delegates through that of the Searcher.
205    * By default the Searcher's Similarity implementation is returned.*/
206   public Similarity getSimilarity(Searcher searcher) {
207     return searcher.getSimilarity();
208   }
209 
210   /** Returns a clone of this query. */
211   public Object clone() {
212     try {
213       return super.clone();
214     } catch (CloneNotSupportedException e) {
215       throw new RuntimeException("Clone not supported: " + e.getMessage());
216     }
217   }
218 
219   public int hashCode() {
220     final int prime = 31;
221     int result = 1;
222     result = prime * result + Float.floatToIntBits(boost);
223     return result;
224   }
225 
226   public boolean equals(Object obj) {
227     if (this == obj)
228       return true;
229     if (obj == null)
230       return false;
231     if (getClass() != obj.getClass())
232       return false;
233     Query other = (Query) obj;
234     if (Float.floatToIntBits(boost) != Float.floatToIntBits(other.boost))
235       return false;
236     return true;
237   }
