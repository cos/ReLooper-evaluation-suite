20  import java.text.Collator;
21  import java.io.IOException;
22  
23  import org.apache.lucene.index.IndexReader;
24  import org.apache.lucene.index.Term;
25  
26  /**
27   * A Query that matches documents within an exclusive range of terms.
28   *
29   * <p>This query matches the documents looking for terms that fall into the
30   * supplied range according to {@link Term#compareTo(Term)}. It is not intended
31   * for numerical ranges, use {@link NumericRangeQuery} instead.
32   *
33   * <p>This query uses {@linkplain
34   * MultiTermQuery#SCORING_BOOLEAN_QUERY_REWRITE}.  If you
35   * want to change this, use the new {@link TermRangeQuery}
36   * instead.
37   *
38   * @deprecated Use {@link TermRangeQuery} for term ranges or
39   * {@link NumericRangeQuery} for numeric ranges instead.
40   * This class will be removed in Lucene 3.0.
41   */
42  public class RangeQuery extends Query {
43    private final TermRangeQuery delegate;
44  
45    /** Constructs a query selecting all terms greater than
46     * <code>lowerTerm</code> but less than <code>upperTerm</code>.
47     * There must be at least one term and either term may be null,
48     * in which case there is no bound on that side, but if there are
49     * two terms, both terms <b>must</b> be for the same field.
50     *
51     * @param lowerTerm The Term at the lower end of the range
52     * @param upperTerm The Term at the upper end of the range
53     * @param inclusive If true, both <code>lowerTerm</code> and
54     *  <code>upperTerm</code> will themselves be included in the range.
55     */
56    public RangeQuery(Term lowerTerm, Term upperTerm, boolean inclusive) {
57      this(lowerTerm, upperTerm, inclusive, null);
58    }
59  
60    /** Constructs a query selecting all terms greater than
61     * <code>lowerTerm</code> but less than <code>upperTerm</code>.
62     * There must be at least one term and either term may be null,
63     * in which case there is no bound on that side, but if there are
64     * two terms, both terms <b>must</b> be for the same field.
65     * <p>
66     * If <code>collator</code> is not null, it will be used to decide whether
67     * index terms are within the given range, rather than using the Unicode code
68     * point order in which index terms are stored.
69     * <p>
70     * <strong>WARNING:</strong> Using this constructor and supplying a non-null
71     * value in the <code>collator</code> parameter will cause every single 
72     * index Term in the Field referenced by lowerTerm and/or upperTerm to be
73     * examined.  Depending on the number of index Terms in this Field, the 
74     * operation could be very slow.
75     *
76     * @param lowerTerm The Term at the lower end of the range
77     * @param upperTerm The Term at the upper end of the range
78     * @param inclusive If true, both <code>lowerTerm</code> and
79     *  <code>upperTerm</code> will themselves be included in the range.
80     * @param collator The collator to use to collate index Terms, to determine
81     *  their membership in the range bounded by <code>lowerTerm</code> and
82     *  <code>upperTerm</code>.
83     */
84    public RangeQuery(Term lowerTerm, Term upperTerm, boolean inclusive, Collator collator) {
85      if (lowerTerm == null && upperTerm == null)
86        throw new IllegalArgumentException("At least one term must be non-null");
87      if (lowerTerm != null && upperTerm != null && lowerTerm.field() != upperTerm.field())
88        throw new IllegalArgumentException("Both terms must have the same field");
89        
90      delegate = new TermRangeQuery(
91        (lowerTerm == null) ? upperTerm.field() : lowerTerm.field(), 
92        (lowerTerm == null) ? null : lowerTerm.text(), 
93        (upperTerm == null) ? null : upperTerm.text(), 
94        inclusive, inclusive,
95        collator
96      );
97      delegate.setRewriteMethod(TermRangeQuery.SCORING_BOOLEAN_QUERY_REWRITE);
98    }
99    
100   public void setBoost(float b) {
101     super.setBoost(b);
102     delegate.setBoost(b);
103   }
104 
105   public Query rewrite(IndexReader reader) throws IOException {
106     return delegate.rewrite(reader);
107   }
108 
109   /** Returns the field name for this query */
110   public String getField() {
111     return delegate.getField();
112   }
113 
114   /** Returns the lower term of this range query. */
115   public Term getLowerTerm() {
116     final String term = delegate.getLowerTerm();
117     return (term == null) ? null : new Term(getField(), term);
118   }
119 
120   /** Returns the upper term of this range query. */
121   public Term getUpperTerm() {
122     final String term = delegate.getUpperTerm();
123     return (term == null) ? null : new Term(getField(), term);
124   }
125 
126   /** Returns <code>true</code> if the range query is inclusive */
127   public boolean isInclusive() {
128     return delegate.includesLower() && delegate.includesUpper();
129   }
130 
131   /** Returns the collator used to determine range inclusion, if any. */
132   public Collator getCollator() {
133     return delegate.getCollator();
134   }
135 
136   /** Prints a user-readable version of this query. */
137   public String toString(String field) {
138     return delegate.toString(field);
139   }
140 
141   /** Returns true iff <code>o</code> is equal to this. */
142   public boolean equals(Object o) {
143     if (this == o) return true;
144     if (!(o instanceof RangeQuery)) return false;
145 
146     final RangeQuery other = (RangeQuery) o;
147     return this.delegate.equals(other.delegate);
148   }
149 
150   /** Returns a hash code value for this object.*/
151   public int hashCode() {
152     return delegate.hashCode();
153   }
154 }
