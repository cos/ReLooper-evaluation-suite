20  import org.apache.lucene.index.IndexReader;
21  
22  import java.io.IOException;
23  import java.util.Set;
24  
25  /**
26   * A query that wraps a filter and simply returns a constant score equal to the
27   * query boost for every document in the filter.
28   *
29   *
30   * @version $Id: ConstantScoreQuery.java 807180 2009-08-24 12:26:43Z markrmiller $
31   */
32  public class ConstantScoreQuery extends Query {
33    protected final Filter filter;
34  
35    public ConstantScoreQuery(Filter filter) {
36      this.filter=filter;
37    }
38  
39    /** Returns the encapsulated filter */
40    public Filter getFilter() {
41      return filter;
42    }
43  
44    public Query rewrite(IndexReader reader) throws IOException {
45      return this;
46    }
47  
48    public void extractTerms(Set terms) {
49      // OK to not add any terms when used for MultiSearcher,
50      // but may not be OK for highlighting
51    }
52  
53    protected class ConstantWeight extends Weight {
54      private Similarity similarity;
55      private float queryNorm;
56      private float queryWeight;
57      
58      public ConstantWeight(Searcher searcher) {
59        this.similarity = getSimilarity(searcher);
60      }
61  
62      public Query getQuery() {
63        return ConstantScoreQuery.this;
64      }
65  
66      public float getValue() {
67        return queryWeight;
68      }
69  
70      public float sumOfSquaredWeights() throws IOException {
71        queryWeight = getBoost();
72        return queryWeight * queryWeight;
73      }
74  
75      public void normalize(float norm) {
76        this.queryNorm = norm;
77        queryWeight *= this.queryNorm;
78      }
79  
80      public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer) throws IOException {
81        return new ConstantScorer(similarity, reader, this);
82      }
83  
84      public Explanation explain(IndexReader reader, int doc) throws IOException {
85        
86        ConstantScorer cs = new ConstantScorer(similarity, reader, this);
87        boolean exists = cs.docIdSetIterator.advance(doc) == doc;
88  
89        ComplexExplanation result = new ComplexExplanation();
90  
91        if (exists) {
92          result.setDescription("ConstantScoreQuery(" + filter
93          + "), product of:");
94          result.setValue(queryWeight);
95          result.setMatch(Boolean.TRUE);
96          result.addDetail(new Explanation(getBoost(), "boost"));
97          result.addDetail(new Explanation(queryNorm,"queryNorm"));
98        } else {
99          result.setDescription("ConstantScoreQuery(" + filter
100         + ") doesn't match id " + doc);
101         result.setValue(0);
102         result.setMatch(Boolean.FALSE);
103       }
104       return result;
105     }
106   }
107 
108   protected class ConstantScorer extends Scorer {
109     final DocIdSetIterator docIdSetIterator;
110     final float theScore;
111     int doc = -1;
112 
113     public ConstantScorer(Similarity similarity, IndexReader reader, Weight w) throws IOException {
114       super(similarity);
115       theScore = w.getValue();
116       DocIdSet docIdSet = filter.getDocIdSet(reader);
117       if (docIdSet == null) {
118         docIdSetIterator = DocIdSet.EMPTY_DOCIDSET.iterator();
119       } else {
120         DocIdSetIterator iter = docIdSet.iterator();
121         if (iter == null) {
122           docIdSetIterator = DocIdSet.EMPTY_DOCIDSET.iterator();
123         } else {
124           docIdSetIterator = iter;
125         }
126       }
127     }
128 
129     /** @deprecated use {@link #nextDoc()} instead. */
130     public boolean next() throws IOException {
131       return docIdSetIterator.nextDoc() != NO_MORE_DOCS;
132     }
133 
134     public int nextDoc() throws IOException {
135       return docIdSetIterator.nextDoc();
136     }
137     
138     /** @deprecated use {@link #docID()} instead. */
139     public int doc() {
140       return docIdSetIterator.doc();
141     }
142     
143     public int docID() {
144       return docIdSetIterator.docID();
145     }
146 
147     public float score() throws IOException {
148       return theScore;
149     }
150 
151     /** @deprecated use {@link #advance(int)} instead. */
152     public boolean skipTo(int target) throws IOException {
153       return docIdSetIterator.advance(target) != NO_MORE_DOCS;
154     }
155 
156     public int advance(int target) throws IOException {
157       return docIdSetIterator.advance(target);
158     }
159     
160     public Explanation explain(int doc) throws IOException {
161       throw new UnsupportedOperationException();
162     }
163   }
164 
165   public Weight createWeight(Searcher searcher) {
166     return new ConstantScoreQuery.ConstantWeight(searcher);
167   }
168 
169   /** Prints a user-readable version of this query. */
170   public String toString(String field) {
171     return "ConstantScore(" + filter.toString()
172       + (getBoost()==1.0 ? ")" : "^" + getBoost());
173   }
174 
175   /** Returns true if <code>o</code> is equal to this. */
176   public boolean equals(Object o) {
177     if (this == o) return true;
178     if (!(o instanceof ConstantScoreQuery)) return false;
179     ConstantScoreQuery other = (ConstantScoreQuery)o;
180     return this.getBoost()==other.getBoost() && filter.equals(other.filter);
181   }
182 
183   /** Returns a hash code value for this object. */
184   public int hashCode() {
185     // Simple add is OK since no existing filter hashcode has a float component.
186     return filter.hashCode() + Float.floatToIntBits(getBoost());
187   }
188 
189 }
