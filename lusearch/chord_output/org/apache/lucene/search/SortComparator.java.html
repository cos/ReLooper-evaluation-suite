20  import org.apache.lucene.index.IndexReader;
21  
22  import java.io.IOException;
23  
24  /**
25   * Abstract base class for sorting hits returned by a Query.
26   * 
27   * <p>
28   * This class should only be used if the other SortField types (SCORE, DOC,
29   * STRING, INT, FLOAT) do not provide an adequate sorting. It maintains an
30   * internal cache of values which could be quite large. The cache is an array of
31   * Comparable, one for each document in the index. There is a distinct
32   * Comparable for each unique term in the field - if some documents have the
33   * same term in the field, the cache array will have entries which reference the
34   * same Comparable.
35   * 
36   * This class will be used as part of a key to a FieldCache value. You must
37   * implement hashCode and equals to avoid an explosion in RAM usage if you use
38   * instances that are not the same instance. If you are searching using the
39   * Remote contrib, the same instance of this class on the client will be a new
40   * instance on every call to the server, so hashCode/equals is very important in
41   * that situation.
42   * 
43   * <p>
44   * Created: Apr 21, 2004 5:08:38 PM
45   * 
46   * 
47   * @version $Id: SortComparator.java 800119 2009-08-02 17:59:21Z markrmiller $
48   * @since 1.4
49   * @deprecated Please use {@link FieldComparatorSource} instead.
50   */
51  public abstract class SortComparator
52  implements SortComparatorSource {
53  
54    // inherit javadocs
55    public ScoreDocComparator newComparator (final IndexReader reader, final String fieldname)
56    throws IOException {
57      final String field = fieldname.intern();
58      final Comparable[] cachedValues = FieldCache.DEFAULT.getCustom (reader, field, SortComparator.this);
59      
60      return new ScoreDocComparator() {
61  
62        public int compare (ScoreDoc i, ScoreDoc j) {
63          return cachedValues[i.doc].compareTo (cachedValues[j.doc]);
64        }
65  
66        public Comparable sortValue (ScoreDoc i) {
67          return cachedValues[i.doc];
68        }
69  
70        public int sortType(){
71          return SortField.CUSTOM;
72        }
73      };
74    }
75  
76    /**
77     * Returns an object which, when sorted according to natural order,
78     * will order the Term values in the correct order.
79     * <p>For example, if the Terms contained integer values, this method
80     * would return <code>new Integer(termtext)</code>.  Note that this
81     * might not always be the most efficient implementation - for this
82     * particular example, a better implementation might be to make a
83     * ScoreDocLookupComparator that uses an internal lookup table of int.
84     * @param termtext The textual value of the term.
85     * @return An object representing <code>termtext</code> that sorts according to the natural order of <code>termtext</code>.
86     * @see Comparable
87     * @see ScoreDocComparator
88     */
89    protected abstract Comparable getComparable (String termtext);
90  
91  }
