20  /**
21   * Lower-level search API. <br>
22   * HitCollectors are primarily meant to be used to implement queries, sorting
23   * and filtering. See {@link Collector} for a lower level and higher performance
24   * (on a multi-segment index) API.
25   * 
26   * @see Searcher#search(Query,HitCollector)
27   * @version $Id: HitCollector.java 764551 2009-04-13 18:33:56Z mikemccand $
28   * @deprecated Please use {@link Collector} instead.
29   */
30  public abstract class HitCollector {
31    /** Called once for every document matching a query, with the document
32     * number and its raw score.
33     *
34     * <P>If, for example, an application wished to collect all of the hits for a
35     * query in a BitSet, then it might:<pre>
36     *   Searcher searcher = new IndexSearcher(indexReader);
37     *   final BitSet bits = new BitSet(indexReader.maxDoc());
38     *   searcher.search(query, new HitCollector() {
39     *       public void collect(int doc, float score) {
40     *         bits.set(doc);
41     *       }
42     *     });
43     * </pre>
44     *
45     * <p>Note: This is called in an inner search loop.  For good search
46     * performance, implementations of this method should not call
47     * {@link Searcher#doc(int)} or
48     * {@link org.apache.lucene.index.IndexReader#document(int)} on every
49     * document number encountered.  Doing so can slow searches by an order
50     * of magnitude or more.
51     * <p>Note: The <code>score</code> passed to this method is a raw score.
52     * In other words, the score will not necessarily be a float whose value is
53     * between 0 and 1.
54     */
55    public abstract void collect(int doc, float score);
56  }
