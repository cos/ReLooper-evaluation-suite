28  /**
29   * The BoostingTermQuery is very similar to the {@link org.apache.lucene.search.spans.SpanTermQuery} except
30   * that it factors in the value of the payload located at each of the positions where the
31   * {@link org.apache.lucene.index.Term} occurs.
32   * <p>
33   * In order to take advantage of this, you must override {@link org.apache.lucene.search.Similarity#scorePayload(String, byte[],int,int)}
34   * which returns 1 by default.
35   * <p>
36   * Payload scores are averaged across term occurrences in the document.  
37   * 
38   * @see org.apache.lucene.search.Similarity#scorePayload(String, byte[], int, int)
39   *
40   * @deprecated See {@link org.apache.lucene.search.payloads.PayloadTermQuery}
41   */
42  public class BoostingTermQuery extends PayloadTermQuery {
43  
44    public BoostingTermQuery(Term term) {
45      this(term, true);
46    }
47  
48    public BoostingTermQuery(Term term, boolean includeSpanScore) {
49      super(term, new AveragePayloadFunction(), includeSpanScore);
50    }
51  
52    public Weight createWeight(Searcher searcher) throws IOException {
53      return new BoostingTermWeight(this, searcher);
54    }
55  
56    protected class BoostingTermWeight extends PayloadTermWeight {
57  
58      public BoostingTermWeight(BoostingTermQuery query, Searcher searcher) throws IOException {
59        super(query, searcher);
60      }
61  
62      public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer) throws IOException {
63        return new PayloadTermSpanScorer((TermSpans) query.getSpans(reader), this,
64            similarity, reader.norms(query.getField()));
65      }
66  
67    }
68  
69  
70    public boolean equals(Object o) {
71      if (!(o instanceof BoostingTermQuery))
72        return false;
73      BoostingTermQuery other = (BoostingTermQuery) o;
74      return (this.getBoost() == other.getBoost())
75              && this.term.equals(other.term);
76    }
77  }
