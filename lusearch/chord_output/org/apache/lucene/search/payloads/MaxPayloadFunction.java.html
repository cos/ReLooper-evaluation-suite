21  /**
22   * Returns the maximum payload score seen, else 1 if there are no payloads on the doc.
23   * <p/>
24   * Is thread safe and completely reusable.
25   *
26   **/
27  public class MaxPayloadFunction extends PayloadFunction {
28    public float currentScore(int docId, String field, int start, int end, int numPayloadsSeen, float currentScore, float currentPayloadScore) {
29      if (numPayloadsSeen == 0) {
30        return currentPayloadScore;
31      } else {
32        return Math.max(currentPayloadScore, currentScore);
33      }
34    }
35  
36    public float docScore(int docId, String field, int numPayloadsSeen, float payloadScore) {
37      return numPayloadsSeen > 0 ? payloadScore : 1;
38    }
39    
40    public int hashCode() {
41      final int prime = 31;
42      int result = 1;
43      result = prime * result + this.getClass().hashCode();
44      return result;
45    }
46  
47    public boolean equals(Object obj) {
48      if (this == obj)
49        return true;
50      if (obj == null)
51        return false;
52      if (getClass() != obj.getClass())
53        return false;
54      return true;
55    }
56  }
