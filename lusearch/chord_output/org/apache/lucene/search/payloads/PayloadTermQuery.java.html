36  /**
37   * This class is very similar to
38   * {@link org.apache.lucene.search.spans.SpanTermQuery} except that it factors
39   * in the value of the payload located at each of the positions where the
40   * {@link org.apache.lucene.index.Term} occurs.
41   * <p>
42   * In order to take advantage of this, you must override
43   * {@link org.apache.lucene.search.Similarity#scorePayload(String, byte[],int,int)}
44   * which returns 1 by default.
45   * <p>
46   * Payload scores are aggregated using a pluggable {@link PayloadFunction}.
47   **/
48  public class PayloadTermQuery extends SpanTermQuery {
49    protected PayloadFunction function;
50    private boolean includeSpanScore;
51  
52    public PayloadTermQuery(Term term, PayloadFunction function) {
53      this(term, function, true);
54    }
55  
56    public PayloadTermQuery(Term term, PayloadFunction function,
57        boolean includeSpanScore) {
58      super(term);
59      this.function = function;
60      this.includeSpanScore = includeSpanScore;
61    }
62  
63    public Weight createWeight(Searcher searcher) throws IOException {
64      return new PayloadTermWeight(this, searcher);
65    }
66  
67    protected class PayloadTermWeight extends SpanWeight {
68  
69      public PayloadTermWeight(PayloadTermQuery query, Searcher searcher)
70          throws IOException {
71        super(query, searcher);
72      }
73  
74      public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder,
75          boolean topScorer) throws IOException {
76        return new PayloadTermSpanScorer((TermSpans) query.getSpans(reader),
77            this, similarity, reader.norms(query.getField()));
78      }
79  
80      protected class PayloadTermSpanScorer extends SpanScorer {
81        // TODO: is this the best way to allocate this?
82        protected byte[] payload = new byte[256];
83        protected TermPositions positions;
84        protected float payloadScore;
85        protected int payloadsSeen;
86  
87        public PayloadTermSpanScorer(TermSpans spans, Weight weight,
88            Similarity similarity, byte[] norms) throws IOException {
89          super(spans, weight, similarity, norms);
90          positions = spans.getPositions();
91        }
92  
93        protected boolean setFreqCurrentDoc() throws IOException {
94          if (!more) {
95            return false;
96          }
97          doc = spans.doc();
98          freq = 0.0f;
99          payloadScore = 0;
100         payloadsSeen = 0;
101         Similarity similarity1 = getSimilarity();
102         while (more && doc == spans.doc()) {
103           int matchLength = spans.end() - spans.start();
104 
105           freq += similarity1.sloppyFreq(matchLength);
106           processPayload(similarity1);
107 
108           more = spans.next();// this moves positions to the next match in this
109                               // document
110         }
111         return more || (freq != 0);
112       }
113 
114       protected void processPayload(Similarity similarity) throws IOException {
115         if (positions.isPayloadAvailable()) {
116           payload = positions.getPayload(payload, 0);
117           payloadScore = function.currentScore(doc, term.field(),
118               spans.start(), spans.end(), payloadsSeen, payloadScore,
119               similarity.scorePayload(doc, term.field(), spans.start(), spans
120                   .end(), payload, 0, positions.getPayloadLength()));
121           payloadsSeen++;
122 
123         } else {
124           // zero out the payload?
125         }
126       }
127 
128       /**
129        * 
130        * @return {@link #getSpanScore()} * {@link #getPayloadScore()}
131        * @throws IOException
132        */
133       public float score() throws IOException {
134 
135         return includeSpanScore ? getSpanScore() * getPayloadScore()
136             : getPayloadScore();
137       }
138 
139       /**
140        * Returns the SpanScorer score only.
141        * <p/>
142        * Should not be overridden without good cause!
143        * 
144        * @return the score for just the Span part w/o the payload
145        * @throws IOException
146        * 
147        * @see #score()
148        */
149       protected float getSpanScore() throws IOException {
150         return super.score();
151       }
152 
153       /**
154        * The score for the payload
155        * 
156        * @return The score, as calculated by
157        *         {@link PayloadFunction#docScore(int, String, int, float)}
158        */
159       protected float getPayloadScore() {
160         return function.docScore(doc, term.field(), payloadsSeen, payloadScore);
161       }
162 
163       public Explanation explain(final int doc) throws IOException {
164         ComplexExplanation result = new ComplexExplanation();
165         Explanation nonPayloadExpl = super.explain(doc);
166         result.addDetail(nonPayloadExpl);
167         // QUESTION: Is there a way to avoid this skipTo call? We need to know
168         // whether to load the payload or not
169         Explanation payloadBoost = new Explanation();
170         result.addDetail(payloadBoost);
171 
172         float payloadScore = getPayloadScore();
173         payloadBoost.setValue(payloadScore);
174         // GSI: I suppose we could toString the payload, but I don't think that
175         // would be a good idea
176         payloadBoost.setDescription("scorePayload(...)");
177         result.setValue(nonPayloadExpl.getValue() * payloadScore);
178         result.setDescription("btq, product of:");
179         result.setMatch(nonPayloadExpl.getValue() == 0 ? Boolean.FALSE
180             : Boolean.TRUE); // LUCENE-1303
181         return result;
182       }
183 
184     }
185   }
186 
187   public int hashCode() {
188     final int prime = 31;
189     int result = super.hashCode();
190     result = prime * result + ((function == null) ? 0 : function.hashCode());
191     result = prime * result + (includeSpanScore ? 1231 : 1237);
192     return result;
193   }
194 
195   public boolean equals(Object obj) {
196     if (this == obj)
197       return true;
198     if (!super.equals(obj))
199       return false;
200     if (getClass() != obj.getClass())
201       return false;
202     PayloadTermQuery other = (PayloadTermQuery) obj;
203     if (function == null) {
204       if (other.function != null)
205         return false;
206     } else if (!function.equals(other.function))
207       return false;
208     if (includeSpanScore != other.includeSpanScore)
209       return false;
210     return true;
211   }
