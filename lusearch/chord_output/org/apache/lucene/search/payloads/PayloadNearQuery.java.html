39  /**
40   * This class is very similar to
41   * {@link org.apache.lucene.search.spans.SpanNearQuery} except that it factors
42   * in the value of the payloads located at each of the positions where the
43   * {@link org.apache.lucene.search.spans.TermSpans} occurs.
44   * <p/>
45   * In order to take advantage of this, you must override
46   * {@link org.apache.lucene.search.Similarity#scorePayload(String, byte[],int,int)}
47   * which returns 1 by default.
48   * <p/>
49   * Payload scores are aggregated using a pluggable {@link PayloadFunction}.
50   * 
51   * @see org.apache.lucene.search.Similarity#scorePayload(String, byte[], int,
52   *      int)
53   */
54  public class PayloadNearQuery extends SpanNearQuery {
55    protected String fieldName;
56    protected PayloadFunction function;
57  
58    public PayloadNearQuery(SpanQuery[] clauses, int slop, boolean inOrder) {
59      this(clauses, slop, inOrder, new AveragePayloadFunction());
60    }
61  
62    public PayloadNearQuery(SpanQuery[] clauses, int slop, boolean inOrder,
63        PayloadFunction function) {
64      super(clauses, slop, inOrder);
65      fieldName = clauses[0].getField(); // all clauses must have same field
66      this.function = function;
67    }
68  
69    public Weight createWeight(Searcher searcher) throws IOException {
70      return new PayloadNearSpanWeight(this, searcher);
71    }
72  
73    public Object clone() {
74      int sz = clauses.size();
75      SpanQuery[] newClauses = new SpanQuery[sz];
76  
77      for (int i = 0; i < sz; i++) {
78        SpanQuery clause = (SpanQuery) clauses.get(i);
79        newClauses[i] = (SpanQuery) clause.clone();
80      }
81      PayloadNearQuery boostingNearQuery = new PayloadNearQuery(newClauses, slop,
82          inOrder);
83      boostingNearQuery.setBoost(getBoost());
84      return boostingNearQuery;
85    }
86  
87    public String toString(String field) {
88      StringBuffer buffer = new StringBuffer();
89      buffer.append("payloadNear([");
90      Iterator i = clauses.iterator();
91      while (i.hasNext()) {
92        SpanQuery clause = (SpanQuery) i.next();
93        buffer.append(clause.toString(field));
94        if (i.hasNext()) {
95          buffer.append(", ");
96        }
97      }
98      buffer.append("], ");
99      buffer.append(slop);
100     buffer.append(", ");
101     buffer.append(inOrder);
102     buffer.append(")");
103     buffer.append(ToStringUtils.boost(getBoost()));
104     return buffer.toString();
105   }

107   // @Override
108   public int hashCode() {
109     final int prime = 31;
110     int result = super.hashCode();
111     result = prime * result + ((fieldName == null) ? 0 : fieldName.hashCode());
112     result = prime * result + ((function == null) ? 0 : function.hashCode());
113     return result;
114   }
115 
116   // @Override
117   public boolean equals(Object obj) {
118     if (this == obj)
119       return true;
120     if (!super.equals(obj))
121       return false;
122     if (getClass() != obj.getClass())
123       return false;
124     PayloadNearQuery other = (PayloadNearQuery) obj;
125     if (fieldName == null) {
126       if (other.fieldName != null)
127         return false;
128     } else if (!fieldName.equals(other.fieldName))
129       return false;
130     if (function == null) {
131       if (other.function != null)
132         return false;
133     } else if (!function.equals(other.function))
134       return false;
135     return true;
136   }

156   public class PayloadNearSpanScorer extends SpanScorer {
157     Spans spans;
158 
159     protected float payloadScore;
160     private int payloadsSeen;
161     Similarity similarity = getSimilarity();
162 
163     protected PayloadNearSpanScorer(Spans spans, Weight weight,
164         Similarity similarity, byte[] norms) throws IOException {
165       super(spans, weight, similarity, norms);
166       this.spans = spans;
167     }
168 
169     // Get the payloads associated with all underlying subspans
170     public void getPayloads(Spans[] subSpans) throws IOException {
171       for (int i = 0; i < subSpans.length; i++) {
172         if (subSpans[i] instanceof NearSpansOrdered) {
173           if (((NearSpansOrdered) subSpans[i]).isPayloadAvailable()) {
174             processPayloads(((NearSpansOrdered) subSpans[i]).getPayload(),
175                 subSpans[i].start(), subSpans[i].end());
176           }
177           getPayloads(((NearSpansOrdered) subSpans[i]).getSubSpans());
178         } else if (subSpans[i] instanceof NearSpansUnordered) {
179           if (((NearSpansUnordered) subSpans[i]).isPayloadAvailable()) {
180             processPayloads(((NearSpansUnordered) subSpans[i]).getPayload(),
181                 subSpans[i].start(), subSpans[i].end());
182           }
183           getPayloads(((NearSpansUnordered) subSpans[i]).getSubSpans());
184         }
185       }
186     }
187 
188     /**
189      * By default, uses the {@link PayloadFunction} to score the payloads, but
190      * can be overridden to do other things.
191      * 
192      * @param payLoads The payloads
193      * @param start The start position of the span being scored
194      * @param end The end position of the span being scored
195      * 
196      * @see Spans
197      */
198     protected void processPayloads(Collection payLoads, int start, int end) {
199       for (Iterator iterator = payLoads.iterator(); iterator.hasNext();) {
200         byte[] thePayload = (byte[]) iterator.next();
201         payloadScore = function.currentScore(doc, fieldName, start, end,
202             payloadsSeen, payloadScore, similarity.scorePayload(doc, fieldName,
203                 spans.start(), spans.end(), thePayload, 0, thePayload.length));
204         ++payloadsSeen;
205       }
206     }
207 
208     //
209     protected boolean setFreqCurrentDoc() throws IOException {
210       if (!more) {
211         return false;
212       }
213       Spans[] spansArr = new Spans[1];
214       spansArr[0] = spans;
215       payloadScore = 0;
216       payloadsSeen = 0;
217       getPayloads(spansArr);
218       return super.setFreqCurrentDoc();
219     }
220 
221     public float score() throws IOException {
222 
223       return super.score()
224           * function.docScore(doc, fieldName, payloadsSeen, payloadScore);
225     }
226 
227     public Explanation explain(int doc) throws IOException {
228       Explanation result = new Explanation();
229       Explanation nonPayloadExpl = super.explain(doc);
230       result.addDetail(nonPayloadExpl);
231       Explanation payloadBoost = new Explanation();
232       result.addDetail(payloadBoost);
233       float avgPayloadScore = (payloadsSeen > 0 ? (payloadScore / payloadsSeen)
234           : 1);
235       payloadBoost.setValue(avgPayloadScore);
236       payloadBoost.setDescription("scorePayload(...)");
237       result.setValue(nonPayloadExpl.getValue() * avgPayloadScore);
238       result.setDescription("bnq, product of:");
239       return result;
240     }
241   }
