20  import java.io.IOException;
21  
22  import org.apache.lucene.document.Document;
23  import org.apache.lucene.document.FieldSelector;
24  import org.apache.lucene.index.CorruptIndexException;
25  import org.apache.lucene.index.Term;
26  
27  /**
28   * The interface for search implementations.
29   * 
30   * <p>
31   * Searchable is the abstract network protocol for searching. Implementations
32   * provide search over a single index, over multiple indices, and over indices
33   * on remote servers.
34   * 
35   * <p>
36   * Queries, filters and sort criteria are designed to be compact so that they
37   * may be efficiently passed to a remote index, with only the top-scoring hits
38   * being returned, rather than every matching hit.
39   * 
40   * <b>NOTE:</b> this interface is kept public for convenience. Since it is not
41   * expected to be implemented directly, it may be changed unexpectedly between
42   * releases.
43   */
44  public interface Searchable {
45    
46    /** Lower-level search API.
47     *
48     * <p>{@link HitCollector#collect(int,float)} is called for every non-zero
49     * scoring document.
50     * <br>HitCollector-based access to remote indexes is discouraged.
51     *
52     * <p>Applications should only use this if they need <i>all</i> of the
53     * matching documents.  The high-level search API ({@link
54     * Searcher#search(Query)}) is usually more efficient, as it skips
55     * non-high-scoring hits.
56     *
57     * @param weight to match documents
58     * @param filter if non-null, used to permit documents to be collected.
59     * @param results to receive hits
60     * @throws BooleanQuery.TooManyClauses
61     * @deprecated use {@link #search(Weight, Filter, Collector)} instead.
62     */
63    void search(Weight weight, Filter filter, HitCollector results)
64    throws IOException;
65  
66    /**
67     * Lower-level search API.
68     * 
69     * <p>
70     * {@link Collector#collect(int)} is called for every document. <br>
71     * Collector-based access to remote indexes is discouraged.
72     * 
73     * <p>
74     * Applications should only use this if they need <i>all</i> of the matching
75     * documents. The high-level search API ({@link Searcher#search(Query)}) is
76     * usually more efficient, as it skips non-high-scoring hits.
77     * 
78     * @param weight
79     *          to match documents
80     * @param filter
81     *          if non-null, used to permit documents to be collected.
82     * @param collector
83     *          to receive hits
84     * @throws BooleanQuery.TooManyClauses
85     */
86    void search(Weight weight, Filter filter, Collector collector) throws IOException;
87  
88    /** Frees resources associated with this Searcher.
89     * Be careful not to call this method while you are still using objects
90     * like {@link Hits}.
91     */
92    void close() throws IOException;
93  
94    /** Expert: Returns the number of documents containing <code>term</code>.
95     * 
96     * @see org.apache.lucene.index.IndexReader#docFreq(Term)
97     */
98    int docFreq(Term term) throws IOException;
99  
100   /** Expert: For each term in the terms array, calculates the number of
101    * documents containing <code>term</code>. Returns an array with these
102    * document frequencies. Used to minimize number of remote calls.
103    */
104   int[] docFreqs(Term[] terms) throws IOException;
105 
106   /** Expert: Returns one greater than the largest possible document number.
107    * 
108    * @see org.apache.lucene.index.IndexReader#maxDoc()
109    */
110   int maxDoc() throws IOException;
111 
112   /** Expert: Low-level search implementation.  Finds the top <code>n</code>
113    * hits for <code>query</code>, applying <code>filter</code> if non-null.
114    *
115    * <p>Applications should usually call {@link Searcher#search(Query)} or
116    * {@link Searcher#search(Query,Filter)} instead.
117    * @throws BooleanQuery.TooManyClauses
118    */
119   TopDocs search(Weight weight, Filter filter, int n) throws IOException;
120 
121   /**
122    * Returns the stored fields of document <code>i</code>.
123    * 
124    * @see org.apache.lucene.index.IndexReader#document(int)
125    * @throws CorruptIndexException if the index is corrupt
126    * @throws IOException if there is a low-level IO error
127    */
128   Document doc(int i) throws CorruptIndexException, IOException;
129 
130   /**
131    * Get the {@link org.apache.lucene.document.Document} at the <code>n</code><sup>th</sup> position. The {@link org.apache.lucene.document.FieldSelector}
132    * may be used to determine what {@link org.apache.lucene.document.Field}s to load and how they should be loaded.
133    * 
134    * <b>NOTE:</b> If the underlying Reader (more specifically, the underlying <code>FieldsReader</code>) is closed before the lazy {@link org.apache.lucene.document.Field} is
135    * loaded an exception may be thrown.  If you want the value of a lazy {@link org.apache.lucene.document.Field} to be available after closing you must
136    * explicitly load it or fetch the Document again with a new loader.
137    * 
138    *  
139    * @param n Get the document at the <code>n</code><sup>th</sup> position
140    * @param fieldSelector The {@link org.apache.lucene.document.FieldSelector} to use to determine what Fields should be loaded on the Document.  May be null, in which case all Fields will be loaded.
141    * @return The stored fields of the {@link org.apache.lucene.document.Document} at the nth position
142    * @throws CorruptIndexException if the index is corrupt
143    * @throws IOException if there is a low-level IO error
144    * 
145    * @see org.apache.lucene.index.IndexReader#document(int, FieldSelector)
146    * @see org.apache.lucene.document.Fieldable
147    * @see org.apache.lucene.document.FieldSelector
148    * @see org.apache.lucene.document.SetBasedFieldSelector
149    * @see org.apache.lucene.document.LoadFirstFieldSelector
150    */
151   Document doc(int n, FieldSelector fieldSelector) throws CorruptIndexException, IOException;
152   
153   /** Expert: called to re-write queries into primitive queries.
154    * @throws BooleanQuery.TooManyClauses
155    */
156   Query rewrite(Query query) throws IOException;
157 
158   /** Expert: low-level implementation method
159    * Returns an Explanation that describes how <code>doc</code> scored against
160    * <code>weight</code>.
161    *
162    * <p>This is intended to be used in developing Similarity implementations,
163    * and, for good performance, should not be displayed with every hit.
164    * Computing an explanation is as expensive as executing the query over the
165    * entire index.
166    * <p>Applications should call {@link Searcher#explain(Query, int)}.
167    * @throws BooleanQuery.TooManyClauses
168    */
169   Explanation explain(Weight weight, int doc) throws IOException;
170 
171   /** Expert: Low-level search implementation with arbitrary sorting.  Finds
172    * the top <code>n</code> hits for <code>query</code>, applying
173    * <code>filter</code> if non-null, and sorting the hits by the criteria in
174    * <code>sort</code>.
175    *
176    * <p>Applications should usually call {@link
177    * Searcher#search(Query,Filter,int,Sort)} instead.
178    * 
179    * @throws BooleanQuery.TooManyClauses
180    */
181   TopFieldDocs search(Weight weight, Filter filter, int n, Sort sort)
182   throws IOException;
183 
184 }
