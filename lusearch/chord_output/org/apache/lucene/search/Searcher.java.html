20  import java.io.IOException;
21  
22  import org.apache.lucene.document.Document;
23  import org.apache.lucene.index.CorruptIndexException;
24  import org.apache.lucene.index.Term;
25  
26  /**
27   * An abstract base class for search implementations. Implements the main search
28   * methods.
29   * 
30   * <p>
31   * Note that you can only access hits from a Searcher as long as it is not yet
32   * closed, otherwise an IOException will be thrown.
33   */
34  public abstract class Searcher implements Searchable {
35  
36    /** Returns the documents matching <code>query</code>. 
37     * @throws BooleanQuery.TooManyClauses
38     * @deprecated Hits will be removed in Lucene 3.0. Use
39     * {@link #search(Query, Filter, int)} instead.
40     */
41    public final Hits search(Query query) throws IOException {
42      return search(query, (Filter)null);
43    }
44  
45    /** Returns the documents matching <code>query</code> and
46     * <code>filter</code>.
47     * @throws BooleanQuery.TooManyClauses
48     * @deprecated Hits will be removed in Lucene 3.0. Use
49     * {@link #search(Query, Filter, int)} instead.
50     */
51    public Hits search(Query query, Filter filter) throws IOException {
52      return new Hits(this, query, filter);
53    }
54  
55    /** Returns documents matching <code>query</code> sorted by
56     * <code>sort</code>.
57     * @throws BooleanQuery.TooManyClauses
58     * @deprecated Hits will be removed in Lucene 3.0. Use 
59     * {@link #search(Query, Filter, int, Sort)} instead.
60     */
61    public Hits search(Query query, Sort sort)
62      throws IOException {
63      return new Hits(this, query, null, sort);
64    }
65  
66    /** Returns documents matching <code>query</code> and <code>filter</code>,
67     * sorted by <code>sort</code>.
68     * @throws BooleanQuery.TooManyClauses
69     * @deprecated Hits will be removed in Lucene 3.0. Use 
70     * {@link #search(Query, Filter, int, Sort)} instead.
71     */
72    public Hits search(Query query, Filter filter, Sort sort)
73      throws IOException {
74      return new Hits(this, query, filter, sort);
75    }
76  
77    /** Search implementation with arbitrary sorting.  Finds
78     * the top <code>n</code> hits for <code>query</code>, applying
79     * <code>filter</code> if non-null, and sorting the hits by the criteria in
80     * <code>sort</code>.
81     * 
82     * <p>NOTE: this does not compute scores by default; use
83     * {@link IndexSearcher#setDefaultFieldSortScoring} to
84     * enable scoring.
85     *
86     * @throws BooleanQuery.TooManyClauses
87     */
88    public TopFieldDocs search(Query query, Filter filter, int n,
89                               Sort sort) throws IOException {
90      return search(createWeight(query), filter, n, sort);
91    }
92  
93    /** Lower-level search API.
94     *
95     * <p>{@link HitCollector#collect(int,float)} is called for every matching
96     * document.
97     *
98     * <p>Applications should only use this if they need <i>all</i> of the
99     * matching documents.  The high-level search API ({@link
100    * Searcher#search(Query)}) is usually more efficient, as it skips
101    * non-high-scoring hits.
102    * <p>Note: The <code>score</code> passed to this method is a raw score.
103    * In other words, the score will not necessarily be a float whose value is
104    * between 0 and 1.
105    * @throws BooleanQuery.TooManyClauses
106    * @deprecated use {@link #search(Query, Collector)} instead.
107    */
108   public void search(Query query, HitCollector results)
109     throws IOException {
110     search(createWeight(query), null, new HitCollectorWrapper(results));
111   }
112 
113   /** Lower-level search API.
114   *
115   * <p>{@link Collector#collect(int)} is called for every matching document.
116   *
117   * <p>Applications should only use this if they need <i>all</i> of the
118   * matching documents.  The high-level search API ({@link
119   * Searcher#search(Query, int)}) is usually more efficient, as it skips
120   * non-high-scoring hits.
121   * <p>Note: The <code>score</code> passed to this method is a raw score.
122   * In other words, the score will not necessarily be a float whose value is
123   * between 0 and 1.
124   * @throws BooleanQuery.TooManyClauses
125   */
126  public void search(Query query, Collector results)
127    throws IOException {
128    search(createWeight(query), null, results);
129  }
130 
131   /** Lower-level search API.
132    *
133    * <p>{@link HitCollector#collect(int,float)} is called for every matching
134    * document.
135    * <br>HitCollector-based access to remote indexes is discouraged.
136    *
137    * <p>Applications should only use this if they need <i>all</i> of the
138    * matching documents.  The high-level search API ({@link
139    * Searcher#search(Query, Filter, int)}) is usually more efficient, as it skips
140    * non-high-scoring hits.
141    *
142    * @param query to match documents
143    * @param filter if non-null, used to permit documents to be collected.
144    * @param results to receive hits
145    * @throws BooleanQuery.TooManyClauses
146    * @deprecated use {@link #search(Query, Filter, Collector)} instead.
147    */
148   public void search(Query query, Filter filter, HitCollector results)
149     throws IOException {
150     search(createWeight(query), filter, new HitCollectorWrapper(results));
151   }
152   
153   /** Lower-level search API.
154    *
155    * <p>{@link Collector#collect(int)} is called for every matching
156    * document.
157    * <br>Collector-based access to remote indexes is discouraged.
158    *
159    * <p>Applications should only use this if they need <i>all</i> of the
160    * matching documents.  The high-level search API ({@link
161    * Searcher#search(Query, Filter, int)}) is usually more efficient, as it skips
162    * non-high-scoring hits.
163    *
164    * @param query to match documents
165    * @param filter if non-null, used to permit documents to be collected.
166    * @param results to receive hits
167    * @throws BooleanQuery.TooManyClauses
168    */
169   public void search(Query query, Filter filter, Collector results)
170   throws IOException {
171     search(createWeight(query), filter, results);
172   }
173 
174   /** Finds the top <code>n</code>
175    * hits for <code>query</code>, applying <code>filter</code> if non-null.
176    *
177    * @throws BooleanQuery.TooManyClauses
178    */
179   public TopDocs search(Query query, Filter filter, int n)
180     throws IOException {
181     return search(createWeight(query), filter, n);
182   }
183 
184   /** Finds the top <code>n</code>
185    * hits for <code>query</code>.
186    *
187    * @throws BooleanQuery.TooManyClauses
188    */
189   public TopDocs search(Query query, int n)
190     throws IOException {
191     return search(query, null, n);
192   }
193 
194   /** Returns an Explanation that describes how <code>doc</code> scored against
195    * <code>query</code>.
196    *
197    * <p>This is intended to be used in developing Similarity implementations,
198    * and, for good performance, should not be displayed with every hit.
199    * Computing an explanation is as expensive as executing the query over the
200    * entire index.
201    */
202   public Explanation explain(Query query, int doc) throws IOException {
203     return explain(createWeight(query), doc);
204   }
205 
206   /** The Similarity implementation used by this searcher. */
207   private Similarity similarity = Similarity.getDefault();
208 
209   /** Expert: Set the Similarity implementation used by this Searcher.
210    *
211    * @see Similarity#setDefault(Similarity)
212    */
213   public void setSimilarity(Similarity similarity) {
214     this.similarity = similarity;
215   }
216 
217   /** Expert: Return the Similarity implementation used by this Searcher.
218    *
219    * <p>This defaults to the current value of {@link Similarity#getDefault()}.
220    */
221   public Similarity getSimilarity() {
222     return this.similarity;
223   }
224 
225   /**
226    * creates a weight for <code>query</code>
227    * @return new weight
228    */
229   protected Weight createWeight(Query query) throws IOException {
230     return query.weight(this);
231   }
232 
233   // inherit javadoc
234   public int[] docFreqs(Term[] terms) throws IOException {
235     int[] result = new int[terms.length];
236     for (int i = 0; i < terms.length; i++) {
237       result[i] = docFreq(terms[i]);
238     }
239     return result;
240   }
241 
242   /* The following abstract methods were added as a workaround for GCJ bug #15411.
243    * http://gcc.gnu.org/bugzilla/show_bug.cgi?id=15411
244    */
245   /**
246    * @deprecated use {@link #search(Weight, Filter, Collector)} instead.
247    */
248   public void search(Weight weight, Filter filter, HitCollector results) throws IOException {
249     search(weight, filter, new HitCollectorWrapper(results));
250   }
251   abstract public void search(Weight weight, Filter filter, Collector results) throws IOException;
252   abstract public void close() throws IOException;
253   abstract public int docFreq(Term term) throws IOException;
254   abstract public int maxDoc() throws IOException;
255   abstract public TopDocs search(Weight weight, Filter filter, int n) throws IOException;
256   abstract public Document doc(int i) throws CorruptIndexException, IOException;
257   abstract public Query rewrite(Query query) throws IOException;
258   abstract public Explanation explain(Weight weight, int doc) throws IOException;
259   abstract public TopFieldDocs search(Weight weight, Filter filter, int n, Sort sort) throws IOException;
260   /* End patch for GCJ bug #15411. */
261 }
