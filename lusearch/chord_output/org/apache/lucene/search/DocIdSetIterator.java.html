20  import java.io.IOException;
21  
22  /**
23   * This abstract class defines methods to iterate over a set of non-decreasing
24   * doc ids. Note that this class assumes it iterates on doc Ids, and therefore
25   * {@link #NO_MORE_DOCS} is set to {@value #NO_MORE_DOCS} in order to be used as
26   * a sentinel object. Implementations of this class are expected to consider
27   * {@link Integer#MAX_VALUE} as an invalid value.
28   */
29  public abstract class DocIdSetIterator {
30    
31    // TODO (3.0): review the javadocs and remove any references to '3.0'.
32    private int doc = -1;
33    
34    /**
35     * When returned by {@link #nextDoc()}, {@link #advance(int)} and
36     * {@link #doc()} it means there are no more docs in the iterator.
37     */
38    public static final int NO_MORE_DOCS = Integer.MAX_VALUE;
39  
40    /**
41     * Unsupported anymore. Call {@link #docID()} instead. This method throws
42     * {@link UnsupportedOperationException} if called.
43     * 
44     * @deprecated use {@link #docID()} instead.
45     */
46    public int doc() {
47      throw new UnsupportedOperationException("Call docID() instead.");
48    }
49  
50    /**
51     * Returns the following:
52     * <ul>
53     * <li>-1 or {@link #NO_MORE_DOCS} if {@link #nextDoc()} or
54     * {@link #advance(int)} were not called yet.
55     * <li>{@link #NO_MORE_DOCS} if the iterator has exhausted.
56     * <li>Otherwise it should return the doc ID it is currently on.
57     * </ul>
58     * <p>
59     * <b>NOTE:</b> in 3.0, this method will become abstract.
60     * 
61     * @since 2.9
62     */
63    public int docID() {
64      return doc;
65    }
66  
67    /**
68     * Unsupported anymore. Call {@link #nextDoc()} instead. This method throws
69     * {@link UnsupportedOperationException} if called.
70     * 
71     * @deprecated use {@link #nextDoc()} instead. This will be removed in 3.0
72     */
73    public boolean next() throws IOException {
74      throw new UnsupportedOperationException("Call nextDoc() instead.");
75    }
76  
77    /**
78     * Unsupported anymore. Call {@link #advance(int)} instead. This method throws
79     * {@link UnsupportedOperationException} if called.
80     * 
81     * @deprecated use {@link #advance(int)} instead. This will be removed in 3.0
82     */
83    public boolean skipTo(int target) throws IOException {
84      throw new UnsupportedOperationException("Call advance() instead.");
85    }
86  
87    /**
88     * Advances to the next document in the set and returns the doc it is
89     * currently on, or {@link #NO_MORE_DOCS} if there are no more docs in the
90     * set.<br>
91     * 
92     * <b>NOTE:</b> in 3.0 this method will become abstract, following the removal
93     * of {@link #next()}. For backward compatibility it is implemented as:
94     * 
95     * <pre>
96     * public int nextDoc() throws IOException {
97     *   return next() ? doc() : NO_MORE_DOCS;
98     * }
99     * </pre>
100    * 
101    * <b>NOTE:</b> after the iterator has exhausted you should not call this
102    * method, as it may result in unpredicted behavior.
103    * 
104    * @since 2.9
105    */
106   public int nextDoc() throws IOException {
107     return doc = next() ? doc() : NO_MORE_DOCS;
108   }
109 
110   /**
111    * Advances to the first beyond the current whose document number is greater
112    * than or equal to <i>target</i>. Returns the current document number or
113    * {@link #NO_MORE_DOCS} if there are no more docs in the set.
114    * <p>
115    * Behaves as if written:
116    * 
117    * <pre>
118    * int advance(int target) {
119    *   int doc;
120    *   while ((doc = nextDoc()) &lt; target) {
121    *   }
122    *   return doc;
123    * }
124    * </pre>
125    * 
126    * Some implementations are considerably more efficient than that.
127    * <p>
128    * <b>NOTE:</b> certain implementations may return a different value (each
129    * time) if called several times in a row with the same target.
130    * <p>
131    * <b>NOTE:</b> this method may be called with {@value #NO_MORE_DOCS} for
132    * efficiency by some Scorers. If your implementation cannot efficiently
133    * determine that it should exhaust, it is recommended that you check for that
134    * value in each call to this method.
135    * <p>
136    * <b>NOTE:</b> after the iterator has exhausted you should not call this
137    * method, as it may result in unpredicted behavior.
138    * <p>
139    * <b>NOTE:</b> in 3.0 this method will become abstract, following the removal
140    * of {@link #skipTo(int)}.
141    * 
142    * @since 2.9
143    */
144   public int advance(int target) throws IOException {
145     if (target == NO_MORE_DOCS) {
146       return doc = NO_MORE_DOCS;
147     }
148     return doc = skipTo(target) ? doc() : NO_MORE_DOCS;
149   }
150 
151 }
