20  import java.io.IOException;
21  
22  /**
23   * A DocIdSet contains a set of doc ids. Implementing classes must
24   * only implement {@link #iterator} to provide access to the set. 
25   */
26  public abstract class DocIdSet {
27  
28    /** An empty {@code DocIdSet} instance for easy use, e.g. in Filters that hit no documents. */
29    public static final DocIdSet EMPTY_DOCIDSET = new DocIdSet() {
30      
31      private final DocIdSetIterator iterator = new DocIdSetIterator() {
32        public int advance(int target) throws IOException { return NO_MORE_DOCS; }
33        public int docID() { return NO_MORE_DOCS; }
34        public int nextDoc() throws IOException { return NO_MORE_DOCS; }
35      };
36      
37      public DocIdSetIterator iterator() {
38        return iterator;
39      }
40      
41      public boolean isCacheable() {
42        return true;
43      }
44    };
45      
46    /** Provides a {@link DocIdSetIterator} to access the set.
47     * This implementation can return <code>null</code> or
48     * <code>{@linkplain #EMPTY_DOCIDSET}.iterator()</code> if there
49     * are no docs that match. */
50    public abstract DocIdSetIterator iterator() throws IOException;
51  
52    /**
53     * This method is a hint for {@link CachingWrapperFilter}, if this <code>DocIdSet</code>
54     * should be cached without copying it into a BitSet. The default is to return
55     * <code>false</code>. If you have an own <code>DocIdSet</code> implementation
56     * that does its iteration very effective and fast without doing disk I/O,
57     * override this method and return <code>true</here>.
58     */
59    public boolean isCacheable() {
60      return false;
61    }
62  }
