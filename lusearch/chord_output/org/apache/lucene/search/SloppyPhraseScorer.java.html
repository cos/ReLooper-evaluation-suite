20  import org.apache.lucene.index.TermPositions;
21  
22  import java.io.IOException;
23  import java.util.HashMap;
24  
25  final class SloppyPhraseScorer extends PhraseScorer {
26      private int slop;
27      private PhrasePositions repeats[];
28      private PhrasePositions tmpPos[]; // for flipping repeating pps.
29      private boolean checkedRepeats;
30  
31      SloppyPhraseScorer(Weight weight, TermPositions[] tps, int[] offsets, Similarity similarity,
32                         int slop, byte[] norms) {
33          super(weight, tps, offsets, similarity, norms);
34          this.slop = slop;
35      }
36  
37      /**
38       * Score a candidate doc for all slop-valid position-combinations (matches) 
39       * encountered while traversing/hopping the PhrasePositions.
40       * <br> The score contribution of a match depends on the distance: 
41       * <br> - highest score for distance=0 (exact match).
42       * <br> - score gets lower as distance gets higher.
43       * <br>Example: for query "a b"~2, a document "x a b a y" can be scored twice: 
44       * once for "a b" (distance=0), and once for "b a" (distance=2).
45       * <br>Possibly not all valid combinations are encountered, because for efficiency  
46       * we always propagate the least PhrasePosition. This allows to base on 
47       * PriorityQueue and move forward faster. 
48       * As result, for example, document "a b c b a"
49       * would score differently for queries "a b c"~4 and "c b a"~4, although 
50       * they really are equivalent. 
51       * Similarly, for doc "a b c b a f g", query "c b"~2 
52       * would get same score as "g f"~2, although "c b"~2 could be matched twice.
53       * We may want to fix this in the future (currently not, for performance reasons).
54       */
55      protected final float phraseFreq() throws IOException {
56          int end = initPhrasePositions();
57          
58          float freq = 0.0f;
59          boolean done = (end<0);
60          while (!done) {
61              PhrasePositions pp = (PhrasePositions) pq.pop();
62              int start = pp.position;
63              int next = ((PhrasePositions) pq.top()).position;
64  
65              boolean tpsDiffer = true;
66              for (int pos = start; pos <= next || !tpsDiffer; pos = pp.position) {
67                  if (pos<=next && tpsDiffer)
68                      start = pos;                  // advance pp to min window
69                  if (!pp.nextPosition()) {
70                      done = true;          // ran out of a term -- done
71                      break;
72                  }
73                  PhrasePositions pp2 = null;
74                  tpsDiffer = !pp.repeats || (pp2 = termPositionsDiffer(pp))==null;
75                  if (pp2!=null && pp2!=pp) {
76                    pp = flip(pp,pp2); // flip pp to pp2
77                  }
78              }
79  
80              int matchLength = end - start;
81              if (matchLength <= slop)
82                  freq += getSimilarity().sloppyFreq(matchLength); // score match
83  
84              if (pp.position > end)
85                  end = pp.position;
86              pq.put(pp);               // restore pq
87          }
88  
89          return freq;
90      }
91      
92      // flip pp2 and pp in the queue: pop until finding pp2, insert back all but pp2, insert pp back.
93      // assumes: pp!=pp2, pp2 in pq, pp not in pq.
94      // called only when there are repeating pps.
95      private PhrasePositions flip(PhrasePositions pp, PhrasePositions pp2) {
96        int n=0;
97        PhrasePositions pp3;
98        //pop until finding pp2
99        while ((pp3=(PhrasePositions)pq.pop()) != pp2) {
100         tmpPos[n++] = pp3;
101       }
102       //insert back all but pp2
103       for (n--; n>=0; n--) {
104         pq.insert(tmpPos[n]);
105       }
106       //insert pp back
107       pq.put(pp);
108       return pp2;
109     }
110 
111     /**
112      * Init PhrasePositions in place.
113      * There is a one time initialization for this scorer:
114      * <br>- Put in repeats[] each pp that has another pp with same position in the doc.
115      * <br>- Also mark each such pp by pp.repeats = true.
116      * <br>Later can consult with repeats[] in termPositionsDiffer(pp), making that check efficient.
117      * In particular, this allows to score queries with no repetitions with no overhead due to this computation.
118      * <br>- Example 1 - query with no repetitions: "ho my"~2
119      * <br>- Example 2 - query with repetitions: "ho my my"~2
120      * <br>- Example 3 - query with repetitions: "my ho my"~2
121      * <br>Init per doc w/repeats in query, includes propagating some repeating pp's to avoid false phrase detection.  
122      * @return end (max position), or -1 if any term ran out (i.e. done) 
123      * @throws IOException 
124      */
125     private int initPhrasePositions() throws IOException {
126         int end = 0;
127         
128         // no repeats at all (most common case is also the simplest one)
129         if (checkedRepeats && repeats==null) {
130             // build queue from list
131             pq.clear();
132             for (PhrasePositions pp = first; pp != null; pp = pp.next) {
133                 pp.firstPosition();
134                 if (pp.position > end)
135                     end = pp.position;
136                 pq.put(pp);         // build pq from list
137             }
138             return end;
139         }
140         
141         // position the pp's
142         for (PhrasePositions pp = first; pp != null; pp = pp.next)
143             pp.firstPosition();
144         
145         // one time initializatin for this scorer
146         if (!checkedRepeats) {
147             checkedRepeats = true;
148             // check for repeats
149             HashMap m = null;
150             for (PhrasePositions pp = first; pp != null; pp = pp.next) {
151                 int tpPos = pp.position + pp.offset;
152                 for (PhrasePositions pp2 = pp.next; pp2 != null; pp2 = pp2.next) {
153                     int tpPos2 = pp2.position + pp2.offset;
154                     if (tpPos2 == tpPos) { 
155                         if (m == null)
156                             m = new HashMap();
157                         pp.repeats = true;
158                         pp2.repeats = true;
159                         m.put(pp,null);
160                         m.put(pp2,null);
161                     }
162                 }
163             }
164             if (m!=null)
165                 repeats = (PhrasePositions[]) m.keySet().toArray(new PhrasePositions[0]);
166         }
167         
168         // with repeats must advance some repeating pp's so they all start with differing tp's       
169         if (repeats!=null) {
170             for (int i = 0; i < repeats.length; i++) {
171                 PhrasePositions pp = repeats[i];
172                 PhrasePositions pp2;
173                 while ((pp2 = termPositionsDiffer(pp)) != null) {
174                   if (!pp2.nextPosition())  // out of pps that do not differ, advance the pp with higher offset 
175                       return -1;           // ran out of a term -- done  
176                 } 
177             }
178         }
179       
180         // build queue from list
181         pq.clear();
182         for (PhrasePositions pp = first; pp != null; pp = pp.next) {
183             if (pp.position > end)
184                 end = pp.position;
185             pq.put(pp);         // build pq from list
186         }
187 
188         if (repeats!=null) {
189           tmpPos = new PhrasePositions[pq.size()];
190         }
191         return end;
192     }
193 
194     /**
195      * We disallow two pp's to have the same TermPosition, thereby verifying multiple occurrences 
196      * in the query of the same word would go elsewhere in the matched doc.
197      * @return null if differ (i.e. valid) otherwise return the higher offset PhrasePositions
198      * out of the first two PPs found to not differ.
199      */
200     private PhrasePositions termPositionsDiffer(PhrasePositions pp) {
201         // efficiency note: a more efficient implementation could keep a map between repeating 
202         // pp's, so that if pp1a, pp1b, pp1c are repeats term1, and pp2a, pp2b are repeats 
203         // of term2, pp2a would only be checked against pp2b but not against pp1a, pp1b, pp1c. 
204         // However this would complicate code, for a rather rare case, so choice is to compromise here.
205         int tpPos = pp.position + pp.offset;
206         for (int i = 0; i < repeats.length; i++) {
207             PhrasePositions pp2 = repeats[i];
208             if (pp2 == pp)
209                 continue;
210             int tpPos2 = pp2.position + pp2.offset;
211             if (tpPos2 == tpPos)
212                 return pp.offset > pp2.offset ? pp : pp2; // do not differ: return the one with higher offset.
213         }
214         return null; 
215     }
216 }
