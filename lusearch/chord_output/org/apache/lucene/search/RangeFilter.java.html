20  import java.text.Collator;
21  
22  /**
23   * A Filter that restricts search results to a range of values in a given
24   * field.
25   *
26   * <p>This filter matches the documents looking for terms that fall into the
27   * supplied range according to {@link String#compareTo(String)}. It is not intended
28   * for numerical ranges, use {@link NumericRangeFilter} instead.
29   *
30   * <p>If you construct a large number of range filters with different ranges but on the 
31   * same field, {@link FieldCacheRangeFilter} may have significantly better performance. 
32   *
33   * @deprecated Use {@link TermRangeFilter} for term ranges or
34   * {@link NumericRangeFilter} for numeric ranges instead.
35   * This class will be removed in Lucene 3.0.
36   */
37  public class RangeFilter extends MultiTermQueryWrapperFilter {
38      
39    /**
40     * @param fieldName The field this range applies to
41     * @param lowerTerm The lower bound on this range
42     * @param upperTerm The upper bound on this range
43     * @param includeLower Does this range include the lower bound?
44     * @param includeUpper Does this range include the upper bound?
45     * @throws IllegalArgumentException if both terms are null or if
46     *  lowerTerm is null and includeLower is true (similar for upperTerm
47     *  and includeUpper)
48     */
49    public RangeFilter(String fieldName, String lowerTerm, String upperTerm,
50                       boolean includeLower, boolean includeUpper) {
51        super(new TermRangeQuery(fieldName, lowerTerm, upperTerm, includeLower, includeUpper));
52    }
53  
54    /**
55     * <strong>WARNING:</strong> Using this constructor and supplying a non-null
56     * value in the <code>collator</code> parameter will cause every single 
57     * index Term in the Field referenced by lowerTerm and/or upperTerm to be
58     * examined.  Depending on the number of index Terms in this Field, the 
59     * operation could be very slow.
60     *
61     * @param lowerTerm The lower bound on this range
62     * @param upperTerm The upper bound on this range
63     * @param includeLower Does this range include the lower bound?
64     * @param includeUpper Does this range include the upper bound?
65     * @param collator The collator to use when determining range inclusion; set
66     *  to null to use Unicode code point ordering instead of collation.
67     * @throws IllegalArgumentException if both terms are null or if
68     *  lowerTerm is null and includeLower is true (similar for upperTerm
69     *  and includeUpper)
70     */
71    public RangeFilter(String fieldName, String lowerTerm, String upperTerm,
72                       boolean includeLower, boolean includeUpper,
73                       Collator collator) {
74        super(new TermRangeQuery(fieldName, lowerTerm, upperTerm, includeLower, includeUpper, collator));
75    }
76  
77    /**
78     * Constructs a filter for field <code>fieldName</code> matching
79     * less than or equal to <code>upperTerm</code>.
80     */
81    public static RangeFilter Less(String fieldName, String upperTerm) {
82        return new RangeFilter(fieldName, null, upperTerm, false, true);
83    }
84  
85    /**
86     * Constructs a filter for field <code>fieldName</code> matching
87     * greater than or equal to <code>lowerTerm</code>.
88     */
89    public static RangeFilter More(String fieldName, String lowerTerm) {
90        return new RangeFilter(fieldName, lowerTerm, null, true, false);
91    }
92  }
