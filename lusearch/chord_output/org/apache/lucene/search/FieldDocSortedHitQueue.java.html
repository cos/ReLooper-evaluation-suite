20  import org.apache.lucene.util.PriorityQueue;
21  
22  import java.text.Collator;
23  import java.util.Locale;
24  
25  /**
26   * Expert: Collects sorted results from Searchable's and collates them.
27   * The elements put into this queue must be of type FieldDoc.
28   *
29   * <p>Created: Feb 11, 2004 2:04:21 PM
30   *
31   * @since   lucene 1.4
32   * @version $Id: FieldDocSortedHitQueue.java 695514 2008-09-15 15:42:11Z otis $
33   */
34  class FieldDocSortedHitQueue
35  extends PriorityQueue {
36  
37      // this cannot contain AUTO fields - any AUTO fields should
38      // have been resolved by the time this class is used.
39      volatile SortField[] fields;
40  
41      // used in the case where the fields are sorted by locale
42      // based strings
43      volatile Collator[] collators;
44  
45  
46      /**
47       * Creates a hit queue sorted by the given list of fields.
48       * @param fields Fieldable names, in priority order (highest priority first).
49       * @param size  The number of hits to retain.  Must be greater than zero.
50       */
51      FieldDocSortedHitQueue (SortField[] fields, int size) {
52          this.fields = fields;
53          this.collators = hasCollators (fields);
54          initialize (size);
55      }
56  
57  
58      /**
59       * Allows redefinition of sort fields if they are <code>null</code>.
60       * This is to handle the case using ParallelMultiSearcher where the
61       * original list contains AUTO and we don't know the actual sort
62       * type until the values come back.  The fields can only be set once.
63       * This method is thread safe.
64       * @param fields
65       */
66      synchronized void setFields (SortField[] fields) {
67          if (this.fields == null) {
68              this.fields = fields;
69              this.collators = hasCollators (fields);
70          }
71      }
72  
73  
74      /** Returns the fields being used to sort. */
75      SortField[] getFields() {
76          return fields;
77      }
78  
79  
80      /** Returns an array of collators, possibly <code>null</code>.  The collators
81       * correspond to any SortFields which were given a specific locale.
82       * @param fields Array of sort fields.
83       * @return Array, possibly <code>null</code>.
84       */
85      private Collator[] hasCollators (final SortField[] fields) {
86          if (fields == null) return null;
87          Collator[] ret = new Collator[fields.length];
88          for (int i=0; i<fields.length; ++i) {
89              Locale locale = fields[i].getLocale();
90              if (locale != null)
91                  ret[i] = Collator.getInstance (locale);
92          }
93          return ret;
94      }
95  
96  
97      /**
98       * Returns whether <code>a</code> is less relevant than <code>b</code>.
99       * @param a ScoreDoc
100      * @param b ScoreDoc
101      * @return <code>true</code> if document <code>a</code> should be sorted after document <code>b</code>.
102      */
103     protected final boolean lessThan (final Object a, final Object b) {
104         final FieldDoc docA = (FieldDoc) a;
105         final FieldDoc docB = (FieldDoc) b;
106         final int n = fields.length;
107         int c = 0;
108         for (int i=0; i<n && c==0; ++i) {
109             final int type = fields[i].getType();
110             switch (type) {
111                 case SortField.SCORE:{
112                     float r1 = ((Float)docA.fields[i]).floatValue();
113                     float r2 = ((Float)docB.fields[i]).floatValue();
114                     if (r1 > r2) c = -1;
115                     if (r1 < r2) c = 1;
116                     break;
117         }
118         case SortField.DOC:
119                 case SortField.INT:{
120                     int i1 = ((Integer)docA.fields[i]).intValue();
121                     int i2 = ((Integer)docB.fields[i]).intValue();
122                     if (i1 < i2) c = -1;
123                     if (i1 > i2) c = 1;
124                     break;
125         }
126         case SortField.LONG:{
127                     long l1 = ((Long)docA.fields[i]).longValue();
128                     long l2 = ((Long)docB.fields[i]).longValue();
129                     if (l1 < l2) c = -1;
130                     if (l1 > l2) c = 1;
131                     break;
132         }
133         case SortField.STRING:{
134                     String s1 = (String) docA.fields[i];
135                     String s2 = (String) docB.fields[i];
136                     // null values need to be sorted first, because of how FieldCache.getStringIndex()
137                     // works - in that routine, any documents without a value in the given field are
138                     // put first.  If both are null, the next SortField is used
139                     if (s1 == null) c = (s2==null) ? 0 : -1;
140                     else if (s2 == null) c = 1;  // 
141                     else if (fields[i].getLocale() == null) {
142                         c = s1.compareTo(s2);
143                     } else {
144                         c = collators[i].compare (s1, s2);
145                     }
146                     break;
147         }
148         case SortField.FLOAT:{
149                     float f1 = ((Float)docA.fields[i]).floatValue();
150                     float f2 = ((Float)docB.fields[i]).floatValue();
151                     if (f1 < f2) c = -1;
152                     if (f1 > f2) c = 1;
153                     break;
154         }
155         case SortField.DOUBLE:{
156                     double d1 = ((Double)docA.fields[i]).doubleValue();
157                     double d2 = ((Double)docB.fields[i]).doubleValue();
158                     if (d1 < d2) c = -1;
159                     if (d1 > d2) c = 1;
160                     break;
161         }
162         case SortField.BYTE:{
163                     int i1 = ((Byte)docA.fields[i]).byteValue();
164                     int i2 = ((Byte)docB.fields[i]).byteValue();
165                     if (i1 < i2) c = -1;
166                     if (i1 > i2) c = 1;
167                     break;
168         }
169         case SortField.SHORT:{
170                     int i1 = ((Short)docA.fields[i]).shortValue();
171                     int i2 = ((Short)docB.fields[i]).shortValue();
172                     if (i1 < i2) c = -1;
173                     if (i1 > i2) c = 1;
174                     break;
175         }
176         case SortField.CUSTOM:{
177                     c = docA.fields[i].compareTo (docB.fields[i]);
178                     break;
179         }
180         case SortField.AUTO:{
181                     // we cannot handle this - even if we determine the type of object (Float or
182                     // Integer), we don't necessarily know how to compare them (both SCORE and
183                     // FLOAT contain floats, but are sorted opposite of each other). Before
184                     // we get here, each AUTO should have been replaced with its actual value.
185                     throw new RuntimeException ("FieldDocSortedHitQueue cannot use an AUTO SortField");
186         }
187         default:{
188                     throw new RuntimeException ("invalid SortField type: "+type);
189         }
190       }
191             if (fields[i].getReverse()) {
192                 c = -c;
193             }
194         }
195 
196     // avoid random sort order that could lead to duplicates (bug #31241):
197     if (c == 0)
198       return docA.doc > docB.doc;
199 
200     return c > 0;
201     }
202 }
