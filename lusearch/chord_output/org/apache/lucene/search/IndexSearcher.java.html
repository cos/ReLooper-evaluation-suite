32  /** Implements search over a single IndexReader.
33   *
34   * <p>Applications usually need only call the inherited
35   * {@link #search(Query,int)}
36   * or {@link #search(Query,Filter,int)} methods. For performance reasons it is 
37   * recommended to open only one IndexSearcher and use it for all of your searches.
38   * 
39   * <p>Note that you can only access the deprecated {@link Hits} from an IndexSearcher as long as it is
40   * not yet closed, otherwise an IOException will be thrown. 
41   *
42   * <a name="thread-safety"></a><p><b>NOTE</b>: {@link
43   * <code>IndexSearcher</code>} instances are completely
44   * thread safe, meaning multiple threads can call any of its
45   * methods, concurrently.  If your application requires
46   * external synchronization, you should <b>not</b>
47   * synchronize on the <code>IndexSearcher</code> instance;
48   * use your own (non-Lucene) objects instead.</p>
49   */
50  public class IndexSearcher extends Searcher {

110   private IndexSearcher(IndexReader r, boolean closeReader) {
111     reader = r;
112     this.closeReader = closeReader;
113 
114     List subReadersList = new ArrayList();
115     gatherSubReaders(subReadersList, reader);
116     subReaders = (IndexReader[]) subReadersList.toArray(new IndexReader[subReadersList.size()]);
117     docStarts = new int[subReaders.length];
118     int maxDoc = 0;
119     for (int i = 0; i < subReaders.length; i++) {
120       docStarts[i] = maxDoc;
121       maxDoc += subReaders[i].maxDoc();
122     }
123   }
124 
125   protected void gatherSubReaders(List allSubReaders, IndexReader r) {
126     ReaderUtil.gatherSubReaders(allSubReaders, r);
127   }

166   public TopDocs search(Weight weight, Filter filter, final int nDocs) throws IOException {
167 
168     if (nDocs <= 0) {
169       throw new IllegalArgumentException("nDocs must be > 0");
170     }
171 
172     TopScoreDocCollector collector = TopScoreDocCollector.create(nDocs, !weight.scoresDocsOutOfOrder());
173     search(weight, filter, collector);
174     return collector.topDocs();
175   }

177   public TopFieldDocs search(Weight weight, Filter filter,
178       final int nDocs, Sort sort) throws IOException {
179     return search(weight, filter, nDocs, sort, true);
180   }

182   /**
183    * Just like {@link #search(Weight, Filter, int, Sort)}, but you choose
184    * whether or not the fields in the returned {@link FieldDoc} instances should
185    * be set by specifying fillFields.<br>
186    *
187    * <p>NOTE: this does not compute scores by default.  If you
188    * need scores, create a {@link TopFieldCollector}
189    * instance by calling {@link TopFieldCollector#create} and
190    * then pass that to {@link #search(Weight, Filter,
191    * Collector)}.</p>
192    */
193   public TopFieldDocs search(Weight weight, Filter filter, final int nDocs,
194                              Sort sort, boolean fillFields)
195       throws IOException {
196     
197     SortField[] fields = sort.fields;
198     boolean legacy = false;
199     for(int i = 0; i < fields.length; i++) {
200       SortField field = fields[i];
201       String fieldname = field.getField();
202       int type = field.getType();
203       // Resolve AUTO into its true type
204       if (type == SortField.AUTO) {
205         int autotype = SortField.detectFieldType(reader, fieldname);
206         if (autotype == SortField.STRING) {
207           fields[i] = new SortField (fieldname, field.getLocale(), field.getReverse());
208         } else {
209           fields[i] = new SortField (fieldname, autotype, field.getReverse());
210         }
211       }
212 
213       if (field.getUseLegacySearch()) {
214         legacy = true;
215       }
216     }
217     
218     if (legacy) {
219       // Search the single top-level reader
220       TopDocCollector collector = new TopFieldDocCollector(reader, sort, nDocs);
221       HitCollectorWrapper hcw = new HitCollectorWrapper(collector);
222       hcw.setNextReader(reader, 0);
223       if (filter == null) {
224         Scorer scorer = weight.scorer(reader, true, true);
225         if (scorer != null) {
226           scorer.score(hcw);
227         }
228       } else {
229         searchWithFilter(reader, weight, filter, hcw);
230       }
231       return (TopFieldDocs) collector.topDocs();
232     }
233     
234     TopFieldCollector collector = TopFieldCollector.create(sort, nDocs,
235         fillFields, fieldSortDoTrackScores, fieldSortDoMaxScore, !weight.scoresDocsOutOfOrder());
236     search(weight, filter, collector);
237     return (TopFieldDocs) collector.topDocs();
238   }

240   public void search(Weight weight, Filter filter, Collector collector)
241       throws IOException {
242     
243     if (filter == null) {
244       for (int i = 0; i < subReaders.length; i++) { // search each subreader
245         collector.setNextReader(subReaders[i], docStarts[i]);
246         Scorer scorer = weight.scorer(subReaders[i], !collector.acceptsDocsOutOfOrder(), true);
247         if (scorer != null) {
248           scorer.score(collector);
249         }
250       }
251     } else {
252       for (int i = 0; i < subReaders.length; i++) { // search each subreader
253         collector.setNextReader(subReaders[i], docStarts[i]);
254         searchWithFilter(subReaders[i], weight, filter, collector);
255       }
256     }
257   }

259   private void searchWithFilter(IndexReader reader, Weight weight,
260       final Filter filter, final Collector collector) throws IOException {
261 
262     assert filter != null;
263     
264     Scorer scorer = weight.scorer(reader, true, false);
265     if (scorer == null) {
266       return;
267     }
268 
269     int docID = scorer.docID();
270     assert docID == -1 || docID == DocIdSetIterator.NO_MORE_DOCS;
271 
272     // CHECKME: use ConjunctionScorer here?
273     DocIdSet filterDocIdSet = filter.getDocIdSet(reader);
274     if (filterDocIdSet == null) {
275       // this means the filter does not accept any documents.
276       return;
277     }
278     
279     DocIdSetIterator filterIter = filterDocIdSet.iterator();
280     if (filterIter == null) {
281       // this means the filter does not accept any documents.
282       return;
283     }
284     int filterDoc = filterIter.nextDoc();
285     int scorerDoc = scorer.advance(filterDoc);
286     
287     collector.setScorer(scorer);
288     while (true) {
289       if (scorerDoc == filterDoc) {
290         // Check if scorer has exhausted, only before collecting.
291         if (scorerDoc == DocIdSetIterator.NO_MORE_DOCS) {
292           break;
293         }
294         collector.collect(scorerDoc);
295         filterDoc = filterIter.nextDoc();
296         scorerDoc = scorer.advance(filterDoc);
297       } else if (scorerDoc > filterDoc) {
298         filterDoc = filterIter.advance(scorerDoc);
299       } else {
300         scorerDoc = scorer.advance(filterDoc);
301       }
302     }
303   }

305   public Query rewrite(Query original) throws IOException {
306     Query query = original;
307     for (Query rewrittenQuery = query.rewrite(reader); rewrittenQuery != query;
308          rewrittenQuery = query.rewrite(reader)) {
309       query = rewrittenQuery;
310     }
311     return query;
312   }

314   public Explanation explain(Weight weight, int doc) throws IOException {
315     int n = ReaderUtil.subIndex(doc, docStarts);
316     int deBasedDoc = doc - docStarts[n];
317     
318     return weight.explain(subReaders[n], deBasedDoc);
319   }

322   private boolean fieldSortDoMaxScore;

324   /** By default, no scores are computed when sorting by
325    *  field (using {@link #search(Query,Filter,int,Sort)}).
326    *  You can change that, per IndexSearcher instance, by
327    *  calling this method.  Note that this will incur a CPU
328    *  cost.
329    * 
330    *  @param doTrackScores If true, then scores are
331    *  returned for every matching document in {@link
332    *  TopFieldDocs}.
333    *
334    *  @param doMaxScore If true, then the max score for all
335    *  matching docs is computed. */
336   public void setDefaultFieldSortScoring(boolean doTrackScores, boolean doMaxScore) {
337     fieldSortDoTrackScores = doTrackScores;
338     fieldSortDoMaxScore = doMaxScore;
339   }
