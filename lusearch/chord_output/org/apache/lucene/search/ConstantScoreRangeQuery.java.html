20  import java.text.Collator;
21  
22  /**
23   * A range query that returns a constant score equal to its boost for
24   * all documents in the exclusive range of terms.
25   *
26   * <p>It does not have an upper bound on the number of clauses covered in the range.
27   *
28   * <p>This query matches the documents looking for terms that fall into the
29   * supplied range according to {@link String#compareTo(String)}. It is not intended
30   * for numerical ranges, use {@link NumericRangeQuery} instead.
31   *
32   * <p>This query is hardwired to {@link MultiTermQuery#CONSTANT_SCORE_AUTO_REWRITE_DEFAULT}.
33   * If you want to change this, use {@link TermRangeQuery} instead.
34   *
35   * @deprecated Use {@link TermRangeQuery} for term ranges or
36   * {@link NumericRangeQuery} for numeric ranges instead.
37   * This class will be removed in Lucene 3.0.
38   * @version $Id: ConstantScoreRangeQuery.java 797694 2009-07-25 00:03:33Z mikemccand $
39   */
40  public class ConstantScoreRangeQuery extends TermRangeQuery
41  {
42  
43    public ConstantScoreRangeQuery(String fieldName, String lowerVal, String upperVal, boolean includeLower, boolean includeUpper)
44    {
45      super(fieldName, lowerVal, upperVal, includeLower, includeUpper);
46      rewriteMethod = CONSTANT_SCORE_AUTO_REWRITE_DEFAULT;
47    }
48  
49    public ConstantScoreRangeQuery(String fieldName, String lowerVal,
50                                   String upperVal, boolean includeLower,
51                                   boolean includeUpper, Collator collator) {
52      super(fieldName, lowerVal, upperVal, includeLower, includeUpper, collator);
53      rewriteMethod = CONSTANT_SCORE_AUTO_REWRITE_DEFAULT;
54    }
55  
56    public String getLowerVal() {
57      return getLowerTerm();
58    }
59  
60    public String getUpperVal() {
61      return getUpperTerm();
62    }
63  
64    /** Changes of mode are not supported by this class (fixed to constant score rewrite mode) */
65    public void setRewriteMethod(RewriteMethod method) {
66      throw new UnsupportedOperationException("Use TermRangeQuery instead to change the rewrite method.");
67    }
68  }
