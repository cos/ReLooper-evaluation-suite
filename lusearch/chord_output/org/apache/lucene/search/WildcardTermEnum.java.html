20  import java.io.IOException;
21  
22  import org.apache.lucene.index.IndexReader;
23  import org.apache.lucene.index.Term;
24  
25  /**
26   * Subclass of FilteredTermEnum for enumerating all terms that match the
27   * specified wildcard filter term.
28   * <p>
29   * Term enumerations are always ordered by Term.compareTo().  Each term in
30   * the enumeration is greater than all that precede it.
31   *
32   * @version $Id: WildcardTermEnum.java 783371 2009-06-10 14:39:56Z mikemccand $
33   */
34  public class WildcardTermEnum extends FilteredTermEnum {
35    final Term searchTerm;
36    final String field;
37    final String text;
38    final String pre;
39    final int preLen;
40    boolean endEnum = false;
41  
42    /**
43     * Creates a new <code>WildcardTermEnum</code>.
44     * <p>
45     * After calling the constructor the enumeration is already pointing to the first 
46     * valid term if such a term exists.
47     */
48    public WildcardTermEnum(IndexReader reader, Term term) throws IOException {
49      super();
50      searchTerm = term;
51      field = searchTerm.field();
52      final String searchTermText = searchTerm.text();
53  
54      final int sidx = searchTermText.indexOf(WILDCARD_STRING);
55      final int cidx = searchTermText.indexOf(WILDCARD_CHAR);
56      int idx = sidx;
57      if (idx == -1) {
58        idx = cidx;
59      }
60      else if (cidx >= 0) {
61        idx = Math.min(idx, cidx);
62      }
63      pre = idx != -1?searchTerm.text().substring(0,idx): "";
64  
65      preLen = pre.length();
66      text = searchTermText.substring(preLen);
67      setEnum(reader.terms(new Term(searchTerm.field(), pre)));
68    }
69  
70    protected final boolean termCompare(Term term) {
71      if (field == term.field()) {
72        String searchText = term.text();
73        if (searchText.startsWith(pre)) {
74          return wildcardEquals(text, 0, searchText, preLen);
75        }
76      }
77      endEnum = true;
78      return false;
79    }
80  
81    public float difference() {
82      return 1.0f;
83    }
84  
85    public final boolean endEnum() {
86      return endEnum;
87    }
88  
89    /********************************************
90     * String equality with support for wildcards
91     ********************************************/
92  
93    public static final char WILDCARD_STRING = '*';
94    public static final char WILDCARD_CHAR = '?';
95  
96    /**
97     * Determines if a word matches a wildcard pattern.
98     * <small>Work released by Granta Design Ltd after originally being done on
99     * company time.</small>
100    */
101   public static final boolean wildcardEquals(String pattern, int patternIdx,
102     String string, int stringIdx)
103   {
104     int p = patternIdx;
105     
106     for (int s = stringIdx; ; ++p, ++s)
107       {
108         // End of string yet?
109         boolean sEnd = (s >= string.length());
110         // End of pattern yet?
111         boolean pEnd = (p >= pattern.length());
112 
113         // If we're looking at the end of the string...
114         if (sEnd)
115         {
116           // Assume the only thing left on the pattern is/are wildcards
117           boolean justWildcardsLeft = true;
118 
119           // Current wildcard position
120           int wildcardSearchPos = p;
121           // While we haven't found the end of the pattern,
122           // and haven't encountered any non-wildcard characters
123           while (wildcardSearchPos < pattern.length() && justWildcardsLeft)
124           {
125             // Check the character at the current position
126             char wildchar = pattern.charAt(wildcardSearchPos);
127             
128             // If it's not a wildcard character, then there is more
129             // pattern information after this/these wildcards.
130             if (wildchar != WILDCARD_CHAR && wildchar != WILDCARD_STRING)
131             {
132               justWildcardsLeft = false;
133             }
134             else
135             {
136               // to prevent "cat" matches "ca??"
137               if (wildchar == WILDCARD_CHAR) {
138                 return false;
139               }
140               
141               // Look at the next character
142               wildcardSearchPos++;
143             }
144           }
145 
146           // This was a prefix wildcard search, and we've matched, so
147           // return true.
148           if (justWildcardsLeft)
149           {
150             return true;
151           }
152         }
153 
154         // If we've gone past the end of the string, or the pattern,
155         // return false.
156         if (sEnd || pEnd)
157         {
158           break;
159         }
160 
161         // Match a single character, so continue.
162         if (pattern.charAt(p) == WILDCARD_CHAR)
163         {
164           continue;
165         }
166 
167         //
168         if (pattern.charAt(p) == WILDCARD_STRING)
169         {
170           // Look at the character beyond the '*'.
171           ++p;
172           // Examine the string, starting at the last character.
173           for (int i = string.length(); i >= s; --i)
174           {
175             if (wildcardEquals(pattern, p, string, i))
176             {
177               return true;
178             }
179           }
180           break;
181         }
182         if (pattern.charAt(p) != string.charAt(s))
183         {
184           break;
185         }
186       }
187       return false;
188   }
189 }
