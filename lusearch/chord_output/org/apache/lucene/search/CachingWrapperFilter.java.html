20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.util.DocIdBitSet;
22  import org.apache.lucene.util.OpenBitSetDISI;
23  import java.util.BitSet;
24  import java.util.WeakHashMap;
25  import java.util.Map;
26  import java.io.IOException;
27  
28  /**
29   * Wraps another filter's result and caches it.  The purpose is to allow
30   * filters to simply filter, and then wrap with this class to add caching.
31   */
32  public class CachingWrapperFilter extends Filter {
33    protected Filter filter;
34  
35    /**
36     * A transient Filter cache.
37     */
38    protected transient Map cache;
39  
40    /**
41     * @param filter Filter to cache results of
42     */
43    public CachingWrapperFilter(Filter filter) {
44      this.filter = filter;
45    }
46  
47    /**
48     * @deprecated Use {@link #getDocIdSet(IndexReader)} instead.
49     */
50    public BitSet bits(IndexReader reader) throws IOException {
51      if (cache == null) {
52        cache = new WeakHashMap();
53      }
54  
55      Object cached = null;
56      synchronized (cache) {  // check cache
57        cached = cache.get(reader);
58      }
59      
60      if (cached != null) {
61        if (cached instanceof BitSet) {
62          return (BitSet) cached;
63        } else if (cached instanceof DocIdBitSet)
64          return ((DocIdBitSet) cached).getBitSet();
65        // It would be nice to handle the DocIdSet case, but that's not really possible
66      }
67  
68      final BitSet bits = filter.bits(reader);
69  
70      synchronized (cache) {  // update cache
71        cache.put(reader, bits);
72      }
73  
74      return bits;
75    }
76  
77    /** Provide the DocIdSet to be cached, using the DocIdSet provided
78     *  by the wrapped Filter.
79     *  <p>This implementation returns the given {@link DocIdSet}, if {@link DocIdSet#isCacheable}
80     *  returns <code>true</code>, else it copies the {@link DocIdSetIterator} into
81     *  an {@link OpenBitSetDISI}.
82     */
83    protected DocIdSet docIdSetToCache(DocIdSet docIdSet, IndexReader reader) throws IOException {
84      if (docIdSet.isCacheable()) {
85        return docIdSet;
86      } else {
87        final DocIdSetIterator it = docIdSet.iterator();
88        // null is allowed to be returned by iterator(),
89        // in this case we wrap with the empty set,
90        // which is cacheable.
91        return (it == null) ? DocIdSet.EMPTY_DOCIDSET : new OpenBitSetDISI(it, reader.maxDoc());
92      }
93    }
94    
95    public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
96      if (cache == null) {
97        cache = new WeakHashMap();
98      }
99  
100     Object cached = null;
101     synchronized (cache) {  // check cache
102       cached = cache.get(reader);
103     }
104 
105     if (cached != null) {
106       if (cached instanceof DocIdSet)
107         return (DocIdSet) cached;
108       else
109         return new DocIdBitSet((BitSet) cached);
110     }
111 
112     final DocIdSet docIdSet = docIdSetToCache(filter.getDocIdSet(reader), reader);
113 
114     if (docIdSet != null) {
115       synchronized (cache) {  // update cache
116         cache.put(reader, docIdSet);
117       }
118     }
119 
120     return docIdSet;
121   }
122 
123   public String toString() {
124     return "CachingWrapperFilter("+filter+")";
125   }
126 
127   public boolean equals(Object o) {
128     if (!(o instanceof CachingWrapperFilter)) return false;
129     return this.filter.equals(((CachingWrapperFilter)o).filter);
130   }
131 
132   public int hashCode() {
133     return filter.hashCode() ^ 0x1117BF25;  
134   }
135 }
