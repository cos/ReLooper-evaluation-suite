26  /**
27   * A range filter built on top of a cached single term field (in {@link FieldCache}).
28   * 
29   * <p>FieldCacheRangeFilter builds a single cache for the field the first time it is used.
30   * Each subsequent FieldCacheRangeFilter on the same field then reuses this cache,
31   * even if the range itself changes. 
32   * 
33   * <p>This means that FieldCacheRangeFilter is much faster (sometimes more than 100x as fast) 
34   * as building a {@link TermRangeFilter} (or {@link ConstantScoreRangeQuery} on a {@link TermRangeFilter})
35   * for each query, if using a {@link #newStringRange}. However, if the range never changes it
36   * is slower (around 2x as slow) than building a CachingWrapperFilter on top of a single TermRangeFilter.
37   *
38   * For numeric data types, this filter may be significantly faster than {@link NumericRangeFilter}.
39   * Furthermore, it does not need the numeric values encoded by {@link NumericField}. But
40   * it has the problem that it only works with exact one value/document (see below).
41   *
42   * <p>As with all {@link FieldCache} based functionality, FieldCacheRangeFilter is only valid for 
43   * fields which exact one term for each document (except for {@link #newStringRange}
44   * where 0 terms are also allowed). Due to a restriction of {@link FieldCache}, for numeric ranges
45   * all terms that do not have a numeric value, 0 is assumed.
46   *
47   * <p>Thus it works on dates, prices and other single value fields but will not work on
48   * regular text fields. It is preferable to use a <code>NOT_ANALYZED</code> field to ensure that
49   * there is only a single term. 
50   *
51   * <p>This class does not have an constructor, use one of the static factory methods available,
52   * that create a correct instance for different data types supported by {@link FieldCache}.
53   */
54

70    }

75    /**
76     * Creates a string range query using {@link FieldCache#getStringIndex}. This works with all
77     * fields containing zero or one term in the field. The range can be half-open by setting one
78     * of the values to <code>null</code>.
79     */
80    public static FieldCacheRangeFilter newStringRange(String field, String lowerVal, String upperVal, boolean includeLower, boolean includeUpper) {
81      return new FieldCacheRangeFilter(field, null, lowerVal, upperVal, includeLower, includeUpper) {
82        public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
83          final FieldCache.StringIndex fcsi = FieldCache.DEFAULT.getStringIndex(reader, field);
84          final int lowerPoint = fcsi.binarySearchLookup((String) lowerVal);
85          final int upperPoint = fcsi.binarySearchLookup((String) upperVal);
86          
87          final int inclusiveLowerPoint, inclusiveUpperPoint;
88  
89          // Hints:
90          // * binarySearchLookup returns 0, if value was null.
91          // * the value is <0 if no exact hit was found, the returned value
92          //   is (-(insertion point) - 1)
93          if (lowerPoint == 0) {
94            assert lowerVal == null;
95            inclusiveLowerPoint = 1;
96          } else if (includeLower && lowerPoint > 0) {
97            inclusiveLowerPoint = lowerPoint;
98          } else if (lowerPoint > 0) {
99            inclusiveLowerPoint = lowerPoint + 1;
100         } else {
101           inclusiveLowerPoint = Math.max(1, -lowerPoint - 1);
102         }
103         
104         if (upperPoint == 0) {
105           assert upperVal == null;
106           inclusiveUpperPoint = Integer.MAX_VALUE;  
107         } else if (includeUpper && upperPoint > 0) {
108           inclusiveUpperPoint = upperPoint;
109         } else if (upperPoint > 0) {
110           inclusiveUpperPoint = upperPoint - 1;
111         } else {
112           inclusiveUpperPoint = -upperPoint - 2;
113         }      
114 
115         if (inclusiveUpperPoint <= 0 || inclusiveLowerPoint > inclusiveUpperPoint)
116           return DocIdSet.EMPTY_DOCIDSET;
117         
118         assert inclusiveLowerPoint > 0 && inclusiveUpperPoint > 0;
119         
120         // for this DocIdSet, we never need to use TermDocs,
121         // because deleted docs have an order of 0 (null entry in StringIndex)
122         return new FieldCacheDocIdSet(reader, false) {
123           final boolean matchDoc(int doc) {
124             return fcsi.order[doc] >= inclusiveLowerPoint && fcsi.order[doc] <= inclusiveUpperPoint;
125           }
126         };
127       }
128     };
129   }

140   /**
141    * Creates a numeric range query using {@link FieldCache#getBytes(IndexReader,String,FieldCache.ByteParser)}. This works with all
142    * byte fields containing exactly one numeric term in the field. The range can be half-open by setting one
143    * of the values to <code>null</code>.
144    */
145   public static FieldCacheRangeFilter newByteRange(String field, FieldCache.ByteParser parser, Byte lowerVal, Byte upperVal, boolean includeLower, boolean includeUpper) {
146     return new FieldCacheRangeFilter(field, parser, lowerVal, upperVal, includeLower, includeUpper) {
147       public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
148         final byte inclusiveLowerPoint, inclusiveUpperPoint;
149         if (lowerVal != null) {
150           final byte i = ((Number) lowerVal).byteValue();
151           if (!includeLower && i == Byte.MAX_VALUE)
152             return DocIdSet.EMPTY_DOCIDSET;
153           inclusiveLowerPoint = (byte) (includeLower ?  i : (i + 1));
154         } else {
155           inclusiveLowerPoint = Byte.MIN_VALUE;
156         }
157         if (upperVal != null) {
158           final byte i = ((Number) upperVal).byteValue();
159           if (!includeUpper && i == Byte.MIN_VALUE)
160             return DocIdSet.EMPTY_DOCIDSET;
161           inclusiveUpperPoint = (byte) (includeUpper ? i : (i - 1));
162         } else {
163           inclusiveUpperPoint = Byte.MAX_VALUE;
164         }
165         
166         if (inclusiveLowerPoint > inclusiveUpperPoint)
167           return DocIdSet.EMPTY_DOCIDSET;
168         
169         final byte[] values = FieldCache.DEFAULT.getBytes(reader, field, (FieldCache.ByteParser) parser);
170         // we only request the usage of termDocs, if the range contains 0
171         return new FieldCacheDocIdSet(reader, (inclusiveLowerPoint <= 0 && inclusiveUpperPoint >= 0)) {
172           boolean matchDoc(int doc) {
173             return values[doc] >= inclusiveLowerPoint && values[doc] <= inclusiveUpperPoint;
174           }
175         };
176       }
177     };
178   }

337    * Creates a numeric range query using {@link FieldCache#getFloats(IndexReader,String,FieldCache.FloatParser)}. This works with all

342     return new FieldCacheRangeFilter(field, parser, lowerVal, upperVal, includeLower, includeUpper) {
343       public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
344         // we transform the floating point numbers to sortable integers
345         // using NumericUtils to easier find the next bigger/lower value
346         final float inclusiveLowerPoint, inclusiveUpperPoint;

351           int i = NumericUtils.floatToSortableInt(f);
352           inclusiveLowerPoint = NumericUtils.sortableIntToFloat( includeLower ?  i : (i + 1) );
353         } else {
354           inclusiveLowerPoint = Float.NEGATIVE_INFINITY;
355         }
356         if (upperVal != null) {
357           float f = ((Number) upperVal).floatValue();
358           if (!includeUpper && f < 0.0f && Float.isInfinite(f))
359             return DocIdSet.EMPTY_DOCIDSET;

362         } else {

364         }
365

433   public final String toString() {
434     final StringBuffer sb = new StringBuffer(field).append(":");
435     return sb.append(includeLower ? '[' : '{')
436       .append((lowerVal == null) ? "*" : lowerVal.toString())
437       .append(" TO ")
438       .append((upperVal == null) ? "*" : upperVal.toString())
439       .append(includeUpper ? ']' : '}')
440       .toString();
441   }

443   public final boolean equals(Object o) {
444     if (this == o) return true;
445     if (!(o instanceof FieldCacheRangeFilter)) return false;
446     FieldCacheRangeFilter other = (FieldCacheRangeFilter) o;
447 
448     if (!this.field.equals(other.field)
449         || this.includeLower != other.includeLower
450         || this.includeUpper != other.includeUpper
451     ) { return false; }
452     if (this.lowerVal != null ? !this.lowerVal.equals(other.lowerVal) : other.lowerVal != null) return false;
453     if (this.upperVal != null ? !this.upperVal.equals(other.upperVal) : other.upperVal != null) return false;
454     if (this.parser != null ? !this.parser.equals(other.parser) : other.parser != null) return false;
455     return true;
456   }

458   public final int hashCode() {
459     int h = field.hashCode();
460     h ^= (lowerVal != null) ? lowerVal.hashCode() : 550356204;
461     h = (h << 1) | (h >>> 31);  // rotate to distinguish lower from upper
462     h ^= (upperVal != null) ? upperVal.hashCode() : -1674416163;
463     h ^= (parser != null) ? parser.hashCode() : -1572457324;
464     h ^= (includeLower ? 1549299360 : -365038026) ^ (includeUpper ? 1721088258 : 1948649653);
465     return h;
466   }

468   static abstract class FieldCacheDocIdSet extends DocIdSet {
469     private final IndexReader reader;
470     private boolean mayUseTermDocs;
471   
472     FieldCacheDocIdSet(IndexReader reader, boolean mayUseTermDocs) {
473       this.reader = reader;
474       this.mayUseTermDocs = mayUseTermDocs;
475     }
476   
477     /** this method checks, if a doc is a hit, should throw AIOBE, when position invalid */
478     abstract boolean matchDoc(int doc) throws ArrayIndexOutOfBoundsException;
479     
480     /** this DocIdSet is cacheable, if it works solely with FieldCache and no TermDocs */
481     public boolean isCacheable() {
482       return !(mayUseTermDocs && reader.hasDeletions());
483     }
484 
485     public DocIdSetIterator iterator() throws IOException {
486       // Synchronization needed because deleted docs BitVector
487       // can change after call to hasDeletions until TermDocs creation.
488       // We only use an iterator with termDocs, when this was requested (e.g. range contains 0)
489       // and the index has deletions
490       final TermDocs termDocs;
491       synchronized(reader) {
492         termDocs = isCacheable() ? null : reader.termDocs(null);
493       }
494       if (termDocs != null) {
495         // a DocIdSetIterator using TermDocs to iterate valid docIds
496         return new DocIdSetIterator() {
497           private int doc = -1;
498           
499           /** @deprecated use {@link #nextDoc()} instead. */
500           public boolean next() throws IOException {
501             return nextDoc() != NO_MORE_DOCS;
502           }
503           
504           /** @deprecated use {@link #advance(int)} instead. */
505           public boolean skipTo(int target) throws IOException {
506             return advance(target) != NO_MORE_DOCS;
507           }
508 
509           /** @deprecated use {@link #docID()} instead. */
510           public int doc() {
511             return termDocs.doc();
512           }
513           
514           public int docID() {
515             return doc;
516           }
517           
518           public int nextDoc() throws IOException {
519             do {
520               if (!termDocs.next())
521                 return doc = NO_MORE_DOCS;
522             } while (!matchDoc(doc = termDocs.doc()));
523             return doc;
524           }
525           
526           public int advance(int target) throws IOException {
527             if (!termDocs.skipTo(target))
528               return doc = NO_MORE_DOCS;
529             while (!matchDoc(doc = termDocs.doc())) { 
530               if (!termDocs.next())
531                 return doc = NO_MORE_DOCS;
532             }
533             return doc;
534           }
535         };
536       } else {
537         // a DocIdSetIterator generating docIds by incrementing a variable -
538         // this one can be used if there are no deletions are on the index
539         return new DocIdSetIterator() {
540           private int doc = -1;
541           
542           /** @deprecated use {@link #nextDoc()} instead. */
543           public boolean next() throws IOException {
544             return nextDoc() != NO_MORE_DOCS;
545           }
546           
547           /** @deprecated use {@link #advance(int)} instead. */
548           public boolean skipTo(int target) throws IOException {
549             return advance(target) != NO_MORE_DOCS;
550           }
551 
552           /** @deprecated use {@link #docID()} instead. */
553           public int doc() {
554             return doc;
555           }
556 
557           public int docID() {
558             return doc;
559           }
560           
561           public int nextDoc() {
562             try {
563               do {
564                 doc++;
565               } while (!matchDoc(doc));
566               return doc;
