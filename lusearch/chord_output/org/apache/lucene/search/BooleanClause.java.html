22  /** A clause in a BooleanQuery. */
23  public class BooleanClause implements java.io.Serializable {
24    
25    /** Specifies how clauses are to occur in matching documents. */
26    public static final class Occur extends Parameter implements java.io.Serializable {
27      
28      private Occur(String name) {
29        // typesafe enum pattern, no public constructor
30        super(name);
31      }
32  
33      public String toString() {
34        if (this == MUST) return "+";
35        if (this == MUST_NOT) return "-";
36        return "";
37      }
38  
39      /** Use this operator for clauses that <i>must</i> appear in the matching documents. */
40      public static final Occur MUST = new Occur("MUST");
41      /** Use this operator for clauses that <i>should</i> appear in the 
42       * matching documents. For a BooleanQuery with no <code>MUST</code> 
43       * clauses one or more <code>SHOULD</code> clauses must match a document 
44       * for the BooleanQuery to match.
45       * @see BooleanQuery#setMinimumNumberShouldMatch
46       */
47      public static final Occur SHOULD = new Occur("SHOULD");
48      /** Use this operator for clauses that <i>must not</i> appear in the matching documents.
49       * Note that it is not possible to search for queries that only consist
50       * of a <code>MUST_NOT</code> clause. */
51      public static final Occur MUST_NOT = new Occur("MUST_NOT");
52      
53    }
54  
55    /** The query whose matching documents are combined by the boolean query.
56     */
57    private Query query;
58  
59    private Occur occur;
60  
61  
62    /** Constructs a BooleanClause.
63    */ 
64    public BooleanClause(Query query, Occur occur) {
65      this.query = query;
66      this.occur = occur;
67      
68    }
69  
70    public Occur getOccur() {
71      return occur;
72    }
73  
74    public void setOccur(Occur occur) {
75      this.occur = occur;
76  
77    }
78  
79    public Query getQuery() {
80      return query;
81    }
82  
83    public void setQuery(Query query) {
84      this.query = query;
85    }
86    
87    public boolean isProhibited() {
88      return Occur.MUST_NOT.equals(occur);
89    }
90  
91    public boolean isRequired() {
92      return Occur.MUST.equals(occur);
93    }
94  
95  
96  
97    /** Returns true if <code>o</code> is equal to this. */
98    public boolean equals(Object o) {
99      if (o == null || !(o instanceof BooleanClause))
100       return false;
101     BooleanClause other = (BooleanClause)o;
102     return this.query.equals(other.query)
103       && this.occur.equals(other.occur);
104   }
105 
106   /** Returns a hash code value for this object.*/
107   public int hashCode() {
108     return query.hashCode() ^ (Occur.MUST.equals(occur)?1:0) ^ (Occur.MUST_NOT.equals(occur)?2:0);
109   }
110 
111 
112   public String toString() {
113     return occur.toString() + query.toString();
114   }
115 }
