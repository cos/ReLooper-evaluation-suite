20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.search.FieldCache;
22  import org.apache.lucene.search.function.DocValues;
23  
24  import java.io.IOException;
25  
26  /**
27   * Expert: obtains single byte field values from the 
28   * {@link org.apache.lucene.search.FieldCache FieldCache}
29   * using <code>getBytes()</code> and makes those values 
30   * available as other numeric types, casting as needed.
31   * 
32   * <p><font color="#FF0000">
33   * WARNING: The status of the <b>search.function</b> package is experimental. 
34   * The APIs introduced here might change in the future and will not be 
35   * supported anymore in such a case.</font>
36   * 
37   * @see org.apache.lucene.search.function.FieldCacheSource for requirements 
38   * on the field. 
39   *
40   * <p><b>NOTE</b>: with the switch in 2.9 to segment-based
41   * searching, if {@link #getValues} is invoked with a
42   * composite (multi-segment) reader, this can easily cause
43   * double RAM usage for the values in the FieldCache.  It's
44   * best to switch your application to pass only atomic
45   * (single segment) readers to this API.  Alternatively, for
46   * a short-term fix, you could wrap your ValueSource using
47   * {@link MultiValueSource}, which costs more CPU per lookup
48   * but will not consume double the FieldCache RAM.</p>
49   */
50  public class ByteFieldSource extends FieldCacheSource {
51    private FieldCache.ByteParser parser;
52  
53    /**
54     * Create a cached byte field source with default string-to-byte parser. 
55     */
56    public ByteFieldSource(String field) {
57      this(field, null);
58    }
59  
60    /**
61     * Create a cached byte field source with a specific string-to-byte parser. 
62     */
63    public ByteFieldSource(String field, FieldCache.ByteParser parser) {
64      super(field);
65      this.parser = parser;
66    }
67  
68    /*(non-Javadoc) @see org.apache.lucene.search.function.ValueSource#description() */
69    public String description() {
70      return "byte(" + super.description() + ')';
71    }
72  
73    /*(non-Javadoc) @see org.apache.lucene.search.function.FieldCacheSource#getCachedValues(org.apache.lucene.search.FieldCache, java.lang.String, org.apache.lucene.index.IndexReader) */
74    public DocValues getCachedFieldValues (FieldCache cache, String field, IndexReader reader) throws IOException {
75      final byte[] arr = cache.getBytes(reader, field, parser);
76      return new DocValues() {
77        /*(non-Javadoc) @see org.apache.lucene.search.function.DocValues#floatVal(int) */
78        public float floatVal(int doc) { 
79          return (float) arr[doc]; 
80        }
81        /*(non-Javadoc) @see org.apache.lucene.search.function.DocValues#intVal(int) */
82        public  int intVal(int doc) { 
83          return arr[doc]; 
84        }
85        /*(non-Javadoc) @see org.apache.lucene.search.function.DocValues#toString(int) */
86        public String toString(int doc) { 
87          return  description() + '=' + intVal(doc);  
88        }
89        /*(non-Javadoc) @see org.apache.lucene.search.function.DocValues#getInnerArray() */
90        Object getInnerArray() {
91          return arr;
92        }
93      };
94    }
95  
96    /*(non-Javadoc) @see org.apache.lucene.search.function.FieldCacheSource#cachedFieldSourceEquals(org.apache.lucene.search.function.FieldCacheSource) */
97    public boolean cachedFieldSourceEquals(FieldCacheSource o) {
98      if (o.getClass() !=  ByteFieldSource.class) {
99        return false;
100     }
101     ByteFieldSource other = (ByteFieldSource)o;
102     return this.parser==null ? 
103       other.parser==null :
104       this.parser.getClass() == other.parser.getClass();
105   }
106 
107   /*(non-Javadoc) @see org.apache.lucene.search.function.FieldCacheSource#cachedFieldSourceHashCode() */
108   public int cachedFieldSourceHashCode() {
109     return parser==null ? 
110       Byte.class.hashCode() : parser.getClass().hashCode();
111   }
112 
113 }
