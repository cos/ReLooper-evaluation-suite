20  import java.io.IOException;
21  
22  import org.apache.lucene.util.ReaderUtil;
23  import org.apache.lucene.index.IndexReader;
24  import org.apache.lucene.search.Explanation;
25  
26  /** This class wraps another ValueSource, but protects
27   *  against accidental double RAM usage in FieldCache when
28   *  a composite reader is passed to {@link #getValues}.
29   *
30   *  <p><b>NOTE</b>: this class adds a CPU penalty to every
31   *  lookup, as it must resolve the incoming document to the
32   *  right sub-reader using a binary search.</p>
33   *
34   *  @deprecated This class is temporary, to ease the
35   *  migration to segment-based searching. Please change your
36   *  code to not pass composite readers to these APIs. */
37  
38  public final class MultiValueSource extends ValueSource {
39  
40    final ValueSource other;
41    public MultiValueSource(ValueSource other) {
42      this.other = other;
43    }
44  
45    public DocValues getValues(IndexReader reader) throws IOException {
46  
47      IndexReader[] subReaders = reader.getSequentialSubReaders();
48      if (subReaders != null) {
49        // This is a composite reader
50        return new MultiDocValues(subReaders);
51      } else {
52        // Already an atomic reader -- just delegate
53        return other.getValues(reader);
54      }
55    }
56  
57    public String description() {
58      return other.description();
59    }
60  
61    public boolean equals(Object o) {
62      if (o instanceof MultiValueSource) {
63        return ((MultiValueSource) o).other.equals(other);
64      } else {
65        return false;
66      }
67    }
68  
69    public int hashCode() {
70      return 31 * other.hashCode();
71    }
72  
73    private final class MultiDocValues extends DocValues {
74  
75      final DocValues[] docValues;
76      final int[] docStarts;
77  
78      MultiDocValues(IndexReader[] subReaders) throws IOException {
79        docValues = new DocValues[subReaders.length];
80        docStarts = new int[subReaders.length];
81        int base = 0;
82        for(int i=0;i<subReaders.length;i++) {
83          docValues[i] = other.getValues(subReaders[i]);
84          docStarts[i] = base;
85          base += subReaders[i].maxDoc();
86        }
87      }
88      
89      public float floatVal(int doc) {
90        final int n = ReaderUtil.subIndex(doc, docStarts);
91        return docValues[n].floatVal(doc-docStarts[n]);
92      }
93  
94      public int intVal(int doc) {
95        final int n = ReaderUtil.subIndex(doc, docStarts);
96        return docValues[n].intVal(doc-docStarts[n]);
97      }
98  
99      public long longVal(int doc) {
100       final int n = ReaderUtil.subIndex(doc, docStarts);
101       return docValues[n].longVal(doc-docStarts[n]);
102     }
103 
104     public double doubleVal(int doc) {
105       final int n = ReaderUtil.subIndex(doc, docStarts);
106       return docValues[n].doubleVal(doc-docStarts[n]);
107     }
108 
109     public String strVal(int doc) {
110       final int n = ReaderUtil.subIndex(doc, docStarts);
111       return docValues[n].strVal(doc-docStarts[n]);
112     }
113 
114     public String toString(int doc) {
115       final int n = ReaderUtil.subIndex(doc, docStarts);
116       return docValues[n].toString(doc-docStarts[n]);
117     }
118 
119     public Explanation explain(int doc) {
120       final int n = ReaderUtil.subIndex(doc, docStarts);
121       return docValues[n].explain(doc-docStarts[n]);
122     }
123   }
124 }
