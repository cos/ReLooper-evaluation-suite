33  /**
34   * Query that sets document score as a programmatic function of several (sub) scores:
35   * <ol>
36   *    <li>the score of its subQuery (any query)</li>
37   *    <li>(optional) the score of its ValueSourceQuery (or queries).
38   *        For most simple/convenient use cases this query is likely to be a 
39   *        {@link org.apache.lucene.search.function.FieldScoreQuery FieldScoreQuery}</li>
40   * </ol>
41   * Subclasses can modify the computation by overriding {@link #getCustomScoreProvider}.
42   * 
43   * <p><font color="#FF0000">
44   * WARNING: The status of the <b>search.function</b> package is experimental. 
45   * The APIs introduced here might change in the future and will not be 
46   * supported anymore in such a case.</font>
47   */
48  public class CustomScoreQuery extends Query {
49  
50    private Query subQuery;
51    private ValueSourceQuery[] valSrcQueries; // never null (empty array if there are no valSrcQueries).
52    private boolean strict = false; // if true, valueSource part of query does not take part in weights normalization.  
53    
54    /**
55     * Create a CustomScoreQuery over input subQuery.
56     * @param subQuery the sub query whose scored is being customed. Must not be null. 
57     */
58    public CustomScoreQuery(Query subQuery) {
59      this(subQuery, new ValueSourceQuery[0]);
60    }
61  
62    /**
63     * Create a CustomScoreQuery over input subQuery and a {@link ValueSourceQuery}.
64     * @param subQuery the sub query whose score is being customized. Must not be null.
65     * @param valSrcQuery a value source query whose scores are used in the custom score
66     * computation. For most simple/convenient use case this would be a 
67     * {@link org.apache.lucene.search.function.FieldScoreQuery FieldScoreQuery}.
68     * This parameter is optional - it can be null.
69     */
70    public CustomScoreQuery(Query subQuery, ValueSourceQuery valSrcQuery) {
71        this(subQuery, valSrcQuery!=null ? // don't want an array that contains a single null.. 
72          new ValueSourceQuery[] {valSrcQuery} : new ValueSourceQuery[0]);
73    }

75    /**
76     * Create a CustomScoreQuery over input subQuery and a {@link ValueSourceQuery}.
77     * @param subQuery the sub query whose score is being customized. Must not be null.
78     * @param valSrcQueries value source queries whose scores are used in the custom score
79     * computation. For most simple/convenient use case these would be 
80     * {@link org.apache.lucene.search.function.FieldScoreQuery FieldScoreQueries}.
81     * This parameter is optional - it can be null or even an empty array.
82     */
83    public CustomScoreQuery(Query subQuery, ValueSourceQuery valSrcQueries[]) {
84      this.subQuery = subQuery;
85      this.valSrcQueries = valSrcQueries!=null?
86          valSrcQueries : new ValueSourceQuery[0];
87      if (subQuery == null) throw new IllegalArgumentException("<subquery> must not be null!");
88    }

90    /*(non-Javadoc) @see org.apache.lucene.search.Query#rewrite(org.apache.lucene.index.IndexReader) */
91    public Query rewrite(IndexReader reader) throws IOException {
92      CustomScoreQuery clone = null;
93      
94      final Query sq = subQuery.rewrite(reader);
95      if (sq != subQuery) {
96        clone = (CustomScoreQuery) clone();
97        clone.subQuery = sq;
98      }
99  
100     for(int i = 0; i < valSrcQueries.length; i++) {
101       final ValueSourceQuery v = (ValueSourceQuery) valSrcQueries[i].rewrite(reader);
102       if (v != valSrcQueries[i]) {
103         if (clone == null) clone = (CustomScoreQuery) clone();
104         clone.valSrcQueries[i] = v;
105       }
106     }
107     
108     return (clone == null) ? this : clone;
109   }

111   /*(non-Javadoc) @see org.apache.lucene.search.Query#extractTerms(java.util.Set) */
112   public void extractTerms(Set terms) {
113     subQuery.extractTerms(terms);
114     for(int i = 0; i < valSrcQueries.length; i++) {
115       valSrcQueries[i].extractTerms(terms);
116     }
117   }

119   /*(non-Javadoc) @see org.apache.lucene.search.Query#clone() */
120   public Object clone() {
121     CustomScoreQuery clone = (CustomScoreQuery)super.clone();
122     clone.subQuery = (Query) subQuery.clone();
123     clone.valSrcQueries = new ValueSourceQuery[valSrcQueries.length];
124     for(int i = 0; i < valSrcQueries.length; i++) {
125       clone.valSrcQueries[i] = (ValueSourceQuery) valSrcQueries[i].clone();
126     }
127     return clone;
128   }

130   /* (non-Javadoc) @see org.apache.lucene.search.Query#toString(java.lang.String) */
131   public String toString(String field) {
132     StringBuffer sb = new StringBuffer(name()).append("(");
133     sb.append(subQuery.toString(field));
134     for(int i = 0; i < valSrcQueries.length; i++) {
135       sb.append(", ").append(valSrcQueries[i].toString(field));
136     }
137     sb.append(")");
138     sb.append(strict?" STRICT" : "");
139     return sb.toString() + ToStringUtils.boost(getBoost());
140   }

142   /** Returns true if <code>o</code> is equal to this. */
143   public boolean equals(Object o) {
144     if (getClass() != o.getClass()) {
145       return false;
146     }
147     CustomScoreQuery other = (CustomScoreQuery)o;
148     if (this.getBoost() != other.getBoost() ||
149         !this.subQuery.equals(other.subQuery) ||
150         this.strict != other.strict ||
151         this.valSrcQueries.length != other.valSrcQueries.length) {
152       return false;
153     }
154     for (int i=0; i<valSrcQueries.length; i++) { //TODO simplify with Arrays.deepEquals() once moving to Java 1.5
155       if (!valSrcQueries[i].equals(other.valSrcQueries[i])) {
156         return false;
157       }
158     }
159     return true;
160   }

162   /** Returns a hash code value for this object. */
163   public int hashCode() {
164     int valSrcHash = 0;
165     for (int i=0; i<valSrcQueries.length; i++) { //TODO simplify with Arrays.deepHashcode() once moving to Java 1.5
166       valSrcHash += valSrcQueries[i].hashCode();
167     }
168     return (getClass().hashCode() + subQuery.hashCode() + valSrcHash) ^
169       Float.floatToIntBits(getBoost()) ^ (strict ? 1234 : 4321);
170   }

201   /**
202    * Compute a custom score by the subQuery score and a number of 
203    * ValueSourceQuery scores.
204    * @deprecated Will be removed in Lucene 3.1.
205    * The doc is relative to the current reader, which is
206    * unknown to CustomScoreQuery when using per-segment search (since Lucene 2.9).
207    * Please override {@link #getCustomScoreProvider} and return a subclass
208    * of {@link CustomScoreProvider} for the given {@link IndexReader}.
209    * @see CustomScoreProvider#customScore(int,float,float[])
210    */
211   public float customScore(int doc, float subQueryScore, float valSrcScores[]) {
212     if (valSrcScores.length == 1) {
213       return customScore(doc, subQueryScore, valSrcScores[0]);
214     }
215     if (valSrcScores.length == 0) {
216       return customScore(doc, subQueryScore, 1);
217     }
218     float score = subQueryScore;
219     for(int i = 0; i < valSrcScores.length; i++) {
220       score *= valSrcScores[i];
221     }
222     return score;
223   }
224 
225   /**
226    * Compute a custom score by the subQuery score and the ValueSourceQuery score.
227    * @deprecated Will be removed in Lucene 3.1.
228    * The doc is relative to the current reader, which is
229    * unknown to CustomScoreQuery when using per-segment search (since Lucene 2.9).
230    * Please override {@link #getCustomScoreProvider} and return a subclass
231    * of {@link CustomScoreProvider} for the given {@link IndexReader}.
232    * @see CustomScoreProvider#customScore(int,float,float)
233    */
234   public float customScore(int doc, float subQueryScore, float valSrcScore) {
235     return subQueryScore * valSrcScore;
236   }

238   /**
239    * Explain the custom score.
240    * @deprecated Will be removed in Lucene 3.1.
241    * The doc is relative to the current reader, which is
242    * unknown to CustomScoreQuery when using per-segment search (since Lucene 2.9).
243    * Please override {@link #getCustomScoreProvider} and return a subclass
244    * of {@link CustomScoreProvider} for the given {@link IndexReader}.
245    * @see CustomScoreProvider#customExplain(int,Explanation,Explanation[])
246    */
247   public Explanation customExplain(int doc, Explanation subQueryExpl, Explanation valSrcExpls[]) {
248     if (valSrcExpls.length == 1) {
249       return customExplain(doc, subQueryExpl, valSrcExpls[0]);
250     }
251     if (valSrcExpls.length == 0) {
252       return subQueryExpl;
253     }
254     float valSrcScore = 1;
255     for (int i = 0; i < valSrcExpls.length; i++) {
256       valSrcScore *= valSrcExpls[i].getValue();
257     }
258     Explanation exp = new Explanation( valSrcScore * subQueryExpl.getValue(), "custom score: product of:");
259     exp.addDetail(subQueryExpl);
260     for (int i = 0; i < valSrcExpls.length; i++) {
261       exp.addDetail(valSrcExpls[i]);
262     }
263     return exp;
264   }

266   /**
267    * Explain the custom score.
268    * @deprecated Will be removed in Lucene 3.1.
269    * The doc is relative to the current reader, which is
270    * unknown to CustomScoreQuery when using per-segment search (since Lucene 2.9).
271    * Please override {@link #getCustomScoreProvider} and return a subclass
272    * of {@link CustomScoreProvider} for the given {@link IndexReader}.
273    * @see CustomScoreProvider#customExplain(int,Explanation,Explanation[])
274    */
275   public Explanation customExplain(int doc, Explanation subQueryExpl, Explanation valSrcExpl) {
276     float valSrcScore = 1;
277     if (valSrcExpl != null) {
278       valSrcScore *= valSrcExpl.getValue();
279     }
280     Explanation exp = new Explanation( valSrcScore * subQueryExpl.getValue(), "custom score: product of:");
281     exp.addDetail(subQueryExpl);
282     exp.addDetail(valSrcExpl);
283     return exp;
284   }

288   private class CustomWeight extends Weight {
289     Similarity similarity;
290     Weight subQueryWeight;
291     Weight[] valSrcWeights;
292     boolean qStrict;
293 
294     public CustomWeight(Searcher searcher) throws IOException {
295       this.similarity = getSimilarity(searcher);
296       this.subQueryWeight = subQuery.weight(searcher);
297       this.valSrcWeights = new Weight[valSrcQueries.length];
298       for(int i = 0; i < valSrcQueries.length; i++) {
299         this.valSrcWeights[i] = valSrcQueries[i].createWeight(searcher);
300       }
301       this.qStrict = strict;
302     }
303 
304     /*(non-Javadoc) @see org.apache.lucene.search.Weight#getQuery() */
305     public Query getQuery() {
306       return CustomScoreQuery.this;
307     }
308 
309     /*(non-Javadoc) @see org.apache.lucene.search.Weight#getValue() */
310     public float getValue() {
311       return getBoost();
312     }
313 
314     /*(non-Javadoc) @see org.apache.lucene.search.Weight#sumOfSquaredWeights() */
315     public float sumOfSquaredWeights() throws IOException {
316       float sum = subQueryWeight.sumOfSquaredWeights();
317       for(int i = 0; i < valSrcWeights.length; i++) {
318         if (qStrict) {
319           valSrcWeights[i].sumOfSquaredWeights(); // do not include ValueSource part in the query normalization
320         } else {
321           sum += valSrcWeights[i].sumOfSquaredWeights();
322         }
323       }
324       sum *= getBoost() * getBoost(); // boost each sub-weight
325       return sum ;
326     }
327 
328     /*(non-Javadoc) @see org.apache.lucene.search.Weight#normalize(float) */
329     public void normalize(float norm) {
330       norm *= getBoost(); // incorporate boost
331       subQueryWeight.normalize(norm);
332       for(int i = 0; i < valSrcWeights.length; i++) {
333         if (qStrict) {
334           valSrcWeights[i].normalize(1); // do not normalize the ValueSource part
335         } else {
336           valSrcWeights[i].normalize(norm);
337         }
338       }
339     }
340 
341     public Scorer scorer(IndexReader reader, boolean scoreDocsInOrder, boolean topScorer) throws IOException {
342       // Pass true for "scoresDocsInOrder", because we
343       // require in-order scoring, even if caller does not,
344       // since we call advance on the valSrcScorers.  Pass
345       // false for "topScorer" because we will not invoke
346       // score(Collector) on these scorers:
347       Scorer subQueryScorer = subQueryWeight.scorer(reader, true, false);
348       if (subQueryScorer == null) {
349         return null;
350       }
351       Scorer[] valSrcScorers = new Scorer[valSrcWeights.length];
352       for(int i = 0; i < valSrcScorers.length; i++) {
353          valSrcScorers[i] = valSrcWeights[i].scorer(reader, true, topScorer);
354       }
355       return new CustomScorer(similarity, reader, this, subQueryScorer, valSrcScorers);
356     }
357 
358     public Explanation explain(IndexReader reader, int doc) throws IOException {
359       Explanation explain = doExplain(reader, doc);
360       return explain == null ? new Explanation(0.0f, "no matching docs") : doExplain(reader, doc);
361     }
362     
363     private Explanation doExplain(IndexReader reader, int doc) throws IOException {
364       Scorer[] valSrcScorers = new Scorer[valSrcWeights.length];
365       for(int i = 0; i < valSrcScorers.length; i++) {
366          valSrcScorers[i] = valSrcWeights[i].scorer(reader, true, false);
367       }
368       Explanation subQueryExpl = subQueryWeight.explain(reader, doc);
369       if (!subQueryExpl.isMatch()) {
370         return subQueryExpl;
371       }
372       // match
373       Explanation[] valSrcExpls = new Explanation[valSrcScorers.length];
374       for(int i = 0; i < valSrcScorers.length; i++) {
375         valSrcExpls[i] = valSrcScorers[i].explain(doc);
376       }
377       Explanation customExp = CustomScoreQuery.this.getCustomScoreProvider(reader).customExplain(doc,subQueryExpl,valSrcExpls);
378       float sc = getValue() * customExp.getValue();
379       Explanation res = new ComplexExplanation(
380         true, sc, CustomScoreQuery.this.toString() + ", product of:");
381       res.addDetail(customExp);
382       res.addDetail(new Explanation(getValue(), "queryBoost")); // actually using the q boost as q weight (== weight value)
383       return res;
384     }
385 
386     public boolean scoresDocsOutOfOrder() {
387       return false;
388     }

390   }

395   /**
396    * A scorer that applies a (callback) function on scores of the subQuery.
397    */
398   private class CustomScorer extends Scorer {
399     private final CustomWeight weight;
400     private final float qWeight;
401     private Scorer subQueryScorer;
402     private Scorer[] valSrcScorers;
403     private IndexReader reader;
404     private final CustomScoreProvider provider;
405     private float vScores[]; // reused in score() to avoid allocating this array for each doc 
406 
407     // constructor
408     private CustomScorer(Similarity similarity, IndexReader reader, CustomWeight w,
409         Scorer subQueryScorer, Scorer[] valSrcScorers) throws IOException {
410       super(similarity);
411       this.weight = w;
412       this.qWeight = w.getValue();
413       this.subQueryScorer = subQueryScorer;
414       this.valSrcScorers = valSrcScorers;
415       this.reader = reader;
416       this.vScores = new float[valSrcScorers.length];
417       this.provider = CustomScoreQuery.this.getCustomScoreProvider(reader);
418     }
419 
420     /** @deprecated use {@link #nextDoc()} instead. */
421     public boolean next() throws IOException {
422       return nextDoc() != NO_MORE_DOCS;
423     }
424 
425     public int nextDoc() throws IOException {
426       int doc = subQueryScorer.nextDoc();
427       if (doc != NO_MORE_DOCS) {
428         for (int i = 0; i < valSrcScorers.length; i++) {
429           valSrcScorers[i].advance(doc);
430         }
431       }
432       return doc;
433     }
434 
435     /** @deprecated use {@link #docID()} instead. */
436     public int doc() {
437       return subQueryScorer.doc();
438     }
439 
440     public int docID() {
441       return subQueryScorer.docID();
442     }
443     
444     /*(non-Javadoc) @see org.apache.lucene.search.Scorer#score() */
445     public float score() throws IOException {
446       for (int i = 0; i < valSrcScorers.length; i++) {
447         vScores[i] = valSrcScorers[i].score();
448       }
449       return qWeight * provider.customScore(subQueryScorer.docID(), subQueryScorer.score(), vScores);
450     }
451 
452     /** @deprecated use {@link #advance(int)} instead. */
453     public boolean skipTo(int target) throws IOException {
454       return advance(target) != NO_MORE_DOCS;
455     }
456 
457     public int advance(int target) throws IOException {
458       int doc = subQueryScorer.advance(target);
459       if (doc != NO_MORE_DOCS) {
460         for (int i = 0; i < valSrcScorers.length; i++) {
461           valSrcScorers[i].advance(doc);
462         }
463       }
464       return doc;
465     }
466     
467     // TODO: remove in 3.0
468     /*(non-Javadoc) @see org.apache.lucene.search.Scorer#explain(int) */
469     public Explanation explain(int doc) throws IOException {
470       Explanation subQueryExpl = weight.subQueryWeight.explain(reader,doc);
471       if (!subQueryExpl.isMatch()) {
472         return subQueryExpl;
473       }
474       // match
475       Explanation[] valSrcExpls = new Explanation[valSrcScorers.length];
476       for(int i = 0; i < valSrcScorers.length; i++) {
477         valSrcExpls[i] = valSrcScorers[i].explain(doc);
478       }
479       Explanation customExp = customExplain(doc,subQueryExpl,valSrcExpls);
480       float sc = qWeight * customExp.getValue();
481       Explanation res = new ComplexExplanation(
482         true, sc, CustomScoreQuery.this.toString() + ", product of:");
483       res.addDetail(customExp);
484       res.addDetail(new Explanation(qWeight, "queryBoost")); // actually using the q boost as q weight (== weight value)
485       return res;
486     }
487   }
488 
489   public Weight createWeight(Searcher searcher) throws IOException {
490     return new CustomWeight(searcher);
491   }

498    * One particular case where this is useful if for testing this query.   
499    * <P>
500    * Note: only has effect when the ValueSource part is not null.
501    */
502   public boolean isStrict() {
503     return strict;
