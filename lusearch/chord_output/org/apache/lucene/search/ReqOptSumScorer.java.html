19  import java.io.IOException;
20  
21  /** A Scorer for queries with a required part and an optional part.
22   * Delays skipTo() on the optional part until a score() is needed.
23   * <br>
24   * This <code>Scorer</code> implements {@link Scorer#skipTo(int)}.
25   */
26  class ReqOptSumScorer extends Scorer {
27    /** The scorers passed from the constructor.
28     * These are set to null as soon as their next() or skipTo() returns false.
29     */
30    private Scorer reqScorer;
31    private Scorer optScorer;
32  
33    /** Construct a <code>ReqOptScorer</code>.
34     * @param reqScorer The required scorer. This must match.
35     * @param optScorer The optional scorer. This is used for scoring only.
36     */
37    public ReqOptSumScorer(
38        Scorer reqScorer,
39        Scorer optScorer)
40    {
41      super(null); // No similarity used.
42      this.reqScorer = reqScorer;
43      this.optScorer = optScorer;
44    }
45  
46    /** @deprecated use {@link #nextDoc()} instead. */
47    public boolean next() throws IOException {
48      return reqScorer.next();
49    }
50  
51    public int nextDoc() throws IOException {
52      return reqScorer.nextDoc();
53    }
54    
55    /** @deprecated use {@link #advance(int)} instead. */
56    public boolean skipTo(int target) throws IOException {
57      return reqScorer.skipTo(target);
58    }
59  
60    public int advance(int target) throws IOException {
61      return reqScorer.advance(target);
62    }
63    
64    /** @deprecated use {@link #docID()} instead. */
65    public int doc() {
66      return reqScorer.doc();
67    }
68  
69    public int docID() {
70      return reqScorer.docID();
71    }
72    
73    /** Returns the score of the current document matching the query.
74     * Initially invalid, until {@link #next()} is called the first time.
75     * @return The score of the required scorer, eventually increased by the score
76     * of the optional scorer when it also matches the current document.
77     */
78    public float score() throws IOException {
79      int curDoc = reqScorer.docID();
80      float reqScore = reqScorer.score();
81      if (optScorer == null) {
82        return reqScore;
83      }
84      
85      int optScorerDoc = optScorer.docID();
86      if (optScorerDoc < curDoc && (optScorerDoc = optScorer.advance(curDoc)) == NO_MORE_DOCS) {
87        optScorer = null;
88        return reqScore;
89      }
90      
91      return optScorerDoc == curDoc ? reqScore + optScorer.score() : reqScore;
92    }
93  
94    /** Explain the score of a document.
95     * TODO: Also show the total score.
96     * See BooleanScorer.explain() on how to do this.
97     */
98    public Explanation explain(int doc) throws IOException {
99      Explanation res = new Explanation();
100     res.setDescription("required, optional");
101     res.addDetail(reqScorer.explain(doc));
102     res.addDetail(optScorer.explain(doc));
103     return res;
104   }
105 }
