65      caches.put(Object.class, new AutoCache(this));

72    public CacheEntry[] getCacheEntries() {
73      List result = new ArrayList(17);
74      Iterator outerKeys = caches.keySet().iterator();
75      while (outerKeys.hasNext()) {
76        Class cacheType = (Class)outerKeys.next();
77        Cache cache = (Cache)caches.get(cacheType);
78        Iterator innerKeys = cache.readerCache.keySet().iterator();
79        while (innerKeys.hasNext()) {
80          // we've now materialized a hard ref
81          Object readerKey = innerKeys.next();
82          // innerKeys was backed by WeakHashMap, sanity check
83          // that it wasn't GCed before we made hard ref
84          if (null != readerKey && cache.readerCache.containsKey(readerKey)) {
85            Map innerCache = ((Map)cache.readerCache.get(readerKey));
86            Iterator entrySetIterator = innerCache.entrySet().iterator();
87            while (entrySetIterator.hasNext()) {
88              Map.Entry mapEntry = (Map.Entry) entrySetIterator.next();
89              Entry entry = (Entry) mapEntry.getKey();
90              result.add(new CacheEntryImpl(readerKey, entry.field,
91                                            cacheType, entry.type,
92                                            entry.custom, entry.locale,
93                                            mapEntry.getValue()));
94            }
95          }
96        }
97      }
98      return (CacheEntry[]) result.toArray(new CacheEntry[result.size()]);
99    }

170   abstract static class Cache {
171     Cache() {
172       this.wrapper = null;
173     }
174 
175     Cache(FieldCache wrapper) {
176       this.wrapper = wrapper;
177     }
178 
179     final FieldCache wrapper;
180 
181     final Map readerCache = new WeakHashMap();
182     
183     protected abstract Object createValue(IndexReader reader, Entry key)
184         throws IOException;
185 
186     public Object get(IndexReader reader, Entry key) throws IOException {
187       Map innerCache;
188       Object value;
189       final Object readerKey = reader.getFieldCacheKey();
190       synchronized (readerCache) {
191         innerCache = (Map) readerCache.get(readerKey);
192         if (innerCache == null) {
193           innerCache = new HashMap();
194           readerCache.put(readerKey, innerCache);
195           value = null;
196         } else {
197           value = innerCache.get(key);
198         }
199         if (value == null) {
200           value = new CreationPlaceholder();
201           innerCache.put(key, value);
202         }
203       }
204       if (value instanceof CreationPlaceholder) {
205         synchronized (value) {
206           CreationPlaceholder progress = (CreationPlaceholder) value;
207           if (progress.value == null) {
208             progress.value = createValue(reader, key);
209             synchronized (readerCache) {
210               innerCache.put(key, progress.value);
211             }
212 
213             // Only check if key.custom (the parser) is
214             // non-null; else, we check twice for a single
215             // call to FieldCache.getXXX
216             if (key.custom != null && wrapper != null) {
217               final PrintStream infoStream = wrapper.getInfoStream();
218               if (infoStream != null) {
219                 printNewInsanity(infoStream, progress.value);
220               }
221             }
222           }
223           return progress.value;
224         }
225       }
226       return value;
227     }
228 
229     private void printNewInsanity(PrintStream infoStream, Object value) {
230       final FieldCacheSanityChecker.Insanity[] insanities = FieldCacheSanityChecker.checkSanity(wrapper);
231       for(int i=0;i<insanities.length;i++) {
232         final FieldCacheSanityChecker.Insanity insanity = insanities[i];
233         final CacheEntry[] entries = insanity.getCacheEntries();
234         for(int j=0;j<entries.length;j++) {
235           if (entries[j].getValue() == value) {
236             // OK this insanity involves our entry
237             infoStream.println("WARNING: new FieldCache insanity created\nDetails: " + insanity.toString());
238             infoStream.println("\nStack:\n");
239             new Throwable().printStackTrace(infoStream);
240             break;
241           }
242         }
243       }
244     }
245   }

274     Entry (String field, Object custom) {

276       this.type = SortField.CUSTOM;
277       this.custom = custom;
278       this.locale = null;
279     }

281     /** 
282      * @deprecated Only (ab)used by FieldSortedHitQueue, 
283      *             remove when FieldSortedHitQueue is removed
284      */
285     Entry (String field, int type, Parser parser) {
286       this.field = StringHelper.intern(field);
287       this.type = type;
288       this.custom = parser;
289       this.locale = null;
290     }
291 
292     /** Two of these are equal iff they reference the same field and type. */
293     public boolean equals (Object o) {
294       if (o instanceof Entry) {
295         Entry other = (Entry) o;
296         if (other.field == field && other.type == type) {
297           if (other.locale == null ? locale == null : other.locale.equals(locale)) {
298             if (other.custom == null) {
299               if (custom == null) return true;
300             } else if (other.custom.equals (custom)) {
301               return true;
302             }
303           }
304         }
305       }
306       return false;
307     }
308 
309     /** Composes a hashcode based on the field and type. */
310     public int hashCode() {
311       return field.hashCode() ^ type ^ (custom==null ? 0 : custom.hashCode()) ^ (locale==null ? 0 : locale.hashCode());
312     }

326   static final class ByteCache extends Cache {
327     ByteCache(FieldCache wrapper) {
328       super(wrapper);
329     }
330     protected Object createValue(IndexReader reader, Entry entryKey)
331         throws IOException {
332       Entry entry = (Entry) entryKey;
333       String field = entry.field;
334       ByteParser parser = (ByteParser) entry.custom;
335       if (parser == null) {
336         return wrapper.getBytes(reader, field, FieldCache.DEFAULT_BYTE_PARSER);
337       }
338       final byte[] retArray = new byte[reader.maxDoc()];
339       TermDocs termDocs = reader.termDocs();
340       TermEnum termEnum = reader.terms (new Term (field));
341       try {
342         do {
343           Term term = termEnum.term();
344           if (term==null || term.field() != field) break;
345           byte termval = parser.parseByte(term.text());
346           termDocs.seek (termEnum);
347           while (termDocs.next()) {
348             retArray[termDocs.doc()] = termval;
349           }
350         } while (termEnum.next());
351       } catch (StopFillCacheException stop) {
352       } finally {
353         termDocs.close();
354         termEnum.close();
355       }
356       return retArray;
357     }
358   };

679   static final class StringIndexCache extends Cache {
680     StringIndexCache(FieldCache wrapper) {
681       super(wrapper);
682     }
683 
684     protected Object createValue(IndexReader reader, Entry entryKey)
685         throws IOException {
686       String field = StringHelper.intern((String) entryKey.field);
687       final int[] retArray = new int[reader.maxDoc()];
688       String[] mterms = new String[reader.maxDoc()+1];
689       TermDocs termDocs = reader.termDocs();
690       TermEnum termEnum = reader.terms (new Term (field));
691       int t = 0;  // current term number
692 
693       // an entry for documents that have no terms in this field
694       // should a document with no terms be at top or bottom?
695       // this puts them at the top - if it is changed, FieldDocSortedHitQueue
696       // needs to change as well.
697       mterms[t++] = null;
698 
699       try {
700         do {
701           Term term = termEnum.term();
702           if (term==null || term.field() != field) break;
703 
704           // store term text
705           // we expect that there is at most one term per document
706           if (t >= mterms.length) throw new RuntimeException ("there are more terms than " +
707                   "documents in field \"" + field + "\", but it's impossible to sort on " +
708                   "tokenized fields");
709           mterms[t] = term.text();
710 
711           termDocs.seek (termEnum);
712           while (termDocs.next()) {
713             retArray[termDocs.doc()] = t;
714           }
715 
716           t++;
717         } while (termEnum.next());
718       } finally {
719         termDocs.close();
720         termEnum.close();
721       }
722 
723       if (t == 0) {
724         // if there are no terms, make the term array
725         // have a single null entry
726         mterms = new String[1];
727       } else if (t < mterms.length) {
728         // if there are less terms than documents,
729         // trim off the dead array space
730         String[] terms = new String[t];
731         System.arraycopy (mterms, 0, terms, 0, t);
732         mterms = terms;
733       }
734 
735       StringIndex value = new StringIndex (retArray, mterms);
736       return value;
737     }
738   };

769       TermEnum enumerator = reader.terms (new Term (field));
770       try {

774         }
