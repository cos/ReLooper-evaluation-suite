31  /**
32   * An abstract {@link Query} that matches documents
33   * containing a subset of terms provided by a {@link
34   * FilteredTermEnum} enumeration.
35   *
36   * <p>This query cannot be used directly; you must subclass
37   * it and define {@link #getEnum} to provide a {@link
38   * FilteredTermEnum} that iterates through the terms to be
39   * matched.
40   *
41   * <p><b>NOTE</b>: if {@link #setRewriteMethod} is either
42   * {@link #CONSTANT_SCORE_BOOLEAN_QUERY_REWRITE} or {@link
43   * #SCORING_BOOLEAN_QUERY_REWRITE}, you may encounter a
44   * {@link BooleanQuery.TooManyClauses} exception during
45   * searching, which happens when the number of terms to be
46   * searched exceeds {@link
47   * BooleanQuery#getMaxClauseCount()}.  Setting {@link
48   * #setRewriteMethod} to {@link #CONSTANT_SCORE_FILTER_REWRITE}
49   * prevents this.
50   *
51   * <p>The recommended rewrite method is {@link
52   * #CONSTANT_SCORE_AUTO_REWRITE_DEFAULT}: it doesn't spend CPU
53   * computing unhelpful scores, and it tries to pick the most
54   * performant rewrite method given the query.
55   *
56   * Note that {@link QueryParser} produces
57   * MultiTermQueries using {@link
58   * #CONSTANT_SCORE_AUTO_REWRITE_DEFAULT} by default.
59   */
60  public abstract class MultiTermQuery extends Query {
61    /* @deprecated move to sub class */
62    protected Term term;
63    protected RewriteMethod rewriteMethod = CONSTANT_SCORE_AUTO_REWRITE_DEFAULT;
64    transient int numberOfTerms = 0;
65  
66    /** Abstract class that defines how the query is rewritten. */
67    public static abstract class RewriteMethod implements Serializable {
68      public abstract Query rewrite(IndexReader reader, MultiTermQuery query) throws IOException;
69    }

71    private static final class ConstantScoreFilterRewrite extends RewriteMethod implements Serializable {
72      public Query rewrite(IndexReader reader, MultiTermQuery query) {
73        Query result = new ConstantScoreQuery(new MultiTermQueryWrapperFilter(query));
74        result.setBoost(query.getBoost());
75        return result;
76      }
77  
78      // Make sure we are still a singleton even after deserializing
79      protected Object readResolve() {
80        return CONSTANT_SCORE_FILTER_REWRITE;
81      }
82    }

84    /** A rewrite method that first creates a private Filter,
85     *  by visiting each term in sequence and marking all docs
86     *  for that term.  Matching documents are assigned a
87     *  constant score equal to the query's boost.
88     * 
89     *  <p> This method is faster than the BooleanQuery
90     *  rewrite methods when the number of matched terms or
91     *  matched documents is non-trivial. Also, it will never
92     *  hit an errant {@link BooleanQuery.TooManyClauses}
93     *  exception.
94     *
95     *  @see #setRewriteMethod */
96    public final static RewriteMethod CONSTANT_SCORE_FILTER_REWRITE = new ConstantScoreFilterRewrite();
97  
98    private static class ScoringBooleanQueryRewrite extends RewriteMethod implements Serializable {
99      public Query rewrite(IndexReader reader, MultiTermQuery query) throws IOException {
100 
101       FilteredTermEnum enumerator = query.getEnum(reader);
102       BooleanQuery result = new BooleanQuery(true);
103       int count = 0;
104       try {
105         do {
106           Term t = enumerator.term();
107           if (t != null) {
108             TermQuery tq = new TermQuery(t); // found a match
109             tq.setBoost(query.getBoost() * enumerator.difference()); // set the boost
110             result.add(tq, BooleanClause.Occur.SHOULD); // add to query
111             count++;
112           }
113         } while (enumerator.next());    
114       } finally {
115         enumerator.close();
116       }
117       query.incTotalNumberOfTerms(count);
118       return result;
119     }
120 
121     // Make sure we are still a singleton even after deserializing
122     protected Object readResolve() {
123       return SCORING_BOOLEAN_QUERY_REWRITE;
124     }
125   }
126 
127   /** A rewrite method that first translates each term into
128    *  {@link BooleanClause.Occur#SHOULD} clause in a
129    *  BooleanQuery, and keeps the scores as computed by the
130    *  query.  Note that typically such scores are
131    *  meaningless to the user, and require non-trivial CPU
132    *  to compute, so it's almost always better to use {@link
133    *  #CONSTANT_SCORE_AUTO_REWRITE_DEFAULT} instead.
134    *
135    *  <p><b>NOTE</b>: This rewrite method will hit {@link
136    *  BooleanQuery.TooManyClauses} if the number of terms
137    *  exceeds {@link BooleanQuery#getMaxClauseCount}.
138    *
139    *  @see #setRewriteMethod */
140   public final static RewriteMethod SCORING_BOOLEAN_QUERY_REWRITE = new ScoringBooleanQueryRewrite();
141 
142   private static class ConstantScoreBooleanQueryRewrite extends ScoringBooleanQueryRewrite implements Serializable {
143     public Query rewrite(IndexReader reader, MultiTermQuery query) throws IOException {
144       // strip the scores off
145       Query result = new ConstantScoreQuery(new QueryWrapperFilter(super.rewrite(reader, query)));
146       result.setBoost(query.getBoost());
147       return result;
148     }
149 
150     // Make sure we are still a singleton even after deserializing
151     protected Object readResolve() {
152       return CONSTANT_SCORE_BOOLEAN_QUERY_REWRITE;
153     }
154   }

156   /** Like {@link #SCORING_BOOLEAN_QUERY_REWRITE} except
157    *  scores are not computed.  Instead, each matching
158    *  document receives a constant score equal to the
159    *  query's boost.
160    * 
161    *  <p><b>NOTE</b>: This rewrite method will hit {@link
162    *  BooleanQuery.TooManyClauses} if the number of terms
163    *  exceeds {@link BooleanQuery#getMaxClauseCount}.
164    *
165    *  @see #setRewriteMethod */
166   public final static RewriteMethod CONSTANT_SCORE_BOOLEAN_QUERY_REWRITE = new ConstantScoreBooleanQueryRewrite();

169   /** A rewrite method that tries to pick the best
170    *  constant-score rewrite method based on term and
171    *  document counts from the query.  If both the number of
172    *  terms and documents is small enough, then {@link
173    *  #CONSTANT_SCORE_BOOLEAN_QUERY_REWRITE} is used.
174    *  Otherwise, {@link #CONSTANT_SCORE_FILTER_REWRITE} is
175    *  used.
176    */
177   public static class ConstantScoreAutoRewrite extends RewriteMethod implements Serializable {
178 
179     // Defaults derived from rough tests with a 20.0 million
180     // doc Wikipedia index.  With more than 350 terms in the
181     // query, the filter method is fastest:
182     public static int DEFAULT_TERM_COUNT_CUTOFF = 350;
183 
184     // If the query will hit more than 1 in 1000 of the docs
185     // in the index (0.1%), the filter method is fastest:
186     public static double DEFAULT_DOC_COUNT_PERCENT = 0.1;
187 
188     private int termCountCutoff = DEFAULT_TERM_COUNT_CUTOFF;
189     private double docCountPercent = DEFAULT_DOC_COUNT_PERCENT;
190 
191     /** If the number of terms in this query is equal to or
192      *  larger than this setting then {@link
193      *  #CONSTANT_SCORE_FILTER_REWRITE} is used. */
194     public void setTermCountCutoff(int count) {
195       termCountCutoff = count;
196     }
197 
198     /** @see #setTermCountCutoff */
199     public int getTermCountCutoff() {
200       return termCountCutoff;
201     }
202 
203     /** If the number of documents to be visited in the
204      *  postings exceeds this specified percentage of the
205      *  maxDoc() for the index, then {@link
206      *  #CONSTANT_SCORE_FILTER_REWRITE} is used.
207      *  @param percent 0.0 to 100.0 */
208     public void setDocCountPercent(double percent) {
209       docCountPercent = percent;
210     }
211 
212     /** @see #setDocCountPercent */
213     public double getDocCountPercent() {
214       return docCountPercent;
215     }
216 
217     public Query rewrite(IndexReader reader, MultiTermQuery query) throws IOException {
218       // Get the enum and start visiting terms.  If we
219       // exhaust the enum before hitting either of the
220       // cutoffs, we use ConstantBooleanQueryRewrite; else,
221       // ConstantFilterRewrite:
222       final Collection pendingTerms = new ArrayList();
223       final int docCountCutoff = (int) ((docCountPercent / 100.) * reader.maxDoc());
224       final int termCountLimit = Math.min(BooleanQuery.getMaxClauseCount(), termCountCutoff);
225       int docVisitCount = 0;
226 
227       FilteredTermEnum enumerator = query.getEnum(reader);
228       try {
229         while(true) {
230           Term t = enumerator.term();
231           if (t != null) {
232             pendingTerms.add(t);
233             // Loading the TermInfo from the terms dict here
234             // should not be costly, because 1) the
235             // query/filter will load the TermInfo when it
236             // runs, and 2) the terms dict has a cache:
237             docVisitCount += reader.docFreq(t);
238           }
239 
240           if (pendingTerms.size() >= termCountLimit || docVisitCount >= docCountCutoff) {
241             // Too many terms -- make a filter.
242             Query result = new ConstantScoreQuery(new MultiTermQueryWrapperFilter(query));
243             result.setBoost(query.getBoost());
244             return result;
245           } else  if (!enumerator.next()) {
246             // Enumeration is done, and we hit a small
247             // enough number of terms & docs -- just make a
248             // BooleanQuery, now
249             Iterator it = pendingTerms.iterator();
250             BooleanQuery bq = new BooleanQuery(true);
251             while(it.hasNext()) {
252               TermQuery tq = new TermQuery((Term) it.next());
253               bq.add(tq, BooleanClause.Occur.SHOULD);
254             }
255             // Strip scores
256             Query result = new ConstantScoreQuery(new QueryWrapperFilter(bq));
257             result.setBoost(query.getBoost());
258             query.incTotalNumberOfTerms(pendingTerms.size());
259             return result;
260           }
261         }
262       } finally {
263         enumerator.close();
264       }
265     }
266     
267     public int hashCode() {
268       final int prime = 1279;
269       return (int) (prime * termCountCutoff + Double.doubleToLongBits(docCountPercent));
270     }
271 
272     public boolean equals(Object obj) {
273       if (this == obj)
274         return true;
275       if (obj == null)
276         return false;
277       if (getClass() != obj.getClass())
278         return false;
279 
280       ConstantScoreAutoRewrite other = (ConstantScoreAutoRewrite) obj;
281       if (other.termCountCutoff != termCountCutoff) {
282         return false;
283       }
284 
285       if (Double.doubleToLongBits(other.docCountPercent) != Double.doubleToLongBits(docCountPercent)) {
286         return false;
287       }
288       
289       return true;
290     }
291   }

349   /**
350    * Expert: Return the number of unique terms visited during execution of the query.
351    * If there are many of them, you may consider using another query type
352    * or optimize your total term count in index.
353    * <p>This method is not thread safe, be sure to only call it when no query is running!
354    * If you re-use the same query instance for another
355    * search, be sure to first reset the term counter
356    * with {@link #clearTotalNumberOfTerms}.
357    * <p>On optimized indexes / no MultiReaders, you get the correct number of
358    * unique terms for the whole index. Use this number to compare different queries.
359    * For non-optimized indexes this number can also be achieved in
360    * non-constant-score mode. In constant-score mode you get the total number of
361    * terms seeked for all segments / sub-readers.
362    * @see #clearTotalNumberOfTerms
363    */
364   public int getTotalNumberOfTerms() {
365     return numberOfTerms;
366   }
367   
368   /**
369    * Expert: Resets the counting of unique terms.
370    * Do this before executing the query/filter.
371    * @see #getTotalNumberOfTerms
372    */
373   public void clearTotalNumberOfTerms() {
374     numberOfTerms = 0;
375   }

377   protected void incTotalNumberOfTerms(int inc) {
378     numberOfTerms += inc;

381   public Query rewrite(IndexReader reader) throws IOException {
382     return rewriteMethod.rewrite(reader, this);

386   /* Prints a user-readable version of this query.
387    * Implemented for back compat in case MultiTermQuery
388    * subclasses do no implement.
389    */
390   public String toString(String field) {
391     StringBuffer buffer = new StringBuffer();
392     if (term != null) {
393       if (!term.field().equals(field)) {
394         buffer.append(term.field());
395         buffer.append(":");
396       }
397       buffer.append(term.text());
398     } else {
399       buffer.append("termPattern:unknown");
400     }
401     buffer.append(ToStringUtils.boost(getBoost()));
402     return buffer.toString();
403   }
