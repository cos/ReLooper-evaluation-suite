20  import java.io.IOException;
21  
22  import org.apache.lucene.index.IndexReader;
23  
24  /**
25   * A {@link Collector} implementation that collects the top-scoring hits,
26   * returning them as a {@link TopDocs}. This is used by {@link IndexSearcher} to
27   * implement {@link TopDocs}-based search. Hits are sorted by score descending
28   * and then (when the scores are tied) docID ascending. When you create an
29   * instance of this collector you should know in advance whether documents are
30   * going to be collected in doc Id order or not.
31   *
32   * <p><b>NOTE</b>: The values {@link Float#NaN} and
33   * {Float#NEGATIVE_INFINITY} are not valid scores.  This
34   * collector will not properly collect hits with such
35   * scores.
36   */
37  public abstract class TopScoreDocCollector extends TopDocsCollector {
38  
39    // Assumes docs are scored in order.
40    private static class InOrderTopScoreDocCollector extends TopScoreDocCollector {
41      private InOrderTopScoreDocCollector(int numHits) {
42        super(numHits);
43      }
44      
45      public void collect(int doc) throws IOException {
46        float score = scorer.score();
47  
48        // This collector cannot handle these scores:
49        assert score != Float.NEGATIVE_INFINITY;
50        assert !Float.isNaN(score);
51  
52        totalHits++;
53        if (score <= pqTop.score) {
54          // Since docs are returned in-order (i.e., increasing doc Id), a document
55          // with equal score to pqTop.score cannot compete since HitQueue favors
56          // documents with lower doc Ids. Therefore reject those docs too.
57          return;
58        }
59        pqTop.doc = doc + docBase;
60        pqTop.score = score;
61        pqTop = (ScoreDoc) pq.updateTop();
62      }
63      
64      public boolean acceptsDocsOutOfOrder() {
65        return false;
66      }
67    }
68  
69    // Assumes docs are scored out of order.
70    private static class OutOfOrderTopScoreDocCollector extends TopScoreDocCollector {
71      private OutOfOrderTopScoreDocCollector(int numHits) {
72        super(numHits);
73      }
74      
75      public void collect(int doc) throws IOException {
76        float score = scorer.score();
77  
78        // This collector cannot handle NaN
79        assert !Float.isNaN(score);
80  
81        totalHits++;
82        doc += docBase;
83        if (score < pqTop.score || (score == pqTop.score && doc > pqTop.doc)) {
84          return;
85        }
86        pqTop.doc = doc;
87        pqTop.score = score;
88        pqTop = (ScoreDoc) pq.updateTop();
89      }
90      
91      public boolean acceptsDocsOutOfOrder() {
92        return true;
93      }
94    }
95  
96    /**
97     * Creates a new {@link TopScoreDocCollector} given the number of hits to
98     * collect and whether documents are scored in order by the input
99     * {@link Scorer} to {@link #setScorer(Scorer)}.
100    *
101    * <p><b>NOTE</b>: The instances returned by this method
102    * pre-allocate a full array of length
103    * <code>numHits</code>, and fill the array with sentinel
104    * objects.
105    */
106   public static TopScoreDocCollector create(int numHits, boolean docsScoredInOrder) {
107     
108     if (docsScoredInOrder) {
109       return new InOrderTopScoreDocCollector(numHits);
110     } else {
111       return new OutOfOrderTopScoreDocCollector(numHits);
112     }
113     
114   }
115   
116   ScoreDoc pqTop;
117   int docBase = 0;
118   Scorer scorer;
119     
120   // prevents instantiation
121   private TopScoreDocCollector(int numHits) {
122     super(new HitQueue(numHits, true));
123     // HitQueue implements getSentinelObject to return a ScoreDoc, so we know
124     // that at this point top() is already initialized.
125     pqTop = (ScoreDoc) pq.top();
126   }
127 
128   protected TopDocs newTopDocs(ScoreDoc[] results, int start) {
129     if (results == null) {
130       return EMPTY_TOPDOCS;
131     }
132     
133     // We need to compute maxScore in order to set it in TopDocs. If start == 0,
134     // it means the largest element is already in results, use its score as
135     // maxScore. Otherwise pop everything else, until the largest element is
136     // extracted and use its score as maxScore.
137     float maxScore = Float.NaN;
138     if (start == 0) {
139       maxScore = results[0].score;
140     } else {
141       for (int i = pq.size(); i > 1; i--) { pq.pop(); }
142       maxScore = ((ScoreDoc) pq.pop()).score;
143     }
144     
145     return new TopDocs(totalHits, results, maxScore);
146   }
147   
148   public void setNextReader(IndexReader reader, int base) {
149     docBase = base;
150   }
151   
152   public void setScorer(Scorer scorer) throws IOException {
153     this.scorer = scorer;
154   }
155 }
