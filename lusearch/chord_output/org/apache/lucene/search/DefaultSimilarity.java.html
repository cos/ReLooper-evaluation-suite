22  /** Expert: Default scoring implementation. */
23  public class DefaultSimilarity extends Similarity {
24  
25    /** Implemented as
26     *  <code>state.getBoost()*lengthNorm(numTerms)</code>, where
27     *  <code>numTerms</code> is {@link FieldInvertState#getLength()} if {@link
28     *  #setDiscountOverlaps} is false, else it's {@link
29     *  FieldInvertState#getLength()} - {@link
30     *  FieldInvertState#getNumOverlap()}.
31     *
32     *  <p><b>WARNING</b>: This API is new and experimental, and may suddenly
33     *  change.</p> */
34    public float computeNorm(String field, FieldInvertState state) {
35      final int numTerms;
36      if (discountOverlaps)
37        numTerms = state.getLength() - state.getNumOverlap();
38      else
39        numTerms = state.getLength();
40      return (float) (state.getBoost() * lengthNorm(field, numTerms));
41    }
42    
43    /** Implemented as <code>1/sqrt(numTerms)</code>. */
44    public float lengthNorm(String fieldName, int numTerms) {
45      return (float)(1.0 / Math.sqrt(numTerms));
46    }
47    
48    /** Implemented as <code>1/sqrt(sumOfSquaredWeights)</code>. */
49    public float queryNorm(float sumOfSquaredWeights) {
50      return (float)(1.0 / Math.sqrt(sumOfSquaredWeights));
51    }
52  
53    /** Implemented as <code>sqrt(freq)</code>. */
54    public float tf(float freq) {
55      return (float)Math.sqrt(freq);
56    }
57      
58    /** Implemented as <code>1 / (distance + 1)</code>. */
59    public float sloppyFreq(int distance) {
60      return 1.0f / (distance + 1);
61    }
62      
63    /** Implemented as <code>log(numDocs/(docFreq+1)) + 1</code>. */
64    public float idf(int docFreq, int numDocs) {
65      return (float)(Math.log(numDocs/(double)(docFreq+1)) + 1.0);
66    }
67      
68    /** Implemented as <code>overlap / maxOverlap</code>. */
69    public float coord(int overlap, int maxOverlap) {
70      return overlap / (float)maxOverlap;
71    }
72  
73    // Default false
74    protected boolean discountOverlaps;
75  
76    /** Determines whether overlap tokens (Tokens with
77     *  0 position increment) are ignored when computing
78     *  norm.  By default this is false, meaning overlap
79     *  tokens are counted just like non-overlap tokens.
80     *
81     *  <p><b>WARNING</b>: This API is new and experimental, and may suddenly
82     *  change.</p>
83     *
84     *  @see #computeNorm
85     */
86    public void setDiscountOverlaps(boolean v) {
87      discountOverlaps = v;
88    }
89  
90    /** @see #setDiscountOverlaps */
91    public boolean getDiscountOverlaps() {
92      return discountOverlaps;
93    }
94  }
