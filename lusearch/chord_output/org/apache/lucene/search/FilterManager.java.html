20  import java.util.Comparator;
21  import java.util.Date;
22  import java.util.HashMap;
23  import java.util.Iterator;
24  import java.util.Map;
25  import java.util.TreeSet;
26  
27  /**
28   * Filter caching singleton.  It can be used 
29   * to save filters locally for reuse.
30   * This class makes it possible to cache Filters even when using RMI, as it
31   * keeps the cache on the searcher side of the RMI connection.
32   * 
33   * Also could be used as a persistent storage for any filter as long as the
34   * filter provides a proper hashCode(), as that is used as the key in the cache.
35   * 
36   * The cache is periodically cleaned up from a separate thread to ensure the
37   * cache doesn't exceed the maximum size.
38   */
39  public class FilterManager {
40  
41    protected static FilterManager manager;
42    
43    /** The default maximum number of Filters in the cache */
44    protected static final int  DEFAULT_CACHE_CLEAN_SIZE = 100;
45    /** The default frequency of cache cleanup */
46    protected static final long DEFAULT_CACHE_SLEEP_TIME = 1000 * 60 * 10;
47  
48    /** The cache itself */
49    protected Map           cache;
50    /** Maximum allowed cache size */
51    protected int           cacheCleanSize;
52    /** Cache cleaning frequency */
53    protected long          cleanSleepTime;
54    /** Cache cleaner that runs in a separate thread */
55    protected FilterCleaner filterCleaner;
56  
57    public synchronized static FilterManager getInstance() {
58      if (manager == null) {
59        manager = new FilterManager();
60      }
61      return manager;
62    }
63  
64    /**
65     * Sets up the FilterManager singleton.
66     */
67    protected FilterManager() {
68      cache            = new HashMap();
69      cacheCleanSize   = DEFAULT_CACHE_CLEAN_SIZE; // Let the cache get to 100 items
70      cleanSleepTime   = DEFAULT_CACHE_SLEEP_TIME; // 10 minutes between cleanings
71  
72      filterCleaner   = new FilterCleaner();
73      Thread fcThread = new Thread(filterCleaner);
74      // set to be a Daemon so it doesn't have to be stopped
75      fcThread.setDaemon(true);
76      fcThread.start();
77    }
78    
79    /**
80     * Sets the max size that cache should reach before it is cleaned up
81     * @param cacheCleanSize maximum allowed cache size
82     */
83    public void setCacheSize(int cacheCleanSize) {
84      this.cacheCleanSize = cacheCleanSize;
85    }
86  
87    /**
88     * Sets the cache cleaning frequency in milliseconds.
89     * @param cleanSleepTime cleaning frequency in milliseconds
90     */
91    public void setCleanThreadSleepTime(long cleanSleepTime) {
92      this.cleanSleepTime  = cleanSleepTime;
93    }
94  
95    /**
96     * Returns the cached version of the filter.  Allows the caller to pass up
97     * a small filter but this will keep a persistent version around and allow
98     * the caching filter to do its job.
99     * 
100    * @param filter The input filter
101    * @return The cached version of the filter
102    */
103   public Filter getFilter(Filter filter) {
104     synchronized(cache) {
105       FilterItem fi = null;
106       fi = (FilterItem)cache.get(new Integer(filter.hashCode()));
107       if (fi != null) {
108         fi.timestamp = new Date().getTime();
109         return fi.filter;
110       }
111       cache.put(new Integer(filter.hashCode()), new FilterItem(filter));
112       return filter;
113     }
114   }
115 
116   /**
117    * Holds the filter and the last time the filter was used, to make LRU-based
118    * cache cleaning possible.
119    * TODO: Clean this up when we switch to Java 1.5
120    */
121   protected class FilterItem {
122     public Filter filter;
123     public long   timestamp;
124 
125     public FilterItem (Filter filter) {        
126       this.filter = filter;
127       this.timestamp = new Date().getTime();
128     }
129   }
130 
131 
132   /**
133    * Keeps the cache from getting too big.
134    * If we were using Java 1.5, we could use LinkedHashMap and we would not need this thread
135    * to clean out the cache.
136    * 
137    * The SortedSet sortedFilterItems is used only to sort the items from the cache,
138    * so when it's time to clean up we have the TreeSet sort the FilterItems by
139    * timestamp.
140    * 
141    * Removes 1.5 * the numbers of items to make the cache smaller.
142    * For example:
143    * If cache clean size is 10, and the cache is at 15, we would remove (15 - 10) * 1.5 = 7.5 round up to 8.
144    * This way we clean the cache a bit more, and avoid having the cache cleaner having to do it frequently.
145    */
146   protected class FilterCleaner implements Runnable  {
147 
148     private boolean running = true;
149     private TreeSet sortedFilterItems;
150 
151     public FilterCleaner() {
152       sortedFilterItems = new TreeSet(new Comparator() {
153         public int compare(Object a, Object b) {
154           if( a instanceof Map.Entry && b instanceof Map.Entry) {
155             FilterItem fia = (FilterItem) ((Map.Entry)a).getValue();
156             FilterItem fib = (FilterItem) ((Map.Entry)b).getValue();
157             if ( fia.timestamp == fib.timestamp ) {
158               return 0;
159             }
160             // smaller timestamp first
161             if ( fia.timestamp < fib.timestamp ) {
162               return -1;
163             }
164             // larger timestamp last
165             return 1;
166           } else {
167             throw new ClassCastException("Objects are not Map.Entry");
168           }
169         }
170       });
171     }
172 
173     public void run () {
174       while (running) {
175 
176         // sort items from oldest to newest 
177         // we delete the oldest filters 
178         if (cache.size() > cacheCleanSize) {
179           // empty the temporary set
180           sortedFilterItems.clear();
181           synchronized (cache) {
182             sortedFilterItems.addAll(cache.entrySet());
183             Iterator it = sortedFilterItems.iterator();
184             int numToDelete = (int) ((cache.size() - cacheCleanSize) * 1.5);
185             int counter = 0;
186             // loop over the set and delete all of the cache entries not used in a while
187             while (it.hasNext() && counter++ < numToDelete) {
188               Map.Entry entry = (Map.Entry)it.next();
189               cache.remove(entry.getKey());
190             }
191           }
192           // empty the set so we don't tie up the memory
193           sortedFilterItems.clear();
194         }
195         // take a nap
196         try {
197           Thread.sleep(cleanSleepTime);
198         } catch (InterruptedException ie) {
199           Thread.currentThread().interrupt();
200           throw new RuntimeException(ie);
201         }
202       }
203     }
204   }
205 }
