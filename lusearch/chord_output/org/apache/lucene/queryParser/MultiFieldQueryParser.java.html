42    /**
43     * Creates a MultiFieldQueryParser. 
44     * Allows passing of a map with term to Boost, and the boost to apply to each term.
45     *
46     * <p>It will, when parse(String query)
47     * is called, construct a query like this (assuming the query consists of
48     * two terms and you specify the two fields <code>title</code> and <code>body</code>):</p>
49     * 
50     * <code>
51     * (title:term1 body:term1) (title:term2 body:term2)
52     * </code>
53     *
54     * <p>When setDefaultOperator(AND_OPERATOR) is set, the result will be:</p>
55     *  
56     * <code>
57     * +(title:term1 body:term1) +(title:term2 body:term2)
58     * </code>
59     * 
60     * <p>When you pass a boost (title=>5 body=>10) you can get </p>
61     * 
62     * <code>
63     * +(title:term1^5.0 body:term1^10.0) +(title:term2^5.0 body:term2^10.0)
64     * </code>
65     *
66     * <p>In other words, all the query's terms must appear, but it doesn't matter in
67     * what fields they appear.</p>
68     *
69     * @deprecated Please use {@link #MultiFieldQueryParser(Version, String[], Analyzer, Map)} instead
70     */
71    public MultiFieldQueryParser(String[] fields, Analyzer analyzer, Map boosts) {
72      this(Version.LUCENE_24, fields, analyzer);
73      this.boosts = boosts;
74    }

159   protected Query getFieldQuery(String field, String queryText, int slop) throws ParseException {
160     if (field == null) {
161       List clauses = new ArrayList();
162       for (int i = 0; i < fields.length; i++) {
163         Query q = super.getFieldQuery(fields[i], queryText);
164         if (q != null) {
165           //If the user passes a map of boosts
166           if (boosts != null) {
167             //Get the boost from the map and apply them
168             Float boost = (Float)boosts.get(fields[i]);
169             if (boost != null) {
170               q.setBoost(boost.floatValue());
171             }
172           }
173           applySlop(q,slop);
174           clauses.add(new BooleanClause(q, BooleanClause.Occur.SHOULD));
175         }
176       }
177       if (clauses.size() == 0)  // happens for stopwords
178         return null;
179       return getBooleanQuery(clauses, true);
180     }
181     Query q = super.getFieldQuery(field, queryText);
182     applySlop(q,slop);
183     return q;
184   }
185 
186   private void applySlop(Query q, int slop) {
187     if (q instanceof PhraseQuery) {
188       ((PhraseQuery) q).setSlop(slop);
189     } else if (q instanceof MultiPhraseQuery) {
190       ((MultiPhraseQuery) q).setSlop(slop);
191     }
192   }

200   protected Query getFuzzyQuery(String field, String termStr, float minSimilarity) throws ParseException
201   {
202     if (field == null) {
203       List clauses = new ArrayList();
204       for (int i = 0; i < fields.length; i++) {
205         clauses.add(new BooleanClause(getFuzzyQuery(fields[i], termStr, minSimilarity),
206             BooleanClause.Occur.SHOULD));
207       }
208       return getBooleanQuery(clauses, true);
209     }
210     return super.getFuzzyQuery(field, termStr, minSimilarity);
211   }

274   /**
275    * Parses a query which searches on the fields specified.
276    * <p>
277    * If x fields are specified, this effectively constructs:
278    * <pre>
279    * <code>
280    * (field1:query1) (field2:query2) (field3:query3)...(fieldx:queryx)
281    * </code>
282    * </pre>
283    * @param matchVersion Lucene version to match; this is passed through to QueryParser.
284    * @param queries Queries strings to parse
285    * @param fields Fields to search on
286    * @param analyzer Analyzer to use
287    * @throws ParseException if query parsing fails
288    * @throws IllegalArgumentException if the length of the queries array differs
289    *  from the length of the fields array
290    */
291   public static Query parse(Version matchVersion, String[] queries, String[] fields,
292       Analyzer analyzer) throws ParseException
293   {
294     if (queries.length != fields.length)
295       throw new IllegalArgumentException("queries.length != fields.length");
296     BooleanQuery bQuery = new BooleanQuery();
297     for (int i = 0; i < fields.length; i++)
298     {
299       QueryParser qp = new QueryParser(matchVersion, fields[i], analyzer);
300       Query q = qp.parse(queries[i]);
301       if (q!=null && // q never null, just being defensive
302           (!(q instanceof BooleanQuery) || ((BooleanQuery)q).getClauses().length>0)) {
303         bQuery.add(q, BooleanClause.Occur.SHOULD);
304       }
305     }
306     return bQuery;
307   }

376   public static Query parse(Version matchVersion, String query, String[] fields,
377       BooleanClause.Occur[] flags, Analyzer analyzer) throws ParseException {
378     if (fields.length != flags.length)
379       throw new IllegalArgumentException("fields.length != flags.length");
380     BooleanQuery bQuery = new BooleanQuery();
381     for (int i = 0; i < fields.length; i++) {
382       QueryParser qp = new QueryParser(matchVersion, fields[i], analyzer);
383       Query q = qp.parse(query);
384       if (q!=null && // q never null, just being defensive 
385           (!(q instanceof BooleanQuery) || ((BooleanQuery)q).getClauses().length>0)) {
386         bQuery.add(q, flags[i]);
387       }
388     }
389     return bQuery;
390   }

469     {

472       if (q!=null && // q never null, just being defensive
473           (!(q instanceof BooleanQuery) || ((BooleanQuery)q).getClauses().length>0)) {
474         bQuery.add(q, flags[i]);
475       }
476     }
