507   protected void addClause(List clauses, int conj, int mods, Query q) {
508     boolean required, prohibited;
509 
510     // If this term is introduced by AND, make the preceding term required,
511     // unless it's already prohibited
512     if (clauses.size() > 0 && conj == CONJ_AND) {
513       BooleanClause c = (BooleanClause) clauses.get(clauses.size()-1);
514       if (!c.isProhibited())
515         c.setOccur(BooleanClause.Occur.MUST);
516     }
517 
518     if (clauses.size() > 0 && operator == AND_OPERATOR && conj == CONJ_OR) {
519       // If this term is introduced by OR, make the preceding term optional,
520       // unless it's prohibited (that means we leave -a OR b but +a OR b-->a OR b)
521       // notice if the input is a OR b, first term is parsed as required; without
522       // this modification a OR b would parsed as +a OR b
523       BooleanClause c = (BooleanClause) clauses.get(clauses.size()-1);
524       if (!c.isProhibited())
525         c.setOccur(BooleanClause.Occur.SHOULD);
526     }
527 
528     // We might have been passed a null query; the term might have been
529     // filtered away by the analyzer.
530     if (q == null)
531       return;
532 
533     if (operator == OR_OPERATOR) {
534       // We set REQUIRED if we're introduced by AND or +; PROHIBITED if
535       // introduced by NOT or -; make sure not to set both.
536       prohibited = (mods == MOD_NOT);
537       required = (mods == MOD_REQ);
538       if (conj == CONJ_AND && !prohibited) {
539         required = true;
540       }
541     } else {
542       // We set PROHIBITED if we're introduced by NOT or -; We set REQUIRED
543       // if not PROHIBITED and not introduced by OR
544       prohibited = (mods == MOD_NOT);
545       required   = (!prohibited && conj != CONJ_OR);
546     }
547     if (required && !prohibited)
548       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST));
549     else if (!required && !prohibited)
550       clauses.add(newBooleanClause(q, BooleanClause.Occur.SHOULD));
551     else if (!required && prohibited)
552       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST_NOT));
553     else
554       throw new RuntimeException("Clause cannot be both required and prohibited");
555   }

1072  /**
1073   * Returns a String where the escape char has been
1074   * removed, or kept only once if there was a double escape.
1075   * 
1076   * Supports escaped unicode characters, e. g. translates
1077   * <code>\\u0041</code> to <code>A</code>.
1078   * 
1079   */
1080  private String discardEscapeChar(String input) throws ParseException {
1081    // Create char array to hold unescaped char sequence
1082    char[] output = new char[input.length()];
1083
1084    // The length of the output can be less than the input
1085    // due to discarded escape chars. This variable holds
1086    // the actual length of the output
1087    int length = 0;
1088
1089    // We remember whether the last processed character was 
1090    // an escape character
1091    boolean lastCharWasEscapeChar = false;
1092
1093    // The multiplier the current unicode digit must be multiplied with.
1094    // E. g. the first digit must be multiplied with 16^3, the second with 16^2...
1095    int codePointMultiplier = 0;
1096
1097    // Used to calculate the codepoint of the escaped unicode character
1098    int codePoint = 0;
1099
1100    for (int i = 0; i < input.length(); i++) {
1101      char curChar = input.charAt(i);
1102      if (codePointMultiplier > 0) {
1103        codePoint += hexToInt(curChar) * codePointMultiplier;
1104        codePointMultiplier >>>= 4;
1105        if (codePointMultiplier == 0) {
1106          output[length++] = (char)codePoint;
1107          codePoint = 0;
1108        }
1109      } else if (lastCharWasEscapeChar) {
1110        if (curChar == 'u') {
1111          // found an escaped unicode character
1112          codePointMultiplier = 16 * 16 * 16;
1113        } else {
1114          // this character was escaped
1115          output[length] = curChar;
1116          length++;
1117        }
1118        lastCharWasEscapeChar = false;
1119      } else {
1120        if (curChar == '\\') {
1121          lastCharWasEscapeChar = true;
1122        } else {
1123          output[length] = curChar;
1124          length++;
1125        }
1126      }
1127    }
1128
1129    if (codePointMultiplier > 0) {
1130      throw new ParseException("Truncated unicode escape sequence.");
1131    }
1132
1133    if (lastCharWasEscapeChar) {
1134      throw new ParseException("Term can not end with escape character.");
1135    }
1136
1137    return new String(output, 0, length);
1138  }

1141  private static final int hexToInt(char c) throws ParseException {
1142    if ('0' <= c && c <= '9') {
1143      return c - '0';
1144    } else if ('a' <= c && c <= 'f'){
1145      return c - 'a' + 10;
1146    } else if ('A' <= c && c <= 'F') {
1147      return c - 'A' + 10;
1148    } else {
1149      throw new ParseException("None-hex character in unicode escape sequence: " + c);
1150    }
1151  }
