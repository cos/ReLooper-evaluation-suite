20  import org.apache.lucene.store.Directory;
21  import org.apache.lucene.store.IndexOutput;
22  import org.apache.lucene.util.StringHelper;
23  import org.apache.lucene.util.UnicodeUtil;
24  
25  import java.io.IOException;
26  
27  final class TermVectorsWriter {
28    
29    private IndexOutput tvx = null, tvd = null, tvf = null;
30    private FieldInfos fieldInfos;
31    final UnicodeUtil.UTF8Result[] utf8Results = new UnicodeUtil.UTF8Result[] {new UnicodeUtil.UTF8Result(),
32                                                                               new UnicodeUtil.UTF8Result()};
33  
34    public TermVectorsWriter(Directory directory, String segment,
35                             FieldInfos fieldInfos)
36      throws IOException {
37      // Open files for TermVector storage
38      tvx = directory.createOutput(segment + "." + IndexFileNames.VECTORS_INDEX_EXTENSION);
39      tvx.writeInt(TermVectorsReader.FORMAT_CURRENT);
40      tvd = directory.createOutput(segment + "." + IndexFileNames.VECTORS_DOCUMENTS_EXTENSION);
41      tvd.writeInt(TermVectorsReader.FORMAT_CURRENT);
42      tvf = directory.createOutput(segment + "." + IndexFileNames.VECTORS_FIELDS_EXTENSION);
43      tvf.writeInt(TermVectorsReader.FORMAT_CURRENT);
44  
45      this.fieldInfos = fieldInfos;
46    }
47  
48    /**
49     * Add a complete document specified by all its term vectors. If document has no
50     * term vectors, add value for tvx.
51     * 
52     * @param vectors
53     * @throws IOException
54     */
55    public final void addAllDocVectors(TermFreqVector[] vectors)
56        throws IOException {
57  
58      tvx.writeLong(tvd.getFilePointer());
59      tvx.writeLong(tvf.getFilePointer());
60  
61      if (vectors != null) {
62        final int numFields = vectors.length;
63        tvd.writeVInt(numFields);
64  
65        long[] fieldPointers = new long[numFields];
66  
67        for (int i=0; i<numFields; i++) {
68          fieldPointers[i] = tvf.getFilePointer();
69  
70          final int fieldNumber = fieldInfos.fieldNumber(vectors[i].getField());
71  
72          // 1st pass: write field numbers to tvd
73          tvd.writeVInt(fieldNumber);
74  
75          final int numTerms = vectors[i].size();
76          tvf.writeVInt(numTerms);
77  
78          final TermPositionVector tpVector;
79  
80          final byte bits;
81          final boolean storePositions;
82          final boolean storeOffsets;
83  
84          if (vectors[i] instanceof TermPositionVector) {
85            // May have positions & offsets
86            tpVector = (TermPositionVector) vectors[i];
87            storePositions = tpVector.size() > 0 && tpVector.getTermPositions(0) != null;
88            storeOffsets = tpVector.size() > 0 && tpVector.getOffsets(0) != null;
89            bits = (byte) ((storePositions ? TermVectorsReader.STORE_POSITIONS_WITH_TERMVECTOR : 0) +
90                           (storeOffsets ? TermVectorsReader.STORE_OFFSET_WITH_TERMVECTOR : 0));
91          } else {
92            tpVector = null;
93            bits = 0;
94            storePositions = false;
95            storeOffsets = false;
96          }
97  
98          tvf.writeVInt(bits);
99  
100         final String[] terms = vectors[i].getTerms();
101         final int[] freqs = vectors[i].getTermFrequencies();
102 
103         int utf8Upto = 0;
104         utf8Results[1].length = 0;
105 
106         for (int j=0; j<numTerms; j++) {
107 
108           UnicodeUtil.UTF16toUTF8(terms[j], 0, terms[j].length(), utf8Results[utf8Upto]);
109           
110           int start = StringHelper.bytesDifference(utf8Results[1-utf8Upto].result,
111                                                    utf8Results[1-utf8Upto].length,
112                                                    utf8Results[utf8Upto].result,
113                                                    utf8Results[utf8Upto].length);
114           int length = utf8Results[utf8Upto].length - start;
115           tvf.writeVInt(start);       // write shared prefix length
116           tvf.writeVInt(length);        // write delta length
117           tvf.writeBytes(utf8Results[utf8Upto].result, start, length);  // write delta bytes
118           utf8Upto = 1-utf8Upto;
119 
120           final int termFreq = freqs[j];
121 
122           tvf.writeVInt(termFreq);
123 
124           if (storePositions) {
125             final int[] positions = tpVector.getTermPositions(j);
126             if (positions == null)
127               throw new IllegalStateException("Trying to write positions that are null!");
128             assert positions.length == termFreq;
129 
130             // use delta encoding for positions
131             int lastPosition = 0;
132             for(int k=0;k<positions.length;k++) {
133               final int position = positions[k];
134               tvf.writeVInt(position-lastPosition);
135               lastPosition = position;
136             }
137           }
138 
139           if (storeOffsets) {
140             final TermVectorOffsetInfo[] offsets = tpVector.getOffsets(j);
141             if (offsets == null)
142               throw new IllegalStateException("Trying to write offsets that are null!");
143             assert offsets.length == termFreq;
144 
145             // use delta encoding for offsets
146             int lastEndOffset = 0;
147             for(int k=0;k<offsets.length;k++) {
148               final int startOffset = offsets[k].getStartOffset();
149               final int endOffset = offsets[k].getEndOffset();
150               tvf.writeVInt(startOffset-lastEndOffset);
151               tvf.writeVInt(endOffset-startOffset);
152               lastEndOffset = endOffset;
153             }
154           }
155         }
156       }
157 
158       // 2nd pass: write field pointers to tvd
159       if (numFields > 1) {
160         long lastFieldPointer = fieldPointers[0];
161         for (int i=1; i<numFields; i++) {
162           final long fieldPointer = fieldPointers[i];
163           tvd.writeVLong(fieldPointer-lastFieldPointer);
164           lastFieldPointer = fieldPointer;
165         }
166       }
167     } else
168       tvd.writeVInt(0);
169   }

171   /**
172    * Do a bulk copy of numDocs documents from reader to our
173    * streams.  This is used to expedite merging, if the
174    * field numbers are congruent.
175    */
176   final void addRawDocuments(TermVectorsReader reader, int[] tvdLengths, int[] tvfLengths, int numDocs) throws IOException {
177     long tvdPosition = tvd.getFilePointer();
178     long tvfPosition = tvf.getFilePointer();
179     long tvdStart = tvdPosition;
180     long tvfStart = tvfPosition;
181     for(int i=0;i<numDocs;i++) {
182       tvx.writeLong(tvdPosition);
183       tvdPosition += tvdLengths[i];
184       tvx.writeLong(tvfPosition);
185       tvfPosition += tvfLengths[i];
186     }
187     tvd.copyBytes(reader.getTvdStream(), tvdPosition-tvdStart);
188     tvf.copyBytes(reader.getTvfStream(), tvfPosition-tvfStart);
189     assert tvd.getFilePointer() == tvdPosition;
190     assert tvf.getFilePointer() == tvfPosition;
191   }
192   
193   /** Close all streams. */
194   final void close() throws IOException {
195     // make an effort to close all streams we can but remember and re-throw
196     // the first exception encountered in this process
197     IOException keep = null;
198     if (tvx != null)
199       try {
200         tvx.close();
201       } catch (IOException e) {
202         if (keep == null) keep = e;
203       }
204     if (tvd != null)
205       try {
206         tvd.close();
207       } catch (IOException e) {
208         if (keep == null) keep = e;
209       }
210     if (tvf != null)
211       try {
212         tvf.close();
213       } catch (IOException e) {
214         if (keep == null) keep = e;
215       }
216     if (keep != null) throw (IOException) keep.fillInStackTrace();
217   }
