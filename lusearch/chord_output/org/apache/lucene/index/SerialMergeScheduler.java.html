20  import java.io.IOException;
21  
22  /** A {@link MergeScheduler} that simply does each merge
23   *  sequentially, using the current thread. */
24  public class SerialMergeScheduler extends MergeScheduler {
25  
26    /** Just do the merges in sequence. We do this
27     * "synchronized" so that even if the application is using
28     * multiple threads, only one merge may run at a time. */
29    synchronized public void merge(IndexWriter writer)
30      throws CorruptIndexException, IOException {
31  
32      while(true) {
33        MergePolicy.OneMerge merge = writer.getNextMerge();
34        if (merge == null)
35          break;
36        writer.merge(merge);
37      }
38    }
39  
40    public void close() {}
41  }
