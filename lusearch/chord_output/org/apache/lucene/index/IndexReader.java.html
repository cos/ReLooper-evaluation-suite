32  /** IndexReader is an abstract class, providing an interface for accessing an
33   index.  Search of an index is done entirely through this abstract interface,
34   so that any subclass which implements it is searchable.
35  
36   <p> Concrete subclasses of IndexReader are usually constructed with a call to
37   one of the static <code>open()</code> methods, e.g. {@link
38   #open(String, boolean)}.
39  
40   <p> For efficiency, in this API documents are often referred to via
41   <i>document numbers</i>, non-negative integers which each name a unique
42   document in the index.  These document numbers are ephemeral--they may change
43   as documents are added to and deleted from an index.  Clients should thus not
44   rely on a given document having the same number between sessions.
45  
46   <p> An IndexReader can be opened on a directory for which an IndexWriter is
47   opened already, but it cannot be used to delete documents from the index then.
48  
49   <p>
50   <b>NOTE</b>: for backwards API compatibility, several methods are not listed 
51   as abstract, but have no useful implementations in this base class and 
52   instead always throw UnsupportedOperationException.  Subclasses are 
53   strongly encouraged to override these methods, but in many cases may not 
54   need to.
55   </p>
56  
57   <p>
58  
59   <b>NOTE</b>: as of 2.4, it's possible to open a read-only
60   IndexReader using one of the static open methods that
61   accepts the boolean readOnly parameter.  Such a reader has
62   better concurrency as it's not necessary to synchronize on
63   the isDeleted method.  Currently the default for readOnly
64   is false, meaning if not specified you will get a
65   read/write IndexReader.  But in 3.0 this default will
66   change to true, meaning you must explicitly specify false
67   if you want to make changes with the resulting IndexReader.
68   </p>
69  
70   <a name="thread-safety"></a><p><b>NOTE</b>: {@link
71   <code>IndexReader</code>} instances are completely thread
72   safe, meaning multiple threads can call any of its methods,
73   concurrently.  If your application requires external
74   synchronization, you should <b>not</b> synchronize on the
75   <code>IndexReader</code> instance; use your own
76   (non-Lucene) objects instead.
77   
78   @version $Id: IndexReader.java 826049 2009-10-16 19:28:55Z mikemccand $
79  */

135   /**
136    * Expert: increments the refCount of this IndexReader
137    * instance.  RefCounts are used to determine when a
138    * reader can be closed safely, i.e. as soon as there are
139    * no more references.  Be sure to always call a
140    * corresponding {@link #decRef}, in a finally clause;
141    * otherwise the reader may never be closed.  Note that
142    * {@link #close} simply calls decRef(), which means that
143    * the IndexReader will not really be closed until {@link
144    * #decRef} has been called for all outstanding
145    * references.
146    *
147    * @see #decRef
148    */
149   public synchronized void incRef() {
150     assert refCount > 0;
151     ensureOpen();
152     refCount++;
153   }
154 
155   /**
156    * Expert: decreases the refCount of this IndexReader
157    * instance.  If the refCount drops to 0, then pending
158    * changes (if any) are committed to the index and this
159    * reader is closed.
160    * 
161    * @throws IOException in case an IOException occurs in commit() or doClose()
162    *
163    * @see #incRef
164    */
165   public synchronized void decRef() throws IOException {
166     assert refCount > 0;
167     ensureOpen();
168     if (refCount == 1) {
169       commit();
170       doClose();
171     }
172     refCount--;
173   }

241     IndexReader r = null;
242     try {
243       r = open(dir, null, null, readOnly, DEFAULT_TERMS_INDEX_DIVISOR);

247     }
248     return new DirectoryOwningReader(r);

479   /**
480    * Refreshes an IndexReader if the index has changed since this instance 
481    * was (re)opened. 
482    * <p>
483    * Opening an IndexReader is an expensive operation. This method can be used
484    * to refresh an existing IndexReader to reduce these costs. This method 
485    * tries to only load segments that have changed or were created after the 
486    * IndexReader was (re)opened.
487    * <p>
488    * If the index has not changed since this instance was (re)opened, then this
489    * call is a NOOP and returns this instance. Otherwise, a new instance is 
490    * returned. The old instance is <b>not</b> closed and remains usable.<br>
491    * <p>   
492    * If the reader is reopened, even though they share
493    * resources internally, it's safe to make changes
494    * (deletions, norms) with the new reader.  All shared
495    * mutable state obeys "copy on write" semantics to ensure
496    * the changes are not seen by other readers.
497    * <p>
498    * You can determine whether a reader was actually reopened by comparing the
499    * old instance with the instance returned by this method: 
500    * <pre>
501    * IndexReader reader = ... 
502    * ...
503    * IndexReader newReader = r.reopen();
504    * if (newReader != reader) {
505    * ...     // reader was reopened
506    *   reader.close(); 
507    * }
508    * reader = newReader;
509    * ...
510    * </pre>
511    *
512    * Be sure to synchronize that code so that other threads,
513    * if present, can never use reader after it has been
514    * closed and before it's switched to newReader.
515    *
516    * <p><b>NOTE</b>: If this reader is a near real-time
517    * reader (obtained from {@link IndexWriter#getReader()},
518    * reopen() will simply call writer.getReader() again for
519    * you, though this may change in the future.
520    * 
521    * @throws CorruptIndexException if the index is corrupt
522    * @throws IOException if there is a low-level IO error
523    */  
524   public synchronized IndexReader reopen() throws CorruptIndexException, IOException {
525     throw new UnsupportedOperationException("This reader does not support reopen().");
526   }

549    * internal state).
550    * <p>
551    * On cloning a reader with pending changes (deletions,
552    * norms), the original reader transfers its write lock to
553    * the cloned reader.  This means only the cloned reader

559    * either the original or the cloned reader: all shared
560    * mutable state obeys "copy on write" semantics to ensure
561    * the changes are not seen by other readers.
562    * <p>
563    * @throws CorruptIndexException if the index is corrupt
564    * @throws IOException if there is a low-level IO error
565    */

1177   * See {@link #deleteDocument(int)} for information about when this deletion will 
1178   * become effective.
1179   *
1180   * @return the number of documents deleted
1181   * @throws StaleReaderException if the index has changed
1182   *  since this reader was opened
1183   * @throws CorruptIndexException if the index is corrupt
1184   * @throws LockObtainFailedException if another writer
1185   *  has this index open (<code>write.lock</code> could not
1186   *  be obtained)
1187   * @throws IOException if there is a low-level IO error
1188   */
1189  public int deleteDocuments(Term term) throws StaleReaderException, CorruptIndexException, LockObtainFailedException, IOException {
1190    ensureOpen();
1191    TermDocs docs = termDocs(term);
1192    if (docs == null) return 0;
1193    int n = 0;
1194    try {
1195      while (docs.next()) {
1196        deleteDocument(docs.doc());
1197        n++;
1198      }
1199    } finally {
1200      docs.close();
1201    }
1202    return n;
1203  }

1379  /**
1380   * Prints the filename and size of each file within a given compound file.
1381   * Add the -extract flag to extract files to the current working directory.
1382   * In order to make the extracted version of the index work, you have to copy
1383   * the segments file from the compound index into the directory where the extracted files are stored.
1384   * @param args Usage: org.apache.lucene.index.IndexReader [-extract] &lt;cfsfile&gt;
1385   */
1386  public static void main(String [] args) {
1387    String filename = null;
1388    boolean extract = false;
1389
1390    for (int i = 0; i < args.length; ++i) {
1391      if (args[i].equals("-extract")) {
1392        extract = true;
1393      } else if (filename == null) {
1394        filename = args[i];
1395      }
1396    }
1397
1398    if (filename == null) {
1399      System.out.println("Usage: org.apache.lucene.index.IndexReader [-extract] <cfsfile>");
1400      return;
1401    }
1402
1403    Directory dir = null;
1404    CompoundFileReader cfr = null;
1405
1406    try {
1407      File file = new File(filename);
1408      String dirname = file.getAbsoluteFile().getParent();
1409      filename = file.getName();
1410      dir = FSDirectory.open(new File(dirname));
1411      cfr = new CompoundFileReader(dir, filename);
1412
1413      String [] files = cfr.list();
1414      Arrays.sort(files);   // sort the array of filename so that the output is more readable
1415
1416      for (int i = 0; i < files.length; ++i) {
1417        long len = cfr.fileLength(files[i]);
1418
1419        if (extract) {
1420          System.out.println("extract " + files[i] + " with " + len + " bytes to local directory...");
1421          IndexInput ii = cfr.openInput(files[i]);
1422
1423          FileOutputStream f = new FileOutputStream(files[i]);
1424
1425          // read and write with a small buffer, which is more effective than reading byte by byte
1426          byte[] buffer = new byte[1024];
1427          int chunk = buffer.length;
1428          while(len > 0) {
1429            final int bufLen = (int) Math.min(chunk, len);
1430            ii.readBytes(buffer, 0, bufLen);
1431            f.write(buffer, 0, bufLen);
1432            len -= bufLen;
1433          }
1434
1435          f.close();
1436          ii.close();
1437        }
1438        else
1439          System.out.println(files[i] + ": " + len + " bytes");
1440      }
1441    } catch (IOException ioe) {
1442      ioe.printStackTrace();
1443    }
1444    finally {
1445      try {
1446        if (dir != null)
1447          dir.close();
1448        if (cfr != null)
1449          cfr.close();
1450      }
1451      catch (IOException ioe) {
1452        ioe.printStackTrace();
1453      }
1454    }
1455  }
