20  import java.io.IOException;
21  
22  /** 
23   * This class keeps track of closing the underlying directory. It is used to wrap
24   * DirectoryReaders, that are created using a String/File parameter
25   * in IndexReader.open() with FSDirectory.getDirectory().
26   * @deprecated This helper class is removed with all String/File
27   * IndexReader.open() methods in Lucene 3.0
28   */
29  final class DirectoryOwningReader extends FilterIndexReader implements Cloneable {
30  
31    DirectoryOwningReader(final IndexReader in) {
32      super(in);
33      this.ref = new SegmentReader.Ref();
34      assert this.ref.refCount() == 1;
35    }
36  
37    private DirectoryOwningReader(final IndexReader in, final SegmentReader.Ref ref) {
38      super(in);
39      this.ref = ref;
40      ref.incRef();
41    }
42  
43    public IndexReader reopen() throws CorruptIndexException, IOException {
44      ensureOpen();
45      final IndexReader r = in.reopen();
46      if (r != in)
47        return new DirectoryOwningReader(r, ref);
48      return this;
49    }
50  
51    public IndexReader reopen(boolean openReadOnly) throws CorruptIndexException, IOException {
52      ensureOpen();
53      final IndexReader r = in.reopen(openReadOnly);
54      if (r != in)
55        return new DirectoryOwningReader(r, ref);
56      return this;
57    }
58  
59    public IndexReader reopen(final IndexCommit commit) throws CorruptIndexException, IOException {
60      ensureOpen();
61      final IndexReader r = in.reopen(commit);
62      if (r != in)
63        return new DirectoryOwningReader(r, ref);
64      return this;
65    }
66  
67    public Object clone() {
68      ensureOpen();
69      return new DirectoryOwningReader((IndexReader) in.clone(), ref);
70    }
71  
72    public IndexReader clone(boolean openReadOnly) throws CorruptIndexException, IOException {
73      ensureOpen();
74      return new DirectoryOwningReader(in.clone(openReadOnly), ref);
75    }
76  
77    protected void doClose() throws IOException {
78      IOException ioe = null;
79      // close the reader, record exception
80      try {
81        super.doClose();
82      } catch (IOException e) {
83        ioe = e;
84      }
85      // close the directory, record exception
86      if (ref.decRef() == 0) {
87        try {
88          in.directory().close();
89        } catch (IOException e) {
90          if (ioe == null) ioe = e;
91        }
92      }
93      // throw the first exception
94      if (ioe != null) throw ioe;
95    }
96  
97    /**
98     * This member contains the ref counter, that is passed to each instance after cloning/reopening,
99     * and is global to all DirectoryOwningReader derived from the original one.
100    * This reuses the class {@link SegmentReader.Ref}
101    */
102   private final SegmentReader.Ref ref;
103 
104 }
