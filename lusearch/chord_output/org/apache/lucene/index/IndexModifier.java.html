164   /**
165    * Initialize an IndexWriter.
166    * @throws CorruptIndexException if the index is corrupt
167    * @throws LockObtainFailedException if another writer
168    *  has this index open (<code>write.lock</code> could not

172   protected void init(Directory directory, Analyzer analyzer, boolean create) throws CorruptIndexException, LockObtainFailedException, IOException {
173     this.directory = directory;
174     synchronized(this.directory) {
175       this.analyzer = analyzer;
176       indexWriter = new IndexWriter(directory, analyzer, create, IndexWriter.MaxFieldLength.LIMITED);
177       open = true;
178     }

191   /**
192    * Close the IndexReader and open an IndexWriter.
193    * @throws CorruptIndexException if the index is corrupt
194    * @throws LockObtainFailedException if another writer
195    *  has this index open (<code>write.lock</code> could not
196    *  be obtained)
197    * @throws IOException if there is a low-level IO error
198    */
199   protected void createIndexWriter() throws CorruptIndexException, LockObtainFailedException, IOException {
200     if (indexWriter == null) {
201       if (indexReader != null) {
202         indexReader.close();
203         indexReader = null;
204       }
205       indexWriter = new IndexWriter(directory, analyzer, false, new IndexWriter.MaxFieldLength(maxFieldLength));
206       // IndexModifier cannot use ConcurrentMergeScheduler
207       // because it synchronizes on the directory which can
208       // cause deadlock
209       indexWriter.setMergeScheduler(new SerialMergeScheduler());
210       indexWriter.setInfoStream(infoStream);
211       indexWriter.setUseCompoundFile(useCompoundFile);
212       if (maxBufferedDocs != IndexWriter.DISABLE_AUTO_FLUSH)
213         indexWriter.setMaxBufferedDocs(maxBufferedDocs);
214       indexWriter.setMergeFactor(mergeFactor);
215     }
216   }

224     if (indexReader == null) {

234    * Make sure all changes are written to disk.
235    * @throws CorruptIndexException if the index is corrupt
236    * @throws LockObtainFailedException if another writer

241   public void flush() throws CorruptIndexException, LockObtainFailedException, IOException {
242     synchronized(directory) {
243       assureOpen();
244       if (indexWriter != null) {
245         indexWriter.close();
246         indexWriter = null;

252       }

566    *
567    * @throws IllegalStateException if the index has been closed before already
568    * @throws CorruptIndexException if the index is corrupt
569    * @throws IOException if there is a low-level IO error
570    */
571   public void close() throws CorruptIndexException, IOException {
572     synchronized(directory) {
573       if (!open)
574         throw new IllegalStateException("Index is closed already");
575       if (indexWriter != null) {
576         indexWriter.close();
577         indexWriter = null;
578       } else if (indexReader != null) {
579         indexReader.close();
580         indexReader = null;
581       }
582       open = false;
583       if (closeDir) {
584         directory.close();
585       }
586       closeDir = false;
587     }
588   }
589 
590   public String toString() {
591     return "Index@" + directory;
592   }
