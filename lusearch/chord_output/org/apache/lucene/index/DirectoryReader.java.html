67    static IndexReader open(final Directory directory, final IndexDeletionPolicy deletionPolicy, final IndexCommit commit, final boolean readOnly,
68                            final int termInfosIndexDivisor) throws CorruptIndexException, IOException {
69      return (IndexReader) new SegmentInfos.FindSegmentsFile(directory) {
70        protected Object doBody(String segmentFileName) throws CorruptIndexException, IOException {
71          SegmentInfos infos = new SegmentInfos();
72          infos.read(directory, segmentFileName);
73          if (readOnly)
74            return new ReadOnlyDirectoryReader(directory, infos, deletionPolicy, termInfosIndexDivisor);
75          else
76            return new DirectoryReader(directory, infos, deletionPolicy, false, termInfosIndexDivisor);
77        }
78      }.run(commit);
79    }

81    /** Construct reading the named set of readers. */
82    DirectoryReader(Directory directory, SegmentInfos sis, IndexDeletionPolicy deletionPolicy, boolean readOnly, int termInfosIndexDivisor) throws IOException {
83      this.directory = directory;
84      this.readOnly = readOnly;
85      this.segmentInfos = sis;
86      this.deletionPolicy = deletionPolicy;
87      this.termInfosIndexDivisor = termInfosIndexDivisor;
88  
89      if (!readOnly) {
90        // We assume that this segments_N was previously
91        // properly sync'd:
92        synced.addAll(sis.files(directory, true));
93      }
94  
95      // To reduce the chance of hitting FileNotFound
96      // (and having to retry), we open segments in
97      // reverse because IndexWriter merges & deletes
98      // the newest segments first.
99  
100     SegmentReader[] readers = new SegmentReader[sis.size()];
101     for (int i = sis.size()-1; i >= 0; i--) {
102       boolean success = false;
103       try {
104         readers[i] = SegmentReader.get(readOnly, sis.info(i), termInfosIndexDivisor);
105         success = true;
106       } finally {
107         if (!success) {
108           // Close all readers we had opened:
109           for(i++;i<sis.size();i++) {
110             try {
111               readers[i].close();
112             } catch (Throwable ignore) {
113               // keep going - we want to clean up as much as possible
114             }
115           }
116         }
117       }
118     }
119 
120     initialize(readers);
121   }

123   // Used by near real-time search
124   DirectoryReader(IndexWriter writer, SegmentInfos infos, int termInfosIndexDivisor) throws IOException {
125     this.directory = writer.getDirectory();
126     this.readOnly = true;
127     this.segmentInfos = infos;
128     segmentInfosStart = (SegmentInfos) infos.clone();
129     this.termInfosIndexDivisor = termInfosIndexDivisor;
130     if (!readOnly) {
131       // We assume that this segments_N was previously
132       // properly sync'd:
133       synced.addAll(infos.files(directory, true));
134     }
135 
136     // IndexWriter synchronizes externally before calling
137     // us, which ensures infos will not change; so there's
138     // no need to process segments in reverse order
139     final int numSegments = infos.size();
140     SegmentReader[] readers = new SegmentReader[numSegments];
141     final Directory dir = writer.getDirectory();
142     int upto = 0;
143 
144     for (int i=0;i<numSegments;i++) {
145       boolean success = false;
146       try {
147         final SegmentInfo info = infos.info(upto);
148         if (info.dir == dir) {
149           readers[upto++] = writer.readerPool.getReadOnlyClone(info, true, termInfosIndexDivisor);
150         }
151         success = true;
152       } finally {
153         if (!success) {
154           // Close all readers we had opened:
155           for(upto--;upto>=0;upto--) {
156             try {
157               readers[upto].close();
158             } catch (Throwable ignore) {
159               // keep going - we want to clean up as much as possible
160             }
161           }
162         }
163       }
164     }
165 
166     this.writer = writer;
167 
168     if (upto < readers.length) {
169       // This means some segments were in a foreign Directory
170       SegmentReader[] newReaders = new SegmentReader[upto];
171       System.arraycopy(readers, 0, newReaders, 0, upto);
172       readers = newReaders;
173     }
174 
175     initialize(readers);
176   }

178   /** This constructor is only used for {@link #reopen()} */
179   DirectoryReader(Directory directory, SegmentInfos infos, SegmentReader[] oldReaders, int[] oldStarts,
180                   Map oldNormsCache, boolean readOnly, boolean doClone, int termInfosIndexDivisor) throws IOException {
181     this.directory = directory;
182     this.readOnly = readOnly;
183     this.segmentInfos = infos;
184     this.termInfosIndexDivisor = termInfosIndexDivisor;
185     if (!readOnly) {
186       // We assume that this segments_N was previously
187       // properly sync'd:
188       synced.addAll(infos.files(directory, true));
189     }
190 
191     // we put the old SegmentReaders in a map, that allows us
192     // to lookup a reader using its segment name
193     Map segmentReaders = new HashMap();
194 
195     if (oldReaders != null) {
196       // create a Map SegmentName->SegmentReader
197       for (int i = 0; i < oldReaders.length; i++) {
198         segmentReaders.put(oldReaders[i].getSegmentName(), new Integer(i));
199       }
200     }
201     
202     SegmentReader[] newReaders = new SegmentReader[infos.size()];
203     
204     // remember which readers are shared between the old and the re-opened
205     // DirectoryReader - we have to incRef those readers
206     boolean[] readerShared = new boolean[infos.size()];
207     
208     for (int i = infos.size() - 1; i>=0; i--) {
209       // find SegmentReader for this segment
210       Integer oldReaderIndex = (Integer) segmentReaders.get(infos.info(i).name);
211       if (oldReaderIndex == null) {
212         // this is a new segment, no old SegmentReader can be reused
213         newReaders[i] = null;
214       } else {
215         // there is an old reader for this segment - we'll try to reopen it
216         newReaders[i] = oldReaders[oldReaderIndex.intValue()];
217       }
218 
219       boolean success = false;
220       try {
221         SegmentReader newReader;
222         if (newReaders[i] == null || infos.info(i).getUseCompoundFile() != newReaders[i].getSegmentInfo().getUseCompoundFile()) {
223 
224           // We should never see a totally new segment during cloning
225           assert !doClone;
226 
227           // this is a new reader; in case we hit an exception we can close it safely
228           newReader = SegmentReader.get(readOnly, infos.info(i), termInfosIndexDivisor);
229         } else {
230           newReader = newReaders[i].reopenSegment(infos.info(i), doClone, readOnly);
231         }
232         if (newReader == newReaders[i]) {
233           // this reader will be shared between the old and the new one,
234           // so we must incRef it
235           readerShared[i] = true;
236           newReader.incRef();
237         } else {
238           readerShared[i] = false;
239           newReaders[i] = newReader;
240         }
241         success = true;
242       } finally {
243         if (!success) {
244           for (i++; i < infos.size(); i++) {
245             if (newReaders[i] != null) {
246               try {
247                 if (!readerShared[i]) {
248                   // this is a new subReader that is not used by the old one,
249                   // we can close it
250                   newReaders[i].close();
251                 } else {
252                   // this subReader is also used by the old reader, so instead
253                   // closing we must decRef it
254                   newReaders[i].decRef();
255                 }
256               } catch (IOException ignore) {
257                 // keep going - we want to clean up as much as possible
258               }
259             }
260           }
261         }
262       }
263     }    
264     
265     // initialize the readers to calculate maxDoc before we try to reuse the old normsCache
266     initialize(newReaders);
267     
268     // try to copy unchanged norms from the old normsCache to the new one
269     if (oldNormsCache != null) {
270       Iterator it = oldNormsCache.entrySet().iterator();
271       while (it.hasNext()) {
272         Map.Entry entry = (Map.Entry) it.next();
273         String field = (String) entry.getKey();
274         if (!hasNorms(field)) {
275           continue;
276         }
277 
278         byte[] oldBytes = (byte[]) entry.getValue();
279 
280         byte[] bytes = new byte[maxDoc()];
281 
282         for (int i = 0; i < subReaders.length; i++) {
283           Integer oldReaderIndex = ((Integer) segmentReaders.get(subReaders[i].getSegmentName()));
284 
285           // this SegmentReader was not re-opened, we can copy all of its norms 
286           if (oldReaderIndex != null &&
287                (oldReaders[oldReaderIndex.intValue()] == subReaders[i] 
288                  || oldReaders[oldReaderIndex.intValue()].norms.get(field) == subReaders[i].norms.get(field))) {
289             // we don't have to synchronize here: either this constructor is called from a SegmentReader,
290             // in which case no old norms cache is present, or it is called from MultiReader.reopen(),
291             // which is synchronized
292             System.arraycopy(oldBytes, oldStarts[oldReaderIndex.intValue()], bytes, starts[i], starts[i+1] - starts[i]);
293           } else {
294             subReaders[i].norms(field, bytes, starts[i]);
295           }
296         }
297 
298         normsCache.put(field, bytes);      // update cache
299       }
300     }
301   }

303   private void initialize(SegmentReader[] subReaders) {
304     this.subReaders = subReaders;
305     starts = new int[subReaders.length + 1];    // build starts array
306     for (int i = 0; i < subReaders.length; i++) {
307       starts[i] = maxDoc;
308       maxDoc += subReaders[i].maxDoc();      // compute maxDocs
309 
310       if (subReaders[i].hasDeletions())
311         hasDeletions = true;
312     }
313     starts[subReaders.length] = maxDoc;
314   }

324   public final synchronized IndexReader clone(boolean openReadOnly) throws CorruptIndexException, IOException {
325     DirectoryReader newReader = doReopen((SegmentInfos) segmentInfos.clone(), true, openReadOnly);
326 
327     if (this != newReader) {
328       newReader.deletionPolicy = deletionPolicy;
329     }
330     newReader.writer = writer;
331     // If we're cloning a non-readOnly reader, move the
332     // writeLock (if there is one) to the new reader:
333     if (!openReadOnly && writeLock != null) {
334       // In near real-time search, reader is always readonly
335       assert writer == null;
336       newReader.writeLock = writeLock;
337       newReader.hasChanges = hasChanges;
338       newReader.hasDeletions = hasDeletions;
339       writeLock = null;
340       hasChanges = false;
341     }
342 
343     return newReader;
344   }

359   private synchronized IndexReader doReopen(final boolean openReadOnly, IndexCommit commit) throws CorruptIndexException, IOException {
360     ensureOpen();
361 
362     assert commit == null || openReadOnly;
363 
364     // If we were obtained by writer.getReader(), re-ask the
365     // writer to get a new reader.
366     if (writer != null) {
367       assert readOnly;
368 
369       if (!openReadOnly) {
370         throw new IllegalArgumentException("a reader obtained from IndexWriter.getReader() can only be reopened with openReadOnly=true (got false)");
371       }
372 
373       if (commit != null) {
374         throw new IllegalArgumentException("a reader obtained from IndexWriter.getReader() cannot currently accept a commit");
375       }
376 
377       if (!writer.isOpen(true)) {
378         throw new AlreadyClosedException("cannot reopen: the IndexWriter this reader was obtained from is now closed");
379       }
380 
381       // TODO: right now we *always* make a new reader; in
382       // the future we could have write make some effort to
383       // detect that no changes have occurred
384       IndexReader reader = writer.getReader();
385       reader.setDisableFakeNorms(getDisableFakeNorms());
386       return reader;
387     }
388 
389     if (commit == null) {
390       if (hasChanges) {
391         // We have changes, which means we are not readOnly:
392         assert readOnly == false;
393         // and we hold the write lock:
394         assert writeLock != null;
395         // so no other writer holds the write lock, which
396         // means no changes could have been done to the index:
397         assert isCurrent();
398 
399         if (openReadOnly) {
400           return (IndexReader) clone(openReadOnly);
401         } else {
402           return this;
403         }
404       } else if (isCurrent()) {
405         if (openReadOnly != readOnly) {
406           // Just fallback to clone
407           return (IndexReader) clone(openReadOnly);
408         } else {
409           return this;
410         }
411       }
412     } else {
413       if (directory != commit.getDirectory())
414         throw new IOException("the specified commit does not match the specified Directory");
415       if (segmentInfos != null && commit.getSegmentsFileName().equals(segmentInfos.getCurrentSegmentFileName())) {
416         if (readOnly != openReadOnly) {
417           // Just fallback to clone
418           return (IndexReader) clone(openReadOnly);
419         } else {
420           return this;
421         }
422       }
423     }
424 
425     return (IndexReader) new SegmentInfos.FindSegmentsFile(directory) {
426       protected Object doBody(String segmentFileName) throws CorruptIndexException, IOException {
427         SegmentInfos infos = new SegmentInfos();
428         infos.read(directory, segmentFileName);
429         return doReopen(infos, false, openReadOnly);
430       }
431     }.run(commit);
432   }

440     }
441     reader.setDisableFakeNorms(getDisableFakeNorms());
442     return reader;
443   }

539   final static int readerIndex(int n, int[] starts, int numSubReaders) {    // find reader for doc n:
540     int lo = 0;                                      // search starts array
541     int hi = numSubReaders - 1;                  // for first element less
542 
543     while (hi >= lo) {
544       int mid = (lo + hi) >>> 1;
545       int midValue = starts[mid];
546       if (n < midValue)
547         hi = mid - 1;
548       else if (n > midValue)
549         lo = mid + 1;
550       else {                                      // found a match
551         while (mid+1 < numSubReaders && starts[mid+1] == midValue) {
552           mid++;                                  // scan to last match
553         }
554         return mid;
555       }
556     }
557     return hi;
558   }

574   public synchronized byte[] norms(String field) throws IOException {
575     ensureOpen();
576     byte[] bytes = (byte[])normsCache.get(field);
577     if (bytes != null)
578       return bytes;          // cache hit
579     if (!hasNorms(field))
580       return getDisableFakeNorms() ? null : fakeNorms();
581 
582     bytes = new byte[maxDoc()];
583     for (int i = 0; i < subReaders.length; i++)
584       subReaders[i].norms(field, bytes, starts[i]);
585     normsCache.put(field, bytes);      // update cache
586     return bytes;
587   }

589   public synchronized void norms(String field, byte[] result, int offset)
590     throws IOException {
591     ensureOpen();
592     byte[] bytes = (byte[])normsCache.get(field);
593     if (bytes==null && !hasNorms(field)) {
594       Arrays.fill(result, offset, result.length, DefaultSimilarity.encodeNorm(1.0f));
595     } else if (bytes != null) {                           // cache hit
596       System.arraycopy(bytes, 0, result, offset, maxDoc());
597     } else {
598       for (int i = 0; i < subReaders.length; i++) {      // read from segments
599         subReaders[i].norms(field, result, offset + starts[i]);
600       }
601     }
602   }

641   /**
642    * Tries to acquire the WriteLock on this directory. this method is only valid if this IndexReader is directory
643    * owner.
644    *
645    * @throws StaleReaderException  if the index has changed since this reader was opened
646    * @throws CorruptIndexException if the index is corrupt
647    * @throws org.apache.lucene.store.LockObtainFailedException
648    *                               if another writer has this index open (<code>write.lock</code> could not be
649    *                               obtained)
650    * @throws IOException           if there is a low-level IO error
651    */
652   protected void acquireWriteLock() throws StaleReaderException, CorruptIndexException, LockObtainFailedException, IOException {
653 
654     if (readOnly) {
655       // NOTE: we should not reach this code w/ the core
656       // IndexReader classes; however, an external subclass
657       // of IndexReader could reach this.
658       ReadOnlySegmentReader.noWrite();
659     }
660 
661     if (segmentInfos != null) {
662       ensureOpen();
663       if (stale)
664         throw new StaleReaderException("IndexReader out of date and no longer valid for delete, undelete, or setNorm operations");
665 
666       if (writeLock == null) {
667         Lock writeLock = directory.makeLock(IndexWriter.WRITE_LOCK_NAME);
668         if (!writeLock.obtain(IndexWriter.WRITE_LOCK_TIMEOUT)) // obtain write lock
669           throw new LockObtainFailedException("Index locked for write: " + writeLock);
670         this.writeLock = writeLock;
671 
672         // we have to check whether index has changed since this reader was opened.
673         // if so, this reader is no longer valid for deletion
674         if (SegmentInfos.readCurrentVersion(directory) > segmentInfos.getVersion()) {
675           stale = true;
676           this.writeLock.release();
677           this.writeLock = null;
678           throw new StaleReaderException("IndexReader out of date and no longer valid for delete, undelete, or setNorm operations");
679         }
680       }
681     }

689   /**
690    * Commit changes resulting from delete, undeleteAll, or setNorm operations
691    * <p/>
692    * If an exception is hit, then either no changes or all changes will have been committed to the index (transactional
693    * semantics).
694    *
695    * @throws IOException if there is a low-level IO error
696    */
697   protected void doCommit(Map commitUserData) throws IOException {
698     if (hasChanges) {
699       segmentInfos.setUserData(commitUserData);
700       // Default deleter (for backwards compatibility) is
701       // KeepOnlyLastCommitDeleter:
702       IndexFileDeleter deleter = new IndexFileDeleter(directory,
703                                                       deletionPolicy == null ? new KeepOnlyLastCommitDeletionPolicy() : deletionPolicy,
704                                                       segmentInfos, null, null);
705 
706       // Checkpoint the state we are about to change, in
707       // case we have to roll back:
708       startCommit();
709 
710       boolean success = false;
711       try {
712         for (int i = 0; i < subReaders.length; i++)
713           subReaders[i].commit();
714 
715         // Sync all files we just wrote
716         Iterator it = segmentInfos.files(directory, false).iterator();
717         while (it.hasNext()) {
718           final String fileName = (String) it.next();
719           if (!synced.contains(fileName)) {
720             assert directory.fileExists(fileName);
721             directory.sync(fileName);
722             synced.add(fileName);
723           }
724         }
725 
726         segmentInfos.commit(directory);
727         success = true;
728       } finally {
729 
730         if (!success) {
731 
732           // Rollback changes that were made to
733           // SegmentInfos but failed to get [fully]
734           // committed.  This way this reader instance
735           // remains consistent (matched to what's
736           // actually in the index):
737           rollbackCommit();
738 
739           // Recompute deletable files & remove them (so
740           // partially written .del files, etc, are
741           // removed):
742           deleter.refresh();
743         }
744       }
745 
746       // Have the deleter remove any now unreferenced
747       // files due to this commit:
748       deleter.checkpoint(segmentInfos, true);
749       deleter.close();
750 
751       if (writeLock != null) {
752         writeLock.release();  // release write lock
753         writeLock = null;
754       }
755     }
756     hasChanges = false;
757   }
758 
759   void startCommit() {
760     rollbackHasChanges = hasChanges;
761     rollbackSegmentInfos = (SegmentInfos) segmentInfos.clone();
762     for (int i = 0; i < subReaders.length; i++) {
763       subReaders[i].startCommit();
764     }
765   }
766 
767   void rollbackCommit() {
768     hasChanges = rollbackHasChanges;
769     for (int i = 0; i < segmentInfos.size(); i++) {
770       // Rollback each segmentInfo.  Because the
771       // SegmentReader holds a reference to the
772       // SegmentInfo we can't [easily] just replace
773       // segmentInfos, so we reset it in place instead:
774       segmentInfos.info(i).reset(rollbackSegmentInfos.info(i));
775     }
776     rollbackSegmentInfos = null;
777     for (int i = 0; i < subReaders.length; i++) {
778       subReaders[i].rollbackCommit();
779     }
780   }

799     normsCache = null;
800     for (int i = 0; i < subReaders.length; i++) {
801       // try to close each reader, even if an exception is thrown

859   /** @see org.apache.lucene.index.IndexReader#listCommits */
860   public static Collection listCommits(Directory dir) throws IOException {
861     final String[] files = dir.listAll();
862 
863     Collection commits = new ArrayList();
864 
865     SegmentInfos latest = new SegmentInfos();
866     latest.read(dir);
867     final long currentGen = latest.getGeneration();
868 
869     commits.add(new ReaderCommit(latest, dir));
870 
871     for(int i=0;i<files.length;i++) {
872 
873       final String fileName = files[i];
874 
875       if (fileName.startsWith(IndexFileNames.SEGMENTS) &&
876           !fileName.equals(IndexFileNames.SEGMENTS_GEN) &&
877           SegmentInfos.generationFromSegmentsFileName(fileName) < currentGen) {
878 
879         SegmentInfos sis = new SegmentInfos();
880         try {
881           // IOException allowed to throw there, in case
882           // segments_N is corrupt
883           sis.read(dir, fileName);
884         } catch (FileNotFoundException fnfe) {
885           // LUCENE-948: on NFS (and maybe others), if
886           // you have writers switching back and forth
887           // between machines, it's very likely that the
888           // dir listing will be stale and will claim a
889           // file segments_X exists when in fact it
890           // doesn't.  So, we catch this and handle it
891           // as if the file does not exist
892           sis = null;
893         }
894 
895         if (sis != null)
896           commits.add(new ReaderCommit(sis, dir));
897       }
898     }
899 
900     return commits;
901   }

955   static class MultiTermEnum extends TermEnum {
956     IndexReader topReader; // used for matching TermEnum to TermDocs
957     private SegmentMergeQueue queue;
958   
959     private Term term;
960     private int docFreq;
961     final SegmentMergeInfo[] matchingSegments; // null terminated array of matching segments
962 
963     public MultiTermEnum(IndexReader topReader, IndexReader[] readers, int[] starts, Term t)
964       throws IOException {
965       this.topReader = topReader;
966       queue = new SegmentMergeQueue(readers.length);
967       matchingSegments = new SegmentMergeInfo[readers.length+1];
968       for (int i = 0; i < readers.length; i++) {
969         IndexReader reader = readers[i];
970         TermEnum termEnum;
971   
972         if (t != null) {
973           termEnum = reader.terms(t);
974         } else
975           termEnum = reader.terms();
976   
977         SegmentMergeInfo smi = new SegmentMergeInfo(starts[i], termEnum, reader);
978         smi.ord = i;
979         if (t == null ? smi.next() : termEnum.term() != null)
980           queue.put(smi);          // initialize queue
981         else
982           smi.close();
983       }
984   
985       if (t != null && queue.size() > 0) {
986         next();
987       }
988     }
989   
990     public boolean next() throws IOException {
991       for (int i=0; i<matchingSegments.length; i++) {
992         SegmentMergeInfo smi = matchingSegments[i];
993         if (smi==null) break;
994         if (smi.next())
995           queue.put(smi);
996         else
997           smi.close(); // done with segment
998       }
999       
1000      int numMatchingSegments = 0;
1001      matchingSegments[0] = null;
1002
1003      SegmentMergeInfo top = (SegmentMergeInfo)queue.top();
1004
1005      if (top == null) {
1006        term = null;
1007        return false;
1008      }
1009  
1010      term = top.term;
1011      docFreq = 0;
1012  
1013      while (top != null && term.compareTo(top.term) == 0) {
1014        matchingSegments[numMatchingSegments++] = top;
1015        queue.pop();
1016        docFreq += top.termEnum.docFreq();    // increment freq
1017        top = (SegmentMergeInfo)queue.top();
1018      }
1019
1020      matchingSegments[numMatchingSegments] = null;
1021      return true;
1022    }
1023  
1024    public Term term() {
1025      return term;
1026    }
1027  
1028    public int docFreq() {
1029      return docFreq;
1030    }
1031  
1032    public void close() throws IOException {
1033      queue.close();
1034    }

1038    IndexReader topReader;  // used for matching TermEnum to TermDocs
1039    protected IndexReader[] readers;
1040    protected int[] starts;
1041    protected Term term;
1042  
1043    protected int base = 0;
1044    protected int pointer = 0;
1045  
1046    private TermDocs[] readerTermDocs;
1047    protected TermDocs current;              // == readerTermDocs[pointer]
1048
1049    private MultiTermEnum tenum;  // the term enum used for seeking... can be null
1050    int matchingSegmentPos;  // position into the matching segments from tenum
1051    SegmentMergeInfo smi;     // current segment mere info... can be null
1052
1053    public MultiTermDocs(IndexReader topReader, IndexReader[] r, int[] s) {
1054      this.topReader = topReader;
1055      readers = r;
1056      starts = s;
1057  
1058      readerTermDocs = new TermDocs[r.length];
1059    }

1061    public int doc() {
1062      return base + current.doc();
1063    }

1065      return current.freq();
1066    }

1068    public void seek(Term term) {
1069      this.term = term;
1070      this.base = 0;
1071      this.pointer = 0;
1072      this.current = null;
1073      this.tenum = null;
1074      this.smi = null;
1075      this.matchingSegmentPos = 0;
1076    }
1077  
1078    public void seek(TermEnum termEnum) throws IOException {
1079      seek(termEnum.term());
1080      if (termEnum instanceof MultiTermEnum) {
1081        tenum = (MultiTermEnum)termEnum;
1082        if (topReader != tenum.topReader)
1083          tenum = null;
1084      }
1085    }
1086  
1087    public boolean next() throws IOException {
1088      for(;;) {
1089        if (current!=null && current.next()) {
1090          return true;
1091        }
1092        else if (pointer < readers.length) {
1093          if (tenum != null) {
1094            smi = tenum.matchingSegments[matchingSegmentPos++];
1095            if (smi==null) {
1096              pointer = readers.length;
1097              return false;
1098            }
1099            pointer = smi.ord;
1100          }
1101          base = starts[pointer];
1102          current = termDocs(pointer++);
1103        } else {
1104          return false;
1105        }
1106      }
1107    }
1108  
1109    /** Optimized implementation. */
1110    public int read(final int[] docs, final int[] freqs) throws IOException {
1111      while (true) {
1112        while (current == null) {
1113          if (pointer < readers.length) {      // try next segment
1114            if (tenum != null) {
1115              smi = tenum.matchingSegments[matchingSegmentPos++];
1116              if (smi==null) {
1117                pointer = readers.length;
1118                return 0;
1119              }
1120              pointer = smi.ord;
1121            }
1122            base = starts[pointer];
1123            current = termDocs(pointer++);
1124          } else {
1125            return 0;
1126          }
1127        }
1128        int end = current.read(docs, freqs);
1129        if (end == 0) {          // none left in segment
1130          current = null;
1131        } else {            // got some
1132          final int b = base;        // adjust doc numbers
1133          for (int i = 0; i < end; i++)
1134           docs[i] += b;
1135          return end;
1136        }
1137      }
1138    }

1140   /* A Possible future optimization could skip entire segments */ 
1141    public boolean skipTo(int target) throws IOException {
1142      for(;;) {
1143        if (current != null && current.skipTo(target-base)) {
1144          return true;
1145        } else if (pointer < readers.length) {
1146          if (tenum != null) {
1147            SegmentMergeInfo smi = tenum.matchingSegments[matchingSegmentPos++];
1148            if (smi==null) {
1149              pointer = readers.length;
1150              return false;
1151            }
1152            pointer = smi.ord;
1153          }
1154          base = starts[pointer];
1155          current = termDocs(pointer++);
1156        } else
1157          return false;
1158      }
1159    }

1161    private TermDocs termDocs(int i) throws IOException {
1162      TermDocs result = readerTermDocs[i];
1163      if (result == null)
1164        result = readerTermDocs[i] = termDocs(readers[i]);
1165      if (smi != null) {
1166        assert(smi.ord == i);
1167        assert(smi.termEnum.term().equals(term));
1168        result.seek(smi.termEnum);
1169      } else {
1170        result.seek(term);
1171      }
1172      return result;
1173    }

1176      throws IOException {
1177      return term==null ? reader.termDocs(null) : reader.termDocs();
1178    }

1181      for (int i = 0; i < readerTermDocs.length; i++) {
1182        if (readerTermDocs[i] != null)
1183          readerTermDocs[i].close();
1184      }
