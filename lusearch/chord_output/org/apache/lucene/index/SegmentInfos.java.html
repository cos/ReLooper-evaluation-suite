46    /** The file format version, a negative number. */
47    /* Works since counter, the old 1st entry, is always >= 0 */
48    public static final int FORMAT = -1;
49  
50    /** This format adds details used for lockless commits.  It differs
51     * slightly from the previous format in that file names
52     * are never re-used (write once).  Instead, each file is
53     * written to the next generation.  For example,
54     * segments_1, segments_2, etc.  This allows us to not use
55     * a commit lock.  See <a
56     * href="http://lucene.apache.org/java/docs/fileformats.html">file
57     * formats</a> for details.
58     */
59    public static final int FORMAT_LOCKLESS = -2;
60  
61    /** This format adds a "hasSingleNormFile" flag into each segment info.
62     * See <a href="http://issues.apache.org/jira/browse/LUCENE-756">LUCENE-756</a>
63     * for details.
64     */
65    public static final int FORMAT_SINGLE_NORM_FILE = -3;
66  
67    /** This format allows multiple segments to share a single
68     * vectors and stored fields file. */
69    public static final int FORMAT_SHARED_DOC_STORE = -4;
70  
71    /** This format adds a checksum at the end of the file to
72     *  ensure all bytes were successfully written. */
73    public static final int FORMAT_CHECKSUM = -5;
74  
75    /** This format adds the deletion count for each segment.
76     *  This way IndexWriter can efficiently report numDocs(). */
77    public static final int FORMAT_DEL_COUNT = -6;
78  
79    /** This format adds the boolean hasProx to record if any
80     *  fields in the segment store prox information (ie, have
81     *  omitTermFreqAndPositions==false) */
82    public static final int FORMAT_HAS_PROX = -7;
83  
84    /** This format adds optional commit userData (String) storage. */
85    public static final int FORMAT_USER_DATA = -8;

88     *  diagnostics storage, and switches userData to Map */

118   /**
119    * Get the generation (N) of the current segments_N file
120    * from a list of files.
121    *
122    * @param files -- array of file names to check
123    */
124   public static long getCurrentSegmentGeneration(String[] files) {
125     if (files == null) {
126       return -1;
127     }
128     long max = -1;
129     for (int i = 0; i < files.length; i++) {
130       String file = files[i];
131       if (file.startsWith(IndexFileNames.SEGMENTS) && !file.equals(IndexFileNames.SEGMENTS_GEN)) {
132         long gen = generationFromSegmentsFileName(file);
133         if (gen > max) {
134           max = gen;
135         }
136       }
137     }
138     return max;
139   }

189   /**
190    * Parse the generation off the segments file name and
191    * return it.
192    */
193   public static long generationFromSegmentsFileName(String fileName) {
194     if (fileName.equals(IndexFileNames.SEGMENTS)) {
195       return 0;
196     } else if (fileName.startsWith(IndexFileNames.SEGMENTS)) {
197       return Long.parseLong(fileName.substring(1+IndexFileNames.SEGMENTS.length()),
198                             Character.MAX_RADIX);
199     } else {
200       throw new IllegalArgumentException("fileName \"" + fileName + "\" is not a segments file");
201     }
202   }

208   public String getNextSegmentFileName() {
209     long nextGeneration;
210 
211     if (generation == -1) {
212       nextGeneration = 1;
213     } else {
214       nextGeneration = generation+1;
215     }
216     return IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS,
217                                                  "",
218                                                  nextGeneration);

221   /**
222    * Read a particular segmentFileName.  Note that this may
223    * throw an IOException if a commit is in process.
224    *
225    * @param directory -- directory containing the segments file
226    * @param segmentFileName -- segment file to load
227    * @throws CorruptIndexException if the index is corrupt
228    * @throws IOException if there is a low-level IO error
229    */
230   public final void read(Directory directory, String segmentFileName) throws CorruptIndexException, IOException {
231     boolean success = false;
232 
233     // Clear any previous segments:
234     clear();
235 
236     ChecksumIndexInput input = new ChecksumIndexInput(directory.openInput(segmentFileName));
237 
238     generation = generationFromSegmentsFileName(segmentFileName);
239 
240     lastGeneration = generation;
241 
242     try {
243       int format = input.readInt();
244       if(format < 0){     // file contains explicit format info
245         // check that it is a format we can understand
246         if (format < CURRENT_FORMAT)
247           throw new CorruptIndexException("Unknown format version: " + format);
248         version = input.readLong(); // read version
249         counter = input.readInt(); // read counter
250       }
251       else{     // file is in old format without explicit format info
252         counter = format;
253       }
254       
255       for (int i = input.readInt(); i > 0; i--) { // read segmentInfos
256         add(new SegmentInfo(directory, format, input));
257       }
258       
259       if(format >= 0){    // in old format the version number may be at the end of the file
260         if (input.getFilePointer() >= input.length())
261           version = System.currentTimeMillis(); // old file format without version number
262         else
263           version = input.readLong(); // read version
264       }
265 
266       if (format <= FORMAT_USER_DATA) {
267         if (format <= FORMAT_DIAGNOSTICS) {
268           userData = input.readStringStringMap();
269         } else if (0 != input.readByte()) {
270           userData = Collections.singletonMap("userData", input.readString());
271         } else {
272           userData = Collections.EMPTY_MAP;
273         }
274       } else {
275         userData = Collections.EMPTY_MAP;
276       }
277 
278       if (format <= FORMAT_CHECKSUM) {
279         final long checksumNow = input.getChecksum();
280         final long checksumThen = input.readLong();
281         if (checksumNow != checksumThen)
282           throw new CorruptIndexException("checksum mismatch in segments file");
283       }
284       success = true;
285     }
286     finally {
287       input.close();
288       if (!success) {
289         // Clear any segment infos we had loaded so we
290         // have a clean slate on retry:
291         clear();
292       }
293     }
294   }

296   /**
297    * This version of read uses the retry logic (for lock-less
298    * commits) to find the right segments file to load.
299    * @throws CorruptIndexException if the index is corrupt
300    * @throws IOException if there is a low-level IO error
301    */
302   public final void read(Directory directory) throws CorruptIndexException, IOException {
303 
304     generation = lastGeneration = -1;
305 
306     new FindSegmentsFile(directory) {
307 
308       protected Object doBody(String segmentFileName) throws CorruptIndexException, IOException {
309         read(directory, segmentFileName);
310         return null;
311       }
312     }.run();
313   }

315   // Only non-null after prepareCommit has been called and
316   // before finishCommit is called
317   ChecksumIndexOutput pendingSegnOutput;
318 
319   private final void write(Directory directory) throws IOException {
320 
321     String segmentFileName = getNextSegmentFileName();
322 
323     // Always advance the generation on write:
324     if (generation == -1) {
325       generation = 1;
326     } else {
327       generation++;
328     }
329 
330     ChecksumIndexOutput segnOutput = new ChecksumIndexOutput(directory.createOutput(segmentFileName));
331 
332     boolean success = false;
333 
334     try {
335       segnOutput.writeInt(CURRENT_FORMAT); // write FORMAT
336       segnOutput.writeLong(++version); // every write changes
337                                    // the index
338       segnOutput.writeInt(counter); // write counter
339       segnOutput.writeInt(size()); // write infos
340       for (int i = 0; i < size(); i++) {
341         info(i).write(segnOutput);
342       }
343       segnOutput.writeStringStringMap(userData);
344       segnOutput.prepareCommit();
345       success = true;
346       pendingSegnOutput = segnOutput;
347     } finally {
348       if (!success) {
349         // We hit an exception above; try to close the file
350         // but suppress any exception:
351         try {
352           segnOutput.close();
353         } catch (Throwable t) {
354           // Suppress so we keep throwing the original exception
355         }
356         try {
357           // Try not to leave a truncated segments_N file in
358           // the index:
359           directory.deleteFile(segmentFileName);
360         } catch (Throwable t) {
361           // Suppress so we keep throwing the original exception
362         }
363       }
364     }
365   }

394   /**
395    * Current version number from segments file.
396    * @throws CorruptIndexException if the index is corrupt
397    * @throws IOException if there is a low-level IO error
398    */
399   public static long readCurrentVersion(Directory directory)
400     throws CorruptIndexException, IOException {
401 
402     return ((Long) new FindSegmentsFile(directory) {
403         protected Object doBody(String segmentFileName) throws CorruptIndexException, IOException {
404 
405           IndexInput input = directory.openInput(segmentFileName);
406 
407           int format = 0;
408           long version = 0;
409           try {
410             format = input.readInt();
411             if(format < 0){
412               if (format < CURRENT_FORMAT)
413                 throw new CorruptIndexException("Unknown format version: " + format);
414               version = input.readLong(); // read version
415             }
416           }
417           finally {
418             input.close();
419           }
420      
421           if(format < 0)
422             return new Long(version);
423 
424           // We cannot be sure about the format of the file.
425           // Therefore we have to read the whole file and cannot simply seek to the version entry.
426           SegmentInfos sis = new SegmentInfos();
427           sis.read(directory, segmentFileName);
428           return new Long(sis.getVersion());
429         }
430       }.run()).longValue();
431   }

520   /**
521    * Utility class for executing code that needs to do
522    * something with the current segments file.  This is
523    * necessary with lock-less commits because from the time
524    * you locate the current segments file name, until you
525    * actually open it, read its contents, or check modified
526    * time, etc., it could have been deleted due to a writer
527    * commit finishing.
528    */
529   public abstract static class FindSegmentsFile {
530     
531     final Directory directory;
532 
533     public FindSegmentsFile(Directory directory) {
534       this.directory = directory;
535     }
536 
537     public Object run() throws CorruptIndexException, IOException {
538       return run(null);
539     }
540     
541     public Object run(IndexCommit commit) throws CorruptIndexException, IOException {
542       if (commit != null) {
543         if (directory != commit.getDirectory())
544           throw new IOException("the specified commit does not match the specified Directory");
545         return doBody(commit.getSegmentsFileName());
546       }
547 
548       String segmentFileName = null;
549       long lastGen = -1;
550       long gen = 0;
551       int genLookaheadCount = 0;
552       IOException exc = null;
553       boolean retry = false;
554 
555       int method = 0;
556 
557       // Loop until we succeed in calling doBody() without
558       // hitting an IOException.  An IOException most likely
559       // means a commit was in process and has finished, in
560       // the time it took us to load the now-old infos files
561       // (and segments files).  It's also possible it's a
562       // true error (corrupt index).  To distinguish these,
563       // on each retry we must see "forward progress" on
564       // which generation we are trying to load.  If we
565       // don't, then the original error is real and we throw
566       // it.
567       
568       // We have three methods for determining the current
569       // generation.  We try the first two in parallel, and
570       // fall back to the third when necessary.
571 
572       while(true) {
573 
574         if (0 == method) {
575 
576           // Method 1: list the directory and use the highest
577           // segments_N file.  This method works well as long
578           // as there is no stale caching on the directory
579           // contents (NOTE: NFS clients often have such stale
580           // caching):
581           String[] files = null;
582 
583           long genA = -1;
584 
585           files = directory.listAll();
586           
587           if (files != null)
588             genA = getCurrentSegmentGeneration(files);
589 
590           message("directory listing genA=" + genA);
591 
592           // Method 2: open segments.gen and read its
593           // contents.  Then we take the larger of the two
594           // gen's.  This way, if either approach is hitting
595           // a stale cache (NFS) we have a better chance of
596           // getting the right generation.
597           long genB = -1;
598           for(int i=0;i<defaultGenFileRetryCount;i++) {
599             IndexInput genInput = null;
600             try {
601               genInput = directory.openInput(IndexFileNames.SEGMENTS_GEN);
602             } catch (FileNotFoundException e) {
603               message("segments.gen open: FileNotFoundException " + e);
604               break;
605             } catch (IOException e) {
606               message("segments.gen open: IOException " + e);
607             }
608   
609             if (genInput != null) {
610               try {
611                 int version = genInput.readInt();
612                 if (version == FORMAT_LOCKLESS) {
613                   long gen0 = genInput.readLong();
614                   long gen1 = genInput.readLong();
615                   message("fallback check: " + gen0 + "; " + gen1);
616                   if (gen0 == gen1) {
617                     // The file is consistent.
618                     genB = gen0;
619                     break;
620                   }
621                 }
622               } catch (IOException err2) {
623                 // will retry
624               } finally {
625                 genInput.close();
626               }
627             }
628             try {
629               Thread.sleep(defaultGenFileRetryPauseMsec);
630             } catch (InterruptedException ie) {
631               // In 3.0 we will change this to throw
632               // InterruptedException instead
633               Thread.currentThread().interrupt();
634               throw new RuntimeException(ie);
635             }
636           }
637 
638           message(IndexFileNames.SEGMENTS_GEN + " check: genB=" + genB);
639 
640           // Pick the larger of the two gen's:
641           if (genA > genB)
642             gen = genA;
643           else
644             gen = genB;
645           
646           if (gen == -1) {
647             // Neither approach found a generation
648             String s;
649             if (files != null) {
650               s = "";
651               for(int i=0;i<files.length;i++)
652                 s += " " + files[i];
653             } else
654               s = " null";
655             throw new FileNotFoundException("no segments* file found in " + directory + ": files:" + s);
656           }
657         }
658 
659         // Third method (fallback if first & second methods
660         // are not reliable): since both directory cache and
661         // file contents cache seem to be stale, just
662         // advance the generation.
663         if (1 == method || (0 == method && lastGen == gen && retry)) {
664 
665           method = 1;
666 
667           if (genLookaheadCount < defaultGenLookaheadCount) {
668             gen++;
669             genLookaheadCount++;
670             message("look ahead increment gen to " + gen);
671           }
672         }
673 
674         if (lastGen == gen) {
675 
676           // This means we're about to try the same
677           // segments_N last tried.  This is allowed,
678           // exactly once, because writer could have been in
679           // the process of writing segments_N last time.
680 
681           if (retry) {
682             // OK, we've tried the same segments_N file
683             // twice in a row, so this must be a real
684             // error.  We throw the original exception we
685             // got.
686             throw exc;
687           } else {
688             retry = true;
689           }
690 
691         } else if (0 == method) {
692           // Segment file has advanced since our last loop, so
693           // reset retry:
694           retry = false;
695         }
696 
697         lastGen = gen;
698 
699         segmentFileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS,
700                                                                 "",
701                                                                 gen);
702 
703         try {
704           Object v = doBody(segmentFileName);
705           if (exc != null) {
706             message("success on " + segmentFileName);
707           }
708           return v;
709         } catch (IOException err) {
710 
711           // Save the original root cause:
712           if (exc == null) {
713             exc = err;
714           }
715 
716           message("primary Exception on '" + segmentFileName + "': " + err + "'; will retry: retry=" + retry + "; gen = " + gen);
717 
718           if (!retry && gen > 1) {
719 
720             // This is our first time trying this segments
721             // file (because retry is false), and, there is
722             // possibly a segments_(N-1) (because gen > 1).
723             // So, check if the segments_(N-1) exists and
724             // try it if so:
725             String prevSegmentFileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS,
726                                                                                "",
727                                                                                gen-1);
728 
729             final boolean prevExists;
730             prevExists = directory.fileExists(prevSegmentFileName);
731 
732             if (prevExists) {
733               message("fallback to prior segment file '" + prevSegmentFileName + "'");
734               try {
735                 Object v = doBody(prevSegmentFileName);
736                 if (exc != null) {
737                   message("success on fallback " + prevSegmentFileName);
738                 }
739                 return v;
740               } catch (IOException err2) {
741                 message("secondary Exception on '" + prevSegmentFileName + "': " + err2 + "'; will retry");
742               }
743             }
744           }
745         }
746       }
747     }

749     /**
750      * Subclass must implement this.  The assumption is an
751      * IOException will be thrown if something goes wrong
752      * during the processing that could have been caused by
753      * a writer committing.
754      */
755     protected abstract Object doBody(String segmentFileName) throws CorruptIndexException, IOException;

777   final void rollbackCommit(Directory dir) throws IOException {
778     if (pendingSegnOutput != null) {
779       try {
780         pendingSegnOutput.close();
781       } catch (Throwable t) {
782         // Suppress so we keep throwing the original exception
783         // in our caller
784       }
785 
786       // Must carefully compute fileName from "generation"
787       // since lastGeneration isn't incremented:
788       try {
789         final String segmentFileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS,
790                                                                              "",
791                                                                              generation);
792         dir.deleteFile(segmentFileName);
793       } catch (Throwable t) {
794         // Suppress so we keep throwing the original exception
795         // in our caller
796       }
797       pendingSegnOutput = null;
798     }
799   }

801   /** Call this to start a commit.  This writes the new
802    *  segments file, but writes an invalid checksum at the
803    *  end, so that it is not visible to readers.  Once this
804    *  is called you must call {@link #finishCommit} to complete
805    *  the commit or {@link #rollbackCommit} to abort it. */
806   final void prepareCommit(Directory dir) throws IOException {
807     if (pendingSegnOutput != null)
808       throw new IllegalStateException("prepareCommit was already called");
809     write(dir);
810   }

812   /** Returns all file names referenced by SegmentInfo
813    *  instances matching the provided Directory (ie files
814    *  associated with any "external" segments are skipped).
815    *  The returned collection is recomputed on each
816    *  invocation.  */
817   public Collection files(Directory dir, boolean includeSegmentsFile) throws IOException {
818     HashSet files = new HashSet();
819     if (includeSegmentsFile) {
820       files.add(getCurrentSegmentFileName());
821     }
822     final int size = size();
823     for(int i=0;i<size;i++) {
824       final SegmentInfo info = info(i);
825       if (info.dir == dir) {
826         files.addAll(info(i).files());
827       }
828     }
829     return files;
830   }
831 
832   final void finishCommit(Directory dir) throws IOException {
833     if (pendingSegnOutput == null)
834       throw new IllegalStateException("prepareCommit was not called");
835     boolean success = false;
836     try {
837       pendingSegnOutput.finishCommit();
838       pendingSegnOutput.close();
839       pendingSegnOutput = null;
840       success = true;
841     } finally {
842       if (!success)
843         rollbackCommit(dir);
844     }
845 
846     // NOTE: if we crash here, we have left a segments_N
847     // file in the directory in a possibly corrupt state (if
848     // some bytes made it to stable storage and others
849     // didn't).  But, the segments_N file includes checksum
850     // at the end, which should catch this case.  So when a
851     // reader tries to read it, it will throw a
852     // CorruptIndexException, which should cause the retry
853     // logic in SegmentInfos to kick in and load the last
854     // good (previous) segments_N-1 file.
855 
856     final String fileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS,
857                                                                   "",
858                                                                   generation);
859     success = false;
860     try {
861       dir.sync(fileName);
862       success = true;
863     } finally {
864       if (!success) {
865         try {
866           dir.deleteFile(fileName);
867         } catch (Throwable t) {
868           // Suppress so we keep throwing the original exception
869         }
870       }
871     }
872 
873     lastGeneration = generation;
874 
875     try {
876       IndexOutput genOutput = dir.createOutput(IndexFileNames.SEGMENTS_GEN);
877       try {
878         genOutput.writeInt(FORMAT_LOCKLESS);
879         genOutput.writeLong(generation);
880         genOutput.writeLong(generation);
881       } finally {
882         genOutput.close();
883       }
884     } catch (Throwable t) {
885       // It's OK if we fail to write this file since it's
886       // used only as one of the retry fallbacks.
887     }
888   }
889 
890   /** Writes & syncs to the Directory dir, taking care to
891    *  remove the segments file on exception */
892   final void commit(Directory dir) throws IOException {
893     prepareCommit(dir);
894     finishCommit(dir);
895   }

897   public synchronized String segString(Directory directory) {
898     StringBuffer buffer = new StringBuffer();
899     final int count = size();
900     for(int i = 0; i < count; i++) {
901       if (i > 0) {
902         buffer.append(' ');
903       }
904       final SegmentInfo info = info(i);
905       buffer.append(info.segString(directory));
906       if (info.dir != directory)
907         buffer.append("**");
908     }
909     return buffer.toString();
910   }
