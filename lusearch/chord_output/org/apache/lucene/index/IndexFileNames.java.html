165   /**
166    * Computes the full file name from base, extension and
167    * generation.  If the generation is -1, the file name is
168    * null.  If it's 0, the file name is <base><extension>.
169    * If it's > 0, the file name is <base>_<generation><extension>.
170    *
171    * @param base -- main part of the file name
172    * @param extension -- extension of the filename (including .)
173    * @param gen -- generation
174    */
175   static final String fileNameFromGeneration(String base, String extension, long gen) {
176     if (gen == SegmentInfo.NO) {
177       return null;
178     } else if (gen == SegmentInfo.WITHOUT_GEN) {
179       return base + extension;
180     } else {
181       return base + "_" + Long.toString(gen, Character.MAX_RADIX) + extension;
182     }
183   }

185   /**
186    * Returns true if the provided filename is one of the doc
187    * store files (ends with an extension in
188    * STORE_INDEX_EXTENSIONS).
189    */
190   static final boolean isDocStoreFile(String fileName) {
191     if (fileName.endsWith(COMPOUND_FILE_STORE_EXTENSION))
192       return true;
193     for(int i=0;i<STORE_INDEX_EXTENSIONS.length;i++)
194       if (fileName.endsWith(STORE_INDEX_EXTENSIONS[i]))
195         return true;
196     return false;
197   }

200     return segmentName + "." + ext;
