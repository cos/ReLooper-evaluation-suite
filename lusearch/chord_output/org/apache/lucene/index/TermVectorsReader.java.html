32    // NOTE: if you make a new format, it must be larger than
33    // the current format
34    static final int FORMAT_VERSION = 2;
35  
36    // Changes to speed up bulk merging of term vectors:
37    static final int FORMAT_VERSION2 = 3;
38  
39    // Changed strings to UTF8 with length-in-bytes not length-in-chars
40    static final int FORMAT_UTF8_LENGTH_IN_BYTES = 4;
41  
42    // NOTE: always change this if you switch to a new format!
43    static final int FORMAT_CURRENT = FORMAT_UTF8_LENGTH_IN_BYTES;
44  
45    //The size in bytes that the FORMAT_VERSION will take up at the beginning of each file 
46    static final int FORMAT_SIZE = 4;
47  
48    static final byte STORE_POSITIONS_WITH_TERMVECTOR = 0x1;
49    static final byte STORE_OFFSET_WITH_TERMVECTOR = 0x2;

75    TermVectorsReader(Directory d, String segment, FieldInfos fieldInfos, int readBufferSize, int docStoreOffset, int size)
76      throws CorruptIndexException, IOException {
77      boolean success = false;
78  
79      try {
80        if (d.fileExists(segment + "." + IndexFileNames.VECTORS_INDEX_EXTENSION)) {
81          tvx = d.openInput(segment + "." + IndexFileNames.VECTORS_INDEX_EXTENSION, readBufferSize);
82          format = checkValidFormat(tvx);
83          tvd = d.openInput(segment + "." + IndexFileNames.VECTORS_DOCUMENTS_EXTENSION, readBufferSize);
84          final int tvdFormat = checkValidFormat(tvd);
85          tvf = d.openInput(segment + "." + IndexFileNames.VECTORS_FIELDS_EXTENSION, readBufferSize);
86          final int tvfFormat = checkValidFormat(tvf);
87  
88          assert format == tvdFormat;
89          assert format == tvfFormat;
90  
91          if (format >= FORMAT_VERSION2) {
92            assert (tvx.length()-FORMAT_SIZE) % 16 == 0;
93            numTotalDocs = (int) (tvx.length() >> 4);
94          } else {
95            assert (tvx.length()-FORMAT_SIZE) % 8 == 0;
96            numTotalDocs = (int) (tvx.length() >> 3);
97          }
98  
99          if (-1 == docStoreOffset) {
100           this.docStoreOffset = 0;
101           this.size = numTotalDocs;
102           assert size == 0 || numTotalDocs == size;
103         } else {
104           this.docStoreOffset = docStoreOffset;
105           this.size = size;
106           // Verify the file is long enough to hold all of our
107           // docs
108           assert numTotalDocs >= size + docStoreOffset: "numTotalDocs=" + numTotalDocs + " size=" + size + " docStoreOffset=" + docStoreOffset;
109         }
110       } else
111         format = 0;
112 
113       this.fieldInfos = fieldInfos;
114       success = true;
115     } finally {
116       // With lock-less commits, it's entirely possible (and
117       // fine) to hit a FileNotFound exception above. In
118       // this case, we want to explicitly close any subset
119       // of things that were opened so that we don't have to
120       // wait for a GC to do so.
121       if (!success) {
122         close();
123       }
124     }
125   }

137   final private void seekTvx(final int docNum) throws IOException {
138     if (format < FORMAT_VERSION2)
139       tvx.seek((docNum + docStoreOffset) * 8L + FORMAT_SIZE);
140     else
141       tvx.seek((docNum + docStoreOffset) * 16L + FORMAT_SIZE);
142   }

144   boolean canReadRawDocs() {
145     return format >= FORMAT_UTF8_LENGTH_IN_BYTES;
146   }

148   /** Retrieve the length (in bytes) of the tvd and tvf
149    *  entries for the next numDocs starting with
150    *  startDocID.  This is used for bulk copying when
151    *  merging segments, if the field numbers are
152    *  congruent.  Once this returns, the tvf & tvd streams
153    *  are seeked to the startDocID. */
154   final void rawDocs(int[] tvdLengths, int[] tvfLengths, int startDocID, int numDocs) throws IOException {
155 
156     if (tvx == null) {
157       Arrays.fill(tvdLengths, 0);
158       Arrays.fill(tvfLengths, 0);
159       return;
160     }
161 
162     // SegmentMerger calls canReadRawDocs() first and should
163     // not call us if that returns false.
164     if (format < FORMAT_VERSION2)
165       throw new IllegalStateException("cannot read raw docs with older term vector formats");
166 
167     seekTvx(startDocID);
168 
169     long tvdPosition = tvx.readLong();
170     tvd.seek(tvdPosition);
171 
172     long tvfPosition = tvx.readLong();
173     tvf.seek(tvfPosition);
174 
175     long lastTvdPosition = tvdPosition;
176     long lastTvfPosition = tvfPosition;
177 
178     int count = 0;
179     while (count < numDocs) {
180       final int docID = docStoreOffset + startDocID + count + 1;
181       assert docID <= numTotalDocs;
182       if (docID < numTotalDocs)  {
183         tvdPosition = tvx.readLong();
184         tvfPosition = tvx.readLong();
185       } else {
186         tvdPosition = tvd.length();
187         tvfPosition = tvf.length();
188         assert count == numDocs-1;
189       }
190       tvdLengths[count] = (int) (tvdPosition-lastTvdPosition);
191       tvfLengths[count] = (int) (tvfPosition-lastTvfPosition);
192       count++;
193       lastTvdPosition = tvdPosition;
194       lastTvfPosition = tvfPosition;
195     }
196   }

198   private int checkValidFormat(IndexInput in) throws CorruptIndexException, IOException
199   {
200     int format = in.readInt();
201     if (format > FORMAT_CURRENT) {
202       throw new CorruptIndexException("Incompatible format version: " + format + " expected " 
203                                       + FORMAT_CURRENT + " or less");
204     }
205     return format;
206   }

208   void close() throws IOException {
209     // make all effort to close up. Keep the first exception
210     // and throw it as a new one.
211     IOException keep = null;
212     if (tvx != null) try { tvx.close(); } catch (IOException e) { if (keep == null) keep = e; }
213     if (tvd != null) try { tvd.close(); } catch (IOException e) { if (keep == null) keep = e; }
214     if (tvf  != null) try {  tvf.close(); } catch (IOException e) { if (keep == null) keep = e; }
215     if (keep != null) throw (IOException) keep.fillInStackTrace();
216   }

226   public void get(int docNum, String field, TermVectorMapper mapper) throws IOException {
227     if (tvx != null) {
228       int fieldNumber = fieldInfos.fieldNumber(field);
229       //We need to account for the FORMAT_SIZE at when seeking in the tvx
230       //We don't need to do this in other seeks because we already have the
231       // file pointer
232       //that was written in another file
233       seekTvx(docNum);
234       //System.out.println("TVX Pointer: " + tvx.getFilePointer());
235       long tvdPosition = tvx.readLong();
236 
237       tvd.seek(tvdPosition);
238       int fieldCount = tvd.readVInt();
239       //System.out.println("Num Fields: " + fieldCount);
240       // There are only a few fields per document. We opt for a full scan
241       // rather then requiring that they be ordered. We need to read through
242       // all of the fields anyway to get to the tvf pointers.
243       int number = 0;
244       int found = -1;
245       for (int i = 0; i < fieldCount; i++) {
246         if (format >= FORMAT_VERSION)
247           number = tvd.readVInt();
248         else
249           number += tvd.readVInt();
250 
251         if (number == fieldNumber)
252           found = i;
253       }
254 
255       // This field, although valid in the segment, was not found in this
256       // document
257       if (found != -1) {
258         // Compute position in the tvf file
259         long position;
260         if (format >= FORMAT_VERSION2)
261           position = tvx.readLong();
262         else
263           position = tvd.readVLong();
264         for (int i = 1; i <= found; i++)
265           position += tvd.readVLong();
266 
267         mapper.setDocumentNumber(docNum);
268         readTermVector(field, position, mapper);
269       } else {
270         //System.out.println("Fieldable not found");
271       }
272     } else {
273       //System.out.println("No tvx file");
274     }
275   }

296   final private String[] readFields(int fieldCount) throws IOException {
297     int number = 0;
298     String[] fields = new String[fieldCount];
299 
300     for (int i = 0; i < fieldCount; i++) {
301       if (format >= FORMAT_VERSION)
302         number = tvd.readVInt();
303       else
304         number += tvd.readVInt();
305 
306       fields[i] = fieldInfos.fieldName(number);
307     }
308 
309     return fields;
310   }

312   // Reads the long[] offsets into TVF; you have to pre-seek
313   // tvx/tvd to the right point
314   final private long[] readTvfPointers(int fieldCount) throws IOException {
315     // Compute position in the tvf file
316     long position;
317     if (format >= FORMAT_VERSION2)
318       position = tvx.readLong();
319     else
320       position = tvd.readVLong();
321 
322     long[] tvfPointers = new long[fieldCount];
323     tvfPointers[0] = position;
324 
325     for (int i = 1; i < fieldCount; i++) {
326       position += tvd.readVLong();
327       tvfPointers[i] = position;
328     }
329 
330     return tvfPointers;
331   }

333   /**
334    * Return all term vectors stored for this document or null if the could not be read in.
335    * 
336    * @param docNum The document number to retrieve the vector for
337    * @return All term frequency vectors
338    * @throws IOException if there is an error reading the term vector files 
339    */
340   TermFreqVector[] get(int docNum) throws IOException {
341     TermFreqVector[] result = null;
342     if (tvx != null) {
343       //We need to offset by
344       seekTvx(docNum);
345       long tvdPosition = tvx.readLong();
346 
347       tvd.seek(tvdPosition);
348       int fieldCount = tvd.readVInt();
349 
350       // No fields are vectorized for this document
351       if (fieldCount != 0) {
352         final String[] fields = readFields(fieldCount);
353         final long[] tvfPointers = readTvfPointers(fieldCount);
354         result = readTermVectors(docNum, fields, tvfPointers);
355       }
356     } else {
357       //System.out.println("No tvx file");
358     }
359     return result;
360   }

364     if (tvx != null) {
365       //We need to offset by

367       seekTvx(docNumber);
368       long tvdPosition = tvx.readLong();
369

376         final long[] tvfPointers = readTvfPointers(fieldCount);

386   private SegmentTermVector[] readTermVectors(int docNum, String fields[], long tvfPointers[])
387           throws IOException {
388     SegmentTermVector res[] = new SegmentTermVector[fields.length];
389     for (int i = 0; i < fields.length; i++) {
390       ParallelArrayTermVectorMapper mapper = new ParallelArrayTermVectorMapper();
391       mapper.setDocumentNumber(docNum);
392       readTermVector(fields[i], tvfPointers[i], mapper);
393       res[i] = (SegmentTermVector) mapper.materializeVector();
394     }
395     return res;
396   }

398   private void readTermVectors(String fields[], long tvfPointers[], TermVectorMapper mapper)
399           throws IOException {
400     for (int i = 0; i < fields.length; i++) {
401       readTermVector(fields[i], tvfPointers[i], mapper);
402     }
403   }

406   /**
407    * 
408    * @param field The field to read in
409    * @param tvfPointer The pointer within the tvf file where we should start reading
410    * @param mapper The mapper used to map the TermVector
411    * @throws IOException
412    */ 
413   private void readTermVector(String field, long tvfPointer, TermVectorMapper mapper)
414           throws IOException {
415 
416     // Now read the data from specified position
417     //We don't need to offset by the FORMAT here since the pointer already includes the offset
418     tvf.seek(tvfPointer);
419 
420     int numTerms = tvf.readVInt();
421     //System.out.println("Num Terms: " + numTerms);
422     // If no terms - return a constant empty termvector. However, this should never occur!
423     if (numTerms == 0) 
424       return;
425     
426     boolean storePositions;
427     boolean storeOffsets;
428     
429     if (format >= FORMAT_VERSION){
430       byte bits = tvf.readByte();
431       storePositions = (bits & STORE_POSITIONS_WITH_TERMVECTOR) != 0;
432       storeOffsets = (bits & STORE_OFFSET_WITH_TERMVECTOR) != 0;
433     }
434     else{
435       tvf.readVInt();
436       storePositions = false;
437       storeOffsets = false;
438     }
439     mapper.setExpectations(field, numTerms, storeOffsets, storePositions);
440     int start = 0;
441     int deltaLength = 0;
442     int totalLength = 0;
443     byte[] byteBuffer;
444     char[] charBuffer;
445     final boolean preUTF8 = format < FORMAT_UTF8_LENGTH_IN_BYTES;
446 
447     // init the buffers
448     if (preUTF8) {
449       charBuffer = new char[10];
450       byteBuffer = null;
451     } else {
452       charBuffer = null;
453       byteBuffer = new byte[20];
454     }
455 
456     for (int i = 0; i < numTerms; i++) {
457       start = tvf.readVInt();
458       deltaLength = tvf.readVInt();
459       totalLength = start + deltaLength;
460 
461       final String term;
462       
463       if (preUTF8) {
464         // Term stored as java chars
465         if (charBuffer.length < totalLength) {
466           char[] newCharBuffer = new char[(int) (1.5*totalLength)];
467           System.arraycopy(charBuffer, 0, newCharBuffer, 0, start);
468           charBuffer = newCharBuffer;
469         }
470         tvf.readChars(charBuffer, start, deltaLength);
471         term = new String(charBuffer, 0, totalLength);
472       } else {
473         // Term stored as utf8 bytes
474         if (byteBuffer.length < totalLength) {
475           byte[] newByteBuffer = new byte[(int) (1.5*totalLength)];
476           System.arraycopy(byteBuffer, 0, newByteBuffer, 0, start);
477           byteBuffer = newByteBuffer;
478         }
479         tvf.readBytes(byteBuffer, start, deltaLength);
480         term = new String(byteBuffer, 0, totalLength, "UTF-8");
481       }
482       int freq = tvf.readVInt();
483       int [] positions = null;
484       if (storePositions) { //read in the positions
485         //does the mapper even care about positions?
486         if (mapper.isIgnoringPositions() == false) {
487           positions = new int[freq];
488           int prevPosition = 0;
489           for (int j = 0; j < freq; j++)
490           {
491             positions[j] = prevPosition + tvf.readVInt();
492             prevPosition = positions[j];
493           }
494         } else {
495           //we need to skip over the positions.  Since these are VInts, I don't believe there is anyway to know for sure how far to skip
496           //
497           for (int j = 0; j < freq; j++)
498           {
499             tvf.readVInt();
500           }
501         }
502       }
503       TermVectorOffsetInfo[] offsets = null;
504       if (storeOffsets) {
505         //does the mapper even care about offsets?
506         if (mapper.isIgnoringOffsets() == false) {
507           offsets = new TermVectorOffsetInfo[freq];
508           int prevOffset = 0;
509           for (int j = 0; j < freq; j++) {
510             int startOffset = prevOffset + tvf.readVInt();
511             int endOffset = startOffset + tvf.readVInt();
512             offsets[j] = new TermVectorOffsetInfo(startOffset, endOffset);
513             prevOffset = endOffset;
514           }
515         } else {
516           for (int j = 0; j < freq; j++){
517             tvf.readVInt();
518             tvf.readVInt();
519           }
520         }
521       }
522       mapper.map(term, freq, offsets, positions);
523     }
524   }

526   protected Object clone() throws CloneNotSupportedException {
527     
528     final TermVectorsReader clone = (TermVectorsReader) super.clone();
529 
530     // These are null when a TermVectorsReader was created
531     // on a segment that did not have term vectors saved
532     if (tvx != null && tvd != null && tvf != null) {
533       clone.tvx = (IndexInput) tvx.clone();
534       clone.tvd = (IndexInput) tvd.clone();
535       clone.tvf = (IndexInput) tvf.clone();
536     }
537     
538     return clone;
539   }

543 /**
544  * Models the existing parallel array structure
545  */
546 class ParallelArrayTermVectorMapper extends TermVectorMapper
547 {
548 
549   private String[] terms;
550   private int[] termFreqs;
551   private int positions[][];
552   private TermVectorOffsetInfo offsets[][];
553   private int currentPosition;
554   private boolean storingOffsets;
555   private boolean storingPositions;
556   private String field;
557 
558   public void setExpectations(String field, int numTerms, boolean storeOffsets, boolean storePositions) {
559     this.field = field;
560     terms = new String[numTerms];
561     termFreqs = new int[numTerms];
562     this.storingOffsets = storeOffsets;
563     this.storingPositions = storePositions;
564     if(storePositions)
565       this.positions = new int[numTerms][];
566     if(storeOffsets)
567       this.offsets = new TermVectorOffsetInfo[numTerms][];
568   }
569 
570   public void map(String term, int frequency, TermVectorOffsetInfo[] offsets, int[] positions) {
571     terms[currentPosition] = term;
572     termFreqs[currentPosition] = frequency;
573     if (storingOffsets)
574     {
575       this.offsets[currentPosition] = offsets;
576     }
577     if (storingPositions)
578     {
579       this.positions[currentPosition] = positions; 
580     }
581     currentPosition++;
582   }
583 
584   /**
585    * Construct the vector
586    * @return The {@link TermFreqVector} based on the mappings.
587    */
588   public TermFreqVector materializeVector() {
589     SegmentTermVector tv = null;
590     if (field != null && terms != null) {
591       if (storingPositions || storingOffsets) {
592         tv = new SegmentTermPositionVector(field, terms, termFreqs, positions, offsets);
593       } else {
594         tv = new SegmentTermVector(field, terms, termFreqs);
595       }
596     }
597     return tv;
598   }
