18  package org.apache.lucene.index;
19  
20  import java.io.IOException;
21  
22  /**
23   * This exception is thrown when an {@link IndexReader}
24   * tries to make changes to the index (via {@link
25   * IndexReader#deleteDocument}, {@link
26   * IndexReader#undeleteAll} or {@link IndexReader#setNorm})
27   * but changes have already been committed to the index
28   * since this reader was instantiated.  When this happens
29   * you must open a new reader on the current index to make
30   * the changes.
31   */
32  public class StaleReaderException extends IOException {
33    public StaleReaderException(String message) {
34      super(message);
35    }
36  }
