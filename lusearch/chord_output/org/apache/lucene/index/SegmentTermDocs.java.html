20  import java.io.IOException;
21  import org.apache.lucene.util.BitVector;
22  import org.apache.lucene.store.IndexInput;
23  
24  class SegmentTermDocs implements TermDocs {
25    protected SegmentReader parent;
26    protected IndexInput freqStream;
27    protected int count;
28    protected int df;
29    protected BitVector deletedDocs;
30    int doc = 0;
31    int freq;
32  
33    private int skipInterval;
34    private int maxSkipLevels;
35    private DefaultSkipListReader skipListReader;
36    
37    private long freqBasePointer;
38    private long proxBasePointer;
39  
40    private long skipPointer;
41    private boolean haveSkipped;
42    
43    protected boolean currentFieldStoresPayloads;
44    protected boolean currentFieldOmitTermFreqAndPositions;
45    
46    protected SegmentTermDocs(SegmentReader parent) {
47      this.parent = parent;
48      this.freqStream = (IndexInput) parent.core.freqStream.clone();
49      synchronized (parent) {
50        this.deletedDocs = parent.deletedDocs;
51      }
52      this.skipInterval = parent.core.getTermsReader().getSkipInterval();
53      this.maxSkipLevels = parent.core.getTermsReader().getMaxSkipLevels();
54    }
55  
56    public void seek(Term term) throws IOException {
57      TermInfo ti = parent.core.getTermsReader().get(term);
58      seek(ti, term);
59    }
60  
61    public void seek(TermEnum termEnum) throws IOException {
62      TermInfo ti;
63      Term term;
64      
65      // use comparison of fieldinfos to verify that termEnum belongs to the same segment as this SegmentTermDocs
66      if (termEnum instanceof SegmentTermEnum && ((SegmentTermEnum) termEnum).fieldInfos == parent.core.fieldInfos) {        // optimized case
67        SegmentTermEnum segmentTermEnum = ((SegmentTermEnum) termEnum);
68        term = segmentTermEnum.term();
69        ti = segmentTermEnum.termInfo();
70      } else  {                                         // punt case
71        term = termEnum.term();
72        ti = parent.core.getTermsReader().get(term);
73      }
74      
75      seek(ti, term);
76    }
77  
78    void seek(TermInfo ti, Term term) throws IOException {
79      count = 0;
80      FieldInfo fi = parent.core.fieldInfos.fieldInfo(term.field);
81      currentFieldOmitTermFreqAndPositions = (fi != null) ? fi.omitTermFreqAndPositions : false;
82      currentFieldStoresPayloads = (fi != null) ? fi.storePayloads : false;
83      if (ti == null) {
84        df = 0;
85      } else {
86        df = ti.docFreq;
87        doc = 0;
88        freqBasePointer = ti.freqPointer;
89        proxBasePointer = ti.proxPointer;
90        skipPointer = freqBasePointer + ti.skipOffset;
91        freqStream.seek(freqBasePointer);
92        haveSkipped = false;
93      }
94    }
95  
96    public void close() throws IOException {
97      freqStream.close();
98      if (skipListReader != null)
99        skipListReader.close();
100   }
101 
102   public final int doc() { return doc; }
103   public final int freq() { return freq; }
104 
105   protected void skippingDoc() throws IOException {
106   }
107 
108   public boolean next() throws IOException {
109     while (true) {
110       if (count == df)
111         return false;
112       final int docCode = freqStream.readVInt();
113       
114       if (currentFieldOmitTermFreqAndPositions) {
115         doc += docCode;
116         freq = 1;
117       } else {
118         doc += docCode >>> 1;       // shift off low bit
119         if ((docCode & 1) != 0)       // if low bit is set
120           freq = 1;         // freq is one
121         else
122           freq = freqStream.readVInt();     // else read freq
123       }
124       
125       count++;
126 
127       if (deletedDocs == null || !deletedDocs.get(doc))
128         break;
129       skippingDoc();
130     }
131     return true;
132   }
133 
134   /** Optimized implementation. */
135   public int read(final int[] docs, final int[] freqs)
136           throws IOException {
137     final int length = docs.length;
138     if (currentFieldOmitTermFreqAndPositions) {
139       return readNoTf(docs, freqs, length);
140     } else {
141       int i = 0;
142       while (i < length && count < df) {
143         // manually inlined call to next() for speed
144         final int docCode = freqStream.readVInt();
145         doc += docCode >>> 1;       // shift off low bit
146         if ((docCode & 1) != 0)       // if low bit is set
147           freq = 1;         // freq is one
148         else
149           freq = freqStream.readVInt();     // else read freq
150         count++;
151 
152         if (deletedDocs == null || !deletedDocs.get(doc)) {
153           docs[i] = doc;
154           freqs[i] = freq;
155           ++i;
156         }
157       }
158       return i;
159     }
160   }
161 
162   private final int readNoTf(final int[] docs, final int[] freqs, final int length) throws IOException {
163     int i = 0;
164     while (i < length && count < df) {
165       // manually inlined call to next() for speed
166       doc += freqStream.readVInt();       
167       count++;
168 
169       if (deletedDocs == null || !deletedDocs.get(doc)) {
170         docs[i] = doc;
171         // Hardware freq to 1 when term freqs were not
172         // stored in the index
173         freqs[i] = 1;
174         ++i;
175       }
176     }
177     return i;
178   }
179  
180   
181   /** Overridden by SegmentTermPositions to skip in prox stream. */
182   protected void skipProx(long proxPointer, int payloadLength) throws IOException {}
183 
184   /** Optimized implementation. */
185   public boolean skipTo(int target) throws IOException {
186     if (df >= skipInterval) {                      // optimized case
187       if (skipListReader == null)
188         skipListReader = new DefaultSkipListReader((IndexInput) freqStream.clone(), maxSkipLevels, skipInterval); // lazily clone
189 
190       if (!haveSkipped) {                          // lazily initialize skip stream
191         skipListReader.init(skipPointer, freqBasePointer, proxBasePointer, df, currentFieldStoresPayloads);
192         haveSkipped = true;
193       }
194 
195       int newCount = skipListReader.skipTo(target); 
196       if (newCount > count) {
197         freqStream.seek(skipListReader.getFreqPointer());
198         skipProx(skipListReader.getProxPointer(), skipListReader.getPayloadLength());
199 
200         doc = skipListReader.getDoc();
201         count = newCount;
202       }      
203     }
204 
205     // done skipping, now just scan
206     do {
207       if (!next())
208         return false;
209     } while (target > doc);
210     return true;
211   }
212 }
