20  import java.io.IOException;
21  
22  import org.apache.lucene.analysis.TokenStream;
23  import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
24  import org.apache.lucene.analysis.tokenattributes.TermAttribute;
25  
26  /** This is a DocFieldConsumer that inverts each field,
27   *  separately, from a Document, and accepts a
28   *  InvertedTermsConsumer to process those terms. */
29  
30  final class DocInverterPerThread extends DocFieldConsumerPerThread {
31    final DocInverter docInverter;
32    final InvertedDocConsumerPerThread consumer;
33    final InvertedDocEndConsumerPerThread endConsumer;
34    //TODO: change to SingleTokenTokenStream after Token was removed
35    final SingleTokenTokenStream singleTokenTokenStream = new SingleTokenTokenStream();
36    
37    static class SingleTokenTokenStream extends TokenStream {
38      TermAttribute termAttribute;
39      OffsetAttribute offsetAttribute;
40      
41      SingleTokenTokenStream() {
42        termAttribute = (TermAttribute) addAttribute(TermAttribute.class);
43        offsetAttribute = (OffsetAttribute) addAttribute(OffsetAttribute.class);
44      }
45      
46      public void reinit(String stringValue, int startOffset,  int endOffset) {
47        termAttribute.setTermBuffer(stringValue);
48        offsetAttribute.setOffset(startOffset, endOffset);
49      }
50      
51      // this is a dummy, to not throw an UOE because this class does not implement any iteration method
52      public boolean incrementToken() {
53        throw new UnsupportedOperationException();
54      }
55    }
56    
57    final DocumentsWriter.DocState docState;
58  
59    final FieldInvertState fieldState = new FieldInvertState();
60  
61    // Used to read a string value for a field
62    final ReusableStringReader stringReader = new ReusableStringReader();
63  
64    public DocInverterPerThread(DocFieldProcessorPerThread docFieldProcessorPerThread, DocInverter docInverter) {
65      this.docInverter = docInverter;
66      docState = docFieldProcessorPerThread.docState;
67      consumer = docInverter.consumer.addThread(this);
68      endConsumer = docInverter.endConsumer.addThread(this);
69    }
70  
71    public void startDocument() throws IOException {
72      consumer.startDocument();
73      endConsumer.startDocument();
74    }
75  
76    public DocumentsWriter.DocWriter finishDocument() throws IOException {
77      // TODO: allow endConsumer.finishDocument to also return
78      // a DocWriter
79      endConsumer.finishDocument();
80      return consumer.finishDocument();
81    }
82  
83    void abort() {
84      try {
85        consumer.abort();
86      } finally {
87        endConsumer.abort();
88      }
89    }
90  
91    public DocFieldConsumerPerField addField(FieldInfo fi) {
92      return new DocInverterPerField(this, fi);
93    }
94  }
