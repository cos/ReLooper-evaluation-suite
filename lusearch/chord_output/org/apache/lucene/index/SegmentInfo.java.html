31  /**
32   * Information about a segment such as it's name, directory, and files related
33   * to the segment.
34   * 
35   * * <p><b>NOTE:</b> This API is new and still experimental
36   * (subject to change suddenly in the next release)</p>
37   */
38  public final class SegmentInfo {
39  
40    static final int NO = -1;          // e.g. no norms; no deletes;
41    static final int YES = 1;          // e.g. have norms; have deletes;
42    static final int CHECK_DIR = 0;    // e.g. must check dir to see if there are norms/deletions
43    static final int WITHOUT_GEN = 0;  // a file name that has no GEN in it. 
44  
45    public String name;                 // unique name in dir
46    public int docCount;                // number of docs in seg
47    public Directory dir;               // where segment resides
48  
49    private boolean preLockless;                    // true if this is a segments file written before
50                                                    // lock-less commits (2.1)
51  
52    private long delGen;                            // current generation of del file; NO if there
53                                                    // are no deletes; CHECK_DIR if it's a pre-2.1 segment
54                                                    // (and we must check filesystem); YES or higher if
55                                                    // there are deletes at generation N
56     
57    private long[] normGen;                         // current generation of each field's norm file.
58                                                    // If this array is null, for lockLess this means no 
59                                                    // separate norms.  For preLockLess this means we must 
60                                                    // check filesystem. If this array is not null, its 
61                                                    // values mean: NO says this field has no separate  
62                                                    // norms; CHECK_DIR says it is a preLockLess segment and    
63                                                    // filesystem must be checked; >= YES says this field  
64                                                    // has separate norms with the specified generation
65  
66    private byte isCompoundFile;                    // NO if it is not; YES if it is; CHECK_DIR if it's
67                                                    // pre-2.1 (ie, must check file system to see
68                                                    // if <name>.cfs and <name>.nrm exist)         
69  
70    private boolean hasSingleNormFile;              // true if this segment maintains norms in a single file; 
71                                                    // false otherwise
72                                                    // this is currently false for segments populated by DocumentWriter
73                                                    // and true for newly created merged segments (both
74                                                    // compound and non compound).
75    
76    private List files;                             // cached list of files that this segment uses
77                                                    // in the Directory
78  
79    long sizeInBytes = -1;                          // total byte size of all of our files (computed on demand)
80  
81    private int docStoreOffset;                     // if this segment shares stored fields & vectors, this
82                                                    // offset is where in that file this segment's docs begin
83    private String docStoreSegment;                 // name used to derive fields/vectors file we share with
84                                                    // other segments
85    private boolean docStoreIsCompoundFile;         // whether doc store files are stored in compound file (*.cfx)
86  
87    private int delCount;                           // How many deleted docs in this segment, or -1 if not yet known
88                                                    // (if it's an older index)
89  
90    private boolean hasProx;                        // True if this segment has any fields with omitTermFreqAndPositions==false
91  
92    private Map diagnostics;

95      return "si: "+dir.toString()+" "+name+" docCount: "+docCount+" delCount: "+delCount+" delFileName: "+getDelFileName();

117   public SegmentInfo(String name, int docCount, Directory dir, boolean isCompoundFile, boolean hasSingleNormFile,
118                      int docStoreOffset, String docStoreSegment, boolean docStoreIsCompoundFile, boolean hasProx) { 
119     this(name, docCount, dir);
120     this.isCompoundFile = (byte) (isCompoundFile ? YES : NO);
121     this.hasSingleNormFile = hasSingleNormFile;
122     preLockless = false;
123     this.docStoreOffset = docStoreOffset;
124     this.docStoreSegment = docStoreSegment;
125     this.docStoreIsCompoundFile = docStoreIsCompoundFile;
126     this.hasProx = hasProx;
127     delCount = 0;
128     assert docStoreOffset == -1 || docStoreSegment != null: "dso=" + docStoreOffset + " dss=" + docStoreSegment + " docCount=" + docCount;
129   }

131   /**
132    * Copy everything from src SegmentInfo into our instance.
133    */
134   void reset(SegmentInfo src) {
135     clearFiles();
136     name = src.name;
137     docCount = src.docCount;
138     dir = src.dir;
139     preLockless = src.preLockless;
140     delGen = src.delGen;
141     docStoreOffset = src.docStoreOffset;
142     docStoreIsCompoundFile = src.docStoreIsCompoundFile;
143     if (src.normGen == null) {
144       normGen = null;
145     } else {
146       normGen = new long[src.normGen.length];
147       System.arraycopy(src.normGen, 0, normGen, 0, src.normGen.length);
148     }
149     isCompoundFile = src.isCompoundFile;
150     hasSingleNormFile = src.hasSingleNormFile;
151     delCount = src.delCount;
152   }

164   /**
165    * Construct a new SegmentInfo instance by reading a
166    * previously saved SegmentInfo from input.
167    *
168    * @param dir directory to load from
169    * @param format format of the segments info file
170    * @param input input handle to read segment info from
171    */
172   SegmentInfo(Directory dir, int format, IndexInput input) throws IOException {
173     this.dir = dir;
174     name = input.readString();
175     docCount = input.readInt();
176     if (format <= SegmentInfos.FORMAT_LOCKLESS) {
177       delGen = input.readLong();
178       if (format <= SegmentInfos.FORMAT_SHARED_DOC_STORE) {
179         docStoreOffset = input.readInt();
180         if (docStoreOffset != -1) {
181           docStoreSegment = input.readString();
182           docStoreIsCompoundFile = (1 == input.readByte());
183         } else {
184           docStoreSegment = name;
185           docStoreIsCompoundFile = false;
186         }
187       } else {
188         docStoreOffset = -1;
189         docStoreSegment = name;
190         docStoreIsCompoundFile = false;
191       }
192       if (format <= SegmentInfos.FORMAT_SINGLE_NORM_FILE) {
193         hasSingleNormFile = (1 == input.readByte());
194       } else {
195         hasSingleNormFile = false;
196       }
197       int numNormGen = input.readInt();
198       if (numNormGen == NO) {
199         normGen = null;
200       } else {
201         normGen = new long[numNormGen];
202         for(int j=0;j<numNormGen;j++) {
203           normGen[j] = input.readLong();
204         }
205       }
206       isCompoundFile = input.readByte();
207       preLockless = (isCompoundFile == CHECK_DIR);
208       if (format <= SegmentInfos.FORMAT_DEL_COUNT) {
209         delCount = input.readInt();
210         assert delCount <= docCount;
211       } else
212         delCount = -1;
213       if (format <= SegmentInfos.FORMAT_HAS_PROX)
214         hasProx = input.readByte() == 1;
215       else
216         hasProx = true;
217 
218       if (format <= SegmentInfos.FORMAT_DIAGNOSTICS) {
219         diagnostics = input.readStringStringMap();
220       } else {
221         diagnostics = Collections.EMPTY_MAP;
222       }
223     } else {
224       delGen = CHECK_DIR;
225       normGen = null;
226       isCompoundFile = CHECK_DIR;
227       preLockless = true;
228       hasSingleNormFile = false;
229       docStoreOffset = -1;
230       docStoreIsCompoundFile = false;
231       docStoreSegment = null;
232       delCount = -1;
233       hasProx = true;
234       diagnostics = Collections.EMPTY_MAP;
235     }
236   }

238   void setNumFields(int numFields) {
239     if (normGen == null) {
240       // normGen is null if we loaded a pre-2.1 segment
241       // file, or, if this segments file hasn't had any
242       // norms set against it yet:
243       normGen = new long[numFields];
244 
245       if (preLockless) {
246         // Do nothing: thus leaving normGen[k]==CHECK_DIR (==0), so that later we know  
247         // we have to check filesystem for norm files, because this is prelockless.
248         
249       } else {
250         // This is a FORMAT_LOCKLESS segment, which means
251         // there are no separate norms:
252         for(int i=0;i<numFields;i++) {
253           normGen[i] = NO;
254         }
255       }
256     }
257   }

259   /** Returns total size in bytes of all of files used by
260    *  this segment. */
261   public long sizeInBytes() throws IOException {
262     if (sizeInBytes == -1) {
263       List files = files();
264       final int size = files.size();
265       sizeInBytes = 0;
266       for(int i=0;i<size;i++) {
267         final String fileName = (String) files.get(i);
268         // We don't count bytes used by a shared doc store
269         // against this segment:
270         if (docStoreOffset == -1 || !IndexFileNames.isDocStoreFile(fileName))
271           sizeInBytes += dir.fileLength(fileName);
272       }
273     }
274     return sizeInBytes;
275   }

277   public boolean hasDeletions()
278     throws IOException {
279     // Cases:
280     //
281     //   delGen == NO: this means this segment was written
282     //     by the LOCKLESS code and for certain does not have
283     //     deletions yet
284     //
285     //   delGen == CHECK_DIR: this means this segment was written by
286     //     pre-LOCKLESS code which means we must check
287     //     directory to see if .del file exists
288     //
289     //   delGen >= YES: this means this segment was written by
290     //     the LOCKLESS code and for certain has
291     //     deletions
292     //
293     if (delGen == NO) {
294       return false;
295     } else if (delGen >= YES) {
296       return true;
297     } else {
298       return dir.fileExists(getDelFileName());
299     }
300   }

302   void advanceDelGen() {
303     // delGen 0 is reserved for pre-LOCKLESS format
304     if (delGen == NO) {
305       delGen = YES;
306     } else {
307       delGen++;
308     }
309     clearFiles();
310   }

312   void clearDelGen() {
313     delGen = NO;
314     clearFiles();
315   }

317   public Object clone () {
318     SegmentInfo si = new SegmentInfo(name, docCount, dir);
319     si.isCompoundFile = isCompoundFile;
320     si.delGen = delGen;
321     si.delCount = delCount;
322     si.hasProx = hasProx;
323     si.preLockless = preLockless;
324     si.hasSingleNormFile = hasSingleNormFile;
325     si.diagnostics = new HashMap(diagnostics);
326     if (normGen != null) {
327       si.normGen = (long[]) normGen.clone();
328     }
329     si.docStoreOffset = docStoreOffset;
330     si.docStoreSegment = docStoreSegment;
331     si.docStoreIsCompoundFile = docStoreIsCompoundFile;
332     return si;
333   }

335   public String getDelFileName() {
336     if (delGen == NO) {
337       // In this case we know there is no deletion filename
338       // against this segment
339       return null;
340     } else {
341       // If delGen is CHECK_DIR, it's the pre-lockless-commit file format
342       return IndexFileNames.fileNameFromGeneration(name, "." + IndexFileNames.DELETES_EXTENSION, delGen); 
343     }
344   }

346   /**
347    * Returns true if this field for this segment has saved a separate norms file (_<segment>_N.sX).
348    *
349    * @param fieldNumber the field index to check
350    */
351   public boolean hasSeparateNorms(int fieldNumber)
352     throws IOException {
353     if ((normGen == null && preLockless) || (normGen != null && normGen[fieldNumber] == CHECK_DIR)) {
354       // Must fallback to directory file exists check:
355       String fileName = name + ".s" + fieldNumber;
356       return dir.fileExists(fileName);
357     } else if (normGen == null || normGen[fieldNumber] == NO) {
358       return false;
359     } else {
360       return true;
361     }
362   }
363 
364   /**
365    * Returns true if any fields in this segment have separate norms.
366    */
367   public boolean hasSeparateNorms()
368     throws IOException {
369     if (normGen == null) {
370       if (!preLockless) {
371         // This means we were created w/ LOCKLESS code and no
372         // norms are written yet:
373         return false;
374       } else {
375         // This means this segment was saved with pre-LOCKLESS
376         // code.  So we must fallback to the original
377         // directory list check:
378         String[] result = dir.list();
379         if (result == null)
380           throw new IOException("cannot read directory " + dir + ": list() returned null");
381         
382         String pattern;
383         pattern = name + ".s";
384         int patternLength = pattern.length();
385         for(int i = 0; i < result.length; i++){
386           if(result[i].startsWith(pattern) && Character.isDigit(result[i].charAt(patternLength)))
387             return true;
388         }
389         return false;
390       }
391     } else {
392       // This means this segment was saved with LOCKLESS
393       // code so we first check whether any normGen's are >= 1
394       // (meaning they definitely have separate norms):
395       for(int i=0;i<normGen.length;i++) {
396         if (normGen[i] >= YES) {
397           return true;
398         }
399       }
400       // Next we look for any == 0.  These cases were
401       // pre-LOCKLESS and must be checked in directory:
402       for(int i=0;i<normGen.length;i++) {
403         if (normGen[i] == CHECK_DIR) {
404           if (hasSeparateNorms(i)) {
405             return true;
406           }
407         }
408       }
409     }
410 
411     return false;
412   }

429   /**
430    * Get the file name for the norms file for this field.
431    *
432    * @param number field index
433    */
434   public String getNormFileName(int number) throws IOException {
435     String prefix;
436 
437     long gen;
438     if (normGen == null) {
439       gen = CHECK_DIR;
440     } else {
441       gen = normGen[number];
442     }
443     
444     if (hasSeparateNorms(number)) {
445       // case 1: separate norm
446       prefix = ".s";
447       return IndexFileNames.fileNameFromGeneration(name, prefix + number, gen);
448     }
449 
450     if (hasSingleNormFile) {
451       // case 2: lockless (or nrm file exists) - single file for all norms 
452       prefix = "." + IndexFileNames.NORMS_EXTENSION;
453       return IndexFileNames.fileNameFromGeneration(name, prefix, WITHOUT_GEN);
454     }
455       
456     // case 3: norm file for each field
457     prefix = ".f";
458     return IndexFileNames.fileNameFromGeneration(name, prefix + number, WITHOUT_GEN);
459   }

490   public int getDelCount() throws IOException {
491     if (delCount == -1) {
492       if (hasDeletions()) {
493         final String delFileName = getDelFileName();
494         delCount = new BitVector(dir, delFileName).count();
495       } else
496         delCount = 0;
497     }
498     assert delCount <= docCount;
499     return delCount;
500   }

535   /**
536    * Save this segment's info.
537    */
538   void write(IndexOutput output)
539     throws IOException {
540     output.writeString(name);
541     output.writeInt(docCount);
542     output.writeLong(delGen);
543     output.writeInt(docStoreOffset);
544     if (docStoreOffset != -1) {
545       output.writeString(docStoreSegment);
546       output.writeByte((byte) (docStoreIsCompoundFile ? 1:0));
547     }
548 
549     output.writeByte((byte) (hasSingleNormFile ? 1:0));
550     if (normGen == null) {
551       output.writeInt(NO);
552     } else {
553       output.writeInt(normGen.length);
554       for(int j = 0; j < normGen.length; j++) {
555         output.writeLong(normGen[j]);
556       }
557     }
558     output.writeByte(isCompoundFile);
559     output.writeInt(delCount);
560     output.writeByte((byte) (hasProx ? 1:0));
561     output.writeStringStringMap(diagnostics);
562   }

578   /*
579    * Return all files referenced by this SegmentInfo.  The
580    * returns List is a locally cached List so you should not
581    * modify it.
582    */
583 
584   public List files() throws IOException {
585 
586     if (files != null) {
587       // Already cached:
588       return files;
589     }
590     
591     files = new ArrayList();
592     
593     boolean useCompoundFile = getUseCompoundFile();
594 
595     if (useCompoundFile) {
596       files.add(name + "." + IndexFileNames.COMPOUND_FILE_EXTENSION);
597     } else {
598       final String[] exts = IndexFileNames.NON_STORE_INDEX_EXTENSIONS;
599       for(int i=0;i<exts.length;i++)
600         addIfExists(files, name + "." + exts[i]);
601     }
602 
603     if (docStoreOffset != -1) {
604       // We are sharing doc stores (stored fields, term
605       // vectors) with other segments
606       assert docStoreSegment != null;
607       if (docStoreIsCompoundFile) {
608         files.add(docStoreSegment + "." + IndexFileNames.COMPOUND_FILE_STORE_EXTENSION);
609       } else {
610         final String[] exts = IndexFileNames.STORE_INDEX_EXTENSIONS;
611         for(int i=0;i<exts.length;i++)
612           addIfExists(files, docStoreSegment + "." + exts[i]);
613       }
614     } else if (!useCompoundFile) {
615       // We are not sharing, and, these files were not
616       // included in the compound file
617       final String[] exts = IndexFileNames.STORE_INDEX_EXTENSIONS;
618       for(int i=0;i<exts.length;i++)
619         addIfExists(files, name + "." + exts[i]);
620     }
621 
622     String delFileName = IndexFileNames.fileNameFromGeneration(name, "." + IndexFileNames.DELETES_EXTENSION, delGen);
623     if (delFileName != null && (delGen >= YES || dir.fileExists(delFileName))) {
624       files.add(delFileName);
625     }
626 
627     // Careful logic for norms files    
628     if (normGen != null) {
629       for(int i=0;i<normGen.length;i++) {
630         long gen = normGen[i];
631         if (gen >= YES) {
632           // Definitely a separate norm file, with generation:
633           files.add(IndexFileNames.fileNameFromGeneration(name, "." + IndexFileNames.SEPARATE_NORMS_EXTENSION + i, gen));
634         } else if (NO == gen) {
635           // No separate norms but maybe plain norms
636           // in the non compound file case:
637           if (!hasSingleNormFile && !useCompoundFile) {
638             String fileName = name + "." + IndexFileNames.PLAIN_NORMS_EXTENSION + i;
639             if (dir.fileExists(fileName)) {
640               files.add(fileName);
641             }
642           }
643         } else if (CHECK_DIR == gen) {
644           // Pre-2.1: we have to check file existence
645           String fileName = null;
646           if (useCompoundFile) {
647             fileName = name + "." + IndexFileNames.SEPARATE_NORMS_EXTENSION + i;
648           } else if (!hasSingleNormFile) {
649             fileName = name + "." + IndexFileNames.PLAIN_NORMS_EXTENSION + i;
650           }
651           if (fileName != null && dir.fileExists(fileName)) {
652             files.add(fileName);
653           }
654         }
655       }
656     } else if (preLockless || (!hasSingleNormFile && !useCompoundFile)) {
657       // Pre-2.1: we have to scan the dir to find all
658       // matching _X.sN/_X.fN files for our segment:
659       String prefix;
660       if (useCompoundFile)
661         prefix = name + "." + IndexFileNames.SEPARATE_NORMS_EXTENSION;
662       else
663         prefix = name + "." + IndexFileNames.PLAIN_NORMS_EXTENSION;
664       int prefixLength = prefix.length();
665       String[] allFiles = dir.listAll();
666       final IndexFileNameFilter filter = IndexFileNameFilter.getFilter();
667       for(int i=0;i<allFiles.length;i++) {
668         String fileName = allFiles[i];
669         if (filter.accept(null, fileName) && fileName.length() > prefixLength && Character.isDigit(fileName.charAt(prefixLength)) && fileName.startsWith(prefix)) {
670           files.add(fileName);
671         }
672       }
673     }
674     return files;
675   }

677   /* Called whenever any change is made that affects which
678    * files this segment has. */
679   private void clearFiles() {
680     files = null;
681     sizeInBytes = -1;

684   /** Used for debugging */
685   public String segString(Directory dir) {
686     String cfs;
687     try {
688       if (getUseCompoundFile())
689         cfs = "c";
690       else
691         cfs = "C";
692     } catch (IOException ioe) {
693       cfs = "?";
694     }
695 
696     String docStore;
697 
698     if (docStoreOffset != -1)

704       cfs +
