23  /** <p>This class implements a {@link MergePolicy} that tries
24   *  to merge segments into levels of exponentially
25   *  increasing size, where each level has fewer segments than
26   *  the value of the merge factor. Whenever extra segments
27   *  (beyond the merge factor upper bound) are encountered,
28   *  all segments within the level are merged. You can get or
29   *  set the merge factor using {@link #getMergeFactor()} and
30   *  {@link #setMergeFactor(int)} respectively.</p>
31   *
32   * <p>This class is abstract and requires a subclass to
33   * define the {@link #size} method which specifies how a
34   * segment's size is determined.  {@link LogDocMergePolicy}
35   * is one subclass that measures size by document count in
36   * the segment.  {@link LogByteSizeMergePolicy} is another
37   * subclass that measures size as the total byte size of the
38   * file(s) for the segment.</p>
39   */
40  
41  public abstract class LogMergePolicy extends MergePolicy {
42  
43    /** Defines the allowed range of log(size) for each
44     *  level.  A level is computed by taking the max segment
45     *  log size, minus LEVEL_LOG_SPAN, and finding all
46     *  segments falling within that range. */
47    public static final double LEVEL_LOG_SPAN = 0.75;
48  
49    /** Default merge factor, which is how many segments are
50     *  merged at a time */
51    public static final int DEFAULT_MERGE_FACTOR = 10;
52  
53    /** Default maximum segment size.  A segment of this size
54     *  or larger will never be merged.  @see setMaxMergeDocs */
55    public static final int DEFAULT_MAX_MERGE_DOCS = Integer.MAX_VALUE;
56  
57    private int mergeFactor = DEFAULT_MERGE_FACTOR;
58  
59    long minMergeSize;
60    long maxMergeSize;
61    int maxMergeDocs = DEFAULT_MAX_MERGE_DOCS;

82    /** <p>Returns the number of segments that are merged at
83     * once and also controls the total number of segments
84     * allowed to accumulate in the index.</p> */
85    public int getMergeFactor() {
86      return mergeFactor;
87    }

89    /** Determines how often segment indices are merged by
90     * addDocument().  With smaller values, less RAM is used
91     * while indexing, and searches on unoptimized indices are
92     * faster, but indexing speed is slower.  With larger
93     * values, more RAM is used during indexing, and while
94     * searches on unoptimized indices are slower, indexing is
95     * faster.  Thus larger values (> 10) are best for batch
96     * index creation, and smaller values (< 10) for indices
97     * that are interactively maintained. */
98    public void setMergeFactor(int mergeFactor) {
99      if (mergeFactor < 2)
100       throw new IllegalArgumentException("mergeFactor cannot be less than 2");
101     this.mergeFactor = mergeFactor;
102   }

156   abstract protected long size(SegmentInfo info) throws IOException;

158   protected long sizeDocs(SegmentInfo info) throws IOException {
159     if (calibrateSizeByDeletes) {
160       int delCount = writer.numDeletedDocs(info);
161       return (info.docCount - (long)delCount);
162     } else {
163       return info.docCount;
164     }
165   }
166   
167   protected long sizeBytes(SegmentInfo info) throws IOException {
168     long byteSize = info.sizeInBytes();
169     if (calibrateSizeByDeletes) {
170       int delCount = writer.numDeletedDocs(info);
171       float delRatio = (info.docCount <= 0 ? 0.0f : ((float)delCount / (float)info.docCount));
172       return (info.docCount <= 0 ?  byteSize : (long)(byteSize * (1.0f - delRatio)));
173     } else {
174       return byteSize;
175     }
176   }

178   private boolean isOptimized(SegmentInfos infos, int maxNumSegments, Set segmentsToOptimize) throws IOException {
179     final int numSegments = infos.size();
180     int numToOptimize = 0;
181     SegmentInfo optimizeInfo = null;
182     for(int i=0;i<numSegments && numToOptimize <= maxNumSegments;i++) {
183       final SegmentInfo info = infos.info(i);
184       if (segmentsToOptimize.contains(info)) {
185         numToOptimize++;
186         optimizeInfo = info;
187       }
188     }
189 
190     return numToOptimize <= maxNumSegments &&
191       (numToOptimize != 1 || isOptimized(optimizeInfo));
192   }

194   /** Returns true if this single info is optimized (has no
195    *  pending norms or deletes, is in the same dir as the
196    *  writer, and matches the current compound file setting */
197   private boolean isOptimized(SegmentInfo info)
198     throws IOException {
199     boolean hasDeletions = writer.numDeletedDocs(info) > 0;
200     return !hasDeletions &&
201       !info.hasSeparateNorms() &&
202       info.dir == writer.getDirectory() &&
203       info.getUseCompoundFile() == useCompoundFile;
204   }

206   /** Returns the merges necessary to optimize the index.
207    *  This merge policy defines "optimized" to mean only one
208    *  segment in the index, where that segment has no
209    *  deletions pending nor separate norms, and it is in
210    *  compound file format if the current useCompoundFile
211    *  setting is true.  This method returns multiple merges
212    *  (mergeFactor at a time) so the {@link MergeScheduler}
213    *  in use may make use of concurrency. */
214   public MergeSpecification findMergesForOptimize(SegmentInfos infos,
215       int maxNumSegments, Set segmentsToOptimize) throws IOException {
216     MergeSpecification spec;
217 
218     assert maxNumSegments > 0;
219 
220     if (!isOptimized(infos, maxNumSegments, segmentsToOptimize)) {
221 
222       // Find the newest (rightmost) segment that needs to
223       // be optimized (other segments may have been flushed
224       // since optimize started):
225       int last = infos.size();
226       while(last > 0) {
227         final SegmentInfo info = infos.info(--last);
228         if (segmentsToOptimize.contains(info)) {
229           last++;
230           break;
231         }
232       }
233 
234       if (last > 0) {
235 
236         spec = new MergeSpecification();
237 
238         // First, enroll all "full" merges (size
239         // mergeFactor) to potentially be run concurrently:
240         while (last - maxNumSegments + 1 >= mergeFactor) {
241           spec.add(new OneMerge(infos.range(last-mergeFactor, last), useCompoundFile));
242           last -= mergeFactor;
243         }
244 
245         // Only if there are no full merges pending do we
246         // add a final partial (< mergeFactor segments) merge:
247         if (0 == spec.merges.size()) {
248           if (maxNumSegments == 1) {
249 
250             // Since we must optimize down to 1 segment, the
251             // choice is simple:
252             if (last > 1 || !isOptimized(infos.info(0)))
253               spec.add(new OneMerge(infos.range(0, last), useCompoundFile));
254           } else if (last > maxNumSegments) {
255 
256             // Take care to pick a partial merge that is
257             // least cost, but does not make the index too
258             // lopsided.  If we always just picked the
259             // partial tail then we could produce a highly
260             // lopsided index over time:
261 
262             // We must merge this many segments to leave
263             // maxNumSegments in the index (from when
264             // optimize was first kicked off):
265             final int finalMergeSize = last - maxNumSegments + 1;
266 
267             // Consider all possible starting points:
268             long bestSize = 0;
269             int bestStart = 0;
270 
271             for(int i=0;i<last-finalMergeSize+1;i++) {
272               long sumSize = 0;
273               for(int j=0;j<finalMergeSize;j++)
274                 sumSize += size(infos.info(j+i));
275               if (i == 0 || (sumSize < 2*size(infos.info(i-1)) && sumSize < bestSize)) {
276                 bestStart = i;
277                 bestSize = sumSize;
278               }
279             }
280 
281             spec.add(new OneMerge(infos.range(bestStart, bestStart+finalMergeSize), useCompoundFile));
282           }
283         }
284         
285       } else
286         spec = null;
287     } else
288       spec = null;
289 
290     return spec;
291   }

293   /**
294    * Finds merges necessary to expunge all deletes from the
295    * index.  We simply merge adjacent segments that have
296    * deletes, up to mergeFactor at a time.
297    */ 
298   public MergeSpecification findMergesToExpungeDeletes(SegmentInfos segmentInfos)
299       throws CorruptIndexException, IOException {
300     final int numSegments = segmentInfos.size();
301 
302     if (verbose())
303       message("findMergesToExpungeDeletes: " + numSegments + " segments");
304 
305     MergeSpecification spec = new MergeSpecification();
306     int firstSegmentWithDeletions = -1;
307     for(int i=0;i<numSegments;i++) {
308       final SegmentInfo info = segmentInfos.info(i);
309       int delCount = writer.numDeletedDocs(info);
310       if (delCount > 0) {
311         if (verbose())
312           message("  segment " + info.name + " has deletions");
313         if (firstSegmentWithDeletions == -1)
314           firstSegmentWithDeletions = i;
315         else if (i - firstSegmentWithDeletions == mergeFactor) {
316           // We've seen mergeFactor segments in a row with
317           // deletions, so force a merge now:
318           if (verbose())
319             message("  add merge " + firstSegmentWithDeletions + " to " + (i-1) + " inclusive");
320           spec.add(new OneMerge(segmentInfos.range(firstSegmentWithDeletions, i), useCompoundFile));
321           firstSegmentWithDeletions = i;
322         }
323       } else if (firstSegmentWithDeletions != -1) {
324         // End of a sequence of segments with deletions, so,
325         // merge those past segments even if it's fewer than
326         // mergeFactor segments
327         if (verbose())
328           message("  add merge " + firstSegmentWithDeletions + " to " + (i-1) + " inclusive");
329         spec.add(new OneMerge(segmentInfos.range(firstSegmentWithDeletions, i), useCompoundFile));
330         firstSegmentWithDeletions = -1;
331       }
332     }
333 
334     if (firstSegmentWithDeletions != -1) {
335       if (verbose())
336         message("  add merge " + firstSegmentWithDeletions + " to " + (numSegments-1) + " inclusive");
337       spec.add(new OneMerge(segmentInfos.range(firstSegmentWithDeletions, numSegments), useCompoundFile));
338     }
339 
340     return spec;
341   }

343   /** Checks if any merges are now necessary and returns a
344    *  {@link MergePolicy.MergeSpecification} if so.  A merge
345    *  is necessary when there are more than {@link
346    *  #setMergeFactor} segments at a given level.  When
347    *  multiple levels have too many segments, this method
348    *  will return multiple merges, allowing the {@link
349    *  MergeScheduler} to use concurrency. */
350   public MergeSpecification findMerges(SegmentInfos infos) throws IOException {
351 
352     final int numSegments = infos.size();
353     if (verbose())
354       message("findMerges: " + numSegments + " segments");
355 
356     // Compute levels, which is just log (base mergeFactor)
357     // of the size of each segment
358     float[] levels = new float[numSegments];
359     final float norm = (float) Math.log(mergeFactor);
360 
361     for(int i=0;i<numSegments;i++) {
362       final SegmentInfo info = infos.info(i);
363       long size = size(info);
364 
365       // Floor tiny segments
366       if (size < 1)
367         size = 1;
368       levels[i] = (float) Math.log(size)/norm;
369     }
370 
371     final float levelFloor;
372     if (minMergeSize <= 0)
373       levelFloor = (float) 0.0;
374     else
375       levelFloor = (float) (Math.log(minMergeSize)/norm);
376 
377     // Now, we quantize the log values into levels.  The
378     // first level is any segment whose log size is within
379     // LEVEL_LOG_SPAN of the max size, or, who has such as
380     // segment "to the right".  Then, we find the max of all
381     // other segments and use that to define the next level
382     // segment, etc.
383 
384     MergeSpecification spec = null;
385 
386     int start = 0;
387     while(start < numSegments) {
388 
389       // Find max level of all segments not already
390       // quantized.
391       float maxLevel = levels[start];
392       for(int i=1+start;i<numSegments;i++) {
393         final float level = levels[i];
394         if (level > maxLevel)
395           maxLevel = level;
396       }
397 
398       // Now search backwards for the rightmost segment that
399       // falls into this level:
400       float levelBottom;
401       if (maxLevel < levelFloor)
402         // All remaining segments fall into the min level
403         levelBottom = -1.0F;
404       else {
405         levelBottom = (float) (maxLevel - LEVEL_LOG_SPAN);
406 
407         // Force a boundary at the level floor
408         if (levelBottom < levelFloor && maxLevel >= levelFloor)
409           levelBottom = levelFloor;
410       }
411 
412       int upto = numSegments-1;
413       while(upto >= start) {
414         if (levels[upto] >= levelBottom) {
415           break;
416         }
417         upto--;
418       }
419       if (verbose())
420         message("  level " + levelBottom + " to " + maxLevel + ": " + (1+upto-start) + " segments");
421 
422       // Finally, record all merges that are viable at this level:
423       int end = start + mergeFactor;
424       while(end <= 1+upto) {
425         boolean anyTooLarge = false;
426         for(int i=start;i<end;i++) {
427           final SegmentInfo info = infos.info(i);
428           anyTooLarge |= (size(info) >= maxMergeSize || sizeDocs(info) >= maxMergeDocs);
429         }
430 
431         if (!anyTooLarge) {
432           if (spec == null)
433             spec = new MergeSpecification();
434           if (verbose())
435             message("    " + start + " to " + end + ": add this merge");
436           spec.add(new OneMerge(infos.range(start, end), useCompoundFile));
437         } else if (verbose())
438           message("    " + start + " to " + end + ": contains segment over maxMergeSize or maxMergeDocs; skipping");
439 
440         start = end;
441         end = start + mergeFactor;
442       }
443 
444       start = 1+upto;
445     }
446 
447     return spec;
448   }

450   /** <p>Determines the largest segment (measured by
451    * document count) that may be merged with other segments.
452    * Small values (e.g., less than 10,000) are best for
453    * interactive indexing, as this limits the length of
454    * pauses while indexing to a few seconds.  Larger values
455    * are best for batched indexing and speedier
456    * searches.</p>
457    *
458    * <p>The default value is {@link Integer#MAX_VALUE}.</p>
459    *
460    * <p>The default merge policy ({@link
461    * LogByteSizeMergePolicy}) also allows you to set this
462    * limit by net size (in MB) of the segment, using {@link
463    * LogByteSizeMergePolicy#setMaxMergeMB}.</p>
464    */
465   public void setMaxMergeDocs(int maxMergeDocs) {
466     this.maxMergeDocs = maxMergeDocs;
467   }

469   /** Returns the largest segment (measured by document
470    *  count) that may be merged with other segments.

472   public int getMaxMergeDocs() {
473     return maxMergeDocs;
