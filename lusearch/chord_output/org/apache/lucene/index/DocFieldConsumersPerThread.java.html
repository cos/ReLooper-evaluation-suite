20  import java.io.IOException;
21  
22  final class DocFieldConsumersPerThread extends DocFieldConsumerPerThread {
23  
24    final DocFieldConsumerPerThread one;
25    final DocFieldConsumerPerThread two;
26    final DocFieldConsumers parent;
27    final DocumentsWriter.DocState docState;
28  
29    public DocFieldConsumersPerThread(DocFieldProcessorPerThread docFieldProcessorPerThread,
30                                      DocFieldConsumers parent, DocFieldConsumerPerThread one, DocFieldConsumerPerThread two) {
31      this.parent = parent;
32      this.one = one;
33      this.two = two;
34      docState = docFieldProcessorPerThread.docState;
35    }
36  
37    public void startDocument() throws IOException {
38      one.startDocument();
39      two.startDocument();
40    }
41  
42    public void abort() {
43      try {
44        one.abort();
45      } finally {
46        two.abort();
47      }
48    }
49  
50    public DocumentsWriter.DocWriter finishDocument() throws IOException {
51      final DocumentsWriter.DocWriter oneDoc = one.finishDocument();
52      final DocumentsWriter.DocWriter twoDoc = two.finishDocument();
53      if (oneDoc == null)
54        return twoDoc;
55      else if (twoDoc == null)
56        return oneDoc;
57      else {
58        DocFieldConsumers.PerDoc both = parent.getPerDoc();
59        both.docID = docState.docID;
60        assert oneDoc.docID == docState.docID;
61        assert twoDoc.docID == docState.docID;
62        both.one = oneDoc;
63        both.two = twoDoc;
64        return both;
65      }
66    }
67  
68    public DocFieldConsumerPerField addField(FieldInfo fi) {
69      return new DocFieldConsumersPerField(this, one.addField(fi), two.addField(fi));
70    }
71  }
