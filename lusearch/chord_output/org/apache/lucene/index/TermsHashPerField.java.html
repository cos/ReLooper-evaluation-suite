54    public TermsHashPerField(DocInverterPerField docInverterPerField, final TermsHashPerThread perThread, final TermsHashPerThread nextPerThread, final FieldInfo fieldInfo) {
55      this.perThread = perThread;
56      intPool = perThread.intPool;
57      charPool = perThread.charPool;
58      bytePool = perThread.bytePool;
59      docState = perThread.docState;
60      fieldState = docInverterPerField.fieldState;
61      this.consumer = perThread.consumer.addField(this, fieldInfo);
62      streamCount = consumer.getStreamCount();
63      numPostingInt = 2*streamCount;
64      this.fieldInfo = fieldInfo;
65      if (nextPerThread != null)
66        nextPerField = (TermsHashPerField) nextPerThread.addField(docInverterPerField, fieldInfo);
67      else
68        nextPerField = null;
69    }

71    void shrinkHash(int targetSize) {
72      assert postingsCompacted || numPostings == 0;
73  
74      // Cannot use ArrayUtil.shrink because we require power
75      // of 2:
76      int newSize = postingsHash.length;
77      while(newSize >= 8 && newSize/4 > targetSize) {
78        newSize /= 2;
79      }
80  
81      if (newSize != postingsHash.length) {
82        postingsHash = new RawPostingList[newSize];
83        postingsHashSize = newSize;
84        postingsHashHalfSize = newSize/2;
85        postingsHashMask = newSize-1;
86      }
87    }

89    public void reset() {
90      if (!postingsCompacted)
91        compactPostings();
92      assert numPostings <= postingsHash.length;
93      if (numPostings > 0) {
94        perThread.termsHash.recyclePostings(postingsHash, numPostings);
95        Arrays.fill(postingsHash, 0, numPostings, null);
96        numPostings = 0;
97      }
98      postingsCompacted = false;
99      if (nextPerField != null)
100       nextPerField.reset();
101   }

103   synchronized public void abort() {
104     reset();
105     if (nextPerField != null)
106       nextPerField.abort();
107   }

109   public void initReader(ByteSliceReader reader, RawPostingList p, int stream) {
110     assert stream < streamCount;
111     final int[] ints = intPool.buffers[p.intStart >> DocumentsWriter.INT_BLOCK_SHIFT];
112     final int upto = p.intStart & DocumentsWriter.INT_BLOCK_MASK;
113     reader.init(bytePool,
114                 p.byteStart+stream*ByteBlockPool.FIRST_LEVEL_SIZE,
115                 ints[upto+stream]);
116   }

118   private synchronized void compactPostings() {
119     int upto = 0;
120     for(int i=0;i<postingsHashSize;i++) {
121       if (postingsHash[i] != null) {
122         if (upto < i) {
123           postingsHash[upto] = postingsHash[i];
124           postingsHash[i] = null;
125         }
126         upto++;
127       }
128     }
129 
130     assert upto == numPostings;
131     postingsCompacted = true;
132   }

134   /** Collapse the hash table & sort in-place. */
135   public RawPostingList[] sortPostings() {
136     compactPostings();
137     quickSort(postingsHash, 0, numPostings-1);
138     return postingsHash;

141   void quickSort(RawPostingList[] postings, int lo, int hi) {
142     if (lo >= hi)
143       return;
144     else if (hi == 1+lo) {
145       if (comparePostings(postings[lo], postings[hi]) > 0) {
146         final RawPostingList tmp = postings[lo];
147         postings[lo] = postings[hi];
148         postings[hi] = tmp;
149       }
150       return;
151     }
152 
153     int mid = (lo + hi) >>> 1;
154 
155     if (comparePostings(postings[lo], postings[mid]) > 0) {
156       RawPostingList tmp = postings[lo];
157       postings[lo] = postings[mid];
158       postings[mid] = tmp;
159     }
160 
161     if (comparePostings(postings[mid], postings[hi]) > 0) {
162       RawPostingList tmp = postings[mid];
163       postings[mid] = postings[hi];
164       postings[hi] = tmp;
165 
166       if (comparePostings(postings[lo], postings[mid]) > 0) {
167         RawPostingList tmp2 = postings[lo];
168         postings[lo] = postings[mid];
169         postings[mid] = tmp2;
170       }
171     }
172 
173     int left = lo + 1;
174     int right = hi - 1;
175 
176     if (left >= right)
177       return;
178 
179     RawPostingList partition = postings[mid];
180 
181     for (; ;) {
182       while (comparePostings(postings[right], partition) > 0)
183         --right;
184 
185       while (left < right && comparePostings(postings[left], partition) <= 0)
186         ++left;
187 
188       if (left < right) {
189         RawPostingList tmp = postings[left];
190         postings[left] = postings[right];
191         postings[right] = tmp;
192         --right;
193       } else {
194         break;
195       }
196     }
197 
198     quickSort(postings, lo, left);
199     quickSort(postings, left + 1, hi);
200   }

202   /** Compares term text for two Posting instance and
203    *  returns -1 if p1 < p2; 1 if p1 > p2; else 0. */
204   int comparePostings(RawPostingList p1, RawPostingList p2) {
205 
206     if (p1 == p2)
207       return 0;
208 
209     final char[] text1 = charPool.buffers[p1.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
210     int pos1 = p1.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
211     final char[] text2 = charPool.buffers[p2.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
212     int pos2 = p2.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
213 
214     assert text1 != text2 || pos1 != pos2;
215 
216     while(true) {
217       final char c1 = text1[pos1++];
218       final char c2 = text2[pos2++];
219       if (c1 != c2) {
220         if (0xffff == c2)
221           return 1;
222         else if (0xffff == c1)
223           return -1;
224         else
225           return c1-c2;
226       } else
227         // This method should never compare equal postings
228         // unless p1==p2
229         assert c1 != 0xffff;
230     }
231   }

233   /** Test whether the text for current RawPostingList p equals
234    *  current tokenText. */
235   private boolean postingEquals(final char[] tokenText, final int tokenTextLen) {
236 
237     final char[] text = perThread.charPool.buffers[p.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
238     assert text != null;
239     int pos = p.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
240 
241     int tokenPos = 0;
242     for(;tokenPos<tokenTextLen;pos++,tokenPos++)
243       if (tokenText[tokenPos] != text[pos])
244         return false;
245     return 0xffff == text[pos];
246   }

266   // Secondary entry point (for 2nd & subsequent TermsHash),
267   // because token text has already been "interned" into
268   // textStart, so we hash by textStart
269   public void add(int textStart) throws IOException {
270 
271     int code = textStart;
272 
273     int hashPos = code & postingsHashMask;
274 
275     assert !postingsCompacted;
276 
277     // Locate RawPostingList in hash
278     p = postingsHash[hashPos];
279 
280     if (p != null && p.textStart != textStart) {
281       // Conflict: keep searching different locations in
282       // the hash table.
283       final int inc = ((code>>8)+code)|1;
284       do {
285         code += inc;
286         hashPos = code & postingsHashMask;
287         p = postingsHash[hashPos];
288       } while (p != null && p.textStart != textStart);
289     }
290 
291     if (p == null) {
292 
293       // First time we are seeing this token since we last
294       // flushed the hash.
295 
296       // Refill?
297       if (0 == perThread.freePostingsCount)
298         perThread.morePostings();
299 
300       // Pull next free RawPostingList from free list
301       p = perThread.freePostings[--perThread.freePostingsCount];
302       assert p != null;
303 
304       p.textStart = textStart;
305           
306       assert postingsHash[hashPos] == null;
307       postingsHash[hashPos] = p;
308       numPostings++;
309 
310       if (numPostings == postingsHashHalfSize)
311         rehashPostings(2*postingsHashSize);
312 
313       // Init stream slices
314       if (numPostingInt + intPool.intUpto > DocumentsWriter.INT_BLOCK_SIZE)
315         intPool.nextBuffer();
316 
317       if (DocumentsWriter.BYTE_BLOCK_SIZE - bytePool.byteUpto < numPostingInt*ByteBlockPool.FIRST_LEVEL_SIZE)
318         bytePool.nextBuffer();
319 
320       intUptos = intPool.buffer;
321       intUptoStart = intPool.intUpto;
322       intPool.intUpto += streamCount;
323 
324       p.intStart = intUptoStart + intPool.intOffset;
325 
326       for(int i=0;i<streamCount;i++) {
327         final int upto = bytePool.newSlice(ByteBlockPool.FIRST_LEVEL_SIZE);
328         intUptos[intUptoStart+i] = upto + bytePool.byteOffset;
329       }
330       p.byteStart = intUptos[intUptoStart];
331 
332       consumer.newTerm(p);
333 
334     } else {
335       intUptos = intPool.buffers[p.intStart >> DocumentsWriter.INT_BLOCK_SHIFT];
336       intUptoStart = p.intStart & DocumentsWriter.INT_BLOCK_MASK;
337       consumer.addTerm(p);
338     }
339   }

341   // Primary entry point (for first TermsHash)
342   void add() throws IOException {
343 
344     assert !postingsCompacted;
345 
346     // We are first in the chain so we must "intern" the
347     // term text into textStart address
348 
349     // Get the text of this term.
350     final char[] tokenText = termAtt.termBuffer();;
351     final int tokenTextLen = termAtt.termLength();
352 
353     // Compute hashcode & replace any invalid UTF16 sequences
354     int downto = tokenTextLen;
355     int code = 0;
356     while (downto > 0) {
357       char ch = tokenText[--downto];
358 
359       if (ch >= UnicodeUtil.UNI_SUR_LOW_START && ch <= UnicodeUtil.UNI_SUR_LOW_END) {
360         if (0 == downto) {
361           // Unpaired
362           ch = tokenText[downto] = UnicodeUtil.UNI_REPLACEMENT_CHAR;
363         } else {
364           final char ch2 = tokenText[downto-1];
365           if (ch2 >= UnicodeUtil.UNI_SUR_HIGH_START && ch2 <= UnicodeUtil.UNI_SUR_HIGH_END) {
366             // OK: high followed by low.  This is a valid
367             // surrogate pair.
368             code = ((code*31) + ch)*31+ch2;
369             downto--;
370             continue;
371           } else {
372             // Unpaired
373             ch = tokenText[downto] = UnicodeUtil.UNI_REPLACEMENT_CHAR;
374           }            
375         }
376       } else if (ch >= UnicodeUtil.UNI_SUR_HIGH_START && (ch <= UnicodeUtil.UNI_SUR_HIGH_END ||
377                                                           ch == 0xffff)) {
378         // Unpaired or 0xffff
379         ch = tokenText[downto] = UnicodeUtil.UNI_REPLACEMENT_CHAR;
380       }
381 
382       code = (code*31) + ch;
383     }
384 
385     int hashPos = code & postingsHashMask;
386 
387     // Locate RawPostingList in hash
388     p = postingsHash[hashPos];
389 
390     if (p != null && !postingEquals(tokenText, tokenTextLen)) {
391       // Conflict: keep searching different locations in
392       // the hash table.
393       final int inc = ((code>>8)+code)|1;
394       do {
395         code += inc;
396         hashPos = code & postingsHashMask;
397         p = postingsHash[hashPos];
398       } while (p != null && !postingEquals(tokenText, tokenTextLen));
399     }
400 
401     if (p == null) {
402 
403       // First time we are seeing this token since we last
404       // flushed the hash.
405       final int textLen1 = 1+tokenTextLen;
406       if (textLen1 + charPool.charUpto > DocumentsWriter.CHAR_BLOCK_SIZE) {
407         if (textLen1 > DocumentsWriter.CHAR_BLOCK_SIZE) {
408           // Just skip this term, to remain as robust as
409           // possible during indexing.  A TokenFilter
410           // can be inserted into the analyzer chain if
411           // other behavior is wanted (pruning the term
412           // to a prefix, throwing an exception, etc).
413 
414           if (docState.maxTermPrefix == null)
415             docState.maxTermPrefix = new String(tokenText, 0, 30);
416 
417           consumer.skippingLongTerm();
418           return;
419         }
420         charPool.nextBuffer();
421       }
422 
423       // Refill?
424       if (0 == perThread.freePostingsCount)
425         perThread.morePostings();
426 
427       // Pull next free RawPostingList from free list
428       p = perThread.freePostings[--perThread.freePostingsCount];
429       assert p != null;
430 
431       final char[] text = charPool.buffer;
432       final int textUpto = charPool.charUpto;
433       p.textStart = textUpto + charPool.charOffset;
434       charPool.charUpto += textLen1;
435       System.arraycopy(tokenText, 0, text, textUpto, tokenTextLen);
436       text[textUpto+tokenTextLen] = 0xffff;
437           
438       assert postingsHash[hashPos] == null;
439       postingsHash[hashPos] = p;
440       numPostings++;
441 
442       if (numPostings == postingsHashHalfSize)
443         rehashPostings(2*postingsHashSize);
444 
445       // Init stream slices
446       if (numPostingInt + intPool.intUpto > DocumentsWriter.INT_BLOCK_SIZE)
447         intPool.nextBuffer();
448 
449       if (DocumentsWriter.BYTE_BLOCK_SIZE - bytePool.byteUpto < numPostingInt*ByteBlockPool.FIRST_LEVEL_SIZE)
450         bytePool.nextBuffer();
451 
452       intUptos = intPool.buffer;
453       intUptoStart = intPool.intUpto;
454       intPool.intUpto += streamCount;
455 
456       p.intStart = intUptoStart + intPool.intOffset;
457 
458       for(int i=0;i<streamCount;i++) {
459         final int upto = bytePool.newSlice(ByteBlockPool.FIRST_LEVEL_SIZE);
460         intUptos[intUptoStart+i] = upto + bytePool.byteOffset;
461       }
462       p.byteStart = intUptos[intUptoStart];
463 
464       consumer.newTerm(p);
465 
466     } else {
467       intUptos = intPool.buffers[p.intStart >> DocumentsWriter.INT_BLOCK_SHIFT];
468       intUptoStart = p.intStart & DocumentsWriter.INT_BLOCK_MASK;
469       consumer.addTerm(p);
470     }
471 
472     if (doNextCall)
473       nextPerField.add(p.textStart);
474   }

479   void writeByte(int stream, byte b) {
480     int upto = intUptos[intUptoStart+stream];
481     byte[] bytes = bytePool.buffers[upto >> DocumentsWriter.BYTE_BLOCK_SHIFT];
482     assert bytes != null;
483     int offset = upto & DocumentsWriter.BYTE_BLOCK_MASK;
484     if (bytes[offset] != 0) {
485       // End of slice; allocate a new one
486       offset = bytePool.allocSlice(bytes, offset);
487       bytes = bytePool.buffer;
488       intUptos[intUptoStart+stream] = offset + bytePool.byteOffset;
489     }
490     bytes[offset] = b;
491     (intUptos[intUptoStart+stream])++;
492   }

501   void writeVInt(int stream, int i) {
502     assert stream < streamCount;
503     while ((i & ~0x7F) != 0) {
504       writeByte(stream, (byte)((i & 0x7f) | 0x80));
505       i >>>= 7;
506     }
507     writeByte(stream, (byte) i);
508   }

516   /** Called when postings hash is too small (> 50%
517    *  occupied) or too large (< 20% occupied). */
518   void rehashPostings(final int newSize) {
519 
520     final int newMask = newSize-1;
521 
522     RawPostingList[] newHash = new RawPostingList[newSize];
523     for(int i=0;i<postingsHashSize;i++) {
524       RawPostingList p0 = postingsHash[i];
525       if (p0 != null) {
526         int code;
527         if (perThread.primary) {
528           final int start = p0.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
529           final char[] text = charPool.buffers[p0.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
530           int pos = start;
531           while(text[pos] != 0xffff)
532             pos++;
533           code = 0;
534           while (pos > start)
535             code = (code*31) + text[--pos];
536         } else
537           code = p0.textStart;
538 
539         int hashPos = code & newMask;
540         assert hashPos >= 0;
541         if (newHash[hashPos] != null) {
542           final int inc = ((code>>8)+code)|1;
543           do {
544             code += inc;
545             hashPos = code & newMask;
546           } while (newHash[hashPos] != null);
547         }
548         newHash[hashPos] = p0;
549       }
550     }
551 
552     postingsHashMask = newMask;
553     postingsHash = newHash;
554     postingsHashSize = newSize;
555     postingsHashHalfSize = newSize >> 1;
556   }
