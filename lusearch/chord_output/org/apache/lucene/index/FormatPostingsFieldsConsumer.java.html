20  import java.io.IOException;
21  
22  /** Abstract API that consumes terms, doc, freq, prox and
23   *  payloads postings.  Concrete implementations of this
24   *  actually do "something" with the postings (write it into
25   *  the index in a specific format).
26   *
27   * NOTE: this API is experimental and will likely change
28   */
29  abstract class FormatPostingsFieldsConsumer {
30  
31    /** Add a new field */
32    abstract FormatPostingsTermsConsumer addField(FieldInfo field) throws IOException;
33  
34    /** Called when we are done adding everything. */
35    abstract void finish() throws IOException;
36  }
