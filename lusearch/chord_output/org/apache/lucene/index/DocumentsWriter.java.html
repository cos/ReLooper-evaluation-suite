44  /**
45   * This class accepts multiple added documents and directly
46   * writes a single segment file.  It does this more
47   * efficiently than creating a single segment per document
48   * (with DocumentWriter) and doing standard merges on those
49   * segments.
50   *
51   * Each added document is passed to the {@link DocConsumer},
52   * which in turn processes the document and interacts with
53   * other consumers in the indexing chain.  Certain
54   * consumers, like {@link StoredFieldsWriter} and {@link
55   * TermVectorsTermsWriter}, digest a document and
56   * immediately write bytes to the "doc store" files (ie,
57   * they do not consume RAM per document, except while they
58   * are processing the document).
59   *
60   * Other consumers, eg {@link FreqProxTermsWriter} and
61   * {@link NormsWriter}, buffer bytes in RAM and flush only
62   * when a new segment is produced.
63  
64   * Once we have used our allowed RAM buffer, or the number
65   * of added docs is large enough (in the case we are
66   * flushing by doc count instead of RAM usage), we create a
67   * real segment and flush it to the Directory.
68   *
69   * Threads:
70   *
71   * Multiple threads are allowed into addDocument at once.
72   * There is an initial synchronized call to getThreadState
73   * which allocates a ThreadState for this thread.  The same
74   * thread will get the same ThreadState over time (thread
75   * affinity) so that if there are consistent patterns (for
76   * example each thread is indexing a different content
77   * source) then we make better use of RAM.  Then
78   * processDocument is called on that ThreadState without
79   * synchronization (most of the "heavy lifting" is in this
80   * call).  Finally the synchronized "finishDocument" is
81   * called to flush changes to the directory.
82   *
83   * When flush is called by IndexWriter, or, we flush
84   * internally when autoCommit=false, we forcefully idle all
85   * threads and flush only once they are all idle.  This
86   * means you can call flush with a given thread even while
87   * other threads are actively adding/deleting documents.
88   *
89   *
90   * Exceptions:
91   *
92   * Because this class directly updates in-memory posting
93   * lists, and flushes stored fields and term vectors
94   * directly to files in the directory, there are certain
95   * limited times when an exception can corrupt this state.
96   * For example, a disk full while flushing stored fields
97   * leaves this file in a corrupt state.  Or, an OOM
98   * exception while appending to the in-memory posting lists
99   * can corrupt that posting list.  We call such exceptions
100  * "aborting exceptions".  In these cases we must call
101  * abort() to discard all docs added since the last flush.
102  *
103  * All other exceptions ("non-aborting exceptions") can
104  * still partially update the index structures.  These
105  * updates are consistent, but, they represent only a part
106  * of the document seen up until the exception was hit.
107  * When this happens, we immediately mark the document as
108  * deleted so that the document is always atomically ("all
109  * or none") added to the index.
110  */

188   static final IndexingChain DefaultIndexingChain = new IndexingChain() {
189 
190     DocConsumer getChain(DocumentsWriter documentsWriter) {
191       /*
192       This is the current indexing chain:
193 
194       DocConsumer / DocConsumerPerThread
195         --> code: DocFieldProcessor / DocFieldProcessorPerThread
196           --> DocFieldConsumer / DocFieldConsumerPerThread / DocFieldConsumerPerField
197             --> code: DocFieldConsumers / DocFieldConsumersPerThread / DocFieldConsumersPerField
198               --> code: DocInverter / DocInverterPerThread / DocInverterPerField
199                 --> InvertedDocConsumer / InvertedDocConsumerPerThread / InvertedDocConsumerPerField
200                   --> code: TermsHash / TermsHashPerThread / TermsHashPerField
201                     --> TermsHashConsumer / TermsHashConsumerPerThread / TermsHashConsumerPerField
202                       --> code: FreqProxTermsWriter / FreqProxTermsWriterPerThread / FreqProxTermsWriterPerField
203                       --> code: TermVectorsTermsWriter / TermVectorsTermsWriterPerThread / TermVectorsTermsWriterPerField
204                 --> InvertedDocEndConsumer / InvertedDocConsumerPerThread / InvertedDocConsumerPerField
205                   --> code: NormsWriter / NormsWriterPerThread / NormsWriterPerField
206               --> code: StoredFieldsWriter / StoredFieldsWriterPerThread / StoredFieldsWriterPerField
207     */
208 
209     // Build up indexing chain:
210 
211       final TermsHashConsumer termVectorsWriter = new TermVectorsTermsWriter(documentsWriter);
212       final TermsHashConsumer freqProxWriter = new FreqProxTermsWriter();
213 
214       final InvertedDocConsumer  termsHash = new TermsHash(documentsWriter, true, freqProxWriter,
215                                                            new TermsHash(documentsWriter, false, termVectorsWriter, null));
216       final NormsWriter normsWriter = new NormsWriter();
217       final DocInverter docInverter = new DocInverter(termsHash, normsWriter);
218       return new DocFieldProcessor(documentsWriter, docInverter);
219     }
220   };

365   /** Closes the current open doc stores an returns the doc
366    *  store segment name.  This returns null if there are *
367    *  no buffered documents. */
368   synchronized String closeDocStore() throws IOException {
369     
370     assert allThreadsIdle();
371 
372     if (infoStream != null)
373       message("closeDocStore: " + openFiles.size() + " files to flush to segment " + docStoreSegment + " numDocs=" + numDocsInStore);
374     
375     boolean success = false;
376 
377     try {
378       initFlushState(true);
379       closedFiles.clear();
380 
381       consumer.closeDocStore(flushState);
382       assert 0 == openFiles.size();
383 
384       String s = docStoreSegment;
385       docStoreSegment = null;
386       docStoreOffset = 0;
387       numDocsInStore = 0;
388       success = true;
389       return s;
390     } finally {
391       if (!success) {
392         abort();
393       }
394     }
395   }

438   /** Called if we hit an exception at a bad time (when
439    *  updating the index files) and must discard all
440    *  currently buffered docs.  This resets our state,
441    *  discarding any docs added since last flush. */
442   synchronized void abort() throws IOException {
443 
444     try {
445       if (infoStream != null)
446         message("docWriter: now abort");
447 
448       // Forcefully remove waiting ThreadStates from line
449       waitQueue.abort();
450 
451       // Wait for all other threads to finish with
452       // DocumentsWriter:
453       pauseAllThreads();
454 
455       try {
456 
457         assert 0 == waitQueue.numWaiting;
458 
459         waitQueue.waitingBytes = 0;
460 
461         try {
462           abortedFiles = openFiles();
463         } catch (Throwable t) {
464           abortedFiles = null;
465         }
466 
467         deletesInRAM.clear();
468 
469         openFiles.clear();
470 
471         for(int i=0;i<threadStates.length;i++)
472           try {
473             threadStates[i].consumer.abort();
474           } catch (Throwable t) {
475           }
476 
477         try {
478           consumer.abort();
479         } catch (Throwable t) {
480         }
481 
482         docStoreSegment = null;
483         numDocsInStore = 0;
484         docStoreOffset = 0;
485 
486         // Reset all postings data
487         doAfterFlush();
488 
489       } finally {
490         resumeAllThreads();
491       }
492     } finally {
493       aborting = false;
494       notifyAll();
495     }
496   }
497 
498   /** Reset after a flush */
499   private void doAfterFlush() throws IOException {
500     // All ThreadStates should be idle when we are called
501     assert allThreadsIdle();
502     threadBindings.clear();
503     waitQueue.reset();
504     segment = null;
505     numDocsInRAM = 0;
506     nextDocID = 0;
507     bufferIsFull = false;
508     flushPending = false;
509     for(int i=0;i<threadStates.length;i++)
510       threadStates[i].doAfterFlush();
511     numBytesUsed = 0;
512   }
513 
514   // Returns true if an abort is in progress
515   synchronized boolean pauseAllThreads() {
516     pauseThreads++;
517     while(!allThreadsIdle()) {
518       try {
519         wait();
520       } catch (InterruptedException ie) {
521         // In 3.0 we will change this to throw
522         // InterruptedException instead
523         Thread.currentThread().interrupt();
524         throw new RuntimeException(ie);
525       }
526     }
527 
528     return aborting;
529   }
530 
531   synchronized void resumeAllThreads() {
532     pauseThreads--;
533     assert pauseThreads >= 0;
534     if (0 == pauseThreads)
535       notifyAll();
536   }

557   /** Flush all pending docs to a new segment */
558   synchronized int flush(boolean closeDocStore) throws IOException {
559 
560     assert allThreadsIdle();
561 
562     assert numDocsInRAM > 0;
563 
564     assert nextDocID == numDocsInRAM;
565     assert waitQueue.numWaiting == 0;
566     assert waitQueue.waitingBytes == 0;
567 
568     initFlushState(false);
569 
570     docStoreOffset = numDocsInStore;
571 
572     if (infoStream != null)
573       message("flush postings as segment " + flushState.segmentName + " numDocs=" + numDocsInRAM);
574     
575     boolean success = false;
576 
577     try {
578 
579       if (closeDocStore) {
580         assert flushState.docStoreSegmentName != null;
581         assert flushState.docStoreSegmentName.equals(flushState.segmentName);
582         closeDocStore();
583         flushState.numDocsInStore = 0;
584       }
585 
586       Collection threads = new HashSet();
587       for(int i=0;i<threadStates.length;i++)
588         threads.add(threadStates[i].consumer);
589       consumer.flush(threads, flushState);
590 
591       if (infoStream != null) {
592         SegmentInfo si = new SegmentInfo(flushState.segmentName, flushState.numDocs, directory);
593         final long newSegmentSize = si.sizeInBytes();
594         String message = "  oldRAMSize=" + numBytesUsed +
595           " newFlushedSize=" + newSegmentSize +
596           " docs/MB=" + nf.format(numDocsInRAM/(newSegmentSize/1024./1024.)) +
597           " new/old=" + nf.format(100.0*newSegmentSize/numBytesUsed) + "%";
598         message(message);
599       }
600 
601       flushedDocCount += flushState.numDocs;
602 
603       doAfterFlush();
604 
605       success = true;
606 
607     } finally {
608       if (!success) {
609         abort();
610       }
611     }
612 
613     assert waitQueue.waitingBytes == 0;
614 
615     return flushState.numDocs;
616   }

627     cfsWriter.close();
628   }

667   /** Returns a free (idle) ThreadState that may be used for
668    * indexing this one document.  This call also pauses if a
669    * flush is pending.  If delTerm is non-null then we
670    * buffer this deleted term after the thread state has
671    * been acquired. */
672   synchronized DocumentsWriterThreadState getThreadState(Document doc, Term delTerm) throws IOException {
673 
674     // First, find a thread state.  If this thread already
675     // has affinity to a specific ThreadState, use that one
676     // again.
677     DocumentsWriterThreadState state = (DocumentsWriterThreadState) threadBindings.get(Thread.currentThread());
678     if (state == null) {
679 
680       // First time this thread has called us since last
681       // flush.  Find the least loaded thread state:
682       DocumentsWriterThreadState minThreadState = null;
683       for(int i=0;i<threadStates.length;i++) {
684         DocumentsWriterThreadState ts = threadStates[i];
685         if (minThreadState == null || ts.numThreads < minThreadState.numThreads)
686           minThreadState = ts;
687       }
688       if (minThreadState != null && (minThreadState.numThreads == 0 || threadStates.length >= MAX_THREAD_STATE)) {
689         state = minThreadState;
690         state.numThreads++;
691       } else {
692         // Just create a new "private" thread state
693         DocumentsWriterThreadState[] newArray = new DocumentsWriterThreadState[1+threadStates.length];
694         if (threadStates.length > 0)
695           System.arraycopy(threadStates, 0, newArray, 0, threadStates.length);
696         state = newArray[threadStates.length] = new DocumentsWriterThreadState(this);
697         threadStates = newArray;
698       }
699       threadBindings.put(Thread.currentThread(), state);
700     }
701 
702     // Next, wait until my thread state is idle (in case
703     // it's shared with other threads) and for threads to
704     // not be paused nor a flush pending:
705     waitReady(state);
706 
707     // Allocate segment name if this is the first doc since
708     // last flush:
709     initSegmentName(false);
710 
711     state.isIdle = false;
712 
713     boolean success = false;
714     try {
715       state.docState.docID = nextDocID;
716 
717       assert writer.testPoint("DocumentsWriter.ThreadState.init start");
718 
719       if (delTerm != null) {
720         addDeleteTerm(delTerm, state.docState.docID);
721         state.doFlushAfter = timeToFlushDeletes();
722       }
723 
724       assert writer.testPoint("DocumentsWriter.ThreadState.init after delTerm");
725 
726       nextDocID++;
727       numDocsInRAM++;
728 
729       // We must at this point commit to flushing to ensure we
730       // always get N docs when we flush by doc count, even if
731       // > 1 thread is adding documents:
732       if (!flushPending &&
733           maxBufferedDocs != IndexWriter.DISABLE_AUTO_FLUSH
734           && numDocsInRAM >= maxBufferedDocs) {
735         flushPending = true;
736         state.doFlushAfter = true;
737       }
738 
739       success = true;
740     } finally {
741       if (!success) {
742         // Forcefully idle this ThreadState:
743         state.isIdle = true;
744         notifyAll();
745         if (state.doFlushAfter) {
746           state.doFlushAfter = false;
747           flushPending = false;
748         }
749       }
750     }
751 
752     return state;
753   }

767   boolean updateDocument(Document doc, Analyzer analyzer, Term delTerm)
768     throws CorruptIndexException, IOException {
769 
770     // This call is synchronized but fast
771     final DocumentsWriterThreadState state = getThreadState(doc, delTerm);
772 
773     final DocState docState = state.docState;
774     docState.doc = doc;
775     docState.analyzer = analyzer;
776 
777     boolean success = false;
778     try {
779       // This call is not synchronized and does all the
780       // work
781       final DocWriter perDoc = state.consumer.processDocument();
782         
783       // This call is synchronized but fast
784       finishDocument(state, perDoc);
785       success = true;
786     } finally {
787       if (!success) {
788         synchronized(this) {
789 
790           if (aborting) {
791             state.isIdle = true;
792             notifyAll();
793             abort();
794           } else {
795             skipDocWriter.docID = docState.docID;
796             boolean success2 = false;
797             try {
798               waitQueue.add(skipDocWriter);
799               success2 = true;
800             } finally {
801               if (!success2) {
802                 state.isIdle = true;
803                 notifyAll();
804                 abort();
805                 return false;
806               }
807             }
808 
809             state.isIdle = true;
810             notifyAll();
811 
812             // If this thread state had decided to flush, we
813             // must clear it so another thread can flush
814             if (state.doFlushAfter) {
815               state.doFlushAfter = false;
816               flushPending = false;
817               notifyAll();
818             }
819 
820             // Immediately mark this document as deleted
821             // since likely it was partially added.  This
822             // keeps indexing as "all or none" (atomic) when
823             // adding a document:
824             addDeleteDocID(state.docState.docID);
825           }
826         }
827       }
828     }
829 
830     return state.doFlushAfter || timeToFlushDeletes();
831   }

982   // Apply buffered delete terms, queries and docIDs to the
983   // provided reader
984   private final synchronized boolean applyDeletes(IndexReader reader, int docIDStart)
985     throws CorruptIndexException, IOException {
986 
987     final int docEnd = docIDStart + reader.maxDoc();
988     boolean any = false;
989 
990     assert checkDeleteTerm(null);
991 
992     // Delete by term
993     Iterator iter = deletesFlushed.terms.entrySet().iterator();
994     TermDocs docs = reader.termDocs();
995     try {
996       while (iter.hasNext()) {
997         Entry entry = (Entry) iter.next();
998         Term term = (Term) entry.getKey();
999         // LUCENE-2086: we should be iterating a TreeMap,
1000        // here, so terms better be in order:
1001        assert checkDeleteTerm(term);
1002        docs.seek(term);
1003        int limit = ((BufferedDeletes.Num) entry.getValue()).getNum();
1004        while (docs.next()) {
1005          int docID = docs.doc();
1006          if (docIDStart+docID >= limit)
1007            break;
1008          reader.deleteDocument(docID);
1009          any = true;
1010        }
1011      }
1012    } finally {
1013      docs.close();
1014    }
1015
1016    // Delete by docID
1017    iter = deletesFlushed.docIDs.iterator();
1018    while(iter.hasNext()) {
1019      int docID = ((Integer) iter.next()).intValue();
1020      if (docID >= docIDStart && docID < docEnd) {
1021        reader.deleteDocument(docID-docIDStart);
1022        any = true;
1023      }
1024    }
1025
1026    // Delete by query
1027    IndexSearcher searcher = new IndexSearcher(reader);
1028    iter = deletesFlushed.queries.entrySet().iterator();
1029    while(iter.hasNext()) {
1030      Entry entry = (Entry) iter.next();
1031      Query query = (Query) entry.getKey();
1032      int limit = ((Integer) entry.getValue()).intValue();
1033      Weight weight = query.weight(searcher);
1034      Scorer scorer = weight.scorer(reader, true, false);
1035      if (scorer != null) {
1036        while(true)  {
1037          int doc = scorer.nextDoc();
1038          if (((long) docIDStart) + doc >= limit)
1039            break;
1040          reader.deleteDocument(doc);
1041          any = true;
1042        }
1043      }
1044    }
1045    searcher.close();
1046    return any;
1047  }

1081  /** Does the synchronized work to finish/flush the
1082   *  inverted document. */
1083  private void finishDocument(DocumentsWriterThreadState perThread, DocWriter docWriter) throws IOException {
1084
1085    if (doBalanceRAM())
1086      // Must call this w/o holding synchronized(this) else
1087      // we'll hit deadlock:
1088      balanceRAM();
1089
1090    synchronized(this) {
1091
1092      assert docWriter == null || docWriter.docID == perThread.docState.docID;
1093
1094      if (aborting) {
1095
1096        // We are currently aborting, and another thread is
1097        // waiting for me to become idle.  We just forcefully
1098        // idle this threadState; it will be fully reset by
1099        // abort()
1100        if (docWriter != null)
1101          try {
1102            docWriter.abort();
1103          } catch (Throwable t) {
1104          }
1105
1106        perThread.isIdle = true;
1107        notifyAll();
1108        return;
1109      }
1110
1111      final boolean doPause;
1112
1113      if (docWriter != null)
1114        doPause = waitQueue.add(docWriter);
1115      else {
1116        skipDocWriter.docID = perThread.docState.docID;
1117        doPause = waitQueue.add(skipDocWriter);
1118      }
1119
1120      if (doPause)
1121        waitForWaitQueue();
1122
1123      if (bufferIsFull && !flushPending) {
1124        flushPending = true;
1125        perThread.doFlushAfter = true;
1126      }
1127
1128      perThread.isIdle = true;
1129      notifyAll();
1130    }
1131  }

1324  /* We have three pools of RAM: Postings, byte blocks
1325   * (holds freq/prox posting data) and char blocks (holds
1326   * characters in the term).  Different docs require
1327   * varying amount of storage from these three classes.
1328   * For example, docs with many unique single-occurrence
1329   * short terms will use up the Postings RAM and hardly any
1330   * of the other two.  Whereas docs with very large terms
1331   * will use alot of char blocks RAM and relatively less of
1332   * the other two.  This method just frees allocations from
1333   * the pools once we are over-budget, which balances the
1334   * pools to match the current docs. */
1335  void balanceRAM() {
1336
1337    // We flush when we've used our target usage
1338    final long flushTrigger = ramBufferSize;
1339
1340    final long deletesRAMUsed = deletesInRAM.bytesUsed+deletesFlushed.bytesUsed;
1341
1342    if (numBytesAlloc+deletesRAMUsed > freeTrigger) {
1343
1344      if (infoStream != null)
1345        message("  RAM: now balance allocations: usedMB=" + toMB(numBytesUsed) +
1346                " vs trigger=" + toMB(flushTrigger) +
1347                " allocMB=" + toMB(numBytesAlloc) +
1348                " deletesMB=" + toMB(deletesRAMUsed) +
1349                " vs trigger=" + toMB(freeTrigger) +
1350                " byteBlockFree=" + toMB(byteBlockAllocator.freeByteBlocks.size()*BYTE_BLOCK_SIZE) +
1351                " charBlockFree=" + toMB(freeCharBlocks.size()*CHAR_BLOCK_SIZE*CHAR_NUM_BYTE));
1352
1353      final long startBytesAlloc = numBytesAlloc + deletesRAMUsed;
1354
1355      int iter = 0;
1356
1357      // We free equally from each pool in 32 KB
1358      // chunks until we are below our threshold
1359      // (freeLevel)
1360
1361      boolean any = true;
1362
1363      while(numBytesAlloc+deletesRAMUsed > freeLevel) {
1364      
1365        synchronized(this) {
1366          if (0 == byteBlockAllocator.freeByteBlocks.size() && 0 == freeCharBlocks.size() && 0 == freeIntBlocks.size() && !any) {
1367            // Nothing else to free -- must flush now.
1368            bufferIsFull = numBytesUsed+deletesRAMUsed > flushTrigger;
1369            if (infoStream != null) {
1370              if (numBytesUsed > flushTrigger)
1371                message("    nothing to free; now set bufferIsFull");
1372              else
1373                message("    nothing to free");
1374            }
1375            assert numBytesUsed <= numBytesAlloc;
1376            break;
1377          }
1378
1379          if ((0 == iter % 4) && byteBlockAllocator.freeByteBlocks.size() > 0) {
1380            byteBlockAllocator.freeByteBlocks.remove(byteBlockAllocator.freeByteBlocks.size()-1);
1381            numBytesAlloc -= BYTE_BLOCK_SIZE;
1382          }
1383
1384          if ((1 == iter % 4) && freeCharBlocks.size() > 0) {
1385            freeCharBlocks.remove(freeCharBlocks.size()-1);
1386            numBytesAlloc -= CHAR_BLOCK_SIZE * CHAR_NUM_BYTE;
1387          }
1388
1389          if ((2 == iter % 4) && freeIntBlocks.size() > 0) {
1390            freeIntBlocks.remove(freeIntBlocks.size()-1);
1391            numBytesAlloc -= INT_BLOCK_SIZE * INT_NUM_BYTE;
1392          }
1393        }
1394
1395        if ((3 == iter % 4) && any)
1396          // Ask consumer to free any recycled state
1397          any = consumer.freeRAM();
1398
1399        iter++;
1400      }
1401
1402      if (infoStream != null)
1403        message("    after free: freedMB=" + nf.format((startBytesAlloc-numBytesAlloc-deletesRAMUsed)/1024./1024.) + " usedMB=" + nf.format((numBytesUsed+deletesRAMUsed)/1024./1024.) + " allocMB=" + nf.format(numBytesAlloc/1024./1024.));
1404      
1405    } else {
1406      // If we have not crossed the 100% mark, but have
1407      // crossed the 95% mark of RAM we are actually
1408      // using, go ahead and flush.  This prevents
1409      // over-allocating and then freeing, with every
1410      // flush.
1411      synchronized(this) {
1412
1413        if (numBytesUsed+deletesRAMUsed > flushTrigger) {
1414          if (infoStream != null)
1415            message("  RAM: now flush @ usedMB=" + nf.format(numBytesUsed/1024./1024.) +
1416                    " allocMB=" + nf.format(numBytesAlloc/1024./1024.) +
1417                    " deletesMB=" + nf.format(deletesRAMUsed/1024./1024.) +
1418                    " triggerMB=" + nf.format(flushTrigger/1024./1024.));
1419
1420          bufferIsFull = true;
1421        }
1422      }
1423    }
1424  }

1426  final WaitQueue waitQueue = new WaitQueue();
1427
1428  private class WaitQueue {
1429    DocWriter[] waiting;
1430    int nextWriteDocID;
1431    int nextWriteLoc;
1432    int numWaiting;
1433    long waitingBytes;
1434
1435    public WaitQueue() {
1436      waiting = new DocWriter[10];
1437    }
1438
1439    synchronized void reset() {
1440      // NOTE: nextWriteLoc doesn't need to be reset
1441      assert numWaiting == 0;
1442      assert waitingBytes == 0;
1443      nextWriteDocID = 0;
1444    }
1445
1446    synchronized boolean doResume() {
1447      return waitingBytes <= waitQueueResumeBytes;
1448    }
1449
1450    synchronized boolean doPause() {
1451      return waitingBytes > waitQueuePauseBytes;
1452    }
1453
1454    synchronized void abort() {
1455      int count = 0;
1456      for(int i=0;i<waiting.length;i++) {
1457        final DocWriter doc = waiting[i];
1458        if (doc != null) {
1459          doc.abort();
1460          waiting[i] = null;
1461          count++;
1462        }
1463      }
1464      waitingBytes = 0;
1465      assert count == numWaiting;
1466      numWaiting = 0;
1467    }
1468
1469    private void writeDocument(DocWriter doc) throws IOException {
1470      assert doc == skipDocWriter || nextWriteDocID == doc.docID;
1471      boolean success = false;
1472      try {
1473        doc.finish();
1474        nextWriteDocID++;
1475        numDocsInStore++;
1476        nextWriteLoc++;
1477        assert nextWriteLoc <= waiting.length;
1478        if (nextWriteLoc == waiting.length)
1479          nextWriteLoc = 0;
1480        success = true;
1481      } finally {
1482        if (!success)
1483          setAborting();
1484      }
1485    }
1486
1487    synchronized public boolean add(DocWriter doc) throws IOException {
1488
1489      assert doc.docID >= nextWriteDocID;
1490
1491      if (doc.docID == nextWriteDocID) {
1492        writeDocument(doc);
1493        while(true) {
1494          doc = waiting[nextWriteLoc];
1495          if (doc != null) {
1496            numWaiting--;
1497            waiting[nextWriteLoc] = null;
1498            waitingBytes -= doc.sizeInBytes();
1499            writeDocument(doc);
1500          } else
1501            break;
1502        }
1503      } else {
1504
1505        // I finished before documents that were added
1506        // before me.  This can easily happen when I am a
1507        // small doc and the docs before me were large, or,
1508        // just due to luck in the thread scheduling.  Just
1509        // add myself to the queue and when that large doc
1510        // finishes, it will flush me:
1511        int gap = doc.docID - nextWriteDocID;
1512        if (gap >= waiting.length) {
1513          // Grow queue
1514          DocWriter[] newArray = new DocWriter[ArrayUtil.getNextSize(gap)];
1515          assert nextWriteLoc >= 0;
1516          System.arraycopy(waiting, nextWriteLoc, newArray, 0, waiting.length-nextWriteLoc);
1517          System.arraycopy(waiting, 0, newArray, waiting.length-nextWriteLoc, nextWriteLoc);
1518          nextWriteLoc = 0;
1519          waiting = newArray;
1520          gap = doc.docID - nextWriteDocID;
1521        }
1522
1523        int loc = nextWriteLoc + gap;
1524        if (loc >= waiting.length)
1525          loc -= waiting.length;
1526
1527        // We should only wrap one time
1528        assert loc < waiting.length;
1529
1530        // Nobody should be in my spot!
1531        assert waiting[loc] == null;
1532        waiting[loc] = doc;
1533        numWaiting++;
1534        waitingBytes += doc.sizeInBytes();
1535      }
1536      
1537      return doPause();
1538    }
1539  }
