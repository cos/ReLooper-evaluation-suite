20  import java.io.IOException;
21  
22  final class TermsHashPerThread extends InvertedDocConsumerPerThread {
23  
24    final TermsHash termsHash;
25    final TermsHashConsumerPerThread consumer;
26    final TermsHashPerThread nextPerThread;
27  
28    final CharBlockPool charPool;
29    final IntBlockPool intPool;
30    final ByteBlockPool bytePool;
31    final boolean primary;
32    final DocumentsWriter.DocState docState;
33  
34    final RawPostingList freePostings[] = new RawPostingList[256];
35    int freePostingsCount;
36  
37    public TermsHashPerThread(DocInverterPerThread docInverterPerThread, final TermsHash termsHash, final TermsHash nextTermsHash, final TermsHashPerThread primaryPerThread) {
38      docState = docInverterPerThread.docState;
39  
40      this.termsHash = termsHash;
41      this.consumer = termsHash.consumer.addThread(this);
42  
43      if (nextTermsHash != null) {
44        // We are primary
45        charPool = new CharBlockPool(termsHash.docWriter);
46        primary = true;
47      } else {
48        charPool = primaryPerThread.charPool;
49        primary = false;
50      }
51  
52      intPool = new IntBlockPool(termsHash.docWriter, termsHash.trackAllocations);
53      bytePool = new ByteBlockPool(termsHash.docWriter.byteBlockAllocator, termsHash.trackAllocations);
54  
55      if (nextTermsHash != null)
56        nextPerThread = nextTermsHash.addThread(docInverterPerThread, this);
57      else
58        nextPerThread = null;
59    }
60  
61    InvertedDocConsumerPerField addField(DocInverterPerField docInverterPerField, final FieldInfo fieldInfo) {
62      return new TermsHashPerField(docInverterPerField, this, nextPerThread, fieldInfo);
63    }
64  
65    synchronized public void abort() {
66      reset(true);
67      consumer.abort();
68      if (nextPerThread != null)
69        nextPerThread.abort();
70    }
71  
72    // perField calls this when it needs more postings:
73    void morePostings() throws IOException {
74      assert freePostingsCount == 0;
75      termsHash.getPostings(freePostings);
76      freePostingsCount = freePostings.length;
77      assert noNullPostings(freePostings, freePostingsCount, "consumer=" + consumer);
78    }
79  
80    private static boolean noNullPostings(RawPostingList[] postings, int count, String details) {
81      for(int i=0;i<count;i++)
82        assert postings[i] != null: "postings[" + i + "] of " + count + " is null: " + details;
83      return true;
84    }
85  
86    public void startDocument() throws IOException {
87      consumer.startDocument();
88      if (nextPerThread != null)
89        nextPerThread.consumer.startDocument();
90    }
91  
92    public DocumentsWriter.DocWriter finishDocument() throws IOException {
93      final DocumentsWriter.DocWriter doc = consumer.finishDocument();
94  
95      final DocumentsWriter.DocWriter doc2;
96      if (nextPerThread != null)
97        doc2 = nextPerThread.consumer.finishDocument();
98      else
99        doc2 = null;
100     if (doc == null)
101       return doc2;
102     else {
103       doc.setNext(doc2);
104       return doc;
105     }
106   }
107 
108   // Clear all state
109   void reset(boolean recyclePostings) {
110     intPool.reset();
111     bytePool.reset();
112 
113     if (primary)
114       charPool.reset();
115 
116     if (recyclePostings) {
117       termsHash.recyclePostings(freePostings, freePostingsCount);
118       freePostingsCount = 0;
119     }
120   }
121 }
