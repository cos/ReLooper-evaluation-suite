20  import java.io.IOException;
21  
22  /** <p>Expert: {@link IndexWriter} uses an instance
23   *  implementing this interface to execute the merges
24   *  selected by a {@link MergePolicy}.  The default
25   *  MergeScheduler is {@link ConcurrentMergeScheduler}.</p>
26   *
27   * <p><b>NOTE:</b> This API is new and still experimental
28   * (subject to change suddenly in the next release)</p>
29   *
30   * <p><b>NOTE</b>: This class typically requires access to
31   * package-private APIs (eg, SegmentInfos) to do its job;
32   * if you implement your own MergePolicy, you'll need to put
33   * it in package org.apache.lucene.index in order to use
34   * these APIs.
35  */
36  
37  public abstract class MergeScheduler {
38  
39    /** Run the merges provided by {@link IndexWriter#getNextMerge()}. */
40    abstract void merge(IndexWriter writer)
41      throws CorruptIndexException, IOException;
42  
43    /** Close this MergeScheduler. */
44    abstract void close()
45      throws CorruptIndexException, IOException;
46  }
