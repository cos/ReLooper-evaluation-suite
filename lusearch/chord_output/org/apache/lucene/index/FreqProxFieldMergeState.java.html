20  import java.io.IOException;
21  
22  // TODO FI: some of this is "generic" to TermsHash* so we
23  // should factor it out so other consumers don't have to
24  // duplicate this code
25  
26  /** Used by DocumentsWriter to merge the postings from
27   *  multiple ThreadStates when creating a segment */
28  final class FreqProxFieldMergeState {
29  
30    final FreqProxTermsWriterPerField field;
31    final int numPostings;
32    final CharBlockPool charPool;
33    final RawPostingList[] postings;
34  
35    private FreqProxTermsWriter.PostingList p;
36    char[] text;
37    int textOffset;
38  
39    private int postingUpto = -1;
40  
41    final ByteSliceReader freq = new ByteSliceReader();
42    final ByteSliceReader prox = new ByteSliceReader();
43  
44    int docID;
45    int termFreq;
46  
47    public FreqProxFieldMergeState(FreqProxTermsWriterPerField field) {
48      this.field = field;
49      this.charPool = field.perThread.termsHashPerThread.charPool;
50      this.numPostings = field.termsHashPerField.numPostings;
51      this.postings = field.termsHashPerField.sortPostings();
52    }
53  
54    boolean nextTerm() throws IOException {
55      postingUpto++;
56      if (postingUpto == numPostings)
57        return false;
58  
59      p = (FreqProxTermsWriter.PostingList) postings[postingUpto];
60      docID = 0;
61  
62      text = charPool.buffers[p.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
63      textOffset = p.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
64  
65      field.termsHashPerField.initReader(freq, p, 0);
66      if (!field.fieldInfo.omitTermFreqAndPositions)
67        field.termsHashPerField.initReader(prox, p, 1);
68  
69      // Should always be true
70      boolean result = nextDoc();
71      assert result;
72  
73      return true;
74    }
75  
76    public boolean nextDoc() throws IOException {
77      if (freq.eof()) {
78        if (p.lastDocCode != -1) {
79          // Return last doc
80          docID = p.lastDocID;
81          if (!field.omitTermFreqAndPositions)
82            termFreq = p.docFreq;
83          p.lastDocCode = -1;
84          return true;
85        } else
86          // EOF
87          return false;
88      }
89  
90      final int code = freq.readVInt();
91      if (field.omitTermFreqAndPositions)
92        docID += code;
93      else {
94        docID += code >>> 1;
95        if ((code & 1) != 0)
96          termFreq = 1;
97        else
98          termFreq = freq.readVInt();
99      }
100 
101     assert docID != p.lastDocID;
102 
103     return true;
104   }
105 }
