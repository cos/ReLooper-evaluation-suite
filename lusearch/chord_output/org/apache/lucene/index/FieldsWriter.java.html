30  final class FieldsWriter
31  {
32    static final byte FIELD_IS_TOKENIZED = 0x1;
33    static final byte FIELD_IS_BINARY = 0x2;
34    static final byte FIELD_IS_COMPRESSED = 0x4;
35  
36    // Original format
37    static final int FORMAT = 0;
38  
39    // Changed strings to UTF8
40    static final int FORMAT_VERSION_UTF8_LENGTH_IN_BYTES = 1;
41  
42    // NOTE: if you introduce a new format, make it 1 higher
43    // than the current one, and always change this if you
44    // switch to a new format!
45    static final int FORMAT_CURRENT = FORMAT_VERSION_UTF8_LENGTH_IN_BYTES;
46    
47      private FieldInfos fieldInfos;
48  
49      private IndexOutput fieldsStream;
50  
51      private IndexOutput indexStream;
52  
53      private boolean doClose;
54  
55      FieldsWriter(Directory d, String segment, FieldInfos fn) throws IOException {
56          fieldInfos = fn;
57  
58          boolean success = false;
59          final String fieldsName = segment + "." + IndexFileNames.FIELDS_EXTENSION;
60          try {
61            fieldsStream = d.createOutput(fieldsName);
62            fieldsStream.writeInt(FORMAT_CURRENT);
63            success = true;
64          } finally {
65            if (!success) {
66              try {
67                close();
68              } catch (Throwable t) {
69                // Suppress so we keep throwing the original exception
70              }
71              try {
72                d.deleteFile(fieldsName);
73              } catch (Throwable t) {
74                // Suppress so we keep throwing the original exception
75              }
76            }
77          }
78  
79          success = false;
80          final String indexName = segment + "." + IndexFileNames.FIELDS_INDEX_EXTENSION;
81          try {
82            indexStream = d.createOutput(indexName);
83            indexStream.writeInt(FORMAT_CURRENT);
84            success = true;
85          } finally {
86            if (!success) {
87              try {
88                close();
89              } catch (IOException ioe) {
90              }
91              try {
92                d.deleteFile(fieldsName);
93              } catch (Throwable t) {
94                // Suppress so we keep throwing the original exception
95              }
96              try {
97                d.deleteFile(indexName);
98              } catch (Throwable t) {
99                // Suppress so we keep throwing the original exception
100             }
101           }
102         }
103 
104         doClose = true;
105     }

107     FieldsWriter(IndexOutput fdx, IndexOutput fdt, FieldInfos fn) {
108         fieldInfos = fn;
109         fieldsStream = fdt;
110         indexStream = fdx;
111         doClose = false;
112     }

118     // Writes the contents of buffer into the fields stream
119     // and adds a new entry for this document into the index
120     // stream.  This assumes the buffer was already written
121     // in the correct fields format.
122     void flushDocument(int numStoredFields, RAMOutputStream buffer) throws IOException {
123       indexStream.writeLong(fieldsStream.getFilePointer());
124       fieldsStream.writeVInt(numStoredFields);
125       buffer.writeTo(fieldsStream);
126     }
127 
128     void skipDocument() throws IOException {
129       indexStream.writeLong(fieldsStream.getFilePointer());
130       fieldsStream.writeVInt(0);
131     }
132 
133     void flush() throws IOException {
134       indexStream.flush();
135       fieldsStream.flush();
136     }

138     final void close() throws IOException {
139       if (doClose) {
140 
141         try {
142           if (fieldsStream != null) {
143             try {
144               fieldsStream.close();
145             } finally {
146               fieldsStream = null;
147             }
148           }
149         } catch (IOException ioe) {
150           try {
151             if (indexStream != null) {
152               try {
153                 indexStream.close();
154               } finally {
155                 indexStream = null;
156               }
157             }
158           } catch (IOException ioe2) {
159             // Ignore so we throw only first IOException hit
160           }
161           throw ioe;
162         } finally {
163           if (indexStream != null) {
164             try {
165               indexStream.close();
166             } finally {
167               indexStream = null;
168             }
169           }
170         }
171       }
172     }

174     final void writeField(FieldInfo fi, Fieldable field) throws IOException {
175       // if the field as an instanceof FieldsReader.FieldForMerge, we're in merge mode
176       // and field.binaryValue() already returns the compressed value for a field
177       // with isCompressed()==true, so we disable compression in that case
178       boolean disableCompression = (field instanceof FieldsReader.FieldForMerge);
179       fieldsStream.writeVInt(fi.number);
180       byte bits = 0;
181       if (field.isTokenized())
182         bits |= FieldsWriter.FIELD_IS_TOKENIZED;
183       if (field.isBinary())
184         bits |= FieldsWriter.FIELD_IS_BINARY;
185       if (field.isCompressed())
186         bits |= FieldsWriter.FIELD_IS_COMPRESSED;
187                 
188       fieldsStream.writeByte(bits);
189                 
190       if (field.isCompressed()) {
191         // compression is enabled for the current field
192         final byte[] data;
193         final int len;
194         final int offset;
195         if (disableCompression) {
196           // optimized case for merging, the data
197           // is already compressed
198           data = field.getBinaryValue();
199           assert data != null;
200           len = field.getBinaryLength();
201           offset = field.getBinaryOffset();  
202         } else {
203           // check if it is a binary field
204           if (field.isBinary()) {
205             data = CompressionTools.compress(field.getBinaryValue(), field.getBinaryOffset(), field.getBinaryLength());
206           } else {
207             byte x[] = field.stringValue().getBytes("UTF-8");
208             data = CompressionTools.compress(x, 0, x.length);
209           }
210           len = data.length;
211           offset = 0;
212         }
213         
214         fieldsStream.writeVInt(len);
215         fieldsStream.writeBytes(data, offset, len);
216       }
217       else {
218         // compression is disabled for the current field
219         if (field.isBinary()) {
220           final byte[] data;
221           final int len;
222           final int offset;
223           data = field.getBinaryValue();
224           len = field.getBinaryLength();
225           offset =  field.getBinaryOffset();
226 
227           fieldsStream.writeVInt(len);
228           fieldsStream.writeBytes(data, offset, len);
229         }
230         else {
231           fieldsStream.writeString(field.stringValue());
232         }
233       }
234     }

236     /** Bulk write a contiguous series of documents.  The
237      *  lengths array is the length (in bytes) of each raw
238      *  document.  The stream IndexInput is the
239      *  fieldsStream from which we should bulk-copy all
240      *  bytes. */
241     final void addRawDocuments(IndexInput stream, int[] lengths, int numDocs) throws IOException {
242       long position = fieldsStream.getFilePointer();
243       long start = position;
244       for(int i=0;i<numDocs;i++) {
245         indexStream.writeLong(position);
246         position += lengths[i];
247       }
248       fieldsStream.copyBytes(stream, position-start);
249       assert fieldsStream.getFilePointer() == position;
250     }
251 
252     final void addDocument(Document doc) throws IOException {
253         indexStream.writeLong(fieldsStream.getFilePointer());
254 
255         int storedCount = 0;
256         Iterator fieldIterator = doc.getFields().iterator();
257         while (fieldIterator.hasNext()) {
258             Fieldable field = (Fieldable) fieldIterator.next();
259             if (field.isStored())
260                 storedCount++;
261         }
262         fieldsStream.writeVInt(storedCount);
263 
264         fieldIterator = doc.getFields().iterator();
265         while (fieldIterator.hasNext()) {
266             Fieldable field = (Fieldable) fieldIterator.next();
267             if (field.isStored())
268               writeField(fieldInfos.fieldInfo(field.name()), field);
269         }
270     }
