28  /**
29   * Gathers all Fieldables for a document under the same
30   * name, updates FieldInfos, and calls per-field consumers
31   * to process field by field.
32   *
33   * Currently, only a single thread visits the fields,
34   * sequentially, for processing.
35   */
36  
37  final class DocFieldProcessorPerThread extends DocConsumerPerThread {
38  
39    float docBoost;
40    int fieldGen;
41    final DocFieldProcessor docFieldProcessor;
42    final FieldInfos fieldInfos;
43    final DocFieldConsumerPerThread consumer;
44  
45    // Holds all fields seen in current doc
46    DocFieldProcessorPerField[] fields = new DocFieldProcessorPerField[1];
47    int fieldCount;
48  
49    // Hash table for all fields ever seen
50    DocFieldProcessorPerField[] fieldHash = new DocFieldProcessorPerField[2];
51    int hashMask = 1;
52    int totalFieldCount;
53  
54    final StoredFieldsWriterPerThread fieldsWriter;
55  
56    final DocumentsWriter.DocState docState;
57  
58    public DocFieldProcessorPerThread(DocumentsWriterThreadState threadState, DocFieldProcessor docFieldProcessor) throws IOException {
59      this.docState = threadState.docState;
60      this.docFieldProcessor = docFieldProcessor;
61      this.fieldInfos = docFieldProcessor.fieldInfos;
62      this.consumer = docFieldProcessor.consumer.addThread(this);
63      fieldsWriter = docFieldProcessor.fieldsWriter.addThread(docState);
64    }
65  
66    public void abort() {
67      for(int i=0;i<fieldHash.length;i++) {
68        DocFieldProcessorPerField field = fieldHash[i];
69        while(field != null) {
70          final DocFieldProcessorPerField next = field.next;
71          field.abort();
72          field = next;
73        }
74      }
75      fieldsWriter.abort();
76      consumer.abort();
77    }
78  
79    public Collection fields() {
80      Collection fields = new HashSet();
81      for(int i=0;i<fieldHash.length;i++) {
82        DocFieldProcessorPerField field = fieldHash[i];
83        while(field != null) {
84          fields.add(field.consumer);
85          field = field.next;
86        }
87      }
88      assert fields.size() == totalFieldCount;
89      return fields;
90    }

92    /** If there are fields we've seen but did not see again
93     *  in the last run, then free them up. */
94  
95    void trimFields(SegmentWriteState state) {
96  
97      for(int i=0;i<fieldHash.length;i++) {
98        DocFieldProcessorPerField perField = fieldHash[i];
99        DocFieldProcessorPerField lastPerField = null;
100 
101       while (perField != null) {
102 
103         if (perField.lastGen == -1) {
104 
105           // This field was not seen since the previous
106           // flush, so, free up its resources now
107 
108           // Unhash
109           if (lastPerField == null)
110             fieldHash[i] = perField.next;
111           else
112             lastPerField.next = perField.next;
113 
114           if (state.docWriter.infoStream != null)
115             state.docWriter.infoStream.println("  purge field=" + perField.fieldInfo.name);
116 
117           totalFieldCount--;
118 
119         } else {
120           // Reset
121           perField.lastGen = -1;
122           lastPerField = perField;
123         }
124 
125         perField = perField.next;
126       }
127     }
128   }

130   private void rehash() {
131     final int newHashSize = (int) (fieldHash.length*2);
132     assert newHashSize > fieldHash.length;
133 
134     final DocFieldProcessorPerField newHashArray[] = new DocFieldProcessorPerField[newHashSize];
135 
136     // Rehash
137     int newHashMask = newHashSize-1;
138     for(int j=0;j<fieldHash.length;j++) {
139       DocFieldProcessorPerField fp0 = fieldHash[j];
140       while(fp0 != null) {
141         final int hashPos2 = fp0.fieldInfo.name.hashCode() & newHashMask;
142         DocFieldProcessorPerField nextFP0 = fp0.next;
143         fp0.next = newHashArray[hashPos2];
144         newHashArray[hashPos2] = fp0;
145         fp0 = nextFP0;
146       }
147     }
148 
149     fieldHash = newHashArray;
150     hashMask = newHashMask;
151   }

153   public DocumentsWriter.DocWriter processDocument() throws IOException {
154 
155     consumer.startDocument();
156     fieldsWriter.startDocument();
157 
158     final Document doc = docState.doc;
159 
160     assert docFieldProcessor.docWriter.writer.testPoint("DocumentsWriter.ThreadState.init start");
161 
162     fieldCount = 0;
163     
164     final int thisFieldGen = fieldGen++;
165 
166     final List docFields = doc.getFields();
167     final int numDocFields = docFields.size();
168 
169     // Absorb any new fields first seen in this document.
170     // Also absorb any changes to fields we had already
171     // seen before (eg suddenly turning on norms or
172     // vectors, etc.):
173 
174     for(int i=0;i<numDocFields;i++) {
175       Fieldable field = (Fieldable) docFields.get(i);
176       final String fieldName = field.name();
177 
178       // Make sure we have a PerField allocated
179       final int hashPos = fieldName.hashCode() & hashMask;
180       DocFieldProcessorPerField fp = fieldHash[hashPos];
181       while(fp != null && !fp.fieldInfo.name.equals(fieldName))
182         fp = fp.next;
183 
184       if (fp == null) {
185 
186         // TODO FI: we need to genericize the "flags" that a
187         // field holds, and, how these flags are merged; it
188         // needs to be more "pluggable" such that if I want
189         // to have a new "thing" my Fields can do, I can
190         // easily add it
191         FieldInfo fi = fieldInfos.add(fieldName, field.isIndexed(), field.isTermVectorStored(),
192                                       field.isStorePositionWithTermVector(), field.isStoreOffsetWithTermVector(),
193                                       field.getOmitNorms(), false, field.getOmitTf());
194 
195         fp = new DocFieldProcessorPerField(this, fi);
196         fp.next = fieldHash[hashPos];
197         fieldHash[hashPos] = fp;
198         totalFieldCount++;
199 
200         if (totalFieldCount >= fieldHash.length/2)
201           rehash();
202       } else
203         fp.fieldInfo.update(field.isIndexed(), field.isTermVectorStored(),
204                             field.isStorePositionWithTermVector(), field.isStoreOffsetWithTermVector(),
205                             field.getOmitNorms(), false, field.getOmitTf());
206 
207       if (thisFieldGen != fp.lastGen) {
208 
209         // First time we're seeing this field for this doc
210         fp.fieldCount = 0;
211 
212         if (fieldCount == fields.length) {
213           final int newSize = fields.length*2;
214           DocFieldProcessorPerField newArray[] = new DocFieldProcessorPerField[newSize];
215           System.arraycopy(fields, 0, newArray, 0, fieldCount);
216           fields = newArray;
217         }
218 
219         fields[fieldCount++] = fp;
220         fp.lastGen = thisFieldGen;
221       }
222 
223       if (fp.fieldCount == fp.fields.length) {
224         Fieldable[] newArray = new Fieldable[fp.fields.length*2];
225         System.arraycopy(fp.fields, 0, newArray, 0, fp.fieldCount);
226         fp.fields = newArray;
227       }
228 
229       fp.fields[fp.fieldCount++] = field;
230       if (field.isStored()) {
231         fieldsWriter.addField(field, fp.fieldInfo);
232       }
233     }
234 
235     // If we are writing vectors then we must visit
236     // fields in sorted order so they are written in
237     // sorted order.  TODO: we actually only need to
238     // sort the subset of fields that have vectors
239     // enabled; we could save [small amount of] CPU
240     // here.
241     quickSort(fields, 0, fieldCount-1);
242 
243     for(int i=0;i<fieldCount;i++)
244       fields[i].consumer.processFields(fields[i].fields, fields[i].fieldCount);
245 
246     if (docState.maxTermPrefix != null && docState.infoStream != null) {
247       docState.infoStream.println("WARNING: document contains at least one immense term (longer than the max length " + DocumentsWriter.MAX_TERM_LENGTH + "), all of which were skipped.  Please correct the analyzer to not produce such terms.  The prefix of the first immense term is: '" + docState.maxTermPrefix + "...'");
248       docState.maxTermPrefix = null;
249     }
250 
251     final DocumentsWriter.DocWriter one = fieldsWriter.finishDocument();
252     final DocumentsWriter.DocWriter two = consumer.finishDocument();
253     if (one == null) {
254       return two;
255     } else if (two == null) {
256       return one;
257     } else {
258       PerDoc both = getPerDoc();
259       both.docID = docState.docID;
260       assert one.docID == docState.docID;
261       assert two.docID == docState.docID;
262       both.one = one;
263       both.two = two;
264       return both;
265     }
266   }

268   void quickSort(DocFieldProcessorPerField[] array, int lo, int hi) {
269     if (lo >= hi)
270       return;
271     else if (hi == 1+lo) {
272       if (array[lo].fieldInfo.name.compareTo(array[hi].fieldInfo.name) > 0) {
273         final DocFieldProcessorPerField tmp = array[lo];
274         array[lo] = array[hi];
275         array[hi] = tmp;
276       }
277       return;
278     }
279 
280     int mid = (lo + hi) >>> 1;
281 
282     if (array[lo].fieldInfo.name.compareTo(array[mid].fieldInfo.name) > 0) {
283       DocFieldProcessorPerField tmp = array[lo];
284       array[lo] = array[mid];
285       array[mid] = tmp;
286     }
287 
288     if (array[mid].fieldInfo.name.compareTo(array[hi].fieldInfo.name) > 0) {
289       DocFieldProcessorPerField tmp = array[mid];
290       array[mid] = array[hi];
291       array[hi] = tmp;
292 
293       if (array[lo].fieldInfo.name.compareTo(array[mid].fieldInfo.name) > 0) {
294         DocFieldProcessorPerField tmp2 = array[lo];
295         array[lo] = array[mid];
296         array[mid] = tmp2;
297       }
298     }
299 
300     int left = lo + 1;
301     int right = hi - 1;
302 
303     if (left >= right)
304       return;
305 
306     DocFieldProcessorPerField partition = array[mid];
307 
308     for (; ;) {
309       while (array[right].fieldInfo.name.compareTo(partition.fieldInfo.name) > 0)
310         --right;
311 
312       while (left < right && array[left].fieldInfo.name.compareTo(partition.fieldInfo.name) <= 0)
313         ++left;
314 
315       if (left < right) {
316         DocFieldProcessorPerField tmp = array[left];
317         array[left] = array[right];
318         array[right] = tmp;
319         --right;
320       } else {
321         break;
322       }
323     }
324 
325     quickSort(array, lo, left);
326     quickSort(array, left + 1, hi);
327   }

329   PerDoc[] docFreeList = new PerDoc[1];
330   int freeCount;
331   int allocCount;
332 
333   synchronized PerDoc getPerDoc() {
334     if (freeCount == 0) {
335       allocCount++;
336       if (allocCount > docFreeList.length) {
337         // Grow our free list up front to make sure we have
338         // enough space to recycle all outstanding PerDoc
339         // instances
340         assert allocCount == 1+docFreeList.length;
341         docFreeList = new PerDoc[ArrayUtil.getNextSize(allocCount)];
342       }
343       return new PerDoc();
344     } else
345       return docFreeList[--freeCount];
346   }
347 
348   synchronized void freePerDoc(PerDoc perDoc) {
349     assert freeCount < docFreeList.length;
350     docFreeList[freeCount++] = perDoc;
351   }

353   class PerDoc extends DocumentsWriter.DocWriter {
354 
355     DocumentsWriter.DocWriter one;
356     DocumentsWriter.DocWriter two;
357 
358     public long sizeInBytes() {
359       return one.sizeInBytes() + two.sizeInBytes();
360     }
361 
362     public void finish() throws IOException {
363       try {
364         try {
365           one.finish();
366         } finally {
367           two.finish();
368         }
369       } finally {
370         freePerDoc(this);
371       }
372     }
373 
374     public void abort() {
375       try {
376         try {
377           one.abort();
378         } finally {
379           two.abort();
380         }
381       } finally {
382         freePerDoc(this);
383       }
384     }
385   }
