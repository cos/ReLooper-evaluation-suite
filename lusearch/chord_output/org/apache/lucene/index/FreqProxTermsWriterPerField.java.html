20  import java.io.IOException;
21  import org.apache.lucene.document.Fieldable;
22  import org.apache.lucene.analysis.tokenattributes.PayloadAttribute;
23  
24  // TODO: break into separate freq and prox writers as
25  // codecs; make separate container (tii/tis/skip/*) that can
26  // be configured as any number of files 1..N
27  final class FreqProxTermsWriterPerField extends TermsHashConsumerPerField implements Comparable {
28  
29    final FreqProxTermsWriterPerThread perThread;
30    final TermsHashPerField termsHashPerField;
31    final FieldInfo fieldInfo;
32    final DocumentsWriter.DocState docState;
33    final FieldInvertState fieldState;
34    boolean omitTermFreqAndPositions;
35    PayloadAttribute payloadAttribute;
36  
37    public FreqProxTermsWriterPerField(TermsHashPerField termsHashPerField, FreqProxTermsWriterPerThread perThread, FieldInfo fieldInfo) {
38      this.termsHashPerField = termsHashPerField;
39      this.perThread = perThread;
40      this.fieldInfo = fieldInfo;
41      docState = termsHashPerField.docState;
42      fieldState = termsHashPerField.fieldState;
43      omitTermFreqAndPositions = fieldInfo.omitTermFreqAndPositions;
44    }
45  
46    int getStreamCount() {
47      if (fieldInfo.omitTermFreqAndPositions)
48        return 1;
49      else
50        return 2;
51    }
52  
53    void finish() {}
54  
55    boolean hasPayloads;
56  
57    void skippingLongTerm() throws IOException {}
58  
59    public int compareTo(Object other0) {
60      FreqProxTermsWriterPerField other = (FreqProxTermsWriterPerField) other0;
61      return fieldInfo.name.compareTo(other.fieldInfo.name);
62    }
63  
64    void reset() {
65      // Record, up front, whether our in-RAM format will be
66      // with or without term freqs:
67      omitTermFreqAndPositions = fieldInfo.omitTermFreqAndPositions;
68      payloadAttribute = null;
69    }
70  
71    boolean start(Fieldable[] fields, int count) {
72      for(int i=0;i<count;i++)
73        if (fields[i].isIndexed())
74          return true;
75      return false;
76    }     
77    
78    void start(Fieldable f) {
79      if (fieldState.attributeSource.hasAttribute(PayloadAttribute.class)) {
80        payloadAttribute = (PayloadAttribute) fieldState.attributeSource.getAttribute(PayloadAttribute.class);
81      } else {
82        payloadAttribute = null;
83      }
84    }
85  
86    final void writeProx(FreqProxTermsWriter.PostingList p, int proxCode) {
87      final Payload payload;
88      if (payloadAttribute == null) {
89        payload = null;
90      } else {
91        payload = payloadAttribute.getPayload();
92      }
93      
94      if (payload != null && payload.length > 0) {
95        termsHashPerField.writeVInt(1, (proxCode<<1)|1);
96        termsHashPerField.writeVInt(1, payload.length);
97        termsHashPerField.writeBytes(1, payload.data, payload.offset, payload.length);
98        hasPayloads = true;      
99      } else
100       termsHashPerField.writeVInt(1, proxCode<<1);
101     p.lastPosition = fieldState.position;
102   }
103 
104   final void newTerm(RawPostingList p0) {
105     // First time we're seeing this term since the last
106     // flush
107     assert docState.testPoint("FreqProxTermsWriterPerField.newTerm start");
108     FreqProxTermsWriter.PostingList p = (FreqProxTermsWriter.PostingList) p0;
109     p.lastDocID = docState.docID;
110     if (omitTermFreqAndPositions) {
111       p.lastDocCode = docState.docID;
112     } else {
113       p.lastDocCode = docState.docID << 1;
114       p.docFreq = 1;
115       writeProx(p, fieldState.position);
116     }
117   }
118 
119   final void addTerm(RawPostingList p0) {
120 
121     assert docState.testPoint("FreqProxTermsWriterPerField.addTerm start");
122 
123     FreqProxTermsWriter.PostingList p = (FreqProxTermsWriter.PostingList) p0;
124 
125     assert omitTermFreqAndPositions || p.docFreq > 0;
126 
127     if (omitTermFreqAndPositions) {
128       if (docState.docID != p.lastDocID) {
129         assert docState.docID > p.lastDocID;
130         termsHashPerField.writeVInt(0, p.lastDocCode);
131         p.lastDocCode = docState.docID - p.lastDocID;
132         p.lastDocID = docState.docID;
133       }
134     } else {
135       if (docState.docID != p.lastDocID) {
136         assert docState.docID > p.lastDocID;
137         // Term not yet seen in the current doc but previously
138         // seen in other doc(s) since the last flush
139 
140         // Now that we know doc freq for previous doc,
141         // write it & lastDocCode
142         if (1 == p.docFreq)
143           termsHashPerField.writeVInt(0, p.lastDocCode|1);
144         else {
145           termsHashPerField.writeVInt(0, p.lastDocCode);
146           termsHashPerField.writeVInt(0, p.docFreq);
147         }
148         p.docFreq = 1;
149         p.lastDocCode = (docState.docID - p.lastDocID) << 1;
150         p.lastDocID = docState.docID;
151         writeProx(p, fieldState.position);
152       } else {
153         p.docFreq++;
154         writeProx(p, fieldState.position-p.lastPosition);
155       }
156     }
157   }
158 
159   public void abort() {}
160 }
