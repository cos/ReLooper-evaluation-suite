30  /** Access to the Fieldable Info file that describes document fields and whether or
31   *  not they are indexed. Each segment has a separate Fieldable Info file. Objects
32   *  of this class are thread-safe for multiple readers, but only one thread can
33   *  be adding documents at a time, with no other reader or writer threads
34   *  accessing this object.
35   */
36  final class FieldInfos {
37  
38    // Used internally (ie not written to *.fnm files) for pre-2.9 files
39    public static final int FORMAT_PRE = -1;
40  
41    // First used in 2.9; prior to 2.9 there was no format header
42    public static final int FORMAT_START = -2;
43  
44    static final int CURRENT_FORMAT = FORMAT_START;
45    
46    static final byte IS_INDEXED = 0x1;
47    static final byte STORE_TERMVECTOR = 0x2;
48    static final byte STORE_POSITIONS_WITH_TERMVECTOR = 0x4;
49    static final byte STORE_OFFSET_WITH_TERMVECTOR = 0x8;
50    static final byte OMIT_NORMS = 0x10;
51    static final byte STORE_PAYLOADS = 0x20;
52    static final byte OMIT_TERM_FREQ_AND_POSITIONS = 0x40;
53    
54    private final ArrayList byNumber = new ArrayList();
55    private final HashMap byName = new HashMap();
56    private int format;
57  
58    FieldInfos() { }
59  
60    /**
61     * Construct a FieldInfos object using the directory and the name of the file
62     * IndexInput
63     * @param d The directory to open the IndexInput from
64     * @param name The name of the file to open the IndexInput from in the Directory
65     * @throws IOException
66     */
67    FieldInfos(Directory d, String name) throws IOException {
68      IndexInput input = d.openInput(name);
69      try {
70        try {
71          read(input, name);
72        } catch (IOException ioe) {
73          if (format == FORMAT_PRE) {
74            // LUCENE-1623: FORMAT_PRE (before there was a
75            // format) may be 2.3.2 (pre-utf8) or 2.4.x (utf8)
76            // encoding; retry with input set to pre-utf8
77            input.seek(0);
78            input.setModifiedUTF8StringsMode();
79            byNumber.clear();
80            byName.clear();
81            try {
82              read(input, name);
83            } catch (Throwable t) {
84              // Ignore any new exception & throw original IOE
85              throw ioe;
86            }
87          } else {
88            // The IOException cannot be caused by
89            // LUCENE-1623, so re-throw it
90            throw ioe;
91          }
92        }
93      } finally {
94        input.close();
95      }
96    }

98    /**
99     * Returns a deep clone of this FieldInfos instance.
100    */
101   synchronized public Object clone() {
102     FieldInfos fis = new FieldInfos();
103     final int numField = byNumber.size();
104     for(int i=0;i<numField;i++) {
105       FieldInfo fi = (FieldInfo) ((FieldInfo) byNumber.get(i)).clone();
106       fis.byNumber.add(fi);
107       fis.byName.put(fi.name, fi);
108     }
109     return fis;
110   }

112   /** Adds field info for a Document. */
113   synchronized public void add(Document doc) {
114     List fields = doc.getFields();
115     Iterator fieldIterator = fields.iterator();
116     while (fieldIterator.hasNext()) {
117       Fieldable field = (Fieldable) fieldIterator.next();
118       add(field.name(), field.isIndexed(), field.isTermVectorStored(), field.isStorePositionWithTermVector(),
119               field.isStoreOffsetWithTermVector(), field.getOmitNorms(), false, field.getOmitTf());
120     }
121   }

123   /** Returns true if any fields do not omitTermFreqAndPositions */
124   boolean hasProx() {
125     final int numFields = byNumber.size();
126     for(int i=0;i<numFields;i++) {
127       final FieldInfo fi = fieldInfo(i);
128       if (fi.isIndexed && !fi.omitTermFreqAndPositions) {
129         return true;
130       }
131     }
132     return false;
133   }

138    * @param names The names of the fields

214    * @param storeOffsetWithTermVector true if the term vector with offsets should be stored
215    * @param omitNorms true if the norms for the indexed field should be omitted

219     add(name, isIndexed, storeTermVector, storePositionWithTermVector,
220         storeOffsetWithTermVector, omitNorms, false, false);

223   /** If the field is not yet known, adds it. If it is known, checks to make
224    *  sure that the isIndexed flag is the same as was given previously for this
225    *  field. If not - marks it as being indexed.  Same goes for the TermVector
226    * parameters.
227    *
228    * @param name The name of the field
229    * @param isIndexed true if the field is indexed
230    * @param storeTermVector true if the term vector should be stored
231    * @param storePositionWithTermVector true if the term vector with positions should be stored
232    * @param storeOffsetWithTermVector true if the term vector with offsets should be stored
233    * @param omitNorms true if the norms for the indexed field should be omitted
234    * @param storePayloads true if payloads should be stored for this field
235    * @param omitTermFreqAndPositions true if term freqs should be omitted for this field
236    */
237   synchronized public FieldInfo add(String name, boolean isIndexed, boolean storeTermVector,
238                        boolean storePositionWithTermVector, boolean storeOffsetWithTermVector,
239                        boolean omitNorms, boolean storePayloads, boolean omitTermFreqAndPositions) {
240     FieldInfo fi = fieldInfo(name);
241     if (fi == null) {
242       return addInternal(name, isIndexed, storeTermVector, storePositionWithTermVector, storeOffsetWithTermVector, omitNorms, storePayloads, omitTermFreqAndPositions);
243     } else {
244       fi.update(isIndexed, storeTermVector, storePositionWithTermVector, storeOffsetWithTermVector, omitNorms, storePayloads, omitTermFreqAndPositions);
245     }
246     return fi;
247   }
248 
249   private FieldInfo addInternal(String name, boolean isIndexed,
250                                 boolean storeTermVector, boolean storePositionWithTermVector, 
251                                 boolean storeOffsetWithTermVector, boolean omitNorms, boolean storePayloads, boolean omitTermFreqAndPositions) {
252     name = StringHelper.intern(name);
253     FieldInfo fi = new FieldInfo(name, isIndexed, byNumber.size(), storeTermVector, storePositionWithTermVector,
254                                  storeOffsetWithTermVector, omitNorms, storePayloads, omitTermFreqAndPositions);
255     byNumber.add(fi);
256     byName.put(name, fi);
257     return fi;
258   }

295   public boolean hasVectors() {
296     boolean hasVectors = false;
297     for (int i = 0; i < size(); i++) {
298       if (fieldInfo(i).storeTermVector) {
299         hasVectors = true;
300         break;
301       }
302     }
303     return hasVectors;
304   }

306   public void write(Directory d, String name) throws IOException {
307     IndexOutput output = d.createOutput(name);
308     try {
309       write(output);
310     } finally {
311       output.close();
312     }
313   }

315   public void write(IndexOutput output) throws IOException {
316     output.writeVInt(CURRENT_FORMAT);
317     output.writeVInt(size());
318     for (int i = 0; i < size(); i++) {
319       FieldInfo fi = fieldInfo(i);
320       byte bits = 0x0;
321       if (fi.isIndexed) bits |= IS_INDEXED;
322       if (fi.storeTermVector) bits |= STORE_TERMVECTOR;
323       if (fi.storePositionWithTermVector) bits |= STORE_POSITIONS_WITH_TERMVECTOR;
324       if (fi.storeOffsetWithTermVector) bits |= STORE_OFFSET_WITH_TERMVECTOR;
325       if (fi.omitNorms) bits |= OMIT_NORMS;
326       if (fi.storePayloads) bits |= STORE_PAYLOADS;
327       if (fi.omitTermFreqAndPositions) bits |= OMIT_TERM_FREQ_AND_POSITIONS;
328       
329       output.writeString(fi.name);
330       output.writeByte(bits);
331     }
332   }

334   private void read(IndexInput input, String fileName) throws IOException {
335     int firstInt = input.readVInt();
336 
337     if (firstInt < 0) {
338       // This is a real format
339       format = firstInt;
340     } else {
341       format = FORMAT_PRE;
342     }
343 
344     if (format != FORMAT_PRE & format != FORMAT_START) {
345       throw new CorruptIndexException("unrecognized format " + format + " in file \"" + fileName + "\"");
346     }
347 
348     int size;
349     if (format == FORMAT_PRE) {
350       size = firstInt;
351     } else {
352       size = input.readVInt(); //read in the size
353     }
354 
355     for (int i = 0; i < size; i++) {
356       String name = StringHelper.intern(input.readString());
357       byte bits = input.readByte();
358       boolean isIndexed = (bits & IS_INDEXED) != 0;
359       boolean storeTermVector = (bits & STORE_TERMVECTOR) != 0;
360       boolean storePositionsWithTermVector = (bits & STORE_POSITIONS_WITH_TERMVECTOR) != 0;
361       boolean storeOffsetWithTermVector = (bits & STORE_OFFSET_WITH_TERMVECTOR) != 0;
362       boolean omitNorms = (bits & OMIT_NORMS) != 0;
363       boolean storePayloads = (bits & STORE_PAYLOADS) != 0;
364       boolean omitTermFreqAndPositions = (bits & OMIT_TERM_FREQ_AND_POSITIONS) != 0;
365       
366       addInternal(name, isIndexed, storeTermVector, storePositionsWithTermVector, storeOffsetWithTermVector, omitNorms, storePayloads, omitTermFreqAndPositions);
367     }
368 
369     if (input.getFilePointer() != input.length()) {
370       throw new CorruptIndexException("did not read all bytes from file \"" + fileName + "\": read " + input.getFilePointer() + " vs size " + input.length());
371     }    
372   }
