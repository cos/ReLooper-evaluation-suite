22  /** This is a {@link LogMergePolicy} that measures size of a
23   *  segment as the number of documents (not taking deletions
24   *  into account). */
25  
26  public class LogDocMergePolicy extends LogMergePolicy {
27  
28    /** Default minimum segment size.  @see setMinMergeDocs */
29    public static final int DEFAULT_MIN_MERGE_DOCS = 1000;
30  
31    public LogDocMergePolicy(IndexWriter writer) {
32      super(writer);
33      minMergeSize = DEFAULT_MIN_MERGE_DOCS;
34  
35      // maxMergeSize is never used by LogDocMergePolicy; set
36      // it to Long.MAX_VALUE to disable it
37      maxMergeSize = Long.MAX_VALUE;
38    }
39    protected long size(SegmentInfo info) throws IOException {
40      return sizeDocs(info);
41    }
42  
43    /** Sets the minimum size for the lowest level segments.
44     * Any segments below this size are considered to be on
45     * the same level (even if they vary drastically in size)
46     * and will be merged whenever there are mergeFactor of
47     * them.  This effectively truncates the "long tail" of
48     * small segments that would otherwise be created into a
49     * single level.  If you set this too large, it could
50     * greatly increase the merging cost during indexing (if
51     * you flush many small segments). */
52    public void setMinMergeDocs(int minMergeDocs) {
53      minMergeSize = minMergeDocs;
54    }
55  
56    /** Get the minimum size for a segment to remain
57     *  un-merged.
58     *  @see #setMinMergeDocs **/
59    public int getMinMergeDocs() {
60      return (int) minMergeSize;
61    }
62  }
