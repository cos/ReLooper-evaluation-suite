20  import java.util.Collection;
21  import java.util.List;
22  import java.util.ArrayList;
23  import java.util.Map;
24  import java.io.IOException;
25  import org.apache.lucene.store.Directory;
26  
27  /** A {@link IndexDeletionPolicy} that wraps around any other
28   *  {@link IndexDeletionPolicy} and adds the ability to hold and
29   *  later release a single "snapshot" of an index.  While
30   *  the snapshot is held, the {@link IndexWriter} will not
31   *  remove any files associated with it even if the index is
32   *  otherwise being actively, arbitrarily changed.  Because
33   *  we wrap another arbitrary {@link IndexDeletionPolicy}, this
34   *  gives you the freedom to continue using whatever {@link
35   *  IndexDeletionPolicy} you would normally want to use with your
36   *  index.  Note that you can re-use a single instance of
37   *  SnapshotDeletionPolicy across multiple writers as long
38   *  as they are against the same index Directory.  Any
39   *  snapshot held when a writer is closed will "survive"
40   *  when the next writer is opened.
41   *
42   * <p><b>WARNING</b>: This API is a new and experimental and
43   * may suddenly change.</p> */
44  
45  public class SnapshotDeletionPolicy implements IndexDeletionPolicy {
46  
47    private IndexCommit lastCommit;
48    private IndexDeletionPolicy primary;
49    private String snapshot;
50  
51    public SnapshotDeletionPolicy(IndexDeletionPolicy primary) {
52      this.primary = primary;
53    }
54  
55    public synchronized void onInit(List commits) throws IOException {
56      primary.onInit(wrapCommits(commits));
57      lastCommit = (IndexCommit) commits.get(commits.size()-1);
58    }
59  
60    public synchronized void onCommit(List commits) throws IOException {
61      primary.onCommit(wrapCommits(commits));
62      lastCommit = (IndexCommit) commits.get(commits.size()-1);
63    }
64  
65    /** Take a snapshot of the most recent commit to the
66     *  index.  You must call release() to free this snapshot.
67     *  Note that while the snapshot is held, the files it
68     *  references will not be deleted, which will consume
69     *  additional disk space in your index. If you take a
70     *  snapshot at a particularly bad time (say just before
71     *  you call optimize()) then in the worst case this could
72     *  consume an extra 1X of your total index size, until
73     *  you release the snapshot. */
74    // TODO 3.0: change this to return IndexCommit instead
75    public synchronized IndexCommitPoint snapshot() {
76      if (snapshot == null)
77        snapshot = lastCommit.getSegmentsFileName();
78      else
79        throw new IllegalStateException("snapshot is already set; please call release() first");
80      return lastCommit;
81    }
82  
83    /** Release the currently held snapshot. */
84    public synchronized void release() {
85      if (snapshot != null)
86        snapshot = null;
87      else
88        throw new IllegalStateException("snapshot was not set; please call snapshot() first");
89    }
90  
91    private class MyCommitPoint extends IndexCommit {
92      IndexCommit cp;
93      MyCommitPoint(IndexCommit cp) {
94        this.cp = cp;
95      }
96      public String getSegmentsFileName() {
97        return cp.getSegmentsFileName();
98      }
99      public Collection getFileNames() throws IOException {
100       return cp.getFileNames();
101     }
102     public Directory getDirectory() {
103       return cp.getDirectory();
104     }
105     public void delete() {
106       synchronized(SnapshotDeletionPolicy.this) {
107         // Suppress the delete request if this commit point is
108         // our current snapshot.
109         if (snapshot == null || !snapshot.equals(getSegmentsFileName()))
110           cp.delete();
111       }
112     }
113     public boolean isDeleted() {
114       return cp.isDeleted();
115     }
116     public long getVersion() {
117       return cp.getVersion();
118     }
119     public long getGeneration() {
120       return cp.getGeneration();
121     }
122     public Map getUserData() throws IOException {
123       return cp.getUserData();
124     }
125   }
126 
127   private List wrapCommits(List commits) {
128     final int count = commits.size();
129     List myCommits = new ArrayList(count);
130     for(int i=0;i<count;i++)
131       myCommits.add(new MyCommitPoint((IndexCommit) commits.get(i)));
132     return myCommits;
133   }
134 }
