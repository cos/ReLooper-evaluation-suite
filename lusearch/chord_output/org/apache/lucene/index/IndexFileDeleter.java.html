33  /*
34   * This class keeps track of each SegmentInfos instance that
35   * is still "live", either because it corresponds to a
36   * segments_N file in the Directory (a "commit", i.e. a
37   * committed SegmentInfos) or because it's an in-memory
38   * SegmentInfos that a writer is actively updating but has
39   * not yet committed.  This class uses simple reference
40   * counting to map the live SegmentInfos instances to
41   * individual files in the Directory.
42   *
43   * When autoCommit=true, IndexWriter currently commits only
44   * on completion of a merge (though this may change with
45   * time: it is not a guarantee).  When autoCommit=false,
46   * IndexWriter only commits when it is closed.  Regardless
47   * of autoCommit, the user may call IndexWriter.commit() to
48   * force a blocking commit.
49   * 
50   * The same directory file may be referenced by more than
51   * one IndexCommit, i.e. more than one SegmentInfos.
52   * Therefore we count how many commits reference each file.
53   * When all the commits referencing a certain file have been
54   * deleted, the refcount for that file becomes zero, and the
55   * file is deleted.
56   *
57   * A separate deletion policy interface
58   * (IndexDeletionPolicy) is consulted on creation (onInit)
59   * and once per commit (onCommit), to decide when a commit
60   * should be removed.
61   * 
62   * It is the business of the IndexDeletionPolicy to choose
63   * when to delete commit points.  The actual mechanics of
64   * file deletion, retrying, etc, derived from the deletion
65   * of commit points is the business of the IndexFileDeleter.
66   * 
67   * The current default deletion policy is {@link
68   * KeepOnlyLastCommitDeletionPolicy}, which removes all
69   * prior commits when a new commit has completed.  This
70   * matches the behavior before 2.2.
71   *
72   * Note that you must hold the write.lock before
73   * instantiating this class.  It opens segments_N file(s)
74   * directly with no retry logic.
75   */
76  
77  final class IndexFileDeleter {

80     * because they are open and we are running on Windows),
81     * so we will retry them again later: */
82    private List deletable;

85     * Counts how many existing commits reference a file.
86     * Maps String to RefCount (class below) instances: */
87    private Map refCounts = new HashMap();

92     * Other policies may leave commit points live for longer
93     * in which case this list would be longer than 1: */
94    private List commits = new ArrayList();

124   /**
125    * Initialize the deleter: find all previous commits in
126    * the Directory, incref the files they reference, call
127    * the policy to let it delete commits.  This will remove
128    * any files not referenced by any of the commits.
129    * @throws CorruptIndexException if the index is corrupt
130    * @throws IOException if there is a low-level IO error
131    */
132   public IndexFileDeleter(Directory directory, IndexDeletionPolicy policy, SegmentInfos segmentInfos, PrintStream infoStream, DocumentsWriter docWriter)
133     throws CorruptIndexException, IOException {
134 
135     this.docWriter = docWriter;
136     this.infoStream = infoStream;
137 
138     if (infoStream != null)
139       message("init: current segments file is \"" + segmentInfos.getCurrentSegmentFileName() + "\"; deletionPolicy=" + policy);
140 
141     this.policy = policy;
142     this.directory = directory;
143 
144     // First pass: walk the files and initialize our ref
145     // counts:
146     long currentGen = segmentInfos.getGeneration();
147     IndexFileNameFilter filter = IndexFileNameFilter.getFilter();
148 
149     String[] files = directory.listAll();
150 
151     CommitPoint currentCommitPoint = null;
152 
153     for(int i=0;i<files.length;i++) {
154 
155       String fileName = files[i];
156 
157       if (filter.accept(null, fileName) && !fileName.equals(IndexFileNames.SEGMENTS_GEN)) {
158 
159         // Add this file to refCounts with initial count 0:
160         getRefCount(fileName);
161 
162         if (fileName.startsWith(IndexFileNames.SEGMENTS)) {
163 
164           // This is a commit (segments or segments_N), and
165           // it's valid (<= the max gen).  Load it, then
166           // incref all files it refers to:
167           if (SegmentInfos.generationFromSegmentsFileName(fileName) <= currentGen) {
168             if (infoStream != null) {
169               message("init: load commit \"" + fileName + "\"");
170             }
171             SegmentInfos sis = new SegmentInfos();
172             try {
173               sis.read(directory, fileName);
174             } catch (FileNotFoundException e) {
175               // LUCENE-948: on NFS (and maybe others), if
176               // you have writers switching back and forth
177               // between machines, it's very likely that the
178               // dir listing will be stale and will claim a
179               // file segments_X exists when in fact it
180               // doesn't.  So, we catch this and handle it
181               // as if the file does not exist
182               if (infoStream != null) {
183                 message("init: hit FileNotFoundException when loading commit \"" + fileName + "\"; skipping this commit point");
184               }
185               sis = null;
186             }
187             if (sis != null) {
188               CommitPoint commitPoint = new CommitPoint(commitsToDelete, directory, sis);
189               if (sis.getGeneration() == segmentInfos.getGeneration()) {
190                 currentCommitPoint = commitPoint;
191               }
192               commits.add(commitPoint);
193               incRef(sis, true);
194             }
195           }
196         }
197       }
198     }
199 
200     if (currentCommitPoint == null) {
201       // We did not in fact see the segments_N file
202       // corresponding to the segmentInfos that was passed
203       // in.  Yet, it must exist, because our caller holds
204       // the write lock.  This can happen when the directory
205       // listing was stale (eg when index accessed via NFS
206       // client with stale directory listing cache).  So we
207       // try now to explicitly open this commit point:
208       SegmentInfos sis = new SegmentInfos();
209       try {
210         sis.read(directory, segmentInfos.getCurrentSegmentFileName());
211       } catch (IOException e) {
212         throw new CorruptIndexException("failed to locate current segments_N file");
213       }
214       if (infoStream != null)
215         message("forced open of current segments file " + segmentInfos.getCurrentSegmentFileName());
216       currentCommitPoint = new CommitPoint(commitsToDelete, directory, sis);
217       commits.add(currentCommitPoint);
218       incRef(sis, true);
219     }
220 
221     // We keep commits list in sorted order (oldest to newest):
222     Collections.sort(commits);
223 
224     // Now delete anything with ref count at 0.  These are
225     // presumably abandoned files eg due to crash of
226     // IndexWriter.
227     Iterator it = refCounts.keySet().iterator();
228     while(it.hasNext()) {
229       String fileName = (String) it.next();
230       RefCount rc = (RefCount) refCounts.get(fileName);
231       if (0 == rc.count) {
232         if (infoStream != null) {
233           message("init: removing unreferenced file \"" + fileName + "\"");
234         }
235         deleteFile(fileName);
236       }
237     }
238 
239     // Finally, give policy a chance to remove things on
240     // startup:
241     policy.onInit(commits);
242 
243     // Always protect the incoming segmentInfos since
244     // sometime it may not be the most recent commit
245     checkpoint(segmentInfos, false);
246     
247     startingCommitDeleted = currentCommitPoint.isDeleted();
248 
249     deleteCommits();
250   }

252   /**
253    * Remove the CommitPoints in the commitsToDelete List by
254    * DecRef'ing all files from each SegmentInfos.
255    */
256   private void deleteCommits() throws IOException {
257 
258     int size = commitsToDelete.size();
259 
260     if (size > 0) {
261 
262       // First decref all files that had been referred to by
263       // the now-deleted commits:
264       for(int i=0;i<size;i++) {
265         CommitPoint commit = (CommitPoint) commitsToDelete.get(i);
266         if (infoStream != null) {
267           message("deleteCommits: now decRef commit \"" + commit.getSegmentsFileName() + "\"");
268         }
269         Iterator it = commit.files.iterator();
270         while(it.hasNext()) {
271           decRef((String) it.next());
272         }
273       }
274       commitsToDelete.clear();
275 
276       // Now compact commits to remove deleted ones (preserving the sort):
277       size = commits.size();
278       int readFrom = 0;
279       int writeTo = 0;
280       while(readFrom < size) {
281         CommitPoint commit = (CommitPoint) commits.get(readFrom);
282         if (!commit.deleted) {
283           if (writeTo != readFrom) {
284             commits.set(writeTo, commits.get(readFrom));
285           }
286           writeTo++;
287         }
288         readFrom++;
289       }
290 
291       while(size > writeTo) {
292         commits.remove(size-1);
293         size--;
294       }
295     }
296   }

298   /**
299    * Writer calls this when it has hit an error and had to
300    * roll back, to tell us that there may now be
301    * unreferenced files in the filesystem.  So we re-list
302    * the filesystem and delete such files.  If segmentName
303    * is non-null, we will only delete files corresponding to
304    * that segment.
305    */
306   public void refresh(String segmentName) throws IOException {
307     String[] files = directory.listAll();
308     IndexFileNameFilter filter = IndexFileNameFilter.getFilter();
309     String segmentPrefix1;
310     String segmentPrefix2;
311     if (segmentName != null) {
312       segmentPrefix1 = segmentName + ".";
313       segmentPrefix2 = segmentName + "_";
314     } else {
315       segmentPrefix1 = null;
316       segmentPrefix2 = null;
317     }
318     
319     for(int i=0;i<files.length;i++) {
320       String fileName = files[i];
321       if (filter.accept(null, fileName) &&
322           (segmentName == null || fileName.startsWith(segmentPrefix1) || fileName.startsWith(segmentPrefix2)) &&
323           !refCounts.containsKey(fileName) &&
324           !fileName.equals(IndexFileNames.SEGMENTS_GEN)) {
325         // Unreferenced file, so remove it
326         if (infoStream != null) {
327           message("refresh [prefix=" + segmentName + "]: removing newly created unreferenced file \"" + fileName + "\"");
328         }
329         deleteFile(fileName);
330       }
331     }
332   }
333 
334   public void refresh() throws IOException {
335     refresh(null);
336   }

338   public void close() throws IOException {
339     // DecRef old files from the last checkpoint, if any:
340     int size = lastFiles.size();
341     if (size > 0) {
342       for(int i=0;i<size;i++)
343         decRef((Collection) lastFiles.get(i));
344       lastFiles.clear();
345     }
346 
347     deletePendingFiles();
348   }
349 
350   private void deletePendingFiles() throws IOException {
351     if (deletable != null) {
352       List oldDeletable = deletable;
353       deletable = null;
354       int size = oldDeletable.size();
355       for(int i=0;i<size;i++) {
356         if (infoStream != null)
357           message("delete pending file " + oldDeletable.get(i));
358         deleteFile((String) oldDeletable.get(i));
359       }
360     }
361   }

363   /**
364    * For definition of "check point" see IndexWriter comments:
365    * "Clarification: Check Points (and commits)".
366    * 
367    * Writer calls this when it has made a "consistent
368    * change" to the index, meaning new files are written to
369    * the index and the in-memory SegmentInfos have been
370    * modified to point to those files.
371    *
372    * This may or may not be a commit (segments_N may or may
373    * not have been written).
374    *
375    * We simply incref the files referenced by the new
376    * SegmentInfos and decref the files we had previously
377    * seen (if any).
378    *
379    * If this is a commit, we also call the policy to give it
380    * a chance to remove other commits.  If any commits are
381    * removed, we decref their files as well.
382    */
383   public void checkpoint(SegmentInfos segmentInfos, boolean isCommit) throws IOException {
384 
385     if (infoStream != null) {
386       message("now checkpoint \"" + segmentInfos.getCurrentSegmentFileName() + "\" [" + segmentInfos.size() + " segments " + "; isCommit = " + isCommit + "]");
387     }
388 
389     // Try again now to delete any previously un-deletable
390     // files (because they were in use, on Windows):
391     deletePendingFiles();
392 
393     // Incref the files:
394     incRef(segmentInfos, isCommit);
395 
396     if (isCommit) {
397       // Append to our commits list:
398       commits.add(new CommitPoint(commitsToDelete, directory, segmentInfos));
399 
400       // Tell policy so it can remove commits:
401       policy.onCommit(commits);
402 
403       // Decref files for commits that were deleted by the policy:
404       deleteCommits();
405     } else {
406 
407       final List docWriterFiles;
408       if (docWriter != null) {
409         docWriterFiles = docWriter.openFiles();
410         if (docWriterFiles != null)
411           // We must incRef these files before decRef'ing
412           // last files to make sure we don't accidentally
413           // delete them:
414           incRef(docWriterFiles);
415       } else
416         docWriterFiles = null;
417 
418       // DecRef old files from the last checkpoint, if any:
419       int size = lastFiles.size();
420       if (size > 0) {
421         for(int i=0;i<size;i++)
422           decRef((Collection) lastFiles.get(i));
423         lastFiles.clear();
424       }
425 
426       // Save files so we can decr on next checkpoint/commit:
427       lastFiles.add(segmentInfos.files(directory, false));
428 
429       if (docWriterFiles != null)
430         lastFiles.add(docWriterFiles);
431     }
432   }

434   void incRef(SegmentInfos segmentInfos, boolean isCommit) throws IOException {
435      // If this is a commit point, also incRef the
436      // segments_N file:
437     Iterator it = segmentInfos.files(directory, isCommit).iterator();
438     while(it.hasNext()) {
439       incRef((String) it.next());
440     }
441   }

443   void incRef(List files) throws IOException {
444     int size = files.size();
445     for(int i=0;i<size;i++) {
446       incRef((String) files.get(i));
447     }
448   }

450   void incRef(String fileName) throws IOException {
451     RefCount rc = getRefCount(fileName);
452     if (infoStream != null && VERBOSE_REF_COUNTS) {
453       message("  IncRef \"" + fileName + "\": pre-incr count is " + rc.count);
454     }
455     rc.IncRef();
456   }
457 
458   void decRef(Collection files) throws IOException {
459     Iterator it = files.iterator();
460     while(it.hasNext()) {
461       decRef((String) it.next());
462     }
463   }
464 
465   void decRef(String fileName) throws IOException {
466     RefCount rc = getRefCount(fileName);
467     if (infoStream != null && VERBOSE_REF_COUNTS) {
468       message("  DecRef \"" + fileName + "\": pre-decr count is " + rc.count);
469     }
470     if (0 == rc.DecRef()) {
471       // This file is no longer referenced by any past
472       // commit points nor by the in-memory SegmentInfos:
473       deleteFile(fileName);
474       refCounts.remove(fileName);
475     }
476   }
477 
478   void decRef(SegmentInfos segmentInfos) throws IOException {
479     Iterator it = segmentInfos.files(directory, false).iterator();
480     while(it.hasNext()) {
481       decRef((String) it.next());
482     }
483   }
484 
485   private RefCount getRefCount(String fileName) {
486     RefCount rc;
487     if (!refCounts.containsKey(fileName)) {
488       rc = new RefCount(fileName);
489       refCounts.put(fileName, rc);
490     } else {
491       rc = (RefCount) refCounts.get(fileName);
492     }
493     return rc;
494   }
495 
496   void deleteFiles(List files) throws IOException {
497     final int size = files.size();
498     for(int i=0;i<size;i++)
499       deleteFile((String) files.get(i));
500   }

502   /** Deletes the specified files, but only if they are new
503    *  (have not yet been incref'd). */
504   void deleteNewFiles(Collection files) throws IOException {
505     final Iterator it = files.iterator();
506     while(it.hasNext()) {
507       final String fileName = (String) it.next();
508       if (!refCounts.containsKey(fileName))
509         deleteFile(fileName);
510     }
511   }
512 
513   void deleteFile(String fileName)
514        throws IOException {
515     try {
516       if (infoStream != null) {
517         message("delete \"" + fileName + "\"");
518       }
519       directory.deleteFile(fileName);
520     } catch (IOException e) {             // if delete fails
521       if (directory.fileExists(fileName)) {
522 
523         // Some operating systems (e.g. Windows) don't
524         // permit a file to be deleted while it is opened
525         // for read (e.g. by another process or thread). So
526         // we assume that when a delete fails it is because
527         // the file is open in another process, and queue
528         // the file for subsequent deletion.
529 
530         if (infoStream != null) {
531           message("IndexFileDeleter: unable to remove file \"" + fileName + "\": " + e.toString() + "; Will re-try later.");
532         }
533         if (deletable == null) {
534           deletable = new ArrayList();
535         }
536         deletable.add(fileName);                  // add to deletable
537       }
538     }
539   }

541   /**
542    * Tracks the reference count for a single index file:
543    */
544   final private static class RefCount {
545 
546     // fileName used only for better assert error messages
547     final String fileName;
548     boolean initDone;
549     RefCount(String fileName) {
550       this.fileName = fileName;
551     }
552 
553     int count;
554 
555     public int IncRef() {
556       if (!initDone) {
557         initDone = true;
558       } else {
559         assert count > 0: "RefCount is 0 pre-increment for file \"" + fileName + "\"";
560       }
561       return ++count;
562     }
563 
564     public int DecRef() {
565       assert count > 0: "RefCount is 0 pre-decrement for file \"" + fileName + "\"";
566       return --count;
567     }
568   }

570   /**
571    * Holds details for each commit point.  This class is
572    * also passed to the deletion policy.  Note: this class
573    * has a natural ordering that is inconsistent with
574    * equals.
575    */
576 
577   final private static class CommitPoint extends IndexCommit implements Comparable {
578 
579     long gen;
580     Collection files;
581     String segmentsFileName;
582     boolean deleted;
583     Directory directory;
584     Collection commitsToDelete;
585     long version;
586     long generation;
587     final boolean isOptimized;
588     final Map userData;
589 
590     public CommitPoint(Collection commitsToDelete, Directory directory, SegmentInfos segmentInfos) throws IOException {
591       this.directory = directory;
592       this.commitsToDelete = commitsToDelete;
593       userData = segmentInfos.getUserData();
594       segmentsFileName = segmentInfos.getCurrentSegmentFileName();
595       version = segmentInfos.getVersion();
596       generation = segmentInfos.getGeneration();
597       files = Collections.unmodifiableCollection(segmentInfos.files(directory, true));
598       gen = segmentInfos.getGeneration();
599       isOptimized = segmentInfos.size() == 1 && !segmentInfos.info(0).hasDeletions();
600 
601       assert !segmentInfos.hasExternalSegments(directory);
602     }
603 
604     public boolean isOptimized() {
605       return isOptimized;
606     }
607 
608     public String getSegmentsFileName() {
609       return segmentsFileName;
610     }
611 
612     public Collection getFileNames() throws IOException {
613       return files;
614     }
615 
616     public Directory getDirectory() {
617       return directory;
618     }
619 
620     public long getVersion() {
621       return version;
622     }
623 
624     public long getGeneration() {
625       return generation;
626     }
627 
628     public Map getUserData() {
629       return userData;
630     }
631 
632     /**
633      * Called only be the deletion policy, to remove this
634      * commit point from the index.
635      */
636     public void delete() {
637       if (!deleted) {
638         deleted = true;
639         commitsToDelete.add(this);
640       }
641     }
642 
643     public boolean isDeleted() {
644       return deleted;
645     }
646 
647     public int compareTo(Object obj) {
648       CommitPoint commit = (CommitPoint) obj;
649       if (gen < commit.gen) {
650         return -1;
651       } else if (gen > commit.gen) {
652         return 1;
653       } else {
654         return 0;
655       }
656     }
657   }
