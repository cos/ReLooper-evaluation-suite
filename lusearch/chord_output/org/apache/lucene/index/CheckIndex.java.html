36  /**
37   * Basic tool and API to check the health of an index and
38   * write a new segments file that removes reference to
39   * problematic segments.
40   * 
41   * <p>As this tool checks every byte in the index, on a large
42   * index it can take quite a long time to run.
43   *
44   * <p><b>WARNING</b>: this tool and API is new and
45   * experimental and is subject to suddenly change in the
46   * next release.  Please make a complete backup of your
47   * index before using this to fix your index!
48   */
49  public class CheckIndex {

298   /** Returns true if index is clean, else false.
299    *  @deprecated Please instantiate a CheckIndex and then use {@link #checkIndex(List)} instead */
300   public static boolean check(Directory dir, boolean doFix, List onlySegments) throws IOException {
301     CheckIndex checker = new CheckIndex(dir);
302     Status status = checker.checkIndex(onlySegments);
303     if (doFix && !status.clean)
304       checker.fixIndex(status);
305 
306     return status.clean;
307   }

322   /** Returns a {@link Status} instance detailing
323    *  the state of the index.
324    * 
325    *  @param onlySegments list of specific segment names to check
326    *
327    *  <p>As this method checks every byte in the specified
328    *  segments, on a large index it can take quite a long
329    *  time to run.
330    *
331    *  <p><b>WARNING</b>: make sure
332    *  you only call this when the index is not opened by any
333    *  writer. */
334   public Status checkIndex(List onlySegments) throws IOException {
335     NumberFormat nf = NumberFormat.getInstance();
336     SegmentInfos sis = new SegmentInfos();
337     Status result = new Status();
338     result.dir = dir;
339     try {
340       sis.read(dir);
341     } catch (Throwable t) {
342       msg("ERROR: could not read any segments file in directory");
343       result.missingSegments = true;
344       if (infoStream != null)
345         t.printStackTrace(infoStream);
346       return result;
347     }
348 
349     final int numSegments = sis.size();
350     final String segmentsFileName = sis.getCurrentSegmentFileName();
351     IndexInput input = null;
352     try {
353       input = dir.openInput(segmentsFileName);
354     } catch (Throwable t) {
355       msg("ERROR: could not open segments file in directory");
356       if (infoStream != null)
357         t.printStackTrace(infoStream);
358       result.cantOpenSegments = true;
359       return result;
360     }
361     int format = 0;
362     try {
363       format = input.readInt();
364     } catch (Throwable t) {
365       msg("ERROR: could not read segment file version in directory");
366       if (infoStream != null)
367         t.printStackTrace(infoStream);
368       result.missingSegmentVersion = true;
369       return result;
370     } finally {
371       if (input != null)
372         input.close();
373     }
374 
375     String sFormat = "";
376     boolean skip = false;
377 
378     if (format == SegmentInfos.FORMAT)
379       sFormat = "FORMAT [Lucene Pre-2.1]";
380     if (format == SegmentInfos.FORMAT_LOCKLESS)
381       sFormat = "FORMAT_LOCKLESS [Lucene 2.1]";
382     else if (format == SegmentInfos.FORMAT_SINGLE_NORM_FILE)
383       sFormat = "FORMAT_SINGLE_NORM_FILE [Lucene 2.2]";
384     else if (format == SegmentInfos.FORMAT_SHARED_DOC_STORE)
385       sFormat = "FORMAT_SHARED_DOC_STORE [Lucene 2.3]";
386     else {
387       if (format == SegmentInfos.FORMAT_CHECKSUM)
388         sFormat = "FORMAT_CHECKSUM [Lucene 2.4]";
389       else if (format == SegmentInfos.FORMAT_DEL_COUNT)
390         sFormat = "FORMAT_DEL_COUNT [Lucene 2.4]";
391       else if (format == SegmentInfos.FORMAT_HAS_PROX)
392         sFormat = "FORMAT_HAS_PROX [Lucene 2.4]";
393       else if (format == SegmentInfos.FORMAT_USER_DATA)
394         sFormat = "FORMAT_USER_DATA [Lucene 2.9]";
395       else if (format == SegmentInfos.FORMAT_DIAGNOSTICS)
396         sFormat = "FORMAT_DIAGNOSTICS [Lucene 2.9]";
397       else if (format < SegmentInfos.CURRENT_FORMAT) {
398         sFormat = "int=" + format + " [newer version of Lucene than this tool]";
399         skip = true;
400       } else {
401         sFormat = format + " [Lucene 1.3 or prior]";
402       }
403     }
404 
405     result.segmentsFileName = segmentsFileName;
406     result.numSegments = numSegments;
407     result.segmentFormat = sFormat;
408     result.userData = sis.getUserData();
409     String userDataString;
410     if (sis.getUserData().size() > 0) {
411       userDataString = " userData=" + sis.getUserData();
412     } else {
413       userDataString = "";
414     }
415 
416     msg("Segments file=" + segmentsFileName + " numSegments=" + numSegments + " version=" + sFormat + userDataString);
417 
418     if (onlySegments != null) {
419       result.partial = true;
420       if (infoStream != null)
421         infoStream.print("\nChecking only these segments:");
422       Iterator it = onlySegments.iterator();
423       while (it.hasNext()) {
424         if (infoStream != null)
425           infoStream.print(" " + it.next());
426       }
427       result.segmentsChecked.addAll(onlySegments);
428       msg(":");
429     }
430 
431     if (skip) {
432       msg("\nERROR: this index appears to be created by a newer version of Lucene than this tool was compiled on; please re-compile this tool on the matching version of Lucene; exiting");
433       result.toolOutOfDate = true;
434       return result;
435     }
436 
437 
438     result.newSegments = (SegmentInfos) sis.clone();
439     result.newSegments.clear();
440 
441     for(int i=0;i<numSegments;i++) {
442       final SegmentInfo info = sis.info(i);
443       if (onlySegments != null && !onlySegments.contains(info.name))
444         continue;
445       Status.SegmentInfoStatus segInfoStat = new Status.SegmentInfoStatus();
446       result.segmentInfos.add(segInfoStat);
447       msg("  " + (1+i) + " of " + numSegments + ": name=" + info.name + " docCount=" + info.docCount);
448       segInfoStat.name = info.name;
449       segInfoStat.docCount = info.docCount;
450 
451       int toLoseDocCount = info.docCount;
452 
453       SegmentReader reader = null;
454 
455       try {
456         msg("    compound=" + info.getUseCompoundFile());
457         segInfoStat.compound = info.getUseCompoundFile();
458         msg("    hasProx=" + info.getHasProx());
459         segInfoStat.hasProx = info.getHasProx();
460         msg("    numFiles=" + info.files().size());
461         segInfoStat.numFiles = info.files().size();
462         msg("    size (MB)=" + nf.format(info.sizeInBytes()/(1024.*1024.)));
463         segInfoStat.sizeMB = info.sizeInBytes()/(1024.*1024.);
464         Map diagnostics = info.getDiagnostics();
465         segInfoStat.diagnostics = diagnostics;
466         if (diagnostics.size() > 0) {
467           msg("    diagnostics = " + diagnostics);
468         }
469 
470         final int docStoreOffset = info.getDocStoreOffset();
471         if (docStoreOffset != -1) {
472           msg("    docStoreOffset=" + docStoreOffset);
473           segInfoStat.docStoreOffset = docStoreOffset;
474           msg("    docStoreSegment=" + info.getDocStoreSegment());
475           segInfoStat.docStoreSegment = info.getDocStoreSegment();
476           msg("    docStoreIsCompoundFile=" + info.getDocStoreIsCompoundFile());
477           segInfoStat.docStoreCompoundFile = info.getDocStoreIsCompoundFile();
478         }
479         final String delFileName = info.getDelFileName();
480         if (delFileName == null){
481           msg("    no deletions");
482           segInfoStat.hasDeletions = false;
483         }
484         else{
485           msg("    has deletions [delFileName=" + delFileName + "]");
486           segInfoStat.hasDeletions = true;
487           segInfoStat.deletionsFileName = delFileName;
488         }
489         if (infoStream != null)
490           infoStream.print("    test: open reader.........");
491         reader = SegmentReader.get(info);
492 
493         segInfoStat.openReaderPassed = true;
494 
495         final int numDocs = reader.numDocs();
496         toLoseDocCount = numDocs;
497         if (reader.hasDeletions()) {
498           if (reader.deletedDocs.count() != info.getDelCount()) {
499             throw new RuntimeException("delete count mismatch: info=" + info.getDelCount() + " vs deletedDocs.count()=" + reader.deletedDocs.count());
500           }
501           if (reader.deletedDocs.count() > reader.maxDoc()) {
502             throw new RuntimeException("too many deleted docs: maxDoc()=" + reader.maxDoc() + " vs deletedDocs.count()=" + reader.deletedDocs.count());
503           }
504           if (info.docCount - numDocs != info.getDelCount()){
505             throw new RuntimeException("delete count mismatch: info=" + info.getDelCount() + " vs reader=" + (info.docCount - numDocs));
506           }
507           segInfoStat.numDeleted = info.docCount - numDocs;
508           msg("OK [" + (segInfoStat.numDeleted) + " deleted docs]");
509         } else {
510           if (info.getDelCount() != 0) {
511             throw new RuntimeException("delete count mismatch: info=" + info.getDelCount() + " vs reader=" + (info.docCount - numDocs));
512           }
513           msg("OK");
514         }
515         if (reader.maxDoc() != info.docCount)
516           throw new RuntimeException("SegmentReader.maxDoc() " + reader.maxDoc() + " != SegmentInfos.docCount " + info.docCount);
517 
518         // Test getFieldNames()
519         if (infoStream != null) {
520           infoStream.print("    test: fields..............");
521         }         
522         Collection fieldNames = reader.getFieldNames(IndexReader.FieldOption.ALL);
523         msg("OK [" + fieldNames.size() + " fields]");
524         segInfoStat.numFields = fieldNames.size();
525         
526         // Test Field Norms
527         segInfoStat.fieldNormStatus = testFieldNorms(fieldNames, reader);
528 
529         // Test the Term Index
530         segInfoStat.termIndexStatus = testTermIndex(info, reader);
531 
532         // Test Stored Fields
533         segInfoStat.storedFieldStatus = testStoredFields(info, reader, nf);
534 
535         // Test Term Vectors
536         segInfoStat.termVectorStatus = testTermVectors(info, reader, nf);
537 
538         // Rethrow the first exception we encountered
539         //  This will cause stats for failed segments to be incremented properly
540         if (segInfoStat.fieldNormStatus.error != null) {
541           throw new RuntimeException("Field Norm test failed");
542         } else if (segInfoStat.termIndexStatus.error != null) {
543           throw new RuntimeException("Term Index test failed");
544         } else if (segInfoStat.storedFieldStatus.error != null) {
545           throw new RuntimeException("Stored Field test failed");
546         } else if (segInfoStat.termVectorStatus.error != null) {
547           throw new RuntimeException("Term Vector test failed");
548         }
549 
550         msg("");
551 
552       } catch (Throwable t) {
553         msg("FAILED");
554         String comment;
555         comment = "fixIndex() would remove reference to this segment";
556         msg("    WARNING: " + comment + "; full exception:");
557         if (infoStream != null)
558           t.printStackTrace(infoStream);
559         msg("");
560         result.totLoseDocCount += toLoseDocCount;
561         result.numBadSegments++;
562         continue;
563       } finally {
564         if (reader != null)
565           reader.close();
566       }
567 
568       // Keeper
569       result.newSegments.add(info.clone());
570     }
571 
572     if (0 == result.numBadSegments) {
573       result.clean = true;
574       msg("No problems were detected with this index.\n");
575     } else
576       msg("WARNING: " + result.numBadSegments + " broken segments (containing " + result.totLoseDocCount + " documents) detected");
577 
578     return result;
579   }

581   /**
582    * Test field norms.
583    */
584   private Status.FieldNormStatus testFieldNorms(Collection fieldNames, SegmentReader reader) {
585     final Status.FieldNormStatus status = new Status.FieldNormStatus();
586 
587     try {
588       // Test Field Norms
589       if (infoStream != null) {
590         infoStream.print("    test: field norms.........");
591       }
592       Iterator it = fieldNames.iterator();
593       final byte[] b = new byte[reader.maxDoc()];
594       while (it.hasNext()) {
595         final String fieldName = (String) it.next();
596         reader.norms(fieldName, b, 0);
597         ++status.totFields;
598       }
599 
600       msg("OK [" + status.totFields + " fields]");
601     } catch (Throwable e) {
602       msg("ERROR [" + String.valueOf(e.getMessage()) + "]");
603       status.error = e;
604       if (infoStream != null) {
605         e.printStackTrace(infoStream);
606       }
607     }
608 
609     return status;
610   }

612   /**
613    * Test the term index.
614    */
615   private Status.TermIndexStatus testTermIndex(SegmentInfo info, SegmentReader reader) {
616     final Status.TermIndexStatus status = new Status.TermIndexStatus();
617 
618     try {
619       if (infoStream != null) {
620         infoStream.print("    test: terms, freq, prox...");
621       }
622 
623       final TermEnum termEnum = reader.terms();
624       final TermPositions termPositions = reader.termPositions();
625 
626       // Used only to count up # deleted docs for this term
627       final MySegmentTermDocs myTermDocs = new MySegmentTermDocs(reader);
628 
629       final int maxDoc = reader.maxDoc();
630 
631       while (termEnum.next()) {
632         status.termCount++;
633         final Term term = termEnum.term();
634         final int docFreq = termEnum.docFreq();
635         termPositions.seek(term);
636         int lastDoc = -1;
637         int freq0 = 0;
638         status.totFreq += docFreq;
639         while (termPositions.next()) {
640           freq0++;
641           final int doc = termPositions.doc();
642           final int freq = termPositions.freq();
643           if (doc <= lastDoc)
644             throw new RuntimeException("term " + term + ": doc " + doc + " <= lastDoc " + lastDoc);
645           if (doc >= maxDoc)
646             throw new RuntimeException("term " + term + ": doc " + doc + " >= maxDoc " + maxDoc);
647 
648           lastDoc = doc;
649           if (freq <= 0)
650             throw new RuntimeException("term " + term + ": doc " + doc + ": freq " + freq + " is out of bounds");
651             
652           int lastPos = -1;
653           status.totPos += freq;
654           for(int j=0;j<freq;j++) {
655             final int pos = termPositions.nextPosition();
656             if (pos < -1)
657               throw new RuntimeException("term " + term + ": doc " + doc + ": pos " + pos + " is out of bounds");
658             if (pos < lastPos)
659               throw new RuntimeException("term " + term + ": doc " + doc + ": pos " + pos + " < lastPos " + lastPos);
660           }
661         }
662 
663         // Now count how many deleted docs occurred in
664         // this term:
665         final int delCount;
666         if (reader.hasDeletions()) {
667           myTermDocs.seek(term);
668           while(myTermDocs.next()) { }
669           delCount = myTermDocs.delCount;
670         } else {
671           delCount = 0; 
672         }
673 
674         if (freq0 + delCount != docFreq) {
675           throw new RuntimeException("term " + term + " docFreq=" + 
676                                      docFreq + " != num docs seen " + freq0 + " + num docs deleted " + delCount);
677         }
678       }
679 
680       msg("OK [" + status.termCount + " terms; " + status.totFreq + " terms/docs pairs; " + status.totPos + " tokens]");
681 
682     } catch (Throwable e) {
683       msg("ERROR [" + String.valueOf(e.getMessage()) + "]");
684       status.error = e;
685       if (infoStream != null) {
686         e.printStackTrace(infoStream);
687       }
688     }
689 
690     return status;
691   }

695    */
696   private Status.StoredFieldStatus testStoredFields(SegmentInfo info, SegmentReader reader, NumberFormat format) {
697     final Status.StoredFieldStatus status = new Status.StoredFieldStatus();
698 
699     try {
700       if (infoStream != null) {
701         infoStream.print("    test: stored fields.......");
702       }
703 
704       // Scan stored fields for all documents
705       for (int j = 0; j < info.docCount; ++j) {
706         if (!reader.isDeleted(j)) {
707           status.docCount++;
708           Document doc = reader.document(j);
709           status.totFields += doc.getFields().size();
710         }
711       }      
712 
713       // Validate docCount
714       if (status.docCount != reader.numDocs()) {
715         throw new RuntimeException("docCount=" + status.docCount + " but saw " + status.docCount + " undeleted docs");
716       }
717 
718       msg("OK [" + status.totFields + " total field count; avg " + 
719           format.format((((float) status.totFields)/status.docCount)) + " fields per doc]");      
720     } catch (Throwable e) {
721       msg("ERROR [" + String.valueOf(e.getMessage()) + "]");
722       status.error = e;
723       if (infoStream != null) {
724         e.printStackTrace(infoStream);
725       }
726     }
727 
728     return status;
729   }

765   /** Repairs the index using previously returned result
766    *  from {@link #checkIndex}.  Note that this does not
767    *  remove any of the unreferenced files after it's done;
768    *  you must separately open an {@link IndexWriter}, which
769    *  deletes unreferenced files when it's created.
770    *
771    * <p><b>WARNING</b>: this writes a
772    *  new segments file into the index, effectively removing
773    *  all documents in broken segments from the index.
774    *  BE CAREFUL.
775    *
776    * <p><b>WARNING</b>: Make sure you only call this when the
777    *  index is not opened  by any writer. */
778   public void fixIndex(Status result) throws IOException {
779     if (result.partial)
780       throw new IllegalArgumentException("can only fix an index that was fully checked (this status checked a subset of segments)");
781     result.newSegments.commit(result.dir);
782   }

796   /** Command-line interface to check and fix an index.
797 
798     <p>
799     Run it like this:
800     <pre>
801     java -ea:org.apache.lucene... org.apache.lucene.index.CheckIndex pathToIndex [-fix] [-segment X] [-segment Y]
802     </pre>
803     <ul>
804     <li><code>-fix</code>: actually write a new segments_N file, removing any problematic segments
805 
806     <li><code>-segment X</code>: only check the specified
807     segment(s).  This can be specified multiple times,
808     to check more than one segment, eg <code>-segment _2
809     -segment _a</code>.  You can't use this with the -fix
810     option.
811     </ul>
812 
813     <p><b>WARNING</b>: <code>-fix</code> should only be used on an emergency basis as it will cause
814                        documents (perhaps many) to be permanently removed from the index.  Always make
815                        a backup copy of your index before running this!  Do not run this tool on an index
816                        that is actively being written to.  You have been warned!
817 
818     <p>                Run without -fix, this tool will open the index, report version information
819                        and report any exceptions it hits and what action it would take if -fix were
820                        specified.  With -fix, this tool will remove any segments that have issues and
821                        write a new segments_N file.  This means all documents contained in the affected
822                        segments will be removed.
823 
824     <p>
825                        This tool exits with exit code 1 if the index cannot be opened or has any
826                        corruption, else 0.
827    */
828   public static void main(String[] args) throws IOException, InterruptedException {
829 
830     boolean doFix = false;
831     List onlySegments = new ArrayList();
832     String indexPath = null;
833     int i = 0;
834     while(i < args.length) {
835       if (args[i].equals("-fix")) {
836         doFix = true;
837         i++;
838       } else if (args[i].equals("-segment")) {
839         if (i == args.length-1) {
840           System.out.println("ERROR: missing name for -segment option");
841           System.exit(1);
842         }
843         onlySegments.add(args[i+1]);
844         i += 2;
845       } else {
846         if (indexPath != null) {
847           System.out.println("ERROR: unexpected extra argument '" + args[i] + "'");
848           System.exit(1);
849         }
850         indexPath = args[i];
851         i++;
852       }
853     }
854 
855     if (indexPath == null) {
856       System.out.println("\nERROR: index path not specified");
857       System.out.println("\nUsage: java org.apache.lucene.index.CheckIndex pathToIndex [-fix] [-segment X] [-segment Y]\n" +
858                          "\n" +
859                          "  -fix: actually write a new segments_N file, removing any problematic segments\n" +
860                          "  -segment X: only check the specified segments.  This can be specified multiple\n" + 
861                          "              times, to check more than one segment, eg '-segment _2 -segment _a'.\n" +
862                          "              You can't use this with the -fix option\n" +
863                          "\n" + 
864                          "**WARNING**: -fix should only be used on an emergency basis as it will cause\n" +
865                          "documents (perhaps many) to be permanently removed from the index.  Always make\n" +
866                          "a backup copy of your index before running this!  Do not run this tool on an index\n" +
867                          "that is actively being written to.  You have been warned!\n" +
868                          "\n" +
869                          "Run without -fix, this tool will open the index, report version information\n" +
870                          "and report any exceptions it hits and what action it would take if -fix were\n" +
871                          "specified.  With -fix, this tool will remove any segments that have issues and\n" + 
872                          "write a new segments_N file.  This means all documents contained in the affected\n" +
873                          "segments will be removed.\n" +
874                          "\n" +
875                          "This tool exits with exit code 1 if the index cannot be opened or has any\n" +
876                          "corruption, else 0.\n");
877       System.exit(1);
878     }
879 
880     if (!assertsOn())
881       System.out.println("\nNOTE: testing will be more thorough if you run java with '-ea:org.apache.lucene...', so assertions are enabled");
882 
883     if (onlySegments.size() == 0)
884       onlySegments = null;
885     else if (doFix) {
886       System.out.println("ERROR: cannot specify both -fix and -segment");
887       System.exit(1);
888     }
889 
890     System.out.println("\nOpening index @ " + indexPath + "\n");
891     Directory dir = null;
892     try {
893       dir = FSDirectory.open(new File(indexPath));
894     } catch (Throwable t) {
895       System.out.println("ERROR: could not open directory \"" + indexPath + "\"; exiting");
896       t.printStackTrace(System.out);
897       System.exit(1);
898     }
899 
900     CheckIndex checker = new CheckIndex(dir);
901     checker.setInfoStream(System.out);
902 
903     Status result = checker.checkIndex(onlySegments);
904     if (result.missingSegments) {
905       System.exit(1);
906     }
907 
908     if (!result.clean) {
909       if (!doFix) {
910         System.out.println("WARNING: would write new segments file, and " + result.totLoseDocCount + " documents would be lost, if -fix were specified\n");
911       } else {
912         System.out.println("WARNING: " + result.totLoseDocCount + " documents will be lost\n");
913         System.out.println("NOTE: will write new segments file in 5 seconds; this will remove " + result.totLoseDocCount + " docs from the index. THIS IS YOUR LAST CHANCE TO CTRL+C!");
914         for(int s=0;s<5;s++) {
915           Thread.sleep(1000);
916           System.out.println("  " + (5-s) + "...");
917         }
918         System.out.println("Writing...");
919         checker.fixIndex(result);
920         System.out.println("OK");
921         System.out.println("Wrote new segments file \"" + result.newSegments.getCurrentSegmentFileName() + "\"");
922       }
923     }
924     System.out.println("");
925 
926     final int exitCode;
927     if (result != null && result.clean == true)
928       exitCode = 0;
929     else
930       exitCode = 1;
931     System.exit(exitCode);
932   }
