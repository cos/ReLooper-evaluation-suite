20  import java.io.IOException;
21  import org.apache.lucene.store.RAMOutputStream;
22  import org.apache.lucene.util.ArrayUtil;
23  
24  /** This is a DocFieldConsumer that writes stored fields. */
25  final class StoredFieldsWriter {
26  
27    FieldsWriter fieldsWriter;
28    final DocumentsWriter docWriter;
29    final FieldInfos fieldInfos;
30    int lastDocID;
31  
32    PerDoc[] docFreeList = new PerDoc[1];
33    int freeCount;
34  
35    public StoredFieldsWriter(DocumentsWriter docWriter, FieldInfos fieldInfos) {
36      this.docWriter = docWriter;
37      this.fieldInfos = fieldInfos;
38    }
39  
40    public StoredFieldsWriterPerThread addThread(DocumentsWriter.DocState docState) throws IOException {
41      return new StoredFieldsWriterPerThread(docState, this);
42    }
43  
44    synchronized public void flush(SegmentWriteState state) throws IOException {
45  
46      if (state.numDocsInStore > 0) {
47        // It's possible that all documents seen in this segment
48        // hit non-aborting exceptions, in which case we will
49        // not have yet init'd the FieldsWriter:
50        initFieldsWriter();
51  
52        // Fill fdx file to include any final docs that we
53        // skipped because they hit non-aborting exceptions
54        fill(state.numDocsInStore - docWriter.getDocStoreOffset());
55      }
56  
57      if (fieldsWriter != null)
58        fieldsWriter.flush();
59    }
60    
61    private void initFieldsWriter() throws IOException {
62      if (fieldsWriter == null) {
63        final String docStoreSegment = docWriter.getDocStoreSegment();
64        if (docStoreSegment != null) {
65          assert docStoreSegment != null;
66          fieldsWriter = new FieldsWriter(docWriter.directory,
67                                          docStoreSegment,
68                                          fieldInfos);
69          docWriter.addOpenFile(docStoreSegment + "." + IndexFileNames.FIELDS_EXTENSION);
70          docWriter.addOpenFile(docStoreSegment + "." + IndexFileNames.FIELDS_INDEX_EXTENSION);
71          lastDocID = 0;
72        }
73      }
74    }
75  
76    synchronized public void closeDocStore(SegmentWriteState state) throws IOException {
77      final int inc = state.numDocsInStore - lastDocID;
78      if (inc > 0) {
79        initFieldsWriter();
80        fill(state.numDocsInStore - docWriter.getDocStoreOffset());
81      }
82  
83      if (fieldsWriter != null) {
84        fieldsWriter.close();
85        fieldsWriter = null;
86        lastDocID = 0;
87        assert state.docStoreSegmentName != null;
88        state.flushedFiles.add(state.docStoreSegmentName + "." + IndexFileNames.FIELDS_EXTENSION);
89        state.flushedFiles.add(state.docStoreSegmentName + "." + IndexFileNames.FIELDS_INDEX_EXTENSION);
90  
91        state.docWriter.removeOpenFile(state.docStoreSegmentName + "." + IndexFileNames.FIELDS_EXTENSION);
92        state.docWriter.removeOpenFile(state.docStoreSegmentName + "." + IndexFileNames.FIELDS_INDEX_EXTENSION);
93  
94        final String fileName = state.docStoreSegmentName + "." + IndexFileNames.FIELDS_INDEX_EXTENSION;
95  
96        if (4+((long) state.numDocsInStore)*8 != state.directory.fileLength(fileName))
97          throw new RuntimeException("after flush: fdx size mismatch: " + state.numDocsInStore + " docs vs " + state.directory.fileLength(fileName) + " length in bytes of " + fileName + " file exists?=" + state.directory.fileExists(fileName));
98      }
99    }
100 
101   int allocCount;
102 
103   synchronized PerDoc getPerDoc() {
104     if (freeCount == 0) {
105       allocCount++;
106       if (allocCount > docFreeList.length) {
107         // Grow our free list up front to make sure we have
108         // enough space to recycle all outstanding PerDoc
109         // instances
110         assert allocCount == 1+docFreeList.length;
111         docFreeList = new PerDoc[ArrayUtil.getNextSize(allocCount)];
112       }
113       return new PerDoc();
114     } else
115       return docFreeList[--freeCount];
116   }
117 
118   synchronized void abort() {
119     if (fieldsWriter != null) {
120       try {
121         fieldsWriter.close();
122       } catch (Throwable t) {
123       }
124       fieldsWriter = null;
125       lastDocID = 0;
126     }
127   }
128 
129   /** Fills in any hole in the docIDs */
130   void fill(int docID) throws IOException {
131     final int docStoreOffset = docWriter.getDocStoreOffset();
132 
133     // We must "catch up" for all docs before us
134     // that had no stored fields:
135     final int end = docID+docStoreOffset;
136     while(lastDocID < end) {
137       fieldsWriter.skipDocument();
138       lastDocID++;
139     }
140   }
141 
142   synchronized void finishDocument(PerDoc perDoc) throws IOException {
143     assert docWriter.writer.testPoint("StoredFieldsWriter.finishDocument start");
144     initFieldsWriter();
145 
146     fill(perDoc.docID);
147 
148     // Append stored fields to the real FieldsWriter:
149     fieldsWriter.flushDocument(perDoc.numStoredFields, perDoc.fdt);
150     lastDocID++;
151     perDoc.reset();
152     free(perDoc);
153     assert docWriter.writer.testPoint("StoredFieldsWriter.finishDocument end");
154   }
155 
156   public boolean freeRAM() {
157     return false;
158   }
159 
160   synchronized void free(PerDoc perDoc) {
161     assert freeCount < docFreeList.length;
162     assert 0 == perDoc.numStoredFields;
163     assert 0 == perDoc.fdt.length();
164     assert 0 == perDoc.fdt.getFilePointer();
165     docFreeList[freeCount++] = perDoc;
166   }
167 
168   class PerDoc extends DocumentsWriter.DocWriter {
169 
170     // TODO: use something more memory efficient; for small
171     // docs the 1024 buffer size of RAMOutputStream wastes alot
172     RAMOutputStream fdt = new RAMOutputStream();
173     int numStoredFields;
174 
175     void reset() {
176       fdt.reset();
177       numStoredFields = 0;
178     }
179 
180     void abort() {
181       reset();
182       free(this);
183     }
184 
185     public long sizeInBytes() {
186       return fdt.sizeInBytes();
187     }
188 
189     public void finish() throws IOException {
190       finishDocument(this);
191     }
192   }
193 }
