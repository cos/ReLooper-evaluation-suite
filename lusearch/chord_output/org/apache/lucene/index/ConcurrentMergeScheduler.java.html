22  import java.io.IOException;

26  /** A {@link MergeScheduler} that runs each merge using a
27   *  separate thread, up until a maximum number of threads
28   *  ({@link #setMaxThreadCount}) at which when a merge is
29   *  needed, the thread(s) that are updating the index will
30   *  pause until one or more merges completes.  This is a
31   *  simple way to use concurrency in the indexing process
32   *  without having to create and manage application level
33   *  threads. */
34  
35  public class ConcurrentMergeScheduler extends MergeScheduler {
36  
37    private int mergeThreadPriority = -1;
38  
39    protected List mergeThreads = new ArrayList();
40  
41    // Max number of threads allowed to be merging at once
42    private int maxThreadCount = 3;
43  
44    protected Directory dir;
45  
46    private boolean closed;
47    protected IndexWriter writer;
48    protected int mergeThreadCount;
49  
50    public ConcurrentMergeScheduler() {
51      if (allInstances != null) {
52        // Only for testing
53        addMyself();
54      }
55    }

57    /** Sets the max # simultaneous threads that may be
58     *  running.  If a merge is necessary yet we already have
59     *  this many threads running, the incoming thread (that
60     *  is calling add/updateDocument) will block until
61     *  a merge thread has completed. */
62    public void setMaxThreadCount(int count) {
63      if (count < 1)
64        throw new IllegalArgumentException("count should be at least 1");
65      maxThreadCount = count;
66    }
67  
68    /** Get the max # simultaneous threads that may be
69     *  running. @see #setMaxThreadCount. */
70    public int getMaxThreadCount() {
71      return maxThreadCount;
72    }
73  
74    /** Return the priority that merge threads run at.  By
75     *  default the priority is 1 plus the priority of (ie,
76     *  slightly higher priority than) the first thread that
77     *  calls merge. */
78    public synchronized int getMergeThreadPriority() {
79      initMergeThreadPriority();
80      return mergeThreadPriority;
81    }
82  
83    /** Return the priority that merge threads run at. */
84    public synchronized void setMergeThreadPriority(int pri) {
85      if (pri > Thread.MAX_PRIORITY || pri < Thread.MIN_PRIORITY)
86        throw new IllegalArgumentException("priority must be in range " + Thread.MIN_PRIORITY + " .. " + Thread.MAX_PRIORITY + " inclusive");
87      mergeThreadPriority = pri;
88  
89      final int numThreads = mergeThreadCount();
90      for(int i=0;i<numThreads;i++) {
91        MergeThread merge = (MergeThread) mergeThreads.get(i);
92        merge.setThreadPriority(pri);
93      }
94    }
95  
96    private boolean verbose() {
97      return writer != null && writer.verbose();
98    }
99    
100   private void message(String message) {
101     if (verbose())
102       writer.message("CMS: " + message);
103   }

105   private synchronized void initMergeThreadPriority() {
106     if (mergeThreadPriority == -1) {
107       // Default to slightly higher priority than our
108       // calling thread
109       mergeThreadPriority = 1+Thread.currentThread().getPriority();
110       if (mergeThreadPriority > Thread.MAX_PRIORITY)
111         mergeThreadPriority = Thread.MAX_PRIORITY;
112     }
113   }
114 
115   public void close() {
116     closed = true;
117   }
118 
119   public synchronized void sync() {
120     while(mergeThreadCount() > 0) {
121       if (verbose())
122         message("now wait for threads; currently " + mergeThreads.size() + " still running");
123       final int count = mergeThreads.size();
124       if (verbose()) {
125         for(int i=0;i<count;i++)
126           message("    " + i + ": " + ((MergeThread) mergeThreads.get(i)));
127       }
128       
129       try {
130         wait();
131       } catch (InterruptedException ie) {
132         // In 3.0 we will change this to throw
133         // InterruptedException instead
134         Thread.currentThread().interrupt();
135         throw new RuntimeException(ie);
136       }
137     }
138   }
139 
140   private synchronized int mergeThreadCount() {
141     int count = 0;
142     final int numThreads = mergeThreads.size();
143     for(int i=0;i<numThreads;i++)
144       if (((MergeThread) mergeThreads.get(i)).isAlive())
145         count++;
146     return count;
147   }

149   public void merge(IndexWriter writer)
150     throws CorruptIndexException, IOException {
151 
152     // TODO: enable this once we are on JRE 1.5
153     // assert !Thread.holdsLock(writer);
154 
155     this.writer = writer;
156 
157     initMergeThreadPriority();
158 
159     dir = writer.getDirectory();
160 
161     // First, quickly run through the newly proposed merges
162     // and add any orthogonal merges (ie a merge not
163     // involving segments already pending to be merged) to
164     // the queue.  If we are way behind on merging, many of
165     // these newly proposed merges will likely already be
166     // registered.
167 
168     if (verbose()) {
169       message("now merge");
170       message("  index: " + writer.segString());
171     }
172     
173     // Iterate, pulling from the IndexWriter's queue of
174     // pending merges, until it's empty:
175     while(true) {
176 
177       // TODO: we could be careful about which merges to do in
178       // the BG (eg maybe the "biggest" ones) vs FG, which
179       // merges to do first (the easiest ones?), etc.
180 
181       MergePolicy.OneMerge merge = writer.getNextMerge();
182       if (merge == null) {
183         if (verbose())
184           message("  no more merges pending; now return");
185         return;
186       }
187 
188       // We do this w/ the primary thread to keep
189       // deterministic assignment of segment names
190       writer.mergeInit(merge);
191 
192       boolean success = false;
193       try {
194         synchronized(this) {
195           final MergeThread merger;
196           while (mergeThreadCount() >= maxThreadCount) {
197             if (verbose())
198               message("    too many merge threads running; stalling...");
199             try {
200               wait();
201             } catch (InterruptedException ie) {
202               // In 3.0 we will change this to throw
203               // InterruptedException instead
204               Thread.currentThread().interrupt();
205               throw new RuntimeException(ie);
206             }
207           }
208 
209           if (verbose())
210             message("  consider merge " + merge.segString(dir));
211       
212           assert mergeThreadCount() < maxThreadCount;
213 
214           // OK to spawn a new merge thread to handle this
215           // merge:
216           merger = getMergeThread(writer, merge);
217           mergeThreads.add(merger);
218           if (verbose())
219             message("    launch new thread [" + merger.getName() + "]");
220 
221           merger.start();
222           success = true;
223         }
224       } finally {
225         if (!success) {
226           writer.mergeFinish(merge);
227         }
228       }
229     }
230   }

232   /** Does the actual merge, by calling {@link IndexWriter#merge} */
233   protected void doMerge(MergePolicy.OneMerge merge)
234     throws IOException {
235     writer.merge(merge);
236   }

238   /** Create and return a new MergeThread */
239   protected synchronized MergeThread getMergeThread(IndexWriter writer, MergePolicy.OneMerge merge) throws IOException {
240     final MergeThread thread = new MergeThread(writer, merge);
241     thread.setThreadPriority(mergeThreadPriority);
242     thread.setDaemon(true);
243     thread.setName("Lucene Merge Thread #" + mergeThreadCount++);
244     return thread;
245   }
246 
247   protected class MergeThread extends Thread {
248 
249     IndexWriter writer;
250     MergePolicy.OneMerge startMerge;
251     MergePolicy.OneMerge runningMerge;
252 
253     public MergeThread(IndexWriter writer, MergePolicy.OneMerge startMerge) throws IOException {
254       this.writer = writer;
255       this.startMerge = startMerge;
256     }
257 
258     public synchronized void setRunningMerge(MergePolicy.OneMerge merge) {
259       runningMerge = merge;
260     }
261 
262     public synchronized MergePolicy.OneMerge getRunningMerge() {
263       return runningMerge;
264     }
265 
266     public void setThreadPriority(int pri) {
267       try {
268         setPriority(pri);
269       } catch (NullPointerException npe) {
270         // Strangely, Sun's JDK 1.5 on Linux sometimes
271         // throws NPE out of here...
272       } catch (SecurityException se) {
273         // Ignore this because we will still run fine with
274         // normal thread priority
275       }
276     }
277 
278     public void run() {
279       
280       // First time through the while loop we do the merge
281       // that we were started with:
282       MergePolicy.OneMerge merge = this.startMerge;
283       
284       try {
285 
286         if (verbose())
287           message("  merge thread: start");
288 
289         while(true) {
290           setRunningMerge(merge);
291           doMerge(merge);
292 
293           // Subsequent times through the loop we do any new
294           // merge that writer says is necessary:
295           merge = writer.getNextMerge();
296           if (merge != null) {
297             writer.mergeInit(merge);
298             if (verbose())
299               message("  merge thread: do another merge " + merge.segString(dir));
300           } else
301             break;
302         }
303 
304         if (verbose())
305           message("  merge thread: done");
306 
307       } catch (Throwable exc) {
308 
309         // Ignore the exception if it was due to abort:
310         if (!(exc instanceof MergePolicy.MergeAbortedException)) {
311           if (!suppressExceptions) {
312             // suppressExceptions is normally only set during
313             // testing.
314             anyExceptions = true;
315             handleMergeException(exc);
316           }
317         }
318       } finally {
319         synchronized(ConcurrentMergeScheduler.this) {
320           ConcurrentMergeScheduler.this.notifyAll();
321           boolean removed = mergeThreads.remove(this);
322           assert removed;
323         }
324       }
325     }
326 
327     public String toString() {
328       MergePolicy.OneMerge merge = getRunningMerge();
329       if (merge == null)
330         merge = startMerge;
331       return "merge thread: " + merge.segString(dir);
332     }
333   }
334 
335   /** Called when an exception is hit in a background merge
336    *  thread */
337   protected void handleMergeException(Throwable exc) {
338     try {
339       // When an exception is hit during merge, IndexWriter
340       // removes any partial files and then allows another
341       // merge to run.  If whatever caused the error is not
342       // transient then the exception will keep happening,
343       // so, we sleep here to avoid saturating CPU in such
344       // cases:
345       Thread.sleep(250);
346     } catch (InterruptedException ie) {
347       Thread.currentThread().interrupt();
348       // In 3.0 this will throw InterruptedException
349       throw new RuntimeException(ie);
350     }
351     throw new MergePolicy.MergeException(exc, dir);
352   }

354   static boolean anyExceptions = false;
355 
356   /** Used for testing */
357   public static boolean anyUnhandledExceptions() {
358     if (allInstances == null) {
359       throw new RuntimeException("setTestMode() was not called; often this is because your test case's setUp method fails to call super.setUp in LuceneTestCase");
360     }
361     synchronized(allInstances) {
362       final int count = allInstances.size();
363       // Make sure all outstanding threads are done so we see
364       // any exceptions they may produce:
365       for(int i=0;i<count;i++)
366         ((ConcurrentMergeScheduler) allInstances.get(i)).sync();
367       boolean v = anyExceptions;
368       anyExceptions = false;
369       return v;
370     }
371   }
372 
373   public static void clearUnhandledExceptions() {
374     synchronized(allInstances) {
375       anyExceptions = false;
376     }
377   }

379   /** Used for testing */
380   private void addMyself() {
381     synchronized(allInstances) {
382       final int size = allInstances.size();
383       int upto = 0;
384       for(int i=0;i<size;i++) {
385         final ConcurrentMergeScheduler other = (ConcurrentMergeScheduler) allInstances.get(i);
386         if (!(other.closed && 0 == other.mergeThreadCount()))
387           // Keep this one for now: it still has threads or
388           // may spawn new threads
389           allInstances.set(upto++, other);
390       }
391       allInstances.subList(upto, allInstances.size()).clear();
392       allInstances.add(this);
393     }
394   }
