20  final class FieldInfo {
21    String name;
22    boolean isIndexed;
23    int number;
24  
25    // true if term vector for this field should be stored
26    boolean storeTermVector;
27    boolean storeOffsetWithTermVector;
28    boolean storePositionWithTermVector;
29  
30    boolean omitNorms; // omit norms associated with indexed fields  
31    boolean omitTermFreqAndPositions;
32    
33    boolean storePayloads; // whether this field stores payloads together with term positions
34  
35    FieldInfo(String na, boolean tk, int nu, boolean storeTermVector, 
36              boolean storePositionWithTermVector,  boolean storeOffsetWithTermVector, 
37              boolean omitNorms, boolean storePayloads, boolean omitTermFreqAndPositions) {
38      name = na;
39      isIndexed = tk;
40      number = nu;
41      if (isIndexed) {
42        this.storeTermVector = storeTermVector;
43        this.storeOffsetWithTermVector = storeOffsetWithTermVector;
44        this.storePositionWithTermVector = storePositionWithTermVector;
45        this.storePayloads = storePayloads;
46        this.omitNorms = omitNorms;
47        this.omitTermFreqAndPositions = omitTermFreqAndPositions;
48      } else { // for non-indexed fields, leave defaults
49        this.storeTermVector = false;
50        this.storeOffsetWithTermVector = false;
51        this.storePositionWithTermVector = false;
52        this.storePayloads = false;
53        this.omitNorms = true;
54        this.omitTermFreqAndPositions = false;
55      }
56    }
57  
58    public Object clone() {
59      return new FieldInfo(name, isIndexed, number, storeTermVector, storePositionWithTermVector,
60                           storeOffsetWithTermVector, omitNorms, storePayloads, omitTermFreqAndPositions);
61    }
62  
63    void update(boolean isIndexed, boolean storeTermVector, boolean storePositionWithTermVector, 
64                boolean storeOffsetWithTermVector, boolean omitNorms, boolean storePayloads, boolean omitTermFreqAndPositions) {
65      if (this.isIndexed != isIndexed) {
66        this.isIndexed = true;                      // once indexed, always index
67      }
68      if (isIndexed) { // if updated field data is not for indexing, leave the updates out
69        if (this.storeTermVector != storeTermVector) {
70          this.storeTermVector = true;                // once vector, always vector
71        }
72        if (this.storePositionWithTermVector != storePositionWithTermVector) {
73          this.storePositionWithTermVector = true;                // once vector, always vector
74        }
75        if (this.storeOffsetWithTermVector != storeOffsetWithTermVector) {
76          this.storeOffsetWithTermVector = true;                // once vector, always vector
77        }
78        if (this.storePayloads != storePayloads) {
79          this.storePayloads = true;
80        }
81        if (this.omitNorms != omitNorms) {
82          this.omitNorms = false;                // once norms are stored, always store
83        }
84        if (this.omitTermFreqAndPositions != omitTermFreqAndPositions) {
85          this.omitTermFreqAndPositions = true;                // if one require omitTermFreqAndPositions at least once, it remains off for life
86        }
87      }
88    }
89  }
