17  package org.apache.lucene.index;
18  
19  import org.apache.lucene.util.AttributeSource;
20  
21  /**
22   * This class tracks the number and position / offset parameters of terms
23   * being added to the index. The information collected in this class is
24   * also used to calculate the normalization factor for a field.
25   * 
26   * <p><b>WARNING</b>: This API is new and experimental, and may suddenly
27   * change.</p>
28   */
29  public final class FieldInvertState {
30    int position;
31    int length;
32    int numOverlap;
33    int offset;
34    float boost;
35    AttributeSource attributeSource;
36  
37    public FieldInvertState() {
38    }
39  
40    public FieldInvertState(int position, int length, int numOverlap, int offset, float boost) {
41      this.position = position;
42      this.length = length;
43      this.numOverlap = numOverlap;
44      this.offset = offset;
45      this.boost = boost;
46    }
47  
48    /**
49     * Re-initialize the state, using this boost value.
50     * @param docBoost boost value to use.
51     */
52    void reset(float docBoost) {
53      position = 0;
54      length = 0;
55      numOverlap = 0;
56      offset = 0;
57      boost = docBoost;
58      attributeSource = null;
59    }
60  
61    /**
62     * Get the last processed term position.
63     * @return the position
64     */
65    public int getPosition() {
66      return position;
67    }
68  
69    /**
70     * Get total number of terms in this field.
71     * @return the length
72     */
73    public int getLength() {
74      return length;
75    }
76  
77    /**
78     * Get the number of terms with <code>positionIncrement == 0</code>.
79     * @return the numOverlap
80     */
81    public int getNumOverlap() {
82      return numOverlap;
83    }
84  
85    /**
86     * Get end offset of the last processed term.
87     * @return the offset
88     */
89    public int getOffset() {
90      return offset;
91    }
92  
93    /**
94     * Get boost value. This is the cumulative product of
95     * document boost and field boost for all field instances
96     * sharing the same field name.
97     * @return the boost
98     */
99    public float getBoost() {
100     return boost;
101   }
102   
103   public AttributeSource getAttributeSource() {
104     return attributeSource;
105   }
106 }
