20  final class CharBlockPool {
21  
22    public char[][] buffers = new char[10][];
23    int numBuffer;
24  
25    int bufferUpto = -1;                        // Which buffer we are upto
26    public int charUpto = DocumentsWriter.CHAR_BLOCK_SIZE;             // Where we are in head buffer
27  
28    public char[] buffer;                              // Current head buffer
29    public int charOffset = -DocumentsWriter.CHAR_BLOCK_SIZE;          // Current head offset
30    final private DocumentsWriter docWriter;
31  
32    public CharBlockPool(DocumentsWriter docWriter) {
33      this.docWriter = docWriter;
34    }
35  
36    public void reset() {
37      docWriter.recycleCharBlocks(buffers, 1+bufferUpto);
38      bufferUpto = -1;
39      charUpto = DocumentsWriter.CHAR_BLOCK_SIZE;
40      charOffset = -DocumentsWriter.CHAR_BLOCK_SIZE;
41    }
42  
43    public void nextBuffer() {
44      if (1+bufferUpto == buffers.length) {
45        char[][] newBuffers = new char[(int) (buffers.length*1.5)][];
46        System.arraycopy(buffers, 0, newBuffers, 0, buffers.length);
47        buffers = newBuffers;
48      }
49      buffer = buffers[1+bufferUpto] = docWriter.getCharBlock();
50      bufferUpto++;
51  
52      charUpto = 0;
53      charOffset += DocumentsWriter.CHAR_BLOCK_SIZE;
54    }
55  }
