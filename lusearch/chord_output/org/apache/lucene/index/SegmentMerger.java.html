35  /**
36   * The SegmentMerger class combines two or more Segments, represented by an IndexReader ({@link #add},
37   * into a single Segment.  After adding the appropriate readers, call the merge method to combine the 
38   * segments.
39   *<P> 
40   * If the compoundFile flag is set, then the segments will be merged into a compound file.
41   *   
42   * 
43   * @see #merge
44   * @see #add
45   */

87    SegmentMerger(IndexWriter writer, String name, MergePolicy.OneMerge merge) {
88      directory = writer.getDirectory();
89      segment = name;
90      if (merge != null) {
91        checkAbort = new CheckAbort(merge, directory);
92      } else {
93        checkAbort = new CheckAbort(null, null) {
94          public void work(double units) throws MergeAbortedException {
95            // do nothing
96          }
97        };
98      }
99      termIndexInterval = writer.getTermIndexInterval();
100   }

133   /**
134    * Merges the readers specified by the {@link #add} method
135    * into the directory passed to the constructor.
136    * @param mergeDocStores if false, we will not merge the
137    * stored fields nor vectors files
138    * @return The number of documents that were merged
139    * @throws CorruptIndexException if the index is corrupt
140    * @throws IOException if there is a low-level IO error
141    */
142   final int merge(boolean mergeDocStores) throws CorruptIndexException, IOException {
143 
144     this.mergeDocStores = mergeDocStores;
145     
146     // NOTE: it's important to add calls to
147     // checkAbort.work(...) if you make any changes to this
148     // method that will spend alot of time.  The frequency
149     // of this check impacts how long
150     // IndexWriter.close(false) takes to actually stop the
151     // threads.
152 
153     mergedDocs = mergeFields();
154     mergeTerms();
155     mergeNorms();
156 
157     if (mergeDocStores && fieldInfos.hasVectors())
158       mergeVectors();
159 
160     return mergedDocs;
161   }

163   /**
164    * close all IndexReaders that have been added.
165    * Should not be called before merge().
166    * @throws IOException
167    */

169     for (Iterator iter = readers.iterator(); iter.hasNext();) {
170       ((IndexReader) iter.next()).close();
171     }

174   final List createCompoundFile(String fileName)
175           throws IOException {
176     CompoundFileWriter cfsWriter =
177       new CompoundFileWriter(directory, fileName, checkAbort);
178 
179     List files =
180       new ArrayList(IndexFileNames.COMPOUND_EXTENSIONS.length + 1);    
181     
182     // Basic files
183     for (int i = 0; i < IndexFileNames.COMPOUND_EXTENSIONS.length; i++) {
184       String ext = IndexFileNames.COMPOUND_EXTENSIONS[i];
185 
186       if (ext.equals(IndexFileNames.PROX_EXTENSION) && !hasProx())
187         continue;
188 
189       if (mergeDocStores || (!ext.equals(IndexFileNames.FIELDS_EXTENSION) &&
190                             !ext.equals(IndexFileNames.FIELDS_INDEX_EXTENSION)))
191         files.add(segment + "." + ext);
192     }
193 
194     // Fieldable norm files
195     for (int i = 0; i < fieldInfos.size(); i++) {
196       FieldInfo fi = fieldInfos.fieldInfo(i);
197       if (fi.isIndexed && !fi.omitNorms) {
198         files.add(segment + "." + IndexFileNames.NORMS_EXTENSION);
199         break;
200       }
201     }
202 
203     // Vector files
204     if (fieldInfos.hasVectors() && mergeDocStores) {
205       for (int i = 0; i < IndexFileNames.VECTOR_EXTENSIONS.length; i++) {
206         files.add(segment + "." + IndexFileNames.VECTOR_EXTENSIONS[i]);
207       }
208     }
209 
210     // Now merge all added files
211     Iterator it = files.iterator();
212     while (it.hasNext()) {
213       cfsWriter.addFile((String) it.next());
214     }
215     
216     // Perform the merge
217     cfsWriter.close();
218    
219     return files;
220   }

222   private void addIndexed(IndexReader reader, FieldInfos fInfos,
223       Collection names, boolean storeTermVectors,
224       boolean storePositionWithTermVector, boolean storeOffsetWithTermVector,
225       boolean storePayloads, boolean omitTFAndPositions)
226       throws IOException {
227     Iterator i = names.iterator();
228     while (i.hasNext()) {
229       String field = (String) i.next();
230       fInfos.add(field, true, storeTermVectors,
231           storePositionWithTermVector, storeOffsetWithTermVector, !reader
232               .hasNorms(field), storePayloads, omitTFAndPositions);
233     }
234   }

236   private SegmentReader[] matchingSegmentReaders;
237   private int[] rawDocLengths;
238   private int[] rawDocLengths2;
239 
240   private void setMatchingSegmentReaders() {
241     // If the i'th reader is a SegmentReader and has
242     // identical fieldName -> number mapping, then this
243     // array will be non-null at position i:
244     int numReaders = readers.size();
245     matchingSegmentReaders = new SegmentReader[numReaders];
246 
247     // If this reader is a SegmentReader, and all of its
248     // field name -> number mappings match the "merged"
249     // FieldInfos, then we can do a bulk copy of the
250     // stored fields:
251     for (int i = 0; i < numReaders; i++) {
252       IndexReader reader = (IndexReader) readers.get(i);
253       if (reader instanceof SegmentReader) {
254         SegmentReader segmentReader = (SegmentReader) reader;
255         boolean same = true;
256         FieldInfos segmentFieldInfos = segmentReader.fieldInfos();
257         int numFieldInfos = segmentFieldInfos.size();
258         for (int j = 0; same && j < numFieldInfos; j++) {
259           same = fieldInfos.fieldName(j).equals(segmentFieldInfos.fieldName(j));
260         }
261         if (same) {
262           matchingSegmentReaders[i] = segmentReader;
263         }
264       }
265     }
266 
267     // Used for bulk-reading raw bytes for stored fields
268     rawDocLengths = new int[MAX_RAW_MERGE_DOCS];
269     rawDocLengths2 = new int[MAX_RAW_MERGE_DOCS];
270   }

272   /**
273    * 
274    * @return The number of documents in all of the readers
275    * @throws CorruptIndexException if the index is corrupt
276    * @throws IOException if there is a low-level IO error
277    */
278   private final int mergeFields() throws CorruptIndexException, IOException {
279 
280     if (!mergeDocStores) {
281       // When we are not merging by doc stores, that means
282       // all segments were written as part of a single
283       // autoCommit=false IndexWriter session, so their field
284       // name -> number mapping are the same.  So, we start
285       // with the fieldInfos of the last segment in this
286       // case, to keep that numbering.
287       final SegmentReader sr = (SegmentReader) readers.get(readers.size()-1);
288       fieldInfos = (FieldInfos) sr.core.fieldInfos.clone();
289     } else {
290       fieldInfos = new FieldInfos();          // merge field names
291     }
292 
293     for (Iterator iter = readers.iterator(); iter.hasNext();) {
294       IndexReader reader = (IndexReader) iter.next();
295       if (reader instanceof SegmentReader) {
296         SegmentReader segmentReader = (SegmentReader) reader;
297         FieldInfos readerFieldInfos = segmentReader.fieldInfos();
298         int numReaderFieldInfos = readerFieldInfos.size();
299         for (int j = 0; j < numReaderFieldInfos; j++) {
300           FieldInfo fi = readerFieldInfos.fieldInfo(j);
301           fieldInfos.add(fi.name, fi.isIndexed, fi.storeTermVector,
302               fi.storePositionWithTermVector, fi.storeOffsetWithTermVector,
303               !reader.hasNorms(fi.name), fi.storePayloads,
304               fi.omitTermFreqAndPositions);
305         }
306       } else {
307         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.TERMVECTOR_WITH_POSITION_OFFSET), true, true, true, false, false);
308         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.TERMVECTOR_WITH_POSITION), true, true, false, false, false);
309         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.TERMVECTOR_WITH_OFFSET), true, false, true, false, false);
310         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.TERMVECTOR), true, false, false, false, false);
311         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.OMIT_TERM_FREQ_AND_POSITIONS), false, false, false, false, true);
312         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.STORES_PAYLOADS), false, false, false, true, false);
313         addIndexed(reader, fieldInfos, reader.getFieldNames(FieldOption.INDEXED), false, false, false, false, false);
314         fieldInfos.add(reader.getFieldNames(FieldOption.UNINDEXED), false);
315       }
316     }
317     fieldInfos.write(directory, segment + ".fnm");
318 
319     int docCount = 0;
320 
321     setMatchingSegmentReaders();
322 
323     if (mergeDocStores) {
324 
325       // for merging we don't want to compress/uncompress the data, so to tell the FieldsReader that we're
326       // in  merge mode, we use this FieldSelector
327       FieldSelector fieldSelectorMerge = new FieldSelector() {
328           public FieldSelectorResult accept(String fieldName) {
329             return FieldSelectorResult.LOAD_FOR_MERGE;
330           }        
331         };
332 
333       // merge field values
334       final FieldsWriter fieldsWriter = new FieldsWriter(directory, segment, fieldInfos);
335 
336       try {
337         int idx = 0;
338         for (Iterator iter = readers.iterator(); iter.hasNext();) {
339           final IndexReader reader = (IndexReader) iter.next();
340           final SegmentReader matchingSegmentReader = matchingSegmentReaders[idx++];
341           FieldsReader matchingFieldsReader = null;
342           if (matchingSegmentReader != null) {
343             final FieldsReader fieldsReader = matchingSegmentReader.getFieldsReader();
344             if (fieldsReader != null && fieldsReader.canReadRawDocs()) {            
345               matchingFieldsReader = fieldsReader;
346             }
347           }
348           if (reader.hasDeletions()) {
349             docCount += copyFieldsWithDeletions(fieldSelectorMerge, fieldsWriter,
350                                                 reader, matchingFieldsReader);
351           } else {
352             docCount += copyFieldsNoDeletions(fieldSelectorMerge, fieldsWriter,
353                                               reader, matchingFieldsReader);
354           }
355         }
356       } finally {
357         fieldsWriter.close();
358       }
359 
360       final String fileName = segment + "." + IndexFileNames.FIELDS_INDEX_EXTENSION;
361       final long fdxFileLength = directory.fileLength(fileName);
362 
363       if (4+((long) docCount)*8 != fdxFileLength)
364         // This is most likely a bug in Sun JRE 1.6.0_04/_05;
365         // we detect that the bug has struck, here, and
366         // throw an exception to prevent the corruption from
367         // entering the index.  See LUCENE-1282 for
368         // details.
369         throw new RuntimeException("mergeFields produced an invalid result: docCount is " + docCount + " but fdx file size is " + fdxFileLength + " file=" + fileName + " file exists?=" + directory.fileExists(fileName) + "; now aborting this merge to prevent index corruption");
370 
371     } else
372       // If we are skipping the doc stores, that means there
373       // are no deletions in any of these segments, so we
374       // just sum numDocs() of each segment to get total docCount
375       for (Iterator iter = readers.iterator(); iter.hasNext();) {
376         docCount += ((IndexReader) iter.next()).numDocs();
377       }
378 
379     return docCount;
380   }

382   private int copyFieldsWithDeletions(final FieldSelector fieldSelectorMerge,
383                                       final FieldsWriter fieldsWriter, final IndexReader reader,
384                                       final FieldsReader matchingFieldsReader)
385     throws IOException, MergeAbortedException, CorruptIndexException {
386     int docCount = 0;
387     final int maxDoc = reader.maxDoc();
388     if (matchingFieldsReader != null) {
389       // We can bulk-copy because the fieldInfos are "congruent"
390       for (int j = 0; j < maxDoc;) {
391         if (reader.isDeleted(j)) {
392           // skip deleted docs
393           ++j;
394           continue;
395         }
396         // We can optimize this case (doing a bulk byte copy) since the field 
397         // numbers are identical
398         int start = j, numDocs = 0;
399         do {
400           j++;
401           numDocs++;
402           if (j >= maxDoc) break;
403           if (reader.isDeleted(j)) {
404             j++;
405             break;
406           }
407         } while(numDocs < MAX_RAW_MERGE_DOCS);
408         
409         IndexInput stream = matchingFieldsReader.rawDocs(rawDocLengths, start, numDocs);
410         fieldsWriter.addRawDocuments(stream, rawDocLengths, numDocs);
411         docCount += numDocs;
412         checkAbort.work(300 * numDocs);
413       }
414     } else {
415       for (int j = 0; j < maxDoc; j++) {
416         if (reader.isDeleted(j)) {
417           // skip deleted docs
418           continue;
419         }
420         // NOTE: it's very important to first assign to doc then pass it to
421         // termVectorsWriter.addAllDocVectors; see LUCENE-1282
422         Document doc = reader.document(j, fieldSelectorMerge);
423         fieldsWriter.addDocument(doc);
424         docCount++;
425         checkAbort.work(300);
426       }
427     }
428     return docCount;
429   }

431   private int copyFieldsNoDeletions(FieldSelector fieldSelectorMerge,
432                                     final FieldsWriter fieldsWriter, final IndexReader reader,
433                                     final FieldsReader matchingFieldsReader)
434     throws IOException, MergeAbortedException, CorruptIndexException {
435     final int maxDoc = reader.maxDoc();
436     int docCount = 0;
437     if (matchingFieldsReader != null) {
438       // We can bulk-copy because the fieldInfos are "congruent"
439       while (docCount < maxDoc) {
440         int len = Math.min(MAX_RAW_MERGE_DOCS, maxDoc - docCount);
441         IndexInput stream = matchingFieldsReader.rawDocs(rawDocLengths, docCount, len);
442         fieldsWriter.addRawDocuments(stream, rawDocLengths, len);
443         docCount += len;
444         checkAbort.work(300 * len);
445       }
446     } else {
447       for (; docCount < maxDoc; docCount++) {
448         // NOTE: it's very important to first assign to doc then pass it to
449         // termVectorsWriter.addAllDocVectors; see LUCENE-1282
450         Document doc = reader.document(docCount, fieldSelectorMerge);
451         fieldsWriter.addDocument(doc);
452         checkAbort.work(300);
453       }
454     }
455     return docCount;
456   }

458   /**
459    * Merge the TermVectors from each of the segments into the new one.
460    * @throws IOException
461    */
462   private final void mergeVectors() throws IOException {
463     TermVectorsWriter termVectorsWriter = 
464       new TermVectorsWriter(directory, segment, fieldInfos);
465 
466     try {
467       int idx = 0;
468       for (Iterator iter = readers.iterator(); iter.hasNext();) {
469         final SegmentReader matchingSegmentReader = matchingSegmentReaders[idx++];
470         TermVectorsReader matchingVectorsReader = null;
471         if (matchingSegmentReader != null) {
472           TermVectorsReader vectorsReader = matchingSegmentReader.getTermVectorsReaderOrig();
473 
474           // If the TV* files are an older format then they cannot read raw docs:
475           if (vectorsReader != null && vectorsReader.canReadRawDocs()) {
476             matchingVectorsReader = vectorsReader;
477           }
478         }
479         final IndexReader reader = (IndexReader) iter.next();
480         if (reader.hasDeletions()) {
481           copyVectorsWithDeletions(termVectorsWriter, matchingVectorsReader, reader);
482         } else {
483           copyVectorsNoDeletions(termVectorsWriter, matchingVectorsReader, reader);
484           
485         }
486       }
487     } finally {
488       termVectorsWriter.close();
489     }
490 
491     final String fileName = segment + "." + IndexFileNames.VECTORS_INDEX_EXTENSION;
492     final long tvxSize = directory.fileLength(fileName);
493 
494     if (4+((long) mergedDocs)*16 != tvxSize)
495       // This is most likely a bug in Sun JRE 1.6.0_04/_05;
496       // we detect that the bug has struck, here, and
497       // throw an exception to prevent the corruption from
498       // entering the index.  See LUCENE-1282 for
499       // details.
500       throw new RuntimeException("mergeVectors produced an invalid result: mergedDocs is " + mergedDocs + " but tvx size is " + tvxSize + " file=" + fileName + " file exists?=" + directory.fileExists(fileName) + "; now aborting this merge to prevent index corruption");
501   }

503   private void copyVectorsWithDeletions(final TermVectorsWriter termVectorsWriter,
504                                         final TermVectorsReader matchingVectorsReader,
505                                         final IndexReader reader)
506     throws IOException, MergeAbortedException {
507     final int maxDoc = reader.maxDoc();
508     if (matchingVectorsReader != null) {
509       // We can bulk-copy because the fieldInfos are "congruent"
510       for (int docNum = 0; docNum < maxDoc;) {
511         if (reader.isDeleted(docNum)) {
512           // skip deleted docs
513           ++docNum;
514           continue;
515         }
516         // We can optimize this case (doing a bulk byte copy) since the field 
517         // numbers are identical
518         int start = docNum, numDocs = 0;
519         do {
520           docNum++;
521           numDocs++;
522           if (docNum >= maxDoc) break;
523           if (reader.isDeleted(docNum)) {
524             docNum++;
525             break;
526           }
527         } while(numDocs < MAX_RAW_MERGE_DOCS);
528         
529         matchingVectorsReader.rawDocs(rawDocLengths, rawDocLengths2, start, numDocs);
530         termVectorsWriter.addRawDocuments(matchingVectorsReader, rawDocLengths, rawDocLengths2, numDocs);
531         checkAbort.work(300 * numDocs);
532       }
533     } else {
534       for (int docNum = 0; docNum < maxDoc; docNum++) {
535         if (reader.isDeleted(docNum)) {
536           // skip deleted docs
537           continue;
538         }
539         
540         // NOTE: it's very important to first assign to vectors then pass it to
541         // termVectorsWriter.addAllDocVectors; see LUCENE-1282
542         TermFreqVector[] vectors = reader.getTermFreqVectors(docNum);
543         termVectorsWriter.addAllDocVectors(vectors);
544         checkAbort.work(300);
545       }
546     }
547   }

549   private void copyVectorsNoDeletions(final TermVectorsWriter termVectorsWriter,
550                                       final TermVectorsReader matchingVectorsReader,
551                                       final IndexReader reader)
552       throws IOException, MergeAbortedException {
553     final int maxDoc = reader.maxDoc();
554     if (matchingVectorsReader != null) {
555       // We can bulk-copy because the fieldInfos are "congruent"
556       int docCount = 0;
557       while (docCount < maxDoc) {
558         int len = Math.min(MAX_RAW_MERGE_DOCS, maxDoc - docCount);
559         matchingVectorsReader.rawDocs(rawDocLengths, rawDocLengths2, docCount, len);
560         termVectorsWriter.addRawDocuments(matchingVectorsReader, rawDocLengths, rawDocLengths2, len);
561         docCount += len;
562         checkAbort.work(300 * len);
563       }
564     } else {
565       for (int docNum = 0; docNum < maxDoc; docNum++) {
566         // NOTE: it's very important to first assign to vectors then pass it to
567         // termVectorsWriter.addAllDocVectors; see LUCENE-1282
568         TermFreqVector[] vectors = reader.getTermFreqVectors(docNum);
569         termVectorsWriter.addAllDocVectors(vectors);
570         checkAbort.work(300);
571       }
572     }

577   private final void mergeTerms() throws CorruptIndexException, IOException {
578 
579     SegmentWriteState state = new SegmentWriteState(null, directory, segment, null, mergedDocs, 0, termIndexInterval);
580 
581     final FormatPostingsFieldsConsumer consumer = new FormatPostingsFieldsWriter(state, fieldInfos);
582 
583     try {
584       queue = new SegmentMergeQueue(readers.size());
585 
586       mergeTermInfos(consumer);
587 
588     } finally {
589       consumer.finish();
590       if (queue != null) queue.close();
591     }
592   }
593 
594   boolean omitTermFreqAndPositions;
595 
596   private final void mergeTermInfos(final FormatPostingsFieldsConsumer consumer) throws CorruptIndexException, IOException {
597     int base = 0;
598     final int readerCount = readers.size();
599     for (int i = 0; i < readerCount; i++) {
600       IndexReader reader = (IndexReader) readers.get(i);
601       TermEnum termEnum = reader.terms();
602       SegmentMergeInfo smi = new SegmentMergeInfo(base, termEnum, reader);
603       int[] docMap  = smi.getDocMap();
604       if (docMap != null) {
605         if (docMaps == null) {
606           docMaps = new int[readerCount][];
607           delCounts = new int[readerCount];
608         }
609         docMaps[i] = docMap;
610         delCounts[i] = smi.reader.maxDoc() - smi.reader.numDocs();
611       }
612       
613       base += reader.numDocs();
614 
615       assert reader.numDocs() == reader.maxDoc() - smi.delCount;
616 
617       if (smi.next())
618         queue.add(smi);               // initialize queue
619       else
620         smi.close();
621     }
622 
623     SegmentMergeInfo[] match = new SegmentMergeInfo[readers.size()];
624 
625     String currentField = null;
626     FormatPostingsTermsConsumer termsConsumer = null;
627 
628     while (queue.size() > 0) {
629       int matchSize = 0;              // pop matching terms
630       match[matchSize++] = (SegmentMergeInfo) queue.pop();
631       Term term = match[0].term;
632       SegmentMergeInfo top = (SegmentMergeInfo) queue.top();
633 
634       while (top != null && term.compareTo(top.term) == 0) {
635         match[matchSize++] = (SegmentMergeInfo) queue.pop();
636         top = (SegmentMergeInfo) queue.top();
637       }
638 
639       if (currentField != term.field) {
640         currentField = term.field;
641         if (termsConsumer != null)
642           termsConsumer.finish();
643         final FieldInfo fieldInfo = fieldInfos.fieldInfo(currentField);
644         termsConsumer = consumer.addField(fieldInfo);
645         omitTermFreqAndPositions = fieldInfo.omitTermFreqAndPositions;
646       }
647 
648       int df = appendPostings(termsConsumer, match, matchSize);       // add new TermInfo
649 
650       checkAbort.work(df/3.0);
651 
652       while (matchSize > 0) {
653         SegmentMergeInfo smi = match[--matchSize];
654         if (smi.next())
655           queue.add(smi);             // restore queue
656         else
657           smi.close();                // done with a segment
658       }
659     }
660   }

672   /** Process postings from multiple segments all positioned on the
673    *  same term. Writes out merged entries into freqOutput and
674    *  the proxOutput streams.
675    *
676    * @param smis array of segments
677    * @param n number of cells in the array actually occupied
678    * @return number of documents across all segments where this term was found
679    * @throws CorruptIndexException if the index is corrupt
680    * @throws IOException if there is a low-level IO error
681    */
682   private final int appendPostings(final FormatPostingsTermsConsumer termsConsumer, SegmentMergeInfo[] smis, int n)
683         throws CorruptIndexException, IOException {
684 
685     final FormatPostingsDocsConsumer docConsumer = termsConsumer.addTerm(smis[0].term.text);
686     int df = 0;
687     for (int i = 0; i < n; i++) {
688       SegmentMergeInfo smi = smis[i];
689       TermPositions postings = smi.getPositions();
690       assert postings != null;
691       int base = smi.base;
692       int[] docMap = smi.getDocMap();
693       postings.seek(smi.termEnum);
694 
695       while (postings.next()) {
696         df++;
697         int doc = postings.doc();
698         if (docMap != null)
699           doc = docMap[doc];                      // map around deletions
700         doc += base;                              // convert to merged space
701 
702         final int freq = postings.freq();
703         final FormatPostingsPositionsConsumer posConsumer = docConsumer.addDoc(doc, freq);
704 
705         if (!omitTermFreqAndPositions) {
706           for (int j = 0; j < freq; j++) {
707             final int position = postings.nextPosition();
708             final int payloadLength = postings.getPayloadLength();
709             if (payloadLength > 0) {
710               if (payloadBuffer == null || payloadBuffer.length < payloadLength)
711                 payloadBuffer = new byte[payloadLength];
712               postings.getPayload(payloadBuffer, 0);
713             }
714             posConsumer.addPosition(position, payloadBuffer, 0, payloadLength);
715           }
716           posConsumer.finish();
717         }
718       }
719     }
720     docConsumer.finish();
721 
722     return df;
723   }

725   private void mergeNorms() throws IOException {
726     byte[] normBuffer = null;
727     IndexOutput output = null;
728     try {
729       int numFieldInfos = fieldInfos.size();
730       for (int i = 0; i < numFieldInfos; i++) {
731         FieldInfo fi = fieldInfos.fieldInfo(i);
732         if (fi.isIndexed && !fi.omitNorms) {
733           if (output == null) { 
734             output = directory.createOutput(segment + "." + IndexFileNames.NORMS_EXTENSION);
735             output.writeBytes(NORMS_HEADER,NORMS_HEADER.length);
736           }
737           for (Iterator iter = readers.iterator(); iter.hasNext();) {
738             IndexReader reader = (IndexReader) iter.next();
739             int maxDoc = reader.maxDoc();
740             if (normBuffer == null || normBuffer.length < maxDoc) {
741               // the buffer is too small for the current segment
742               normBuffer = new byte[maxDoc];
743             }
744             reader.norms(fi.name, normBuffer, 0);
745             if (!reader.hasDeletions()) {
746               //optimized case for segments without deleted docs
747               output.writeBytes(normBuffer, maxDoc);
748             } else {
749               // this segment has deleted docs, so we have to
750               // check for every doc if it is deleted or not
751               for (int k = 0; k < maxDoc; k++) {
752                 if (!reader.isDeleted(k)) {
753                   output.writeByte(normBuffer[k]);
754                 }
755               }
756             }
757             checkAbort.work(maxDoc);
758           }
759         }
760       }
761     } finally {
762       if (output != null) { 
763         output.close();
764       }
765     }
766   }

768   static class CheckAbort {
769     private double workCount;
770     private MergePolicy.OneMerge merge;
771     private Directory dir;
772     public CheckAbort(MergePolicy.OneMerge merge, Directory dir) {
773       this.merge = merge;
774       this.dir = dir;
775     }

777     /**
778      * Records the fact that roughly units amount of work
779      * have been done since this method was last called.
780      * When adding time-consuming code into SegmentMerger,
781      * you should test different values for units to ensure
782      * that the time in between calls to merge.checkAborted
783      * is up to ~ 1 second.
784      */
785     public void work(double units) throws MergePolicy.MergeAbortedException {
786       workCount += units;
787       if (workCount >= 10000.0) {
788         merge.checkAborted(dir);
789         workCount = 0;
790       }
791     }
792   }
