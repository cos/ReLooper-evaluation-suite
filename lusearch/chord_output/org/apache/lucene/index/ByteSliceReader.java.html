20  import org.apache.lucene.store.IndexInput;
21  import org.apache.lucene.store.IndexOutput;
22  import java.io.IOException;
23  
24  /* IndexInput that knows how to read the byte slices written
25   * by Posting and PostingVector.  We read the bytes in
26   * each slice until we hit the end of that slice at which
27   * point we read the forwarding address of the next slice
28   * and then jump to it.*/
29  final class ByteSliceReader extends IndexInput {
30    ByteBlockPool pool;
31    int bufferUpto;
32    byte[] buffer;
33    public int upto;
34    int limit;
35    int level;
36    public int bufferOffset;
37  
38    public int endIndex;
39  
40    public void init(ByteBlockPool pool, int startIndex, int endIndex) {
41  
42      assert endIndex-startIndex >= 0;
43      assert startIndex >= 0;
44      assert endIndex >= 0;
45  
46      this.pool = pool;
47      this.endIndex = endIndex;
48  
49      level = 0;
50      bufferUpto = startIndex / DocumentsWriter.BYTE_BLOCK_SIZE;
51      bufferOffset = bufferUpto * DocumentsWriter.BYTE_BLOCK_SIZE;
52      buffer = pool.buffers[bufferUpto];
53      upto = startIndex & DocumentsWriter.BYTE_BLOCK_MASK;
54  
55      final int firstSize = ByteBlockPool.levelSizeArray[0];
56  
57      if (startIndex+firstSize >= endIndex) {
58        // There is only this one slice to read
59        limit = endIndex & DocumentsWriter.BYTE_BLOCK_MASK;
60      } else
61        limit = upto+firstSize-4;
62    }
63  
64    public boolean eof() {
65      assert upto + bufferOffset <= endIndex;
66      return upto + bufferOffset == endIndex;
67    }
68  
69    public byte readByte() {
70      assert !eof();
71      assert upto <= limit;
72      if (upto == limit)
73        nextSlice();
74      return buffer[upto++];
75    }
76  
77    public long writeTo(IndexOutput out) throws IOException {
78      long size = 0;
79      while(true) {
80        if (limit + bufferOffset == endIndex) {
81          assert endIndex - bufferOffset >= upto;
82          out.writeBytes(buffer, upto, limit-upto);
83          size += limit-upto;
84          break;
85        } else {
86          out.writeBytes(buffer, upto, limit-upto);
87          size += limit-upto;
88          nextSlice();
89        }
90      }
91  
92      return size;
93    }
94  
95    public void nextSlice() {
96  
97      // Skip to our next slice
98      final int nextIndex = ((buffer[limit]&0xff)<<24) + ((buffer[1+limit]&0xff)<<16) + ((buffer[2+limit]&0xff)<<8) + (buffer[3+limit]&0xff);
99  
100     level = ByteBlockPool.nextLevelArray[level];
101     final int newSize = ByteBlockPool.levelSizeArray[level];
102 
103     bufferUpto = nextIndex / DocumentsWriter.BYTE_BLOCK_SIZE;
104     bufferOffset = bufferUpto * DocumentsWriter.BYTE_BLOCK_SIZE;
105 
106     buffer = pool.buffers[bufferUpto];
107     upto = nextIndex & DocumentsWriter.BYTE_BLOCK_MASK;
108 
109     if (nextIndex + newSize >= endIndex) {
110       // We are advancing to the final slice
111       assert endIndex - nextIndex > 0;
112       limit = endIndex - bufferOffset;
113     } else {
114       // This is not the final slice (subtract 4 for the
115       // forwarding address at the end of this new slice)
116       limit = upto+newSize-4;
117     }
118   }
119 
120   public void readBytes(byte[] b, int offset, int len) {
121     while(len > 0) {
122       final int numLeft = limit-upto;
123       if (numLeft < len) {
124         // Read entire slice
125         System.arraycopy(buffer, upto, b, offset, numLeft);
126         offset += numLeft;
127         len -= numLeft;
128         nextSlice();
129       } else {
130         // This slice is the last one
131         System.arraycopy(buffer, upto, b, offset, len);
132         upto += len;
133         break;
134       }
135     }
136   }
137 
138   public long getFilePointer() {throw new RuntimeException("not implemented");}
139   public long length() {throw new RuntimeException("not implemented");}
140   public void seek(long pos) {throw new RuntimeException("not implemented");}
141   public void close() {throw new RuntimeException("not implemented");}
142 }
