27  final class TermVectorsTermsWriterPerField extends TermsHashConsumerPerField {

29    final TermVectorsTermsWriterPerThread perThread;

31    final TermVectorsTermsWriter termsWriter;
32    final FieldInfo fieldInfo;
33    final DocumentsWriter.DocState docState;
34    final FieldInvertState fieldState;

36    boolean doVectors;
37    boolean doVectorPositions;
38    boolean doVectorOffsets;

41    OffsetAttribute offsetAttribute = null;

43    public TermVectorsTermsWriterPerField(TermsHashPerField termsHashPerField, TermVectorsTermsWriterPerThread perThread, FieldInfo fieldInfo) {
44      this.termsHashPerField = termsHashPerField;
45      this.perThread = perThread;
46      this.termsWriter = perThread.termsWriter;
47      this.fieldInfo = fieldInfo;
48      docState = termsHashPerField.docState;
49      fieldState = termsHashPerField.fieldState;
50    }
51  
52    int getStreamCount() {
53      return 2;
54    }
55  
56    boolean start(Fieldable[] fields, int count) {
57      doVectors = false;
58      doVectorPositions = false;
59      doVectorOffsets = false;
60  
61      for(int i=0;i<count;i++) {
62        Fieldable field = fields[i];
63        if (field.isIndexed() && field.isTermVectorStored()) {
64          doVectors = true;
65          doVectorPositions |= field.isStorePositionWithTermVector();
66          doVectorOffsets |= field.isStoreOffsetWithTermVector();
67        }
68      }
69  
70      if (doVectors) {
71        if (perThread.doc == null) {
72          perThread.doc = termsWriter.getPerDoc();
73          perThread.doc.docID = docState.docID;
74          assert perThread.doc.numVectorFields == 0;
75          assert 0 == perThread.doc.tvf.length();
76          assert 0 == perThread.doc.tvf.getFilePointer();
77        } else {
78          assert perThread.doc.docID == docState.docID;
79  
80          if (termsHashPerField.numPostings != 0)
81            // Only necessary if previous doc hit a
82            // non-aborting exception while writing vectors in
83            // this field:
84            termsHashPerField.reset();
85        }
86      }
87  
88      // TODO: only if needed for performance
89      //perThread.postingsCount = 0;
90  
91      return doVectors;
92    }     
93  
94    public void abort() {}

96    /** Called once per field per document if term vectors
97     *  are enabled, to write the vectors to
98     *  RAMOutputStream, which is then quickly flushed to
99     *  * the real term vectors files in the Directory. */
100   void finish() throws IOException {
101 
102     assert docState.testPoint("TermVectorsTermsWriterPerField.finish start");
103 
104     final int numPostings = termsHashPerField.numPostings;
105 
106     assert numPostings >= 0;
107 
108     if (!doVectors || numPostings == 0)
109       return;
110 
111     if (numPostings > maxNumPostings)
112       maxNumPostings = numPostings;
113 
114     final IndexOutput tvf = perThread.doc.tvf;
115 
116     // This is called once, after inverting all occurrences
117     // of a given field in the doc.  At this point we flush
118     // our hash into the DocWriter.
119 
120     assert fieldInfo.storeTermVector;
121     assert perThread.vectorFieldsInOrder(fieldInfo);
122 
123     perThread.doc.addField(termsHashPerField.fieldInfo.number);
124 
125     final RawPostingList[] postings = termsHashPerField.sortPostings();
126 
127     tvf.writeVInt(numPostings);
128     byte bits = 0x0;
129     if (doVectorPositions)
130       bits |= TermVectorsReader.STORE_POSITIONS_WITH_TERMVECTOR;
131     if (doVectorOffsets) 
132       bits |= TermVectorsReader.STORE_OFFSET_WITH_TERMVECTOR;
133     tvf.writeByte(bits);
134 
135     int encoderUpto = 0;
136     int lastTermBytesCount = 0;
137 
138     final ByteSliceReader reader = perThread.vectorSliceReader;
139     final char[][] charBuffers = perThread.termsHashPerThread.charPool.buffers;
140     for(int j=0;j<numPostings;j++) {
141       final TermVectorsTermsWriter.PostingList posting = (TermVectorsTermsWriter.PostingList) postings[j];
142       final int freq = posting.freq;
143           
144       final char[] text2 = charBuffers[posting.textStart >> DocumentsWriter.CHAR_BLOCK_SHIFT];
145       final int start2 = posting.textStart & DocumentsWriter.CHAR_BLOCK_MASK;
146 
147       // We swap between two encoders to save copying
148       // last Term's byte array
149       final UnicodeUtil.UTF8Result utf8Result = perThread.utf8Results[encoderUpto];
150 
151       // TODO: we could do this incrementally
152       UnicodeUtil.UTF16toUTF8(text2, start2, utf8Result);
153       final int termBytesCount = utf8Result.length;
154 
155       // TODO: UTF16toUTF8 could tell us this prefix
156       // Compute common prefix between last term and
157       // this term
158       int prefix = 0;
159       if (j > 0) {
160         final byte[] lastTermBytes = perThread.utf8Results[1-encoderUpto].result;
161         final byte[] termBytes = perThread.utf8Results[encoderUpto].result;
162         while(prefix < lastTermBytesCount && prefix < termBytesCount) {
163           if (lastTermBytes[prefix] != termBytes[prefix])
164             break;
165           prefix++;
166         }
167       }
168       encoderUpto = 1-encoderUpto;
169       lastTermBytesCount = termBytesCount;
170 
171       final int suffix = termBytesCount - prefix;
172       tvf.writeVInt(prefix);
173       tvf.writeVInt(suffix);
174       tvf.writeBytes(utf8Result.result, prefix, suffix);
175       tvf.writeVInt(freq);
176 
177       if (doVectorPositions) {
178         termsHashPerField.initReader(reader, posting, 0);
179         reader.writeTo(tvf);
180       }
181 
182       if (doVectorOffsets) {
183         termsHashPerField.initReader(reader, posting, 1);
184         reader.writeTo(tvf);
185       }
186     }
187 
188     termsHashPerField.reset();
189     perThread.termsHashPerThread.reset(false);
190   }
191 
192   void shrinkHash() {
193     termsHashPerField.shrinkHash(maxNumPostings);
194     maxNumPostings = 0;
195   }

197   void start(Fieldable f) {
198     if (doVectorOffsets) {
199       offsetAttribute = (OffsetAttribute) fieldState.attributeSource.addAttribute(OffsetAttribute.class);
200     } else {
201       offsetAttribute = null;
202     }
203   }
204 
205   void newTerm(RawPostingList p0) {
206 
207     assert docState.testPoint("TermVectorsTermsWriterPerField.newTerm start");
208 
209     TermVectorsTermsWriter.PostingList p = (TermVectorsTermsWriter.PostingList) p0;
210 
211     p.freq = 1;
212 
213     if (doVectorOffsets) {
214       int startOffset = fieldState.offset + offsetAttribute.startOffset();;
215       int endOffset = fieldState.offset + offsetAttribute.endOffset();
216       
217       termsHashPerField.writeVInt(1, startOffset);
218       termsHashPerField.writeVInt(1, endOffset - startOffset);
219       p.lastOffset = endOffset;
220     }
221 
222     if (doVectorPositions) {
223       termsHashPerField.writeVInt(0, fieldState.position);
224       p.lastPosition = fieldState.position;
225     }
226   }
227 
228   void addTerm(RawPostingList p0) {
229 
230     assert docState.testPoint("TermVectorsTermsWriterPerField.addTerm start");
231 
232     TermVectorsTermsWriter.PostingList p = (TermVectorsTermsWriter.PostingList) p0;
233     p.freq++;
234 
235     if (doVectorOffsets) {
236       int startOffset = fieldState.offset + offsetAttribute.startOffset();;
237       int endOffset = fieldState.offset + offsetAttribute.endOffset();
238       
239       termsHashPerField.writeVInt(1, startOffset - p.lastOffset);
240       termsHashPerField.writeVInt(1, endOffset - startOffset);
241       p.lastOffset = endOffset;
242     }
243 
244     if (doVectorPositions) {
245       termsHashPerField.writeVInt(0, fieldState.position - p.lastPosition);
246       p.lastPosition = fieldState.position;
247     }
248   }
