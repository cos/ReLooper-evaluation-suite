19  import java.util.Comparator;
20  
21  /**
22   * Compares {@link org.apache.lucene.index.TermVectorEntry}s first by frequency and then by
23   * the term (case-sensitive)
24   *
25   **/
26  public class TermVectorEntryFreqSortedComparator implements Comparator {
27    public int compare(Object object, Object object1) {
28      int result = 0;
29      TermVectorEntry entry = (TermVectorEntry) object;
30      TermVectorEntry entry1 = (TermVectorEntry) object1;
31      result = entry1.getFrequency() - entry.getFrequency();
32      if (result == 0)
33      {
34        result = entry.getTerm().compareTo(entry1.getTerm());
35        if (result == 0)
36        {
37          result = entry.getField().compareTo(entry1.getField());
38        }
39      }
40      return result;
41    }
42  }
