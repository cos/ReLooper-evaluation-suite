20  import org.apache.lucene.store.Directory;
21  import org.apache.lucene.store.IndexOutput;
22  import org.apache.lucene.store.IndexInput;
23  import java.util.LinkedList;
24  import java.util.HashSet;
25  import java.util.Iterator;
26  import java.io.IOException;
27  
28  
29  /**
30   * Combines multiple files into a single compound file.
31   * The file format:<br>
32   * <ul>
33   *     <li>VInt fileCount</li>
34   *     <li>{Directory}
35   *         fileCount entries with the following structure:</li>
36   *         <ul>
37   *             <li>long dataOffset</li>
38   *             <li>String fileName</li>
39   *         </ul>
40   *     <li>{File Data}
41   *         fileCount entries with the raw data of the corresponding file</li>
42   * </ul>
43   *
44   * The fileCount integer indicates how many files are contained in this compound
45   * file. The {directory} that follows has that many entries. Each directory entry
46   * contains a long pointer to the start of this file's data section, and a String
47   * with that file's name.
48   *
49   *
50   * @version $Id: CompoundFileWriter.java 690539 2008-08-30 17:33:06Z mikemccand $
51   */
52  final class CompoundFileWriter {
53  
54      private static final class FileEntry {
55          /** source file */
56          String file;
57  
58          /** temporary holder for the start of directory entry for this file */
59          long directoryOffset;
60  
61          /** temporary holder for the start of this file's data section */
62          long dataOffset;
63      }
64  
65  
66      private Directory directory;
67      private String fileName;
68      private HashSet ids;
69      private LinkedList entries;
70      private boolean merged = false;
71      private SegmentMerger.CheckAbort checkAbort;
72  
73      /** Create the compound stream in the specified file. The file name is the
74       *  entire name (no extensions are added).
75       *  @throws NullPointerException if <code>dir</code> or <code>name</code> is null
76       */
77      public CompoundFileWriter(Directory dir, String name) {
78        this(dir, name, null);
79      }
80  
81      CompoundFileWriter(Directory dir, String name, SegmentMerger.CheckAbort checkAbort) {
82          if (dir == null)
83              throw new NullPointerException("directory cannot be null");
84          if (name == null)
85              throw new NullPointerException("name cannot be null");
86          this.checkAbort = checkAbort;
87          directory = dir;
88          fileName = name;
89          ids = new HashSet();
90          entries = new LinkedList();
91      }
92  
93      /** Returns the directory of the compound file. */
94      public Directory getDirectory() {
95          return directory;
96      }
97  
98      /** Returns the name of the compound file. */
99      public String getName() {
100         return fileName;
101     }
102 
103     /** Add a source stream. <code>file</code> is the string by which the 
104      *  sub-stream will be known in the compound stream.
105      * 
106      *  @throws IllegalStateException if this writer is closed
107      *  @throws NullPointerException if <code>file</code> is null
108      *  @throws IllegalArgumentException if a file with the same name
109      *   has been added already
110      */
111     public void addFile(String file) {
112         if (merged)
113             throw new IllegalStateException(
114                 "Can't add extensions after merge has been called");
115 
116         if (file == null)
117             throw new NullPointerException(
118                 "file cannot be null");
119 
120         if (! ids.add(file))
121             throw new IllegalArgumentException(
122                 "File " + file + " already added");
123 
124         FileEntry entry = new FileEntry();
125         entry.file = file;
126         entries.add(entry);
127     }
128 
129     /** Merge files with the extensions added up to now.
130      *  All files with these extensions are combined sequentially into the
131      *  compound stream. After successful merge, the source files
132      *  are deleted.
133      *  @throws IllegalStateException if close() had been called before or
134      *   if no file has been added to this object
135      */
136     public void close() throws IOException {
137         if (merged)
138             throw new IllegalStateException(
139                 "Merge already performed");
140 
141         if (entries.isEmpty())
142             throw new IllegalStateException(
143                 "No entries to merge have been defined");
144 
145         merged = true;
146 
147         // open the compound stream
148         IndexOutput os = null;
149         try {
150             os = directory.createOutput(fileName);
151 
152             // Write the number of entries
153             os.writeVInt(entries.size());
154 
155             // Write the directory with all offsets at 0.
156             // Remember the positions of directory entries so that we can
157             // adjust the offsets later
158             Iterator it = entries.iterator();
159             long totalSize = 0;
160             while(it.hasNext()) {
161                 FileEntry fe = (FileEntry) it.next();
162                 fe.directoryOffset = os.getFilePointer();
163                 os.writeLong(0);    // for now
164                 os.writeString(fe.file);
165                 totalSize += directory.fileLength(fe.file);
166             }
167 
168             // Pre-allocate size of file as optimization --
169             // this can potentially help IO performance as
170             // we write the file and also later during
171             // searching.  It also uncovers a disk-full
172             // situation earlier and hopefully without
173             // actually filling disk to 100%:
174             final long finalLength = totalSize+os.getFilePointer();
175             os.setLength(finalLength);
176 
177             // Open the files and copy their data into the stream.
178             // Remember the locations of each file's data section.
179             byte buffer[] = new byte[16384];
180             it = entries.iterator();
181             while(it.hasNext()) {
182                 FileEntry fe = (FileEntry) it.next();
183                 fe.dataOffset = os.getFilePointer();
184                 copyFile(fe, os, buffer);
185             }
186 
187             // Write the data offsets into the directory of the compound stream
188             it = entries.iterator();
189             while(it.hasNext()) {
190                 FileEntry fe = (FileEntry) it.next();
191                 os.seek(fe.directoryOffset);
192                 os.writeLong(fe.dataOffset);
193             }
194 
195             assert finalLength == os.length();
196 
197             // Close the output stream. Set the os to null before trying to
198             // close so that if an exception occurs during the close, the
199             // finally clause below will not attempt to close the stream
200             // the second time.
201             IndexOutput tmp = os;
202             os = null;
203             tmp.close();
204 
205         } finally {
206             if (os != null) try { os.close(); } catch (IOException e) { }
207         }
208     }
209 
210     /** Copy the contents of the file with specified extension into the
211      *  provided output stream. Use the provided buffer for moving data
212      *  to reduce memory allocation.
213      */
214     private void copyFile(FileEntry source, IndexOutput os, byte buffer[])
215     throws IOException
216     {
217         IndexInput is = null;
218         try {
219             long startPtr = os.getFilePointer();
220 
221             is = directory.openInput(source.file);
222             long length = is.length();
223             long remainder = length;
224             int chunk = buffer.length;
225 
226             while(remainder > 0) {
227                 int len = (int) Math.min(chunk, remainder);
228                 is.readBytes(buffer, 0, len, false);
229                 os.writeBytes(buffer, len);
230                 remainder -= len;
231                 if (checkAbort != null)
232                   // Roughly every 2 MB we will check if
233                   // it's time to abort
234                   checkAbort.work(80);
235             }
236 
237             // Verify that remainder is 0
238             if (remainder != 0)
239                 throw new IOException(
240                     "Non-zero remainder length after copying: " + remainder
241                     + " (id: " + source.file + ", length: " + length
242                     + ", buffer size: " + chunk + ")");
243 
244             // Verify that the output length diff is equal to original file
245             long endPtr = os.getFilePointer();
246             long diff = endPtr - startPtr;
247             if (diff != length)
248                 throw new IOException(
249                     "Difference in the output file offsets " + diff
250                     + " does not match the original file length " + length);
251 
252         } finally {
253             if (is != null) is.close();
254         }
255     }
256 }
