33  /** Writes norms.  Each thread X field accumulates the norms
34   *  for the doc/fields it saw, then the flush method below
35   *  merges all of these together into a single _X.nrm file.
36   */
37  
38  final class NormsWriter extends InvertedDocEndConsumer {
39  
40    private static final byte defaultNorm = Similarity.encodeNorm(1.0f);
41    private FieldInfos fieldInfos;
42    public InvertedDocEndConsumerPerThread addThread(DocInverterPerThread docInverterPerThread) {
43      return new NormsWriterPerThread(docInverterPerThread, this);
44    }
45  
46    public void abort() {}
47  
48    // We only write the _X.nrm file at flush
49    void files(Collection files) {}
50  
51    void setFieldInfos(FieldInfos fieldInfos) {
52      this.fieldInfos = fieldInfos;
53    }
54  
55    /** Produce _X.nrm if any document had a field with norms
56     *  not disabled */
57    public void flush(Map threadsAndFields, SegmentWriteState state) throws IOException {
58  
59      final Map byField = new HashMap();
60  
61      // Typically, each thread will have encountered the same
62      // field.  So first we collate by field, ie, all
63      // per-thread field instances that correspond to the
64      // same FieldInfo
65      final Iterator it = threadsAndFields.entrySet().iterator();
66      while(it.hasNext()) {
67        Map.Entry entry = (Map.Entry) it.next();
68  
69        Collection fields = (Collection) entry.getValue();
70        Iterator fieldsIt = fields.iterator();
71  
72        while(fieldsIt.hasNext()) {
73          NormsWriterPerField perField = (NormsWriterPerField) fieldsIt.next();
74  
75          if (perField.upto > 0) {
76            // It has some norms
77            List l = (List) byField.get(perField.fieldInfo);
78            if (l == null) {
79              l = new ArrayList();
80              byField.put(perField.fieldInfo, l);
81            }
82            l.add(perField);
83          } else
84            // Remove this field since we haven't seen it
85            // since the previous flush
86            fieldsIt.remove();
87        }
88      }
89  
90      final String normsFileName = state.segmentName + "." + IndexFileNames.NORMS_EXTENSION;
91      state.flushedFiles.add(normsFileName);
92      IndexOutput normsOut = state.directory.createOutput(normsFileName);
93  
94      try {
95        normsOut.writeBytes(SegmentMerger.NORMS_HEADER, 0, SegmentMerger.NORMS_HEADER.length);
96  
97        final int numField = fieldInfos.size();
98  
99        int normCount = 0;
100 
101       for(int fieldNumber=0;fieldNumber<numField;fieldNumber++) {
102 
103         final FieldInfo fieldInfo = fieldInfos.fieldInfo(fieldNumber);
104 
105         List toMerge = (List) byField.get(fieldInfo);
106         int upto = 0;
107         if (toMerge != null) {
108 
109           final int numFields = toMerge.size();
110 
111           normCount++;
112 
113           final NormsWriterPerField[] fields = new NormsWriterPerField[numFields];
114           int[] uptos = new int[numFields];
115 
116           for(int j=0;j<numFields;j++)
117             fields[j] = (NormsWriterPerField) toMerge.get(j);
118 
119           int numLeft = numFields;
120               
121           while(numLeft > 0) {
122 
123             assert uptos[0] < fields[0].docIDs.length : " uptos[0]=" + uptos[0] + " len=" + (fields[0].docIDs.length);
124 
125             int minLoc = 0;
126             int minDocID = fields[0].docIDs[uptos[0]];
127 
128             for(int j=1;j<numLeft;j++) {
129               final int docID = fields[j].docIDs[uptos[j]];
130               if (docID < minDocID) {
131                 minDocID = docID;
132                 minLoc = j;
133               }
134             }
135 
136             assert minDocID < state.numDocs;
137 
138             // Fill hole
139             for(;upto<minDocID;upto++)
140               normsOut.writeByte(defaultNorm);
141 
142             normsOut.writeByte(fields[minLoc].norms[uptos[minLoc]]);
143             (uptos[minLoc])++;
144             upto++;
145 
146             if (uptos[minLoc] == fields[minLoc].upto) {
147               fields[minLoc].reset();
148               if (minLoc != numLeft-1) {
149                 fields[minLoc] = fields[numLeft-1];
150                 uptos[minLoc] = uptos[numLeft-1];
151               }
152               numLeft--;
153             }
154           }
155           
156           // Fill final hole with defaultNorm
157           for(;upto<state.numDocs;upto++)
158             normsOut.writeByte(defaultNorm);
159         } else if (fieldInfo.isIndexed && !fieldInfo.omitNorms) {
160           normCount++;
161           // Fill entire field with default norm:
162           for(;upto<state.numDocs;upto++)
163             normsOut.writeByte(defaultNorm);
164         }
165 
166         assert 4+normCount*state.numDocs == normsOut.getFilePointer() : ".nrm file size mismatch: expected=" + (4+normCount*state.numDocs) + " actual=" + normsOut.getFilePointer();
167       }
168 
169     } finally {
170       normsOut.close();
171     }
172   }
173 
174   void closeDocStore(SegmentWriteState state) {}
175 }
