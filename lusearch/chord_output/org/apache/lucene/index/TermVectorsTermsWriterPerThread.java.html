20  import org.apache.lucene.util.UnicodeUtil;
21  
22  final class TermVectorsTermsWriterPerThread extends TermsHashConsumerPerThread {
23  
24    final TermVectorsTermsWriter termsWriter;
25    final TermsHashPerThread termsHashPerThread;
26    final DocumentsWriter.DocState docState;
27  
28    TermVectorsTermsWriter.PerDoc doc;
29  
30    public TermVectorsTermsWriterPerThread(TermsHashPerThread termsHashPerThread, TermVectorsTermsWriter termsWriter) {
31      this.termsWriter = termsWriter;
32      this.termsHashPerThread = termsHashPerThread;
33      docState = termsHashPerThread.docState;
34    }
35    
36    // Used by perField when serializing the term vectors
37    final ByteSliceReader vectorSliceReader = new ByteSliceReader();
38  
39    final UnicodeUtil.UTF8Result utf8Results[] = {new UnicodeUtil.UTF8Result(),
40                                                  new UnicodeUtil.UTF8Result()};
41  
42    public void startDocument() {
43      assert clearLastVectorFieldName();
44      if (doc != null) {
45        doc.reset();
46        doc.docID = docState.docID;
47      }
48    }
49  
50    public DocumentsWriter.DocWriter finishDocument() {
51      try {
52        return doc;
53      } finally {
54        doc = null;
55      }
56    }
57  
58    public TermsHashConsumerPerField addField(TermsHashPerField termsHashPerField, FieldInfo fieldInfo) {
59      return new TermVectorsTermsWriterPerField(termsHashPerField, this, fieldInfo);
60    }
61  
62    public void abort() {
63      if (doc != null) {
64        doc.abort();
65        doc = null;
66      }
67    }
68  
69    // Called only by assert
70    final boolean clearLastVectorFieldName() {
71      lastVectorFieldName = null;
72      return true;
73    }
74  
75    // Called only by assert
76    String lastVectorFieldName;
77    final boolean vectorFieldsInOrder(FieldInfo fi) {
78      try {
79        if (lastVectorFieldName != null)
80          return lastVectorFieldName.compareTo(fi.name) < 0;
81        else
82          return true;
83      } finally {
84        lastVectorFieldName = fi.name;
85      }
86    }
87  }
