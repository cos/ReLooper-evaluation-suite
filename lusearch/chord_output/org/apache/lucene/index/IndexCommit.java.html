20  import java.util.Collection;
21  import java.util.Map;
22  import java.io.IOException;
23  import org.apache.lucene.store.Directory;
24  
25  /**
26   * <p>Expert: represents a single commit into an index as seen by the
27   * {@link IndexDeletionPolicy} or {@link IndexReader}.</p>
28   *
29   * <p> Changes to the content of an index are made visible
30   * only after the writer who made that change commits by
31   * writing a new segments file
32   * (<code>segments_N</code>). This point in time, when the
33   * action of writing of a new segments file to the directory
34   * is completed, is an index commit.</p>
35   *
36   * <p>Each index commit point has a unique segments file
37   * associated with it. The segments file associated with a
38   * later index commit point would have a larger N.</p>
39   *
40   * <p><b>WARNING</b>: This API is a new and experimental and
41   * may suddenly change. </p>
42  */
43  
44  public abstract class IndexCommit implements IndexCommitPoint {
45  
46    /**
47     * Get the segments file (<code>segments_N</code>) associated 
48     * with this commit point.
49     */
50    public abstract String getSegmentsFileName();
51  
52    /**
53     * Returns all index files referenced by this commit point.
54     */
55    public abstract Collection getFileNames() throws IOException;
56  
57    /**
58     * Returns the {@link Directory} for the index.
59     */
60    public abstract Directory getDirectory();
61    
62    /**
63     * Delete this commit point.  This only applies when using
64     * the commit point in the context of IndexWriter's
65     * IndexDeletionPolicy.
66     * <p>
67     * Upon calling this, the writer is notified that this commit 
68     * point should be deleted. 
69     * <p>
70     * Decision that a commit-point should be deleted is taken by the {@link IndexDeletionPolicy} in effect
71     * and therefore this should only be called by its {@link IndexDeletionPolicy#onInit onInit()} or 
72     * {@link IndexDeletionPolicy#onCommit onCommit()} methods.
73    */
74    public void delete() {
75      throw new UnsupportedOperationException("This IndexCommit does not support this method.");
76    }
77  
78    public boolean isDeleted() {
79      throw new UnsupportedOperationException("This IndexCommit does not support this method.");
80    }
81  
82    /**
83     * Returns true if this commit is an optimized index.
84     */
85    public boolean isOptimized() {
86      throw new UnsupportedOperationException("This IndexCommit does not support this method.");
87    }
88  
89    /**
90     * Two IndexCommits are equal if both their Directory and versions are equal.
91     */
92    public boolean equals(Object other) {
93      if (other instanceof IndexCommit) {
94        IndexCommit otherCommit = (IndexCommit) other;
95        return otherCommit.getDirectory().equals(getDirectory()) && otherCommit.getVersion() == getVersion();
96      } else
97        return false;
98    }
99  
100   public int hashCode() {
101     return getDirectory().hashCode() + getSegmentsFileName().hashCode();
102   }
103 
104   /** Returns the version for this IndexCommit.  This is the
105    *  same value that {@link IndexReader#getVersion} would
106    *  return if it were opened on this commit. */
107   public long getVersion() {
108     throw new UnsupportedOperationException("This IndexCommit does not support this method.");
109   }
110 
111   /** Returns the generation (the _N in segments_N) for this
112    *  IndexCommit */
113   public long getGeneration() {
114     throw new UnsupportedOperationException("This IndexCommit does not support this method.");
115   }
116 
117   /** Convenience method that returns the last modified time
118    *  of the segments_N file corresponding to this index
119    *  commit, equivalent to
120    *  getDirectory().fileModified(getSegmentsFileName()). */
121   public long getTimestamp() throws IOException {
122     return getDirectory().fileModified(getSegmentsFileName());
123   }
124 
125   /** Returns userData, previously passed to {@link
126    *  IndexWriter#commit(Map)} for this commit.  Map is
127    *  String -> String. */
128   public Map getUserData() throws IOException {
129     throw new UnsupportedOperationException("This IndexCommit does not support this method.");
130   }
131 }
