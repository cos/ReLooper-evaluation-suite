20  import java.util.HashMap;
21  import java.util.Map;
22  import java.util.TreeMap;
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Iterator;
26  import java.util.Map.Entry;
27  
28  /** Holds buffered deletes, by docID, term or query.  We
29   *  hold two instances of this class: one for the deletes
30   *  prior to the last flush, the other for deletes after
31   *  the last flush.  This is so if we need to abort
32   *  (discard all buffered docs) we can also discard the
33   *  buffered deletes yet keep the deletes done during
34   *  previously flushed segments. */
35  class BufferedDeletes {
36    int numTerms;
37    Map terms;
38    Map queries = new HashMap();
39    List docIDs = new ArrayList();
40    long bytesUsed;
41    private final boolean doTermSort;
42  
43    public BufferedDeletes(boolean doTermSort) {
44      this.doTermSort = doTermSort;
45      if (doTermSort) {
46        terms = new TreeMap();
47      } else {
48        terms = new HashMap();
49      }
50    }
51  
52    // Number of documents a delete term applies to.
53    final static class Num {
54      private int num;
55  
56      Num(int num) {
57        this.num = num;
58      }
59  
60      int getNum() {
61        return num;
62      }
63  
64      void setNum(int num) {
65        // Only record the new number if it's greater than the
66        // current one.  This is important because if multiple
67        // threads are replacing the same doc at nearly the
68        // same time, it's possible that one thread that got a
69        // higher docID is scheduled before the other
70        // threads.
71        if (num > this.num)
72          this.num = num;
73      }
74    }
75  
76    int size() {
77      // We use numTerms not terms.size() intentionally, so
78      // that deletes by the same term multiple times "count",
79      // ie if you ask to flush every 1000 deletes then even
80      // dup'd terms are counted towards that 1000
81      return numTerms + queries.size() + docIDs.size();
82    }
83  
84    void update(BufferedDeletes in) {
85      numTerms += in.numTerms;
86      bytesUsed += in.bytesUsed;
87      terms.putAll(in.terms);
88      queries.putAll(in.queries);
89      docIDs.addAll(in.docIDs);
90      in.clear();
91    }
92      
93    void clear() {
94      terms.clear();
95      queries.clear();
96      docIDs.clear();
97      numTerms = 0;
98      bytesUsed = 0;
99    }
100 
101   void addBytesUsed(long b) {
102     bytesUsed += b;
103   }
104 
105   boolean any() {
106     return terms.size() > 0 || docIDs.size() > 0 || queries.size() > 0;
107   }
108 
109   // Remaps all buffered deletes based on a completed
110   // merge
111   synchronized void remap(MergeDocIDRemapper mapper,
112                           SegmentInfos infos,
113                           int[][] docMaps,
114                           int[] delCounts,
115                           MergePolicy.OneMerge merge,
116                           int mergeDocCount) {
117 
118     final Map newDeleteTerms;
119 
120     // Remap delete-by-term
121     if (terms.size() > 0) {
122       if (doTermSort) {
123         newDeleteTerms = new TreeMap();
124       } else {
125         newDeleteTerms = new HashMap();
126       }
127       Iterator iter = terms.entrySet().iterator();
128       while(iter.hasNext()) {
129         Entry entry = (Entry) iter.next();
130         Num num = (Num) entry.getValue();
131         newDeleteTerms.put(entry.getKey(),
132                            new Num(mapper.remap(num.getNum())));
133       }
134     } else
135       newDeleteTerms = null;
136 
137     // Remap delete-by-docID
138     final List newDeleteDocIDs;
139 
140     if (docIDs.size() > 0) {
141       newDeleteDocIDs = new ArrayList(docIDs.size());
142       Iterator iter = docIDs.iterator();
143       while(iter.hasNext()) {
144         Integer num = (Integer) iter.next();
145         newDeleteDocIDs.add(new Integer(mapper.remap(num.intValue())));
146       }
147     } else
148       newDeleteDocIDs = null;
149 
150     // Remap delete-by-query
151     final HashMap newDeleteQueries;
152     
153     if (queries.size() > 0) {
154       newDeleteQueries = new HashMap(queries.size());
155       Iterator iter = queries.entrySet().iterator();
156       while(iter.hasNext()) {
157         Entry entry = (Entry) iter.next();
158         Integer num = (Integer) entry.getValue();
159         newDeleteQueries.put(entry.getKey(),
160                              new Integer(mapper.remap(num.intValue())));
161       }
162     } else
163       newDeleteQueries = null;
164 
165     if (newDeleteTerms != null)
166       terms = newDeleteTerms;
167     if (newDeleteDocIDs != null)
168       docIDs = newDeleteDocIDs;
169     if (newDeleteQueries != null)
170       queries = newDeleteQueries;
171   }
172 }
