20  /** Consumes doc & freq, writing them using the current
21   *  index file format */
22  
23  import java.io.IOException;
24  
25  import org.apache.lucene.util.UnicodeUtil;
26  import org.apache.lucene.store.IndexOutput;
27  
28  final class FormatPostingsDocsWriter extends FormatPostingsDocsConsumer {
29  
30    final IndexOutput out;
31    final FormatPostingsTermsWriter parent;
32    final FormatPostingsPositionsWriter posWriter;
33    final DefaultSkipListWriter skipListWriter;
34    final int skipInterval;
35    final int totalNumDocs;
36  
37    boolean omitTermFreqAndPositions;
38    boolean storePayloads;
39    long freqStart;
40    FieldInfo fieldInfo;
41  
42    FormatPostingsDocsWriter(SegmentWriteState state, FormatPostingsTermsWriter parent) throws IOException {
43      super();
44      this.parent = parent;
45      final String fileName = IndexFileNames.segmentFileName(parent.parent.segment, IndexFileNames.FREQ_EXTENSION);
46      state.flushedFiles.add(fileName);
47      out = parent.parent.dir.createOutput(fileName);
48      totalNumDocs = parent.parent.totalNumDocs;
49  
50      // TODO: abstraction violation
51      skipInterval = parent.parent.termsOut.skipInterval;
52      skipListWriter = parent.parent.skipListWriter;
53      skipListWriter.setFreqOutput(out);
54  
55      posWriter = new FormatPostingsPositionsWriter(state, this);
56    }
57  
58    void setField(FieldInfo fieldInfo) {
59      this.fieldInfo = fieldInfo;
60      omitTermFreqAndPositions = fieldInfo.omitTermFreqAndPositions;
61      storePayloads = fieldInfo.storePayloads;
62      posWriter.setField(fieldInfo);
63    }
64  
65    int lastDocID;
66    int df;
67  
68    /** Adds a new doc in this term.  If this returns null
69     *  then we just skip consuming positions/payloads. */
70    FormatPostingsPositionsConsumer addDoc(int docID, int termDocFreq) throws IOException {
71  
72      final int delta = docID - lastDocID;
73  
74      if (docID < 0 || (df > 0 && delta <= 0))
75        throw new CorruptIndexException("docs out of order (" + docID + " <= " + lastDocID + " )");
76  
77      if ((++df % skipInterval) == 0) {
78        // TODO: abstraction violation
79        skipListWriter.setSkipData(lastDocID, storePayloads, posWriter.lastPayloadLength);
80        skipListWriter.bufferSkip(df);
81      }
82  
83      assert docID < totalNumDocs: "docID=" + docID + " totalNumDocs=" + totalNumDocs;
84  
85      lastDocID = docID;
86      if (omitTermFreqAndPositions)
87        out.writeVInt(delta);
88      else if (1 == termDocFreq)
89        out.writeVInt((delta<<1) | 1);
90      else {
91        out.writeVInt(delta<<1);
92        out.writeVInt(termDocFreq);
93      }
94  
95      return posWriter;
96    }
97  
98    private final TermInfo termInfo = new TermInfo();  // minimize consing
99    final UnicodeUtil.UTF8Result utf8 = new UnicodeUtil.UTF8Result();
100 
101   /** Called when we are done adding docs to this term */
102   void finish() throws IOException {
103     long skipPointer = skipListWriter.writeSkip(out);
104 
105     // TODO: this is abstraction violation -- we should not
106     // peek up into parents terms encoding format
107     termInfo.set(df, parent.freqStart, parent.proxStart, (int) (skipPointer - parent.freqStart));
108 
109     // TODO: we could do this incrementally
110     UnicodeUtil.UTF16toUTF8(parent.currentTerm, parent.currentTermStart, utf8);
111 
112     if (df > 0) {
113       parent.termsOut.add(fieldInfo.number,
114                           utf8.result,
115                           utf8.length,
116                           termInfo);
117     }
118 
119     lastDocID = 0;
120     df = 0;
121   }
122 
123   void close() throws IOException {
124     out.close();
125     posWriter.close();
126   }
127 }
