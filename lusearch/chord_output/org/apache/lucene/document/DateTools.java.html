20  import java.text.ParseException;
21  import java.text.SimpleDateFormat;
22  import java.util.Calendar;
23  import java.util.Date;
24  import java.util.TimeZone;
25  import java.util.Locale;
26  import org.apache.lucene.search.NumericRangeQuery; // for javadocs
27  import org.apache.lucene.util.NumericUtils; // for javadocs

29  /**
30   * Provides support for converting dates to strings and vice-versa.
31   * The strings are structured so that lexicographic sorting orders 
32   * them by date, which makes them suitable for use as field values 
33   * and search terms.
34   * 
35   * <P>This class also helps you to limit the resolution of your dates. Do not
36   * save dates with a finer resolution than you really need, as then
37   * RangeQuery and PrefixQuery will require more memory and become slower.
38   * 
39   * <P>Compared to {@link DateField} the strings generated by the methods
40   * in this class take slightly more space, unless your selected resolution
41   * is set to <code>Resolution.DAY</code> or lower.
42   *
43   * <P>
44   * Another approach is {@link NumericUtils}, which provides
45   * a sortable binary representation (prefix encoded) of numeric values, which
46   * date/time are.
47   * For indexing a {@link Date} or {@link Calendar}, just get the unix timestamp as
48   * <code>long</code> using {@link Date#getTime} or {@link Calendar#getTimeInMillis} and
49   * index this as a numeric value with {@link NumericField}
50   * and use {@link NumericRangeQuery} to query it.
51   */
52  public class DateTools {
53    
54    private final static TimeZone GMT = TimeZone.getTimeZone("GMT");
55  
56    private static final SimpleDateFormat YEAR_FORMAT = new SimpleDateFormat("yyyy", Locale.US);
57    private static final SimpleDateFormat MONTH_FORMAT = new SimpleDateFormat("yyyyMM", Locale.US);
58    private static final SimpleDateFormat DAY_FORMAT = new SimpleDateFormat("yyyyMMdd", Locale.US);
59    private static final SimpleDateFormat HOUR_FORMAT = new SimpleDateFormat("yyyyMMddHH", Locale.US);
60    private static final SimpleDateFormat MINUTE_FORMAT = new SimpleDateFormat("yyyyMMddHHmm", Locale.US);
61    private static final SimpleDateFormat SECOND_FORMAT = new SimpleDateFormat("yyyyMMddHHmmss", Locale.US);
62    private static final SimpleDateFormat MILLISECOND_FORMAT = new SimpleDateFormat("yyyyMMddHHmmssSSS", Locale.US);
63    static {
64      // times need to be normalized so the value doesn't depend on the 
65      // location the index is created/used:
66      YEAR_FORMAT.setTimeZone(GMT);
67      MONTH_FORMAT.setTimeZone(GMT);
68      DAY_FORMAT.setTimeZone(GMT);
69      HOUR_FORMAT.setTimeZone(GMT);
70      MINUTE_FORMAT.setTimeZone(GMT);
71      SECOND_FORMAT.setTimeZone(GMT);
72      MILLISECOND_FORMAT.setTimeZone(GMT);
73    }
74  
75    private static final Calendar calInstance = Calendar.getInstance(GMT);
76    
77    // cannot create, the class has static methods only
78    private DateTools() {}
79  
80    /**
81     * Converts a Date to a string suitable for indexing.
82     * 
83     * @param date the date to be converted
84     * @param resolution the desired resolution, see
85     *  {@link #round(Date, DateTools.Resolution)}
86     * @return a string in format <code>yyyyMMddHHmmssSSS</code> or shorter,
87     *  depending on <code>resolution</code>; using GMT as timezone 
88     */
89    public static synchronized String dateToString(Date date, Resolution resolution) {
90      return timeToString(date.getTime(), resolution);
91    }
92  
93    /**
94     * Converts a millisecond time to a string suitable for indexing.
95     * 
96     * @param time the date expressed as milliseconds since January 1, 1970, 00:00:00 GMT
97     * @param resolution the desired resolution, see
98     *  {@link #round(long, DateTools.Resolution)}
99     * @return a string in format <code>yyyyMMddHHmmssSSS</code> or shorter,
100    *  depending on <code>resolution</code>; using GMT as timezone
101    */
102   public static synchronized String timeToString(long time, Resolution resolution) {
103     calInstance.setTimeInMillis(round(time, resolution));
104     Date date = calInstance.getTime();
105     
106     if (resolution == Resolution.YEAR) {
107       return YEAR_FORMAT.format(date);
108     } else if (resolution == Resolution.MONTH) {
109       return MONTH_FORMAT.format(date);
110     } else if (resolution == Resolution.DAY) {
111       return DAY_FORMAT.format(date);
112     } else if (resolution == Resolution.HOUR) {
113       return HOUR_FORMAT.format(date);
114     } else if (resolution == Resolution.MINUTE) {
115       return MINUTE_FORMAT.format(date);
116     } else if (resolution == Resolution.SECOND) {
117       return SECOND_FORMAT.format(date);
118     } else if (resolution == Resolution.MILLISECOND) {
119       return MILLISECOND_FORMAT.format(date);
120     }
121     
122     throw new IllegalArgumentException("unknown resolution " + resolution);
123   }
124   
125   /**
126    * Converts a string produced by <code>timeToString</code> or
127    * <code>dateToString</code> back to a time, represented as the
128    * number of milliseconds since January 1, 1970, 00:00:00 GMT.
129    * 
130    * @param dateString the date string to be converted
131    * @return the number of milliseconds since January 1, 1970, 00:00:00 GMT
132    * @throws ParseException if <code>dateString</code> is not in the 
133    *  expected format 
134    */
135   public static synchronized long stringToTime(String dateString) throws ParseException {
136     return stringToDate(dateString).getTime();
137   }
138 
139   /**
140    * Converts a string produced by <code>timeToString</code> or
141    * <code>dateToString</code> back to a time, represented as a
142    * Date object.
143    * 
144    * @param dateString the date string to be converted
145    * @return the parsed time as a Date object 
146    * @throws ParseException if <code>dateString</code> is not in the 
147    *  expected format 
148    */
149   public static synchronized Date stringToDate(String dateString) throws ParseException {
150     if (dateString.length() == 4) {
151       return YEAR_FORMAT.parse(dateString);
152     } else if (dateString.length() == 6) {
153       return MONTH_FORMAT.parse(dateString);
154     } else if (dateString.length() == 8) {
155       return DAY_FORMAT.parse(dateString);
156     } else if (dateString.length() == 10) {
157       return HOUR_FORMAT.parse(dateString);
158     } else if (dateString.length() == 12) {
159       return MINUTE_FORMAT.parse(dateString);
160     } else if (dateString.length() == 14) {
161       return SECOND_FORMAT.parse(dateString);
162     } else if (dateString.length() == 17) {
163       return MILLISECOND_FORMAT.parse(dateString);
164     }
165     throw new ParseException("Input is not valid date string: " + dateString, 0);
166   }
167   
168   /**
169    * Limit a date's resolution. For example, the date <code>2004-09-21 13:50:11</code>
170    * will be changed to <code>2004-09-01 00:00:00</code> when using
171    * <code>Resolution.MONTH</code>. 
172    * 
173    * @param resolution The desired resolution of the date to be returned
174    * @return the date with all values more precise than <code>resolution</code>
175    *  set to 0 or 1
176    */
177   public static synchronized Date round(Date date, Resolution resolution) {
178     return new Date(round(date.getTime(), resolution));
179   }
180   
181   /**
182    * Limit a date's resolution. For example, the date <code>1095767411000</code>
183    * (which represents 2004-09-21 13:50:11) will be changed to 
184    * <code>1093989600000</code> (2004-09-01 00:00:00) when using
185    * <code>Resolution.MONTH</code>.
186    * 
187    * @param resolution The desired resolution of the date to be returned
188    * @return the date with all values more precise than <code>resolution</code>
189    *  set to 0 or 1, expressed as milliseconds since January 1, 1970, 00:00:00 GMT
190    */
191   public static synchronized long round(long time, Resolution resolution) {
192     calInstance.setTimeInMillis(time);
193     
194     if (resolution == Resolution.YEAR) {
195       calInstance.set(Calendar.MONTH, 0);
196       calInstance.set(Calendar.DAY_OF_MONTH, 1);
197       calInstance.set(Calendar.HOUR_OF_DAY, 0);
198       calInstance.set(Calendar.MINUTE, 0);
199       calInstance.set(Calendar.SECOND, 0);
200       calInstance.set(Calendar.MILLISECOND, 0);
201     } else if (resolution == Resolution.MONTH) {
202       calInstance.set(Calendar.DAY_OF_MONTH, 1);
203       calInstance.set(Calendar.HOUR_OF_DAY, 0);
204       calInstance.set(Calendar.MINUTE, 0);
205       calInstance.set(Calendar.SECOND, 0);
206       calInstance.set(Calendar.MILLISECOND, 0);
207     } else if (resolution == Resolution.DAY) {
208       calInstance.set(Calendar.HOUR_OF_DAY, 0);
209       calInstance.set(Calendar.MINUTE, 0);
210       calInstance.set(Calendar.SECOND, 0);
211       calInstance.set(Calendar.MILLISECOND, 0);
212     } else if (resolution == Resolution.HOUR) {
213       calInstance.set(Calendar.MINUTE, 0);
214       calInstance.set(Calendar.SECOND, 0);
215       calInstance.set(Calendar.MILLISECOND, 0);
216     } else if (resolution == Resolution.MINUTE) {
217       calInstance.set(Calendar.SECOND, 0);
218       calInstance.set(Calendar.MILLISECOND, 0);
219     } else if (resolution == Resolution.SECOND) {
220       calInstance.set(Calendar.MILLISECOND, 0);
221     } else if (resolution == Resolution.MILLISECOND) {
222       // don't cut off anything
223     } else {
224       throw new IllegalArgumentException("unknown resolution " + resolution);
225     }
226     return calInstance.getTimeInMillis();
227   }
228 
229   /** Specifies the time granularity. */
230   public static class Resolution {
231     
232     public static final Resolution YEAR = new Resolution("year");
233     public static final Resolution MONTH = new Resolution("month");
234     public static final Resolution DAY = new Resolution("day");
235     public static final Resolution HOUR = new Resolution("hour");
236     public static final Resolution MINUTE = new Resolution("minute");
237     public static final Resolution SECOND = new Resolution("second");
238     public static final Resolution MILLISECOND = new Resolution("millisecond");
239 
240     private String resolution;
241 
242     private Resolution() {
243     }
244     
245     private Resolution(String resolution) {
246       this.resolution = resolution;
247     }
248     
249     public String toString() {
250       return resolution;
251     }
252 
253   }
