20  import org.apache.lucene.search.PrefixQuery;
21  import org.apache.lucene.search.TermRangeQuery;
22  import org.apache.lucene.search.NumericRangeQuery; // for javadocs
23  import org.apache.lucene.util.NumericUtils; // for javadocs
24  
25  import java.util.Date;   // for javadoc
26  import java.util.Calendar;   // for javadoc
27  
28  // do not remove in 3.0, needed for reading old indexes!
29  
30  /**
31   * Provides support for converting dates to strings and vice-versa.
32   * The strings are structured so that lexicographic sorting orders by date,
33   * which makes them suitable for use as field values and search terms.
34   *
35   * <P>Note that this class saves dates with millisecond granularity,
36   * which is bad for {@link TermRangeQuery} and {@link PrefixQuery}, as those
37   * queries are expanded to a BooleanQuery with a potentially large number
38   * of terms when searching. Thus you might want to use
39   * {@link DateTools} instead.
40   *
41   * <P>
42   * Note: dates before 1970 cannot be used, and therefore cannot be
43   * indexed when using this class. See {@link DateTools} for an
44   * alternative without such a limitation.
45   *
46   * <P>
47   * Another approach is {@link NumericUtils}, which provides
48   * a sortable binary representation (prefix encoded) of numeric values, which
49   * date/time are.
50   * For indexing a {@link Date} or {@link Calendar}, just get the unix timestamp as
51   * <code>long</code> using {@link Date#getTime} or {@link Calendar#getTimeInMillis} and
52   * index this as a numeric value with {@link NumericField}
53   * and use {@link NumericRangeQuery} to query it.
54   *
55   * @deprecated If you build a new index, use {@link DateTools} or 
56   * {@link NumericField} instead.
57   * This class is included for use with existing
58   * indices and will be removed in a future release.
59   */
60  public class DateField {
61    
62    private DateField() {}
63  
64    // make date strings long enough to last a millenium
65    private static int DATE_LEN = Long.toString(1000L*365*24*60*60*1000,
66                             Character.MAX_RADIX).length();
67  
68    public static String MIN_DATE_STRING() {
69      return timeToString(0);
70    }
71  
72    public static String MAX_DATE_STRING() {
73      char[] buffer = new char[DATE_LEN];
74      char c = Character.forDigit(Character.MAX_RADIX-1, Character.MAX_RADIX);
75      for (int i = 0 ; i < DATE_LEN; i++)
76        buffer[i] = c;
77      return new String(buffer);
78    }
79  
80    /**
81     * Converts a Date to a string suitable for indexing.
82     * @throws RuntimeException if the date specified in the
83     * method argument is before 1970
84     */
85    public static String dateToString(Date date) {
86      return timeToString(date.getTime());
87    }
88    /**
89     * Converts a millisecond time to a string suitable for indexing.
90     * @throws RuntimeException if the time specified in the
91     * method argument is negative, that is, before 1970
92     */
93    public static String timeToString(long time) {
94      if (time < 0)
95        throw new RuntimeException("time '" + time + "' is too early, must be >= 0");
96  
97      String s = Long.toString(time, Character.MAX_RADIX);
98  
99      if (s.length() > DATE_LEN)
100       throw new RuntimeException("time '" + time + "' is too late, length of string " +
101           "representation must be <= " + DATE_LEN);
102 
103     // Pad with leading zeros
104     if (s.length() < DATE_LEN) {
105       StringBuffer sb = new StringBuffer(s);
106       while (sb.length() < DATE_LEN)
107         sb.insert(0, 0);
108       s = sb.toString();
109     }
110 
111     return s;
112   }
113 
114   /** Converts a string-encoded date into a millisecond time. */
115   public static long stringToTime(String s) {
116     return Long.parseLong(s, Character.MAX_RADIX);
117   }
118   /** Converts a string-encoded date into a Date object. */
119   public static Date stringToDate(String s) {
120     return new Date(stringToTime(s));
121   }
122 }
