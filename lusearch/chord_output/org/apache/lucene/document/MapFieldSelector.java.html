20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  
24  /**
25   * A {@link FieldSelector} based on a Map of field names to {@link FieldSelectorResult}s
26   *
27   */
28  public class MapFieldSelector implements FieldSelector {
29      
30      Map fieldSelections;
31      
32      /** Create a a MapFieldSelector
33       * @param fieldSelections maps from field names (String) to {@link FieldSelectorResult}s
34       */
35      public MapFieldSelector(Map fieldSelections) {
36          this.fieldSelections = fieldSelections;
37      }
38      
39      /** Create a a MapFieldSelector
40       * @param fields fields to LOAD.  List of Strings.  All other fields are NO_LOAD.
41       */
42      public MapFieldSelector(List fields) {
43          fieldSelections = new HashMap(fields.size()*5/3);
44          for (int i=0; i<fields.size(); i++)
45              fieldSelections.put(fields.get(i), FieldSelectorResult.LOAD);
46      }
47      
48      /** Create a a MapFieldSelector
49       * @param fields fields to LOAD.  All other fields are NO_LOAD.
50       */
51      public MapFieldSelector(String[] fields) {
52          fieldSelections = new HashMap(fields.length*5/3);
53          for (int i=0; i<fields.length; i++)
54              fieldSelections.put(fields[i], FieldSelectorResult.LOAD);
55      }
56      
57      /** Load field according to its associated value in fieldSelections
58       * @param field a field name
59       * @return the fieldSelections value that field maps to or NO_LOAD if none.
60       */
61      public FieldSelectorResult accept(String field) {
62          FieldSelectorResult selection = (FieldSelectorResult) fieldSelections.get(field);
63          return selection!=null ? selection : FieldSelectorResult.NO_LOAD;
64      }
65      
66  }
