20  /**

23   *
24   **/
25  public interface FieldSelector extends Serializable {
26  
27    /**
28     * 
29     * @param fieldName the field to accept or reject
30     * @return an instance of {@link FieldSelectorResult}

32     */
33    FieldSelectorResult accept(String fieldName);
34  }
