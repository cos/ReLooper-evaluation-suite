19  import org.apache.lucene.analysis.TokenStream;
20  import org.apache.lucene.index.FieldInvertState; // for javadocs
21  
22  import java.io.Reader;
23  import java.io.Serializable;
24  
25  /**
26   * Synonymous with {@link Field}.
27   *
28   * <p><bold>WARNING</bold>: This interface may change within minor versions, despite Lucene's backward compatibility requirements.
29   * This means new methods may be added from version to version.  This change only affects the Fieldable API; other backwards
30   * compatibility promises remain intact. For example, Lucene can still
31   * read and write indices created within the same major version.
32   * </p>
33   *
34   **/
35  public interface Fieldable extends Serializable {
36    /** Sets the boost factor hits on this field.  This value will be
37     * multiplied into the score of all hits on this this field of this
38     * document.
39     *
40     * <p>The boost is multiplied by {@link org.apache.lucene.document.Document#getBoost()} of the document
41     * containing this field.  If a document has multiple fields with the same
42     * name, all such values are multiplied together.  This product is then
43     * used to compute the norm factor for the field.  By
44     * default, in the {@link
45     * org.apache.lucene.search.Similarity#computeNorm(String,
46     * FieldInvertState)} method, the boost value is multiplied
47     * by the {@link
48     * org.apache.lucene.search.Similarity#lengthNorm(String,
49     * int)} and then rounded by {@link org.apache.lucene.search.Similarity#encodeNorm(float)} before it is stored in the
50     * index.  One should attempt to ensure that this product does not overflow
51     * the range of that encoding.
52     *
53     * @see org.apache.lucene.document.Document#setBoost(float)
54     * @see org.apache.lucene.search.Similarity#computeNorm(String, FieldInvertState)
55     * @see org.apache.lucene.search.Similarity#encodeNorm(float)
56     */
57    void setBoost(float boost);
58  
59    /** Returns the boost factor for hits for this field.
60     *
61     * <p>The default value is 1.0.
62     *
63     * <p>Note: this value is not stored directly with the document in the index.
64     * Documents returned from {@link org.apache.lucene.index.IndexReader#document(int)} and
65     * {@link org.apache.lucene.search.Hits#doc(int)} may thus not have the same value present as when
66     * this field was indexed.
67     *
68     * @see #setBoost(float)
69     */
70    float getBoost();
71  
72    /** Returns the name of the field as an interned string.
73     * For example "date", "title", "body", ...
74     */
75    String name();
76  
77    /** The value of the field as a String, or null.
78     * <p>
79     * For indexing, if isStored()==true, the stringValue() will be used as the stored field value
80     * unless isBinary()==true, in which case binaryValue() will be used.
81     *
82     * If isIndexed()==true and isTokenized()==false, this String value will be indexed as a single token.
83     * If isIndexed()==true and isTokenized()==true, then tokenStreamValue() will be used to generate indexed tokens if not null,
84     * else readerValue() will be used to generate indexed tokens if not null, else stringValue() will be used to generate tokens.
85     */
86    public String stringValue();
87    
88    /** The value of the field as a Reader, which can be used at index time to generate indexed tokens.
89     * @see #stringValue()
90     */
91    public Reader readerValue();
92    
93    /** The value of the field in Binary, or null.
94     * @see #stringValue()
95     */
96    public byte[] binaryValue();
97    
98    /** The TokenStream for this field to be used when indexing, or null.
99     * @see #stringValue()
100    */
101   public TokenStream tokenStreamValue();
102 
103   /** True if the value of the field is to be stored in the index for return
104     with search hits. */
105   boolean  isStored();
106 
107   /** True if the value of the field is to be indexed, so that it may be
108     searched on. */
109   boolean  isIndexed();
110 
111   /** True if the value of the field should be tokenized as text prior to
112     indexing.  Un-tokenized fields are indexed as a single word and may not be
113     Reader-valued. */
114   boolean  isTokenized();
115 
116   /** True if the value of the field is stored and compressed within the index */
117   boolean  isCompressed();
118 
119   /** True if the term or terms used to index this field are stored as a term
120    *  vector, available from {@link org.apache.lucene.index.IndexReader#getTermFreqVector(int,String)}.
121    *  These methods do not provide access to the original content of the field,
122    *  only to terms used to index it. If the original content must be
123    *  preserved, use the <code>stored</code> attribute instead.
124    *
125    * @see org.apache.lucene.index.IndexReader#getTermFreqVector(int, String)
126    */
127   boolean isTermVectorStored();
128 
129   /**
130    * True if terms are stored as term vector together with their offsets 
131    * (start and end positon in source text).
132    */
133   boolean isStoreOffsetWithTermVector();
134 
135   /**
136    * True if terms are stored as term vector together with their token positions.
137    */
138   boolean isStorePositionWithTermVector();
139 
140   /** True if the value of the field is stored as binary */
141   boolean  isBinary();
142 
143   /** True if norms are omitted for this indexed field */
144   boolean getOmitNorms();
145 
146   /** Expert:
147    *
148    * If set, omit normalization factors associated with this indexed field.
149    * This effectively disables indexing boosts and length normalization for this field.
150    */
151   void setOmitNorms(boolean omitNorms);
152 
153   /** @deprecated Renamed to {@link AbstractField#setOmitTermFreqAndPositions} */
154   void setOmitTf(boolean omitTf);
155 
156   /** @deprecated Renamed to {@link AbstractField#getOmitTermFreqAndPositions} */
157   boolean getOmitTf();
158 
159   /**
160    * Indicates whether a Field is Lazy or not.  The semantics of Lazy loading are such that if a Field is lazily loaded, retrieving
161    * it's values via {@link #stringValue()} or {@link #binaryValue()} is only valid as long as the {@link org.apache.lucene.index.IndexReader} that
162    * retrieved the {@link Document} is still open.
163    *  
164    * @return true if this field can be loaded lazily
165    */
166   boolean isLazy();
167   
168   /**
169    * Returns offset into byte[] segment that is used as value, if Field is not binary
170    * returned value is undefined
171    * @return index of the first character in byte[] segment that represents this Field value
172    */
173   abstract int getBinaryOffset();
174   
175   /**
176    * Returns length of byte[] segment that is used as value, if Field is not binary
177    * returned value is undefined
178    * @return length of byte[] segment that represents this Field value
179    */
180   abstract int getBinaryLength();
181 
182   /**
183    * Return the raw byte[] for the binary field.  Note that
184    * you must also call {@link #getBinaryLength} and {@link
185    * #getBinaryOffset} to know which range of bytes in this
186    * returned array belong to the field.
187    * @return reference to the Field value as byte[].
188    */
189   abstract byte[] getBinaryValue();
190 
191   /**
192    * Return the raw byte[] for the binary field.  Note that
193    * you must also call {@link #getBinaryLength} and {@link
194    * #getBinaryOffset} to know which range of bytes in this
195    * returned array belong to the field.<p>
196    * About reuse: if you pass in the result byte[] and it is
197    * used, likely the underlying implementation will hold
198    * onto this byte[] and return it in future calls to
199    * {@link #binaryValue()} or {@link #getBinaryValue()}.
200    * So if you subsequently re-use the same byte[] elsewhere
201    * it will alter this Fieldable's value.
202    * @param result  User defined buffer that will be used if
203    *  possible.  If this is null or not large enough, a new
204    *  buffer is allocated
205    * @return reference to the Field value as byte[].
206    */
207   abstract byte[] getBinaryValue(byte[] result);
208 }
