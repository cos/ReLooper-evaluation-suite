20  import java.util.zip.Deflater;
21  import java.util.zip.Inflater;
22  import java.util.zip.DataFormatException;
23  import java.io.ByteArrayOutputStream;
24  import org.apache.lucene.util.UnicodeUtil;
25  
26  /** Simple utility class providing static methods to
27   *  compress and decompress binary data for stored fields.
28   *  This class uses java.util.zip.Deflater and Inflater
29   *  classes to compress and decompress, which is the same
30   *  format previously used by the now deprecated
31   *  Field.Store.COMPRESS.
32   */
33  
34  public class CompressionTools {
35  
36    // Export only static methods
37    private CompressionTools() {}
38  
39    /** Compresses the specified byte range using the
40     *  specified compressionLevel (constants are defined in
41     *  java.util.zip.Deflater). */
42    public static byte[] compress(byte[] value, int offset, int length, int compressionLevel) {
43  
44      /* Create an expandable byte array to hold the compressed data.
45       * You cannot use an array that's the same size as the orginal because
46       * there is no guarantee that the compressed data will be smaller than
47       * the uncompressed data. */
48      ByteArrayOutputStream bos = new ByteArrayOutputStream(length);
49  
50      Deflater compressor = new Deflater();
51  
52      try {
53        compressor.setLevel(compressionLevel);
54        compressor.setInput(value, offset, length);
55        compressor.finish();
56  
57        // Compress the data
58        final byte[] buf = new byte[1024];
59        while (!compressor.finished()) {
60          int count = compressor.deflate(buf);
61          bos.write(buf, 0, count);
62        }
63      } finally {
64        compressor.end();
65      }
66  
67      return bos.toByteArray();
68    }
69  
70    /** Compresses the specified byte range, with default BEST_COMPRESSION level */
71    public static byte[] compress(byte[] value, int offset, int length) {
72      return compress(value, offset, length, Deflater.BEST_COMPRESSION);
73    }
74    
75    /** Compresses all bytes in the array, with default BEST_COMPRESSION level */
76    public static byte[] compress(byte[] value) {
77      return compress(value, 0, value.length, Deflater.BEST_COMPRESSION);
78    }
79  
80    /** Compresses the String value, with default BEST_COMPRESSION level */
81    public static byte[] compressString(String value) {
82      return compressString(value, Deflater.BEST_COMPRESSION);
83    }
84  
85    /** Compresses the String value using the specified
86     *  compressionLevel (constants are defined in
87     *  java.util.zip.Deflater). */
88    public static byte[] compressString(String value, int compressionLevel) {
89      UnicodeUtil.UTF8Result result = new UnicodeUtil.UTF8Result();
90      UnicodeUtil.UTF16toUTF8(value, 0, value.length(), result);
91      return compress(result.result, 0, result.length, compressionLevel);
92    }
93  
94    /** Decompress the byte array previously returned by
95     *  compress */
96    public static byte[] decompress(byte[] value) throws DataFormatException {
97      // Create an expandable byte array to hold the decompressed data
98      ByteArrayOutputStream bos = new ByteArrayOutputStream(value.length);
99  
100     Inflater decompressor = new Inflater();
101 
102     try {
103       decompressor.setInput(value);
104 
105       // Decompress the data
106       final byte[] buf = new byte[1024];
107       while (!decompressor.finished()) {
108         int count = decompressor.inflate(buf);
109         bos.write(buf, 0, count);
110       }
111     } finally {  
112       decompressor.end();
113     }
114     
115     return bos.toByteArray();
116   }
117 
118   /** Decompress the byte array previously returned by
119    *  compressString back into a String */
120   public static String decompressString(byte[] value) throws DataFormatException {
121     UnicodeUtil.UTF16Result result = new UnicodeUtil.UTF16Result();
122     final byte[] bytes = decompress(value);
123     UnicodeUtil.UTF8toUTF16(bytes, 0, bytes.length, result);
124     return new String(result.result, 0, result.length);
125   }
126 }
