20  import org.apache.lucene.document.NumericField; // for javadocs
21  import org.apache.lucene.search.NumericRangeQuery; // for javadocs
22  import org.apache.lucene.util.NumericUtils; // for javadocs
23  
24  // do not remove this class in 3.0, it may be needed to decode old indexes!
25  
26  /**
27   * Provides support for converting longs to Strings, and back again. The strings
28   * are structured so that lexicographic sorting order is preserved.
29   * 
30   * <p>
31   * That is, if l1 is less than l2 for any two longs l1 and l2, then
32   * NumberTools.longToString(l1) is lexicographically less than
33   * NumberTools.longToString(l2). (Similarly for "greater than" and "equals".)
34   * 
35   * <p>
36   * This class handles <b>all</b> long values (unlike
37   * {@link org.apache.lucene.document.DateField}).
38   * 
39   * @deprecated For new indexes use {@link NumericUtils} instead, which
40   * provides a sortable binary representation (prefix encoded) of numeric
41   * values.
42   * To index and efficiently query numeric values use {@link NumericField}
43   * and {@link NumericRangeQuery}.
44   * This class is included for use with existing
45   * indices and will be removed in a future release.
46   */
47  public class NumberTools {
48  
49      private static final int RADIX = 36;
50  
51      private static final char NEGATIVE_PREFIX = '-';
52  
53      // NB: NEGATIVE_PREFIX must be < POSITIVE_PREFIX
54      private static final char POSITIVE_PREFIX = '0';
55  
56      //NB: this must be less than
57      /**
58       * Equivalent to longToString(Long.MIN_VALUE)
59       */
60      public static final String MIN_STRING_VALUE = NEGATIVE_PREFIX
61              + "0000000000000";
62  
63      /**
64       * Equivalent to longToString(Long.MAX_VALUE)
65       */
66      public static final String MAX_STRING_VALUE = POSITIVE_PREFIX
67              + "1y2p0ij32e8e7";
68  
69      /**
70       * The length of (all) strings returned by {@link #longToString}
71       */
72      public static final int STR_SIZE = MIN_STRING_VALUE.length();
73  
74      /**
75       * Converts a long to a String suitable for indexing.
76       */
77      public static String longToString(long l) {
78  
79          if (l == Long.MIN_VALUE) {
80              // special case, because long is not symmetric around zero
81              return MIN_STRING_VALUE;
82          }
83  
84          StringBuffer buf = new StringBuffer(STR_SIZE);
85  
86          if (l < 0) {
87              buf.append(NEGATIVE_PREFIX);
88              l = Long.MAX_VALUE + l + 1;
89          } else {
90              buf.append(POSITIVE_PREFIX);
91          }
92          String num = Long.toString(l, RADIX);
93  
94          int padLen = STR_SIZE - num.length() - buf.length();
95          while (padLen-- > 0) {
96              buf.append('0');
97          }
98          buf.append(num);
99  
100         return buf.toString();
101     }
102 
103     /**
104      * Converts a String that was returned by {@link #longToString} back to a
105      * long.
106      * 
107      * @throws IllegalArgumentException
108      *             if the input is null
109      * @throws NumberFormatException
110      *             if the input does not parse (it was not a String returned by
111      *             longToString()).
112      */
113     public static long stringToLong(String str) {
114         if (str == null) {
115             throw new NullPointerException("string cannot be null");
116         }
117         if (str.length() != STR_SIZE) {
118             throw new NumberFormatException("string is the wrong size");
119         }
120 
121         if (str.equals(MIN_STRING_VALUE)) {
122             return Long.MIN_VALUE;
123         }
124 
125         char prefix = str.charAt(0);
126         long l = Long.parseLong(str.substring(1), RADIX);
127 
128         if (prefix == POSITIVE_PREFIX) {
129             // nop
130         } else if (prefix == NEGATIVE_PREFIX) {
131             l = l - Long.MAX_VALUE - 1;
132         } else {
133             throw new NumberFormatException(
134                     "string does not begin with the correct prefix");
135         }
136 
137         return l;
138     }
139 }
