28  /**
29    A field is a section of a Document.  Each field has two parts, a name and a
30    value.  Values may be free text, provided as a String or as a Reader, or they
31    may be atomic keywords, which are not further processed.  Such keywords may
32    be used to represent dates, urls, etc.  Fields are optionally stored in the
33    index, so that they may be returned with hits on the document.
34    */
35  
36  public final class Field extends AbstractField implements Fieldable, Serializable {
37    
38    /** Specifies whether and how a field should be stored. */
39    public static final class Store extends Parameter implements Serializable {
40  
41      private Store(String name) {
42        super(name);
43      }
44  
45      /** Store the original field value in the index in a compressed form. This is
46       * useful for long documents and for binary valued fields.
47       * @deprecated Please use {@link CompressionTools} instead.
48       * For string fields that were previously indexed and stored using compression,
49       * the new way to achieve this is: First add the field indexed-only (no store)
50       * and additionally using the same field name as a binary, stored field
51       * with {@link CompressionTools#compressString}.
52       */
53      public static final Store COMPRESS = new Store("COMPRESS");
54  
55      /** Store the original field value in the index. This is useful for short texts
56       * like a document's title which should be displayed with the results. The
57       * value is stored in its original form, i.e. no analyzer is used before it is
58       * stored.
59       */
60      public static final Store YES = new Store("YES");
61  
62      /** Do not store the field value in the index. */
63      public static final Store NO = new Store("NO");
64    }

66    /** Specifies whether and how a field should be indexed. */
67    public static final class Index extends Parameter implements Serializable {
68  
69      private Index(String name) {
70        super(name);
71      }
72  
73      /** Do not index the field value. This field can thus not be searched,
74       * but one can still access its contents provided it is
75       * {@link Field.Store stored}. */
76      public static final Index NO = new Index("NO");
77  
78      /** Index the tokens produced by running the field's
79       * value through an Analyzer.  This is useful for
80       * common text. */
81      public static final Index ANALYZED = new Index("ANALYZED");
82  
83      /** @deprecated this has been renamed to {@link #ANALYZED} */
84      public static final Index TOKENIZED = ANALYZED;
85  
86      /** Index the field's value without using an Analyzer, so it can be searched.
87       * As no analyzer is used the value will be stored as a single term. This is
88       * useful for unique Ids like product numbers.
89       */
90      public static final Index NOT_ANALYZED = new Index("NOT_ANALYZED");
91  
92      /** @deprecated This has been renamed to {@link #NOT_ANALYZED} */
93      public static final Index UN_TOKENIZED = NOT_ANALYZED;
94  
95      /** Expert: Index the field's value without an Analyzer,
96       * and also disable the storing of norms.  Note that you
97       * can also separately enable/disable norms by calling
98       * {@link Field#setOmitNorms}.  No norms means that
99       * index-time field and document boosting and field
100      * length normalization are disabled.  The benefit is
101      * less memory usage as norms take up one byte of RAM
102      * per indexed field for every document in the index,
103      * during searching.  Note that once you index a given
104      * field <i>with</i> norms enabled, disabling norms will
105      * have no effect.  In other words, for this to have the
106      * above described effect on a field, all instances of
107      * that field must be indexed with NOT_ANALYZED_NO_NORMS
108      * from the beginning. */
109     public static final Index NOT_ANALYZED_NO_NORMS = new Index("NOT_ANALYZED_NO_NORMS");
110 
111     /** @deprecated This has been renamed to
112      *  {@link #NOT_ANALYZED_NO_NORMS} */
113     public static final Index NO_NORMS = NOT_ANALYZED_NO_NORMS;
114 
115     /** Expert: Index the tokens produced by running the
116      *  field's value through an Analyzer, and also
117      *  separately disable the storing of norms.  See
118      *  {@link #NOT_ANALYZED_NO_NORMS} for what norms are
119      *  and why you may want to disable them. */
120     public static final Index ANALYZED_NO_NORMS = new Index("ANALYZED_NO_NORMS");
121   }

139      * Store the term vector + token position information
140      * 
141      * @see #YES

143     public static final TermVector WITH_POSITIONS = new TermVector("WITH_POSITIONS");
144     
145     /**
146      * Store the term vector + Token offset information
147      * 
148      * @see #YES
149      */ 
150     public static final TermVector WITH_OFFSETS = new TermVector("WITH_OFFSETS");
151     
152     /**
153      * Store the term vector + Token position and offset information
154      * 
155      * @see #YES
156      * @see #WITH_POSITIONS
157      * @see #WITH_OFFSETS
158      */

160   }

163   /** The value of the field as a String, or null.  If null, the Reader value or
164    * binary value is used.  Exactly one of stringValue(),
165    * readerValue(), and getBinaryValue() must be set. */
166   public String stringValue()   { return fieldsData instanceof String ? (String)fieldsData : null; }
167   
168   /** The value of the field as a Reader, or null.  If null, the String value or
169    * binary value is used.  Exactly one of stringValue(),
170    * readerValue(), and getBinaryValue() must be set. */
171   public Reader readerValue()   { return fieldsData instanceof Reader ? (Reader)fieldsData : null; }
172   
173   /** The value of the field in Binary, or null.  If null, the Reader value,
174    * or String value is used. Exactly one of stringValue(),
175    * readerValue(), and getBinaryValue() must be set.
176    * @deprecated This method must allocate a new byte[] if
177    * the {@link AbstractField#getBinaryOffset()} is non-zero
178    * or {@link AbstractField#getBinaryLength()} is not the
179    * full length of the byte[]. Please use {@link
180    * AbstractField#getBinaryValue()} instead, which simply
181    * returns the byte[].
182    */ 
183   public byte[] binaryValue() {
184     if (!isBinary)
185       return null;
186     final byte[] data = (byte[]) fieldsData;
187     if (binaryOffset == 0 && data.length == binaryLength)
188       return data; //Optimization
189     
190     final byte[] ret = new byte[binaryLength];
191     System.arraycopy(data, binaryOffset, ret, 0, binaryLength);
192     return ret;    
193   }

195   /** The TokesStream for this field to be used when indexing, or null.  If null, the Reader value
196    * or String value is analyzed to produce the indexed tokens. */
197   public TokenStream tokenStreamValue()   { return tokenStream; }

200   /** <p>Expert: change the value of this field.  This can
201    *  be used during indexing to re-use a single Field
202    *  instance to improve indexing speed by avoiding GC cost
203    *  of new'ing and reclaiming Field instances.  Typically
204    *  a single {@link Document} instance is re-used as
205    *  well.  This helps most on small documents.</p>
206    * 
207    *  <p>Each Field instance should only be used once
208    *  within a single {@link Document} instance.  See <a
209    *  href="http://wiki.apache.org/lucene-java/ImproveIndexingSpeed">ImproveIndexingSpeed</a>
210    *  for details.</p> */
211   public void setValue(String value) {
212     if (isBinary) {
213       throw new IllegalArgumentException("cannot set a String value on a binary field");
214     }
215     fieldsData = value;
216   }
217 
218   /** Expert: change the value of this field.  See <a href="#setValue(java.lang.String)">setValue(String)</a>. */
219   public void setValue(Reader value) {
220     if (isBinary) {
221       throw new IllegalArgumentException("cannot set a Reader value on a binary field");
222     }
223     if (isStored) {
224       throw new IllegalArgumentException("cannot set a Reader value on a stored field");
225     }
226     fieldsData = value;
227   }

229   /** Expert: change the value of this field.  See <a href="#setValue(java.lang.String)">setValue(String)</a>. */
230   public void setValue(byte[] value) {
231     if (!isBinary) {
232       throw new IllegalArgumentException("cannot set a byte[] value on a non-binary field");
233     }
234     fieldsData = value;
235     binaryLength = value.length;
236     binaryOffset = 0;
237   }

239   /** Expert: change the value of this field.  See <a href="#setValue(java.lang.String)">setValue(String)</a>. */
240   public void setValue(byte[] value, int offset, int length) {
241     if (!isBinary) {
242       throw new IllegalArgumentException("cannot set a byte[] value on a non-binary field");
243     }
244     fieldsData = value;
245     binaryLength = length;
246     binaryOffset = offset;
247   }

308   /**
309    * Create a field by specifying its name, value and how it will
310    * be saved in the index.
311    * 
312    * @param name The name of the field
313    * @param internName Whether to .intern() name or not
314    * @param value The string to process
315    * @param store Whether <code>value</code> should be stored in the index
316    * @param index Whether the field should be indexed, and if so, if it should
317    *  be tokenized before indexing 
318    * @param termVector Whether term vector should be stored
319    * @throws NullPointerException if name or value is <code>null</code>
320    * @throws IllegalArgumentException in any of the following situations:
321    * <ul> 
322    *  <li>the field is neither stored nor indexed</li> 
323    *  <li>the field is not indexed but termVector is <code>TermVector.YES</code></li>
324    * </ul> 
325    */ 
326   public Field(String name, boolean internName, String value, Store store, Index index, TermVector termVector) {
327     if (name == null)
328       throw new NullPointerException("name cannot be null");
329     if (value == null)
330       throw new NullPointerException("value cannot be null");
331     if (name.length() == 0 && value.length() == 0)
332       throw new IllegalArgumentException("name and value cannot both be empty");
333     if (index == Index.NO && store == Store.NO)
334       throw new IllegalArgumentException("it doesn't make sense to have a field that "
335          + "is neither indexed nor stored");
336     if (index == Index.NO && termVector != TermVector.NO)
337       throw new IllegalArgumentException("cannot store term vector information "
338          + "for a field that is not indexed");
339           
340     if (internName) // field names are optionally interned
341       name = StringHelper.intern(name);
342     
343     this.name = name; 
344     
345     this.fieldsData = value;
346 
347     if (store == Store.YES){
348       this.isStored = true;
349       this.isCompressed = false;
350     }
351     else if (store == Store.COMPRESS) {
352       this.isStored = true;
353       this.isCompressed = true;
354     }
355     else if (store == Store.NO){
356       this.isStored = false;
357       this.isCompressed = false;
358     }
359     else
360       throw new IllegalArgumentException("unknown store parameter " + store);
361    
362     if (index == Index.NO) {
363       this.isIndexed = false;
364       this.isTokenized = false;
365       this.omitTermFreqAndPositions = false;
366       this.omitNorms = true;
367     } else if (index == Index.ANALYZED) {
368       this.isIndexed = true;
369       this.isTokenized = true;
370     } else if (index == Index.NOT_ANALYZED) {
371       this.isIndexed = true;
372       this.isTokenized = false;
373     } else if (index == Index.NOT_ANALYZED_NO_NORMS) {
374       this.isIndexed = true;
375       this.isTokenized = false;
376       this.omitNorms = true;
377     } else if (index == Index.ANALYZED_NO_NORMS) {
378       this.isIndexed = true;
379       this.isTokenized = true;
380       this.omitNorms = true;
381     } else {
382       throw new IllegalArgumentException("unknown index parameter " + index);
383     }
384     
385     this.isBinary = false;
386 
387     setStoreTermVector(termVector);
388   }

404   /**
405    * Create a tokenized and indexed field that is not stored, optionally with 
406    * storing term vectors.  The Reader is read only when the Document is added to the index,
407    * i.e. you may not close the Reader until {@link IndexWriter#addDocument(Document)}

416     if (name == null)
417       throw new NullPointerException("name cannot be null");
418     if (reader == null)
419       throw new NullPointerException("reader cannot be null");
420     
421     this.name = StringHelper.intern(name);        // field names are interned
422     this.fieldsData = reader;
423     
424     this.isStored = false;
425     this.isCompressed = false;
426     
427     this.isIndexed = true;
428     this.isTokenized = true;
429     
430     this.isBinary = false;
431     
432     setStoreTermVector(termVector);
433   }

496   /**
497    * Create a stored field with binary value. Optionally the value may be compressed.
498    * 
499    * @param name The name of the field
500    * @param value The binary value
501    * @param offset Starting offset in value where this Field's bytes are
502    * @param length Number of bytes to use for this Field, starting at offset
503    * @param store How <code>value</code> should be stored (compressed or not)
504    * @throws IllegalArgumentException if store is <code>Store.NO</code> 
505    */
506   public Field(String name, byte[] value, int offset, int length, Store store) {
507 
508     if (name == null)
509       throw new IllegalArgumentException("name cannot be null");
510     if (value == null)
511       throw new IllegalArgumentException("value cannot be null");
512     
513     this.name = StringHelper.intern(name);        // field names are interned
514     fieldsData = value;
515     
516     if (store == Store.YES) {
517       isStored = true;
518       isCompressed = false;
519     }
520     else if (store == Store.COMPRESS) {
521       isStored = true;
522       isCompressed = true;
523     }
524     else if (store == Store.NO)
525       throw new IllegalArgumentException("binary values can't be unstored");
526     else
527       throw new IllegalArgumentException("unknown store parameter " + store);
528     
529     isIndexed   = false;
530     isTokenized = false;
531     omitTermFreqAndPositions = false;
532     omitNorms = true;
533     
534     isBinary    = true;
535     binaryLength = length;
536     binaryOffset = offset;
537     
538     setStoreTermVector(TermVector.NO);
539   }
