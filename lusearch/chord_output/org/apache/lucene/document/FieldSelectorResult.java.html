20  /**
21   *  Provides information about what should be done with this Field 
22   *
23   **/
24  //Replace with an enumerated type in 1.5
25  public final class FieldSelectorResult implements Serializable {
26  
27      /**
28       * Load this {@link Field} every time the {@link Document} is loaded, reading in the data as it is encountered.
29       *  {@link Document#getField(String)} and {@link Document#getFieldable(String)} should not return null.
30       *<p/>
31       * {@link Document#add(Fieldable)} should be called by the Reader.
32        */
33    public transient static final FieldSelectorResult LOAD = new FieldSelectorResult(0);
34      /**
35       * Lazily load this {@link Field}.  This means the {@link Field} is valid, but it may not actually contain its data until
36       * invoked.  {@link Document#getField(String)} SHOULD NOT BE USED.  {@link Document#getFieldable(String)} is safe to use and should
37       * return a valid instance of a {@link Fieldable}.
38       *<p/>
39       * {@link Document#add(Fieldable)} should be called by the Reader.
40       */
41    public transient static final FieldSelectorResult LAZY_LOAD = new FieldSelectorResult(1);
42      /**
43       * Do not load the {@link Field}.  {@link Document#getField(String)} and {@link Document#getFieldable(String)} should return null.
44       * {@link Document#add(Fieldable)} is not called.
45       * <p/>
46       * {@link Document#add(Fieldable)} should not be called by the Reader.
47       */
48    public transient static final FieldSelectorResult NO_LOAD = new FieldSelectorResult(2);
49      /**
50       * Load this field as in the {@link #LOAD} case, but immediately return from {@link Field} loading for the {@link Document}.  Thus, the
51       * Document may not have its complete set of Fields.  {@link Document#getField(String)} and {@link Document#getFieldable(String)} should
52       * both be valid for this {@link Field}
53       * <p/>
54       * {@link Document#add(Fieldable)} should be called by the Reader.
55       */
56    public transient static final FieldSelectorResult LOAD_AND_BREAK = new FieldSelectorResult(3);
57      /**
58       * Behaves much like {@link #LOAD} but does not uncompress any compressed data.  This is used for internal purposes.
59       * {@link Document#getField(String)} and {@link Document#getFieldable(String)} should not return null.
60       * <p/>
61       * {@link Document#add(Fieldable)} should be called by
62       * the Reader.
63       * @deprecated This is an internal option only, and is
64       *    no longer needed now that {@link CompressionTools}
65       *    is used for field compression.
66       */
67    public transient static final FieldSelectorResult LOAD_FOR_MERGE = new FieldSelectorResult(4);
68  
69       /** Expert:  Load the size of this {@link Field} rather than its value.
70         * Size is measured as number of bytes required to store the field == bytes for a binary or any compressed value, and 2*chars for a String value.
71        * The size is stored as a binary value, represented as an int in a byte[], with the higher order byte first in [0]
72        */
73    public transient static final FieldSelectorResult SIZE = new FieldSelectorResult(5);
74  
75    /** Expert: Like {@link #SIZE} but immediately break from the field loading loop, i.e., stop loading further fields, after the size is loaded */         
76    public transient static final FieldSelectorResult SIZE_AND_BREAK = new FieldSelectorResult(6);
77  
78  
79  
80    private int id;
81  
82    private FieldSelectorResult(int id) {
83      this.id = id;
84    }
85  
86    public boolean equals(Object o) {
87      if (this == o) return true;
88      if (o == null || getClass() != o.getClass()) return false;
89  
90      final FieldSelectorResult that = (FieldSelectorResult) o;
91  
92      if (id != that.id) return false;
93  
94      return true;
95    }
96  
97    public int hashCode() {
98      return id;
99    }
100 }
