18  package org.apache.lucene.analysis;
19  
20  import java.io.IOException;
21  
22  
23  /**
24   * Works in conjunction with the SinkTokenizer to provide the ability to set aside tokens
25   * that have already been analyzed.  This is useful in situations where multiple fields share
26   * many common analysis steps and then go their separate ways.
27   * <p/>
28   * It is also useful for doing things like entity extraction or proper noun analysis as
29   * part of the analysis workflow and saving off those tokens for use in another field.
30   *
31   * <pre>
32  SinkTokenizer sink1 = new SinkTokenizer();
33  SinkTokenizer sink2 = new SinkTokenizer();
34  
35  TokenStream source1 = new TeeTokenFilter(new TeeTokenFilter(new WhitespaceTokenizer(reader1), sink1), sink2);
36  TokenStream source2 = new TeeTokenFilter(new TeeTokenFilter(new WhitespaceTokenizer(reader2), sink1), sink2);
37  
38  TokenStream final1 = new LowerCaseFilter(source1);
39  TokenStream final2 = source2;
40  TokenStream final3 = new EntityDetect(sink1);
41  TokenStream final4 = new URLDetect(sink2);
42  
43  d.add(new Field("f1", final1));
44  d.add(new Field("f2", final2));
45  d.add(new Field("f3", final3));
46  d.add(new Field("f4", final4));
47   * </pre>
48   * In this example, <code>sink1</code> and <code>sink2<code> will both get tokens from both
49   * <code>reader1</code> and <code>reader2</code> after whitespace tokenizer
50   * and now we can further wrap any of these in extra analysis, and more "sources" can be inserted if desired.
51   * It is important, that tees are consumed before sinks (in the above example, the field names must be
52   * less the sink's field names).
53   * Note, the EntityDetect and URLDetect TokenStreams are for the example and do not currently exist in Lucene
54   <p/>
55   *
56   * See <a href="http://issues.apache.org/jira/browse/LUCENE-1058">LUCENE-1058</a>.
57   * <p/>
58   * WARNING: {@link TeeTokenFilter} and {@link SinkTokenizer} only work with the old TokenStream API.
59   * If you switch to the new API, you need to use {@link TeeSinkTokenFilter} instead, which offers 
60   * the same functionality.
61  
62   * @see SinkTokenizer
63   * @deprecated Use {@link TeeSinkTokenFilter} instead
64   **/
65  public class TeeTokenFilter extends TokenFilter {
66    SinkTokenizer sink;
67  
68    public TeeTokenFilter(TokenStream input, SinkTokenizer sink) {
69      super(input);
70      this.sink = sink;
71    }
72    
73    public Token next(final Token reusableToken) throws IOException {
74      assert reusableToken != null;
75      Token nextToken = input.next(reusableToken);
76      sink.add(nextToken);
77      return nextToken;
78    }
79  
80  }
