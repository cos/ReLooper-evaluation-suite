26  /**
27   * A simple class that stores Strings as char[]'s in a
28   * hash table.  Note that this is not a general purpose
29   * class.  For example, it cannot remove items from the
30   * set, nor does it resize its hash table to be smaller,
31   * etc.  It is designed to be quick to test if a char[]
32   * is in the set without the necessity of converting it
33   * to a String first.
34   */
35  
36  public class CharArraySet extends AbstractSet {
37    private final static int INIT_SIZE = 8;
38    private char[][] entries;
39    private int count;
40    private final boolean ignoreCase;
41  
42    /** Create set with enough capacity to hold startSize
43     *  terms */
44    public CharArraySet(int startSize, boolean ignoreCase) {
45      this.ignoreCase = ignoreCase;
46      int size = INIT_SIZE;
47      while(startSize + (startSize>>2) > size)
48        size <<= 1;
49      entries = new char[size][];
50    }
51  
52   /** Create set from a Collection of char[] or String */
53    public CharArraySet(Collection c, boolean ignoreCase) {
54      this(c.size(), ignoreCase);
55      addAll(c);
56    }
57    /** Create set from entries */
58    private CharArraySet(char[][] entries, boolean ignoreCase, int count){
59      this.entries = entries;
60      this.ignoreCase = ignoreCase;
61      this.count = count;
62    }
63  
64    /** true if the <code>len</code> chars of <code>text</code> starting at <code>off</code>
65     * are in the set */
66    public boolean contains(char[] text, int off, int len) {
67      return entries[getSlot(text, off, len)] != null;
68    }
69  
70    /** true if the <code>CharSequence</code> is in the set */
71    public boolean contains(CharSequence cs) {
72      return entries[getSlot(cs)] != null;
73    }
74  
75    private int getSlot(char[] text, int off, int len) {
76      int code = getHashCode(text, off, len);
77      int pos = code & (entries.length-1);
78      char[] text2 = entries[pos];
79      if (text2 != null && !equals(text, off, len, text2)) {
80        final int inc = ((code>>8)+code)|1;
81        do {
82          code += inc;
83          pos = code & (entries.length-1);
84          text2 = entries[pos];
85        } while (text2 != null && !equals(text, off, len, text2));
86      }
87      return pos;
88    }
89  
90    /** Returns true if the String is in the set */  
91    private int getSlot(CharSequence text) {
92      int code = getHashCode(text);
93      int pos = code & (entries.length-1);
94      char[] text2 = entries[pos];
95      if (text2 != null && !equals(text, text2)) {
96        final int inc = ((code>>8)+code)|1;
97        do {
98          code += inc;
99          pos = code & (entries.length-1);
100         text2 = entries[pos];
101       } while (text2 != null && !equals(text, text2));
102     }
103     return pos;
104   }
105 
106   /** Add this CharSequence into the set */
107   public boolean add(CharSequence text) {
108     return add(text.toString()); // could be more efficient
109   }
110   
111   /** Add this String into the set */
112   public boolean add(String text) {
113     return add(text.toCharArray());
114   }
115 
116   /** Add this char[] directly to the set.
117    * If ignoreCase is true for this Set, the text array will be directly modified.
118    * The user should never modify this text array after calling this method.
119    */
120   public boolean add(char[] text) {
121     if (ignoreCase)
122       for(int i=0;i<text.length;i++)
123         text[i] = Character.toLowerCase(text[i]);
124     int slot = getSlot(text, 0, text.length);
125     if (entries[slot] != null) return false;
126     entries[slot] = text;
127     count++;
128 
129     if (count + (count>>2) > entries.length) {
130       rehash();
131     }
132 
133     return true;
134   }
135 
136   private boolean equals(char[] text1, int off, int len, char[] text2) {
137     if (len != text2.length)
138       return false;
139     if (ignoreCase) {
140       for(int i=0;i<len;i++) {
141         if (Character.toLowerCase(text1[off+i]) != text2[i])
142           return false;
143       }
144     } else {
145       for(int i=0;i<len;i++) {
146         if (text1[off+i] != text2[i])
147           return false;
148       }
149     }
150     return true;
151   }
152 
153   private boolean equals(CharSequence text1, char[] text2) {
154     int len = text1.length();
155     if (len != text2.length)
156       return false;
157     if (ignoreCase) {
158       for(int i=0;i<len;i++) {
159         if (Character.toLowerCase(text1.charAt(i)) != text2[i])
160           return false;
161       }
162     } else {
163       for(int i=0;i<len;i++) {
164         if (text1.charAt(i) != text2[i])
165           return false;
166       }
167     }
168     return true;
169   }
170 
171   private void rehash() {
172     final int newSize = 2*entries.length;
173     char[][] oldEntries = entries;
174     entries = new char[newSize][];
175 
176     for(int i=0;i<oldEntries.length;i++) {
177       char[] text = oldEntries[i];
178       if (text != null) {
179         // todo: could be faster... no need to compare strings on collision
180         entries[getSlot(text,0,text.length)] = text;
181       }
182     }
183   }
184   
185   private int getHashCode(char[] text, int offset, int len) {
186     int code = 0;
187     final int stop = offset + len;
188     if (ignoreCase) {
189       for (int i=offset; i<stop; i++) {
190         code = code*31 + Character.toLowerCase(text[i]);
191       }
192     } else {
193       for (int i=offset; i<stop; i++) {
194         code = code*31 + text[i];
195       }
196     }
197     return code;
198   }
199 
200   private int getHashCode(CharSequence text) {
201     int code = 0;
202     int len = text.length();
203     if (ignoreCase) {
204       for (int i=0; i<len; i++) {
205         code = code*31 + Character.toLowerCase(text.charAt(i));
206       }
207     } else {
208       for (int i=0; i<len; i++) {
209         code = code*31 + text.charAt(i);
210       }
211     }
212     return code;
213   }

217     return count;

221     return count==0;

224   public boolean contains(Object o) {
225     if (o instanceof char[]) {
226       char[] text = (char[])o;
227       return contains(text, 0, text.length);
228     } 
229     return contains(o.toString());
230   }

232   public boolean add(Object o) {
233     if (o instanceof char[]) {
234       return add((char[])o);
235     }
236     return add(o.toString());
237   }

239   /**
240    * Returns an unmodifiable {@link CharArraySet}. This allows to provide
241    * unmodifiable views of internal sets for "read-only" use.
242    * 
243    * @param set
244    *          a set for which the unmodifiable set is returned.
245    * @return an new unmodifiable {@link CharArraySet}.
246    * @throws NullPointerException
247    *           if the given set is <code>null</code>.
248    */
249   public static CharArraySet unmodifiableSet(CharArraySet set) {
250     if (set == null)
251       throw new NullPointerException("Given set is null");
252     /*
253      * Instead of delegating calls to the given set copy the low-level values to
254      * the unmodifiable Subclass
255      */
256     return new UnmodifiableCharArraySet(set.entries, set.ignoreCase, set.count);
257   }

259   /** The Iterator<String> for this set.  Strings are constructed on the fly, so
260    * use <code>nextCharArray</code> for more efficient access. */
261   public class CharArraySetIterator implements Iterator {
262     int pos=-1;
263     char[] next;
264     CharArraySetIterator() {
265       goNext();
266     }
267 
268     private void goNext() {
269       next = null;
270       pos++;
271       while (pos < entries.length && (next=entries[pos]) == null) pos++;
272     }
273 
274     public boolean hasNext() {
275       return next != null;
276     }
277 
278     /** do not modify the returned char[] */
279     public char[] nextCharArray() {
280       char[] ret = next;
281       goNext();
282       return ret;
283     }
284 
285     /** Returns the next String, as a Set<String> would...
286      * use nextCharArray() for better efficiency. */
287     public Object next() {
288       return new String(nextCharArray());
289     }
290 
291     public void remove() {
292       throw new UnsupportedOperationException();
293     }
294   }
295 
296 
297   public Iterator iterator() {
298     return new CharArraySetIterator();
299   }

312       super(entries, ignoreCase, count);

324       throw new UnsupportedOperationException();
325     }

327     public boolean add(CharSequence text) {
328       throw new UnsupportedOperationException();

331     public boolean add(String text) {
332       throw new UnsupportedOperationException();
