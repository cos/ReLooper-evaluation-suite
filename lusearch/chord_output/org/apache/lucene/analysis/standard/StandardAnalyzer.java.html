34   * <p>You must specify the required {@link Version}
35   * compatibility when creating StandardAnalyzer:
36   * <ul>
37   *   <li> As of 2.9, StopFilter preserves position
38   *        increments
39   *   <li> As of 2.4, Tokens incorrectly identified as acronyms
40   *        are corrected (see <a href="https://issues.apache.org/jira/browse/LUCENE-1068">LUCENE-1608</a>
41   * </ul>

255   private final void init(Version matchVersion) {
256     setOverridesTokenStreamMethod(StandardAnalyzer.class);
257     if (matchVersion.onOrAfter(Version.LUCENE_29)) {
258       enableStopPositionIncrements = true;
259     } else {
260       useDefaultStopPositionIncrements = true;
261     }
262     if (matchVersion.onOrAfter(Version.LUCENE_24)) {
263       replaceInvalidAcronym = defaultReplaceInvalidAcronym;
264     } else {
265       replaceInvalidAcronym = false;
266     }
267   }

269   /** Constructs a {@link StandardTokenizer} filtered by a {@link
270   StandardFilter}, a {@link LowerCaseFilter} and a {@link StopFilter}. */
271   public TokenStream tokenStream(String fieldName, Reader reader) {
272     StandardTokenizer tokenStream = new StandardTokenizer(reader, replaceInvalidAcronym);
273     tokenStream.setMaxTokenLength(maxTokenLength);
274     TokenStream result = new StandardFilter(tokenStream);
275     result = new LowerCaseFilter(result);
276     if (useDefaultStopPositionIncrements) {
277       result = new StopFilter(result, stopSet);
278     } else {
279       result = new StopFilter(enableStopPositionIncrements, result, stopSet);
280     }
281     return result;
282   }

284   private static final class SavedStreams {
285     StandardTokenizer tokenStream;
286     TokenStream filteredTokenStream;
287   }

289   /** Default maximum allowed token length */
290   public static final int DEFAULT_MAX_TOKEN_LENGTH = 255;
291 
292   private int maxTokenLength = DEFAULT_MAX_TOKEN_LENGTH;
293 
294   /**
295    * Set maximum allowed token length.  If a token is seen
296    * that exceeds this length then it is discarded.  This
297    * setting only takes effect the next time tokenStream or
298    * reusableTokenStream is called.
299    */
300   public void setMaxTokenLength(int length) {
301     maxTokenLength = length;
302   }
303     
304   /**
305    * @see #setMaxTokenLength
306    */
307   public int getMaxTokenLength() {
308     return maxTokenLength;
309   }

311   /** @deprecated Use {@link #tokenStream} instead */
312   public TokenStream reusableTokenStream(String fieldName, Reader reader) throws IOException {
313     if (overridesTokenStreamMethod) {
314       // LUCENE-1678: force fallback to tokenStream() if we
315       // have been subclassed and that subclass overrides
316       // tokenStream but not reusableTokenStream
317       return tokenStream(fieldName, reader);
318     }
319     SavedStreams streams = (SavedStreams) getPreviousTokenStream();
320     if (streams == null) {
321       streams = new SavedStreams();
322       setPreviousTokenStream(streams);
323       streams.tokenStream = new StandardTokenizer(reader);
324       streams.filteredTokenStream = new StandardFilter(streams.tokenStream);
325       streams.filteredTokenStream = new LowerCaseFilter(streams.filteredTokenStream);
326       if (useDefaultStopPositionIncrements) {
327         streams.filteredTokenStream = new StopFilter(streams.filteredTokenStream, stopSet);
328       } else {
329         streams.filteredTokenStream = new StopFilter(enableStopPositionIncrements, streams.filteredTokenStream, stopSet);
330       }
331     } else {
332       streams.tokenStream.reset(reader);
333     }
334     streams.tokenStream.setMaxTokenLength(maxTokenLength);
335     
336     streams.tokenStream.setReplaceInvalidAcronym(replaceInvalidAcronym);
337 
338     return streams.filteredTokenStream;
339   }
