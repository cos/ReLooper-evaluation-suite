32  /** A grammar-based tokenizer constructed with JFlex
33   *
34   * <p> This should be a good tokenizer for most European-language documents:
35   *
36   * <ul>
37   *   <li>Splits words at punctuation characters, removing punctuation. However, a 
38   *     dot that's not followed by whitespace is considered part of a token.
39   *   <li>Splits words at hyphens, unless there's a number in the token, in which case
40   *     the whole token is interpreted as a product number and is not split.
41   *   <li>Recognizes email addresses and internet hostnames as one token.
42   * </ul>
43   *
44   * <p>Many applications have specific tokenizer needs.  If this tokenizer does
45   * not suit your application, please consider copying this source code
46   * directory to your project and maintaining your own grammar-based tokenizer.
47   *
48   * <a name="version"/>
49   * <p>You must specify the required {@link Version}
50   * compatibility when creating StandardAnalyzer:
51   * <ul>
52   *   <li> As of 2.4, Tokens incorrectly identified as acronyms
53   *        are corrected (see <a href="https://issues.apache.org/jira/browse/LUCENE-1068">LUCENE-1608</a>
54   * </ul>
55   */
56  
57  public class StandardTokenizer extends Tokenizer {
58    /** A private instance of the JFlex-constructed scanner */
59    private final StandardTokenizerImpl scanner;

61    public static final int ALPHANUM          = 0;
62    public static final int APOSTROPHE        = 1;
63    public static final int ACRONYM           = 2;
64    public static final int COMPANY           = 3;
65    public static final int EMAIL             = 4;
66    public static final int HOST              = 5;
67    public static final int NUM               = 6;
68    public static final int CJ                = 7;

72     *             as ACRONYMs. It is deprecated and will be removed in the next

82      "<COMPANY>",

198   private void init(Reader input, boolean replaceInvalidAcronym) {
199     this.replaceInvalidAcronym = replaceInvalidAcronym;
200     this.input = input;    
201     termAtt = (TermAttribute) addAttribute(TermAttribute.class);
202     offsetAtt = (OffsetAttribute) addAttribute(OffsetAttribute.class);
203     posIncrAtt = (PositionIncrementAttribute) addAttribute(PositionIncrementAttribute.class);
204     typeAtt = (TypeAttribute) addAttribute(TypeAttribute.class);
205   }
206 
207   private void init(Reader input, Version matchVersion) {
208     if (matchVersion.onOrAfter(Version.LUCENE_24)) {
209       init(input, true);
210     } else {
211       init(input, false);
212     }
213   }
214   
215   // this tokenizer generates three attributes:
216   // offset, positionIncrement and type
217   private TermAttribute termAtt;
218   private OffsetAttribute offsetAtt;
219   private PositionIncrementAttribute posIncrAtt;
220   private TypeAttribute typeAtt;
221 
222   /*
223    * (non-Javadoc)
224    *
225    * @see org.apache.lucene.analysis.TokenStream#next()
226    */
227   public final boolean incrementToken() throws IOException {
228     clearAttributes();
229     int posIncr = 1;
230 
231     while(true) {
232       int tokenType = scanner.getNextToken();
233 
234       if (tokenType == StandardTokenizerImpl.YYEOF) {
235         return false;
236       }
237 
238       if (scanner.yylength() <= maxTokenLength) {
239         posIncrAtt.setPositionIncrement(posIncr);
240         scanner.getText(termAtt);
241         final int start = scanner.yychar();
242         offsetAtt.setOffset(correctOffset(start), correctOffset(start+termAtt.termLength()));
243         // This 'if' should be removed in the next release. For now, it converts
244         // invalid acronyms to HOST. When removed, only the 'else' part should
245         // remain.
246         if (tokenType == StandardTokenizerImpl.ACRONYM_DEP) {
247           if (replaceInvalidAcronym) {
248             typeAtt.setType(StandardTokenizerImpl.TOKEN_TYPES[StandardTokenizerImpl.HOST]);
249             termAtt.setTermLength(termAtt.termLength() - 1); // remove extra '.'
250           } else {
251             typeAtt.setType(StandardTokenizerImpl.TOKEN_TYPES[StandardTokenizerImpl.ACRONYM]);
252           }
253         } else {
254           typeAtt.setType(StandardTokenizerImpl.TOKEN_TYPES[tokenType]);
255         }
256         return true;
257       } else
258         // When we skip a too-long term, we still increment the
259         // position increment
260         posIncr++;
261     }
262   }
263   
264   public final void end() {
265     // set final offset
266     int finalOffset = correctOffset(scanner.yychar() + scanner.yylength());
267     offsetAtt.setOffset(finalOffset, finalOffset);
268   }
