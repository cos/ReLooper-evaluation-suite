20  import java.io.IOException;
21  import java.io.Reader;
22  
23  import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
24  import org.apache.lucene.analysis.tokenattributes.TermAttribute;
25  import org.apache.lucene.util.AttributeSource;
26  
27  /** An abstract base class for simple, character-oriented tokenizers.*/
28  public abstract class CharTokenizer extends Tokenizer {
29    public CharTokenizer(Reader input) {
30      super(input);
31      offsetAtt = (OffsetAttribute) addAttribute(OffsetAttribute.class);
32      termAtt = (TermAttribute) addAttribute(TermAttribute.class);
33    }
34  
35    public CharTokenizer(AttributeSource source, Reader input) {
36      super(source, input);
37      offsetAtt = (OffsetAttribute) addAttribute(OffsetAttribute.class);
38      termAtt = (TermAttribute) addAttribute(TermAttribute.class);
39    }
40  
41    public CharTokenizer(AttributeFactory factory, Reader input) {
42      super(factory, input);
43      offsetAtt = (OffsetAttribute) addAttribute(OffsetAttribute.class);
44      termAtt = (TermAttribute) addAttribute(TermAttribute.class);
45    }
46    
47    private int offset = 0, bufferIndex = 0, dataLen = 0;
48    private static final int MAX_WORD_LEN = 255;
49    private static final int IO_BUFFER_SIZE = 4096;
50    private final char[] ioBuffer = new char[IO_BUFFER_SIZE];
51    
52    private TermAttribute termAtt;
53    private OffsetAttribute offsetAtt;
54  
55    /** Returns true iff a character should be included in a token.  This
56     * tokenizer generates as tokens adjacent sequences of characters which
57     * satisfy this predicate.  Characters for which this is false are used to
58     * define token boundaries and are not included in tokens. */
59    protected abstract boolean isTokenChar(char c);
60  
61    /** Called on each token character to normalize it before it is added to the
62     * token.  The default implementation does nothing.  Subclasses may use this
63     * to, e.g., lowercase tokens. */
64    protected char normalize(char c) {
65      return c;
66    }
67  
68    public final boolean incrementToken() throws IOException {
69      clearAttributes();
70      int length = 0;
71      int start = bufferIndex;
72      char[] buffer = termAtt.termBuffer();
73      while (true) {
74  
75        if (bufferIndex >= dataLen) {
76          offset += dataLen;
77          dataLen = input.read(ioBuffer);
78          if (dataLen == -1) {
79            dataLen = 0;                            // so next offset += dataLen won't decrement offset
80            if (length > 0)
81              break;
82            else
83              return false;
84          }
85          bufferIndex = 0;
86        }
87  
88        final char c = ioBuffer[bufferIndex++];
89  
90        if (isTokenChar(c)) {               // if it's a token char
91  
92          if (length == 0)                 // start of token
93            start = offset + bufferIndex - 1;
94          else if (length == buffer.length)
95            buffer = termAtt.resizeTermBuffer(1+length);
96  
97          buffer[length++] = normalize(c); // buffer it, normalized
98  
99          if (length == MAX_WORD_LEN)      // buffer overflow!
100           break;
101 
102       } else if (length > 0)             // at non-Letter w/ chars
103         break;                           // return 'em
104     }
105 
106     termAtt.setTermLength(length);
107     offsetAtt.setOffset(correctOffset(start), correctOffset(start+length));
108     return true;
109   }
110   
111   public final void end() {
112     // set final offset
113     int finalOffset = correctOffset(offset);
114     offsetAtt.setOffset(finalOffset, finalOffset);
115   }
116 
117   /** @deprecated Will be removed in Lucene 3.0. This method is final, as it should
118    * not be overridden. Delegates to the backwards compatibility layer. */
119   public final Token next(final Token reusableToken) throws IOException {
120     return super.next(reusableToken);
121   }
122 
123   /** @deprecated Will be removed in Lucene 3.0. This method is final, as it should
124    * not be overridden. Delegates to the backwards compatibility layer. */
125   public final Token next() throws IOException {
126     return super.next();
127   }
128 
129   public void reset(Reader input) throws IOException {
130     super.reset(input);
131     bufferIndex = 0;
132     offset = 0;
133     dataLen = 0;
134   }
135 }
