18  package org.apache.lucene.analysis;
19  
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.Iterator;
23  import java.util.List;
24  
25  
26  /**
27   * A SinkTokenizer can be used to cache Tokens for use in an Analyzer
28   * <p/>
29   * WARNING: {@link TeeTokenFilter} and {@link SinkTokenizer} only work with the old TokenStream API.
30   * If you switch to the new API, you need to use {@link TeeSinkTokenFilter} instead, which offers 
31   * the same functionality.
32   * @see TeeTokenFilter
33   * @deprecated Use {@link TeeSinkTokenFilter} instead
34   *
35   **/
36  public class SinkTokenizer extends Tokenizer {
37    protected List/*<Token>*/ lst = new ArrayList/*<Token>*/();
38    protected Iterator/*<Token>*/ iter;
39  
40    public SinkTokenizer(List/*<Token>*/ input) {
41      this.lst = input;
42      if (this.lst == null) this.lst = new ArrayList/*<Token>*/();
43    }
44  
45    public SinkTokenizer() {
46      this.lst = new ArrayList/*<Token>*/();
47    }
48  
49    public SinkTokenizer(int initCap){
50      this.lst = new ArrayList/*<Token>*/(initCap);
51    }
52  
53    /**
54     * Get the tokens in the internal List.
55     * <p/>
56     * WARNING: Adding tokens to this list requires the {@link #reset()} method to be called in order for them
57     * to be made available.  Also, this Tokenizer does nothing to protect against {@link java.util.ConcurrentModificationException}s
58     * in the case of adds happening while {@link #next(org.apache.lucene.analysis.Token)} is being called.
59     * <p/>
60     * WARNING: Since this SinkTokenizer can be reset and the cached tokens made available again, do not modify them. Modify clones instead.
61     *
62     * @return A List of {@link org.apache.lucene.analysis.Token}s
63     */
64    public List/*<Token>*/ getTokens() {
65      return lst;
66    }
67  
68    /**
69     * Returns the next token out of the list of cached tokens
70     * @return The next {@link org.apache.lucene.analysis.Token} in the Sink.
71     * @throws IOException
72     */
73    public Token next(final Token reusableToken) throws IOException {
74      assert reusableToken != null;
75      if (iter == null) iter = lst.iterator();
76      // Since this TokenStream can be reset we have to maintain the tokens as immutable
77      if (iter.hasNext()) {
78        Token nextToken = (Token) iter.next();
79        return (Token) nextToken.clone();
80      }
81      return null;
82    }
83  
84    /**
85     * Override this method to cache only certain tokens, or new tokens based
86     * on the old tokens.
87     *
88     * @param t The {@link org.apache.lucene.analysis.Token} to add to the sink
89     */
90    public void add(Token t) {
91      if (t == null) return;
92      lst.add((Token) t.clone());
93    }
94  
95    public void close() throws IOException {
96      //nothing to close
97      input = null;
98      lst = null;
99    }
100 
101   /**
102    * Reset the internal data structures to the start at the front of the list of tokens.  Should be called
103    * if tokens were added to the list after an invocation of {@link #next(Token)}
104    * @throws IOException
105    */
106   public void reset() throws IOException {
107     iter = lst.iterator();
108   }
109 }
