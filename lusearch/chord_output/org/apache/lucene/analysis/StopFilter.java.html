113   /**
114    * Construct a token stream filtering the given input.
115    * If <code>stopWords</code> is an instance of {@link CharArraySet} (true if
116    * <code>makeStopSet()</code> was used to construct the set) it will be directly used
117    * and <code>ignoreCase</code> will be ignored since <code>CharArraySet</code>
118    * directly controls case sensitivity.
119    * <p/>
120    * If <code>stopWords</code> is not an instance of {@link CharArraySet},
121    * a new CharArraySet will be constructed and <code>ignoreCase</code> will be
122    * used to specify the case sensitivity of that set.
123    *
124    * @param enablePositionIncrements true if token positions should record the removed stop words
125    * @param input Input TokenStream
126    * @param stopWords The set of Stop Words.
127    * @param ignoreCase -Ignore case when stopping.
128    */
129   public StopFilter(boolean enablePositionIncrements, TokenStream input, Set stopWords, boolean ignoreCase)
130   {
131     super(input);
132     if (stopWords instanceof CharArraySet) {
133       this.stopWords = (CharArraySet)stopWords;
134     } else {
135       this.stopWords = new CharArraySet(stopWords.size(), ignoreCase);
136       this.stopWords.addAll(stopWords);
137     }
138     this.enablePositionIncrements = enablePositionIncrements;
139     init();
140   }

195   /**
196    * 
197    * @param stopWords An array of stopwords
198    * @param ignoreCase If true, all words are lower cased first.  
199    * @return a Set containing the words
200    */    
201   public static final Set makeStopSet(String[] stopWords, boolean ignoreCase) {
202     CharArraySet stopSet = new CharArraySet(stopWords.length, ignoreCase);
203     stopSet.addAll(Arrays.asList(stopWords));
204     return stopSet;
205   }

211    * @return A Set containing the words

213   public static final Set makeStopSet(List/*<String>*/ stopWords, boolean ignoreCase){
214     CharArraySet stopSet = new CharArraySet(stopWords.size(), ignoreCase);
215     stopSet.addAll(stopWords);
216     return stopSet;

219   /**
220    * Returns the next input Token whose term() is not a stop word.
221    */
222   public final boolean incrementToken() throws IOException {
223     // return the first non-stop word found
224     int skippedPositions = 0;
225     while (input.incrementToken()) {
226       if (!stopWords.contains(termAtt.termBuffer(), 0, termAtt.termLength())) {
227         if (enablePositionIncrements) {
228           posIncrAtt.setPositionIncrement(posIncrAtt.getPositionIncrement() + skippedPositions);
229         }
230         return true;
231       }
232       skippedPositions += posIncrAtt.getPositionIncrement();
233     }
234     // reached EOS -- return null
235     return false;
236   }

246   /**
247    * Returns version-dependent default for
248    * enablePositionIncrements.  Analyzers that embed
249    * StopFilter use this method when creating the
250    * StopFilter.  Prior to 2.9, this returns {@link #getEnablePositionIncrementsDefault}.
251    * On 2.9 or later, it returns true.
252    */
253   public static boolean getEnablePositionIncrementsVersionDefault(Version matchVersion) {
254     if (matchVersion.onOrAfter(Version.LUCENE_29)) {
255       return true;
256     } else {
257       return ENABLE_POSITION_INCREMENTS_DEFAULT;
258     }
259   }

263    * <p>
264    * Note: behavior of a single StopFilter instance can be modified 
265    * with {@link #setEnablePositionIncrements(boolean)}.
266    * This static method allows control over behavior of classes using StopFilters internally,

268    * if used with the no-arg ctor.
269    * <p>
270    * Default : false.
271    * @see #setEnablePositionIncrements(boolean).
272    * @deprecated Please specify this when you create the StopFilter
273    */
274   public static void setEnablePositionIncrementsDefault(boolean defaultValue) {
275     ENABLE_POSITION_INCREMENTS_DEFAULT = defaultValue;
276   }

285   /**
286    * If <code>true</code>, this StopFilter will preserve
287    * positions of the incoming tokens (ie, accumulate and
288    * set position increments of the removed stop tokens).
289    * Generally, <code>true</code> is best as it does not
290    * lose information (positions of the original tokens)
291    * during indexing.
292    * 
293    * <p> When set, when a token is stopped
294    * (omitted), the position increment of the following
295    * token is incremented.
296    *
297    * <p> <b>NOTE</b>: be sure to also
298    * set {@link QueryParser#setEnablePositionIncrements} if
299    * you use QueryParser to create queries.
300    */
301   public void setEnablePositionIncrements(boolean enable) {
302     this.enablePositionIncrements = enable;
303   }
