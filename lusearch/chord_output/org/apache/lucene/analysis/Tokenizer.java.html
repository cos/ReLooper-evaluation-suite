20  import org.apache.lucene.util.AttributeSource;
21  
22  import java.io.Reader;
23  import java.io.IOException;
24  
25  /** A Tokenizer is a TokenStream whose input is a Reader.
26    <p>
27    This is an abstract class; subclasses must override {@link #incrementToken()}
28    <p>
29    NOTE: Subclasses overriding {@link #incrementToken()} must
30    call {@link AttributeSource#clearAttributes()} before
31    setting attributes.
32    Subclasses overriding {@link #next(Token)} must call
33    {@link Token#clear()} before setting Token attributes. 
34   */
35  public abstract class Tokenizer extends TokenStream {
36    /** The text source for this Tokenizer. */
37    protected Reader input;
38  
39    /** Construct a tokenizer with null input. */
40    protected Tokenizer() {}
41    
42    /** Construct a token stream processing the given input. */
43    protected Tokenizer(Reader input) {
44      this.input = CharReader.get(input);
45    }
46    
47    /** Construct a tokenizer with null input using the given AttributeFactory. */
48    protected Tokenizer(AttributeFactory factory) {
49      super(factory);
50    }
51  
52    /** Construct a token stream processing the given input using the given AttributeFactory. */
53    protected Tokenizer(AttributeFactory factory, Reader input) {
54      super(factory);
55      this.input = CharReader.get(input);
56    }
57  
58    /** Construct a token stream processing the given input using the given AttributeSource. */
59    protected Tokenizer(AttributeSource source) {
60      super(source);
61    }
62  
63    /** Construct a token stream processing the given input using the given AttributeSource. */
64    protected Tokenizer(AttributeSource source, Reader input) {
65      super(source);
66      this.input = CharReader.get(input);
67    }
68    
69    /** By default, closes the input Reader. */
70    public void close() throws IOException {
71      input.close();
72    }
73    
74    /** Return the corrected offset. If {@link #input} is a {@link CharStream} subclass
75     * this method calls {@link CharStream#correctOffset}, else returns <code>currentOff</code>.
76     * @param currentOff offset as seen in the output
77     * @return corrected offset based on the input
78     * @see CharStream#correctOffset
79     */
80    protected final int correctOffset(int currentOff) {
81      return (input instanceof CharStream) ? ((CharStream) input).correctOffset(currentOff) : currentOff;
82    }
83  
84    /** Expert: Reset the tokenizer to a new reader.  Typically, an
85     *  analyzer (in its reusableTokenStream method) will use
86     *  this to re-use a previously created tokenizer. */
87    public void reset(Reader input) throws IOException {
88      this.input = input;
89    }
90  }
