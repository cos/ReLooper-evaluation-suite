18  package org.apache.lucene.analysis;
19  
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  /**
24   * Base utility class for implementing a {@link CharFilter}.
25   * You subclass this, and then record mappings by calling
26   * {@link #addOffCorrectMap}, and then invoke the correct
27   * method to correct an offset.
28   *
29   * <p><b>NOTE</b>: This class is not particularly efficient.
30   * For example, a new class instance is created for every
31   * call to {@link #addOffCorrectMap}, which is then appended
32   * to a private list.
33   */
34  public abstract class BaseCharFilter extends CharFilter {
35  
36    //private List<OffCorrectMap> pcmList;
37    private List pcmList;
38    
39    public BaseCharFilter(CharStream in) {
40      super(in);
41    }
42  
43    /** Retrieve the corrected offset.  Note that this method
44     *  is slow, if you correct positions far before the most
45     *  recently added position, as it's a simple linear
46     *  search backwards through all offset corrections added
47     *  by {@link #addOffCorrectMap}. */
48    protected int correct(int currentOff) {
49      if (pcmList == null || pcmList.isEmpty()) {
50        return currentOff;
51      }
52      for (int i = pcmList.size() - 1; i >= 0; i--) {
53        if (currentOff >= ((OffCorrectMap) pcmList.get(i)).off) {
54          return currentOff + ((OffCorrectMap) pcmList.get(i)).cumulativeDiff;
55        }
56      }
57      return currentOff;
58    }
59    
60    protected int getLastCumulativeDiff() {
61      return pcmList == null || pcmList.isEmpty() ?
62        0 : ((OffCorrectMap)pcmList.get(pcmList.size() - 1)).cumulativeDiff;
63    }
64  
65    protected void addOffCorrectMap(int off, int cumulativeDiff) {
66      if (pcmList == null) {
67        pcmList = new ArrayList();
68      }
69      pcmList.add(new OffCorrectMap(off, cumulativeDiff));
70    }
71  
72    static class OffCorrectMap {
73  
74      int off;
75      int cumulativeDiff;
76  
77      OffCorrectMap(int off, int cumulativeDiff) {
78        this.off = off;
79        this.cumulativeDiff = cumulativeDiff;
80      }
81  
82      public String toString() {
83        StringBuffer sb = new StringBuffer();
84        sb.append('(');
85        sb.append(off);
86        sb.append(',');
87        sb.append(cumulativeDiff);
88        sb.append(')');
89        return sb.toString();
90      }
91    }
92  }
