18  package org.apache.lucene.analysis;
19  
20  import java.io.Reader;
21  
22  /**
23   * CharStream adds {@link #correctOffset}
24   * functionality over {@link Reader}.  All Tokenizers accept a
25   * CharStream instead of {@link Reader} as input, which enables
26   * arbitrary character based filtering before tokenization. 
27   * The {@link #correctOffset} method fixed offsets to account for
28   * removal or insertion of characters, so that the offsets
29   * reported in the tokens match the character offsets of the
30   * original Reader.
31   */
32  public abstract class CharStream extends Reader {
33  
34    /**
35     * Called by CharFilter(s) and Tokenizer to correct token offset.
36     *
37     * @param currentOff offset as seen in the output
38     * @return corrected offset based on the input
39     */
40    public abstract int correctOffset(int currentOff);
41  }
