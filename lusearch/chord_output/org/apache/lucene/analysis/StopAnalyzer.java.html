28  /** Filters {@link LetterTokenizer} with {@link
29   * LowerCaseFilter} and {@link StopFilter}.
30   *
31   * <a name="version"/>
32   * <p>You must specify the required {@link Version}
33   * compatibility when creating StopAnalyzer:
34   * <ul>
35   *   <li> As of 2.9, position increments are preserved
36   * </ul>
37  */
38  
39  public final class StopAnalyzer extends Analyzer {
40    private final Set/*<String>*/ stopWords;
41    // @deprecated
42    private final boolean useDefaultStopPositionIncrement;
43    private final boolean enablePositionIncrements;

45    /** An array containing some common English words that are not usually useful
46      for searching. 
47      @deprecated Use {@link #ENGLISH_STOP_WORDS_SET} instead */
48    public static final String[] ENGLISH_STOP_WORDS = {
49      "a", "an", "and", "are", "as", "at", "be", "but", "by",
50      "for", "if", "in", "into", "is", "it",
51      "no", "not", "of", "on", "or", "such",
52      "that", "the", "their", "then", "there", "these",
53      "they", "this", "to", "was", "will", "with"
54    };
55    
56    /** An unmodifiable set containing some common English words that are not usually useful
57    for searching.*/
58    public static final Set/*<String>*/ ENGLISH_STOP_WORDS_SET;
59    
60    static {
61        final String[] stopWords = new String[]{
62          "a", "an", "and", "are", "as", "at", "be", "but", "by",
63          "for", "if", "in", "into", "is", "it",
64          "no", "not", "of", "on", "or", "such",
65          "that", "the", "their", "then", "there", "these",
66          "they", "this", "to", "was", "will", "with"
67        };
68        final CharArraySet stopSet = new CharArraySet(stopWords.length, false);
69      stopSet.addAll(Arrays.asList(stopWords));  
70        ENGLISH_STOP_WORDS_SET = CharArraySet.unmodifiableSet(stopSet); 
71    }
72    
73    /** Builds an analyzer which removes words in
74     * ENGLISH_STOP_WORDS.
75     * @deprecated Use {@link #StopAnalyzer(Version)} instead */
76    public StopAnalyzer() {
77      stopWords = ENGLISH_STOP_WORDS_SET;
78      useDefaultStopPositionIncrement = true;
79      enablePositionIncrements = false;
80    }

84    public StopAnalyzer(Version matchVersion) {
85      stopWords = ENGLISH_STOP_WORDS_SET;
86      useDefaultStopPositionIncrement = false;
87      enablePositionIncrements = StopFilter.getEnablePositionIncrementsVersionDefault(matchVersion);
88    }

210   /** Filters LowerCaseTokenizer with StopFilter. */
211   public TokenStream tokenStream(String fieldName, Reader reader) {
212     if (useDefaultStopPositionIncrement) {
213       return new StopFilter(new LowerCaseTokenizer(reader), stopWords);
214     } else {
215       return new StopFilter(enablePositionIncrements, new LowerCaseTokenizer(reader), stopWords);
216     }
217   }
218 
219   /** Filters LowerCaseTokenizer with StopFilter. */
220   private class SavedStreams {
221     Tokenizer source;
222     TokenStream result;
223   };
224   public TokenStream reusableTokenStream(String fieldName, Reader reader) throws IOException {
225     SavedStreams streams = (SavedStreams) getPreviousTokenStream();
226     if (streams == null) {
227       streams = new SavedStreams();
228       streams.source = new LowerCaseTokenizer(reader);
229       if (useDefaultStopPositionIncrement) {
230         streams.result = new StopFilter(streams.source, stopWords);
231       } else {
232         streams.result = new StopFilter(enablePositionIncrements, streams.source, stopWords);
233       }
234       setPreviousTokenStream(streams);
235     } else
236       streams.source.reset(reader);
237     return streams.result;
238   }
