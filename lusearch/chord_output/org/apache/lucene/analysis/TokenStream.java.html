37  /**
38   * A <code>TokenStream</code> enumerates the sequence of tokens, either from
39   * {@link Field}s of a {@link Document} or from query text.
40   * <p>
41   * This is an abstract class; concrete subclasses are:
42   * <ul>
43   * <li>{@link Tokenizer}, a <code>TokenStream</code> whose input is a Reader; and
44   * <li>{@link TokenFilter}, a <code>TokenStream</code> whose input is another
45   * <code>TokenStream</code>.
46   * </ul>
47   * A new <code>TokenStream</code> API has been introduced with Lucene 2.9. This API
48   * has moved from being {@link Token}-based to {@link Attribute}-based. While
49   * {@link Token} still exists in 2.9 as a convenience class, the preferred way
50   * to store the information of a {@link Token} is to use {@link AttributeImpl}s.
51   * <p>
52   * <code>TokenStream</code> now extends {@link AttributeSource}, which provides
53   * access to all of the token {@link Attribute}s for the <code>TokenStream</code>.
54   * Note that only one instance per {@link AttributeImpl} is created and reused
55   * for every token. This approach reduces object creation and allows local
56   * caching of references to the {@link AttributeImpl}s. See
57   * {@link #incrementToken()} for further details.
58   * <p>
59   * <b>The workflow of the new <code>TokenStream</code> API is as follows:</b>
60   * <ol>
61   * <li>Instantiation of <code>TokenStream</code>/{@link TokenFilter}s which add/get
62   * attributes to/from the {@link AttributeSource}.
63   * <li>The consumer calls {@link TokenStream#reset()}.
64   * <li>The consumer retrieves attributes from the stream and stores local
65   * references to all attributes it wants to access.
66   * <li>The consumer calls {@link #incrementToken()} until it returns false
67   * consuming the attributes after each call.
68   * <li>The consumer calls {@link #end()} so that any end-of-stream operations
69   * can be performed.
70   * <li>The consumer calls {@link #close()} to release any resource when finished
71   * using the <code>TokenStream</code>.
72   * </ol>
73   * To make sure that filters and consumers know which attributes are available,
74   * the attributes must be added during instantiation. Filters and consumers are
75   * not required to check for availability of attributes in
76   * {@link #incrementToken()}.
77   * <p>
78   * You can find some example code for the new API in the analysis package level
79   * Javadoc.
80   * <p>
81   * Sometimes it is desirable to capture a current state of a <code>TokenStream</code>,
82   * e.g., for buffering purposes (see {@link CachingTokenFilter},
83   * {@link TeeSinkTokenFilter}). For this usecase
84   * {@link AttributeSource#captureState} and {@link AttributeSource#restoreState}
85   * can be used.
86   */
87  public abstract class TokenStream extends AttributeSource {

102   /** @deprecated Remove this when old API is removed! */
103   private static final class MethodSupport {
104     final boolean hasIncrementToken, hasReusableNext, hasNext;
105 
106     MethodSupport(Class clazz) {
107       hasIncrementToken = isMethodOverridden(clazz, "incrementToken", METHOD_NO_PARAMS);
108       hasReusableNext = isMethodOverridden(clazz, "next", METHOD_TOKEN_PARAM);
109       hasNext = isMethodOverridden(clazz, "next", METHOD_NO_PARAMS);
110     }
111     
112     private static boolean isMethodOverridden(Class clazz, String name, Class[] params) {
113       try {
114         return clazz.getMethod(name, params).getDeclaringClass() != TokenStream.class;
115       } catch (NoSuchMethodException e) {
116         // should not happen
117         throw new RuntimeException(e);
118       }
119     }
120     
121     private static final Class[] METHOD_NO_PARAMS = new Class[0];
122     private static final Class[] METHOD_TOKEN_PARAM = new Class[]{Token.class};
123   }
124       
125   /** @deprecated Remove this when old API is removed! */
126   private static final IdentityHashMap/*<Class<? extends TokenStream>,MethodSupport>*/ knownMethodSupport = new IdentityHashMap();
127   
128   /** @deprecated Remove this when old API is removed! */
129   private static MethodSupport getSupportedMethods(Class clazz) {
130     MethodSupport supportedMethods;
131     synchronized(knownMethodSupport) {
132       supportedMethods = (MethodSupport) knownMethodSupport.get(clazz);
133       if (supportedMethods == null) {
134         knownMethodSupport.put(clazz, supportedMethods = new MethodSupport(clazz));
135       }
136     }
137     return supportedMethods;
138   }

140   /** @deprecated Remove this when old API is removed! */
141   private static final class TokenWrapperAttributeFactory extends AttributeFactory {
142     private final AttributeFactory delegate;
143   
144     private TokenWrapperAttributeFactory(AttributeFactory delegate) {
145       this.delegate = delegate;
146     }
147   
148     public AttributeImpl createAttributeInstance(Class attClass) {
149       return attClass.isAssignableFrom(TokenWrapper.class)
150         ? new TokenWrapper()
151         : delegate.createAttributeInstance(attClass);
152     }
153     
154     // this is needed for TeeSinkTokenStream's check for compatibility of AttributeSource,
155     // so two TokenStreams using old API have the same AttributeFactory wrapped by this one.
156     public boolean equals(Object other) {
157       if (this == other) return true;
158       if (other instanceof TokenWrapperAttributeFactory) {
159         final TokenWrapperAttributeFactory af = (TokenWrapperAttributeFactory) other;
160         return this.delegate.equals(af.delegate);
161       }
162       return false;
163     }
164     
165     public int hashCode() {
166       return delegate.hashCode() ^ 0x0a45ff31;
167     }
168   }

182   /**
183    * A TokenStream that uses the same attributes as the supplied one.
184    */

186     super(input);
187     tokenWrapper = initTokenWrapper(input);
188     check();
189   }

191   /**
192    * A TokenStream using the supplied AttributeFactory for creating new {@link Attribute} instances.
193    */
194   protected TokenStream(AttributeFactory factory) {
195     super(onlyUseNewAPI
196       ? factory
197       : new TokenWrapperAttributeFactory(factory)
198     );
199     tokenWrapper = initTokenWrapper(null);
200     check();
201   }

203   /** @deprecated Remove this when old API is removed! */
204   private TokenWrapper initTokenWrapper(AttributeSource input) {
205     if (onlyUseNewAPI) {
206       // no wrapper needed
207       return null;
208     } else {
209       // if possible get the wrapper from the filter's input stream
210       if (input instanceof TokenStream && ((TokenStream) input).tokenWrapper != null) {
211         return ((TokenStream) input).tokenWrapper;
212       }
213       // check that all attributes are implemented by the same TokenWrapper instance
214       final Attribute att = addAttribute(TermAttribute.class);
215       if (att instanceof TokenWrapper &&
216         addAttribute(TypeAttribute.class) == att &&
217         addAttribute(PositionIncrementAttribute.class) == att &&
218         addAttribute(FlagsAttribute.class) == att &&
219         addAttribute(OffsetAttribute.class) == att &&
220         addAttribute(PayloadAttribute.class) == att
221       ) {
222         return (TokenWrapper) att;
223       } else {
224         throw new UnsupportedOperationException(
225           "If onlyUseNewAPI is disabled, all basic Attributes must be implemented by the internal class "+
226           "TokenWrapper. Please make sure, that all TokenStreams/TokenFilters in this chain have been "+
227           "instantiated with this flag disabled and do not add any custom instances for the basic Attributes!"
228         );
229       }
230     }
231   }
232 
233   /** @deprecated Remove this when old API is removed! */
234   private void check() {
235     if (onlyUseNewAPI && !supportedMethods.hasIncrementToken) {
236       throw new UnsupportedOperationException(getClass().getName()+" does not implement incrementToken() which is needed for onlyUseNewAPI.");
237     }
238 
239     // a TokenStream subclass must at least implement one of the methods!
240     if (!(supportedMethods.hasIncrementToken || supportedMethods.hasNext || supportedMethods.hasReusableNext)) {
241       throw new UnsupportedOperationException(getClass().getName()+" does not implement any of incrementToken(), next(Token), next().");
242     }
243   }

245   /**
246    * For extra performance you can globally enable the new
247    * {@link #incrementToken} API using {@link Attribute}s. There will be a
248    * small, but in most cases negligible performance increase by enabling this,
249    * but it only works if <b>all</b> <code>TokenStream</code>s use the new API and
250    * implement {@link #incrementToken}. This setting can only be enabled
251    * globally.
252    * <P>
253    * This setting only affects <code>TokenStream</code>s instantiated after this
254    * call. All <code>TokenStream</code>s already created use the other setting.
255    * <P>
256    * All core {@link Analyzer}s are compatible with this setting, if you have
257    * your own <code>TokenStream</code>s that are also compatible, you should enable
258    * this.
259    * <P>
260    * When enabled, tokenization may throw {@link UnsupportedOperationException}
261    * s, if the whole tokenizer chain is not compatible eg one of the
262    * <code>TokenStream</code>s does not implement the new <code>TokenStream</code> API.
263    * <P>
264    * The default is <code>false</code>, so there is the fallback to the old API
265    * available.
266    * 
267    * @deprecated This setting will no longer be needed in Lucene 3.0 as the old
268    *             API will be removed.
269    */
270   public static void setOnlyUseNewAPI(boolean onlyUseNewAPI) {
271     TokenStream.onlyUseNewAPI = onlyUseNewAPI;
272   }

285   /**
286    * Consumers (i.e., {@link IndexWriter}) use this method to advance the stream to
287    * the next token. Implementing classes must implement this method and update
288    * the appropriate {@link AttributeImpl}s with the attributes of the next
289    * token.
290    * <P>
291    * The producer must make no assumptions about the attributes after the method
292    * has been returned: the caller may arbitrarily change it. If the producer
293    * needs to preserve the state for subsequent calls, it can use
294    * {@link #captureState} to create a copy of the current attribute state.
295    * <p>
296    * This method is called for every token of a document, so an efficient
297    * implementation is crucial for good performance. To avoid calls to
298    * {@link #addAttribute(Class)} and {@link #getAttribute(Class)} or downcasts,
299    * references to all {@link AttributeImpl}s that this stream uses should be
300    * retrieved during instantiation.
301    * <p>
302    * To ensure that filters and consumers know which attributes are available,
303    * the attributes must be added during instantiation. Filters and consumers
304    * are not required to check for availability of attributes in
305    * {@link #incrementToken()}.
306    * 
307    * @return false for end of stream; true otherwise
308    * 
309    *         <p>
310    *         <b>Note that this method will be defined abstract in Lucene
311    *         3.0.</b>
312    */
313   public boolean incrementToken() throws IOException {
314     assert tokenWrapper != null;
315     
316     final Token token;
317     if (supportedMethods.hasReusableNext) {
318       token = next(tokenWrapper.delegate);
319     } else {
320       assert supportedMethods.hasNext;
321       token = next();
322     }
323     if (token == null) return false;
324     tokenWrapper.delegate = token;
325     return true;
326   }

328   /**
329    * This method is called by the consumer after the last token has been
330    * consumed, after {@link #incrementToken()} returned <code>false</code>
331    * (using the new <code>TokenStream</code> API). Streams implementing the old API
332    * should upgrade to use this feature.
333    * <p/>
334    * This method can be used to perform any end-of-stream operations, such as
335    * setting the final offset of a stream. The final offset of a stream might
336    * differ from the offset of the last token eg in case one or more whitespaces
337    * followed after the last token, but a {@link WhitespaceTokenizer} was used.
338    * 
339    * @throws IOException
340    */
341   public void end() throws IOException {
342     // do nothing by default
343   }

345   /**
346    * Returns the next token in the stream, or null at EOS. When possible, the
347    * input Token should be used as the returned Token (this gives fastest
348    * tokenization performance), but this is not required and a new Token may be
349    * returned. Callers may re-use a single Token instance for successive calls
350    * to this method.
351    * <p>
352    * This implicitly defines a "contract" between consumers (callers of this
353    * method) and producers (implementations of this method that are the source
354    * for tokens):
355    * <ul>
356    * <li>A consumer must fully consume the previously returned {@link Token}
357    * before calling this method again.</li>
358    * <li>A producer must call {@link Token#clear()} before setting the fields in
359    * it and returning it</li>
360    * </ul>
361    * Also, the producer must make no assumptions about a {@link Token} after it
362    * has been returned: the caller may arbitrarily change it. If the producer
363    * needs to hold onto the {@link Token} for subsequent calls, it must clone()
364    * it before storing it. Note that a {@link TokenFilter} is considered a
365    * consumer.
366    * 
367    * @param reusableToken a {@link Token} that may or may not be used to return;
368    *        this parameter should never be null (the callee is not required to
369    *        check for null before using it, but it is a good idea to assert that
370    *        it is not null.)
371    * @return next {@link Token} in the stream or null if end-of-stream was hit
372    * @deprecated The new {@link #incrementToken()} and {@link AttributeSource}
373    *             APIs should be used instead.
374    */
375   public Token next(final Token reusableToken) throws IOException {
376     assert reusableToken != null;
377     
378     if (tokenWrapper == null)
379       throw new UnsupportedOperationException("This TokenStream only supports the new Attributes API.");
380     
381     if (supportedMethods.hasIncrementToken) {
382       tokenWrapper.delegate = reusableToken;
383       return incrementToken() ? tokenWrapper.delegate : null;
384     } else {
385       assert supportedMethods.hasNext;
386       return next();
387     }
388   }
389 
390   /**
391    * Returns the next {@link Token} in the stream, or null at EOS.
392    * 
393    * @deprecated The returned Token is a "full private copy" (not re-used across
394    *             calls to {@link #next()}) but will be slower than calling
395    *             {@link #next(Token)} or using the new {@link #incrementToken()}
396    *             method with the new {@link AttributeSource} API.
397    */
398   public Token next() throws IOException {
399     if (tokenWrapper == null)
400       throw new UnsupportedOperationException("This TokenStream only supports the new Attributes API.");
401     
402     final Token nextToken;
403     if (supportedMethods.hasIncrementToken) {
404       final Token savedDelegate = tokenWrapper.delegate;
405       tokenWrapper.delegate = new Token();
406       nextToken = incrementToken() ? tokenWrapper.delegate : null;
407       tokenWrapper.delegate = savedDelegate;
408     } else {
409       assert supportedMethods.hasReusableNext;
410       nextToken = next(new Token());
411     }
412     
413     if (nextToken != null) {
414       Payload p = nextToken.getPayload();
415       if (p != null) {
416         nextToken.setPayload((Payload) p.clone());
417       }
418     }
419     return nextToken;
420   }

422   /**
423    * Resets this stream to the beginning. This is an optional operation, so
424    * subclasses may or may not implement this method. {@link #reset()} is not needed for
425    * the standard indexing process. However, if the tokens of a
426    * <code>TokenStream</code> are intended to be consumed more than once, it is
427    * necessary to implement {@link #reset()}. Note that if your TokenStream
428    * caches tokens and feeds them back again after a reset, it is imperative
429    * that you clone the tokens when you store them away (on the first pass) as
430    * well as when you return them (on future passes after {@link #reset()}).
431    */
432   public void reset() throws IOException {}

434   /** Releases resources associated with this stream. */
435   public void close() throws IOException {}
