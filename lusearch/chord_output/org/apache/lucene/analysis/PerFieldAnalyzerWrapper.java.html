20  import java.io.Reader;
21  import java.io.IOException;
22  import java.util.Map;
23  import java.util.HashMap;
24  
25  /**
26   * This analyzer is used to facilitate scenarios where different
27   * fields require different analysis techniques.  Use {@link #addAnalyzer}
28   * to add a non-default analyzer on a field name basis.
29   * 
30   * <p>Example usage:
31   * 
32   * <pre>
33   *   PerFieldAnalyzerWrapper aWrapper =
34   *      new PerFieldAnalyzerWrapper(new StandardAnalyzer());
35   *   aWrapper.addAnalyzer("firstname", new KeywordAnalyzer());
36   *   aWrapper.addAnalyzer("lastname", new KeywordAnalyzer());
37   * </pre>
38   * 
39   * <p>In this example, StandardAnalyzer will be used for all fields except "firstname"
40   * and "lastname", for which KeywordAnalyzer will be used.
41   * 
42   * <p>A PerFieldAnalyzerWrapper can be used like any other analyzer, for both indexing
43   * and query parsing.
44   */
45  public class PerFieldAnalyzerWrapper extends Analyzer {
46    private Analyzer defaultAnalyzer;
47    private Map analyzerMap = new HashMap();
48  
49  
50    /**
51     * Constructs with default analyzer.
52     *
53     * @param defaultAnalyzer Any fields not specifically
54     * defined to use a different analyzer will use the one provided here.
55     */
56    public PerFieldAnalyzerWrapper(Analyzer defaultAnalyzer) {
57      this(defaultAnalyzer, null);
58    }
59    
60    /**
61     * Constructs with default analyzer and a map of analyzers to use for 
62     * specific fields.
63     *
64     * @param defaultAnalyzer Any fields not specifically
65     * defined to use a different analyzer will use the one provided here.
66     * @param fieldAnalyzers a Map (String field name to the Analyzer) to be 
67     * used for those fields 
68     */
69    public PerFieldAnalyzerWrapper(Analyzer defaultAnalyzer, 
70        Map /*<String, Analyzer>*/ fieldAnalyzers) {
71      this.defaultAnalyzer = defaultAnalyzer;
72      if (fieldAnalyzers != null) {
73        analyzerMap.putAll(fieldAnalyzers);
74      }
75      setOverridesTokenStreamMethod(PerFieldAnalyzerWrapper.class);
76    }
77    
78  
79    /**
80     * Defines an analyzer to use for the specified field.
81     *
82     * @param fieldName field name requiring a non-default analyzer
83     * @param analyzer non-default analyzer to use for field
84     */
85    public void addAnalyzer(String fieldName, Analyzer analyzer) {
86      analyzerMap.put(fieldName, analyzer);
87    }
88  
89    public TokenStream tokenStream(String fieldName, Reader reader) {
90      Analyzer analyzer = (Analyzer) analyzerMap.get(fieldName);
91      if (analyzer == null) {
92        analyzer = defaultAnalyzer;
93      }
94  
95      return analyzer.tokenStream(fieldName, reader);
96    }
97    
98    public TokenStream reusableTokenStream(String fieldName, Reader reader) throws IOException {
99      if (overridesTokenStreamMethod) {
100       // LUCENE-1678: force fallback to tokenStream() if we
101       // have been subclassed and that subclass overrides
102       // tokenStream but not reusableTokenStream
103       return tokenStream(fieldName, reader);
104     }
105     Analyzer analyzer = (Analyzer) analyzerMap.get(fieldName);
106     if (analyzer == null)
107       analyzer = defaultAnalyzer;
108 
109     return analyzer.reusableTokenStream(fieldName, reader);
110   }
111   
112   /** Return the positionIncrementGap from the analyzer assigned to fieldName */
113   public int getPositionIncrementGap(String fieldName) {
114     Analyzer analyzer = (Analyzer) analyzerMap.get(fieldName);
115     if (analyzer == null)
116       analyzer = defaultAnalyzer;
117     return analyzer.getPositionIncrementGap(fieldName);
118   }
119   
120   public String toString() {
121     return "PerFieldAnalyzerWrapper(" + analyzerMap + ", default=" + defaultAnalyzer + ")";
122   }
123 }
