20  import java.io.Reader;
21  import java.io.IOException;
22  import java.lang.reflect.Method;
23  
24  import org.apache.lucene.util.CloseableThreadLocal;
25  import org.apache.lucene.store.AlreadyClosedException;
26  
27  import org.apache.lucene.document.Fieldable;
28  
29  /** An Analyzer builds TokenStreams, which analyze text.  It thus represents a
30   *  policy for extracting index terms from text.
31   *  <p>
32   *  Typical implementations first build a Tokenizer, which breaks the stream of
33   *  characters from the Reader into raw Tokens.  One or more TokenFilters may
34   *  then be applied to the output of the Tokenizer.
35   */
36  public abstract class Analyzer {
37    /** Creates a TokenStream which tokenizes all the text in the provided
38     * Reader.  Must be able to handle null field name for
39     * backward compatibility.
40     */
41    public abstract TokenStream tokenStream(String fieldName, Reader reader);
42  
43    /** Creates a TokenStream that is allowed to be re-used
44     *  from the previous time that the same thread called
45     *  this method.  Callers that do not need to use more
46     *  than one TokenStream at the same time from this
47     *  analyzer should use this method for better
48     *  performance.
49     */
50    public TokenStream reusableTokenStream(String fieldName, Reader reader) throws IOException {
51      return tokenStream(fieldName, reader);
52    }
53  
54    private CloseableThreadLocal tokenStreams = new CloseableThreadLocal();
55  
56    /** Used by Analyzers that implement reusableTokenStream
57     *  to retrieve previously saved TokenStreams for re-use
58     *  by the same thread. */
59    protected Object getPreviousTokenStream() {
60      try {
61        return tokenStreams.get();
62      } catch (NullPointerException npe) {
63        if (tokenStreams == null) {
64          throw new AlreadyClosedException("this Analyzer is closed");
65        } else {
66          throw npe;
67        }
68      }
69    }
70  
71    /** Used by Analyzers that implement reusableTokenStream
72     *  to save a TokenStream for later re-use by the same
73     *  thread. */
74    protected void setPreviousTokenStream(Object obj) {
75      try {
76        tokenStreams.set(obj);
77      } catch (NullPointerException npe) {
78        if (tokenStreams == null) {
79          throw new AlreadyClosedException("this Analyzer is closed");
80        } else {
81          throw npe;
82        }
83      }
84    }
85  
86    protected boolean overridesTokenStreamMethod;
87  
88    /** @deprecated This is only present to preserve
89     *  back-compat of classes that subclass a core analyzer
90     *  and override tokenStream but not reusableTokenStream */
91    protected void setOverridesTokenStreamMethod(Class baseClass) {
92  
93      final Class[] params = new Class[2];
94      params[0] = String.class;
95      params[1] = Reader.class;
96      
97      try {
98        Method m = this.getClass().getMethod("tokenStream", params);
99        if (m != null) {
100         overridesTokenStreamMethod = m.getDeclaringClass() != baseClass;
101       } else {
102         overridesTokenStreamMethod = false;
103       }
104     } catch (NoSuchMethodException nsme) {
105       overridesTokenStreamMethod = false;
106     }
107   }
108 
109 
110   /**
111    * Invoked before indexing a Fieldable instance if
112    * terms have already been added to that field.  This allows custom
113    * analyzers to place an automatic position increment gap between
114    * Fieldable instances using the same field name.  The default value
115    * position increment gap is 0.  With a 0 position increment gap and
116    * the typical default token position increment of 1, all terms in a field,
117    * including across Fieldable instances, are in successive positions, allowing
118    * exact PhraseQuery matches, for instance, across Fieldable instance boundaries.
119    *
120    * @param fieldName Fieldable name being indexed.
121    * @return position increment gap, added to the next token emitted from {@link #tokenStream(String,Reader)}
122    */
123   public int getPositionIncrementGap(String fieldName)
124   {
125     return 0;
126   }
127 
128   /**
129    * Just like {@link #getPositionIncrementGap}, except for
130    * Token offsets instead.  By default this returns 1 for
131    * tokenized fields and, as if the fields were joined
132    * with an extra space character, and 0 for un-tokenized
133    * fields.  This method is only called if the field
134    * produced at least one token for indexing.
135    *
136    * @param field the field just indexed
137    * @return offset gap, added to the next token emitted from {@link #tokenStream(String,Reader)}
138    */
139   public int getOffsetGap(Fieldable field) {
140     if (field.isTokenized())
141       return 1;
142     else
143       return 0;
144   }
145 
146   /** Frees persistent resources used by this Analyzer */
147   public void close() {
148     tokenStreams.close();
149     tokenStreams = null;
150   }
151 }
