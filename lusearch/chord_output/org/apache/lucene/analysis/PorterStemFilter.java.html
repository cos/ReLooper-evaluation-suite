20  import java.io.IOException;
21  
22  import org.apache.lucene.analysis.tokenattributes.TermAttribute;
23  
24  /** Transforms the token stream as per the Porter stemming algorithm.
25      Note: the input to the stemming filter must already be in lower case,
26      so you will need to use LowerCaseFilter or LowerCaseTokenizer farther
27      down the Tokenizer chain in order for this to work properly!
28      <P>
29      To use this filter with other analyzers, you'll want to write an
30      Analyzer class that sets up the TokenStream chain as you want it.
31      To use this with LowerCaseTokenizer, for example, you'd write an
32      analyzer like this:
33      <P>
34      <PRE>
35      class MyAnalyzer extends Analyzer {
36        public final TokenStream tokenStream(String fieldName, Reader reader) {
37          return new PorterStemFilter(new LowerCaseTokenizer(reader));
38        }
39      }
40      </PRE>
41  */
42  public final class PorterStemFilter extends TokenFilter {
43    private PorterStemmer stemmer;
44    private TermAttribute termAtt;
45  
46    public PorterStemFilter(TokenStream in) {
47      super(in);
48      stemmer = new PorterStemmer();
49      termAtt = (TermAttribute) addAttribute(TermAttribute.class);
50    }
51  
52    public final boolean incrementToken() throws IOException {
53      if (!input.incrementToken())
54        return false;
55  
56      if (stemmer.stem(termAtt.termBuffer(), 0, termAtt.termLength()))
57        termAtt.setTermBuffer(stemmer.getResultBuffer(), 0, stemmer.getResultLength());
58      return true;
59    }
60  }
