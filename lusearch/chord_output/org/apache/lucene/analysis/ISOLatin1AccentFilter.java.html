22  /**
23   * A filter that replaces accented characters in the ISO Latin 1 character set 
24   * (ISO-8859-1) by their unaccented equivalent. The case will not be altered.
25   * <p>
26   * For instance, '&agrave;' will be replaced by 'a'.
27   * <p>
28   * 
29   * @deprecated in favor of {@link ASCIIFoldingFilter} which covers a superset 
30   * of Latin 1. This class will be removed in Lucene 3.0.
31   */
32  public class ISOLatin1AccentFilter extends TokenFilter {
33    public ISOLatin1AccentFilter(TokenStream input) {
34      super(input);
35      termAtt = (TermAttribute) addAttribute(TermAttribute.class);
36    }
37  
38    private char[] output = new char[256];
39    private int outputPos;
40    private TermAttribute termAtt;
41      
42    public final boolean incrementToken() throws java.io.IOException {    
43      if (input.incrementToken()) {
44        final char[] buffer = termAtt.termBuffer();
45        final int length = termAtt.termLength();
46        // If no characters actually require rewriting then we
47        // just return token as-is:
48        for(int i=0;i<length;i++) {
49          final char c = buffer[i];
50          if (c \u00c0= 'À' && c \uFB06= 'ﬆ') {
51            removeAccents(buffer, length);
52            termAtt.setTermBuffer(output, 0, outputPos);
53            break;
54          }
55        }
56        return true;
57      } else
58        return false;
59    }
60  
61    /** @deprecated Will be removed in Lucene 3.0. This method is final, as it should
62     * not be overridden. Delegates to the backwards compatibility layer. */
63    public final Token next(final Token reusableToken) throws java.io.IOException {
64      return super.next(reusableToken);
65    }
66  
67    /** @deprecated Will be removed in Lucene 3.0. This method is final, as it should
68     * not be overridden. Delegates to the backwards compatibility layer. */
69    public final Token next() throws java.io.IOException {
70      return super.next();
71    }

73    /**
74     * To replace accented characters in a String by unaccented equivalents.
75     */
76    public final void removeAccents(char[] input, int length) {
77  
78      // Worst-case length required:
79      final int maxSizeNeeded = 2*length;
80  
81      int size = output.length;
82      while (size < maxSizeNeeded)
83        size *= 2;
84  
85      if (size != output.length)
86        output = new char[size];
87  
88      outputPos = 0;
89  
90      int pos = 0;
91  
92      for (int i=0; i<length; i++, pos++) {
93        final char c = input[pos];
94  
95        // Quick test: if it's not in range then just keep
96        // current character
97        if (c\u00c0< 'À' || c\uFB06> 'ﬆ')
98          output[outputPos++] = c;
99        else {
100         switch (c) {
101         ca\u00C0e 'À' : // À
102         ca\u00C1e 'Á' : // Á
103         ca\u00C2e 'Â' : // Â
104         ca\u00C3e 'Ã' : // Ã
105         ca\u00C4e 'Ä' : // Ä
106         ca\u00C5e 'Å' : // Å
107           output[outputPos++] = 'A';
108           break;
109         ca\u00C6e 'Æ' : // Æ
110           output[outputPos++] = 'A';
111           output[outputPos++] = 'E';
112           break;
113         ca\u00C7e 'Ç' : // Ç
114           output[outputPos++] = 'C';
115           break;
116         ca\u00C8e 'È' : // È
117         ca\u00C9e 'É' : // É
118         ca\u00CAe 'Ê' : // Ê
119         ca\u00CBe 'Ë' : // Ë
120           output[outputPos++] = 'E';
121           break;
122         ca\u00CCe 'Ì' : // Ì
123         ca\u00CDe 'Í' : // Í
124         ca\u00CEe 'Î' : // Î
125         ca\u00CFe 'Ï' : // Ï
126           output[outputPos++] = 'I';
127           break;
128         ca\u0132e 'Ĳ' : // Ĳ
129             output[outputPos++] = 'I';
130             output[outputPos++] = 'J';
131             break;
132         ca\u00D0e 'Ð' : // Ð
133           output[outputPos++] = 'D';
134           break;
135         ca\u00D1e 'Ñ' : // Ñ
136           output[outputPos++] = 'N';
137           break;
138         ca\u00D2e 'Ò' : // Ò
139         ca\u00D3e 'Ó' : // Ó
140         ca\u00D4e 'Ô' : // Ô
141         ca\u00D5e 'Õ' : // Õ
142         ca\u00D6e 'Ö' : // Ö
143         ca\u00D8e 'Ø' : // Ø
144           output[outputPos++] = 'O';
145           break;
146         ca\u0152e 'Œ' : // Œ
147           output[outputPos++] = 'O';
148           output[outputPos++] = 'E';
149           break;
150         ca\u00DEe 'Þ' : // Þ
151           output[outputPos++] = 'T';
152           output[outputPos++] = 'H';
153           break;
154         ca\u00D9e 'Ù' : // Ù
155         ca\u00DAe 'Ú' : // Ú
156         ca\u00DBe 'Û' : // Û
157         ca\u00DCe 'Ü' : // Ü
158           output[outputPos++] = 'U';
159           break;
160         ca\u00DDe 'Ý' : // Ý
161         ca\u0178e 'Ÿ' : // Ÿ
162           output[outputPos++] = 'Y';
163           break;
164         ca\u00E0e 'à' : // à
165         ca\u00E1e 'á' : // á
166         ca\u00E2e 'â' : // â
167         ca\u00E3e 'ã' : // ã
168         ca\u00E4e 'ä' : // ä
169         ca\u00E5e 'å' : // å
170           output[outputPos++] = 'a';
171           break;
172         ca\u00E6e 'æ' : // æ
173           output[outputPos++] = 'a';
174           output[outputPos++] = 'e';
175           break;
176         ca\u00E7e 'ç' : // ç
177           output[outputPos++] = 'c';
178           break;
179         ca\u00E8e 'è' : // è
180         ca\u00E9e 'é' : // é
181         ca\u00EAe 'ê' : // ê
182         ca\u00EBe 'ë' : // ë
183           output[outputPos++] = 'e';
184           break;
185         ca\u00ECe 'ì' : // ì
186         ca\u00EDe 'í' : // í
187         ca\u00EEe 'î' : // î
188         ca\u00EFe 'ï' : // ï
189           output[outputPos++] = 'i';
190           break;
191         ca\u0133e 'ĳ' : // ĳ
192             output[outputPos++] = 'i';
193             output[outputPos++] = 'j';
194             break;
195         ca\u00F0e 'ð' : // ð
196           output[outputPos++] = 'd';
197           break;
198         ca\u00F1e 'ñ' : // ñ
199           output[outputPos++] = 'n';
200           break;
201         ca\u00F2e 'ò' : // ò
202         ca\u00F3e 'ó' : // ó
203         ca\u00F4e 'ô' : // ô
204         ca\u00F5e 'õ' : // õ
205         ca\u00F6e 'ö' : // ö
206         ca\u00F8e 'ø' : // ø
207           output[outputPos++] = 'o';
208           break;
209         ca\u0153e 'œ' : // œ
210           output[outputPos++] = 'o';
211           output[outputPos++] = 'e';
212           break;
213         ca\u00DFe 'ß' : // ß
214           output[outputPos++] = 's';
215           output[outputPos++] = 's';
216           break;
217         ca\u00FEe 'þ' : // þ
218           output[outputPos++] = 't';
219           output[outputPos++] = 'h';
220           break;
221         ca\u00F9e 'ù' : // ù
222         ca\u00FAe 'ú' : // ú
223         ca\u00FBe 'û' : // û
224         ca\u00FCe 'ü' : // ü
225           output[outputPos++] = 'u';
226           break;
227         ca\u00FDe 'ý' : // ý
228         ca\u00FFe 'ÿ' : // ÿ
229           output[outputPos++] = 'y';
230           break;
231         ca\uFB00e 'ﬀ': // ﬀ
232             output[outputPos++] = 'f';
233             output[outputPos++] = 'f';
234             break;
235         ca\uFB01e 'ﬁ': // ﬁ
236             output[outputPos++] = 'f';
237             output[outputPos++] = 'i';
238             break;
239         ca\uFB02e 'ﬂ': // ﬂ
240             output[outputPos++] = 'f';
241             output[outputPos++] = 'l';
242             break;
243         // following 2 are commented as they can break the maxSizeNeeded (and doing *3 could be expensive)
244 //        ca\uFB03e 'ﬃ': // ﬃ
245 //            output[outputPos++] = 'f';
246 //            output[outputPos++] = 'f';
247 //            output[outputPos++] = 'i';
248 //            break;
249 //        ca\uFB04e 'ﬄ': // ﬄ
250 //            output[outputPos++] = 'f';
251 //            output[outputPos++] = 'f';
252 //            output[outputPos++] = 'l';
253 //            break;
254         ca\uFB05e 'ﬅ': // ﬅ
255             output[outputPos++] = 'f';
256             output[outputPos++] = 't';
257             break;
258         ca\uFB06e 'ﬆ': // ﬆ
259             output[outputPos++] = 's';
260             output[outputPos++] = 't';
261           break;
262         default :
263           output[outputPos++] = c;
264           break;
265         }
266       }
267     }
268   }
