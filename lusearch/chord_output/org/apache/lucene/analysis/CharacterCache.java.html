18  package org.apache.lucene.analysis;
19  
20  /**
21   * Replacement for Java 1.5 Character.valueOf()
22   * @deprecated Move to Character.valueOf() in 3.0
23   */
24  class CharacterCache {
25  
26    private static final Character cache[] = new Character[128];
27  
28    static {
29      for (int i = 0; i < cache.length; i++) {
30        cache[i] = new Character((char) i);
31      }
32    }
33  
34    /**
35     * Returns a Character instance representing the given char value
36     * 
37     * @param c
38     *          a char value
39     * @return a Character representation of the given char value.
40     */
41    public static Character valueOf(char c) {
42      if (c < cache.length) {
43        return cache[(int) c];
44      }
45      return new Character(c);
46    }
47  }
