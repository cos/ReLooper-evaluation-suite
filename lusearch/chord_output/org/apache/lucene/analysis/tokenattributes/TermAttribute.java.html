20  import org.apache.lucene.util.Attribute;
21  
22  /**
23   * The term text of a Token.
24   */
25  public interface TermAttribute extends Attribute {
26    /** Returns the Token's term text.
27     * 
28     * This method has a performance penalty
29     * because the text is stored internally in a char[].  If
30     * possible, use {@link #termBuffer()} and {@link
31     * #termLength()} directly instead.  If you really need a
32     * String, use this method, which is nothing more than
33     * a convenience call to <b>new String(token.termBuffer(), 0, token.termLength())</b>
34     */
35    public String term();
36    
37    /** Copies the contents of buffer, starting at offset for
38     *  length characters, into the termBuffer array.
39     *  @param buffer the buffer to copy
40     *  @param offset the index in the buffer of the first character to copy
41     *  @param length the number of characters to copy
42     */
43    public void setTermBuffer(char[] buffer, int offset, int length);
44  
45    /** Copies the contents of buffer into the termBuffer array.
46     *  @param buffer the buffer to copy
47     */
48    public void setTermBuffer(String buffer);
49  
50    /** Copies the contents of buffer, starting at offset and continuing
51     *  for length characters, into the termBuffer array.
52     *  @param buffer the buffer to copy
53     *  @param offset the index in the buffer of the first character to copy
54     *  @param length the number of characters to copy
55     */
56    public void setTermBuffer(String buffer, int offset, int length);
57    
58    /** Returns the internal termBuffer character array which
59     *  you can then directly alter.  If the array is too
60     *  small for your token, use {@link
61     *  #resizeTermBuffer(int)} to increase it.  After
62     *  altering the buffer be sure to call {@link
63     *  #setTermLength} to record the number of valid
64     *  characters that were placed into the termBuffer. */
65    public char[] termBuffer();
66  
67    /** Grows the termBuffer to at least size newSize, preserving the
68     *  existing content. Note: If the next operation is to change
69     *  the contents of the term buffer use
70     *  {@link #setTermBuffer(char[], int, int)},
71     *  {@link #setTermBuffer(String)}, or
72     *  {@link #setTermBuffer(String, int, int)}
73     *  to optimally combine the resize with the setting of the termBuffer.
74     *  @param newSize minimum size of the new termBuffer
75     *  @return newly created termBuffer with length >= newSize
76     */
77    public char[] resizeTermBuffer(int newSize);
78  
79    /** Return number of valid characters (length of the term)
80     *  in the termBuffer array. */
81    public int termLength();
82    
83    /** Set number of valid characters (length of the term) in
84     *  the termBuffer array. Use this to truncate the termBuffer
85     *  or to synchronize with external manipulation of the termBuffer.
86     *  Note: to grow the size of the array,
87     *  use {@link #resizeTermBuffer(int)} first.
88     *  @param length the truncated length
89     */
90    public void setTermLength(int length);
91  }
