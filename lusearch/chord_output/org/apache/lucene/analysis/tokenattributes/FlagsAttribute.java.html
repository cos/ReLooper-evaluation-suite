20  import org.apache.lucene.analysis.Tokenizer;
21  import org.apache.lucene.util.Attribute;
22  
23  /**
24   * This attribute can be used to pass different flags down the {@link Tokenizer} chain,
25   * eg from one TokenFilter to another one. 
26   */
27  public interface FlagsAttribute extends Attribute {
28    /**
29     * EXPERIMENTAL:  While we think this is here to stay, we may want to change it to be a long.
30     * <p/>
31     *
32     * Get the bitset for any bits that have been set.  This is completely distinct from {@link TypeAttribute#type()}, although they do share similar purposes.
33     * The flags can be used to encode information about the token for use by other {@link org.apache.lucene.analysis.TokenFilter}s.
34     *
35     *
36     * @return The bits
37     */
38    public int getFlags();
39  
40    /**
41     * @see #getFlags()
42     */
43    public void setFlags(int flags);  
44  }
