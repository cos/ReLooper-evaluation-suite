20  import java.io.BufferedReader;
21  import java.io.File;
22  import java.io.FileReader;
23  import java.io.IOException;
24  import java.io.Reader;
25  import java.util.HashMap;
26  import java.util.HashSet;
27  
28  /**
29   * Loader for text files that represent a list of stopwords.
30   *
31   *
32   * @version $Id: WordlistLoader.java 706342 2008-10-20 17:19:29Z gsingers $
33   */
34  public class WordlistLoader {
35  
36    /**
37     * Loads a text file and adds every line as an entry to a HashSet (omitting
38     * leading and trailing whitespace). Every line of the file should contain only
39     * one word. The words need to be in lowercase if you make use of an
40     * Analyzer which uses LowerCaseFilter (like StandardAnalyzer).
41     *
42     * @param wordfile File containing the wordlist
43     * @return A HashSet with the file's words
44     */
45    public static HashSet getWordSet(File wordfile) throws IOException {
46      HashSet result = new HashSet();
47      FileReader reader = null;
48      try {
49        reader = new FileReader(wordfile);
50        result = getWordSet(reader);
51      }
52      finally {
53        if (reader != null)
54          reader.close();
55      }
56      return result;
57    }
58  
59    /**
60     * Loads a text file and adds every non-comment line as an entry to a HashSet (omitting
61     * leading and trailing whitespace). Every line of the file should contain only
62     * one word. The words need to be in lowercase if you make use of an
63     * Analyzer which uses LowerCaseFilter (like StandardAnalyzer).
64     *
65     * @param wordfile File containing the wordlist
66     * @param comment The comment string to ignore
67     * @return A HashSet with the file's words
68     */
69    public static HashSet getWordSet(File wordfile, String comment) throws IOException {
70      HashSet result = new HashSet();
71      FileReader reader = null;
72      try {
73        reader = new FileReader(wordfile);
74        result = getWordSet(reader, comment);
75      }
76      finally {
77        if (reader != null)
78          reader.close();
79      }
80      return result;
81    }
82  
83  
84    /**
85     * Reads lines from a Reader and adds every line as an entry to a HashSet (omitting
86     * leading and trailing whitespace). Every line of the Reader should contain only
87     * one word. The words need to be in lowercase if you make use of an
88     * Analyzer which uses LowerCaseFilter (like StandardAnalyzer).
89     *
90     * @param reader Reader containing the wordlist
91     * @return A HashSet with the reader's words
92     */
93    public static HashSet getWordSet(Reader reader) throws IOException {
94      HashSet result = new HashSet();
95      BufferedReader br = null;
96      try {
97        if (reader instanceof BufferedReader) {
98          br = (BufferedReader) reader;
99        } else {
100         br = new BufferedReader(reader);
101       }
102       String word = null;
103       while ((word = br.readLine()) != null) {
104         result.add(word.trim());
105       }
106     }
107     finally {
108       if (br != null)
109         br.close();
110     }
111     return result;
112   }
113 
114   /**
115    * Reads lines from a Reader and adds every non-comment line as an entry to a HashSet (omitting
116    * leading and trailing whitespace). Every line of the Reader should contain only
117    * one word. The words need to be in lowercase if you make use of an
118    * Analyzer which uses LowerCaseFilter (like StandardAnalyzer).
119    *
120    * @param reader Reader containing the wordlist
121    * @param comment The string representing a comment.
122    * @return A HashSet with the reader's words
123    */
124   public static HashSet getWordSet(Reader reader, String comment) throws IOException {
125     HashSet result = new HashSet();
126     BufferedReader br = null;
127     try {
128       if (reader instanceof BufferedReader) {
129         br = (BufferedReader) reader;
130       } else {
131         br = new BufferedReader(reader);
132       }
133       String word = null;
134       while ((word = br.readLine()) != null) {
135         if (word.startsWith(comment) == false){
136           result.add(word.trim());
137         }
138       }
139     }
140     finally {
141       if (br != null)
142         br.close();
143     }
144     return result;
145   }

149   /**
150    * Reads a stem dictionary. Each line contains:
151    * <pre>word<b>\t</b>stem</pre>
152    * (i.e. two tab seperated words)
153    *
154    * @return stem dictionary that overrules the stemming algorithm
155    * @throws IOException 
156    */
157   public static HashMap getStemDict(File wordstemfile) throws IOException {
158     if (wordstemfile == null)
159       throw new NullPointerException("wordstemfile may not be null");
160     HashMap result = new HashMap();
161     BufferedReader br = null;
162     FileReader fr = null;
163     try {
164       fr = new FileReader(wordstemfile);
165       br = new BufferedReader(fr);
166       String line;
167       while ((line = br.readLine()) != null) {
168         String[] wordstem = line.split("\t", 2);
169         result.put(wordstem[0], wordstem[1]);
170       }
171     } finally {
172       if (fr != null)
173         fr.close();
174       if (br != null)
175         br.close();
176     }
177     return result;
178   }
