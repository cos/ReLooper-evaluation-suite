60    public ASCIIFoldingFilter(TokenStream input)
61    {
62      super(input);
63      termAtt = (TermAttribute) addAttribute(TermAttribute.class);

66    private char[] output = new char[512];
67    private int outputPos;
68    private TermAttribute termAtt;
69  
70    public boolean incrementToken() throws IOException {
71      if (input.incrementToken()) {
72        final char[] buffer = termAtt.termBuffer();
73        final int length = termAtt.termLength();
74  
75        // If no characters actually require rewriting then we
76        // just return token as-is:
77        for(int i = 0 ; i < length ; ++i) {
78          final char c = buffer[i];
79          if (c \u0080= '')
80          {
81            foldToASCII(buffer, length);
82            termAtt.setTermBuffer(output, 0, outputPos);
83            break;
84          }
85        }
86        return true;
87      } else {
88        return false;
89      }
90    }
